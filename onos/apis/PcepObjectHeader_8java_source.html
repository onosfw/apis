170  /**
171  * Writes Byte stream of PCEP object header to channel buffer.
172  *
173  * @param cb output channel buffer
174  * @return objLenIndex object length index in channel buffer
175  */
176  public int write(ChannelBuffer cb) {
177 
178  cb.writeByte(this.objClass);
179  byte temp = (byte) (this.objType << OBJECT_TYPE_SHIFT_VALUE);
180  if (this.bPFlag) {
181  temp = (byte) (temp | PFLAG_SET);
182  }
183  if (this.bIFlag) {
184  temp = (byte) (temp | IFLAG_SET);
185  }
186  cb.writeByte(temp);
187  int objLenIndex = cb.writerIndex();
188  cb.writeShort((short) 0);
189  return objLenIndex;
190  }

192  /**
193  * Read from channel buffer and Returns PCEP Objects header.
194  *
195  * @param cb of type channel buffer
196  * @return PCEP Object header
197  */
198  public static PcepObjectHeader read(ChannelBuffer cb) {
199 
200  byte objClass;
201  byte objType;
202  boolean bPFlag;
203  boolean bIFlag;
204  short objLen;
205  objClass = cb.readByte();
206  byte temp = cb.readByte();
207  bIFlag = (temp & IFLAG_SET) == IFLAG_SET;
208  bPFlag = (temp & PFLAG_SET) == PFLAG_SET;
209  objType = (byte) (temp >> OBJECT_TYPE_SHIFT_VALUE);
210  objLen = cb.readShort();
211  return new PcepObjectHeader(objClass, objType, bPFlag, bIFlag, objLen);
212  }
