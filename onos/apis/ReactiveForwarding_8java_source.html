403  /**
404  * Packet processor responsible for forwarding packets along their paths.
405  */
406  private class ReactivePacketProcessor implements PacketProcessor {
407 
408  @Override
409  public void process(PacketContext context) {
410  // Stop processing if the packet has been handled, since we
411  // can't do any more to it.
412 
413  if (context.isHandled()) {
414  return;
415  }
416 
417  InboundPacket pkt = context.inPacket();
418  Ethernet ethPkt = pkt.parsed();
419 
420  if (ethPkt == null) {
421  return;
422  }
423 
424  // Bail if this is deemed to be a control packet.
425  if (isControlPacket(ethPkt)) {
426  return;
427  }
428 
429  // Skip IPv6 multicast packet when IPv6 forward is disabled.
430  if (!ipv6Forwarding && isIpv6Multicast(ethPkt)) {
431  return;
432  }
433 
434  HostId id = HostId.hostId(ethPkt.getDestinationMAC());
435 
436  // Do not process link-local addresses in any way.
437  if (id.mac().isLinkLocal()) {
438  return;
439  }
440 
441  // Do not process IPv4 multicast packets, let mfwd handle them
442  if (ignoreIpv4McastPackets && ethPkt.getEtherType() == Ethernet.TYPE_IPV4) {
443  if (id.mac().isMulticast()) {
444  return;
445  }
446  }
447 
448  // Do we know who this is for? If not, flood and bail.
449  Host dst = hostService.getHost(id);
450  if (dst == null) {
451  flood(context);
452  return;
453  }
454 
455  // Are we on an edge switch that our destination is on? If so,
456  // simply forward out to the destination and bail.
457  if (pkt.receivedFrom().deviceId().equals(dst.location().deviceId())) {
458  if (!context.inPacket().receivedFrom().port().equals(dst.location().port())) {
459  installRule(context, dst.location().port());
460  }
461  return;
462  }
463 
464  // Otherwise, get a set of paths that lead from here to the
465  // destination edge switch.
466  Set<Path> paths =
467  topologyService.getPaths(topologyService.currentTopology(),
468  pkt.receivedFrom().deviceId(),
469  dst.location().deviceId());
470  if (paths.isEmpty()) {
471  // If there are no paths, flood and bail.
472  flood(context);
473  return;
474  }
475 
476  // Otherwise, pick a path that does not lead back to where we
477  // came from; if no such path, flood and bail.
478  Path path = pickForwardPathIfPossible(paths, pkt.receivedFrom().port());
479  if (path == null) {
480  log.warn("Don't know where to go from here {} for {} -> {}",
481  pkt.receivedFrom(), ethPkt.getSourceMAC(), ethPkt.getDestinationMAC());
482  flood(context);
483  return;
484  }
485 
486  // Otherwise forward and be done with it.
487  installRule(context, path.src().port());
488  }
489 
490  }

492  // Indicates whether this is a control packet, e.g. LLDP, BDDP
493  private boolean isControlPacket(Ethernet eth) {
494  short type = eth.getEtherType();
495  return type == Ethernet.TYPE_LLDP || type == Ethernet.TYPE_BSN;
496  }

498  // Indicated whether this is an IPv6 multicast packet.
499  private boolean isIpv6Multicast(Ethernet eth) {
500  return eth.getEtherType() == Ethernet.TYPE_IPV6 && eth.isMulticast();
501  }
502 
503  // Selects a path from the given set that does not lead back to the
504  // specified port if possible.
505  private Path pickForwardPathIfPossible(Set<Path> paths, PortNumber notToPort) {
506  Path lastPath = null;
507  for (Path path : paths) {
508  lastPath = path;
509  if (!path.src().port().equals(notToPort)) {
510  return path;
511  }
512  }
513  return lastPath;
514  }

532  // Install a rule forwarding the packet to the specified port.
533  private void installRule(PacketContext context, PortNumber portNumber) {
534  //
535  // We don't support (yet) buffer IDs in the Flow Service so
536  // packet out first.
537  //
538  Ethernet inPkt = context.inPacket().parsed();
539  TrafficSelector.Builder selectorBuilder = DefaultTrafficSelector.builder();
540 
541  // If PacketOutOnly or ARP packet than forward directly to output port
542  if (packetOutOnly || inPkt.getEtherType() == Ethernet.TYPE_ARP) {
543  packetOut(context, portNumber);
544  return;
545  }
546 
547  //
548  // If matchDstMacOnly
549  // Create flows matching dstMac only
550  // Else
551  // Create flows with default matching and include configured fields
552  //
553  if (matchDstMacOnly) {
554  selectorBuilder.matchEthDst(inPkt.getDestinationMAC());
555  } else {
556  selectorBuilder.matchInPort(context.inPacket().receivedFrom().port())
557  .matchEthSrc(inPkt.getSourceMAC())
558  .matchEthDst(inPkt.getDestinationMAC());
559 
560  // If configured Match Vlan ID
561  if (matchVlanId && inPkt.getVlanID() != Ethernet.VLAN_UNTAGGED) {
562  selectorBuilder.matchVlanId(VlanId.vlanId(inPkt.getVlanID()));
563  }
564 
565  //
566  // If configured and EtherType is IPv4 - Match IPv4 and
567  // TCP/UDP/ICMP fields
568  //
569  if (matchIpv4Address && inPkt.getEtherType() == Ethernet.TYPE_IPV4) {
570  IPv4 ipv4Packet = (IPv4) inPkt.getPayload();
571  byte ipv4Protocol = ipv4Packet.getProtocol();
572  Ip4Prefix matchIp4SrcPrefix =
573  Ip4Prefix.valueOf(ipv4Packet.getSourceAddress(),
574  Ip4Prefix.MAX_MASK_LENGTH);
575  Ip4Prefix matchIp4DstPrefix =
576  Ip4Prefix.valueOf(ipv4Packet.getDestinationAddress(),
577  Ip4Prefix.MAX_MASK_LENGTH);
578  selectorBuilder.matchEthType(Ethernet.TYPE_IPV4)
579  .matchIPSrc(matchIp4SrcPrefix)
580  .matchIPDst(matchIp4DstPrefix);
581 
582  if (matchIpv4Dscp) {
583  byte dscp = ipv4Packet.getDscp();
584  byte ecn = ipv4Packet.getEcn();
585  selectorBuilder.matchIPDscp(dscp).matchIPEcn(ecn);
586  }
587 
588  if (matchTcpUdpPorts && ipv4Protocol == IPv4.PROTOCOL_TCP) {
589  TCP tcpPacket = (TCP) ipv4Packet.getPayload();
590  selectorBuilder.matchIPProtocol(ipv4Protocol)
591  .matchTcpSrc(TpPort.tpPort(tcpPacket.getSourcePort()))
592  .matchTcpDst(TpPort.tpPort(tcpPacket.getDestinationPort()));
593  }
594  if (matchTcpUdpPorts && ipv4Protocol == IPv4.PROTOCOL_UDP) {
595  UDP udpPacket = (UDP) ipv4Packet.getPayload();
596  selectorBuilder.matchIPProtocol(ipv4Protocol)
597  .matchUdpSrc(TpPort.tpPort(udpPacket.getSourcePort()))
598  .matchUdpDst(TpPort.tpPort(udpPacket.getDestinationPort()));
599  }
600  if (matchIcmpFields && ipv4Protocol == IPv4.PROTOCOL_ICMP) {
601  ICMP icmpPacket = (ICMP) ipv4Packet.getPayload();
602  selectorBuilder.matchIPProtocol(ipv4Protocol)
603  .matchIcmpType(icmpPacket.getIcmpType())
604  .matchIcmpCode(icmpPacket.getIcmpCode());
605  }
606  }
607 
608  //
609  // If configured and EtherType is IPv6 - Match IPv6 and
610  // TCP/UDP/ICMP fields
611  //
612  if (matchIpv6Address && inPkt.getEtherType() == Ethernet.TYPE_IPV6) {
613  IPv6 ipv6Packet = (IPv6) inPkt.getPayload();
614  byte ipv6NextHeader = ipv6Packet.getNextHeader();
615  Ip6Prefix matchIp6SrcPrefix =
616  Ip6Prefix.valueOf(ipv6Packet.getSourceAddress(),
617  Ip6Prefix.MAX_MASK_LENGTH);
618  Ip6Prefix matchIp6DstPrefix =
619  Ip6Prefix.valueOf(ipv6Packet.getDestinationAddress(),
620  Ip6Prefix.MAX_MASK_LENGTH);
621  selectorBuilder.matchEthType(Ethernet.TYPE_IPV6)
622  .matchIPv6Src(matchIp6SrcPrefix)
623  .matchIPv6Dst(matchIp6DstPrefix);
624 
625  if (matchIpv6FlowLabel) {
626  selectorBuilder.matchIPv6FlowLabel(ipv6Packet.getFlowLabel());
627  }
628 
629  if (matchTcpUdpPorts && ipv6NextHeader == IPv6.PROTOCOL_TCP) {
630  TCP tcpPacket = (TCP) ipv6Packet.getPayload();
631  selectorBuilder.matchIPProtocol(ipv6NextHeader)
632  .matchTcpSrc(TpPort.tpPort(tcpPacket.getSourcePort()))
633  .matchTcpDst(TpPort.tpPort(tcpPacket.getDestinationPort()));
634  }
635  if (matchTcpUdpPorts && ipv6NextHeader == IPv6.PROTOCOL_UDP) {
636  UDP udpPacket = (UDP) ipv6Packet.getPayload();
637  selectorBuilder.matchIPProtocol(ipv6NextHeader)
638  .matchUdpSrc(TpPort.tpPort(udpPacket.getSourcePort()))
639  .matchUdpDst(TpPort.tpPort(udpPacket.getDestinationPort()));
640  }
641  if (matchIcmpFields && ipv6NextHeader == IPv6.PROTOCOL_ICMP6) {
642  ICMP6 icmp6Packet = (ICMP6) ipv6Packet.getPayload();
643  selectorBuilder.matchIPProtocol(ipv6NextHeader)
644  .matchIcmpv6Type(icmp6Packet.getIcmpType())
645  .matchIcmpv6Code(icmp6Packet.getIcmpCode());
646  }
647  }
648  }
649  TrafficTreatment treatment = DefaultTrafficTreatment.builder()
650  .setOutput(portNumber)
651  .build();
652 
653  ForwardingObjective forwardingObjective = DefaultForwardingObjective.builder()
654  .withSelector(selectorBuilder.build())
655  .withTreatment(treatment)
656  .withPriority(flowPriority)
657  .withFlag(ForwardingObjective.Flag.VERSATILE)
658  .fromApp(appId)
659  .makeTemporary(flowTimeout)
660  .add();
661 
662  flowObjectiveService.forward(context.inPacket().receivedFrom().deviceId(),
663  forwardingObjective);
664 
665  //
666  // If packetOutOfppTable
667  // Send packet back to the OpenFlow pipeline to match installed flow
668  // Else
669  // Send packet direction on the appropriate port
670  //
671  if (packetOutOfppTable) {
672  packetOut(context, PortNumber.TABLE);
673  } else {
674  packetOut(context, portNumber);
675  }

695  private void fixBlackhole(ConnectPoint egress) {
696  Set<FlowEntry> rules = getFlowRulesFrom(egress);
697  Set<SrcDstPair> pairs = findSrcDstPairs(rules);
698 
699  Map<DeviceId, Set<Path>> srcPaths = new HashMap<>();
700 
701  for (SrcDstPair sd : pairs) {
702  // get the edge deviceID for the src host
703  Host srcHost = hostService.getHost(HostId.hostId(sd.src));
704  Host dstHost = hostService.getHost(HostId.hostId(sd.dst));
705  if (srcHost != null && dstHost != null) {
706  DeviceId srcId = srcHost.location().deviceId();
707  DeviceId dstId = dstHost.location().deviceId();
708  log.trace("SRC ID is " + srcId + ", DST ID is " + dstId);
709 
710  cleanFlowRules(sd, egress.deviceId());
711 
712  Set<Path> shortestPaths = srcPaths.get(srcId);
713  if (shortestPaths == null) {
714  shortestPaths = topologyService.getPaths(topologyService.currentTopology(),
715  egress.deviceId(), srcId);
716  srcPaths.put(srcId, shortestPaths);
717  }
718  backTrackBadNodes(shortestPaths, dstId, sd);
719  }
720  }

723  // Backtracks from link down event to remove flows that lead to blackhole
724  private void backTrackBadNodes(Set<Path> shortestPaths, DeviceId dstId, SrcDstPair sd) {
725  for (Path p : shortestPaths) {
726  List<Link> pathLinks = p.links();
727  for (int i = 0; i < pathLinks.size(); i = i + 1) {
728  Link curLink = pathLinks.get(i);
729  DeviceId curDevice = curLink.src().deviceId();
730 
731  // skipping the first link because this link's src has already been pruned beforehand
732  if (i != 0) {
733  cleanFlowRules(sd, curDevice);
734  }
735 
736  Set<Path> pathsFromCurDevice =
737  topologyService.getPaths(topologyService.currentTopology(),
738  curDevice, dstId);
739  if (pickForwardPathIfPossible(pathsFromCurDevice, curLink.src().port()) != null) {
740  break;
741  } else {
742  if (i + 1 == pathLinks.size()) {
743  cleanFlowRules(sd, curLink.dst().deviceId());
744  }
745  }
746  }
747  }
748  }

750  // Removes flow rules off specified device with specific SrcDstPair
751  private void cleanFlowRules(SrcDstPair pair, DeviceId id) {
752  log.trace("Searching for flow rules to remove from: " + id);
753  log.trace("Removing flows w/ SRC=" + pair.src + ", DST=" + pair.dst);
754  for (FlowEntry r : flowRuleService.getFlowEntries(id)) {
755  boolean matchesSrc = false, matchesDst = false;
756  for (Instruction i : r.treatment().allInstructions()) {
757  if (i.type() == Instruction.Type.OUTPUT) {
758  // if the flow has matching src and dst
759  for (Criterion cr : r.selector().criteria()) {
760  if (cr.type() == Criterion.Type.ETH_DST) {
761  if (((EthCriterion) cr).mac().equals(pair.dst)) {
762  matchesDst = true;
763  }
764  } else if (cr.type() == Criterion.Type.ETH_SRC) {
765  if (((EthCriterion) cr).mac().equals(pair.src)) {
766  matchesSrc = true;
767  }
768  }
769  }
770  }
771  }
772  if (matchesDst && matchesSrc) {
773  log.trace("Removed flow rule from device: " + id);
774  flowRuleService.removeFlowRules((FlowRule) r);
775  }
776  }

780  // Returns a set of src/dst MAC pairs extracted from the specified set of flow entries
781  private Set<SrcDstPair> findSrcDstPairs(Set<FlowEntry> rules) {
782  ImmutableSet.Builder<SrcDstPair> builder = ImmutableSet.builder();
783  for (FlowEntry r : rules) {
784  MacAddress src = null, dst = null;
785  for (Criterion cr : r.selector().criteria()) {
786  if (cr.type() == Criterion.Type.ETH_DST) {
787  dst = ((EthCriterion) cr).mac();
788  } else if (cr.type() == Criterion.Type.ETH_SRC) {
789  src = ((EthCriterion) cr).mac();
790  }
791  }
792  builder.add(new SrcDstPair(src, dst));
793  }
794  return builder.build();
