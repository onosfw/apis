41  /**

43  *
44  * @return set of CordVtnNodeConfig or null
45  */
46  public Set<CordVtnNodeConfig> cordVtnNodes() {
47  Set<CordVtnNodeConfig> nodes = Sets.newHashSet();
48 
49  JsonNode jsonNodes = object.get(CORDVTN_NODES);
50  if (jsonNodes == null) {
51  return null;
52  }
53  jsonNodes.forEach(jsonNode -> nodes.add(new CordVtnNodeConfig(
54  jsonNode.path(HOSTNAME).asText(),
55  IpAddress.valueOf(jsonNode.path(OVSDB_IP).asText()),
56  TpPort.tpPort(jsonNode.path(OVSDB_PORT).asInt()),
57  DeviceId.deviceId(jsonNode.path(BRIDGE_ID).asText()))));
58 
59  return nodes;
60  }
