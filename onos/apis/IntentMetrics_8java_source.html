144  @Override
145  public void event(IntentEvent event) {
146  synchronized (lastEvents) {
147  switch (event.type()) {
148  case INSTALL_REQ:
149  intentSubmittedEventMetric.eventReceived();
150  break;
151  case INSTALLED:
152  intentInstalledEventMetric.eventReceived();
153  break;
154  case FAILED:
155  intentFailedEventMetric.eventReceived();
156  break;
157  case WITHDRAW_REQ:
158  intentWithdrawRequestedEventMetric.eventReceived();
159  break;
160  case WITHDRAWN:
161  intentWithdrawnEventMetric.eventReceived();
162  break;
163  case PURGED:
164  intentPurgedEventMetric.eventReceived();
165  break;
166  default:
167  break;
168  }
169 
170  //
171  // Keep only the last N events, where N = LAST_EVENTS_MAX_N
172  //
173  while (lastEvents.size() >= LAST_EVENTS_MAX_N) {
174  lastEvents.remove();
175  }
176  lastEvents.add(event);
177  }
178 
179  log.debug("Intent Event: time = {} type = {} event = {}",
180  event.time(), event.type(), event);
181  }
