117  /**
118  * Tests a pair of hosts with 8 hops between them.
119  */
120  @Test
121  public void testSingleLongPathCompilation() {
122 
123  HostToHostIntent intent = makeIntent(HOST_ONE,
124  HOST_TWO);
125  assertThat(intent, is(notNullValue()));
126 
127  String[] hops = {HOST_ONE, "h1", "h2", "h3", "h4", "h5", "h6", "h7", "h8", HOST_TWO};
128  HostToHostIntentCompiler compiler = makeCompiler(hops);
129  assertThat(compiler, is(notNullValue()));
130 
131  List<Intent> result = compiler.compile(intent, null, null);
132  assertThat(result, is(Matchers.notNullValue()));
133  assertThat(result, hasSize(2));
134  Intent forwardResultIntent = result.get(0);
135  assertThat(forwardResultIntent instanceof PathIntent, is(true));
136  Intent reverseResultIntent = result.get(1);
137  assertThat(reverseResultIntent instanceof PathIntent, is(true));
138 
139  if (forwardResultIntent instanceof PathIntent) {
140  PathIntent forwardPathIntent = (PathIntent) forwardResultIntent;
141  assertThat(forwardPathIntent.path().links(), hasSize(9));
142  assertThat(forwardPathIntent.path().links(), linksHasPath(HOST_ONE, "h1"));
143  assertThat(forwardPathIntent.path().links(), linksHasPath("h1", "h2"));
144  assertThat(forwardPathIntent.path().links(), linksHasPath("h2", "h3"));
145  assertThat(forwardPathIntent.path().links(), linksHasPath("h3", "h4"));
146  assertThat(forwardPathIntent.path().links(), linksHasPath("h4", "h5"));
147  assertThat(forwardPathIntent.path().links(), linksHasPath("h5", "h6"));
148  assertThat(forwardPathIntent.path().links(), linksHasPath("h6", "h7"));
149  assertThat(forwardPathIntent.path().links(), linksHasPath("h7", "h8"));
150  assertThat(forwardPathIntent.path().links(), linksHasPath("h8", HOST_TWO));
151  }
152 
153  if (reverseResultIntent instanceof PathIntent) {
154  PathIntent reversePathIntent = (PathIntent) reverseResultIntent;
155  assertThat(reversePathIntent.path().links(), hasSize(9));
156  assertThat(reversePathIntent.path().links(), linksHasPath("h1", HOST_ONE));
157  assertThat(reversePathIntent.path().links(), linksHasPath("h2", "h1"));
158  assertThat(reversePathIntent.path().links(), linksHasPath("h3", "h2"));
159  assertThat(reversePathIntent.path().links(), linksHasPath("h4", "h3"));
160  assertThat(reversePathIntent.path().links(), linksHasPath("h5", "h4"));
161  assertThat(reversePathIntent.path().links(), linksHasPath("h6", "h5"));
162  assertThat(reversePathIntent.path().links(), linksHasPath("h7", "h6"));
163  assertThat(reversePathIntent.path().links(), linksHasPath("h8", "h7"));
164  assertThat(reversePathIntent.path().links(), linksHasPath(HOST_TWO, "h8"));
165  }
166  }
