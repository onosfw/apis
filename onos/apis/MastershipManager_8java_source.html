198  @Override
199  public void balanceRoles() {
200  List<ControllerNode> nodes = newArrayList(clusterService.getNodes());
201  Map<ControllerNode, Set<DeviceId>> controllerDevices = new HashMap<>();
202  int deviceCount = 0;
203 
204  // Create buckets reflecting current ownership.
205  for (ControllerNode node : nodes) {
206  if (clusterService.getState(node.id()) == ACTIVE) {
207  Set<DeviceId> devicesOf = new HashSet<>(getDevicesOf(node.id()));
208  deviceCount += devicesOf.size();
209  controllerDevices.put(node, devicesOf);
210  log.info("Node {} has {} devices.", node.id(), devicesOf.size());
211  }
212  }
213 
214  // Now re-balance the buckets until they are roughly even.
215  List<CompletableFuture<Void>> balanceBucketsFutures = Lists.newLinkedList();
216  int rounds = controllerDevices.keySet().size();
217  for (int i = 0; i < rounds; i++) {
218  // Iterate over the buckets and find the smallest and the largest.
219  ControllerNode smallest = findBucket(true, controllerDevices);
220  ControllerNode largest = findBucket(false, controllerDevices);
221  balanceBucketsFutures.add(balanceBuckets(smallest, largest, controllerDevices, deviceCount));
222  }
223  CompletableFuture<Void> balanceRolesFuture = CompletableFuture.allOf(
224  balanceBucketsFutures.toArray(new CompletableFuture[balanceBucketsFutures.size()]));
225 
226  Futures.getUnchecked(balanceRolesFuture);
227  }

229  private ControllerNode findBucket(boolean min,
230  Map<ControllerNode, Set<DeviceId>> controllerDevices) {
231  int xSize = min ? Integer.MAX_VALUE : -1;
232  ControllerNode xNode = null;
233  for (ControllerNode node : controllerDevices.keySet()) {
234  int size = controllerDevices.get(node).size();
235  if ((min && size < xSize) || (!min && size > xSize)) {
236  xSize = size;
237  xNode = node;
238  }
239  }
240  return xNode;
241  }
242 
243  private CompletableFuture<Void> balanceBuckets(ControllerNode smallest, ControllerNode largest,
244  Map<ControllerNode, Set<DeviceId>> controllerDevices,
245  int deviceCount) {
246  Collection<DeviceId> minBucket = controllerDevices.get(smallest);
247  Collection<DeviceId> maxBucket = controllerDevices.get(largest);
248  int bucketCount = controllerDevices.keySet().size();
249 
250  int delta = (maxBucket.size() - minBucket.size()) / 2;
251  delta = Math.min(deviceCount / bucketCount, delta);
252 
253  List<CompletableFuture<Void>> setRoleFutures = Lists.newLinkedList();
254 
255  if (delta > 0) {
256  log.info("Attempting to move {} nodes from {} to {}...", delta,
257  largest.id(), smallest.id());
258 
259  int i = 0;
260  Iterator<DeviceId> it = maxBucket.iterator();
261  while (it.hasNext() && i < delta) {
262  DeviceId deviceId = it.next();
263  log.info("Setting {} as the master for {}", smallest.id(), deviceId);
264  setRoleFutures.add(setRole(smallest.id(), deviceId, MASTER));
265  controllerDevices.get(smallest).add(deviceId);
266  it.remove();
267  i++;
268  }
269  }
270 
271  return CompletableFuture.allOf(setRoleFutures.toArray(new CompletableFuture[setRoleFutures.size()]));
272  }
