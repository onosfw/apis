150  @Activate
151  public void activate() {
152  localNodeId = clusterService.getLocalNode().id();
153 
154  Map<String, Set<NodeId>> partitionMap = Maps.newHashMap();
155  clusterMetadataService.getClusterMetadata().getPartitions().forEach(p -> {
156  partitionMap.put(p.getName(), Sets.newHashSet(p.getMembers()));
157  });
158 
159 
160  String[] activeNodeUris = partitionMap.values()
161  .stream()
162  .reduce((s1, s2) -> Sets.union(s1, s2))
163  .get()
164  .stream()
165  .map(this::nodeIdToUri)
166  .toArray(String[]::new);
167 
168  String localNodeUri = nodeIdToUri(clusterMetadataService.getLocalNode().id());
169  Protocol protocol = new CopycatCommunicationProtocol(clusterService, clusterCommunicator);
170 
171  ClusterConfig clusterConfig = new ClusterConfig()
172  .withProtocol(protocol)
173  .withElectionTimeout(electionTimeoutMillis(activeNodeUris))
174  .withHeartbeatInterval(heartbeatTimeoutMillis(activeNodeUris))
175  .withMembers(activeNodeUris)
176  .withLocalMember(localNodeUri);
177 
178  CopycatConfig copycatConfig = new CopycatConfig()
179  .withName("onos")
180  .withClusterConfig(clusterConfig)
181  .withDefaultSerializer(new DatabaseSerializer())
182  .withDefaultExecutor(Executors.newSingleThreadExecutor(new NamedThreadFactory("copycat-coordinator-%d")));
183 
184  coordinator = new DefaultClusterCoordinator(copycatConfig.resolve());
185 
186  DatabaseConfig inMemoryDatabaseConfig =
187  newDatabaseConfig(BASE_PARTITION_NAME, newInMemoryLog(), activeNodeUris);
188  inMemoryDatabase = coordinator
189  .getResource(inMemoryDatabaseConfig.getName(), inMemoryDatabaseConfig.resolve(clusterConfig)
190  .withSerializer(copycatConfig.getDefaultSerializer())
191  .withDefaultExecutor(copycatConfig.getDefaultExecutor()));
192 
193  List<Database> partitions = partitionMap.entrySet()
194  .stream()
195  .map(entry -> {
196  String[] replicas = entry.getValue().stream().map(this::nodeIdToUri).toArray(String[]::new);
197  return newDatabaseConfig(entry.getKey(), newPersistentLog(), replicas);
198  })
199  .map(config -> {
200  Database db = coordinator.getResource(config.getName(), config.resolve(clusterConfig)
201  .withSerializer(copycatConfig.getDefaultSerializer())
202  .withDefaultExecutor(copycatConfig.getDefaultExecutor()));
203  return db;
204  })
205  .collect(Collectors.toList());
206 
207  partitionedDatabase = new PartitionedDatabase("onos-store", partitions);
208 
209  CompletableFuture<Void> status = coordinator.open()
210  .thenCompose(v -> CompletableFuture.allOf(inMemoryDatabase.open(), partitionedDatabase.open())
211  .whenComplete((db, error) -> {
212  if (error != null) {
213  log.error("Failed to initialize database.", error);
214  } else {
215  log.info("Successfully initialized database.");
216  }
217  }));
218 
219  Futures.getUnchecked(status);
220 
221  transactionManager = new TransactionManager(partitionedDatabase, consistentMapBuilder());
222  partitionedDatabase.setTransactionManager(transactionManager);
223 
224  log.info("Started");
225  }
226 
227  @Deactivate
228  public void deactivate() {
229  CompletableFuture.allOf(inMemoryDatabase.close(), partitionedDatabase.close())
230  .thenCompose(v -> coordinator.close())
231  .whenComplete((result, error) -> {
232  if (error != null) {
233  log.warn("Failed to cleanly close databases.", error);
234  } else {
235  log.info("Successfully closed databases.");
236  }
237  });
238  ImmutableList.copyOf(maps.values()).forEach(this::unregisterMap);
239  if (applicationService != null) {
240  applicationService.removeListener(appListener);
241  }
242  log.info("Stopped");
243  }

393  private static <T> T complete(CompletableFuture<T> future) {
394  try {
395  return future.get(DATABASE_OPERATION_TIMEOUT_MILLIS, TimeUnit.MILLISECONDS);
396  } catch (InterruptedException e) {
397  Thread.currentThread().interrupt();

426  private class InternalApplicationListener implements ApplicationListener {
427  @Override
428  public void event(ApplicationEvent event) {
429  if (event.type() == APP_UNINSTALLED || event.type() == APP_DEACTIVATED) {
430  ApplicationId appId = event.subject().id();
431  List<DefaultAsyncConsistentMap> mapsToRemove;
432  synchronized (mapsByApplication) {
433  mapsToRemove = ImmutableList.copyOf(mapsByApplication.get(appId));
434  }
435  mapsToRemove.forEach(DatabaseManager.this::unregisterMap);
436  if (event.type() == APP_UNINSTALLED) {
437  mapsToRemove.stream().filter(map -> map.purgeOnUninstall()).forEach(map -> map.clear());
438  }
439  }
440  }
441  }
