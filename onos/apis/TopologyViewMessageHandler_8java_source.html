360 
361  if (type.equals(DEVICE)) {

366  HostId hid = hostId(id);
367  pp = hostDetails(hid, sid);
368  overlayCache.currentOverlay().modifyHostDetails(pp, hid);
369  }
370 
371  sendMessage(envelope(SHOW_DETAILS, sid, json(pp)));
372  }
