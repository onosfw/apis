147  // As per RFC the length of object should be
148  // multiples of 4
149  int pad = length % 4;
150  if (pad != 0) {
151  pad = 4 - pad;
152  for (int i = 0; i < pad; i++) {
153  cb.writeByte((byte) 0);
154  }
155  length = length + pad;
156  }
