74  public void checkIsDisjoint(Path<TestVertex, TestEdge> p, Map<TestEdge, Integer> risks) {
75  assertTrue("The path is not a DisjointPathPair", (p instanceof DisjointPathPair));
76  DisjointPathPair<TestVertex, TestEdge> q = (DisjointPathPair) p;
77  Set<Integer> p1Risks = new HashSet<>();
78  for (TestEdge e : q.edges()) {
79  p1Risks.add(risks.get(e));
80  }
81  if (!q.hasBackup()) {
82  return;
83  }
84  Path<TestVertex, TestEdge> pq = q.secondary();
85  for (TestEdge e: pq.edges()) {
86  assertTrue("The paths are not disjoint", !p1Risks.contains(risks.get(e)));
87  }
88  }
