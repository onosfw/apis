134  // OvsdbNodeId(IP) is used in the adaptor while DeviceId(ovsdb:IP)
135  // is used in the core. So DeviceId need be changed to OvsdbNodeId.
136  private OvsdbNodeId changeDeviceIdToNodeId(DeviceId deviceId) {
137  String[] splits = deviceId.toString().split(":");
138  if (splits == null || splits.length < 1) {
139  return null;
140  }
141  IpAddress ipAddress = IpAddress.valueOf(splits[1]);
142  return new OvsdbNodeId(ipAddress, 0);
143  }

145  // Used for getting OvsdbClientService.
146  private OvsdbClientService getOvsdbClientService(DriverHandler handler) {
147  OvsdbController ovsController = handler.get(OvsdbController.class);
148  DeviceId deviceId = handler.data().deviceId();
149  OvsdbNodeId nodeId = changeDeviceIdToNodeId(deviceId);
150  return ovsController.getOvsdbClient(nodeId);
151  }
