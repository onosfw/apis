40  private static final String FMT =
41  "id=%s, mac=%s, location=%s/%s, vlan=%s, ip(s)=%s%s";
42 
43  @Override
44  protected void execute() {
45  HostService service = get(HostService.class);
46  if (outputJson()) {
47  print("%s", json(getSortedHosts(service)));
48  } else {
49  for (Host host : getSortedHosts(service)) {

52  }
53  }

55  // Produces JSON structure.
56  private JsonNode json(Iterable<Host> hosts) {
57  ObjectMapper mapper = new ObjectMapper();
58  ArrayNode result = mapper.createArrayNode();
59 
60  hosts.forEach(host -> result.add(jsonForEntity(host, Host.class)));
61  return result;
62  }

64  /**

67  * @param service device service

70  protected List<Host> getSortedHosts(HostService service) {
71  List<Host> hosts = newArrayList(service.getHosts());
