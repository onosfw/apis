120  /**
121  * Write Error Object List and Open Object to channel buffer.
122  *
123  * @param cb output channel buffer
124  * @return length of written Error object list with open
125  * @throws PcepParseException when mandatory fields are not set
126  */
127  public int write(ChannelBuffer cb) throws PcepParseException {
128  int iLenStartIndex = cb.writerIndex();
129  boolean bIsErrObjListFound = false;
130 
131  //<error-obj-list> is mandatory , if not present throw exception.
132  if (llerrorObjList != null) {
133  ListIterator<PcepErrorObject> errObjListIterator = llerrorObjList.listIterator();
134  while (errObjListIterator.hasNext()) {
135  errObjListIterator.next().write(cb);
136  bIsErrObjListFound = true;
137  }
138  }
139 
140  if (!bIsErrObjListFound) {
141  throw new PcepParseException("<error-obj-list> is mandatory.");
142  }
143 
144  //Open Object is optional , if present write.
145  if (openObject != null) {
146  openObject.write(cb);
147  }
148 
149  return cb.writerIndex() - iLenStartIndex;
150  }
