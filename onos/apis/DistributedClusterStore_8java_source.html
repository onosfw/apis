197  private void heartbeat() {
198  try {
199  Set<ControllerNode> peers = allNodes.values()
200  .stream()
201  .filter(node -> !(node.id().equals(localNode.id())))
202  .collect(Collectors.toSet());
203  byte[] hbMessagePayload = SERIALIZER.encode(new HeartbeatMessage(localNode, peers));
204  peers.forEach((node) -> {
205  heartbeatToPeer(hbMessagePayload, node);
206  State currentState = nodeStates.get(node.id());
207  double phi = failureDetector.phi(node.id());
208  if (phi >= PHI_FAILURE_THRESHOLD) {
209  if (currentState == State.ACTIVE) {
210  updateState(node.id(), State.INACTIVE);
211  notifyStateChange(node.id(), State.ACTIVE, State.INACTIVE);
212  }
213  } else {
214  if (currentState == State.INACTIVE) {
215  updateState(node.id(), State.ACTIVE);
216  notifyStateChange(node.id(), State.INACTIVE, State.ACTIVE);
217  }
218  }
219  });
220  } catch (Exception e) {
221  log.debug("Failed to send heartbeat", e);
222  }
223  }
224 
225  private void notifyStateChange(NodeId nodeId, State oldState, State newState) {
226  ControllerNode node = allNodes.get(nodeId);
227  if (newState == State.ACTIVE) {
228  notifyDelegate(new ClusterEvent(ClusterEvent.Type.INSTANCE_ACTIVATED, node));
229  } else {
230  notifyDelegate(new ClusterEvent(ClusterEvent.Type.INSTANCE_DEACTIVATED, node));
231  }
232  }
233 
234  private void heartbeatToPeer(byte[] messagePayload, ControllerNode peer) {
235  Endpoint remoteEp = new Endpoint(peer.ip(), peer.tcpPort());
236  messagingService.sendAsync(remoteEp, HEARTBEAT_MESSAGE, messagePayload).whenComplete((result, error) -> {
237  if (error != null) {
238  log.trace("Sending heartbeat to {} failed", remoteEp, error);
239  }
240  });
241  }

247  failureDetector.report(hb.source().id());
