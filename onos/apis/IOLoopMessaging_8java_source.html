100  /**
101  * Activates IO Loops.
102  *
103  * @param localEp local end-point
104  * @throws IOException is activation fails
105  */
106  public void start(Endpoint localEp) throws IOException {
107  if (started.get()) {
108  log.warn("IOMessaging is already running at {}", localEp);
109  return;
110  }
111  this.localEp = localEp;
112  streams.setLifo(false);
113  this.acceptorLoop = new DefaultAcceptorLoop(new InetSocketAddress(localEp.host().toString(), localEp.port()));
114 
115  for (int i = 0; i < NUM_WORKERS; i++) {
116  ioLoops.add(new DefaultIOLoop(this::dispatchLocally));
117  }
118 
119  ioLoops.forEach(ioThreadPool::execute);
120  acceptorThreadPool.execute(acceptorLoop);
121  ioLoops.forEach(loop -> loop.awaitStart(TIMEOUT));
122  acceptorLoop.awaitStart(TIMEOUT);
123  started.set(true);
124  }
125 
126  /**
127  * Shuts down IO loops.
128  */
129  public void stop() {
130  if (started.get()) {
131  ioLoops.forEach(SelectorLoop::shutdown);
132  acceptorLoop.shutdown();
133  ioThreadPool.shutdown();
134  acceptorThreadPool.shutdown();
135  started.set(false);
136  }
137  }

150  protected CompletableFuture<Void> sendAsync(Endpoint ep, DefaultMessage message) {
151  CompletableFuture<Void> future = new CompletableFuture<>();
152  if (ep.equals(localEp)) {
153  dispatchLocally(message);
154  future.complete(null);
155  return future;
156  }
157 
158  DefaultMessageStream stream = null;
159  try {
160  stream = streams.borrowObject(ep);
161  stream.write(message);
162  future.complete(null);
163  } catch (Exception e) {
164  future.completeExceptionally(e);
165  } finally {
166  try {
167  streams.returnObject(ep, stream);
168  } catch (Exception e) {
169  log.warn("Failed to return stream to pool");
170  }
171  }
172  return future;
173  }

175  @Override
176  public CompletableFuture<byte[]> sendAndReceive(
177  Endpoint ep,
178  String type,
179  byte[] payload) {
180  CompletableFuture<byte[]> response = new CompletableFuture<>();
181  Long messageId = messageIdGenerator.incrementAndGet();
182  responseFutures.put(messageId, response);
183  DefaultMessage message = new DefaultMessage(messageId, localEp, type, payload);
184  try {
185  sendAsync(ep, message);
186  } catch (Exception e) {
187  responseFutures.invalidate(messageId);
188  response.completeExceptionally(e);

238  protected void dispatchLocally(DefaultMessage message) {
239  String type = message.type();
240  if (REPLY_MESSAGE_TYPE.equals(type)) {
241  try {
242  CompletableFuture<byte[]> futureResponse =
243  responseFutures.getIfPresent(message.id());
244  if (futureResponse != null) {
245  futureResponse.complete(message.payload());
246  } else {
247  log.warn("Received a reply for message id:[{}]. "
248  + " from {}. But was unable to locate the"
249  + " request handle", message.id(), message.sender());
250  }
251  } finally {
252  responseFutures.invalidate(message.id());
253  }
254  return;
255  }
256  Consumer<DefaultMessage> handler = handlers.get(type);
257  if (handler != null) {
258  handler.accept(message);
259  } else {
260  log.debug("No handler registered for {}", type);
261  }
262  }

270  /**

277  private DefaultMessageStream createConnection(Endpoint ep, DefaultIOLoop loop) throws IOException {
278  SocketAddress sa = new InetSocketAddress(ep.host().toString(), ep.port());
279  SocketChannel ch = SocketChannel.open();
280  ch.configureBlocking(false);
