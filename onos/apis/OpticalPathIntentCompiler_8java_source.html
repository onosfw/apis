74  @Override
75  public List<Intent> compile(OpticalPathIntent intent, List<Intent> installable,
76  Set<LinkResourceAllocations> resources) {
77  log.debug("Compiling optical path intent between {} and {}", intent.src(), intent.dst());
78 
79  // Create rules for forward and reverse path
80  List<FlowRule> rules = createRules(intent);
81  if (intent.isBidirectional()) {
82  rules.addAll(createReverseRules(intent));
83  }
84 
85  return Collections.singletonList(new FlowRuleIntent(appId, rules, intent.resources()));
86  }

88  /**
89  * Create rules for the forward path of the intent.
90  *
91  * @param intent the intent
92  * @return list of flow rules
93  */
94  private List<FlowRule> createRules(OpticalPathIntent intent) {
95  TrafficSelector.Builder selectorBuilder = DefaultTrafficSelector.builder();
96  selectorBuilder.matchInPort(intent.src().port());
97 
98  List<FlowRule> rules = new LinkedList<>();
99  ConnectPoint current = intent.src();
100 
101  for (Link link : intent.path().links()) {
102  TrafficTreatment.Builder treatmentBuilder = DefaultTrafficTreatment.builder();
103  treatmentBuilder.add(Instructions.modL0Lambda(intent.lambda()));
104  treatmentBuilder.setOutput(link.src().port());
105 
106  FlowRule rule = DefaultFlowRule.builder()
107  .forDevice(current.deviceId())
108  .withSelector(selectorBuilder.build())
109  .withTreatment(treatmentBuilder.build())
110  .withPriority(intent.priority())
111  .fromApp(appId)
112  .makePermanent()
113  .build();
114 
115  rules.add(rule);
116 
117  current = link.dst();
118  selectorBuilder.matchInPort(link.dst().port());
119  selectorBuilder.add(Criteria.matchLambda(intent.lambda()));
120  selectorBuilder.add(Criteria.matchOchSignalType(intent.signalType()));
121  }
122 
123  // Build the egress ROADM rule
124  TrafficTreatment.Builder treatmentLast = DefaultTrafficTreatment.builder();
125  treatmentLast.setOutput(intent.dst().port());
126 
127  FlowRule rule = new DefaultFlowRule.Builder()
128  .forDevice(intent.dst().deviceId())
129  .withSelector(selectorBuilder.build())
130  .withTreatment(treatmentLast.build())
131  .withPriority(intent.priority())
132  .fromApp(appId)
133  .makePermanent()
134  .build();
135  rules.add(rule);
136 
137  return rules;
138  }
