64  /**

72  protected Intent(ApplicationId appId,

76  checkState(idGenerator != null, "Id generator is not bound.");
77  checkArgument(priority <= MAX_PRIORITY && priority >= MIN_PRIORITY);
78  this.id = IntentId.valueOf(idGenerator.getNewId());
