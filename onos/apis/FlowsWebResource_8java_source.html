59  /**

64  @GET

66  public Response getFlows() {
67  final Iterable<Device> devices = get(DeviceService.class).getDevices();
68  for (final Device device : devices) {
69  final Iterable<FlowEntry> flowEntries = service.getFlowEntries(device.id());
70  if (flowEntries != null) {
71  for (final FlowEntry entry : flowEntries) {

130  /**
131  * Create new flow rule. Creates and installs a new flow rule for the
132  * specified device.
133  * @rsModel FlowsPost
134  * @param deviceId device identifier
135  * @param stream flow rule JSON
136  * @return status of the request - CREATED if the JSON is correct,
137  * BAD_REQUEST if the JSON is invalid
138  */
139  @POST
140  @Path("{deviceId}")
141  @Consumes(MediaType.APPLICATION_JSON)
142  @Produces(MediaType.APPLICATION_JSON)
143  public Response createFlow(@PathParam("deviceId") String deviceId,
144  InputStream stream) {
145  URI location;
146  try {
147  ObjectNode jsonTree = (ObjectNode) mapper().readTree(stream);
148  JsonNode specifiedDeviceId = jsonTree.get("deviceId");
149  if (specifiedDeviceId != null &&
150  !specifiedDeviceId.asText().equals(deviceId)) {
151  throw new IllegalArgumentException(
152  "Invalid deviceId in flow creation request");
153  }
154  jsonTree.put("deviceId", deviceId);
155  FlowRule rule = codec(FlowRule.class).decode(jsonTree, this);
156  service.applyFlowRules(rule);
157  location = new URI(Long.toString(rule.id().value()));
158  } catch (IOException | URISyntaxException ex) {
159  throw new IllegalArgumentException(ex);
160  }
161 
162  return Response
163  .created(location)
164  .build();
165  }

167  /**

169  *
170  * @param deviceId device identifier
171  * @param flowId flow rule identifier

173  @DELETE

175  @Path("{deviceId}/{flowId}")
176  public void deleteFlowByDeviceIdAndFlowId(@PathParam("deviceId") String deviceId,
177  @PathParam("flowId") long flowId) {
178  final Iterable<FlowEntry> flowEntries =
179  service.getFlowEntries(DeviceId.deviceId(deviceId));
180 
181  if (!flowEntries.iterator().hasNext()) {
182  throw new ItemNotFoundException(DEVICE_NOT_FOUND);
183  }
184 
185  StreamSupport.stream(flowEntries.spliterator(), false)
186  .filter(entry -> entry.id().value() == flowId)
187  .forEach(service::removeFlowRules);
