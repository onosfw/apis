124  @Activate
125  public synchronized void activate(ComponentContext context) {
126  cfgService.registerProperties(DefaultTopologyProvider.class);
127  executor = newFixedThreadPool(MAX_THREADS, groupedThreads("onos/topo", "build-%d"));
128  accumulator = new TopologyChangeAccumulator();
129  logConfig("Configured");

158  @Modified
159  public void modified(ComponentContext context) {
160  if (context == null) {
161  accumulator = new TopologyChangeAccumulator();
162  logConfig("Reconfigured");
163  return;
164  }
165 
166  Dictionary<?, ?> properties = context.getProperties();
167  int newMaxEvents, newMaxBatchMs, newMaxIdleMs;
168  try {
169  String s = get(properties, "maxEvents");
170  newMaxEvents = isNullOrEmpty(s) ? maxEvents : Integer.parseInt(s.trim());
171 
172  s = get(properties, "maxBatchMs");
173  newMaxBatchMs = isNullOrEmpty(s) ? maxBatchMs : Integer.parseInt(s.trim());
174 
175  s = get(properties, "maxIdleMs");
176  newMaxIdleMs = isNullOrEmpty(s) ? maxIdleMs : Integer.parseInt(s.trim());
177 
178  } catch (NumberFormatException | ClassCastException e) {
179  newMaxEvents = DEFAULT_MAX_EVENTS;
180  newMaxBatchMs = DEFAULT_MAX_BATCH_MS;
181  newMaxIdleMs = DEFAULT_MAX_IDLE_MS;
182  }
183 
184  if (newMaxEvents != maxEvents || newMaxBatchMs != maxBatchMs || newMaxIdleMs != maxIdleMs) {
185  maxEvents = newMaxEvents;
186  maxBatchMs = newMaxBatchMs;
187  maxIdleMs = newMaxIdleMs;
188  accumulator = maxEvents > 1 ? new TopologyChangeAccumulator() : null;
189  logConfig("Reconfigured");
190  }
191  }

199  @Override

204  /**
205  * Triggers assembly of topology data citing the specified events as the
206  * reason.
207  *
208  * @param reasons events which triggered the topology change
209  */
210  private synchronized void triggerTopologyBuild(List<Event> reasons) {
211  if (executor != null) {
212  executor.execute(new TopologyBuilderTask(reasons));
213  }
214  }
215 
216  // Builds the topology using the latest device and link information
217  // and citing the specified events as reasons for the change.
218  private void buildTopology(List<Event> reasons) {
219  if (isStarted) {
220  GraphDescription desc =
221  new DefaultGraphDescription(System.nanoTime(),
222  System.currentTimeMillis(),
223  deviceService.getAvailableDevices(),
224  linkService.getActiveLinks());
225  providerService.topologyChanged(desc, reasons);
226  }
227  }
228 
229  private void processEvent(Event event) {
230  if (accumulator != null) {
231  accumulator.add(event);
232  } else {
233  triggerTopologyBuild(ImmutableList.of(event));
234  }
235  }

237  // Callback for device events

239  @Override
240  public void event(DeviceEvent event) {
241  DeviceEvent.Type type = event.type();
242  if (type == DEVICE_ADDED || type == DEVICE_REMOVED ||
243  type == DEVICE_AVAILABILITY_CHANGED) {
244  processEvent(event);
245  }
246  }
247  }
