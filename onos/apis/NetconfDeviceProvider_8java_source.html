133  @Override
134  public boolean isReachable(DeviceId deviceId) {
135  Map<DeviceId, NetconfDevice> devices = controller.getDevicesMap();
136 
137  NetconfDevice netconfDevice = null;
138  for (DeviceId key : devices.keySet()) {
139  if (key.equals(deviceId)) {
140  netconfDevice = controller.getDevicesMap().get(key);
141  }
142  }
143  if (netconfDevice == null) {
144  log.warn("BAD REQUEST: the requested device id: "
145  + deviceId.toString()
146  + " is not associated to any NETCONF Device");
147  return false;
148  }
149  return netconfDevice.isActive();
150  }

190  private void connectDevices() {
191  NetconfProviderConfig cfg = cfgService.getConfig(appId, NetconfProviderConfig.class);
192  if (cfg != null) {
193  log.info("cfg {}", cfg);
194  try {
195  cfg.getDevicesAddresses().stream().forEach(addr -> controller
196  .connectDevice(new NetconfDeviceInfo(addr.name(),
197  addr.password(),
198  addr.ip(),
199  addr.port())));
200  } catch (ConfigException e) {
201  log.error("Cannot read config error " + e);
202  }
203  }
