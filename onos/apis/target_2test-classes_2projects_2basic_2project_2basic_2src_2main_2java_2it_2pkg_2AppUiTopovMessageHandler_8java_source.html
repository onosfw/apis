102  private final class DisplayStartHandler extends RequestHandler {
103  public DisplayStartHandler() {
104  super(SAMPLE_TOPOV_DISPLAY_START);
105  }
106 
107  @Override
108  public void process(long sid, ObjectNode payload) {
109  String mode = string(payload, MODE);
110 
111  log.debug("Start Display: mode [{}]", mode);
112  clearState();
113  clearForMode();
114 
115  switch (mode) {
116  case "mouse":
117  currentMode = Mode.MOUSE;
118  cancelTask();
119  sendMouseData();
120  break;
121 
122  case "link":
123  currentMode = Mode.LINK;
124  scheduleTask();
125  initLinkSet();
126  sendLinkData();
127  break;
128 
129  default:
130  currentMode = Mode.IDLE;
131  cancelTask();
132  break;
133  }
134  }
135  }

176  private void updateForMode(String id) {
177  log.debug("host service: {}", hostService);
178  log.debug("device service: {}", deviceService);
179 
180  try {
181  HostId hid = HostId.hostId(id);
182  log.debug("host id {}", hid);
183  elementOfNote = hostService.getHost(hid);
184  log.debug("host element {}", elementOfNote);
185 
186  } catch (Exception e) {
187  try {
188  DeviceId did = DeviceId.deviceId(id);
189  log.debug("device id {}", did);
190  elementOfNote = deviceService.getDevice(did);
191  log.debug("device element {}", elementOfNote);
192 
193  } catch (Exception e2) {
194  log.debug("Unable to process ID [{}]", id);
195  elementOfNote = null;
196  }
197  }
198 
199  switch (currentMode) {
200  case MOUSE:
201  sendMouseData();
202  break;
203 
204  case LINK:
205  sendLinkData();
206  break;
207 
208  default:
209  break;
210  }
211 
212  }

223  private void sendMouseData() {
224  if (elementOfNote != null && elementOfNote instanceof Device) {
225  DeviceId devId = (DeviceId) elementOfNote.id();
226  Set<Link> links = linkService.getDeviceEgressLinks(devId);
227  Highlights highlights = fromLinks(links, devId);

240  private NodeBadge createBadge(int n) {
241  Status status = n > 3 ? Status.ERROR : Status.WARN;
242  String noun = n > 3 ? "(critical)" : "(problematic)";
243  String msg = "Egress links: " + n + " " + noun;
244  return NodeBadge.number(status, n, msg);
245  }

247  private Highlights fromLinks(Set<Link> links, DeviceId devId) {
248  DemoLinkMap linkMap = new DemoLinkMap();
249  if (links != null) {
250  log.debug("Processing {} links", links.size());
251  links.forEach(linkMap::add);
252  } else {
253  log.debug("No egress links found for device {}", devId);
254  }
255 
256  Highlights highlights = new Highlights();
257 
258  for (DemoLink dlink : linkMap.biLinks()) {
259  dlink.makeImportant().setLabel("Yo!");
260  highlights.add(dlink.highlight(null));
261  }
262  return highlights;
263  }

265  private void initLinkSet() {
266  Set<Link> links = new HashSet<>();
267  for (Link link : linkService.getActiveLinks()) {
268  links.add(link);
269  }
270  linkSet = links.toArray(new Link[links.size()]);
271  linkIndex = 0;
272  log.debug("initialized link set to {}", linkSet.length);
273  }
274 
275  private void sendLinkData() {
276  DemoLinkMap linkMap = new DemoLinkMap();
277  for (Link link : linkSet) {
278  linkMap.add(link);
279  }
280  DemoLink dl = linkMap.add(linkSet[linkIndex]);
281  dl.makeImportant().setLabel(Integer.toString(linkIndex));
282  log.debug("sending link data (index {})", linkIndex);
283 
284  linkIndex += 1;
285  if (linkIndex >= linkSet.length) {
286  linkIndex = 0;
287  }
288 
289  Highlights highlights = new Highlights();
290  for (DemoLink dlink : linkMap.biLinks()) {
291  highlights.add(dlink.highlight(null));
292  }
293 
294  sendHighlights(highlights);
295  }

297  private synchronized void scheduleTask() {
298  if (demoTask == null) {
299  log.debug("Starting up demo task...");
300  demoTask = new DisplayUpdateTask();
301  timer.schedule(demoTask, UPDATE_PERIOD_MS, UPDATE_PERIOD_MS);
302  } else {
303  log.debug("(demo task already running");
304  }
305  }
306 
307  private synchronized void cancelTask() {
308  if (demoTask != null) {
309  demoTask.cancel();
310  demoTask = null;
311  }
312  }
