177  private class InternalStoreDelegate implements ApplicationStoreDelegate {
178  @Override
179  public void notify(ApplicationEvent event) {
180  ApplicationEvent.Type type = event.type();
181  Application app = event.subject();
182  try {
183  if (type == APP_ACTIVATED) {
184  if (installAppFeatures(app)) {
185  log.info("Application {} has been activated", app.id().name());
186  }
187 
188  } else if (type == APP_DEACTIVATED) {
189  if (uninstallAppFeatures(app)) {
190  log.info("Application {} has been deactivated", app.id().name());
191  }
192 
193  } else if (type == APP_INSTALLED) {
194  if (installAppArtifacts(app)) {
195  log.info("Application {} has been installed", app.id().name());
196  }
197 
198  } else if (type == APP_UNINSTALLED) {
199  if (uninstallAppFeatures(app) || uninstallAppArtifacts(app)) {
200  log.info("Application {} has been uninstalled", app.id().name());
201  }
202 
203  }
204  post(event);
205 
206  } catch (Exception e) {
207  log.warn("Unable to perform operation on application " + app.id().name(), e);
208  }
209  }
210  }

212  // The following methods are fully synchronized to guard against remote vs.
213  // locally induced feature service interactions.

215  // Installs all feature repositories required by the specified app.
216  private synchronized boolean installAppArtifacts(Application app) throws Exception {
217  if (app.featuresRepo().isPresent() &&
218  featuresService.getRepository(app.featuresRepo().get()) == null) {
219  featuresService.addRepository(app.featuresRepo().get());
220  return true;
221  }
222  return false;
223  }
224 
225  // Uninstalls all the feature repositories required by the specified app.
226  private synchronized boolean uninstallAppArtifacts(Application app) throws Exception {
227  if (app.featuresRepo().isPresent() &&
228  featuresService.getRepository(app.featuresRepo().get()) != null) {
229  featuresService.removeRepository(app.featuresRepo().get());
230  return true;
231  }
232  return false;
233  }
234 
235  // Installs all features that define the specified app.
236  private synchronized boolean installAppFeatures(Application app) throws Exception {
237  boolean changed = false;
238  for (String name : app.features()) {
239  Feature feature = featuresService.getFeature(name);
240 
241  // If we see an attempt at activation of a non-existent feature
242  // attempt to install the app artifacts first and then retry.
243  // This can be triggered by a race condition between different ONOS
244  // instances "installing" the apps from disk at their own pace.
245  // Perhaps there is a more elegant solution to be explored in the
246  // future.
247  if (feature == null) {
248  installAppArtifacts(app);
249  feature = featuresService.getFeature(name);
250  }
251 
252  if (feature != null && !featuresService.isInstalled(feature)) {
253  featuresService.installFeature(name);
254  changed = true;
255  } else if (feature == null) {
256  log.warn("Feature {} not found", name);
257  }
258  }
259  return changed;
260  }
261 
262  // Uninstalls all features that define the specified app.
263  private synchronized boolean uninstallAppFeatures(Application app) throws Exception {
264  boolean changed = false;
265  invokeHook(deactivateHooks.get(app.id().name()), app.id());
266  for (String name : app.features()) {
267  Feature feature = featuresService.getFeature(name);
268  if (feature != null && featuresService.isInstalled(feature)) {
269  featuresService.uninstallFeature(name);
270  changed = true;
271  } else if (feature == null) {
272  log.warn("Feature {} not found", name);
273  }
274  }
275  return changed;
276  }

278  // Invokes the specified function, if not null.
279  private void invokeHook(Runnable hook, ApplicationId appId) {
280  if (hook != null) {
281  try {
282  hook.run();
283  } catch (Exception e) {
284  log.warn("Deactivate hook for application {} encountered an error",
285  appId.name(), e);
286  }
287  }
