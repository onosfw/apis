56  @Override
57  protected void decode(
58  ChannelHandlerContext context,
59  ByteBuf buffer,
60  List<Object> out) throws Exception {
61 
62  switch (state()) {
63  case READ_MESSAGE_PREAMBLE:
64  preamble = buffer.readInt();
65  if (preamble != correctPreamble) {
66  throw new IllegalStateException("This message had an incorrect preamble.");
67  }
68  checkpoint(DecoderState.READ_MESSAGE_ID);
69  case READ_MESSAGE_ID:
70  messageId = buffer.readLong();
71  checkpoint(DecoderState.READ_SENDER_IP_VERSION);
72  case READ_SENDER_IP_VERSION:
73  ipVersion = buffer.readByte() == 0x0 ? Version.INET : Version.INET6;
74  checkpoint(DecoderState.READ_SENDER_IP);
75  case READ_SENDER_IP:
76  byte[] octets = new byte[IpAddress.byteLength(ipVersion)];
77  buffer.readBytes(octets);
78  senderIp = IpAddress.valueOf(ipVersion, octets);
79  checkpoint(DecoderState.READ_SENDER_PORT);
80  case READ_SENDER_PORT:
81  senderPort = buffer.readInt();
82  checkpoint(DecoderState.READ_MESSAGE_TYPE_LENGTH);
83  case READ_MESSAGE_TYPE_LENGTH:
84  messageTypeLength = buffer.readInt();
85  checkpoint(DecoderState.READ_MESSAGE_TYPE);
86  case READ_MESSAGE_TYPE:
87  byte[] messageTypeBytes = new byte[messageTypeLength];
88  buffer.readBytes(messageTypeBytes);
89  messageType = new String(messageTypeBytes, Charsets.UTF_8);
90  checkpoint(DecoderState.READ_CONTENT_LENGTH);
91  case READ_CONTENT_LENGTH:
92  contentLength = buffer.readInt();
93  checkpoint(DecoderState.READ_CONTENT);
94  case READ_CONTENT:
95  //TODO Perform a sanity check on the size before allocating
96  byte[] payload = new byte[contentLength];
97  buffer.readBytes(payload);
98  InternalMessage message = new InternalMessage(messageId,
99  new Endpoint(senderIp, senderPort),
100  messageType,
101  payload);
102  out.add(message);
103  checkpoint(DecoderState.READ_MESSAGE_PREAMBLE);
104  break;
105  default:
106  checkState(false, "Must not be here");
107  }
108  }
