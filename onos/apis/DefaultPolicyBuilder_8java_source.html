117  public static List<Permission> convertToJavaPermissions(Set<org.onosproject.security.Permission> permissions) {
118  List<Permission> result = Lists.newArrayList();
119  for (org.onosproject.security.Permission perm : permissions) {
120  Permission javaPerm = getPermission(perm);
121  if (javaPerm != null) {
122  if (javaPerm instanceof AppPermission) {
123  if (((AppPermission) javaPerm).getType() != null) {
124  AppPermission ap = (AppPermission) javaPerm;
125  result.add(ap);
126  if (serviceDirectory.containsKey(ap.getType())) {
127  for (String service : serviceDirectory.get(ap.getType())) {
128  result.add(new ServicePermission(service, ServicePermission.GET));
129  }
130  }
131  }
132  } else if (javaPerm instanceof ServicePermission) {
133  if (!javaPerm.getName().contains(SecurityAdminService.class.getName())) {
134  result.add(javaPerm);
135  }
136  } else {
137  result.add(javaPerm);
138  }
139 
140  }
141  }
142  return result;
143  }

366  private static Permission getPermission(org.onosproject.security.Permission permission) {
367 
368  String classname = permission.getClassName();
369  String name = permission.getName();
370  String actions = permission.getActions();
371 
372  if (classname == null || name == null) {
373  return null;
374  }
375  classname = classname.trim();
376  name = name.trim();
377  actions = actions.trim();
378 
379  if (AppPermission.class.getName().equals(classname)) {
380  return new AppPermission(name);
381  } else if (FilePermission.class.getName().equals(classname)) {
382  return new FilePermission(name, actions);
383  } else if (SerializablePermission.class.getName().equals(classname)) {
384  return new SerializablePermission(name, actions);
385  } else if (NetPermission.class.getName().equals(classname)) {
386  return new NetPermission(name, actions);
387  } else if (RuntimePermission.class.getName().equals(classname)) {
388  return new RuntimePermission(name, actions);
389  } else if (SocketPermission.class.getName().equals(classname)) {
390  return new SocketPermission(name, actions);
391  } else if (SQLPermission.class.getName().equals(classname)) {
392  return new SQLPermission(name, actions);
393  } else if (PropertyPermission.class.getName().equals(classname)) {
394  return new PropertyPermission(name, actions);
395  } else if (LoggingPermission.class.getName().equals(classname)) {
396  return new LoggingPermission(name, actions);
397  } else if (SSLPermission.class.getName().equals(classname)) {
398  return new SSLPermission(name, actions);
399  } else if (AuthPermission.class.getName().equals(classname)) {
400  return new AuthPermission(name, actions);
401  } else if (PrivateCredentialPermission.class.getName().equals(classname)) {
402  return new PrivateCredentialPermission(name, actions);
403  } else if (DelegationPermission.class.getName().equals(classname)) {
404  return new DelegationPermission(name, actions);
405  } else if (javax.security.auth.kerberos.ServicePermission.class.getName().equals(classname)) {
406  return new javax.security.auth.kerberos.ServicePermission(name, actions);
407  } else if (AudioPermission.class.getName().equals(classname)) {
408  return new AudioPermission(name, actions);
409  } else if (AdaptPermission.class.getName().equals(classname)) {
410  return new AdaptPermission(name, actions);
411  } else if (BundlePermission.class.getName().equals(classname)) {
412  return new BundlePermission(name, actions);
413  } else if (CapabilityPermission.class.getName().equals(classname)) {
414  return new CapabilityPermission(name, actions);
415  } else if (PackagePermission.class.getName().equals(classname)) {
416  return new PackagePermission(name, actions);
417  } else if (ServicePermission.class.getName().equals(classname)) {
418  return new ServicePermission(name, actions);

421  //AllPermission, SecurityPermission, UnresolvedPermission
