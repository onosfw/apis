47 /**
48  * This component cleans up intents that have encountered errors or otherwise
49  * stalled during installation or withdrawal.
50  * <p>
51  * It periodically polls (based on configured period) for pending and CORRUPT
52  * intents from the store and retries. It also listens for CORRUPT event
53  * notifications, which signify errors in processing, and retries.
54  * </p>
55  */
56 @Component(immediate = true)
57 public class IntentCleanup implements Runnable, IntentListener {
58

90  @Activate

110  @Modified
111  public void modified(ComponentContext context) {
112  Dictionary<?, ?> properties = context != null ? context.getProperties() : new Properties();
113 
114  int newPeriod;
115  boolean newEnabled;
116  try {
117  String s = get(properties, "period");
118  newPeriod = isNullOrEmpty(s) ? period : Integer.parseInt(s.trim());
119 
120  s = get(properties, "retryThreshold");
121  retryThreshold = isNullOrEmpty(s) ? retryThreshold : Integer.parseInt(s.trim());
122 
123  s = get(properties, "enabled");
124  newEnabled = isNullOrEmpty(s) ? enabled : Boolean.parseBoolean(s.trim());
125  } catch (NumberFormatException e) {
126  log.warn(e.getMessage());
127  newPeriod = period;
128  newEnabled = enabled;
129  }
130 
131  // Any change in the following parameters implies hard restart
132  if (newPeriod != period || enabled != newEnabled) {
133  period = newPeriod;
134  enabled = newEnabled;
135  adjustRate();
136  }
137 
138  log.info("Settings: enabled={}, period={}, retryThreshold={}",
139  enabled, period, retryThreshold);
140  }

142  protected void adjustRate() {
143  if (timerTask != null) {
144  timerTask.cancel();
145  timerTask = null;
146  }
147 
148  if (enabled) {
149  timerTask = new TimerTask() {
150  @Override
151  public void run() {
152  executor.submit(IntentCleanup.this);
153  }
154  };
155 
156  periodMs = period * 1_000; //convert to ms
157  timer.scheduleAtFixedRate(timerTask, periodMs, periodMs);
158  }
159  }

162  @Override

171  private void resubmitCorrupt(IntentData intentData, boolean checkThreshold) {
172  if (checkThreshold && intentData.errorCount() >= retryThreshold) {
173  return; // threshold met or exceeded
174  }
175 
176  switch (intentData.request()) {
177  case INSTALL_REQ:
178  service.submit(intentData.intent());
179  break;
180  case WITHDRAW_REQ:
181  service.withdraw(intentData.intent());
182  break;
183  default:
184  log.warn("Trying to resubmit corrupt/failed intent {} in state {} with request {}",
185  intentData.key(), intentData.state(), intentData.request());
186  break;
187  }
188  }
189 
190  private void resubmitPendingRequest(IntentData intentData) {
191  switch (intentData.request()) {
192  case INSTALL_REQ:
193  service.submit(intentData.intent());
194  break;
195  case WITHDRAW_REQ:
196  service.withdraw(intentData.intent());
197  break;
198  default:
199  log.warn("Failed to resubmit pending intent {} in state {} with request {}",
200  intentData.key(), intentData.state(), intentData.request());
201  break;
202  }
203  }
204 
205  /**
206  * Iterates through corrupt, failed and pending intents and
207  * re-submit/withdraw appropriately.
208  */
209  private void cleanup() {
210  int corruptCount = 0, failedCount = 0, stuckCount = 0, pendingCount = 0;
211 
212  for (IntentData intentData : store.getIntentData(true, periodMs)) {
213  switch (intentData.state()) {
214  case FAILED:
215  resubmitCorrupt(intentData, false);
216  failedCount++;
217  break;
218  case CORRUPT:
219  resubmitCorrupt(intentData, false);
220  corruptCount++;
221  break;
222  case INSTALLING: //FALLTHROUGH
223  case WITHDRAWING:
224  resubmitPendingRequest(intentData);
225  stuckCount++;
226  break;
227  default:
228  //NOOP
229  break;
230  }
231  }
232 
233  for (IntentData intentData : store.getPendingData(true, periodMs)) {
234  resubmitPendingRequest(intentData);
235  stuckCount++;
236  }
237 
238  if (corruptCount + failedCount + stuckCount + pendingCount > 0) {
239  log.debug("Intent cleanup ran and resubmitted {} corrupt, {} failed, {} stuck, and {} pending intents",
240  corruptCount, failedCount, stuckCount, pendingCount);
241  }
242  }

244  @Override
245  public void event(IntentEvent event) {
246  // this is the fast path for CORRUPT intents, retry on event notification.
247  //TODO we might consider using the timer to back off for subsequent retries
248  if (enabled && event.type() == IntentEvent.Type.CORRUPT) {
249  Key key = event.subject().key();
250  if (store.isMaster(key)) {
251  IntentData data = store.getIntentData(event.subject().key());
252  resubmitCorrupt(data, true);
253  }
254  }
255  }
