178  * @param routeUpdates collection of route updates to process

180  private void update(Collection<RouteUpdate> routeUpdates) {
181  try {

189  /**
190  * Thread for handling route updates.
191  */
192  private void doUpdatesThread() {
193  boolean interrupted = false;
194  try {
195  while (!interrupted) {
196  try {
197  Collection<RouteUpdate> routeUpdates =
198  routeUpdatesQueue.take();
199  processRouteUpdates(routeUpdates);
200  } catch (InterruptedException e) {
201  log.error("Interrupted while taking from updates queue", e);
202  interrupted = true;
203  } catch (Exception e) {
204  log.error("exception", e);
205  }
206  }
207  } finally {
208  if (interrupted) {
209  Thread.currentThread().interrupt();
210  }
211  }

302  /**
303  * Processes route updates.
304  *
305  * @param routeUpdates the route updates to process
306  */
307  void processRouteUpdates(Collection<RouteUpdate> routeUpdates) {
308  synchronized (this) {
309  Collection<IpPrefix> withdrawPrefixes = new LinkedList<>();
310  Collection<FibUpdate> fibUpdates = new LinkedList<>();
311  Collection<FibUpdate> fibWithdraws = new LinkedList<>();
312 
313  for (RouteUpdate update : routeUpdates) {
314  switch (update.type()) {
315  case UPDATE:
316 
317  FibEntry fib = processRouteAdd(update.routeEntry(),
318  withdrawPrefixes);
319  if (fib != null) {
320  fibUpdates.add(new FibUpdate(FibUpdate.Type.UPDATE, fib));
321  }
322 
323  break;
324  case DELETE:
325  processRouteDelete(update.routeEntry(), withdrawPrefixes);
326 
327  break;
328  default:
329  log.error("Unknown update Type: {}", update.type());
330  break;
331  }
332  }
333 
334  withdrawPrefixes.forEach(p -> fibWithdraws.add(new FibUpdate(
335  FibUpdate.Type.DELETE, new FibEntry(p, null, null))));
336 
337  if (!fibUpdates.isEmpty() || !fibWithdraws.isEmpty()) {
338  fibComponent.update(fibUpdates, fibWithdraws);
339  }
340  }
341  }

343  /**
344  * Processes adding a route entry.
345  * <p>
346  * The route entry is added to the radix tree. If there was an existing
347  * next hop for this prefix, but the next hop was different, then the
348  * old route entry is deleted.
349  * </p>
350  * <p>
351  * NOTE: Currently, we don't handle routes if the next hop is within the
352  * SDN domain.
353  * </p>
354  *
355  * @param routeEntry the route entry to add
356  * @param withdrawPrefixes the collection of accumulated prefixes whose
357  * intents will be withdrawn
358  * @return the corresponding FIB entry change, or null
359  */
360  private FibEntry processRouteAdd(RouteEntry routeEntry,
361  Collection<IpPrefix> withdrawPrefixes) {
362  log.debug("Processing route add: {}", routeEntry);
363 
364  // Find the old next-hop if we are updating an old route entry
365  IpAddress oldNextHop = null;
366  RouteEntry oldRouteEntry = findRibRoute(routeEntry.prefix());
367  if (oldRouteEntry != null) {
368  oldNextHop = oldRouteEntry.nextHop();
369  }
370 
371  // Add the new route to the RIB
372  addRibRoute(routeEntry);
373 
374  if (oldNextHop != null) {
375  if (oldNextHop.equals(routeEntry.nextHop())) {
376  return null; // No change
377  }
378  //
379  // Update an existing nexthop for the prefix.
380  // We need to remove the old flows for this prefix from the
381  // switches before the new flows are added.
382  //
383  withdrawPrefixes.add(routeEntry.prefix());
384  }
385 
386  if (routingConfigurationService.isIpPrefixLocal(routeEntry.prefix())) {
387  // Route originated by local SDN domain
388  // We don't handle these here, reactive routing APP will handle
389  // these
390  log.debug("Own route {} to {}",
391  routeEntry.prefix(), routeEntry.nextHop());
392  return null;
393  }
394 
395  //
396  // Find the MAC address of next hop router for this route entry.
397  // If the MAC address can not be found in ARP cache, then this prefix
398  // will be put in routesWaitingOnArp queue. Otherwise, generate
399  // a new route intent.
400  //
401 
402  // Monitor the IP address for updates of the MAC address
403  hostService.startMonitoringIp(routeEntry.nextHop());
404 
405  // Check if we know the MAC address of the next hop
406  MacAddress nextHopMacAddress = ip2Mac.get(routeEntry.nextHop());
407  if (nextHopMacAddress == null) {
408  Set<Host> hosts = hostService.getHostsByIp(routeEntry.nextHop());
409  if (!hosts.isEmpty()) {
410  nextHopMacAddress = hosts.iterator().next().mac();
411  }
412  if (nextHopMacAddress != null) {
413  ip2Mac.put(routeEntry.nextHop(), nextHopMacAddress);
414  }
415  }
416  if (nextHopMacAddress == null) {
417  routesWaitingOnArp.put(routeEntry.nextHop(), routeEntry);
418  return null;
419  }
420  return new FibEntry(routeEntry.prefix(), routeEntry.nextHop(),
421  nextHopMacAddress);
422  }
423 
424  /**
425  * Processes the deletion of a route entry.
426  * <p>
427  * The prefix for the routing entry is removed from radix tree.
428  * If the operation is successful, the prefix is added to the collection
429  * of prefixes whose intents that will be withdrawn.
430  * </p>
431  *
432  * @param routeEntry the route entry to delete
433  * @param withdrawPrefixes the collection of accumulated prefixes whose
434  * intents will be withdrawn
435  */
436  private void processRouteDelete(RouteEntry routeEntry,
437  Collection<IpPrefix> withdrawPrefixes) {
438  log.debug("Processing route delete: {}", routeEntry);
439  boolean isRemoved = removeRibRoute(routeEntry.prefix());
440 
441  if (isRemoved) {
442  //
443  // Only withdraw intents if an entry was actually removed from the
444  // tree. If no entry was removed, the <prefix, nexthop> wasn't
445  // there so it's probably already been removed and we don't
446  // need to do anything.
447  //
448  withdrawPrefixes.add(routeEntry.prefix());
449  }
450 
451  routesWaitingOnArp.remove(routeEntry.nextHop(), routeEntry);
452  }
453 
454  /**
455  * Signals the Router that the MAC to IP mapping has potentially been
456  * updated. This has the effect of updating the MAC address for any
457  * installed prefixes if it has changed, as well as installing any pending
458  * prefixes that were waiting for MAC resolution.
459  *
460  * @param ipAddress the IP address that an event was received for
461  * @param macAddress the most recently known MAC address for the IP address
462  */
463  private void updateMac(IpAddress ipAddress, MacAddress macAddress) {
464  log.debug("Received updated MAC info: {} => {}", ipAddress,
465  macAddress);
466 
467  //
468  // We synchronize on "this" to prevent changes to the Radix tree
469  // while we're pushing intents. If the tree changes, the
470  // tree and the intents could get out of sync.
471  //
472  synchronized (this) {
473  Collection<FibUpdate> submitFibEntries = new LinkedList<>();
474 
475  Set<RouteEntry> routesToPush =
476  routesWaitingOnArp.removeAll(ipAddress);
477 
478  for (RouteEntry routeEntry : routesToPush) {
479  // These will always be adds
480  RouteEntry foundRouteEntry = findRibRoute(routeEntry.prefix());
481  if (foundRouteEntry != null &&
482  foundRouteEntry.nextHop().equals(routeEntry.nextHop())) {
483  // We only push FIB updates if the prefix is still in the
484  // radix tree and the next hop is the same as our entry.
485  // The prefix could have been removed while we were waiting
486  // for the ARP, or the next hop could have changed.
487  submitFibEntries.add(new FibUpdate(FibUpdate.Type.UPDATE,
488  new FibEntry(routeEntry.prefix(),
489  ipAddress, macAddress)));
490  } else {
491  log.debug("{} has been revoked before the MAC was resolved",
492  routeEntry);
493  }
494  }
495 
496  if (!submitFibEntries.isEmpty()) {
497  fibComponent.update(submitFibEntries, Collections.emptyList());
498  }
499 
500  ip2Mac.put(ipAddress, macAddress);
501  }
502  }

507  class InternalHostListener implements HostListener {
508  @Override
509  public void event(HostEvent event) {
510  log.debug("Received HostEvent {}", event);
511 
512  Host host = event.subject();
513  switch (event.type()) {
514  case HOST_ADDED:
515  // FALLTHROUGH
516  case HOST_UPDATED:
517  for (IpAddress ipAddress : host.ipAddresses()) {
518  updateMac(ipAddress, host.mac());
519  }
520  break;
521  case HOST_REMOVED:
522  for (IpAddress ipAddress : host.ipAddresses()) {
523  ip2Mac.remove(ipAddress);
524  }
525  break;
526  default:
527  break;
528  }
529  }
