286  localInfo.setAddress(ctx.getChannel().getLocalAddress());

289  // Assign the local and remote IPv4 addresses
290  InetAddress inetAddr;
291  if (localInfo.address() instanceof InetSocketAddress) {
292  inetAddr = ((InetSocketAddress) localInfo.address()).getAddress();
293  localInfo.setIp4Address(Ip4Address.valueOf(inetAddr.getAddress()));
294  }
295  if (remoteInfo.address() instanceof InetSocketAddress) {
296  inetAddr = ((InetSocketAddress) remoteInfo.address()).getAddress();
297  remoteInfo.setIp4Address(Ip4Address.valueOf(inetAddr.getAddress()));
298  }
299 
300  log.debug("BGP Session Connected from {} on {}",
301  remoteInfo.address(), localInfo.address());
302  if (!bgpSessionManager.peerConnected(this)) {
303  log.debug("Cannot setup BGP Session Connection from {}. Closing...",
304  remoteInfo.address());
305  ctx.getChannel().close();
306  }

308  //
309  // Assign the local BGP ID
310  // NOTE: This should be configuration-based
311  //

333  /**
334  * Processes the channel being disconnected.
335  */
336  private void processChannelDisconnected() {
337  //
338  // Withdraw the routes advertised by this BGP peer
339  //
340  // NOTE: We must initialize the RIB-IN before propagating the withdraws
341  // for further processing. Otherwise, the BGP Decision Process
342  // will use those routes again.
343  //
344  Collection<BgpRouteEntry> deletedRoutes4 = bgpRibIn4.values();
345  Collection<BgpRouteEntry> deletedRoutes6 = bgpRibIn6.values();
346  bgpRibIn4 = new ConcurrentHashMap<>();
347  bgpRibIn6 = new ConcurrentHashMap<>();
348 
349  // Push the updates to the BGP Merged RIB
350  BgpRouteSelector bgpRouteSelector =
351  bgpSessionManager.getBgpRouteSelector();
352  Collection<BgpRouteEntry> addedRoutes = Collections.emptyList();
353  bgpRouteSelector.routeUpdates(this, addedRoutes, deletedRoutes4);
354  bgpRouteSelector.routeUpdates(this, addedRoutes, deletedRoutes6);
355 
356  bgpSessionManager.peerDisconnected(this);
357  }

359  /**
360  * Restarts the BGP KeepaliveTimer.
361  *
362  * @param ctx the Channel Handler Context to use
363  */
364  void restartKeepaliveTimer(ChannelHandlerContext ctx) {
365  long localKeepaliveInterval = 0;
366 
367  //
368  // Compute the local Keepalive interval
369  //
370  if (localInfo.holdtime() != 0) {
371  localKeepaliveInterval = Math.max(localInfo.holdtime() /
372  BgpConstants.BGP_KEEPALIVE_PER_HOLD_INTERVAL,
373  BgpConstants.BGP_KEEPALIVE_MIN_INTERVAL);
374  }
375 
376  // Restart the Keepalive timer
377  if (localKeepaliveInterval == 0) {
378  return; // Nothing to do
379  }
380  keepaliveTimeout = timer.newTimeout(new TransmitKeepaliveTask(ctx),
381  localKeepaliveInterval,
382  TimeUnit.SECONDS);
383  }
