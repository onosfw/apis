135  @Override
136  public synchronized void register(UiExtension extension) {
137  if (!extensions.contains(extension)) {
138  extensions.add(extension);
139  for (UiView view : extension.views()) {
140  views.put(view.id(), extension);
141  }
142  }
143  }
144 
145  @Override
146  public synchronized void unregister(UiExtension extension) {
147  extensions.remove(extension);
148  extension.views().stream()
149  .map(UiView::id).collect(toSet()).forEach(views::remove);
150  }
151 
152  @Override
153  public synchronized List<UiExtension> getExtensions() {
