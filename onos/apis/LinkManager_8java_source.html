250  // returns a LinkDescription made from the union of the BasicLinkConfig
251  // annotations if it exists
252  private LinkDescription validateLink(LinkDescription linkDescription) {
253  // TODO Investigate whether this can be made more efficient
254  BasicLinkConfig cfg = networkConfigService.getConfig(linkKey(linkDescription.src(),
255  linkDescription.dst()),
256  BasicLinkConfig.class);
257  BasicLinkConfig cfgTwo = networkConfigService.getConfig(linkKey(linkDescription.dst(),
258  linkDescription.src()),
259  BasicLinkConfig.class);
260  if (isAllowed(cfg) && isAllowed(cfgTwo)) {
261  return BasicLinkOperator.combine(cfg, linkDescription);
262  } else {
263  log.trace("Link " + linkDescription.toString() + " is not allowed");
264  return null;
265  }
266  }

303  // Removes all links in the specified set and emits appropriate events.
304  private void removeLinks(Set<Link> links, boolean isSoftRemove) {
305  for (Link link : links) {
306  LinkEvent event = isSoftRemove ?
307  store.removeOrDownLink(link.src(), link.dst()) :
308  store.removeLink(link.src(), link.dst());
309  if (event != null) {
310  log.info("Link {} removed/vanished", event.subject());
311  post(event);
312  }
313  }
314  }

324  // listens for NetworkConfigEvents of type BasicLinkConfig and removes
325  // links that the config does not allow
326  private class InternalNetworkConfigListener implements NetworkConfigListener {
327 
328  @Override
329  public boolean isRelevant(NetworkConfigEvent event) {
330  return event.configClass().equals(BasicLinkConfig.class)
331  && (event.type() == NetworkConfigEvent.Type.CONFIG_ADDED
332  || event.type() == NetworkConfigEvent.Type.CONFIG_UPDATED);
333  }
334 
335  @Override
336  public void event(NetworkConfigEvent event) {
337  LinkKey lk = (LinkKey) event.subject();
338  BasicLinkConfig cfg = networkConfigService.getConfig(lk, BasicLinkConfig.class);
339 
340  if (!isAllowed(cfg)) {
341  log.info("Kicking out links between {} and {}", lk.src(), lk.dst());
342  removeLink(lk.src(), lk.dst());
343  removeLink(lk.dst(), lk.src());
344  return;
345  }
346  Link link = getLink(lk.src(), lk.dst());
347  LinkDescription fldesc;
348  LinkDescription rldesc;
349  if (link == null) {
350  fldesc = BasicLinkOperator.descriptionOf(lk.src(), lk.dst(), cfg);
351  rldesc = BasicLinkOperator.descriptionOf(lk.dst(), lk.src(), cfg);
352  } else {
353  fldesc = BasicLinkOperator.combine(cfg,
354  BasicLinkOperator.descriptionOf(lk.src(), lk.dst(), link));
355  rldesc = BasicLinkOperator.combine(cfg,
356  BasicLinkOperator.descriptionOf(lk.dst(), lk.src(), link));
357  }
358  // XXX think of sane way to fetch the LinkProvider
359  store.createOrUpdateLink(ProviderId.NONE, fldesc);
360  store.createOrUpdateLink(ProviderId.NONE, rldesc);
361  }
