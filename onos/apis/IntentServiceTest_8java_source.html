128  /**
129  * Validates that the test event listener has received the following events
130  * for the specified intent. Events received for other intents will not be
131  * considered.
132  *
133  * @param intent intent subject
134  * @param types list of event types for which events are expected
135  */
136  protected void validateEvents(Intent intent, IntentEvent.Type... types) {
137  Iterator<IntentEvent> events = listener.events.iterator();
138  for (IntentEvent.Type type : types) {
139  IntentEvent event = events.hasNext() ? events.next() : null;
140  if (event == null) {
141  fail("expected event not found: " + type);
142  } else if (intent.equals(event.subject())) {
143  assertEquals("incorrect state", type, event.type());
144  }
145  }
146 
147  // Remainder of events should not apply to this intent; make sure.
148  while (events.hasNext()) {
149  assertFalse("unexpected event for intent",
150  intent.equals(events.next().subject()));
151  }
152  }
