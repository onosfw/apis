101  @Override
102  public void event(IntentEvent event) {
103  events.put(event.type(), event);
104  if (latchMap.containsKey(event.type())) {
105  latchMap.get(event.type()).countDown();

107  }
108 
109  public int getCounts(IntentEvent.Type type) {
