162  protected CompletableFuture<Void> sendAsync(Endpoint ep, InternalMessage message) {
163  CompletableFuture<Void> future = new CompletableFuture<>();
164  try {
165  if (ep.equals(localEp)) {
166  dispatchLocally(message);
167  future.complete(null);
168  } else {
169  Channel channel = null;
170  try {
171  channel = channels.borrowObject(ep);
172  channel.writeAndFlush(message).addListener(channelFuture -> {
173  if (!channelFuture.isSuccess()) {
174  future.completeExceptionally(channelFuture.cause());
175  } else {
176  future.complete(null);
177  }
178  });
179  } finally {
180  channels.returnObject(ep, channel);
181  }
182  }
183  } catch (Exception e) {
184  future.completeExceptionally(e);
185  }
186  return future;
187  }

251  private void startAcceptingConnections() throws InterruptedException {
252  ServerBootstrap b = new ServerBootstrap();
253  b.option(ChannelOption.WRITE_BUFFER_HIGH_WATER_MARK, 32 * 1024);
254  b.option(ChannelOption.WRITE_BUFFER_LOW_WATER_MARK, 8 * 1024);
255  b.option(ChannelOption.SO_RCVBUF, 1048576);
256  b.option(ChannelOption.TCP_NODELAY, true);
257  b.option(ChannelOption.ALLOCATOR, PooledByteBufAllocator.DEFAULT);
258  b.group(serverGroup, clientGroup);
259  b.channel(serverChannelClass);
260  if (enableNettyTls) {
261  b.childHandler(new SslServerCommunicationChannelInitializer());
262  } else {
263  b.childHandler(new OnosCommunicationChannelInitializer());
264  }
265  b.option(ChannelOption.SO_BACKLOG, 128);
266  b.childOption(ChannelOption.SO_KEEPALIVE, true);
267 
268  // Bind and start to accept incoming connections.
269  b.bind(localEp.port()).sync().addListener(future -> {
270  if (future.isSuccess()) {
271  log.info("{} accepting incoming connections on port {}", localEp.host(), localEp.port());
272  } else {
273  log.warn("{} failed to bind to port {}", localEp.host(), localEp.port(), future.cause());
274  }
275  });

334  KeyStore ts = KeyStore.getInstance("JKS");
335  ts.load(new FileInputStream(tsLocation), tsPwd);
336  tmFactory.init(ts);
337 
338  KeyManagerFactory kmf = KeyManagerFactory.getInstance(KeyManagerFactory.getDefaultAlgorithm());
339  KeyStore ks = KeyStore.getInstance("JKS");
340  ks.load(new FileInputStream(ksLocation), ksPwd);
341  kmf.init(ks, ksPwd);
342 
343  SSLContext serverContext = SSLContext.getInstance("TLS");
344  serverContext.init(kmf.getKeyManagers(), tmFactory.getTrustManagers(), null);
345 
346  SSLEngine serverSslEngine = serverContext.createSSLEngine();
347 
348  serverSslEngine.setNeedClientAuth(true);
349  serverSslEngine.setUseClientMode(false);
350  serverSslEngine.setEnabledProtocols(serverSslEngine.getSupportedProtocols());
351  serverSslEngine.setEnabledCipherSuites(serverSslEngine.getSupportedCipherSuites());
352  serverSslEngine.setEnableSessionCreation(true);
353 
354  channel.pipeline().addLast("ssl", new io.netty.handler.ssl.SslHandler(serverSslEngine))
355  .addLast("encoder", encoder)
356  .addLast("decoder", new MessageDecoder(preamble))
357  .addLast("handler", dispatcher);

429  private void dispatchLocally(InternalMessage message) throws IOException {
430  String type = message.type();
431  if (REPLY_MESSAGE_TYPE.equals(type)) {
432  try {
433  CompletableFuture<byte[]> futureResponse =
434  responseFutures.getIfPresent(message.id());
435  if (futureResponse != null) {
436  futureResponse.complete(message.payload());
437  } else {
438  log.warn("Received a reply for message id:[{}]. "
439  + " from {}. But was unable to locate the"
440  + " request handle", message.id(), message.sender());
441  }
442  } finally {
443  responseFutures.invalidate(message.id());
444  }
445  return;
446  }
447  Consumer<InternalMessage> handler = handlers.get(type);
448  if (handler != null) {
449  handler.accept(message);
450  } else {
451  log.debug("No handler registered for {}", type);
452  }
453  }
