49  // Provides an out-of-thread simulation of intent submit life-cycle
50  private void executeSubmit(final Intent intent) {
51  registerSubclassCompilerIfNeeded(intent);
52  executor.execute(() -> {

82  private <T extends Intent> void executeCompilingPhase(T intent) {
83  setState(intent, IntentState.COMPILING);
84  try {
85  // For the fake, we compile using a single level pass
86  List<Intent> installable = new ArrayList<>();
87  for (Intent compiled : getCompiler(intent).compile(intent, null, null)) {
88  installable.add(compiled);
89  }
90  executeInstallingPhase(intent, installable);
91 
92  } catch (IntentException e) {
93  setState(intent, IntentState.FAILED);
94  dispatch(new IntentEvent(IntentEvent.Type.FAILED, intent));
95  }
96  }
97 
98  private void executeInstallingPhase(Intent intent,
99  List<Intent> installable) {
100  setState(intent, IntentState.INSTALLING);
101  try {
102  setState(intent, IntentState.INSTALLED);
103  putInstallable(intent.key(), installable);
104  dispatch(new IntentEvent(IntentEvent.Type.INSTALLED, intent));
105 
106  } catch (IntentException e) {
107  setState(intent, IntentState.FAILED);
108  dispatch(new IntentEvent(IntentEvent.Type.FAILED, intent));
109  }
110  }

112  private void executeWithdrawingPhase(Intent intent,
113  List<Intent> installable) {
114  setState(intent, IntentState.WITHDRAWING);
115  try {
116  removeInstallable(intent.key());
117  setState(intent, IntentState.WITHDRAWN);

119  } catch (IntentException e) {
120  // FIXME: Rework this to always go from WITHDRAWING to WITHDRAWN!

123  }

245  private void registerSubclassCompilerIfNeeded(Intent intent) {
246  if (!compilers.containsKey(intent.getClass())) {
247  Class<?> cls = intent.getClass();
248  while (cls != Object.class) {
249  // As long as we're within the Intent class descendants
250  if (Intent.class.isAssignableFrom(cls)) {
251  IntentCompiler<?> compiler = compilers.get(cls);
252  if (compiler != null) {
253  compilers.put(intent.getClass(), compiler);
254  return;
255  }
256  }
257  cls = cls.getSuperclass();
258  }
259  }
260  }
