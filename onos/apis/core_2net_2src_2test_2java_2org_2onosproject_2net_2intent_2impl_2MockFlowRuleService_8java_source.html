46  @Override
47  public void apply(FlowRuleOperations ops) {
48  AtomicBoolean thisSuccess = new AtomicBoolean(success);
49  ops.stages().forEach(stage -> stage.forEach(flow -> {
50  if (errorFlow == flow.rule().id().value()) {
51  thisSuccess.set(false);
52  } else {
53  switch (flow.type()) {
54  case ADD:
55  case MODIFY: //TODO is this the right behavior for modify?
56  flows.add(flow.rule());
57  break;
58  case REMOVE:
59  flows.remove(flow.rule());
60  break;
61  default:
62  break;
63  }
64  }
65  }));
66  if (thisSuccess.get()) {
67  ops.callback().onSuccess(ops);
68  } else {
69  ops.callback().onError(ops);
70  }
71  }
