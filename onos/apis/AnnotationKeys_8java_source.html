139  /**

142  * If parsing fails, the returned value will be 1.0.
143  *
144  * @param annotated annotated object whose annotated value is obtained
145  * @param key key of annotation
146  * @return double value of annotated object for the specified key
147  */
148  public static double getAnnotatedValue(Annotated annotated, String key) {
149  double value;
150  try {
151  value = Double.parseDouble(annotated.annotations().value(key));
152  } catch (NumberFormatException e) {
153  value = 1.0;
154  }
155  return value;
