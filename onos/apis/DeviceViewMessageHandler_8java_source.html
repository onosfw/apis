146  private void populateRow(TableModel.Row row, Device dev,
147  DeviceService ds, MastershipService ms) {
148  DeviceId id = dev.id();
149  boolean available = ds.isAvailable(id);
150  String iconId = available ? ICON_ID_ONLINE : ICON_ID_OFFLINE;
151 
152  row.cell(ID, id)
153  .cell(NAME, deviceName(dev))
154  .cell(AVAILABLE, available)
155  .cell(AVAILABLE_IID, iconId)
156  .cell(TYPE_IID, getTypeIconId(dev))
157  .cell(MFR, dev.manufacturer())
158  .cell(HW, dev.hwVersion())
159  .cell(SW, dev.swVersion())
160  .cell(PROTOCOL, deviceProtocol(dev))
161  .cell(NUM_PORTS, ds.getPorts(id).size())
162  .cell(MASTER_ID, ms.getMasterFor(id));
163  }

166  // handler for selected device detail requests
167  private final class DetailRequestHandler extends RequestHandler {
168  private DetailRequestHandler() {
169  super(DEV_DETAILS_REQ);
170  }
171 
172  @Override
173  public void process(long sid, ObjectNode payload) {
174  String id = string(payload, ID, ZERO_URI);
175 
176  DeviceId deviceId = deviceId(id);
177  DeviceService service = get(DeviceService.class);
178  MastershipService ms = get(MastershipService.class);
179  Device device = service.getDevice(deviceId);
180  ObjectNode data = objectNode();
181 
182  data.put(ID, deviceId.toString());
183  data.put(NAME, deviceName(device));
184  data.put(TYPE, capitalizeFully(device.type().toString()));
185  data.put(TYPE_IID, getTypeIconId(device));
186  data.put(MFR, device.manufacturer());
187  data.put(HW, device.hwVersion());
188  data.put(SW, device.swVersion());
189  data.put(SERIAL, device.serialNumber());
190  data.put(CHASSIS_ID, device.chassisId().toString());
191  data.put(MASTER_ID, ms.getMasterFor(deviceId).toString());
192  data.put(PROTOCOL, deviceProtocol(device));
193 
194  ArrayNode ports = arrayNode();
195 
196  List<Port> portList = new ArrayList<>(service.getPorts(deviceId));
197  Collections.sort(portList, (p1, p2) -> {
198  long delta = p1.number().toLong() - p2.number().toLong();
199  return delta == 0 ? 0 : (delta < 0 ? -1 : +1);
200  });
201 
202  for (Port p : portList) {
203  ports.add(portData(p, deviceId));
204  }
205  data.set(PORTS, ports);
206 
207  ObjectNode rootNode = objectNode();
208  rootNode.set(DETAILS, data);
209  sendMessage(DEV_DETAILS_RESP, 0, rootNode);
210  }
211 
212  private ObjectNode portData(Port p, DeviceId id) {
213  ObjectNode port = objectNode();
214  LinkService ls = get(LinkService.class);
215  String name = p.annotations().value(AnnotationKeys.PORT_NAME);
216 
217  port.put(ID, capitalizeFully(p.number().toString()));
218  port.put(TYPE, capitalizeFully(p.type().toString()));
219  port.put(SPEED, p.portSpeed());
220  port.put(ENABLED, p.isEnabled());
221  port.put(NAME, name != null ? name : "");
222 
223  Set<Link> links = ls.getEgressLinks(new ConnectPoint(id, p.number()));
224  if (!links.isEmpty()) {
225  StringBuilder egressLinks = new StringBuilder();
226  for (Link l : links) {
227  ConnectPoint dest = l.dst();
228  egressLinks.append(dest.elementId()).append("/")
229  .append(dest.port()).append(" ");
230  }
231  port.put(LINK_DEST, egressLinks.toString());
232  }
233 
234  return port;
235  }
236  }

245  @Override
246  public void process(long sid, ObjectNode payload) {
247  DeviceId deviceId = deviceId(string(payload, ID, ZERO_URI));
248  String name = emptyToNull(string(payload, NAME, null));
249  log.debug("Name change request: {} -- '{}'", deviceId, name);
250 
251  NetworkConfigService service = get(NetworkConfigService.class);
252  BasicDeviceConfig cfg =
253  service.addConfig(deviceId, BasicDeviceConfig.class);
254 
255  // Name attribute missing from the payload (or empty string)
256  // means that the friendly name should be unset.
257  cfg.name(name);
258  cfg.apply();
259  sendMessage(DEV_NAME_CHANGE_RESP, 0, payload);
260  }
