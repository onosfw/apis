84  /**
85  * Indicates if this connection is idle.

87  * @return true if idle or closed
88  */
89  synchronized boolean isIdle() {
90  long quietFor = System.currentTimeMillis() - lastActive;
91  boolean idle = quietFor > MAX_AGE_MS;
92  if (idle || (connection != null && !connection.isOpen())) {
93  log.debug("IDLE (or closed) websocket [{} ms]", quietFor);
94  return true;
95  } else if (connection != null) {
96  try {
97  control.sendControl(PING, PING_DATA, 0, PING_DATA.length);
98  } catch (IOException e) {
99  log.warn("Unable to send ping message due to: ", e);
100  }
101  }
102  return false;
103  }

178  // Creates new message handlers.
179  private synchronized void createHandlersAndOverlays() {
180  log.debug("creating handlers and overlays...");
181  handlers = new HashMap<>();
182  overlayCache = new TopoOverlayCache();
183 
184  UiExtensionService service = directory.get(UiExtensionService.class);
185  service.getExtensions().forEach(ext -> {
186  UiMessageHandlerFactory factory = ext.messageHandlerFactory();
187  if (factory != null) {
188  factory.newHandlers().forEach(handler -> {
189  handler.init(this, directory);
190  handler.messageTypes().forEach(type -> handlers.put(type, handler));
191 
192  // need to inject the overlay cache into topology message handler
193  if (handler instanceof TopologyViewMessageHandler) {
194  ((TopologyViewMessageHandler) handler).setOverlayCache(overlayCache);
195  }
196  });
197  }
198 
199  UiTopoOverlayFactory overlayFactory = ext.topoOverlayFactory();
200  if (overlayFactory != null) {
201  overlayFactory.newOverlays().forEach(overlayCache::add);
202  }
203  });
204  log.debug("#handlers = {}, #overlays = {}", handlers.size(),
205  overlayCache.size());
206  }

208  // Destroys message handlers.

210  log.debug("destroying handlers and overlays...");
211  handlers.forEach((type, handler) -> handler.destroy());
212  handlers.clear();
213 
214  if (overlayCache != null) {
215  overlayCache.destroy();
216  overlayCache = null;
217  }
218  }

220  // Sends cluster node/instance information to allow GUI to fail-over.
221  private void sendInstanceData() {
222  ClusterService service = directory.get(ClusterService.class);
223  ArrayNode instances = mapper.createArrayNode();
224 
225  for (ControllerNode node : service.getNodes()) {
226  ObjectNode instance = mapper.createObjectNode()
227  .put("id", node.id().toString())
228  .put("ip", node.ip().toString())
229  .put(TopoConstants.Glyphs.UI_ATTACHED,
230  node.equals(service.getLocalNode()));
231  instances.add(instance);
232  }
233 
234  ObjectNode payload = mapper.createObjectNode();
235  payload.set("clusterNodes", instances);
236  sendMessage("bootstrap", 0, payload);
237  }
