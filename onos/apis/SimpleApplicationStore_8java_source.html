48  * Manages inventory of network control applications.

50 @Component(immediate = true)
51 @Service
52 public class SimpleApplicationStore extends ApplicationArchive implements ApplicationStore {
53 
54  private final Logger log = getLogger(getClass());
55 
56  // App inventory & states
57  private final ConcurrentMap<ApplicationId, DefaultApplication> apps = new ConcurrentHashMap<>();
58  private final ConcurrentMap<ApplicationId, ApplicationState> states = new ConcurrentHashMap<>();
59  private final ConcurrentMap<ApplicationId, Set<Permission>> permissions = new ConcurrentHashMap<>();
60 
61  @Reference(cardinality = ReferenceCardinality.MANDATORY_UNARY)
62  protected ApplicationIdStore idStore;
63 
64  @Activate
65  public void activate() {
66  loadFromDisk();
67  log.info("Started");
68  }
69 
70  private void loadFromDisk() {
71  for (String name : getApplicationNames()) {
72  ApplicationId appId = idStore.registerApplication(name);
73  ApplicationDescription appDesc = getApplicationDescription(name);
74  DefaultApplication app =
75  new DefaultApplication(appId, appDesc.version(),
76  appDesc.description(), appDesc.origin(),
77  appDesc.role(), appDesc.permissions(),
78  appDesc.featuresRepo(), appDesc.features(),
79  appDesc.requiredApps());
80  apps.put(appId, app);
81  states.put(appId, isActive(name) ? INSTALLED : ACTIVE);
82  // load app permissions
83  }
84  }
85 
86  @Deactivate
87  public void deactivate() {
88  apps.clear();
89  states.clear();
90  permissions.clear();
91  log.info("Stopped");
92  }
93

113 
114  @Override
115  public Application create(InputStream appDescStream) {
116  ApplicationDescription appDesc = saveApplication(appDescStream);
117  ApplicationId appId = idStore.registerApplication(appDesc.name());
118  DefaultApplication app =
119  new DefaultApplication(appId, appDesc.version(), appDesc.description(),
120  appDesc.origin(), appDesc.role(), appDesc.permissions(),
121  appDesc.featuresRepo(), appDesc.features(),
122  appDesc.requiredApps());
123  apps.put(appId, app);
124  states.put(appId, INSTALLED);
125  delegate.notify(new ApplicationEvent(APP_INSTALLED, app));
126  return app;
127  }
128 
129  @Override
130  public void remove(ApplicationId appId) {
131  Application app = apps.remove(appId);
132  if (app != null) {
133  states.remove(appId);
134  delegate.notify(new ApplicationEvent(APP_UNINSTALLED, app));
135  purgeApplication(app.id().name());
136  }
137  }
138 
139  @Override
140  public void activate(ApplicationId appId) {
141  Application app = apps.get(appId);
142  if (app != null) {
143  setActive(appId.name());
144  states.put(appId, ACTIVE);
145  delegate.notify(new ApplicationEvent(APP_ACTIVATED, app));
146  }
147  }
148 
149  @Override
150  public void deactivate(ApplicationId appId) {
151  Application app = apps.get(appId);
152  if (app != null) {
153  clearActive(appId.name());
154  states.put(appId, INSTALLED);
155  delegate.notify(new ApplicationEvent(APP_DEACTIVATED, app));
156  }
157  }
158 
159  @Override
160  public Set<Permission> getPermissions(ApplicationId appId) {
161  return permissions.get(appId);
162  }
163 
164  @Override
165  public void setPermissions(ApplicationId appId, Set<Permission> permissions) {
166  Application app = getApplication(appId);
167  if (app != null) {
168  this.permissions.put(appId, permissions);
169  delegate.notify(new ApplicationEvent(APP_PERMISSIONS_CHANGED, app));
170  }
171  }
172 }
