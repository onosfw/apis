252  private class HostLocationTracker implements EventuallyConsistentMapListener<HostId, DefaultHost> {
253  @Override
254  public void event(EventuallyConsistentMapEvent<HostId, DefaultHost> event) {
255  DefaultHost host = checkNotNull(event.value());
256  if (event.type() == PUT) {
257  Host prevHost = prevHosts.put(host.id(), host);
258  if (prevHost == null) {
259  notifyDelegate(new HostEvent(HOST_ADDED, host));
260  } else if (!Objects.equals(prevHost.location(), host.location())) {
261  notifyDelegate(new HostEvent(HOST_MOVED, host, prevHost));
262  } else if (!Objects.equals(prevHost, host)) {
263  notifyDelegate(new HostEvent(HOST_UPDATED, host, prevHost));
264  }
265  } else if (event.type() == REMOVE) {
266  if (prevHosts.remove(host.id()) != null) {
267  notifyDelegate(new HostEvent(HOST_REMOVED, host));
268  }
269  }
270  }
271  }
