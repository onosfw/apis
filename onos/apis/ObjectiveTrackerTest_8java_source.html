114  @Override
115  public void triggerCompile(Iterable<Key> intentKeys,
116  boolean compileAllFailed) {
117  intentIdsFromEvent = Lists.newArrayList(intentKeys);
118  compileAllFailedFromEvent = compileAllFailed;
119  latch.countDown();
120  }
