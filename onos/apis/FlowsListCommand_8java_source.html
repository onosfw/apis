80  /**

97  // Produces JSON object with the flows of the given device.
98  private ObjectNode json(ObjectMapper mapper,
99  Device device, List<FlowEntry> flows) {
100  ObjectNode result = mapper.createObjectNode();
101  ArrayNode array = mapper.createArrayNode();
102 
103  flows.forEach(flow -> array.add(jsonForEntity(flow, FlowEntry.class)));
104 
105  result.put("device", device.id().toString())
106  .put("flowCount", flows.size())
107  .set("flows", array);
108  return result;

111  /**
112  * Returns the list of devices sorted using the device ID URIs.
113  *
114  * @param deviceService device service
115  * @param service flow rule service
116  * @return sorted device list
117  */
118  protected SortedMap<Device, List<FlowEntry>> getSortedFlows(DeviceService deviceService,
119  FlowRuleService service) {
120  SortedMap<Device, List<FlowEntry>> flows = new TreeMap<>(Comparators.ELEMENT_COMPARATOR);
121  List<FlowEntry> rules;
122  FlowEntryState s = null;
123  if (state != null && !state.equals("any")) {
124  s = FlowEntryState.valueOf(state.toUpperCase());
125  }
126  Iterable<Device> devices = null;
127  if (uri == null) {
128  devices = deviceService.getDevices();
129  } else {
130  Device dev = deviceService.getDevice(DeviceId.deviceId(uri));
131  devices = (dev == null) ? deviceService.getDevices()
132  : Collections.singletonList(dev);
133  }
134  for (Device d : devices) {
135  if (s == null) {
136  rules = newArrayList(service.getFlowEntries(d.id()));
137  } else {
138  rules = newArrayList();
139  for (FlowEntry f : service.getFlowEntries(d.id())) {
140  if (f.state().equals(s)) {
141  rules.add(f);
142  }
143  }
144  }
145  rules.sort(Comparators.FLOW_RULE_COMPARATOR);
146  flows.put(d, rules);
147  }
148  return flows;
149  }

151  /**
152  * Prints flows.
153  *
154  * @param d the device
155  * @param flows the set of flows for that device
156  * @param coreService core system service
157  */
158  protected void printFlows(Device d, List<FlowEntry> flows,
159  CoreService coreService) {
160  boolean empty = flows == null || flows.isEmpty();
161  print("deviceId=%s, flowRuleCount=%d", d.id(), empty ? 0 : flows.size());
162  if (!empty) {
163  for (FlowEntry f : flows) {
164  ApplicationId appId = coreService.getAppId(f.appId());
165  print(FMT, Long.toHexString(f.id().value()), f.state(),
166  f.bytes(), f.packets(), f.life(), f.priority(), f.tableId(),
167  appId != null ? appId.name() : "<none>",
168  f.payLoad() == null ? null : f.payLoad().payLoad().toString());
169  print(SFMT, f.selector().criteria());
170  print(TFMT, f.treatment());
171  }
172  }
173  }
