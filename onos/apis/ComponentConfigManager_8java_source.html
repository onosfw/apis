110  @Override
111  public void registerProperties(Class<?> componentClass) {
112  checkPermission(CONFIG_WRITE);
113 
114  String componentName = componentClass.getName();
115  String resourceName = componentClass.getSimpleName() + RESOURCE_EXT;
116  try (InputStream ris = componentClass.getResourceAsStream(resourceName)) {
117  checkArgument(ris != null, "Property definitions not found at resource %s",
118  resourceName);
119 
120  // Read the definitions
121  Set<ConfigProperty> defs = ConfigPropertyDefinitions.read(ris);
122 
123  // Produce a new map of the properties and register it.
124  Map<String, ConfigProperty> map = Maps.newConcurrentMap();
125  defs.forEach(p -> map.put(p.name(), p));
126 
127  properties.put(componentName, map);
128  loadExistingValues(componentName);
129  } catch (IOException e) {
130  log.error("Unable to read property definitions from resource " + resourceName, e);
131  }
132  }
133 
134  @Override
135  public void unregisterProperties(Class<?> componentClass, boolean clear) {
136  checkPermission(CONFIG_WRITE);

141  if (clear && cps != null) {
142  cps.keySet().forEach(name -> store.unsetProperty(componentName, name));

145  }

236  // Locates the property in the component map and replaces it with an
237  // updated copy.
238  private void set(String componentName, String name, String value) {
239  Map<String, ConfigProperty> map = properties.get(componentName);
240  if (map != null) {
241  ConfigProperty prop = map.get(name);
242  if (prop != null) {
243  map.put(name, ConfigProperty.setProperty(prop, value));
244  accumulator.add(componentName);
245  return;
246  }
247  }
248  log.warn("Unable to set non-existent property {} for component {}",
249  name, componentName);
250  }
251 
252  // Locates the property in the component map and replaces it with an
253  // reset copy.
254  private void reset(String componentName, String name) {
255  Map<String, ConfigProperty> map = properties.get(componentName);
256  if (map != null) {
257  ConfigProperty prop = map.get(name);
258  if (prop != null) {
259  map.put(name, ConfigProperty.resetProperty(prop));
260  accumulator.add(componentName);
261  return;
262  }
263  log.warn("Unable to reset non-existent property {} for component {}",
264  name, componentName);
265  }
266  }

268  // Loads existing property values that may have been set.
269  private void loadExistingValues(String componentName) {
270  try {
271  Configuration cfg = cfgAdmin.getConfiguration(componentName, null);
272  Map<String, ConfigProperty> map = properties.get(componentName);
273  Dictionary<String, Object> props = cfg.getProperties();
274  if (props != null) {
275  Enumeration<String> it = props.keys();
276  while (it.hasMoreElements()) {
277  String name = it.nextElement();
278  ConfigProperty p = map.get(name);
279  if (p != null) {
280  map.put(name, ConfigProperty.setProperty(p, (String) props.get(name)));
281  }
282  }
283  }
284  } catch (IOException e) {
285  log.error("Unable to get configuration for " + componentName, e);
286  }
287 
288  }

290  // FIXME: This should be a slightly deferred execution to allow changing
291  // values just once per component when a number of updates arrive shortly
292  // after each other.
293  private void triggerUpdate(String componentName) {
294  try {
295  Configuration cfg = cfgAdmin.getConfiguration(componentName, null);
296  Map<String, ConfigProperty> map = properties.get(componentName);
297  Dictionary<String, Object> props = new Hashtable<>();
298  map.values().forEach(p -> props.put(p.name(), p.value()));
299  cfg.update(props);
300  } catch (IOException e) {
301  log.warn("Unable to update configuration for " + componentName, e);
302  }
303  }
