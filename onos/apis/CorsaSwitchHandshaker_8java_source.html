78  @Override
79  public void processDriverHandshakeMessage(OFMessage m) {
80  if (!startDriverHandshakeCalled) {
81  throw new SwitchDriverSubHandshakeNotStarted();
82  }
83  if (handshakeComplete.get()) {
84  throw new SwitchDriverSubHandshakeCompleted(m);
85  }
86  if (m.getType() == OFType.BARRIER_REPLY &&
87  m.getXid() == barrierXid) {
88  handshakeComplete.set(true);
