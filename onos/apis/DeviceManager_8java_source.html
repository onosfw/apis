344  @Override
345  public void deviceDisconnected(DeviceId deviceId) {
346  checkNotNull(deviceId, DEVICE_ID_NULL);
347  checkValidity();
348 
349  log.info("Device {} disconnected from this node", deviceId);
350 
351  List<Port> ports = store.getPorts(deviceId);
352  final Device device = getDevice(deviceId);
353 
354  List<PortDescription> descs = ports.stream().map(
355  port -> (!(Device.Type.ROADM.equals(device.type()))) ?
356  new DefaultPortDescription(port.number(), false,
357  port.type(), port.portSpeed()) :
358  OpticalPortOperator.descriptionOf(port, false)
359  ).collect(Collectors.toList());
360 
361  store.updatePorts(this.provider().id(), deviceId, descs);
362  try {
363  if (mastershipService.isLocalMaster(deviceId)) {
364  post(store.markOffline(deviceId));
365  }
366  } catch (IllegalStateException e) {
367  log.warn("Failed to mark {} offline", deviceId);
368  // only the MASTER should be marking off-line in normal cases,
369  // but if I was the last STANDBY connection, etc. and no one else
370  // was there to mark the device offline, this instance may need to
371  // temporarily request for Master Role and mark offline.
372 
373  //there are times when this node will correctly have mastership, BUT
374  //that isn't reflected in the ClockManager before the device disconnects.
375  //we want to let go of the device anyways, so make sure this happens.
376 
377  // FIXME: Store semantics leaking out as IllegalStateException.
378  // Consider revising store API to handle this scenario.
379  CompletableFuture<MastershipRole> roleFuture = mastershipService.requestRoleFor(deviceId);
380  roleFuture.whenComplete((role, error) -> {
381  MastershipTerm term = termService.getMastershipTerm(deviceId);
382  // TODO: Move this type of check inside device clock manager, etc.
383  if (term != null && localNodeId.equals(term.master())) {
384  log.info("Retry marking {} offline", deviceId);
385  post(store.markOffline(deviceId));
386  } else {
387  log.info("Failed again marking {} offline. {}", deviceId, role);
388  }
389  });
390  } finally {
391  try {
392  //relinquish master role and ability to be backup.
393  mastershipService.relinquishMastership(deviceId).get();
394  } catch (InterruptedException e) {
395  log.warn("Interrupted while reliquishing role for {}", deviceId);
396  Thread.currentThread().interrupt();
397  } catch (ExecutionException e) {
398  log.error("Exception thrown while relinquishing role for {}", deviceId, e);
399  }
400  }
401  }

556  /**
557  * Reaasert role for specified device connected to this node.
558  *
559  * @param did device identifier
560  * @param nextRole role to apply. If NONE is specified,
561  * it will ask mastership service for a role and apply it.
562  */
563  private void reassertRole(final DeviceId did,
564  final MastershipRole nextRole) {
565 
566  MastershipRole myNextRole = nextRole;
567  if (myNextRole == NONE) {
568  mastershipService.requestRoleFor(did);
569  MastershipTerm term = termService.getMastershipTerm(did);
570  if (term != null && localNodeId.equals(term.master())) {
571  myNextRole = MASTER;
572  } else {
573  myNextRole = STANDBY;
574  }
575  }
576 
577  switch (myNextRole) {
578  case MASTER:
579  final Device device = getDevice(did);
580  if ((device != null) && !isAvailable(did)) {
581  //flag the device as online. Is there a better way to do this?
582  DefaultDeviceDescription deviceDescription
583  = new DefaultDeviceDescription(did.uri(),
584  device.type(),
585  device.manufacturer(),
586  device.hwVersion(),
587  device.swVersion(),
588  device.serialNumber(),
589  device.chassisId());
590  DeviceEvent devEvent =
591  store.createOrUpdateDevice(device.providerId(), did,
592  deviceDescription);
593  post(devEvent);
594  }
595  // TODO: should apply role only if there is mismatch
596  log.debug("Applying role {} to {}", myNextRole, did);
597  if (!applyRoleAndProbe(did, MASTER)) {
598  log.warn("Unsuccessful applying role {} to {}", myNextRole, did);
599  // immediately failed to apply role
600  mastershipService.relinquishMastership(did);
601  // FIXME disconnect?
602  }
603  break;
604  case STANDBY:
605  log.debug("Applying role {} to {}", myNextRole, did);
606  if (!applyRoleAndProbe(did, STANDBY)) {
607  log.warn("Unsuccessful applying role {} to {}", myNextRole, did);
608  // immediately failed to apply role
609  mastershipService.relinquishMastership(did);
610  // FIXME disconnect?
611  }
612  break;
613  case NONE:
614  default:
615  // should never reach here
616  log.error("You didn't see anything. I did not exist.");
617  break;
618  }
619  }

621  private void handleMastershipEvent(MastershipEvent event) {
622  if (event.type() != MastershipEvent.Type.MASTER_CHANGED) {
623  // Don't care if backup list changed.
624  return;
625  }
626 
627  final DeviceId did = event.subject();
628 
629  // myRole suggested by MastershipService
630  MastershipRole myNextRole;
631  if (localNodeId.equals(event.roleInfo().master())) {
632  // confirm latest info
633  MastershipTerm term = termService.getMastershipTerm(did);
634  final boolean iHaveControl = term != null && localNodeId.equals(term.master());
635  if (iHaveControl) {
636  myNextRole = MASTER;
637  } else {
638  myNextRole = STANDBY;
639  }
640  } else if (event.roleInfo().backups().contains(localNodeId)) {
641  myNextRole = STANDBY;
642  } else {
643  myNextRole = NONE;
644  }
645 
646  final boolean isReachable = isReachable(did);
647  if (!isReachable) {
648  // device is not connected to this node
649  if (myNextRole != NONE) {
650  log.warn("Node was instructed to be {} role for {}, "
651  + "but this node cannot reach the device. "
652  + "Relinquishing role. ",
653  myNextRole, did);
654  mastershipService.relinquishMastership(did);
655  }
656  return;
657  }
658 
659  // device is connected to this node:
660  if (store.getDevice(did) != null) {
661  reassertRole(did, myNextRole);
662  } else {
663  log.debug("Device is not yet/no longer in the store: {}", did);
664  }
665  }

729  @Override
730  public void event(NetworkConfigEvent event) {
731  DeviceEvent de = null;
732  if (event.configClass().equals(BasicDeviceConfig.class)) {
733  log.info("Detected Device network config event {}", event.type());
734  DeviceId did = (DeviceId) event.subject();
735  BasicDeviceConfig cfg = networkConfigService.getConfig(did, BasicDeviceConfig.class);
736 
737  if (!isAllowed(cfg)) {
738  kickOutBadDevice(did);
739  } else {
740  Device dev = getDevice(did);
741  DeviceDescription desc = (dev == null) ? null : BasicDeviceOperator.descriptionOf(dev);
742  desc = BasicDeviceOperator.combine(cfg, desc);
743  if (desc != null && getProvider(did) != null) {
744  de = store.createOrUpdateDevice(getProvider(did).id(), did, desc);
745  }
746  }
747  }
748  if (event.configClass().equals(OpticalPortConfig.class)) {
749  ConnectPoint cpt = (ConnectPoint) event.subject();
750  DeviceId did = cpt.deviceId();
751  Port dpt = getPort(did, cpt.port());
752 
753  if (dpt != null) {
754  OpticalPortConfig opc = networkConfigService.getConfig(cpt, OpticalPortConfig.class);
755  PortDescription desc = OpticalPortOperator.descriptionOf(dpt);
756  desc = OpticalPortOperator.combine(opc, desc);
757  if (desc != null && getProvider(did) != null) {
758  de = store.updatePortStatus(getProvider(did).id(), did, desc);
759  }
760  }
761  }
762 
763  if (de != null) {
764  post(de);
765  }
766  }
