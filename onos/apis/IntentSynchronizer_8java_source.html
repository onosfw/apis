149  /**
150  * Signals the synchronizer that the leadership has changed.
151  *
152  * @param isLeader true if this instance is now the leader, otherwise false
153  */
154  public void leaderChanged(boolean isLeader) {
155  log.debug("Leader changed: {}", isLeader);
156 
157  if (!isLeader) {
158  this.isElectedLeader = false;
159  this.isActivatedLeader = false;
160  return; // Nothing to do
161  }
162  this.isActivatedLeader = false;
163  this.isElectedLeader = true;
164 
165  // Run the synchronization method off-thread
166  intentsSynchronizerExecutor.execute(this::synchronizeIntents);
167  }
168 
169  private void synchronizeIntents() {
170  Map<Key, Intent> serviceIntents = new HashMap<>();
171  intentService.getIntents().forEach(i -> {
172  if (i.appId().equals(appId)) {
173  serviceIntents.put(i.key(), i);
174  }
175  });
176 
177  List<Intent> intentsToAdd = new LinkedList<>();
178  List<Intent> intentsToRemove = new LinkedList<>();
179 
180  for (Intent localIntent : intents.values()) {
181  Intent serviceIntent = serviceIntents.remove(localIntent.key());
182  if (serviceIntent == null) {
183  intentsToAdd.add(localIntent);
184  } else {
185  IntentState state = intentService.getIntentState(serviceIntent.key());
186  if (!IntentUtils.equals(serviceIntent, localIntent) || state == null ||
187  state == IntentState.WITHDRAW_REQ ||
188  state == IntentState.WITHDRAWING ||
189  state == IntentState.WITHDRAWN) {
190  intentsToAdd.add(localIntent);
191  }
192  }
193  }
194 
195  for (Intent serviceIntent : serviceIntents.values()) {
196  IntentState state = intentService.getIntentState(serviceIntent.key());
197  if (state != null && state != IntentState.WITHDRAW_REQ
198  && state != IntentState.WITHDRAWING
199  && state != IntentState.WITHDRAWN) {
200  intentsToRemove.add(serviceIntent);
201  }
202  }
203 
204  log.debug("Intent Synchronizer: submitting {}, withdrawing {}",
205  intentsToAdd.size(), intentsToRemove.size());
206 
207  // Withdraw Intents
208  for (Intent intent : intentsToRemove) {
209  intentService.withdraw(intent);
210  log.trace("Intent Synchronizer: withdrawing intent: {}",
211  intent);
212  }
213  if (!isElectedLeader) {
214  log.debug("Intent Synchronizer: cannot withdraw intents: " +
215  "not elected leader anymore");
216  isActivatedLeader = false;
217  return;
218  }
219 
220  // Add Intents
221  for (Intent intent : intentsToAdd) {
222  intentService.submit(intent);
223  log.trace("Intent Synchronizer: submitting intent: {}",
224  intent);
225  }
226  if (!isElectedLeader) {
227  log.debug("Intent Synchronizer: cannot submit intents: " +
228  "not elected leader anymore");
229  isActivatedLeader = false;
230  return;
231  }
232 
233  if (isElectedLeader) {
234  isActivatedLeader = true; // Allow push of Intents
235  } else {
236  isActivatedLeader = false;
237  }
238  log.debug("Intent synchronization completed");
239  }
