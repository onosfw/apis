215  // returns a HostDescription made from the union of the BasicHostConfig
216  // annotations if it exists
217  private HostDescription validateHost(HostDescription hostDescription, HostId hostId) {
218  BasicHostConfig cfg = networkConfigService.getConfig(hostId, BasicHostConfig.class);
219  checkState(cfg == null || cfg.isAllowed(), "Host {} is not allowed", hostId);

221  return BasicHostOperator.combine(cfg, hostDescription);

247  // listens for NetworkConfigEvents of type BasicHostConfig and removes
248  // links that the config does not allow
249  private class InternalNetworkConfigListener implements NetworkConfigListener {
250  @Override
251  public void event(NetworkConfigEvent event) {
252  if ((event.type() == NetworkConfigEvent.Type.CONFIG_ADDED ||
253  event.type() == NetworkConfigEvent.Type.CONFIG_UPDATED) &&
254  event.configClass().equals(BasicHostConfig.class)) {
255  log.info("Detected Host network config event {}", event.type());
256  kickOutBadHost(((HostId) event.subject()));
257  }
258  }
259  }
260 
261  // checks if the specified host is allowed by the BasicHostConfig
262  // and if not, removes it
263  private void kickOutBadHost(HostId hostId) {
264  BasicHostConfig cfg = networkConfigService.getConfig(hostId, BasicHostConfig.class);
265  if (cfg != null && !cfg.isAllowed()) {
266  Host badHost = getHost(hostId);
267  if (badHost != null) {
268  removeHost(hostId);
269  } else {
270  log.info("Failed removal: Host {} does not exist", hostId);
271  }
272  }
