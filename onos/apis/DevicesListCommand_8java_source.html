33 /**

36 @Command(scope = "onos", name = "devices",
37  description = "Lists all infrastructure devices")
38 public class DevicesListCommand extends AbstractShellCommand {

40  private static final String FMT =
41  "id=%s, available=%s, role=%s, type=%s, mfr=%s, hw=%s, sw=%s, serial=%s%s";
42 
43  @Override
44  protected void execute() {
45  DeviceService service = get(DeviceService.class);
46  if (outputJson()) {
47  print("%s", json(getSortedDevices(service)));
48  } else {
49  for (Device device : getSortedDevices(service)) {
50  printDevice(service, device);

52  }
53  }

55  /**
56  * Returns JSON node representing the specified devices.
57  *
58  * @param devices collection of devices
59  * @return JSON node
60  */
61  private JsonNode json(Iterable<Device> devices) {
62  ObjectMapper mapper = new ObjectMapper();
63  ArrayNode result = mapper.createArrayNode();
64  for (Device device : devices) {
65  result.add(jsonForEntity(device, Device.class));
66  }
67  return result;
68  }

70  /**

72  *
73  * @param service device service
74  * @return sorted device list

76  public static List<Device> getSortedDevices(DeviceService service) {
77  List<Device> devices = newArrayList(service.getDevices());
