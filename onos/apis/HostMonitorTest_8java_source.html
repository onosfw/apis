180  ConnectPoint cp = new ConnectPoint(devId, portNum);

184  replay(hostManager);
185 
186  InterfaceService interfaceService = createMock(InterfaceService.class);
187  expect(interfaceService.getMatchingInterface(TARGET_IPV4_ADDR))

189  .anyTimes();
190  replay(interfaceService);
191 
192  TestPacketService packetService = new TestPacketService();
193 
194 
195  // Run the test
196  hostMonitor = new HostMonitor(packetService, hostManager, interfaceService, edgePortService);
197 
198  hostMonitor.addMonitoringFor(TARGET_IPV4_ADDR);
199  hostMonitor.run(null);
200 
201 
202  // Check that a packet was sent to our PacketService and that it has
203  // the properties we expect
204  assertEquals(1, packetService.packets.size());
205  OutboundPacket packet = packetService.packets.get(0);
206 
207  // Check the output port is correct
208  assertEquals(1, packet.treatment().immediate().size());
209  Instruction instruction = packet.treatment().immediate().get(0);
210  assertTrue(instruction instanceof OutputInstruction);
211  OutputInstruction oi = (OutputInstruction) instruction;
212  assertEquals(portNum, oi.port());
213 
214  // Check the output packet is correct (well the important bits anyway)
215  final byte[] pktData = new byte[packet.data().remaining()];
216  packet.data().get(pktData);
217  Ethernet eth = Ethernet.deserializer().deserialize(pktData, 0, pktData.length);
218  assertEquals(Ethernet.VLAN_UNTAGGED, eth.getVlanID());
219  ARP arp = (ARP) eth.getPayload();
220  assertArrayEquals(SOURCE_IPV4_ADDR.toOctets(),
221  arp.getSenderProtocolAddress());
222  assertArrayEquals(sourceMac.toBytes(),
223  arp.getSenderHardwareAddress());
224  assertArrayEquals(TARGET_IPV4_ADDR.toOctets(),
225  arp.getTargetProtocolAddress());
226  }
