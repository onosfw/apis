51  // Default number of millis a sink can take to process an event.
52  private static final long DEFAULT_EXECUTE_MS = 5_000; // ms
53  private static final long WATCHDOG_MS = 250; // ms
54 
55  private final BlockingQueue<Event> events = new LinkedBlockingQueue<>();

109  // Auxiliary event dispatching loop that feeds off the events queue.
110  private class DispatchLoop implements Runnable {
111  private volatile boolean stopped;
112 
113  @Override
114  public void run() {
115  stopped = false;
116  log.info("Dispatch loop initiated");
117  while (!stopped) {
118  try {
119  // Fetch the next event and if it is the kill-pill, bail
120  Event event = events.take();
121  if (event == KILL_PILL) {
122  break;
123  }
124  process(event);
125  } catch (InterruptedException e) {
126  log.warn("Dispatch loop interrupted");
127  } catch (Exception e) {
128  log.warn("Error encountered while dispatching event:", e);
129  }
130  }
131  log.info("Dispatch loop terminated");
132  }
133 
134  // Locate the sink for the event class and use it to process the event
135  @SuppressWarnings("unchecked")
136  private void process(Event event) {
137  EventSink sink = getSink(event.getClass());
138  if (sink != null) {
139  lastSink = sink;
140  lastStart = System.currentTimeMillis();
141  sink.process(event);
142  lastStart = 0;
143  } else {
144  log.warn("No sink registered for event class {}",
145  event.getClass().getName());
146  }
147  }
148 
149  void stop() {
150  stopped = true;
151  }
152  }
153 
154  // Monitors event sinks to make sure none take too long to execute.
155  private class Watchdog extends TimerTask {
156  @Override
157  public void run() {
158  long delta = System.currentTimeMillis() - lastStart;
159  if (lastStart > 0 && delta > maxProcessMillis) {
160  lastStart = 0;
161  log.warn("Event sink {} exceeded execution time limit: {} ms; spawning new dispatch loop",
162  lastSink.getClass().getName(), delta);
163 
164  // Notify the sink that it has exceeded its time limit.
165  lastSink.onProcessLimit();
166 
167  // Cancel the old dispatch loop and submit a new one.
168  dispatchLoop.stop();
169  dispatchLoop = new DispatchLoop();
170  dispatchFuture.cancel(true);
171  dispatchFuture = executor.submit(dispatchLoop);
172  }
173  }
174  }
