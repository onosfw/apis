170  /**
171  * Records an event.
172  *
173  * @param event the event to record
174  * @param eventMetric the Event Metric to use
175  */
176  private void recordEvent(Event event, EventMetric eventMetric) {
177  synchronized (lastEvents) {
178  eventMetric.eventReceived();
179 
180  //
181  // Keep only the last N events, where N = LAST_EVENTS_MAX_N
182  //
183  while (lastEvents.size() >= LAST_EVENTS_MAX_N) {
184  lastEvents.remove();
185  }
186  lastEvents.add(event);
187  }
188  }

190  /**
191  * Inner Device Event Listener class.
192  */
193  private class InnerDeviceListener implements DeviceListener {
194  @Override
195  public void event(DeviceEvent event) {
196  // Ignore PORT_STATS_UPDATED probe event from interfering with
197  // other device event timestamps
198  if (event.type() == DeviceEvent.Type.PORT_STATS_UPDATED) {
199  log.debug("PORT_STATS_UPDATED event ignored from metrics");
200  } else {
201  recordEvent(event, topologyDeviceEventMetric);
202  log.debug("Device Event: time = {} type = {} event = {}",
203  event.time(), event.type(), event);
204  }
205  }
206  }

232  /**

235  private class InnerTopologyListener implements TopologyListener {
236  @Override
237  public void event(TopologyEvent event) {
238  recordEvent(event, topologyGraphEventMetric);
239  log.debug("Topology Event: time = {} type = {} event = {}",
240  event.time(), event.type(), event);
241  for (Event reason : event.reasons()) {
242  recordEvent(event, topologyGraphReasonsEventMetric);
243  log.debug("Topology Event Reason: time = {} type = {} event = {}",
244  reason.time(), reason.type(), reason);
245  }
246  }
