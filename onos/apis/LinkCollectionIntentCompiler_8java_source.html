72  @Override
73  public List<Intent> compile(LinkCollectionIntent intent, List<Intent> installable,
74  Set<LinkResourceAllocations> resources) {
75  SetMultimap<DeviceId, PortNumber> inputPorts = HashMultimap.create();
76  SetMultimap<DeviceId, PortNumber> outputPorts = HashMultimap.create();
77 
78  for (Link link : intent.links()) {
79  inputPorts.put(link.dst().deviceId(), link.dst().port());
80  outputPorts.put(link.src().deviceId(), link.src().port());
81  }
82 
83  for (ConnectPoint ingressPoint : intent.ingressPoints()) {
84  inputPorts.put(ingressPoint.deviceId(), ingressPoint.port());
85  }
86 
87  for (ConnectPoint egressPoint : intent.egressPoints()) {
88  outputPorts.put(egressPoint.deviceId(), egressPoint.port());
89  }
90 
91  List<FlowRule> rules = new ArrayList<>();
92  for (DeviceId deviceId: outputPorts.keys()) {
93  rules.addAll(createRules(intent, deviceId, inputPorts.get(deviceId), outputPorts.get(deviceId)));
94  }
95  return Collections.singletonList(new FlowRuleIntent(appId, rules, intent.resources()));
96  }
97 
98  private List<FlowRule> createRules(LinkCollectionIntent intent, DeviceId deviceId,
99  Set<PortNumber> inPorts, Set<PortNumber> outPorts) {
100  Set<PortNumber> ingressPorts = intent.ingressPoints().stream()
101  .filter(point -> point.deviceId().equals(deviceId))
102  .map(ConnectPoint::port)
103  .collect(Collectors.toSet());
104 
105  TrafficTreatment.Builder defaultTreatmentBuilder = DefaultTrafficTreatment.builder();
106  outPorts.stream()
107  .forEach(defaultTreatmentBuilder::setOutput);
108  TrafficTreatment defaultTreatment = defaultTreatmentBuilder.build();
109 
110  TrafficTreatment.Builder ingressTreatmentBuilder = DefaultTrafficTreatment.builder(intent.treatment());
111  outPorts.stream()
112  .forEach(ingressTreatmentBuilder::setOutput);
113  TrafficTreatment ingressTreatment = ingressTreatmentBuilder.build();
114 
115  List<FlowRule> rules = new ArrayList<>(inPorts.size());
116  for (PortNumber inPort: inPorts) {
117  TrafficSelector selector = DefaultTrafficSelector.builder(intent.selector()).matchInPort(inPort).build();
118  TrafficTreatment treatment;
119  if (ingressPorts.contains(inPort)) {
120  treatment = ingressTreatment;
121  } else {
122  treatment = defaultTreatment;
123  }
124 
125  FlowRule rule = DefaultFlowRule.builder()
126  .forDevice(deviceId)
127  .withSelector(selector)
128  .withTreatment(treatment)
129  .withPriority(intent.priority())
130  .fromApp(appId)
131  .makePermanent()
132  .build();
133  rules.add(rule);
134  }
135 
136  return rules;
137  }
