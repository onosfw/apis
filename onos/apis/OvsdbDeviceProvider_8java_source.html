107  @Override
108  public boolean isReachable(DeviceId deviceId) {
109  OvsdbClientService ovsdbClient = controller.getOvsdbClient(changeDeviceIdToNodeId(deviceId));
110  return !(ovsdbClient == null || !ovsdbClient.isConnected());
111  }

113  private class InnerOvsdbNodeListener implements OvsdbNodeListener {
114 
115  @Override
116  public void nodeAdded(OvsdbNodeId nodeId) {
117  checkNotNull(nodeId, ISNOTNULL);
118  DeviceId deviceId = DeviceId.deviceId(nodeId.toString());
119  URI uri = URI.create(nodeId.toString());
120  ChassisId cid = new ChassisId();
121  String ipAddress = nodeId.getIpAddress();
122  SparseAnnotations annotations = DefaultAnnotations.builder()
123  .set("ipaddress", ipAddress).build();
124  DeviceDescription deviceDescription = new DefaultDeviceDescription(
125  uri,
126  Device.Type.CONTROLLER,
127  UNKNOWN, UNKNOWN,
128  UNKNOWN, UNKNOWN,
129  cid,
130  annotations);
131  providerService.deviceConnected(deviceId, deviceDescription);
132 
133  }
134 
135  @Override
136  public void nodeRemoved(OvsdbNodeId nodeId) {
137  checkNotNull(nodeId, ISNOTNULL);
138  DeviceId deviceId = DeviceId.deviceId(nodeId.toString());
139  providerService.deviceDisconnected(deviceId);
140 
141  }
142  }
143 
144  private OvsdbNodeId changeDeviceIdToNodeId(DeviceId deviceId) {
145  String[] strings = deviceId.toString().split(":");
146  if (strings.length < 1) {
147  return null;
148  }
149  return new OvsdbNodeId(IpAddress.valueOf(strings[1]), 0);
150  }
