342  private V removeAndNotify(K key, V value) {
343  Timestamp timestamp = timestampProvider.apply(key, value);
344  Optional<MapValue<V>> tombstone = tombstonesDisabled || timestamp == null
345  ? Optional.empty() : Optional.of(MapValue.tombstone(timestamp));
346  MapValue<V> previousValue = removeInternal(key, Optional.ofNullable(value), tombstone);
347  if (previousValue != null) {
348  notifyPeers(new UpdateEntry<>(key, tombstone.orElse(null)),
349  peerUpdateFunction.apply(key, previousValue.get()));
350  if (previousValue.isAlive()) {
351  notifyListeners(new EventuallyConsistentMapEvent<>(mapName, REMOVE, key, previousValue.get()));
352  }
353  }
354  return previousValue != null ? previousValue.get() : null;
355  }

357  private MapValue<V> removeInternal(K key, Optional<V> value, Optional<MapValue<V>> tombstone) {
358  checkState(!destroyed, destroyedMessage);
359  checkNotNull(key, ERROR_NULL_KEY);
360  checkNotNull(value, ERROR_NULL_VALUE);
361  tombstone.ifPresent(v -> checkState(v.isTombstone()));
362 
363  counter.incrementCount();
364  AtomicBoolean updated = new AtomicBoolean(false);
365  AtomicReference<MapValue<V>> previousValue = new AtomicReference<>();
366  items.compute(key, (k, existing) -> {
367  boolean valueMatches = true;
368  if (value.isPresent() && existing != null && existing.isAlive()) {
369  valueMatches = Objects.equals(value.get(), existing.get());
370  }
371  if (existing == null) {
372  log.trace("ECMap Remove: Existing value for key {} is already null", k);
373  }
374  if (valueMatches) {
375  if (existing == null) {
376  updated.set(tombstone.isPresent());
377  } else {
378  updated.set(!tombstone.isPresent() || tombstone.get().isNewerThan(existing));
379  }
380  }
381  if (updated.get()) {
382  previousValue.set(existing);
383  return tombstone.orElse(null);
384  } else {
385  return existing;
386  }
387  });
388  return previousValue.get();
389  }

391  @Override
392  public V compute(K key, BiFunction<K, V, V> recomputeFunction) {
393  checkState(!destroyed, destroyedMessage);
394  checkNotNull(key, ERROR_NULL_KEY);
395  checkNotNull(recomputeFunction, "Recompute function cannot be null");
396 
397  AtomicBoolean updated = new AtomicBoolean(false);
398  AtomicReference<MapValue<V>> previousValue = new AtomicReference<>();
399  MapValue<V> computedValue = items.compute(key, (k, mv) -> {
400  previousValue.set(mv);
401  V newRawValue = recomputeFunction.apply(key, mv == null ? null : mv.get());
402  MapValue<V> newValue = new MapValue<>(newRawValue, timestampProvider.apply(key, newRawValue));
403  if (mv == null || newValue.isNewerThan(mv)) {
404  updated.set(true);
405  return newValue;
406  } else {
407  return mv;
408  }
409  });
410  if (updated.get()) {
411  notifyPeers(new UpdateEntry<>(key, computedValue), peerUpdateFunction.apply(key, computedValue.get()));
412  EventuallyConsistentMapEvent.Type updateType = computedValue.isTombstone() ? REMOVE : PUT;
413  V value = computedValue.isTombstone()
414  ? previousValue.get() == null ? null : previousValue.get().get()
415  : computedValue.get();
416  if (value != null) {
417  notifyListeners(new EventuallyConsistentMapEvent<>(mapName, updateType, key, value));
418  }
419  }
420  return computedValue.get();
421  }

573  private void handleAntiEntropyAdvertisement(AntiEntropyAdvertisement<K> ad) {
574  if (destroyed || underHighLoad()) {
575  return;
576  }
577  try {
578  if (log.isTraceEnabled()) {
579  log.trace("Received anti-entropy advertisement from {} for {} with {} entries in it",
580  mapName, ad.sender(), ad.digest().size());
581  }
582  antiEntropyCheckLocalItems(ad).forEach(this::notifyListeners);
583 
584  if (!lightweightAntiEntropy) {
585  // if remote ad has any entries that the local copy is missing, actively sync
586  // TODO: Missing keys is not the way local copy can be behind.
587  if (Sets.difference(ad.digest().keySet(), items.keySet()).size() > 0) {
588  // TODO: Send ad for missing keys and for entries that are stale
589  sendAdvertisementToPeer(ad.sender());
590  }
591  }
592  } catch (Exception e) {
593  log.warn("Error handling anti-entropy advertisement", e);
594  }
595  }

597  /**
598  * Processes anti-entropy ad from peer by taking following actions:
599  * 1. If peer has an old entry, updates peer.
600  * 2. If peer indicates an entry is removed and has a more recent
601  * timestamp than the local entry, update local state.
602  */
603  private List<EventuallyConsistentMapEvent<K, V>> antiEntropyCheckLocalItems(
604  AntiEntropyAdvertisement<K> ad) {
605  final List<EventuallyConsistentMapEvent<K, V>> externalEvents = Lists.newLinkedList();
606  final NodeId sender = ad.sender();
607  items.forEach((key, localValue) -> {
608  MapValue.Digest remoteValueDigest = ad.digest().get(key);
609  if (remoteValueDigest == null || localValue.isNewerThan(remoteValueDigest.timestamp())) {
610  // local value is more recent, push to sender
611  queueUpdate(new UpdateEntry<>(key, localValue), ImmutableList.of(sender));
612  }
613  if (remoteValueDigest != null
614  && remoteValueDigest.isNewerThan(localValue.digest())
615  && remoteValueDigest.isTombstone()) {
616  MapValue<V> tombstone = MapValue.tombstone(remoteValueDigest.timestamp());
617  MapValue<V> previousValue = removeInternal(key,
618  Optional.empty(),
619  Optional.of(tombstone));
620  if (previousValue != null && previousValue.isAlive()) {
621  externalEvents.add(new EventuallyConsistentMapEvent<>(mapName, REMOVE, key, previousValue.get()));
622  }
623  }
624  });
625  return externalEvents;
626  }

628  private void processUpdates(Collection<UpdateEntry<K, V>> updates) {
629  if (destroyed) {
630  return;
631  }
632  updates.forEach(update -> {
633  final K key = update.key();
634  final MapValue<V> value = update.value();
635  if (value == null || value.isTombstone()) {
636  MapValue<V> previousValue = removeInternal(key, Optional.empty(), Optional.ofNullable(value));
637  if (previousValue != null && previousValue.isAlive()) {
638  notifyListeners(new EventuallyConsistentMapEvent<>(mapName, REMOVE, key, previousValue.get()));
639  }
640  } else if (putInternal(key, value)) {
641  notifyListeners(new EventuallyConsistentMapEvent<>(mapName, PUT, key, value.get()));
642  }
643  });
644  }
