195  private Application loadFromDisk(String appName) {
196  for (int i = 0; i < MAX_LOAD_RETRIES; i++) {
197  try {
198  // Directly return if app already exists
199  ApplicationId appId = getId(appName);
200  if (appId != null) {
201  return getApplication(appId);
202  }
203 
204  ApplicationDescription appDesc = getApplicationDescription(appName);
205  boolean success = appDesc.requiredApps().stream()
206  .noneMatch(requiredApp -> loadFromDisk(requiredApp) == null);
207  return success ? create(appDesc, false) : null;
208  } catch (Exception e) {
209  log.warn("Unable to load application {} from disk; retrying", appName);
210  randomDelay(RETRY_DELAY_MS); //FIXME: This is a deliberate hack; fix in Falcon
211  }
212  }
213  return null;
214  }

373  /**
374  * Listener to application state distributed map changes.
375  */
376  private final class InternalAppStatesListener
377  implements EventuallyConsistentMapListener<Application, InternalState> {
378  @Override
379  public void event(EventuallyConsistentMapEvent<Application, InternalState> event) {
380  // If we do not have a delegate, refuse to process any events entirely.
381  // This is to allow the anti-entropy to kick in and process the events
382  // perhaps a bit later, but with opportunity to notify delegate.
383  if (delegate == null) {
384  return;
385  }
386 
387  Application app = event.key();
388  InternalState state = event.value();
389 
390  if (event.type() == PUT) {
391  if (state == INSTALLED) {
392  fetchBitsIfNeeded(app);
393  delegate.notify(new ApplicationEvent(APP_INSTALLED, app));
394 
395  } else if (state == ACTIVATED) {
396  installAppIfNeeded(app);
397  setActive(app.id().name());
398  delegate.notify(new ApplicationEvent(APP_ACTIVATED, app));
399 
400  } else if (state == DEACTIVATED) {
401  clearActive(app.id().name());
402  delegate.notify(new ApplicationEvent(APP_DEACTIVATED, app));
403  }
404  } else if (event.type() == REMOVE) {
405  delegate.notify(new ApplicationEvent(APP_UNINSTALLED, app));
406  purgeApplication(app.id().name());
407  }
408  }
409  }

442  /**
443  * Fetches the bits from the cluster peers.
444  */
445  private void fetchBits(Application app) {
446  ControllerNode localNode = clusterService.getLocalNode();
447  CountDownLatch latch = new CountDownLatch(1);
448 
449  // FIXME: send message with name & version to make sure we don't get served old bits
450 
451  log.info("Downloading bits for application {}", app.id().name());
452  for (ControllerNode node : clusterService.getNodes()) {
453  if (latch.getCount() == 0) {
454  break;
455  }
456  if (node.equals(localNode)) {
457  continue;
458  }
459  clusterCommunicator.sendAndReceive(app.id().name(),
460  APP_BITS_REQUEST,
461  s -> s.getBytes(Charsets.UTF_8),
462  Function.identity(),
463  node.id())
464  .whenCompleteAsync((bits, error) -> {
465  if (error == null && latch.getCount() > 0) {
466  saveApplication(new ByteArrayInputStream(bits));
467  log.info("Downloaded bits for application {} from node {}",
468  app.id().name(), node.id());
469  latch.countDown();
470  } else if (error != null) {
471  log.warn("Unable to fetch bits for application {} from node {}",
472  app.id().name(), node.id());
473  }
474  }, executor);
475  }
476 
477  try {
478  if (!latch.await(FETCH_TIMEOUT_MS, MILLISECONDS)) {
479  log.warn("Unable to fetch bits for application {}", app.id().name());
480  }
481  } catch (InterruptedException e) {
482  log.warn("Interrupted while fetching bits for application {}", app.id().name());
483  }
484  }
