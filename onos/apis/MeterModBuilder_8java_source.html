102  private OFMeterMod.Builder builderMeterMod() {
103  OFMeterMod.Builder builder = factory.buildMeterMod();
104  int flags = 0;
105  if (burst) {
106  // covering loxi short comings.
107  flags |= 1 << OFMeterFlags.BURST.ordinal();
108  }
109  switch (unit) {
110  case PKTS_PER_SEC:
111  flags |= 1 << OFMeterFlags.PKTPS.ordinal();
112  break;
113  case KB_PER_SEC:
114  flags |= 1 << OFMeterFlags.KBPS.ordinal();
115  break;
116  default:
117  log.warn("Unknown unit type {}", unit);
118  }
119  //FIXME: THIS WILL CHANGE IN OF1.4 to setBands.
120  builder.setMeters(buildBands());
121  builder.setFlags(flags)
122  .setMeterId(id)
123  .setXid(xid);
124  return builder;
125  }
126 
127  private List<OFMeterBand> buildBands() {
128  return bands.stream().map(b -> {
129  switch (b.type()) {
130  case DROP:
131  OFMeterBandDrop.Builder dropBuilder =
132  factory.meterBands().buildDrop();
133  if (burst) {
134  dropBuilder.setBurstSize(b.burst());
135  }
136  dropBuilder.setRate(b.rate());
137  return dropBuilder.build();
138  case REMARK:
139  OFMeterBandDscpRemark.Builder remarkBand =
140  factory.meterBands().buildDscpRemark();
141  if (burst) {
142  remarkBand.setBurstSize(b.burst());
143  }
144  remarkBand.setRate(b.rate());
145  remarkBand.setPrecLevel(b.dropPrecedence());
146  return remarkBand.build();
147  default:
148  log.warn("Unknown band type {}", b.type());
149  return null;
150  }
151  }).filter(value -> value != null).collect(Collectors.toList());
152  }

158  }
