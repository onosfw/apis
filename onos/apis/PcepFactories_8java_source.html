66  * +-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
67  * | Ver | Flags | |

75  packetVersion = (byte) (packetVersion >> SHIFT_FLAG);

82  break;
83  default:
84  throw new PcepParseException("Unknown Packet version: " + packetVersion);
85  }
