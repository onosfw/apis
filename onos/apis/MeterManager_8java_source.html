115  @Override
116  public Meter submit(MeterRequest request) {
117 
118  MeterId id = allocateMeterId(request.deviceId());
119 
120  Meter.Builder mBuilder = DefaultMeter.builder()
121  .forDevice(request.deviceId())
122  .fromApp(request.appId())
123  .withBands(request.bands())
124  .withId(id)
125  .withUnit(request.unit());
126 
127  if (request.isBurst()) {
128  mBuilder.burst();
129  }
130  DefaultMeter m = (DefaultMeter) mBuilder.build();
131  m.setState(MeterState.PENDING_ADD);
132  store.storeMeter(m).whenComplete((result, error) ->
133  onComplete.accept(request, result, error));

135  }

167  private MeterId allocateMeterId(DeviceId deviceId) {
168  long id = meterIdCounters.compute(deviceId, (k, v) -> {
169  if (v == null) {
170  return allocateCounter(k);

172  return v;
173  }).incrementAndGet();
174 
175  return MeterId.meterId((int) id);

208  .collect(Collectors.toMap(Meter::id, m -> m));

228  @Override
229  public void notify(MeterEvent event) {
230  DeviceId deviceId = event.subject().deviceId();
231  MeterProvider p = getProvider(event.subject().deviceId());
232  switch (event.type()) {
233  case METER_ADD_REQ:
234  p.performMeterOperation(deviceId, new MeterOperation(event.subject(),
235  MeterOperation.Type.ADD));
236  break;
237  case METER_REM_REQ:
238  p.performMeterOperation(deviceId, new MeterOperation(event.subject(),
239  MeterOperation.Type.REMOVE));
240  break;
241  default:
242  log.warn("Unknown meter event {}", event.type());
243  }
