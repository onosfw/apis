49  /**
50  * This locks map has a structure similar to the "tables" map above and
51  * holds all the provisional updates made during a transaction's prepare phase.
52  * The entry value is represented as the tuple: (transactionId, newValue)
53  * If newValue == null that signifies this update is attempting to
54  * delete the existing value.
55  * This map also serves as a lock on the entries that are being updated.
56  * The presence of a entry in this map indicates that element is
57  * participating in a transaction and is currently locked for updates.
58  */
59  private Map<String, Map<String, Update>> locks;

129  @Override
130  public Result<UpdateResult<String, byte[]>> mapUpdate(
131  String mapName,
132  String key,
133  Match<byte[]> valueMatch,
134  Match<Long> versionMatch,
135  byte[] value) {
136  if (isLockedForUpdates(mapName, key)) {
137  return Result.locked();
138  }
139  Versioned<byte[]> currentValue = getMap(mapName).get(key);
140  if (!valueMatch.matches(currentValue == null ? null : currentValue.value()) ||
141  !versionMatch.matches(currentValue == null ? null : currentValue.version())) {
142  return Result.ok(new UpdateResult<>(false, mapName, key, currentValue, currentValue));
143  } else {
144  if (value == null) {
145  if (currentValue == null) {
146  return Result.ok(new UpdateResult<>(false, mapName, key, null, null));
147  } else {
148  getMap(mapName).remove(key);
149  return Result.ok(new UpdateResult<>(true, mapName, key, currentValue, null));
150  }
151  }
152  Versioned<byte[]> newValue = new Versioned<>(value, ++nextVersion);
153  getMap(mapName).put(key, newValue);
154  return Result.ok(new UpdateResult<>(true, mapName, key, currentValue, newValue));
155  }
156  }

234  @Override
235  public boolean prepare(Transaction transaction) {
236  if (transaction.updates().stream().anyMatch(update ->
237  isLockedByAnotherTransaction(update.mapName(),
238  update.key(),
239  transaction.id()))) {
240  return false;
241  }
242 
243  if (transaction.updates().stream().allMatch(this::isUpdatePossible)) {
244  transaction.updates().forEach(update -> doProvisionalUpdate(update, transaction.id()));
245  return true;
246  }
247  return false;
248  }

278  private boolean isUpdatePossible(DatabaseUpdate update) {
279  Versioned<byte[]> existingEntry = mapGet(update.mapName(), update.key());
280  switch (update.type()) {
281  case PUT:
282  case REMOVE:
283  return true;
284  case PUT_IF_ABSENT:
285  return existingEntry == null;
286  case PUT_IF_VERSION_MATCH:
287  return existingEntry != null && existingEntry.version() == update.currentVersion();
288  case PUT_IF_VALUE_MATCH:
289  return existingEntry != null && Arrays.equals(existingEntry.value(), update.currentValue());
290  case REMOVE_IF_VERSION_MATCH:
291  return existingEntry == null || existingEntry.version() == update.currentVersion();
292  case REMOVE_IF_VALUE_MATCH:
293  return existingEntry == null || Arrays.equals(existingEntry.value(), update.currentValue());
294  default:
295  throw new IllegalStateException("Unsupported type: " + update.type());
296  }
297  }

299  private void doProvisionalUpdate(DatabaseUpdate update, long transactionId) {
300  Map<String, Update> lockMap = getLockMap(update.mapName());
301  switch (update.type()) {
302  case PUT:
303  case PUT_IF_ABSENT:
304  case PUT_IF_VERSION_MATCH:
305  case PUT_IF_VALUE_MATCH:
306  lockMap.put(update.key(), new Update(transactionId, update.value()));
307  break;
308  case REMOVE:
309  case REMOVE_IF_VERSION_MATCH:
310  case REMOVE_IF_VALUE_MATCH:
311  lockMap.put(update.key(), new Update(transactionId, null));
312  break;
313  default:
314  throw new IllegalStateException("Unsupported type: " + update.type());
315  }
316  }

318  private UpdateResult<String, byte[]> commitProvisionalUpdate(DatabaseUpdate update, long transactionId) {
319  String mapName = update.mapName();
320  String key = update.key();
321  Update provisionalUpdate = getLockMap(mapName).get(key);
322  if (Objects.equal(transactionId, provisionalUpdate.transactionId())) {
323  getLockMap(mapName).remove(key);
324  } else {
325  throw new IllegalStateException("Invalid transaction Id");
326  }
327  return mapUpdate(mapName, key, Match.any(), Match.any(), provisionalUpdate.value()).value();
328  }
329 
330  private void undoProvisionalUpdate(DatabaseUpdate update, long transactionId) {
331  String mapName = update.mapName();
332  String key = update.key();
333  Update provisionalUpdate = getLockMap(mapName).get(key);
334  if (provisionalUpdate == null) {
335  return;
336  }
337  if (Objects.equal(transactionId, provisionalUpdate.transactionId())) {
338  getLockMap(mapName).remove(key);
339  }
340  }

342  private boolean isLockedByAnotherTransaction(String mapName, String key, long transactionId) {
343  Update update = getLockMap(mapName).get(key);
344  return update != null && !Objects.equal(transactionId, update.transactionId());
345  }
