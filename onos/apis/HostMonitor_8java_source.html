53 /**
54  * Monitors hosts on the dataplane to detect changes in host data.
55  * <p>
56  * The HostMonitor can monitor hosts that have already been detected for
57  * changes. At an application's request, it can also monitor and actively
58  * probe for hosts that have not yet been detected (specified by IP address).
59  * </p>
60  */
61 public class HostMonitor implements TimerTask {
62 
63  private Logger log = LoggerFactory.getLogger(getClass());

80  /**
81  * Creates a new host monitor.
82  *
83  * @param packetService packet service used to send packets on the data plane

89  public HostMonitor(PacketService packetService, HostManager hostManager,

93  this.packetService = packetService;

95  this.interfaceService = interfaceService;
96  this.edgePortService = edgePortService;

143  * Registers a host provider with the host monitor. The monitor can use the

152  @Override
153  public void run(Timeout timeout) throws Exception {
154  for (IpAddress ip : monitoredAddresses) {
155  Set<Host> hosts = hostManager.getHostsByIp(ip);
156 
157  if (hosts.isEmpty()) {
158  sendRequest(ip);
159  } else {
160  for (Host host : hosts) {
161  HostProvider provider = hostProviders.get(host.providerId());
162  if (provider == null) {
163  hostProviders.remove(host.providerId(), null);
164  } else {
165  provider.triggerProbe(host);
166  }
167  }
168  }
169  }
170 
171  synchronized (this) {
172  this.timeout = Timer.getTimer().newTimeout(this, probeRate, TimeUnit.MILLISECONDS);
173  }
174  }

176  /**
177  * Sends an ARP or NDP request for the given IP address.
178  *
179  * @param targetIp IP address to send the request for
180  */
181  private void sendRequest(IpAddress targetIp) {
182  Interface intf = interfaceService.getMatchingInterface(targetIp);
183 
184  if (intf == null) {
185  return;
186  }
187 
188  if (!edgePortService.isEdgePoint(intf.connectPoint())) {
189  log.warn("Attempt to send probe out non-edge port: {}", intf);
190  return;
191  }
192 
193  for (InterfaceIpAddress ia : intf.ipAddresses()) {
194  if (ia.subnetAddress().contains(targetIp)) {
195  sendProbe(intf.connectPoint(), targetIp, ia.ipAddress(),
196  intf.mac(), intf.vlan());
197  }
198  }
199  }
200 
201  private void sendProbe(ConnectPoint connectPoint,
202  IpAddress targetIp,
203  IpAddress sourceIp, MacAddress sourceMac,
204  VlanId vlan) {
205  Ethernet probePacket = null;
206 
207  if (targetIp.isIp4()) {
208  // IPv4: Use ARP
209  probePacket = buildArpRequest(targetIp, sourceIp, sourceMac, vlan);
210  } else {
211  // IPv6: Use Neighbor Discovery
212  probePacket = buildNdpRequest(targetIp, sourceIp, sourceMac, vlan);
213  }
214 
215  TrafficTreatment treatment = DefaultTrafficTreatment.builder()
216  .setOutput(connectPoint.port())
217  .build();
218 
219  OutboundPacket outboundPacket =
220  new DefaultOutboundPacket(connectPoint.deviceId(), treatment,
221  ByteBuffer.wrap(probePacket.serialize()));
222 
223  packetService.emit(outboundPacket);
224  }
225 
226  private Ethernet buildArpRequest(IpAddress targetIp, IpAddress sourceIp,
227  MacAddress sourceMac, VlanId vlan) {
228 
229  ARP arp = new ARP();
230  arp.setHardwareType(ARP.HW_TYPE_ETHERNET)
231  .setHardwareAddressLength((byte) Ethernet.DATALAYER_ADDRESS_LENGTH)
232  .setProtocolType(ARP.PROTO_TYPE_IP)
233  .setProtocolAddressLength((byte) IpAddress.INET_BYTE_LENGTH)
234  .setOpCode(ARP.OP_REQUEST);
235 
236  arp.setSenderHardwareAddress(sourceMac.toBytes())
237  .setSenderProtocolAddress(sourceIp.toOctets())
238  .setTargetHardwareAddress(ZERO_MAC_ADDRESS)
239  .setTargetProtocolAddress(targetIp.toOctets());
240 
241  Ethernet ethernet = new Ethernet();
242  ethernet.setEtherType(Ethernet.TYPE_ARP)
243  .setDestinationMACAddress(MacAddress.BROADCAST)
244  .setSourceMACAddress(sourceMac)
245  .setPayload(arp);
246 
247  if (!vlan.equals(VlanId.NONE)) {
248  ethernet.setVlanID(vlan.toShort());
249  }
250 
251  ethernet.setPad(true);
252 
253  return ethernet;
254  }
255 
256  private Ethernet buildNdpRequest(IpAddress targetIp, IpAddress sourceIp,
257  MacAddress sourceMac, VlanId vlan) {
258 
259  // Create the Ethernet packet
260  Ethernet ethernet = new Ethernet();
261  ethernet.setEtherType(Ethernet.TYPE_IPV6)
262  .setDestinationMACAddress(MacAddress.BROADCAST)
263  .setSourceMACAddress(sourceMac);
264  if (!vlan.equals(VlanId.NONE)) {
265  ethernet.setVlanID(vlan.toShort());
266  }
267 
268  //
269  // Create the IPv6 packet
270  //
271  // TODO: The destination IP address should be the
272  // solicited-node multicast address
273  IPv6 ipv6 = new IPv6();
274  ipv6.setSourceAddress(sourceIp.toOctets());
275  ipv6.setDestinationAddress(targetIp.toOctets());
276  ipv6.setHopLimit((byte) 255);
277 
278  // Create the ICMPv6 packet
279  ICMP6 icmp6 = new ICMP6();
280  icmp6.setIcmpType(ICMP6.NEIGHBOR_SOLICITATION);
281  icmp6.setIcmpCode((byte) 0);
282 
283  // Create the Neighbor Solicitation packet
284  NeighborSolicitation ns = new NeighborSolicitation();
285  ns.setTargetAddress(targetIp.toOctets());
286  ns.addOption(NeighborDiscoveryOptions.TYPE_SOURCE_LL_ADDRESS,
287  sourceMac.toBytes());
288 
289  icmp6.setPayload(ns);
290  ipv6.setPayload(icmp6);
291  ethernet.setPayload(ipv6);
292 
293  return ethernet;
294  }
