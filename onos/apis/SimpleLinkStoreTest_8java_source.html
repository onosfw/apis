392  private void removeOrDownLink(boolean isDurable) {
393  final ConnectPoint d1P1 = new ConnectPoint(DID1, P1);
394  final ConnectPoint d2P2 = new ConnectPoint(DID2, P2);
395  LinkKey linkId1 = LinkKey.linkKey(d1P1, d2P2);
396  LinkKey linkId2 = LinkKey.linkKey(d2P2, d1P1);
397 
398  putLink(linkId1, DIRECT, isDurable ? DA1 : A1);
399  putLink(linkId2, DIRECT, isDurable ? DA2 : A2);
400 
401  // DID1,P1 => DID2,P2
402  // DID2,P2 => DID1,P1
403  // DID1,P2 => DID2,P3
404 
405  LinkEvent event = linkStore.removeOrDownLink(d1P1, d2P2);
406  assertEquals(isDurable ? LINK_UPDATED : LINK_REMOVED, event.type());
407  assertAnnotationsEquals(event.subject().annotations(), isDurable ? DA1 : A1);
408  LinkEvent event2 = linkStore.removeOrDownLink(d1P1, d2P2);
409  assertNull(event2);
410 
411  assertLink(linkId2, DIRECT, linkStore.getLink(d2P2, d1P1));
412  assertAnnotationsEquals(linkStore.getLink(d2P2, d1P1).annotations(),
413  isDurable ? DA2 : A2);
414 
415  // annotations, etc. should not survive remove
416  if (!isDurable) {
417  putLink(linkId1, DIRECT);
418  assertLink(linkId1, DIRECT, linkStore.getLink(d1P1, d2P2));
419  assertAnnotationsEquals(linkStore.getLink(d1P1, d2P2).annotations());
420  }
421  }

500  final CountDownLatch addLatch = new CountDownLatch(1);
501  LinkStoreDelegate checkAdd = event -> {
502  assertEquals(LINK_ADDED, event.type());
503  assertLink(linkId1, INDIRECT, event.subject());
504  addLatch.countDown();
505  };
506  final CountDownLatch updateLatch = new CountDownLatch(1);
507  LinkStoreDelegate checkUpdate = event -> {
508  assertEquals(LINK_UPDATED, event.type());
509  assertLink(linkId1, DIRECT, event.subject());
510  updateLatch.countDown();
511  };
512  final CountDownLatch removeLatch = new CountDownLatch(1);
513  LinkStoreDelegate checkRemove = event -> {
514  assertEquals(LINK_REMOVED, event.type());
515  assertLink(linkId1, DIRECT, event.subject());
516  removeLatch.countDown();
517  };
518 
519  linkStore.setDelegate(checkAdd);
520  putLink(linkId1, INDIRECT);
521  assertTrue("Add event fired", addLatch.await(1, TimeUnit.SECONDS));
522 
523  linkStore.unsetDelegate(checkAdd);
524  linkStore.setDelegate(checkUpdate);
525  putLink(linkId1, DIRECT);
526  assertTrue("Update event fired", updateLatch.await(1, TimeUnit.SECONDS));
527 
528  linkStore.unsetDelegate(checkUpdate);
529  linkStore.setDelegate(checkRemove);
530  linkStore.removeOrDownLink(d1P1, d2P2);
531  assertTrue("Remove event fired", removeLatch.await(1, TimeUnit.SECONDS));
