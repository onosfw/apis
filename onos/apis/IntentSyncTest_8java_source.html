146  /**
147  * Tests the synchronization behavior of intent synchronizer. We set up
148  * a discrepancy between the intent service state and the intent
149  * synchronizer's state and ensure that this is reconciled correctly.
150  *
151  * @throws TestUtilsException
152  */
153  @Test
154  public void testIntentSync() throws TestUtilsException {
155 
156  //
157  // Construct routes and intents.
158  // This test simulates the following cases during the master change
159  // time interval:
160  // 1. RouteEntry1 did not change and the intent also did not change.
161  // 2. RouteEntry2 was deleted, but the intent was not deleted.
162  // 3. RouteEntry3 was newly added, and the intent was also submitted.
163  // 4. RouteEntry4 was updated to RouteEntry4Update, and the intent was
164  // also updated to a new one.
165  // 5. RouteEntry5 did not change, but its intent id changed.
166  // 6. RouteEntry6 was newly added, but the intent was not submitted.
167  //
168  RouteEntry routeEntry1 = new RouteEntry(
169  Ip4Prefix.valueOf("1.1.1.0/24"),
170  Ip4Address.valueOf("192.168.10.1"));
171 
172  RouteEntry routeEntry2 = new RouteEntry(
173  Ip4Prefix.valueOf("2.2.2.0/24"),
174  Ip4Address.valueOf("192.168.20.1"));
175 
176  RouteEntry routeEntry3 = new RouteEntry(
177  Ip4Prefix.valueOf("3.3.3.0/24"),
178  Ip4Address.valueOf("192.168.30.1"));
179 
180  RouteEntry routeEntry4 = new RouteEntry(
181  Ip4Prefix.valueOf("4.4.4.0/24"),
182  Ip4Address.valueOf("192.168.30.1"));
183 
184  RouteEntry routeEntry4Update = new RouteEntry(
185  Ip4Prefix.valueOf("4.4.4.0/24"),
186  Ip4Address.valueOf("192.168.20.1"));
187 
188  RouteEntry routeEntry5 = new RouteEntry(
189  Ip4Prefix.valueOf("5.5.5.0/24"),
190  Ip4Address.valueOf("192.168.10.1"));
191 
192  RouteEntry routeEntry6 = new RouteEntry(
193  Ip4Prefix.valueOf("6.6.6.0/24"),
194  Ip4Address.valueOf("192.168.10.1"));
195 
196  RouteEntry routeEntry7 = new RouteEntry(
197  Ip4Prefix.valueOf("7.7.7.0/24"),
198  Ip4Address.valueOf("192.168.10.1"));
199 
200  MultiPointToSinglePointIntent intent1 = intentBuilder(
201  routeEntry1.prefix(), "00:00:00:00:00:01", SW1_ETH1);
202  MultiPointToSinglePointIntent intent2 = intentBuilder(
203  routeEntry2.prefix(), "00:00:00:00:00:02", SW2_ETH1);
204  MultiPointToSinglePointIntent intent3 = intentBuilder(
205  routeEntry3.prefix(), "00:00:00:00:00:03", SW3_ETH1);
206  MultiPointToSinglePointIntent intent4 = intentBuilder(
207  routeEntry4.prefix(), "00:00:00:00:00:03", SW3_ETH1);
208  MultiPointToSinglePointIntent intent4Update = intentBuilder(
209  routeEntry4Update.prefix(), "00:00:00:00:00:02", SW2_ETH1);
210  MultiPointToSinglePointIntent intent5 = intentBuilder(
211  routeEntry5.prefix(), "00:00:00:00:00:01", SW1_ETH1);
212  MultiPointToSinglePointIntent intent7 = intentBuilder(
213  routeEntry7.prefix(), "00:00:00:00:00:01", SW1_ETH1);
214 
215  // Compose a intent, which is equal to intent5 but the id is different.
216  MultiPointToSinglePointIntent intent5New =
217  staticIntentBuilder(intent5, routeEntry5, "00:00:00:00:00:01");
218  assertThat(IntentUtils.equals(intent5, intent5New), is(true));
219  assertFalse(intent5.equals(intent5New));
220 
221  MultiPointToSinglePointIntent intent6 = intentBuilder(
222  routeEntry6.prefix(), "00:00:00:00:00:01", SW1_ETH1);
223 
224  // Set up expectation
225  Set<Intent> intents = new HashSet<>();
226  intents.add(intent1);
227  expect(intentService.getIntentState(intent1.key()))
228  .andReturn(IntentState.INSTALLED).anyTimes();
229  intents.add(intent2);
230  expect(intentService.getIntentState(intent2.key()))
231  .andReturn(IntentState.INSTALLED).anyTimes();
232  intents.add(intent4);
233  expect(intentService.getIntentState(intent4.key()))
234  .andReturn(IntentState.INSTALLED).anyTimes();
235  intents.add(intent5);
236  expect(intentService.getIntentState(intent5.key()))
237  .andReturn(IntentState.INSTALLED).anyTimes();
238  intents.add(intent7);
239  expect(intentService.getIntentState(intent7.key()))
240  .andReturn(IntentState.WITHDRAWING).anyTimes();
241  expect(intentService.getIntents()).andReturn(intents).anyTimes();
242 
243  // These are the operations that should be done to the intentService
244  // during synchronization
245  intentService.withdraw(intent2);
246  intentService.submit(intent3);
247  intentService.submit(intent4Update);
248  intentService.submit(intent6);
249  intentService.submit(intent7);
250  replay(intentService);
251 
252  // Start the test
253 
254  // Simulate some input from the clients. The intent synchronizer has not
255  // gained the global leadership yet, but it will remember this input for
256  // when it does.
257  intentSynchronizer.submit(intent1);
258  intentSynchronizer.submit(intent2);
259  intentSynchronizer.withdraw(intent2);
260  intentSynchronizer.submit(intent3);
261  intentSynchronizer.submit(intent4);
262  intentSynchronizer.submit(intent4Update);
263  intentSynchronizer.submit(intent5);
264  intentSynchronizer.submit(intent6);
265  intentSynchronizer.submit(intent7);
266 
267  // Give the leadership to the intent synchronizer. It will now attempt
268  // to synchronize the intents in the store with the intents it has
269  // recorded based on the earlier user input.
270  intentSynchronizer.leaderChanged(true);
271 
272  verify(intentService);
273  }

293 
294  // Test the submit

296 
297  verify(intentService);
298

300  // that it does not submit any intents to the intent service when we
301  // call the submit API
302  reset(intentService);
303  replay(intentService);
304 
305  intentSynchronizer.leaderChanged(false);
306 
307  intentSynchronizer.submit(intent);
308 
309  verify(intentService);

352  /**
353  * MultiPointToSinglePointIntent builder.
354  *
355  * @param ipPrefix the ipPrefix to match
356  * @param nextHopMacAddress to which the destination MAC address in packet
357  * should be rewritten
358  * @param egressPoint to which packets should be sent
359  * @return the constructed MultiPointToSinglePointIntent
360  */
361  private MultiPointToSinglePointIntent intentBuilder(IpPrefix ipPrefix,
362  String nextHopMacAddress, ConnectPoint egressPoint) {
363 
364  TrafficSelector.Builder selectorBuilder =
365  DefaultTrafficSelector.builder();
366  if (ipPrefix.isIp4()) {
367  selectorBuilder.matchEthType(Ethernet.TYPE_IPV4);
368  selectorBuilder.matchIPDst(ipPrefix);
369  } else {
370  selectorBuilder.matchEthType(Ethernet.TYPE_IPV6);
371  selectorBuilder.matchIPv6Dst(ipPrefix);
372  }
373 
374  TrafficTreatment.Builder treatmentBuilder =
375  DefaultTrafficTreatment.builder();
376  treatmentBuilder.setEthDst(MacAddress.valueOf(nextHopMacAddress));
377 
378  Set<ConnectPoint> ingressPoints = new HashSet<>();
379  for (Interface intf : interfaces) {
380  if (!intf.connectPoint().equals(egressPoint)) {
381  ConnectPoint srcPort = intf.connectPoint();
382  ingressPoints.add(srcPort);
383  }
384  }
385  MultiPointToSinglePointIntent intent =
386  MultiPointToSinglePointIntent.builder()
387  .appId(APPID)
388  .key(Key.of(ipPrefix.toString(), APPID))
389  .selector(selectorBuilder.build())
390  .treatment(treatmentBuilder.build())
391  .ingressPoints(ingressPoints)
392  .egressPoint(egressPoint)
393  .constraints(SdnIpFib.CONSTRAINTS)
394  .build();
395  return intent;
396  }

398  /**
399  * A static MultiPointToSinglePointIntent builder, the returned intent is
400  * equal to the input intent except that the id is different.
401  *
402  * @param intent the intent to be used for building a new intent
403  * @param routeEntry the relative routeEntry of the intent
404  * @return the newly constructed MultiPointToSinglePointIntent
405  * @throws TestUtilsException
406  */
407  private MultiPointToSinglePointIntent staticIntentBuilder(
408  MultiPointToSinglePointIntent intent, RouteEntry routeEntry,
409  String nextHopMacAddress) throws TestUtilsException {
410 
411  // Use a different egress ConnectPoint with that in intent
412  // to generate a different id
413  MultiPointToSinglePointIntent intentNew = intentBuilder(
414  routeEntry.prefix(), nextHopMacAddress, SW2_ETH1);
415  TestUtils.setField(intentNew, "egressPoint", intent.egressPoint());
416  TestUtils.setField(intentNew,
417  "ingressPoints", intent.ingressPoints());
418  return intentNew;
419  }
