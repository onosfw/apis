100  @Override
101  public Ip4Address suggestIP(HostId hostId, Ip4Address requestedIP) {
102 
103  IpAssignment assignmentInfo;
104  if (allocationMap.containsKey(hostId)) {
105  assignmentInfo = allocationMap.get(hostId).value();
106  IpAssignment.AssignmentStatus status = assignmentInfo.assignmentStatus();
107  Ip4Address ipAddr = assignmentInfo.ipAddress();
108 
109  if (assignmentInfo.rangeNotEnforced()) {
110  return assignmentInfo.ipAddress();
111  } else if (status == IpAssignment.AssignmentStatus.Option_Assigned ||
112  status == IpAssignment.AssignmentStatus.Option_Requested) {
113  // Client has a currently Active Binding.
114  if (ipWithinRange(ipAddr)) {
115  return ipAddr;
116  }
117 
118  } else if (status == IpAssignment.AssignmentStatus.Option_Expired) {
119  // Client has a Released or Expired Binding.
120  if (freeIPPool.contains(ipAddr)) {
121  assignmentInfo = IpAssignment.builder()
122  .ipAddress(ipAddr)
123  .timestamp(new Date())
124  .leasePeriod(timeoutForPendingAssignments)
125  .assignmentStatus(IpAssignment.AssignmentStatus.Option_Requested)
126  .build();
127  if (freeIPPool.remove(ipAddr)) {
128  allocationMap.put(hostId, assignmentInfo);
129  return ipAddr;
130  }
131  }
132  }
133  } else if (requestedIP.toInt() != 0) {
134  // Client has requested an IP.
135  if (freeIPPool.contains(requestedIP)) {
136  assignmentInfo = IpAssignment.builder()
137  .ipAddress(requestedIP)
138  .timestamp(new Date())
139  .leasePeriod(timeoutForPendingAssignments)
140  .assignmentStatus(IpAssignment.AssignmentStatus.Option_Requested)
141  .build();
142  if (freeIPPool.remove(requestedIP)) {
143  allocationMap.put(hostId, assignmentInfo);
144  return requestedIP;
145  }
146  }
147  }
148 
149  // Allocate a new IP from the server's pool of available IP.
150  Ip4Address nextIPAddr = fetchNextIP();
151  if (nextIPAddr != null) {
152  assignmentInfo = IpAssignment.builder()
153  .ipAddress(nextIPAddr)
154  .timestamp(new Date())
155  .leasePeriod(timeoutForPendingAssignments)
156  .assignmentStatus(IpAssignment.AssignmentStatus.Option_Requested)
157  .build();
158 
159  allocationMap.put(hostId, assignmentInfo);
160  }
161  return nextIPAddr;
162 
163  }

165  @Override
166  public boolean assignIP(HostId hostId, Ip4Address ipAddr, int leaseTime, boolean rangeNotEnforced,
167  List<Ip4Address> addressList) {
168 
169  IpAssignment assignmentInfo;
170 
171  log.debug("Assign IP Called w/ Ip4Address: {}, HostId: {}", ipAddr.toString(), hostId.mac().toString());
172 
173  if (allocationMap.containsKey(hostId)) {
174 
175  assignmentInfo = allocationMap.get(hostId).value();
176  IpAssignment.AssignmentStatus status = assignmentInfo.assignmentStatus();
177 
178  if (Objects.equals(assignmentInfo.ipAddress(), ipAddr) && ipWithinRange(ipAddr)) {
179 
180  if (status == IpAssignment.AssignmentStatus.Option_Assigned ||
181  status == IpAssignment.AssignmentStatus.Option_Requested) {
182  // Client has a currently active binding with the server.
183  assignmentInfo = IpAssignment.builder()
184  .ipAddress(ipAddr)
185  .timestamp(new Date())
186  .leasePeriod(leaseTime)
187  .assignmentStatus(IpAssignment.AssignmentStatus.Option_Assigned)
188  .build();
189  allocationMap.put(hostId, assignmentInfo);
190  return true;
191  } else if (status == IpAssignment.AssignmentStatus.Option_Expired) {
192  // Client has an expired binding with the server.
193  if (freeIPPool.contains(ipAddr)) {
194  assignmentInfo = IpAssignment.builder()
195  .ipAddress(ipAddr)
196  .timestamp(new Date())
197  .leasePeriod(leaseTime)
198  .assignmentStatus(IpAssignment.AssignmentStatus.Option_Assigned)
199  .build();
200  if (freeIPPool.remove(ipAddr)) {
201  allocationMap.put(hostId, assignmentInfo);
202  return true;
203  }
204  }
205  }
206  }
207  } else if (freeIPPool.contains(ipAddr)) {
208  assignmentInfo = IpAssignment.builder()
209  .ipAddress(ipAddr)
210  .timestamp(new Date())
211  .leasePeriod(leaseTime)
212  .assignmentStatus(IpAssignment.AssignmentStatus.Option_Assigned)
213  .build();
214  if (freeIPPool.remove(ipAddr)) {
215  allocationMap.put(hostId, assignmentInfo);
216  return true;
217  }
218  } else if (rangeNotEnforced) {
219  assignmentInfo = IpAssignment.builder()
220  .ipAddress(ipAddr)
221  .timestamp(new Date())
222  .leasePeriod(leaseTime)
223  .rangeNotEnforced(true)
224  .assignmentStatus(IpAssignment.AssignmentStatus.Option_RangeNotEnforced)
225  .subnetMask((Ip4Address) addressList.toArray()[0])
226  .dhcpServer((Ip4Address) addressList.toArray()[1])
227  .routerAddress((Ip4Address) addressList.toArray()[2])
228  .domainServer((Ip4Address) addressList.toArray()[3])
229  .build();
230  allocationMap.put(hostId, assignmentInfo);
231  return true;
232  }
233  return false;
234  }

236  @Override
237  public Ip4Address releaseIP(HostId hostId) {
238  if (allocationMap.containsKey(hostId)) {
239  IpAssignment newAssignment = IpAssignment.builder(allocationMap.get(hostId).value())
240  .assignmentStatus(IpAssignment.AssignmentStatus.Option_Expired)
241  .build();
242  Ip4Address freeIP = newAssignment.ipAddress();
243  allocationMap.put(hostId, newAssignment);
244  if (ipWithinRange(freeIP)) {
245  freeIPPool.add(freeIP);
246  }
247  return freeIP;
248  }
249  return null;
250  }

257  @Override
258  public Map<HostId, IpAssignment> listAssignedMapping() {
259 
260  Map<HostId, IpAssignment> validMapping = new HashMap<>();
261  IpAssignment assignment;
262  for (Map.Entry<HostId, Versioned<IpAssignment>> entry: allocationMap.entrySet()) {
263  assignment = entry.getValue().value();
264  if (assignment.assignmentStatus() == IpAssignment.AssignmentStatus.Option_Assigned
265  || assignment.assignmentStatus() == IpAssignment.AssignmentStatus.Option_RangeNotEnforced) {
266  validMapping.put(entry.getKey(), assignment);
267  }
268  }
269  return validMapping;
270  }

288  @Override
289  public boolean removeStaticIP(MacAddress macID) {
290  HostId host = HostId.hostId(macID);
291  if (allocationMap.containsKey(host)) {
292  IpAssignment assignment = allocationMap.get(host).value();
293 
294  if (assignment.rangeNotEnforced()) {
295  allocationMap.remove(host);
296  return true;
297  }
298 
299  Ip4Address freeIP = assignment.ipAddress();
300  if (assignment.leasePeriod() < 0) {
301  allocationMap.remove(host);
302  if (ipWithinRange(freeIP)) {
303  freeIPPool.add(freeIP);
304  }
305  return true;
306  }
307  }
308  return false;
309  }

316  @Override
317  public void populateIPPoolfromRange(Ip4Address startIP, Ip4Address endIP) {
318  // Clear all entries from previous range.
319  allocationMap.clear();
320  freeIPPool.clear();
321  startIPRange = startIP;
322  endIPRange = endIP;
323 
324  int lastIP = endIP.toInt();
325  Ip4Address nextIP;
326  for (int loopCounter = startIP.toInt(); loopCounter <= lastIP; loopCounter++) {
327  nextIP = Ip4Address.valueOf(loopCounter);
328  freeIPPool.add(nextIP);
329  }
330  }

337  /**
338  * Fetches the next available IP from the free pool pf IPs.
339  *
340  * @return the next available IP address
341  */
342  private Ip4Address fetchNextIP() {
343  for (Ip4Address freeIP : freeIPPool) {
344  if (freeIPPool.remove(freeIP)) {
345  return freeIP;
346  }
347  }
348  return null;
349  }

351  /**
352  * Returns true if the given ip is within the range of available IPs.
353  *
354  * @param ip given ip address
355  * @return true if within range, false otherwise
356  */
357  private boolean ipWithinRange(Ip4Address ip) {
358  if ((ip.toInt() >= startIPRange.toInt()) && (ip.toInt() <= endIPRange.toInt())) {
359  return true;
360  }
361  return false;
