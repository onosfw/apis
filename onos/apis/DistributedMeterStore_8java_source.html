139  @Override
140  public CompletableFuture<MeterStoreResult> deleteMeter(Meter meter) {
141  CompletableFuture<MeterStoreResult> future = new CompletableFuture<>();
142  MeterKey key = MeterKey.key(meter.deviceId(), meter.id());
143  futures.put(key, future);
144 
145  MeterData data = new MeterData(meter, null, local);
146 
147  // update the state of the meter. It will be pruned by observing
148  // that it has been removed from the dataplane.
149  try {
150  if (meters.computeIfPresent(key, (k, v) -> data) == null) {
151  future.complete(MeterStoreResult.success());

161  @Override
162  public CompletableFuture<MeterStoreResult> updateMeter(Meter meter) {
163  CompletableFuture<MeterStoreResult> future = new CompletableFuture<>();
164  MeterKey key = MeterKey.key(meter.deviceId(), meter.id());
165  futures.put(key, future);
166 
167  MeterData data = new MeterData(meter, null, local);
168  try {
169  if (meters.computeIfPresent(key, (k, v) -> data) == null) {
170  future.complete(MeterStoreResult.fail(MeterFailReason.INVALID_METER));

178  @Override
179  public void updateMeterState(Meter meter) {
180  MeterKey key = MeterKey.key(meter.deviceId(), meter.id());
181  meters.computeIfPresent(key, (k, v) -> {
182  DefaultMeter m = (DefaultMeter) v.meter();
183  m.setState(meter.state());

219  private class InternalMapEventListener implements MapEventListener<MeterKey, MeterData> {
220  @Override
221  public void event(MapEvent<MeterKey, MeterData> event) {
222  MeterKey key = event.key();
223  MeterData data = event.value().value();
224  NodeId master = mastershipService.getMasterFor(data.meter().deviceId());
225  switch (event.type()) {
226  case INSERT:
227  case UPDATE:
228  switch (data.meter().state()) {
229  case PENDING_ADD:
230  case PENDING_REMOVE:
231  if (!data.reason().isPresent() && local.equals(master)) {
232  notifyDelegate(
233  new MeterEvent(data.meter().state() == MeterState.PENDING_ADD ?
234  MeterEvent.Type.METER_ADD_REQ : MeterEvent.Type.METER_REM_REQ,
235  data.meter()));
236  } else if (data.reason().isPresent() && local.equals(data.origin())) {
237  MeterStoreResult msr = MeterStoreResult.fail(data.reason().get());
238  //TODO: No future -> no friend
239  futures.get(key).complete(msr);
240  }
241  break;
242  case ADDED:
243  if (local.equals(data.origin()) && data.meter().state() == MeterState.PENDING_ADD) {
244  futures.remove(key).complete(MeterStoreResult.success());
245  }
246  break;
247  case REMOVED:
248  if (local.equals(data.origin()) && data.meter().state() == MeterState.PENDING_REMOVE) {
249  futures.remove(key).complete(MeterStoreResult.success());
250  }
251  break;
252  default:
253  log.warn("Unknown meter state type {}", data.meter().state());
254  }
255  break;
256  case REMOVE:
257  //Only happens at origin so we do not need to care.
258  break;
259  default:
260  log.warn("Unknown Map event type {}", event.type());
261  }
262 
263  }
264  }
