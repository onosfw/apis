41 /**
42  * Default Transactional Map implementation that provides a repeatable reads
43  * transaction isolation level.
44  *
45  * @param <K> key type
46  * @param <V> value type.
47  */
48 public class DefaultTransactionalMap<K, V> implements TransactionalMap<K, V> {

50  private final TransactionContext txContext;

87  @Override
88  public V get(K key) {
89  checkState(txContext.isOpen(), TX_CLOSED_ERROR);
90  checkNotNull(key, ERROR_NULL_KEY);
91  if (deleteSet.contains(key)) {
92  return null;
93  }
94  V latest = writeCache.get(key);
95  if (latest != null) {
96  return latest;
97  } else {
98  Versioned<V> v = readCache.computeIfAbsent(key, k -> backingMap.get(k));
99  return v != null ? v.value() : null;
100  }
101  }
102 
103  @Override
104  public V put(K key, V value) {
105  checkState(txContext.isOpen(), TX_CLOSED_ERROR);
106  checkNotNull(value, ERROR_NULL_VALUE);
107 
108  V latest = get(key);
109  writeCache.put(key, value);
110  deleteSet.remove(key);
111  return latest;
112  }
113 
114  @Override
115  public V remove(K key) {
116  checkState(txContext.isOpen(), TX_CLOSED_ERROR);
117  V latest = get(key);
118  if (latest != null) {
119  writeCache.remove(key);
120  deleteSet.add(key);
121  }
122  return latest;
123  }
124 
125  @Override
126  public boolean remove(K key, V value) {
127  checkState(txContext.isOpen(), TX_CLOSED_ERROR);
128  checkNotNull(value, ERROR_NULL_VALUE);
129  V latest = get(key);
130  if (Objects.equal(value, latest)) {
131  remove(key);
132  return true;
133  }
134  return false;
135  }

137  @Override
138  public boolean replace(K key, V oldValue, V newValue) {
139  checkState(txContext.isOpen(), TX_CLOSED_ERROR);
140  checkNotNull(oldValue, ERROR_NULL_VALUE);
141  checkNotNull(newValue, ERROR_NULL_VALUE);
142  V latest = get(key);

150  @Override

152  checkState(txContext.isOpen(), TX_CLOSED_ERROR);
153  checkNotNull(value, ERROR_NULL_VALUE);

161  protected List<DatabaseUpdate> prepareDatabaseUpdates() {
162  List<DatabaseUpdate> updates = Lists.newLinkedList();
163  deleteSet.forEach(key -> {
164  Versioned<V> original = readCache.get(key);
165  if (original != null) {
166  updates.add(DatabaseUpdate.newBuilder()
167  .withMapName(name)
168  .withType(DatabaseUpdate.Type.REMOVE_IF_VERSION_MATCH)
169  .withKey(keyCache.getUnchecked(key))
170  .withCurrentVersion(original.version())
171  .build());
172  }
173  });
174  writeCache.forEach((key, value) -> {
175  Versioned<V> original = readCache.get(key);
176  if (original == null) {
177  updates.add(DatabaseUpdate.newBuilder()
178  .withMapName(name)
179  .withType(DatabaseUpdate.Type.PUT_IF_ABSENT)
180  .withKey(keyCache.getUnchecked(key))
181  .withValue(serializer.encode(value))
182  .build());
183  } else {
184  updates.add(DatabaseUpdate.newBuilder()
185  .withMapName(name)
186  .withType(DatabaseUpdate.Type.PUT_IF_VERSION_MATCH)
187  .withKey(keyCache.getUnchecked(key))
188  .withCurrentVersion(original.version())
189  .withValue(serializer.encode(value))
190  .build());
191  }
192  });
193  return updates;
194  }
