62  @Override
63  public void forward(ForwardingObjective fwd) {
64  FlowRuleOperations.Builder flowBuilder = FlowRuleOperations.builder();
65 
66  if (fwd.flag() != ForwardingObjective.Flag.VERSATILE) {
67  throw new UnsupportedOperationException(
68  "Only VERSATILE is supported.");
69  }
70 
71  TrafficSelector selector = fwd.selector();
72  TrafficTreatment treatment = fwd.treatment();
73  if ((fwd.treatment().deferred().size() == 0) &&
74  (fwd.treatment().immediate().size() == 0) &&
75  (fwd.treatment().tableTransition() == null) &&
76  (!fwd.treatment().clearedDeferred())) {
77  TrafficTreatment.Builder flowTreatment = DefaultTrafficTreatment.builder();
78  flowTreatment.add(Instructions.createDrop());
79  treatment = flowTreatment.build();
80  }
81 
82  FlowRule.Builder ruleBuilder = DefaultFlowRule.builder()
83  .forDevice(deviceId)
84  .withSelector(selector)
85  .withTreatment(treatment)
86  .fromApp(fwd.appId())
87  .withPriority(fwd.priority());
88 
89  if (fwd.permanent()) {
90  ruleBuilder.makePermanent();
91  } else {
92  ruleBuilder.makeTemporary(fwd.timeout());
93  }
94 
95 
96  switch (fwd.op()) {
97 
98  case ADD:
99  flowBuilder.add(ruleBuilder.build());
100  break;
101  case REMOVE:
102  flowBuilder.remove(ruleBuilder.build());
103  break;
104  default:
105  log.warn("Unknown operation {}", fwd.op());
106  }
107 
108  flowRuleService.apply(flowBuilder.build(new FlowRuleOperationsContext() {
109  @Override
110  public void onSuccess(FlowRuleOperations ops) {
111  if (fwd.context().isPresent()) {
112  fwd.context().get().onSuccess(fwd);
113  }
114  }
115 
116  @Override
117  public void onError(FlowRuleOperations ops) {
118  if (fwd.context().isPresent()) {
119  fwd.context().get().onError(fwd, ObjectiveError.FLOWINSTALLATIONFAILED);
120  }
121  }
122  }));
123 
124  }
