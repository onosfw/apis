65  public static NextHop read(ChannelBuffer cb) throws BgpParseException {
66  Ip4Address nextHop;
67  ChannelBuffer tempCb = cb.copy();
68  Validation parseFlags = Validation.parseAttributeHeader(cb);
69 
70  if (cb.readableBytes() < parseFlags.getLength()) {
71  Validation.validateLen(BgpErrorType.UPDATE_MESSAGE_ERROR, BgpErrorType.ATTRIBUTE_LENGTH_ERROR,
72  parseFlags.getLength());
73  }
74  int len = parseFlags.isShort() ? parseFlags.getLength() + Constants.TYPE_AND_LEN_AS_SHORT : parseFlags
75  .getLength() + Constants.TYPE_AND_LEN_AS_BYTE;
76  ChannelBuffer data = tempCb.readBytes(len);
77  if (parseFlags.getFirstBit() && !parseFlags.getSecondBit() && parseFlags.getThirdBit()) {
78  throw new BgpParseException(BgpErrorType.UPDATE_MESSAGE_ERROR, BgpErrorType.ATTRIBUTE_FLAGS_ERROR, data);
79  }
80 
81  InetAddress ipAddress = Validation.toInetAddress(parseFlags.getLength(), cb);
82  if (ipAddress.isMulticastAddress()) {
83  throw new BgpParseException("Multicast address is not supported");
84  }
85 
86  nextHop = Ip4Address.valueOf(ipAddress);
87  return new NextHop(nextHop);
88  }
