49 /* Generic validation
50  *
51  * [--type--][--len---]
52  *
53  * \todo This function needs removed in favor of specific validation.
54  *
55  * See: RFC 791
56  */
57 static int IPV4OptValidateGeneric(Packet *p, const IPV4Opt *o)
58 {
59  switch (o->type) {
60  /* See: RFC 4782 */
61  case IPV4_OPT_QS:
62  if (p->IPV4_OPTS[p->IPV4_OPTS_CNT].len < IPV4_OPT_QS_MIN) {
63  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_INVALID_LEN);
64  return -1;
65  }
66  break;
67  /* See: RFC 1108 */
68  case IPV4_OPT_SEC:
69  if (p->IPV4_OPTS[p->IPV4_OPTS_CNT].len != IPV4_OPT_SEC_LEN) {
70  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_INVALID_LEN);
71  return -1;
72  }
73  break;
74  case IPV4_OPT_SID:
75  if (p->IPV4_OPTS[p->IPV4_OPTS_CNT].len != IPV4_OPT_SID_LEN) {
76  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_INVALID_LEN);
77  return -1;
78  }
79  break;
80  /* See: RFC 2113 */
81  case IPV4_OPT_RTRALT:
82  if (p->IPV4_OPTS[p->IPV4_OPTS_CNT].len != IPV4_OPT_RTRALT_LEN) {
83  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_INVALID_LEN);
84  return -1;
85  }
86  break;
87  default:
88  /* Should never get here unless there is a coding error */
89  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_UNKNOWN);
90  return -1;
91  }
92 
93  return 0;
94 }

96 /* Validate route type options
97  *
98  * [--type--][--len---][--ptr---][address1]...[addressN]
99  *
100  * See: RFC 791
101  */
102 static int IPV4OptValidateRoute(Packet *p, const IPV4Opt *o)
103 {
104  uint8_t ptr;
105 
106  /* Check length */
107  if (unlikely(o->len < IPV4_OPT_ROUTE_MIN)) {
108  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_INVALID_LEN);
109  return -1;
110  }
111 
112  /* Data is required */
113  if (unlikely(o->data == NULL)) {
114  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_MALFORMED);
115  return -1;
116  }
117  ptr = *o->data;
118 
119  /* Address pointer is 1 based and points at least after type+len+ptr,
120  * must be a incremented by 4 bytes (address size) and cannot extend
121  * past option length.
122  */
123  if (unlikely((ptr < 4) || (ptr % 4) || (ptr > o->len + 1))) {
124  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_MALFORMED);
125  return -1;
126  }
127 
128  return 0;
129 }

131 /* Validate timestamp type options
132  *
133  * [--type--][--len---][--ptr---][ovfl][flag][rec1----...]...[recN----...]
134  * NOTE: rec could be 4 (ts only) or 8 (ip+ts) bytes in length.
135  *
136  * See: RFC 781
137  */
138 static int IPV4OptValidateTimestamp(Packet *p, const IPV4Opt *o)
139 {
140  uint8_t ptr;
141  uint8_t flag;
142  uint8_t rec_size;
143 
144  /* Check length */
145  if (unlikely(o->len < IPV4_OPT_TS_MIN)) {
146  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_INVALID_LEN);
147  return -1;
148  }
149 
150  /* Data is required */
151  if (unlikely(o->data == NULL)) {
152  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_MALFORMED);
153  return -1;
154  }
155  ptr = *o->data;
156 
157  /* We need the flag to determine what is in the option payload */
158  if (unlikely(ptr < 5)) {
159  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_MALFORMED);
160  return -1;
161  }
162  flag = *(o->data + 3) & 0x00ff;
163 
164  /* A flag of 1|3 means we have both the ip+ts in each record */
165  rec_size = ((flag == 1) || (flag == 3)) ? 8 : 4;
166 
167  /* Address pointer is 1 based and points at least after
168  * type+len+ptr+ovfl+flag, must be incremented by by the rec_size
169  * and cannot extend past option length.
170  */
171  if (unlikely(((ptr - 5) % rec_size) || (ptr > o->len + 1))) {
172  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_MALFORMED);
173  return -1;
174  }
175 
176  return 0;
177 }

179 /* Validate CIPSO option
180  *
181  * [--type--][--len---][--doi---][tags--...]
182  *
183  * See: draft-ietf-cipso-ipsecurity-01.txt
184  * See: FIPS 188 (tags 6 & 7)
185  */
186 static int IPV4OptValidateCIPSO(Packet *p, const IPV4Opt *o)
187 {
188 // uint32_t doi;
189  uint8_t *tag;
190  uint16_t len;
191 
192  /* Check length */
193  if (unlikely(o->len < IPV4_OPT_CIPSO_MIN)) {
194  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_INVALID_LEN);
195  return -1;
196  }
197 
198  /* Data is required */
199  if (unlikely(o->data == NULL)) {
200  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_MALFORMED);
201  return -1;
202  }
203 // doi = *o->data;
204  tag = o->data + 4;
205  len = o->len - 1 - 1 - 4; /* Length of tags after header */
206 
207 
208 #if 0
209  /* Domain of Interest (DOI) of 0 is reserved and thus invalid */
210  /** \todo Aparently a DOI of zero is fine in practice - verify. */
211  if (doi == 0) {
212  ENGINE_SET_EVENT(p,IPV4_OPT_MALFORMED);
213  return -1;
214  }
215 #endif
216 
217  /* NOTE: We know len has passed min tests prior to this call */
218 
219  /* Check that tags are formatted correctly
220  * [-ttype--][--tlen--][-tagdata-...]
221  */
222  while (len) {
223  uint8_t ttype;
224  uint8_t tlen;
225 
226  /* Tag header must fit within option length */
227  if (unlikely(len < 2)) {
228  //printf("CIPSO tag header too large %" PRIu16 " < 2\n", len);
229  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_MALFORMED);
230  return -1;
231  }
232 
233  /* Tag header is type+len */
234  ttype = *(tag++);
235  tlen = *(tag++);
236 
237  /* Tag length must fit within the option length */
238  if (unlikely(tlen > len)) {
239  //printf("CIPSO tag len too large %" PRIu8 " > %" PRIu16 "\n", tlen, len);
240  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_MALFORMED);
241  return -1;
242  }
243 
244  switch(ttype) {
245  case 1:
246  case 2:
247  case 5:
248  case 6:
249  case 7:
250  /* Tag is at least 4 and at most the remainder of option len */
251  if (unlikely((tlen < 4) || (tlen > len))) {
252  //printf("CIPSO tag %" PRIu8 " bad tlen=%" PRIu8 " len=%" PRIu8 "\n", ttype, tlen, len);
253  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_MALFORMED);
254  return -1;
255  }
256 
257  /* The alignment octet is always 0 except tag
258  * type 7, which has no such field.
259  */
260  if (unlikely((ttype != 7) && (*tag != 0))) {
261  //printf("CIPSO tag %" PRIu8 " ao=%" PRIu8 "\n", ttype, tlen);
262  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_MALFORMED);
263  return -1;
264  }
265 
266  /* Skip the rest of the tag payload */
267  tag += tlen - 2;
268  len -= tlen;
269 
270  continue;
271  case 0:
272  /* Tag type 0 is reserved and thus invalid */
273  /** \todo Wireshark marks this a padding, but spec says reserved. */
274  ENGINE_SET_INVALID_EVENT(p,IPV4_OPT_MALFORMED);
275  return -1;
276  default:
277  //printf("CIPSO tag %" PRIu8 " unknown tag\n", ttype);
278  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_MALFORMED);
279  /** \todo May not want to return error here on unknown tag type (at least not for 3|4) */
280  return -1;
281  }
282  }
283 
284  return 0;
285 }

287 /**
288  * Decode/Validate IPv4 Options.
289  */
290 static int DecodeIPV4Options(Packet *p, uint8_t *pkt, uint16_t len)
291 {
292  uint16_t plen = len;
293 
294  p->IPV4_OPTS_CNT = 0;
295 
296 #ifdef DEBUG
297  if (SCLogDebugEnabled()) {
298  uint16_t i;
299  char buf[256] = "";
300  int offset = 0;
301 
302  for (i = 0; i < len; i++) {
303  offset += snprintf(buf + offset, (sizeof(buf) - offset), "%02" PRIx8 " ", pkt[i]);
304  }
305  SCLogDebug("IPV4OPTS: { %s}", buf);
306  }
307 #endif
308 
309  /* Options length must be padded to 8byte boundary */
310  if (plen % 8) {
311  ENGINE_SET_EVENT(p,IPV4_OPT_PAD_REQUIRED);
312  /* Warn - we can keep going */
313  }
314 
315  while (plen)
316  {
317  /* single byte options */
318  if (*pkt == IPV4_OPT_EOL) {
319  /** \todo What if more data exist after EOL (possible covert channel or data leakage)? */
320  SCLogDebug("IPV4OPT %" PRIu16 " len 1 @ %" PRIu16 "/%" PRIu16 "",
321  *pkt, (len - plen), (len - 1));
322  break;
323  } else if (*pkt == IPV4_OPT_NOP) {
324  SCLogDebug("IPV4OPT %" PRIu16 " len 1 @ %" PRIu16 "/%" PRIu16 "",
325  *pkt, (len - plen), (len - 1));
326  pkt++;
327  plen--;
328 
329  /* multibyte options */
330  } else {
331  if (unlikely(plen < 2)) {
332  /** \todo What if padding is non-zero (possible covert channel or data leakage)? */
333  /** \todo Spec seems to indicate EOL required if there is padding */
334  ENGINE_SET_EVENT(p,IPV4_OPT_EOL_REQUIRED);
335  break;
336  }
337 
338  /* Option length is too big for packet */
339  if (unlikely(*(pkt+1) > plen)) {
340  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_INVALID_LEN);
341  return -1;
342  }
343 
344  p->IPV4_OPTS[p->IPV4_OPTS_CNT].type = *pkt;
345  p->IPV4_OPTS[p->IPV4_OPTS_CNT].len = *(pkt+1);
346  if (plen > 2)
347  p->IPV4_OPTS[p->IPV4_OPTS_CNT].data = (pkt+2);
348  else
349  p->IPV4_OPTS[p->IPV4_OPTS_CNT].data = NULL;
350 
351  SCLogDebug("IPV4OPT %" PRIu16 " len %" PRIu16 " @ %" PRIu16 "/%" PRIu16 "",
352  p->IPV4_OPTS[p->IPV4_OPTS_CNT].type, p->IPV4_OPTS[p->IPV4_OPTS_CNT].len,
353  (len - plen), (len - 1));
354 
355  /* we already know that the total options len is valid,
356  * so here the len of the specific option must be bad.
357  * Also check for invalid lengths 0 and 1. */
358  if (unlikely(p->IPV4_OPTS[p->IPV4_OPTS_CNT].len > plen ||
359  p->IPV4_OPTS[p->IPV4_OPTS_CNT].len < 2)) {
360  ENGINE_SET_INVALID_EVENT(p, IPV4_OPT_INVALID_LEN);
361  return -1;
362  }
363 
364  /* we are parsing the most commonly used opts to prevent
365  * us from having to walk the opts list for these all the
366  * time. */
367  /** \todo Figure out which IP options are more common and list them first */
368  switch (p->IPV4_OPTS[p->IPV4_OPTS_CNT].type) {
369  case IPV4_OPT_TS:
370  if (p->ip4vars.o_ts != NULL) {
371  ENGINE_SET_EVENT(p,IPV4_OPT_DUPLICATE);
372  /* Warn - we can keep going */
373  break;
374  } else if (IPV4OptValidateTimestamp(p,&p->IPV4_OPTS[p->IPV4_OPTS_CNT])) {
375  return -1;
376  }
377  p->ip4vars.o_ts = &p->IPV4_OPTS[p->IPV4_OPTS_CNT];
378  break;
379  case IPV4_OPT_RR:
380  if (p->ip4vars.o_rr != NULL) {
381  ENGINE_SET_EVENT(p,IPV4_OPT_DUPLICATE);
382  /* Warn - we can keep going */
383  break;
384  } else if (IPV4OptValidateRoute(p,&p->IPV4_OPTS[p->IPV4_OPTS_CNT]) != 0) {
385  return -1;
386  }
387  p->ip4vars.o_rr = &p->IPV4_OPTS[p->IPV4_OPTS_CNT];
388  break;
389  case IPV4_OPT_QS:
390  if (p->ip4vars.o_qs != NULL) {
391  ENGINE_SET_EVENT(p,IPV4_OPT_DUPLICATE);
392  /* Warn - we can keep going */
393  break;
394  } else if (IPV4OptValidateGeneric(p, &p->IPV4_OPTS[p->IPV4_OPTS_CNT])) {
395  return -1;
396  }
397  p->ip4vars.o_qs = &p->IPV4_OPTS[p->IPV4_OPTS_CNT];
398  break;
399  case IPV4_OPT_SEC:
400  if (p->ip4vars.o_sec != NULL) {
401  ENGINE_SET_EVENT(p,IPV4_OPT_DUPLICATE);
402  /* Warn - we can keep going */
403  break;
404  } else if (IPV4OptValidateGeneric(p, &p->IPV4_OPTS[p->IPV4_OPTS_CNT])) {
405  return -1;
406  }
407  p->ip4vars.o_sec = &p->IPV4_OPTS[p->IPV4_OPTS_CNT];
408  break;
409  case IPV4_OPT_LSRR:
410  if (p->ip4vars.o_lsrr != NULL) {
411  ENGINE_SET_EVENT(p,IPV4_OPT_DUPLICATE);
412  /* Warn - we can keep going */
413  break;
414  } else if (IPV4OptValidateRoute(p,&p->IPV4_OPTS[p->IPV4_OPTS_CNT]) != 0) {
415  return -1;
416  }
417  p->ip4vars.o_lsrr = &p->IPV4_OPTS[p->IPV4_OPTS_CNT];
418  break;
419  case IPV4_OPT_CIPSO:
420  if (p->ip4vars.o_cipso != NULL) {
421  ENGINE_SET_EVENT(p,IPV4_OPT_DUPLICATE);
422  /* Warn - we can keep going */
423  break;
424  } else if (IPV4OptValidateCIPSO(p,&p->IPV4_OPTS[p->IPV4_OPTS_CNT]) != 0) {
425  return -1;
426  }
427  p->ip4vars.o_cipso = &p->IPV4_OPTS[p->IPV4_OPTS_CNT];
428  break;
429  case IPV4_OPT_SID:
430  if (p->ip4vars.o_sid != NULL) {
431  ENGINE_SET_EVENT(p,IPV4_OPT_DUPLICATE);
432  /* Warn - we can keep going */
433  break;
434  } else if (IPV4OptValidateGeneric(p, &p->IPV4_OPTS[p->IPV4_OPTS_CNT])) {
435  return -1;
436  }
437  p->ip4vars.o_sid = &p->IPV4_OPTS[p->IPV4_OPTS_CNT];
438  break;
439  case IPV4_OPT_SSRR:
440  if (p->ip4vars.o_ssrr != NULL) {
441  ENGINE_SET_EVENT(p,IPV4_OPT_DUPLICATE);
442  /* Warn - we can keep going */
443  break;
444  } else if (IPV4OptValidateRoute(p,&p->IPV4_OPTS[p->IPV4_OPTS_CNT]) != 0) {
445  return -1;
446  }
447  p->ip4vars.o_ssrr = &p->IPV4_OPTS[p->IPV4_OPTS_CNT];
448  break;
449  case IPV4_OPT_RTRALT:
450  if (p->ip4vars.o_rtralt != NULL) {
451  ENGINE_SET_EVENT(p,IPV4_OPT_DUPLICATE);
452  /* Warn - we can keep going */
453  break;
454  } else if (IPV4OptValidateGeneric(p, &p->IPV4_OPTS[p->IPV4_OPTS_CNT])) {
455  return -1;
456  }
457  p->ip4vars.o_rtralt = &p->IPV4_OPTS[p->IPV4_OPTS_CNT];
458  break;
459  default:
460  SCLogDebug("IPV4OPT <unknown> (%" PRIu8 ") len %" PRIu8 "",
461  p->IPV4_OPTS[p->IPV4_OPTS_CNT].type,
462  p->IPV4_OPTS[p->IPV4_OPTS_CNT].len);
463  ENGINE_SET_EVENT(p,IPV4_OPT_INVALID);
464  /* Warn - we can keep going */
465  break;
466  }
467 
468  pkt += p->IPV4_OPTS[p->IPV4_OPTS_CNT].len;
469  plen -= (p->IPV4_OPTS[p->IPV4_OPTS_CNT].len);
470  p->IPV4_OPTS_CNT++;
471  }
472  }
473 
474  return 0;
475 }

477 static int DecodeIPV4Packet(Packet *p, uint8_t *pkt, uint16_t len)
478 {
479  if (unlikely(len < IPV4_HEADER_LEN)) {
480  ENGINE_SET_INVALID_EVENT(p, IPV4_PKT_TOO_SMALL);
481  return -1;
482  }
483 
484  if (unlikely(IP_GET_RAW_VER(pkt) != 4)) {
485  SCLogDebug("wrong ip version %" PRIu8 "",IP_GET_RAW_VER(pkt));
486  ENGINE_SET_INVALID_EVENT(p, IPV4_WRONG_IP_VER);
487  return -1;
488  }
489 
490  p->ip4h = (IPV4Hdr *)pkt;
491 
492  if (unlikely(IPV4_GET_HLEN(p) < IPV4_HEADER_LEN)) {
493  ENGINE_SET_INVALID_EVENT(p, IPV4_HLEN_TOO_SMALL);
494  return -1;
495  }
496 
497  if (unlikely(IPV4_GET_IPLEN(p) < IPV4_GET_HLEN(p))) {
498  ENGINE_SET_INVALID_EVENT(p, IPV4_IPLEN_SMALLER_THAN_HLEN);
499  return -1;
500  }
501 
502  if (unlikely(len < IPV4_GET_IPLEN(p))) {
503  ENGINE_SET_INVALID_EVENT(p, IPV4_TRUNC_PKT);
504  return -1;
505  }
506 
507  /* set the address struct */
508  SET_IPV4_SRC_ADDR(p,&p->src);
509  SET_IPV4_DST_ADDR(p,&p->dst);
510 
511  /* save the options len */
512  uint8_t ip_opt_len = IPV4_GET_HLEN(p) - IPV4_HEADER_LEN;
513  if (ip_opt_len > 0) {
514  DecodeIPV4Options(p, pkt + IPV4_HEADER_LEN, ip_opt_len);
515  }
516 
517  return 0;
518 }

520 int DecodeIPV4(ThreadVars *tv, DecodeThreadVars *dtv, Packet *p, uint8_t *pkt, uint16_t len, PacketQueue *pq)
521 {
522  StatsIncr(tv, dtv->counter_ipv4);
523 
524  SCLogDebug("pkt %p len %"PRIu16"", pkt, len);
525 
526  /* do the actual decoding */
527  if (unlikely(DecodeIPV4Packet (p, pkt, len) < 0)) {
528  SCLogDebug("decoding IPv4 packet failed");
529  p->ip4h = NULL;
530  return TM_ECODE_FAILED;
531  }
532  p->proto = IPV4_GET_IPPROTO(p);
533 
534  /* If a fragment, pass off for re-assembly. */
535  if (unlikely(IPV4_GET_IPOFFSET(p) > 0 || IPV4_GET_MF(p) == 1)) {
536  Packet *rp = Defrag(tv, dtv, p, pq);
537  if (rp != NULL) {
538  PacketEnqueue(pq, rp);
539  }
540  p->flags |= PKT_IS_FRAGMENT;
541  return TM_ECODE_OK;
542  }
543 
544  /* do hdr test, process hdr rules */
545 
546 #ifdef DEBUG
547  if (SCLogDebugEnabled()) { /* only convert the addresses if debug is really enabled */
548  /* debug print */
549  char s[16], d[16];
550  PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p), s, sizeof(s));
551  PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p), d, sizeof(d));
552  SCLogDebug("IPV4 %s->%s PROTO: %" PRIu32 " OFFSET: %" PRIu32 " RF: %" PRIu32 " DF: %" PRIu32 " MF: %" PRIu32 " ID: %" PRIu32 "", s,d,
553  IPV4_GET_IPPROTO(p), IPV4_GET_IPOFFSET(p), IPV4_GET_RF(p),
554  IPV4_GET_DF(p), IPV4_GET_MF(p), IPV4_GET_IPID(p));
555  }
556 #endif /* DEBUG */
557 
558  /* check what next decoder to invoke */
559  switch (IPV4_GET_IPPROTO(p)) {
560  case IPPROTO_TCP:
561  DecodeTCP(tv, dtv, p, pkt + IPV4_GET_HLEN(p),
562  IPV4_GET_IPLEN(p) - IPV4_GET_HLEN(p), pq);
563  break;
564  case IPPROTO_UDP:
565  DecodeUDP(tv, dtv, p, pkt + IPV4_GET_HLEN(p),
566  IPV4_GET_IPLEN(p) - IPV4_GET_HLEN(p), pq);
567  break;
568  case IPPROTO_ICMP:
569  DecodeICMPV4(tv, dtv, p, pkt + IPV4_GET_HLEN(p),
570  IPV4_GET_IPLEN(p) - IPV4_GET_HLEN(p), pq);
571  break;
572  case IPPROTO_GRE:
573  DecodeGRE(tv, dtv, p, pkt + IPV4_GET_HLEN(p),
574  IPV4_GET_IPLEN(p) - IPV4_GET_HLEN(p), pq);
575  break;
576  case IPPROTO_SCTP:
577  DecodeSCTP(tv, dtv, p, pkt + IPV4_GET_HLEN(p),
578  IPV4_GET_IPLEN(p) - IPV4_GET_HLEN(p), pq);
579  break;
580  case IPPROTO_IPV6:
581  {
582  if (pq != NULL) {
583  /* spawn off tunnel packet */
