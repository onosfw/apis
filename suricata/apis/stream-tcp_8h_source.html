146  /* If we are on IPS mode, and got a drop action triggered from

156 /**
157  * \brief Function to flip the direction When we missed the SYN packet,
158  * SYN/ACK is considered as sent by server, but our engine flagged the
159  * packet as from client for the host whose packet is received first in
160  * the session.
161  *
162  * \param ssn TcpSession to whom this packet belongs
163  * \param p Packet whose flag has to be changed
164  */
165 static inline void StreamTcpPacketSwitchDir(TcpSession *ssn, Packet *p)
166 {
167  SCLogDebug("ssn %p: switching pkt direction", ssn);
168 
169  if (PKT_IS_TOSERVER(p)) {
170  p->flowflags &= ~FLOW_PKT_TOSERVER;
171  p->flowflags |= FLOW_PKT_TOCLIENT;
172 
173  if (p->flowflags & FLOW_PKT_TOSERVER_FIRST) {
174  p->flowflags &= ~FLOW_PKT_TOSERVER_FIRST;
175  p->flowflags |= FLOW_PKT_TOCLIENT_FIRST;
176  }
177  } else {
178  p->flowflags &= ~FLOW_PKT_TOCLIENT;
179  p->flowflags |= FLOW_PKT_TOSERVER;
180 
181  if (p->flowflags & FLOW_PKT_TOCLIENT_FIRST) {
182  p->flowflags &= ~FLOW_PKT_TOCLIENT_FIRST;
183  p->flowflags |= FLOW_PKT_TOSERVER_FIRST;
184  }
185  }
186 }

198 static inline int StreamNeedsReassembly(TcpSession *ssn, int direction)
199 {
200  /* server tcp state */
201  if (direction) {
202  if (ssn->server.seg_list != NULL &&
203  (!(ssn->server.seg_list_tail->flags & SEGMENTTCP_FLAG_RAW_PROCESSED) ||
204  !(ssn->server.seg_list_tail->flags & SEGMENTTCP_FLAG_APPLAYER_PROCESSED)) ) {
205  return STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_REASSEMBLY;
206  } else if (ssn->toclient_smsg_head != NULL) {
207  return STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_ONLY_DETECTION;
208  } else {
209  return STREAM_HAS_UNPROCESSED_SEGMENTS_NONE;
210  }
211  } else {
212  if (ssn->client.seg_list != NULL &&
213  (!(ssn->client.seg_list_tail->flags & SEGMENTTCP_FLAG_RAW_PROCESSED) ||
214  !(ssn->client.seg_list_tail->flags & SEGMENTTCP_FLAG_APPLAYER_PROCESSED)) ) {
215  return STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_REASSEMBLY;
216  } else if (ssn->toserver_smsg_head != NULL) {
217  return STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_ONLY_DETECTION;
218  } else {
219  return STREAM_HAS_UNPROCESSED_SEGMENTS_NONE;
220  }
221  }
222 }
