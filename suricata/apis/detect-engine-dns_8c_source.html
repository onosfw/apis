100  * \param det_ctx Detection engine thread context
101  * \param s Signature to inspect

110 int DetectEngineInspectGenericList(ThreadVars *tv,
111  const DetectEngineCtx *de_ctx,
112  DetectEngineThreadCtx *det_ctx,
113  const Signature *s, Flow *f, const uint8_t flags,
114  void *alstate, void *txv, uint64_t tx_id, const int list)
115 {
116  KEYWORD_PROFILING_SET_LIST(det_ctx, list);
117 
118  SigMatchData *smd = s->sm_arrays[list];
119  SCLogDebug("running match functions, sm %p", smd);
120  if (smd != NULL) {
121  while (1) {
122  int match = 0;
123  KEYWORD_PROFILING_START;
124  match = sigmatch_table[smd->type].
125  AppLayerTxMatch(tv, det_ctx, f, flags, alstate, txv, s, smd->ctx);
126  KEYWORD_PROFILING_END(det_ctx, smd->type, (match == 1));
127 
128  if (match == 0)
129  return DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
130  if (match == 2) {
131  return DETECT_ENGINE_INSPECT_SIG_CANT_MATCH;
132  }
133 
134  if (smd->is_last)
135  break;
136  smd++;
137  }
138  }
139 
140  return DETECT_ENGINE_INSPECT_SIG_MATCH;
141 }
