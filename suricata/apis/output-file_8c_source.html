58 int OutputRegisterFileLogger(const char *name, FileLogger LogFunc, OutputCtx *output_ctx)
59 {
60  int module_id = TmModuleGetIdByName(name);
61  if (module_id < 0)
62  return -1;
63 
64  OutputFileLogger *op = SCMalloc(sizeof(*op));
65  if (op == NULL)
66  return -1;
67  memset(op, 0x00, sizeof(*op));
68 
69  op->LogFunc = LogFunc;
70  op->output_ctx = output_ctx;
71  op->name = name;
72  op->module_id = (TmmId) module_id;
73 
74  if (list == NULL)
75  list = op;
76  else {
77  OutputFileLogger *t = list;
78  while (t->next)
79  t = t->next;
80  t->next = op;
81  }
82 
83  SCLogDebug("OutputRegisterTxLogger happy");
84  return 0;
85 }

87 static TmEcode OutputFileLog(ThreadVars *tv, Packet *p, void *thread_data, PacketQueue *pq, PacketQueue *postpq)
88 {
89  BUG_ON(thread_data == NULL);
90  BUG_ON(list == NULL);
91 
92  OutputLoggerThreadData *op_thread_data = (OutputLoggerThreadData *)thread_data;
93  OutputFileLogger *logger = list;
94  OutputLoggerThreadStore *store = op_thread_data->store;
95 
96  BUG_ON(logger == NULL && store != NULL);
97  BUG_ON(logger != NULL && store == NULL);
98  BUG_ON(logger == NULL && store == NULL);
99 
100  uint8_t flags = 0;
101  Flow * const f = p->flow;
102 
103  /* no flow, no files */
104  if (f == NULL) {
105  SCReturnInt(TM_ECODE_OK);
106  }
107 
108  if (p->flowflags & FLOW_PKT_TOCLIENT)
109  flags |= STREAM_TOCLIENT;
110  else
111  flags |= STREAM_TOSERVER;
112 
113  int file_close = (p->flags & PKT_PSEUDO_STREAM_END) ? 1 : 0;
114  int file_trunc = 0;
115 
116  FLOWLOCK_WRLOCK(f); // < need write lock for FilePrune below
117  file_trunc = StreamTcpReassembleDepthReached(p);
118 
119  FileContainer *ffc = AppLayerParserGetFiles(p->proto, f->alproto,
120  f->alstate, flags);
121  SCLogDebug("ffc %p", ffc);
122  if (ffc != NULL) {
123  File *ff;
124  for (ff = ffc->head; ff != NULL; ff = ff->next) {
125  if (ff->flags & FILE_LOGGED)
126  continue;
127 
128  SCLogDebug("ff %p", ff);
129 
130  if (file_trunc && ff->state < FILE_STATE_CLOSED)
131  ff->state = FILE_STATE_TRUNCATED;
132 
133  if (file_close && ff->state < FILE_STATE_CLOSED)
134  ff->state = FILE_STATE_TRUNCATED;
135 
136  if (ff->state == FILE_STATE_CLOSED ||
137  ff->state == FILE_STATE_TRUNCATED ||
138  ff->state == FILE_STATE_ERROR)
139  {
140  int file_logged = 0;
141 
142  if (FileForceMagic() && ff->magic == NULL) {
143  FilemagicGlobalLookup(ff);
144  }
145 
146  logger = list;
147  store = op_thread_data->store;
148  while (logger && store) {
149  BUG_ON(logger->LogFunc == NULL);
150 
151  SCLogDebug("logger %p", logger);
152  PACKET_PROFILING_TMM_START(p, logger->module_id);
153  logger->LogFunc(tv, store->thread_data, (const Packet *)p, (const File *)ff);
154  PACKET_PROFILING_TMM_END(p, logger->module_id);
155  file_logged = 1;
156 
157  logger = logger->next;
158  store = store->next;
159 
160  BUG_ON(logger == NULL && store != NULL);
161  BUG_ON(logger != NULL && store == NULL);
162  }
163 
164  if (file_logged) {
165  ff->flags |= FILE_LOGGED;
166  }
167  }
168  }
169 
170  FilePrune(ffc);
171  }
172 
173  FLOWLOCK_UNLOCK(f);
174  return TM_ECODE_OK;
175 }

177 /** \brief thread init for the tx logger
178  * This will run the thread init functions for the individual registered
179  * loggers */
180 static TmEcode OutputFileLogThreadInit(ThreadVars *tv, void *initdata, void **data)
181 {
182  OutputLoggerThreadData *td = SCMalloc(sizeof(*td));
183  if (td == NULL)
184  return TM_ECODE_FAILED;
185  memset(td, 0x00, sizeof(*td));
186 
187  *data = (void *)td;
188 
189  SCLogDebug("OutputFileLogThreadInit happy (*data %p)", *data);
190 
191  OutputFileLogger *logger = list;
192  while (logger) {
193  TmModule *tm_module = TmModuleGetByName((char *)logger->name);
194  if (tm_module == NULL) {
195  SCLogError(SC_ERR_INVALID_ARGUMENT,
196  "TmModuleGetByName for %s failed", logger->name);
197  exit(EXIT_FAILURE);
198  }
199 
200  if (tm_module->ThreadInit) {
201  void *retptr = NULL;
202  if (tm_module->ThreadInit(tv, (void *)logger->output_ctx, &retptr) == TM_ECODE_OK) {
203  OutputLoggerThreadStore *ts = SCMalloc(sizeof(*ts));
204 /* todo */ BUG_ON(ts == NULL);
205  memset(ts, 0x00, sizeof(*ts));
206 
207  /* store thread handle */
208  ts->thread_data = retptr;
209 
210  if (td->store == NULL) {
211  td->store = ts;
212  } else {
213  OutputLoggerThreadStore *tmp = td->store;
214  while (tmp->next != NULL)
215  tmp = tmp->next;
216  tmp->next = ts;
217  }
218 
219  SCLogDebug("%s is now set up", logger->name);
220  }
221  }
222 
223  logger = logger->next;
224  }
225 
226  return TM_ECODE_OK;
227 }

229 static TmEcode OutputFileLogThreadDeinit(ThreadVars *tv, void *thread_data)
230 {
231  OutputLoggerThreadData *op_thread_data = (OutputLoggerThreadData *)thread_data;
232  OutputLoggerThreadStore *store = op_thread_data->store;
233  OutputFileLogger *logger = list;
234 
235  while (logger && store) {
236  TmModule *tm_module = TmModuleGetByName((char *)logger->name);
237  if (tm_module == NULL) {
238  SCLogError(SC_ERR_INVALID_ARGUMENT,
239  "TmModuleGetByName for %s failed", logger->name);
240  exit(EXIT_FAILURE);
241  }
242 
243  if (tm_module->ThreadDeinit) {
244  tm_module->ThreadDeinit(tv, store->thread_data);
245  }
246 
247  OutputLoggerThreadStore *next_store = store->next;
248  SCFree(store);
249  store = next_store;
250  logger = logger->next;
251  }
252  return TM_ECODE_OK;
