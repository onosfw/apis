47 /**
48  * \brief Main decoding function for PPPOE Discovery packets
49  */
50 int DecodePPPOEDiscovery(ThreadVars *tv, DecodeThreadVars *dtv, Packet *p, uint8_t *pkt, uint16_t len, PacketQueue *pq)
51 {
52  StatsIncr(tv, dtv->counter_pppoe);
53 
54  if (len < PPPOE_DISCOVERY_HEADER_MIN_LEN) {
55  ENGINE_SET_INVALID_EVENT(p, PPPOE_PKT_TOO_SMALL);
56  return TM_ECODE_FAILED;
57  }
58 
59  p->pppoedh = (PPPOEDiscoveryHdr *)pkt;
60  if (p->pppoedh == NULL)
61  return TM_ECODE_FAILED;
62 
63  /* parse the PPPOE code */
64  switch (p->pppoedh->pppoe_code)
65  {
66  case PPPOE_CODE_PADI:
67  break;
68  case PPPOE_CODE_PADO:
69  break;
70  case PPPOE_CODE_PADR:
71  break;
72  case PPPOE_CODE_PADS:
73  break;
74  case PPPOE_CODE_PADT:
75  break;
76  default:
77  SCLogDebug("unknown PPPOE code: 0x%0"PRIX8"", p->pppoedh->pppoe_code);
78  ENGINE_SET_INVALID_EVENT(p, PPPOE_WRONG_CODE);
79  return TM_ECODE_OK;
80  }
81 
82  /* parse any tags we have in the packet */
83 
84  uint16_t tag_length = 0;
85  PPPOEDiscoveryTag* pppoedt = (PPPOEDiscoveryTag*) (p->pppoedh + PPPOE_DISCOVERY_HEADER_MIN_LEN);
86 
87  uint16_t pppoe_length = ntohs(p->pppoedh->pppoe_length);
88  uint16_t packet_length = len - PPPOE_DISCOVERY_HEADER_MIN_LEN ;
89 
90  SCLogDebug("pppoe_length %"PRIu16", packet_length %"PRIu16"",
91  pppoe_length, packet_length);
92 
93  if (pppoe_length > packet_length) {
94  SCLogDebug("malformed PPPOE tags");
95  ENGINE_SET_INVALID_EVENT(p, PPPOE_MALFORMED_TAGS);
96  return TM_ECODE_OK;
97  }
98 
99  while (pppoedt < (PPPOEDiscoveryTag*) (pkt + (len - sizeof(PPPOEDiscoveryTag))) && pppoe_length >=4 && packet_length >=4)
100  {
101 #ifdef DEBUG
102  uint16_t tag_type = ntohs(pppoedt->pppoe_tag_type);
103 #endif
104  tag_length = ntohs(pppoedt->pppoe_tag_length);
105 
106  SCLogDebug ("PPPoE Tag type %x, length %u", tag_type, tag_length);
107 
108  if (pppoe_length >= (4 + tag_length)) {
109  pppoe_length -= (4 + tag_length);
110  } else {
111  pppoe_length = 0; // don't want an underflow
112  }
113 
114  if (packet_length >= 4 + tag_length) {
115  packet_length -= (4 + tag_length);
116  } else {
117  packet_length = 0; // don't want an underflow
118  }
119 
120  pppoedt = pppoedt + (4 + tag_length);
121  }
122 
123  return TM_ECODE_OK;
124 }

126 /**
127  * \brief Main decoding function for PPPOE Session packets
128  */
129 int DecodePPPOESession(ThreadVars *tv, DecodeThreadVars *dtv, Packet *p, uint8_t *pkt, uint16_t len, PacketQueue *pq)
130 {
131  StatsIncr(tv, dtv->counter_pppoe);
132 
133  if (len < PPPOE_SESSION_HEADER_LEN) {
134  ENGINE_SET_INVALID_EVENT(p, PPPOE_PKT_TOO_SMALL);
135  return TM_ECODE_FAILED;
136  }
137 
138  p->pppoesh = (PPPOESessionHdr *)pkt;
139  if (p->pppoesh == NULL)
140  return TM_ECODE_FAILED;
141 
142  SCLogDebug("PPPOE VERSION %" PRIu32 " TYPE %" PRIu32 " CODE %" PRIu32 " SESSIONID %" PRIu32 " LENGTH %" PRIu32 "",
143  PPPOE_SESSION_GET_VERSION(p->pppoesh), PPPOE_SESSION_GET_TYPE(p->pppoesh), p->pppoesh->pppoe_code, ntohs(p->pppoesh->session_id), ntohs(p->pppoesh->pppoe_length));
144 
145  /* can't use DecodePPP() here because we only get a single 2-byte word to indicate protocol instead of the full PPP header */
146 
147  if (ntohs(p->pppoesh->pppoe_length) > 0) {
148  /* decode contained PPP packet */
149 
150  switch (ntohs(p->pppoesh->protocol))
151  {
152  case PPP_VJ_COMP:
153  case PPP_IPX:
154  case PPP_OSI:
155  case PPP_NS:
156  case PPP_DECNET:
157  case PPP_APPLE:
158  case PPP_BRPDU:
159  case PPP_STII:
160  case PPP_VINES:
161  case PPP_HELLO:
162  case PPP_LUXCOM:
163  case PPP_SNS:
164  case PPP_MPLS_UCAST:
165  case PPP_MPLS_MCAST:
166  case PPP_IPCP:
167  case PPP_OSICP:
168  case PPP_NSCP:
169  case PPP_DECNETCP:
170  case PPP_APPLECP:
171  case PPP_IPXCP:
172  case PPP_STIICP:
173  case PPP_VINESCP:
174  case PPP_IPV6CP:
175  case PPP_MPLSCP:
176  case PPP_LCP:
177  case PPP_PAP:
178  case PPP_LQM:
179  case PPP_CHAP:
180  ENGINE_SET_EVENT(p,PPP_UNSUP_PROTO);
181  break;
182 
183  case PPP_VJ_UCOMP:
184 
185  if(len < (PPPOE_SESSION_HEADER_LEN + IPV4_HEADER_LEN)) {
186  ENGINE_SET_INVALID_EVENT(p, PPPVJU_PKT_TOO_SMALL);
187  return TM_ECODE_OK;
188  }
189 
190  if(IPV4_GET_RAW_VER((IPV4Hdr *)(pkt + PPPOE_SESSION_HEADER_LEN)) == 4) {
191  DecodeIPV4(tv, dtv, p, pkt + PPPOE_SESSION_HEADER_LEN, len - PPPOE_SESSION_HEADER_LEN, pq );
192  }
193  break;
194 
195  case PPP_IP:
196  if(len < (PPPOE_SESSION_HEADER_LEN + IPV4_HEADER_LEN)) {
197  ENGINE_SET_INVALID_EVENT(p, PPPIPV4_PKT_TOO_SMALL);
198  return TM_ECODE_OK;
199  }
200 
201  DecodeIPV4(tv, dtv, p, pkt + PPPOE_SESSION_HEADER_LEN, len - PPPOE_SESSION_HEADER_LEN, pq );
202  break;
203 
204  /* PPP IPv6 was not tested */
205  case PPP_IPV6:
206  if(len < (PPPOE_SESSION_HEADER_LEN + IPV6_HEADER_LEN)) {
207  ENGINE_SET_INVALID_EVENT(p, PPPIPV6_PKT_TOO_SMALL);
208  return TM_ECODE_OK;
209  }
210 
211  DecodeIPV6(tv, dtv, p, pkt + PPPOE_SESSION_HEADER_LEN, len - PPPOE_SESSION_HEADER_LEN, pq );
212  break;
213 
214  default:
215  SCLogDebug("unknown PPP protocol: %" PRIx32 "",ntohs(p->ppph->protocol));
216  ENGINE_SET_INVALID_EVENT(p, PPP_WRONG_TYPE);
217  return TM_ECODE_OK;
218  }
219  }
220  return TM_ECODE_OK;
221 }
