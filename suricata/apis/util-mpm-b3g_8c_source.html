269 static int B3gAddPattern(MpmCtx *mpm_ctx, uint8_t *pat, uint16_t patlen, uint16_t offset, uint16_t depth, uint32_t pid, uint32_t sid, uint8_t flags)
270 {
271  B3gCtx *ctx = (B3gCtx *)mpm_ctx->ctx;
272 
273  if (patlen == 0)
274  return 0;
275 
276  /* get a memory piece */
277  B3gPattern *p = B3gInitHashLookup(ctx, pat, patlen, flags);
278  if (p == NULL) {
279  p = B3gAllocPattern(mpm_ctx);
280  if (p == NULL)
281  goto error;
282 
283  p->len = patlen;
284  p->flags = flags;
285  p->id = pid;
286 
287  /* setup the case insensitive part of the pattern */
288  p->ci = SCMalloc(patlen);
289  if (p->ci == NULL)
290  goto error;
291 
292  mpm_ctx->memory_cnt++;
293  mpm_ctx->memory_size += patlen;
294  memcpy_tolower(p->ci, pat, patlen);
295 
296  /* setup the case sensitive part of the pattern */
297  if (p->flags & MPM_PATTERN_FLAG_NOCASE) {
298  /* nocase means no difference between cs and ci */
299  p->cs = p->ci;
300  } else {
301  if (memcmp(p->ci,pat,p->len) == 0) {
302  /* no diff between cs and ci: pat is lowercase */
303  p->cs = p->ci;
304  } else {
305  p->cs = SCMalloc(patlen);
306  if (p->cs == NULL)
307  goto error;
308 
309  mpm_ctx->memory_cnt++;
310  mpm_ctx->memory_size += patlen;
311  memcpy(p->cs, pat, patlen);
312  }
313  }
314 
315  //printf("B3gAddPattern: ci \""); prt(p->ci,p->len);
316  //printf("\" cs \""); prt(p->cs,p->len);
317  //printf("\" prefix_ci %" PRIu32 ", prefix_cs %" PRIu32 "\n", p->prefix_ci, p->prefix_cs);
318 
319  p->sids_size = 1;
320  p->sids = SCMalloc(p->sids_size * sizeof(SigIntId));
321  BUG_ON(p->sids == NULL);
322  p->sids[0] = sid;
323  mpm_ctx->memory_cnt++;
324  mpm_ctx->memory_size += sizeof(SigIntId);
325 
326  /* put in the pattern hash */
327  B3gInitHashAdd(ctx, p);
328 
329  if (mpm_ctx->pattern_cnt == 65535) {
330  printf("Max search words reached\n");
331  exit(1);
332  }
333  mpm_ctx->pattern_cnt++;
334 
335  if (mpm_ctx->maxlen < patlen) mpm_ctx->maxlen = patlen;
336  if (mpm_ctx->minlen == 0) mpm_ctx->minlen = patlen;
337  else if (mpm_ctx->minlen > patlen) mpm_ctx->minlen = patlen;
338  } else {
339  /* Multiple sids for the same pid, so keep an array of sids. */
340 
341  /* TODO figure out how we can be called multiple times for the
342  * same CTX with the same sid */
343  int found = 0;
344  uint32_t x = 0;
345  for (x = 0; x < p->sids_size; x++) {
346  if (p->sids[x] == sid) {
347  found = 1;
348  break;
349  }
350  }
351  if (!found) {
352  SigIntId *sids = SCRealloc(p->sids, (sizeof(SigIntId) * (p->sids_size + 1)));
353  BUG_ON(sids == NULL);
354  p->sids = sids;
355  p->sids[p->sids_size] = sid;
356  p->sids_size++;
357  mpm_ctx->memory_size += sizeof(SigIntId);
358  }
359  }
360 
361  return 0;
362 
363 error:
364  B3gFreePattern(mpm_ctx, p);
365  return -1;
366 }

381 static uint32_t B3gBloomHash(void *data, uint16_t datalen, uint8_t iter, uint32_t hash_size)
382 {
383  uint8_t *d = (uint8_t *)data;
384  uint16_t i;
385  uint32_t hash = (uint32_t)u8_tolower(*d);
386 
387  for (i = 1; i < datalen; i++) {
388  d++;
389  hash += (u8_tolower(*d)) ^ i;
390  }
391  hash <<= (iter+1);
392 
393  hash %= hash_size;
394  return hash;
395 }

534  } while (thi != NULL);
535  }
536 
537  return;
538 error:

587  /* populate it with the patterns in the hash */
588  uint32_t i = 0, p = 0;
589  for (i = 0; i < INIT_HASH_SIZE; i++) {
590  B3gPattern *node = ctx->init_hash[i], *nnode = NULL;
591  for ( ; node != NULL; ) {
592  nnode = node->next;
593  node->next = NULL;

596

598  node = nnode;
599  }
600  }

602  SCFree(ctx->init_hash);
603  ctx->init_hash = NULL;
604 
605  /* set 'm' to the smallest pattern size */
606  ctx->m = mpm_ctx->minlen;
607 
608  /* make sure 'm' stays in bounds
609  m can be max WORD_SIZE - 1 */
610  if (ctx->m >= B3G_WORD_SIZE) {
611  ctx->m = B3G_WORD_SIZE - 1;
612  }
613  if (ctx->m < 3) ctx->m = 3;
614 
615 
616  ctx->hash_size = b3g_hash_size;
617  B3gPrepareHash(mpm_ctx);
618  B3gBuildMatchArray(mpm_ctx);
619 
620  if (ctx->pat_1_cnt) {
621  ctx->Search = B3gSearch1;
622  if (ctx->pat_2_cnt) {
623  ctx->Search = B3gSearch12;
624  ctx->MBSearch = b3g_func;
625  }

628  ctx->Search = B3gSearch2;
629  ctx->MBSearch = b3g_func;
630  }

633  return 0;
634 error:

907 #ifdef B3G_COUNTERS
908  B3gThreadCtx *tctx = (B3gThreadCtx *)mpm_thread_ctx->ctx;
909 #endif
910  uint32_t pos = ctx->m - B3G_Q + 1, matches = 0;
911  B3G_TYPE d;
912 
913  COUNT(tctx->stat_calls++);
914  COUNT(tctx->stat_m_total+=ctx->m);
915 
916  if (buflen < ctx->m)
917  return 0;

920  if (pmq) {
921  bitarray = alloca(pmq->pattern_id_bitarray_size);
922  memset(bitarray, 0, pmq->pattern_id_bitarray_size);
923  }
924 
925  while (pos <= (uint32_t)(buflen - B3G_Q + 1)) {
926  uint16_t h = B3G_HASH(u8_tolower(buf[pos - 1]), u8_tolower(buf[pos]),u8_tolower(buf[pos + 1]));
927  d = ctx->B3G[h];
928 
929  if (d != 0) {
930  COUNT(tctx->stat_d0++);
931  uint32_t j = pos;
932  uint32_t first = pos - (ctx->m - B3G_Q + 1);
933 
934  do {
935  j = j - 1;
936  if (d >= (uint32_t)(1 << (ctx->m - 1))) {
937  if (j > first) pos = j;
938  else {
939  /* get our patterns from the hash */
940  h = B3G_HASH(u8_tolower(buf[j + ctx->m - 3]), u8_tolower(buf[j + ctx->m - 2]),u8_tolower(buf[j + ctx->m - 1]));
941

943  COUNT(tctx->stat_pminlen_calls++);
944  COUNT(tctx->stat_pminlen_total+=ctx->pminlen[h]);
945 
946  if ((buflen - j) < ctx->pminlen[h]) {
947  goto skip_loop;
948  } else {
949  COUNT(tctx->stat_bloom_calls++);
950 
951  if (BloomFilterTest(ctx->bloom[h], buf+j, ctx->pminlen[h]) == 0) {
952  COUNT(tctx->stat_bloom_hits++);
953 
954  //printf("Bloom: %p, buflen %" PRIu32 ", pos %" PRIu32 ", p_min_len %" PRIu32 "\n", ctx->bloom[h], buflen, pos, ctx->pminlen[h]);
955  goto skip_loop;
956  }
957  }
958  }
959 
960  B3gHashItem *hi = ctx->hash[h], *thi;
961  for (thi = hi; thi != NULL; thi = thi->nxt) {
962  COUNT(tctx->stat_d0_hashloop++);
963  B3gPattern *p = ctx->parray[thi->idx];
964 
965  if (p->flags & MPM_PATTERN_FLAG_NOCASE) {
966  if (buflen - j < p->len)
967  continue;
968 
969  if (memcmp_lowercase(p->ci, buf+j, p->len) == 0) {
970  COUNT(tctx->stat_loop_match++);
971 
972  matches += MpmVerifyMatch(mpm_thread_ctx, pmq, p->id, bitarray, p->sids, p->sids_size);
973  } else {
974  COUNT(tctx->stat_loop_no_match++);
975  }
976  } else {
977  if (buflen - j < p->len)
978  continue;
979 
980  if (memcmp(p->cs, buf+j, p->len) == 0) {
981  COUNT(tctx->stat_loop_match++);
982 
983  matches += MpmVerifyMatch(mpm_thread_ctx, pmq, p->id, bitarray, p->sids, p->sids_size);
984  } else {
985  COUNT(tctx->stat_loop_no_match++);
986  }
987  }
988  }
989 skip_loop:
990  //printf("output at pos %" PRIu32 ": ", j); prt(buf + (j), ctx->m); printf("\n");
991  ; // gcc doesn't like the goto label without this :-S
992  }
993  }
994 
995  if (j == 0)
996  break;
997 
998  h = B3G_HASH(u8_tolower(buf[j - 1]), u8_tolower(buf[j - 0]),u8_tolower(buf[j+1]));
999  d = (d << 1) & ctx->B3G[h];
1000  } while (d != 0);
1001  }
1002  COUNT(tctx->stat_num_shift++);
1003  COUNT(tctx->stat_total_shift += (ctx->m - B3G_Q + 1));
1004  pos = pos + ctx->m - B3G_Q + 1;
1005  }
1006  return matches;
1007 }
