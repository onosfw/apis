59 /**
60  * \brief Set the parent on stand-by until the child is ready
61  *
62  * \param pid pid of the child process to wait
63  */
64 static void WaitForChild (pid_t pid)
65 {
66  int status;
67  SCLogDebug("Daemon: Parent waiting for child to be ready...");
68  /* Wait until child signals is ready */
69  while (sigflag == 0) {
70  if (waitpid(pid, &status, WNOHANG)) {
71  /* Check if the child is still there, otherwise the parent should exit */
72  if (WIFEXITED(status) || WIFSIGNALED(status)) {
73  SCLogError(SC_ERR_DAEMON, "Child died unexpectedly");
74  exit(EXIT_FAILURE);
75  }
76  }
77  /* sigsuspend(); */
78  sleep(1);
79  }
80 }

101 /**
102  * \brief Daemonize the process
103  *
104  */
105 void Daemonize (void)
106 {
107  pid_t pid, sid;
108 
109  /* Register the signal handler */
110  signal(SIGUSR1, SignalHandlerSigusr1);
111 
112  /** \todo We should check if wie allow more than 1 instance
113  to run simultaneously. Maybe change the behaviour
114  through conf file */
115 
116  /* Creates a new process */
117  pid = fork();
118 
119  if (pid < 0) {
120  /* Fork error */
121  SCLogError(SC_ERR_DAEMON, "Error forking the process");
122  exit(EXIT_FAILURE);
123  } else if (pid == 0) {
124  /* Child continues here */
125  char *daemondir;
126 
127  umask(027);
128 
129  sid = setsid();
130  if (sid < 0) {
131  SCLogError(SC_ERR_DAEMON, "Error creating new session");
132  exit(EXIT_FAILURE);
133  }
134 
135  if (ConfGet("daemon-directory", &daemondir) == 1) {
136  if ((chdir(daemondir)) < 0) {
137  SCLogError(SC_ERR_DAEMON, "Error changing to working directory");
138  exit(EXIT_FAILURE);
139  }
140  }
141 #ifndef OS_WIN32
142  else {
143  if (chdir("/") < 0) {
144  SCLogError(SC_ERR_DAEMON, "Error changing to working directory '/'");
145  }
146  }
147 #endif
148 
149  SetupLogging();
150 
151  /* Child is ready, tell its parent */
152  TellWaitingParent(getppid());
153 
154  /* Daemon is up and running */
155  SCLogDebug("Daemon is running");
156  return;
157  }
158  /* Parent continues here, waiting for child to be ready */
159  SCLogDebug("Parent is waiting for child to be ready");
160  WaitForChild(pid);
161 
162  /* Parent exits */
163  SCLogDebug("Child is ready, parent exiting");
164  exit(EXIT_SUCCESS);
165 
166 }
167 
168 #endif /* ifndef OS_WIN32 */
