177  }

208 /* calculate the hash key for this packet
209  *
210  * we're using:
211  * hash_rand -- set at init time
212  * source port
213  * destination port
214  * source address
215  * destination address
216  * recursion level -- for tunnels, make sure different tunnel layers can
217  * never get mixed up.
218  *
219  * For ICMP we only consider UNREACHABLE errors atm.
220  */
221 static inline uint32_t FlowGetKey(const Packet *p)
222 {
223  uint32_t key;
224 
225  if (p->ip4h != NULL) {
226  if (p->tcph != NULL || p->udph != NULL) {
227  FlowHashKey4 fhk;
228  if (p->src.addr_data32[0] > p->dst.addr_data32[0]) {
229  fhk.src = p->src.addr_data32[0];
230  fhk.dst = p->dst.addr_data32[0];
231  } else {
232  fhk.src = p->dst.addr_data32[0];
233  fhk.dst = p->src.addr_data32[0];
234  }
235  if (p->sp > p->dp) {
236  fhk.sp = p->sp;
237  fhk.dp = p->dp;
238  } else {
239  fhk.sp = p->dp;
240  fhk.dp = p->sp;
241  }
242  fhk.proto = (uint16_t)p->proto;
243  fhk.recur = (uint16_t)p->recursion_level;
244  fhk.vlan_id[0] = p->vlan_id[0];
245  fhk.vlan_id[1] = p->vlan_id[1];
246 
247  uint32_t hash = hashword(fhk.u32, 5, flow_config.hash_rand);
248  key = hash % flow_config.hash_size;
249 
250  } else if (ICMPV4_DEST_UNREACH_IS_VALID(p)) {
251  uint32_t psrc = IPV4_GET_RAW_IPSRC_U32(ICMPV4_GET_EMB_IPV4(p));
252  uint32_t pdst = IPV4_GET_RAW_IPDST_U32(ICMPV4_GET_EMB_IPV4(p));
253  FlowHashKey4 fhk;
254  if (psrc > pdst) {
255  fhk.src = psrc;
256  fhk.dst = pdst;
257  } else {
258  fhk.src = pdst;
259  fhk.dst = psrc;
260  }
261  if (p->icmpv4vars.emb_sport > p->icmpv4vars.emb_dport) {
262  fhk.sp = p->icmpv4vars.emb_sport;
263  fhk.dp = p->icmpv4vars.emb_dport;
264  } else {
265  fhk.sp = p->icmpv4vars.emb_dport;
266  fhk.dp = p->icmpv4vars.emb_sport;
267  }
268  fhk.proto = (uint16_t)ICMPV4_GET_EMB_PROTO(p);
269  fhk.recur = (uint16_t)p->recursion_level;
270  fhk.vlan_id[0] = p->vlan_id[0];
271  fhk.vlan_id[1] = p->vlan_id[1];
272 
273  uint32_t hash = hashword(fhk.u32, 5, flow_config.hash_rand);
274  key = hash % flow_config.hash_size;
275 
276  } else {
277  FlowHashKey4 fhk;
278  if (p->src.addr_data32[0] > p->dst.addr_data32[0]) {
279  fhk.src = p->src.addr_data32[0];
280  fhk.dst = p->dst.addr_data32[0];
281  } else {
282  fhk.src = p->dst.addr_data32[0];
283  fhk.dst = p->src.addr_data32[0];
284  }
285  fhk.sp = 0xfeed;
286  fhk.dp = 0xbeef;
287  fhk.proto = (uint16_t)p->proto;
288  fhk.recur = (uint16_t)p->recursion_level;
289  fhk.vlan_id[0] = p->vlan_id[0];
290  fhk.vlan_id[1] = p->vlan_id[1];
291 
292  uint32_t hash = hashword(fhk.u32, 5, flow_config.hash_rand);
293  key = hash % flow_config.hash_size;
294  }
295  } else if (p->ip6h != NULL) {
296  FlowHashKey6 fhk;
297  if (FlowHashRawAddressIPv6GtU32(p->src.addr_data32, p->dst.addr_data32)) {
298  fhk.src[0] = p->src.addr_data32[0];
299  fhk.src[1] = p->src.addr_data32[1];
300  fhk.src[2] = p->src.addr_data32[2];
301  fhk.src[3] = p->src.addr_data32[3];
302  fhk.dst[0] = p->dst.addr_data32[0];
303  fhk.dst[1] = p->dst.addr_data32[1];
304  fhk.dst[2] = p->dst.addr_data32[2];
305  fhk.dst[3] = p->dst.addr_data32[3];
306  } else {
307  fhk.src[0] = p->dst.addr_data32[0];
308  fhk.src[1] = p->dst.addr_data32[1];
309  fhk.src[2] = p->dst.addr_data32[2];
310  fhk.src[3] = p->dst.addr_data32[3];
311  fhk.dst[0] = p->src.addr_data32[0];
312  fhk.dst[1] = p->src.addr_data32[1];
313  fhk.dst[2] = p->src.addr_data32[2];
314  fhk.dst[3] = p->src.addr_data32[3];
315  }
316  if (p->sp > p->dp) {
317  fhk.sp = p->sp;
318  fhk.dp = p->dp;
319  } else {
320  fhk.sp = p->dp;
321  fhk.dp = p->sp;
322  }
323  fhk.proto = (uint16_t)p->proto;
324  fhk.recur = (uint16_t)p->recursion_level;
325  fhk.vlan_id[0] = p->vlan_id[0];
326  fhk.vlan_id[1] = p->vlan_id[1];
327 
328  uint32_t hash = hashword(fhk.u32, 11, flow_config.hash_rand);
329  key = hash % flow_config.hash_size;
330  } else
331  key = 0;
332 
333  return key;
334 }

350 /**
351  * \brief See if a ICMP packet belongs to a flow by comparing the embedded
352  * packet in the ICMP error packet to the flow.
353  *
354  * \param f flow
355  * \param p ICMP packet
356  *
357  * \retval 1 match
358  * \retval 0 no match
359  */
360 static inline int FlowCompareICMPv4(Flow *f, const Packet *p)
361 {
362  if (ICMPV4_DEST_UNREACH_IS_VALID(p)) {
363  /* first check the direction of the flow, in other words, the client ->
364  * server direction as it's most likely the ICMP error will be a
365  * response to the clients traffic */
366  if ((f->src.addr_data32[0] == IPV4_GET_RAW_IPSRC_U32( ICMPV4_GET_EMB_IPV4(p) )) &&
367  (f->dst.addr_data32[0] == IPV4_GET_RAW_IPDST_U32( ICMPV4_GET_EMB_IPV4(p) )) &&
368  f->sp == p->icmpv4vars.emb_sport &&
369  f->dp == p->icmpv4vars.emb_dport &&
370  f->proto == ICMPV4_GET_EMB_PROTO(p) &&
371  f->recursion_level == p->recursion_level &&
372  f->vlan_id[0] == p->vlan_id[0] &&
373  f->vlan_id[1] == p->vlan_id[1])
374  {
375  return 1;
376 
377  /* check the less likely case where the ICMP error was a response to
378  * a packet from the server. */
379  } else if ((f->dst.addr_data32[0] == IPV4_GET_RAW_IPSRC_U32( ICMPV4_GET_EMB_IPV4(p) )) &&
380  (f->src.addr_data32[0] == IPV4_GET_RAW_IPDST_U32( ICMPV4_GET_EMB_IPV4(p) )) &&
381  f->dp == p->icmpv4vars.emb_sport &&
382  f->sp == p->icmpv4vars.emb_dport &&
383  f->proto == ICMPV4_GET_EMB_PROTO(p) &&
384  f->recursion_level == p->recursion_level &&
385  f->vlan_id[0] == p->vlan_id[0] &&
386  f->vlan_id[1] == p->vlan_id[1])
387  {
388  return 1;
389  }
390 
391  /* no match, fall through */
392  } else {
393  /* just treat ICMP as a normal proto for now */
394  return CMP_FLOW(f, p);
395  }
396 
397  return 0;
398 }

402 static inline int FlowCompare(Flow *f, const Packet *p)
403 {
404  if (p->proto == IPPROTO_ICMP) {
405  return FlowCompareICMPv4(f, p);
406  } else if (p->proto == IPPROTO_TCP) {
407  if (CMP_FLOW(f, p) == 0)
408  return 0;
409 
410  /* if this session is 'reused', we don't return it anymore,
411  * so return false on the compare */
412  if (f->flags & FLOW_TCP_REUSED)
413  return 0;
414 
415  if (handle_tcp_reuse == 1) {
416  /* lets see if we need to consider the existing session reuse */
417  if (unlikely(TcpSessionPacketSsnReuse(p, f, f->protoctx) == 1)) {
418  /* okay, we need to setup a new flow for this packet.
419  * Flag the flow that it's been replaced by a new one */
420  f->flags |= FLOW_TCP_REUSED;
421  SCLogDebug("flow obsolete: TCP reuse will use a new flow "
422  "starting with packet %"PRIu64, p->pcap_cnt);
423  return 0;
424  }
425  }
426  return 1;
427  } else {
428  return CMP_FLOW(f, p);
429  }

453 /**
454  * \brief Get a new flow
455  *
456  * Get a new flow. We're checking memcap first and will try to make room
457  * if the memcap is reached.
458  *
459  * \param tv thread vars
460  * \param dtv decode thread vars (for flow log api thread data)
461  *
462  * \retval f *LOCKED* flow on succes, NULL on error.
463  */
464 static Flow *FlowGetNew(ThreadVars *tv, DecodeThreadVars *dtv, const Packet *p)
465 {
466  Flow *f = NULL;
467 
468  if (FlowCreateCheck(p) == 0) {
469  return NULL;
470  }
471 
472  /* get a flow from the spare queue */
473  f = FlowDequeue(&flow_spare_q);
474  if (f == NULL) {
475  /* If we reached the max memcap, we get a used flow */
476  if (!(FLOW_CHECK_MEMCAP(sizeof(Flow) + FlowStorageSize()))) {
477  /* declare state of emergency */
478  if (!(SC_ATOMIC_GET(flow_flags) & FLOW_EMERGENCY)) {
479  SC_ATOMIC_OR(flow_flags, FLOW_EMERGENCY);
480 
481  /* under high load, waking up the flow mgr each time leads
482  * to high cpu usage. Flows are not timed out much faster if
483  * we check a 1000 times a second. */
484  FlowWakeupFlowManagerThread();
485  }
486 
487  f = FlowGetUsedFlow(tv, dtv);
488  if (f == NULL) {
489  /* max memcap reached, so increments the counter */
490  if (tv != NULL && dtv != NULL) {
491  StatsIncr(tv, dtv->counter_flow_memcap);
492  }
493 
494  /* very rare, but we can fail. Just giving up */
495  return NULL;
496  }
497 
498  /* freed a flow, but it's unlocked */
499  } else {
500  /* now see if we can alloc a new flow */
501  f = FlowAlloc();
502  if (f == NULL) {
503  if (tv != NULL && dtv != NULL) {
504  StatsIncr(tv, dtv->counter_flow_memcap);
505  }
506  return NULL;
507  }
508 
509  /* flow is initialized but *unlocked* */
510  }
511  } else {
512  /* flow has been recycled before it went into the spare queue */
513 
514  /* flow is initialized (recylced) but *unlocked* */
515  }
516 
517  FLOWLOCK_WRLOCK(f);
518  return f;
519 }

521 Flow *FlowGetFlowFromHashByPacket(const Packet *p)
522 {
523  Flow *f = NULL;
524 
525  /* get the key to our bucket */
526  uint32_t key = FlowGetKey(p);
527  /* get our hash bucket and lock it */
528  FlowBucket *fb = &flow_hash[key];
529  FBLOCK_LOCK(fb);
530 
531  SCLogDebug("fb %p fb->head %p", fb, fb->head);
532 
533  f = FlowGetNew(NULL, NULL, p);
534  if (f != NULL) {
535  /* flow is locked */
536  if (fb->head == NULL) {
537  fb->head = f;
538  fb->tail = f;
539  } else {
540  f->hprev = fb->tail;
541  fb->tail->hnext = f;
542  fb->tail = f;
543  }
544 
545  /* got one, now lock, initialize and return */
546  FlowInit(f, p);
547  f->fb = fb;
548  /* update the last seen timestamp of this flow */
549  COPY_TIMESTAMP(&p->ts,&f->lastts);
550 
551  }
552  FBLOCK_UNLOCK(fb);
553  return f;
554 }

636 /** \brief Get Flow for packet
637  *
638  * Hash retrieval function for flows. Looks up the hash bucket containing the
639  * flow pointer. Then compares the packet with the found flow to see if it is
640  * the flow we need. If it isn't, walk the list until the right flow is found.
641  *
642  * If the flow is not found or the bucket was emtpy, a new flow is taken from
643  * the queue. FlowDequeue() will alloc new flows as long as we stay within our
644  * memcap limit.
645  *
646  * The p->flow pointer is updated to point to the flow.
647  *
648  * \param tv thread vars
649  * \param dtv decode thread vars (for flow log api thread data)
650  *
651  * \retval f *LOCKED* flow or NULL
652  */

657

663

665 
666  FlowHashCountIncr;
667

669  if (fb->head == NULL) {
670  f = FlowGetNew(tv, dtv, p);
671  if (f == NULL) {
672  FBLOCK_UNLOCK(fb);
673  FlowHashCountUpdate;
674  return NULL;
675  }
676 
677  /* flow is locked */
678  fb->head = f;
679  fb->tail = f;
680 
681  /* got one, now lock, initialize and return */
682  FlowInit(f, p);
683  f->fb = fb;
684 
685  /* update the last seen timestamp of this flow */
686  COPY_TIMESTAMP(&p->ts,&f->lastts);
687 
688  FBLOCK_UNLOCK(fb);
689  FlowHashCountUpdate;
690  return f;
691  }
692 
693  /* ok, we have a flow in the bucket. Let's find out if it is our flow */
694  f = fb->head;
695

697  if (FlowCompare(f, p) == 0) {
698  Flow *pf = NULL; /* previous flow */
699

702

708  if (f == NULL) {
709  FBLOCK_UNLOCK(fb);
710  FlowHashCountUpdate;
711  return NULL;
712  }
713  fb->tail = f;
714 
715  /* flow is locked */
716 
717  f->hprev = pf;
718 
719  /* initialize and return */
720  FlowInit(f, p);
721  f->fb = fb;
722 
723  /* update the last seen timestamp of this flow */
724  COPY_TIMESTAMP(&p->ts,&f->lastts);
725 
726  FBLOCK_UNLOCK(fb);
727  FlowHashCountUpdate;
728  return f;
729  }
730 
731  if (FlowCompare(f, p) != 0) {
732  /* we found our flow, lets put it on top of the

736  }

771 /** \internal
772  * \brief Get a flow from the hash directly.
773  *
774  * Called in conditions where the spare queue is empty and memcap is reached.
775  *
776  * Walks the hash until a flow can be freed. Timeouts are disregarded, use_cnt
777  * is adhered to. "flow_prune_idx" atomic int makes sure we don't start at the
778  * top each time since that would clear the top of the hash leading to longer
779  * and longer search times under high pressure (observed).
780  *
781  * \param tv thread vars
782  * \param dtv decode thread vars (for flow log api thread data)
783  *
784  * \retval f flow or NULL
785  */
786 static Flow *FlowGetUsedFlow(ThreadVars *tv, DecodeThreadVars *dtv)
787 {
788  uint32_t idx = SC_ATOMIC_GET(flow_prune_idx) % flow_config.hash_size;
789  uint32_t cnt = flow_config.hash_size;
790 
791  while (cnt--) {
792  if (++idx >= flow_config.hash_size)
793  idx = 0;
794 
795  FlowBucket *fb = &flow_hash[idx];
796 
797  if (FBLOCK_TRYLOCK(fb) != 0)
798  continue;
799 
800  Flow *f = fb->tail;
801  if (f == NULL) {
802  FBLOCK_UNLOCK(fb);
803  continue;
804  }
805 
806  if (FLOWLOCK_TRYWRLOCK(f) != 0) {
807  FBLOCK_UNLOCK(fb);
808  continue;
809  }
810 
811  /** never prune a flow that is used by a packet or stream msg
812  * we are currently processing in one of the threads */
813  if (SC_ATOMIC_GET(f->use_cnt) > 0) {
814  FBLOCK_UNLOCK(fb);
815  FLOWLOCK_UNLOCK(f);
816  continue;
817  }
818 
819  /* remove from the hash */
820  if (f->hprev != NULL)
821  f->hprev->hnext = f->hnext;
822  if (f->hnext != NULL)
823  f->hnext->hprev = f->hprev;
824  if (fb->head == f)
825  fb->head = f->hnext;
826  if (fb->tail == f)
827  fb->tail = f->hprev;
828 
829  f->hnext = NULL;
830  f->hprev = NULL;
831  f->fb = NULL;
832  FBLOCK_UNLOCK(fb);
833 
834  int state = SC_ATOMIC_GET(f->flow_state);
835  if (state == FLOW_STATE_NEW)
836  f->flow_end_flags |= FLOW_END_FLAG_STATE_NEW;
837  else if (state == FLOW_STATE_ESTABLISHED)
838  f->flow_end_flags |= FLOW_END_FLAG_STATE_ESTABLISHED;
839  else if (state == FLOW_STATE_CLOSED)
840  f->flow_end_flags |= FLOW_END_FLAG_STATE_CLOSED;
841 
842  f->flow_end_flags |= FLOW_END_FLAG_FORCED;
843 
844  if (SC_ATOMIC_GET(flow_flags) & FLOW_EMERGENCY)
845  f->flow_end_flags |= FLOW_END_FLAG_EMERGENCY;
846 
847  /* invoke flow log api */
848  if (dtv && dtv->output_flow_thread_data)
849  (void)OutputFlowLog(tv, dtv->output_flow_thread_data, f);
850 
851  FlowClearMemory(f, f->protomap);
852 
853  FLOWLOCK_UNLOCK(f);
854 
855  (void) SC_ATOMIC_ADD(flow_prune_idx, (flow_config.hash_size - cnt));
856  return f;
857  }
858 
859  return NULL;
860 }
