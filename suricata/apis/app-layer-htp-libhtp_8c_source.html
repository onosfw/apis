55 /**
56  * \brief A direct flick off libhtp-0.5.x htp_parse_positive_integer_whitespace().
57  */
58 static int64_t SC_htp_parse_positive_integer_whitespace(unsigned char *data, size_t len, int base)
59 {
60  if (len == 0) return -1003;
61 
62  size_t last_pos;
63  size_t pos = 0;
64 
65  // Ignore LWS before
66  while ((pos < len) && (SC_htp_is_lws(data[pos]))) pos++;
67  if (pos == len) return -1001;
68 
69  int64_t r = bstr_util_mem_to_pint(data + pos, len - pos, base, &last_pos);
70  if (r < 0) return r;
71 
72  // Move after the last digit
73  pos += last_pos;
74 
75  // Ignore LWS after
76  while (pos < len) {
77  if (!SC_htp_is_lws(data[pos])) {
78  return -1002;
79  }
80 
81  pos++;
82  }
83 
84  return r;
85 }

95 /**
96  * \brief Generates the normalized uri.
97  *
98  * Libhtp doesn't recreate the whole normalized uri and save it.
99  * That duty has now been passed to us. A lot of this code has been
100  * copied from libhtp.
101  *
102  * Keep an eye out on the tx->parsed_uri struct and how the parameters
103  * in it are generated, just in case some modifications are made to
104  * them in the future.
105  *
106  * \param uri_include_all boolean to indicate if scheme, username/password,
107  hostname and port should be part of the buffer
108  */
109 bstr *SCHTPGenerateNormalizedUri(htp_tx_t *tx, htp_uri_t *uri, int uri_include_all)
110 {
111  if (uri == NULL)
112  return NULL;
113 
114  // On the first pass determine the length of the final string
115  size_t len = 0;
116 
117  if (uri_include_all) {
118  if (uri->scheme != NULL) {
119  len += bstr_len(uri->scheme);
120  len += 3; // "://"
121  }
122 
123  if ((uri->username != NULL) || (uri->password != NULL)) {
124  if (uri->username != NULL) {
125  len += bstr_len(uri->username);
126  }
127 
128  len += 1; // ":"
129 
130  if (uri->password != NULL) {
131  len += bstr_len(uri->password);
132  }
133 
134  len += 1; // "@"
135  }
136 
137  if (uri->hostname != NULL) {
138  len += bstr_len(uri->hostname);
139  }
140 
141  if (uri->port != NULL) {
142  len += 1; // ":"
143  len += bstr_len(uri->port);
144  }
145  }
146 
147  if (uri->path != NULL) {
148  len += bstr_len(uri->path);
149  }
150 
151  if (uri->query != NULL) {
152  len += 1; // "?"
153  len += bstr_len(uri->query);
154  }
155 
156  if (uri->fragment != NULL) {
157  len += 1; // "#"

159  }

216  }

219 }
