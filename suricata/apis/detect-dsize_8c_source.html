93 /**
94  * \internal
95  * \brief This function is used to match flags on a packet with those passed via dsize:
96  *
97  * \param t pointer to thread vars
98  * \param det_ctx pointer to the pattern matcher thread
99  * \param p pointer to the current packet
100  * \param s pointer to the Signature
101  * \param m pointer to the sigmatch
102  *
103  * \retval 0 no match
104  * \retval 1 match
105  */
106 int DetectDsizeMatch (ThreadVars *t, DetectEngineThreadCtx *det_ctx, Packet *p, Signature *s, const SigMatchCtx *ctx)
107 {
108  SCEnter();
109  int ret = 0;
110 
111  if (PKT_IS_PSEUDOPKT(p)) {
112  SCReturnInt(0);
113  }
114 
115  const DetectDsizeData *dd = (const DetectDsizeData *)ctx;
116 
117  SCLogDebug("p->payload_len %"PRIu16"", p->payload_len);
118 
119  if (dd->mode == DETECTDSIZE_EQ && dd->dsize == p->payload_len)
120  ret = 1;
121  else if (dd->mode == DETECTDSIZE_LT && p->payload_len < dd->dsize)
122  ret = 1;
123  else if (dd->mode == DETECTDSIZE_GT && p->payload_len > dd->dsize)
124  ret = 1;
125  else if (dd->mode == DETECTDSIZE_RA && p->payload_len > dd->dsize && p->payload_len < dd->dsize2)
126  ret = 1;
127 
128  SCReturnInt(ret);
129 }

131 /**
132  * \internal
133  * \brief This function is used to parse dsize options passed via dsize: keyword
134  *
135  * \param rawstr Pointer to the user provided dsize options
136  *
137  * \retval dd pointer to DetectDsizeData on success
138  * \retval NULL on failure
139  */
140 DetectDsizeData *DetectDsizeParse (char *rawstr)
141 {
142  DetectDsizeData *dd = NULL;
143 #define MAX_SUBSTRINGS 30
144  int ret = 0, res = 0;
145  int ov[MAX_SUBSTRINGS];
146  char mode[2] = "";
147  char value1[6] = "";
148  char value2[6] = "";
149  char range[3] = "";
150 
151  ret = pcre_exec(parse_regex, parse_regex_study, rawstr, strlen(rawstr), 0, 0, ov, MAX_SUBSTRINGS);
152  if (ret < 3 || ret > 5) {
153  SCLogError(SC_ERR_PCRE_MATCH,"Parse error %s", rawstr);
154  goto error;
155  }
156 
157  res = pcre_copy_substring((char *)rawstr, ov, MAX_SUBSTRINGS, 1, mode, sizeof(mode));
158  if (res < 0) {
159  SCLogError(SC_ERR_PCRE_GET_SUBSTRING,"pcre_copy_substring failed");
160  goto error;
161  }
162  SCLogDebug("mode \"%s\"", mode);
163 
164  res = pcre_copy_substring((char *)rawstr, ov, MAX_SUBSTRINGS, 2, value1, sizeof(value1));
165  if (res < 0) {
166  SCLogError(SC_ERR_PCRE_GET_SUBSTRING,"pcre_copy_substring failed");
167  goto error;
168  }
169  SCLogDebug("value1 \"%s\"", value1);
170 
171  if (ret > 3) {
172  res = pcre_copy_substring((char *)rawstr, ov, MAX_SUBSTRINGS, 3, range, sizeof(range));
173  if (res < 0) {
174  SCLogError(SC_ERR_PCRE_GET_SUBSTRING,"pcre_copy_substring failed");
175  goto error;
176  }
177  SCLogDebug("range \"%s\"", range);
178 
179  if (ret > 4) {
180  res = pcre_copy_substring((char *)rawstr, ov, MAX_SUBSTRINGS, 4, value2, sizeof(value2));
181  if (res < 0) {
182  SCLogError(SC_ERR_PCRE_GET_SUBSTRING,"pcre_copy_substring failed");
183  goto error;
184  }
185  SCLogDebug("value2 \"%s\"", value2);
186  }
187  }
188 
189  dd = SCMalloc(sizeof(DetectDsizeData));
190  if (unlikely(dd == NULL))
191  goto error;
192  dd->dsize = 0;
193  dd->dsize2 = 0;
194  dd->mode = DETECTDSIZE_EQ; // default
195 
196  if (strlen(mode) > 0) {
197  if (mode[0] == '<')
198  dd->mode = DETECTDSIZE_LT;
199  else if (mode[0] == '>')
200  dd->mode = DETECTDSIZE_GT;
201  else
202  dd->mode = DETECTDSIZE_EQ;
203  }
204 
205  if (strcmp("<>", range) == 0) {
206  if (strlen(mode) != 0) {
207  SCLogError(SC_ERR_INVALID_ARGUMENT,"Range specified but mode also set");
208  goto error;
209  }
210  dd->mode = DETECTDSIZE_RA;
211  }
212 
213  /** set the first dsize value */
214  if (ByteExtractStringUint16(&dd->dsize,10,strlen(value1),value1) <= 0) {
215  SCLogError(SC_ERR_INVALID_ARGUMENT, "Invalid size value1:\"%s\"", value1);
216  goto error;
217  }
218 
219  /** set the second dsize value if specified */
220  if (strlen(value2) > 0) {
221  if (dd->mode != DETECTDSIZE_RA) {
222  SCLogError(SC_ERR_INVALID_ARGUMENT,"Multiple dsize values specified but mode is not range");
223  goto error;
224  }
225 
226  if (ByteExtractStringUint16(&dd->dsize2,10,strlen(value2),value2) <= 0) {
227  SCLogError(SC_ERR_INVALID_ARGUMENT,"Invalid size value2:\"%s\"",value2);
228  goto error;
229  }
230 
231  if (dd->dsize2 <= dd->dsize) {
232  SCLogError(SC_ERR_INVALID_ARGUMENT,"dsize2:%"PRIu16" <= dsize:%"PRIu16"",dd->dsize2,dd->dsize);
233  goto error;
234  }
235  }
236 
237  SCLogDebug("dsize parsed successfully dsize: %"PRIu16" dsize2: %"PRIu16"",dd->dsize,dd->dsize2);
238  return dd;
239 
240 error:
241  if (dd)
242  SCFree(dd);
243  return NULL;
244 }

246 /**
247  * \internal
248  * \brief this function is used to add the parsed dsize into the current signature
249  *
250  * \param de_ctx pointer to the Detection Engine Context
251  * \param s pointer to the Current Signature
252  * \param rawstr pointer to the user provided flags options
253  *
254  * \retval 0 on Success
255  * \retval -1 on Failure
256  */
257 static int DetectDsizeSetup (DetectEngineCtx *de_ctx, Signature *s, char *rawstr)
258 {
259  DetectDsizeData *dd = NULL;
260  SigMatch *sm = NULL;
261 
262  if (SigMatchGetLastSMFromLists(s, 2,
263  DETECT_DSIZE,
264  s->sm_lists_tail[DETECT_SM_LIST_MATCH]) != NULL) {
265  SCLogError(SC_ERR_INVALID_SIGNATURE, "Can't use 2 or more dsizes in "
266  "the same sig. Invalidating signature.");
267  goto error;
268  }
269 
270  SCLogDebug("\'%s\'", rawstr);
271 
272  dd = DetectDsizeParse(rawstr);
273  if (dd == NULL) {
274  SCLogError(SC_ERR_INVALID_ARGUMENT,"Parsing \'%s\' failed", rawstr);
275  goto error;
276  }
277 
278  /* Okay so far so good, lets get this into a SigMatch
279  * and put it in the Signature. */
280  sm = SigMatchAlloc();
281  if (sm == NULL){
282  SCLogError(SC_ERR_MEM_ALLOC, "Failed to allocate memory for SigMatch");
283  SCFree(dd);
284  goto error;
285  }
286 
287  sm->type = DETECT_DSIZE;
288  sm->ctx = (SigMatchCtx *)dd;
289 
290  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);
291 
292  SCLogDebug("dd->dsize %"PRIu16", dd->dsize2 %"PRIu16", dd->mode %"PRIu8"",
293  dd->dsize, dd->dsize2, dd->mode);
294  /* tell the sig it has a dsize to speed up engine init */
295  s->flags |= SIG_FLAG_REQUIRE_PACKET;
296  s->flags |= SIG_FLAG_DSIZE;
297 
298  if (s->dsize_sm == NULL) {
299  s->dsize_sm = sm;
300  }
301 
302  return 0;
303 
304 error:
305  return -1;
306 }
