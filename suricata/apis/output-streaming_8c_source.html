101 int Streamer(void *cbdata, Flow *f, uint8_t *data, uint32_t data_len, uint64_t tx_id, uint8_t flags)
102 {
103  StreamerCallbackData *streamer_cbdata = (StreamerCallbackData *)cbdata;
104  BUG_ON(streamer_cbdata == NULL);
105  OutputStreamingLogger *logger = streamer_cbdata->logger;
106  OutputLoggerThreadStore *store = streamer_cbdata->store;
107  ThreadVars *tv = streamer_cbdata->tv;
108 #ifdef PROFILING
109  Packet *p = streamer_cbdata->p;
110 #endif
111  BUG_ON(logger == NULL);
112  BUG_ON(store == NULL);
113 
114  while (logger && store) {
115  BUG_ON(logger->LogFunc == NULL);
116 
117  if (logger->type == streamer_cbdata->type) {
118  SCLogDebug("logger %p", logger);
119  PACKET_PROFILING_TMM_START(p, logger->module_id);
120  logger->LogFunc(tv, store->thread_data, (const Flow *)f, data, data_len, tx_id, flags);
121  PACKET_PROFILING_TMM_END(p, logger->module_id);
122  }
123 
124  logger = logger->next;
125  store = store->next;
126 
127  BUG_ON(logger == NULL && store != NULL);
128  BUG_ON(logger != NULL && store == NULL);
129  }
130 
131  return 0;
132 }

143 int HttpBodyIterator(Flow *f, int close, void *cbdata, uint8_t iflags)
144 {
145  SCLogDebug("called with %p, %d, %p, %02x", f, close, cbdata, iflags);
146 
147  HtpState *s = f->alstate;
148  if (s != NULL && s->conn != NULL) {
149  int tx_progress_done_value_ts =
150  AppLayerParserGetStateProgressCompletionStatus(IPPROTO_TCP,
151  ALPROTO_HTTP, STREAM_TOSERVER);
152  int tx_progress_done_value_tc =
153  AppLayerParserGetStateProgressCompletionStatus(IPPROTO_TCP,
154  ALPROTO_HTTP, STREAM_TOCLIENT);
155 
156  // for each tx
157  uint64_t tx_id = 0;
158  uint64_t total_txs = AppLayerParserGetTxCnt(f->proto, f->alproto, f->alstate);
159  SCLogDebug("s->conn %p", s->conn);
160  for (tx_id = 0; tx_id < total_txs; tx_id++) { // TODO optimization store log tx
161  htp_tx_t *tx = AppLayerParserGetTx(f->proto, f->alproto, f->alstate, tx_id);
162  if (tx != NULL) {
163  int tx_done = 0;
164  int tx_logged = 0;
165 
166  int tx_progress_ts = AppLayerParserGetStateProgress(
167  IPPROTO_TCP, ALPROTO_HTTP, tx, FlowGetDisruptionFlags(f, STREAM_TOSERVER));
168  if (tx_progress_ts >= tx_progress_done_value_ts) {
169  int tx_progress_tc = AppLayerParserGetStateProgress(
170  IPPROTO_TCP, ALPROTO_HTTP, tx, FlowGetDisruptionFlags(f, STREAM_TOCLIENT));
171  if (tx_progress_tc >= tx_progress_done_value_tc) {
172  tx_done = 1;
173  }
174  }
175 
176  SCLogDebug("tx %p", tx);
177  HtpTxUserData *htud = (HtpTxUserData *) htp_tx_get_user_data(tx);
178  if (htud != NULL) {
179  SCLogDebug("htud %p", htud);
180  HtpBody *body = NULL;
181  if (iflags & OUTPUT_STREAMING_FLAG_TOCLIENT)
182  body = &htud->request_body;
183  else if (iflags & OUTPUT_STREAMING_FLAG_TOSERVER)
184  body = &htud->response_body;
185 
186  if (body == NULL) {
187  SCLogDebug("no body");
188  goto next;
189  }
190  if (body->first == NULL) {
191  SCLogDebug("no body chunks");
192  goto next;
193  }
194  if (body->last->logged == 1) {
195  SCLogDebug("all logged already");
196  goto next;
197  }
198 
199  // for each chunk
200  HtpBodyChunk *chunk = body->first;
201  for ( ; chunk != NULL; chunk = chunk->next) {
202  if (chunk->logged) {
203  SCLogDebug("logged %d", chunk->logged);
204  continue;
205  }
206 
207  uint8_t flags = iflags | OUTPUT_STREAMING_FLAG_TRANSACTION;
208  if (chunk->stream_offset == 0)
209  flags |= OUTPUT_STREAMING_FLAG_OPEN;
210  /* if we need to close and we're at the last segment in the list
211  * we add the 'close' flag so the logger can close up. */
212  if ((tx_done || close) && chunk->next == NULL) {
213  flags |= OUTPUT_STREAMING_FLAG_CLOSE;
214  }
215 
216  // invoke Streamer
217  Streamer(cbdata, f, chunk->data, (uint32_t)chunk->len, tx_id, flags);
218  //PrintRawDataFp(stdout, chunk->data, chunk->len);
219  chunk->logged = 1;
220  tx_logged = 1;
221  }
222 
223  next:
224  /* if we need to close we need to invoke the Streamer for sure. If we
225  * logged no chunks, we call the Streamer with NULL data so it can
226  * close up. */
227  if (tx_logged == 0 && (close||tx_done)) {
228  Streamer(cbdata, f, NULL, 0, tx_id,
229  OUTPUT_STREAMING_FLAG_CLOSE|OUTPUT_STREAMING_FLAG_TRANSACTION);
230  }
231  }
232  }
233  }
234  }
235 
236 
237  return 0;
238 }

240 int StreamIterator(Flow *f, TcpStream *stream, int close, void *cbdata, uint8_t iflags)
241 {
242  SCLogDebug("called with %p, %d, %p, %02x", f, close, cbdata, iflags);
243  int logged = 0;
244 
245  /* optimization: don't iterate list if we've logged all,
246  * so check the last segment's flags */
247  if (stream->seg_list_tail != NULL &&
248  (!(stream->seg_list_tail->flags & SEGMENTTCP_FLAG_LOGAPI_PROCESSED)))
249  {
250  TcpSegment *seg = stream->seg_list;
251  while (seg) {
252  uint8_t flags = iflags;
253 
254  if (seg->flags & SEGMENTTCP_FLAG_LOGAPI_PROCESSED) {
255  seg = seg->next;
256  continue;
257  }
258 
259  if (SEQ_GT(seg->seq + seg->payload_len, stream->last_ack)) {
260  SCLogDebug("seg not (fully) acked yet");
261  break;
262  }
263 
264  if (seg->seq == stream->isn + 1)
265  flags |= OUTPUT_STREAMING_FLAG_OPEN;
266  /* if we need to close and we're at the last segment in the list
267  * we add the 'close' flag so the logger can close up. */
268  if (close && seg->next == NULL)
269  flags |= OUTPUT_STREAMING_FLAG_CLOSE;
270 
271  Streamer(cbdata, f, seg->payload, (uint32_t)seg->payload_len, 0, flags);
272 
273  seg->flags |= SEGMENTTCP_FLAG_LOGAPI_PROCESSED;
274 
275  seg = seg->next;
276 
277  logged = 1;
278  }
279  }
280 
281  /* if we need to close we need to invoke the Streamer for sure. If we
282  * logged no segments, we call the Streamer with NULL data so it can
283  * close up. */
284  if (logged == 0 && close) {
285  Streamer(cbdata, f, NULL, 0, 0, OUTPUT_STREAMING_FLAG_CLOSE);
286  }
287 
288  return 0;
289 }

291 static TmEcode OutputStreamingLog(ThreadVars *tv, Packet *p, void *thread_data, PacketQueue *pq, PacketQueue *postpq)
292 {
293  BUG_ON(thread_data == NULL);
294  BUG_ON(list == NULL);
295 
296  OutputLoggerThreadData *op_thread_data = (OutputLoggerThreadData *)thread_data;
297  OutputStreamingLogger *logger = list;
298  OutputLoggerThreadStore *store = op_thread_data->store;
299 
300  StreamerCallbackData streamer_cbdata = { logger, store, tv, p , 0};
301 
302  BUG_ON(logger == NULL && store != NULL);
303  BUG_ON(logger != NULL && store == NULL);
304  BUG_ON(logger == NULL && store == NULL);
305 
306  uint8_t flags = 0;
307  Flow * const f = p->flow;
308 
309  /* no flow, no streaming */
310  if (f == NULL) {
311  SCReturnInt(TM_ECODE_OK);
312  }
313 
314  if (p->flowflags & FLOW_PKT_TOCLIENT)
315  flags |= OUTPUT_STREAMING_FLAG_TOCLIENT;
316  else
317  flags |= OUTPUT_STREAMING_FLAG_TOSERVER;
318 
319  FLOWLOCK_WRLOCK(f);
320 
321  if (op_thread_data->loggers & (1<<STREAMING_TCP_DATA)) {
322  TcpSession *ssn = f->protoctx;
323  if (ssn) {
324  int close = (ssn->state >= TCP_CLOSED);
325  close |= ((p->flags & PKT_PSEUDO_STREAM_END) ? 1 : 0);
326  SCLogDebug("close ? %s", close ? "yes" : "no");
327 
328  TcpStream *stream = flags & OUTPUT_STREAMING_FLAG_TOSERVER ? &ssn->client : &ssn->server;
329 
330  streamer_cbdata.type = STREAMING_TCP_DATA;
331  StreamIterator(p->flow, stream, close, (void *)&streamer_cbdata, flags);
332  }
333  }
334  if (op_thread_data->loggers & (1<<STREAMING_HTTP_BODIES)) {
335  if (f->alproto == ALPROTO_HTTP && f->alstate != NULL) {
336  int close = 0;
337  TcpSession *ssn = f->protoctx;
338  if (ssn) {
339  close = (ssn->state >= TCP_CLOSED);
340  close |= ((p->flags & PKT_PSEUDO_STREAM_END) ? 1 : 0);
341  }
342  SCLogDebug("close ? %s", close ? "yes" : "no");
343  streamer_cbdata.type = STREAMING_HTTP_BODIES;
344  HttpBodyIterator(f, close, (void *)&streamer_cbdata, flags);
345  }
346  }
347 
348  FLOWLOCK_UNLOCK(f);
349  return TM_ECODE_OK;
350 }

352 /** \brief thread init for the tx logger
353  * This will run the thread init functions for the individual registered

355 static TmEcode OutputStreamingLogThreadInit(ThreadVars *tv, void *initdata, void **data) {
356  OutputLoggerThreadData *td = SCMalloc(sizeof(*td));
357  if (td == NULL)
358  return TM_ECODE_FAILED;
359  memset(td, 0x00, sizeof(*td));
360 
361  *data = (void *)td;
362 
363  SCLogDebug("OutputStreamingLogThreadInit happy (*data %p)", *data);
364 
365  OutputStreamingLogger *logger = list;
366  while (logger) {
367  TmModule *tm_module = TmModuleGetByName((char *)logger->name);
368  if (tm_module == NULL) {
369  SCLogError(SC_ERR_INVALID_ARGUMENT,
370  "TmModuleGetByName for %s failed", logger->name);
371  exit(EXIT_FAILURE);
372  }
373 
374  if (tm_module->ThreadInit) {
375  void *retptr = NULL;
376  if (tm_module->ThreadInit(tv, (void *)logger->output_ctx, &retptr) == TM_ECODE_OK) {
377  OutputLoggerThreadStore *ts = SCMalloc(sizeof(*ts));
378 /* todo */ BUG_ON(ts == NULL);
379  memset(ts, 0x00, sizeof(*ts));
380 
381  /* store thread handle */
382  ts->thread_data = retptr;
383 
384  if (td->store == NULL) {
385  td->store = ts;
386  } else {
387  OutputLoggerThreadStore *tmp = td->store;
388  while (tmp->next != NULL)
389  tmp = tmp->next;
390  tmp->next = ts;
391  }
392 
393  SCLogInfo("%s is now set up", logger->name);
394  }
395  }
396 
397  td->loggers |= (1<<logger->type);
398 
399  logger = logger->next;
400  }
401 
402  return TM_ECODE_OK;
403 }
