42 /**
43  * \brief Function to decode Teredo packets
44  *
45  * \retval TM_ECODE_FAILED if packet is not a Teredo packet, TM_ECODE_OK if it is
46  */
47 int DecodeTeredo(ThreadVars *tv, DecodeThreadVars *dtv, Packet *p, uint8_t *pkt, uint16_t len, PacketQueue *pq)
48 {
49 
50  uint8_t *start = pkt;
51 
52  /* Is this packet to short to contain an IPv6 packet ? */
53  if (len < IPV6_HEADER_LEN)
54  return TM_ECODE_FAILED;
55 
56  /* Teredo encapsulate IPv6 in UDP and can add some custom message
57  * part before the IPv6 packet. In our case, we just want to get
58  * over an ORIGIN indication. So we just make one offset if needed. */
59  if (start[0] == 0x0) {
60  switch (start[1]) {
61  /* origin indication: compatible with tunnel */
62  case 0x0:
63  /* offset is coherent with len and presence of an IPv6 header */
64  if (len >= TEREDO_ORIG_INDICATION_LENGTH + IPV6_HEADER_LEN)
65  start += TEREDO_ORIG_INDICATION_LENGTH;
66  else
67  return TM_ECODE_FAILED;
68  break;
69  /* authentication: negotiation not real tunnel */
70  case 0x1:
71  return TM_ECODE_FAILED;
72  /* this case is not possible in Teredo: not that protocol */
73  default:
74  return TM_ECODE_FAILED;
75  }
76  }
77 
78  /* There is no specific field that we can check to prove that the packet
79  * is a Teredo packet. We've zapped here all the possible Teredo header
80  * and we should have an IPv6 packet at the start pointer.
81  * We then can only do two checks before sending the encapsulated packets
82  * to decoding:
83  * - The packet has a protocol version which is IPv6.
84  * - The IPv6 length of the packet matches what remains in buffer.
85  */
86  if (IP_GET_RAW_VER(start) == 6) {
87  IPV6Hdr *thdr = (IPV6Hdr *)start;
88  if (len == IPV6_HEADER_LEN +
89  IPV6_GET_RAW_PLEN(thdr) + (start - pkt)) {
90  if (pq != NULL) {
91  int blen = len - (start - pkt);
92  /* spawn off tunnel packet */
93  Packet *tp = PacketTunnelPktSetup(tv, dtv, p, start, blen,
94  DECODE_TUNNEL_IPV6, pq);
95  if (tp != NULL) {
96  PKT_SET_SRC(tp, PKT_SRC_DECODER_TEREDO);
97  /* add the tp to the packet queue. */
98  PacketEnqueue(pq,tp);
99  StatsIncr(tv, dtv->counter_teredo);
100  return TM_ECODE_OK;
101  }
102  }
103  }
104  return TM_ECODE_FAILED;
105  }
106 
107  return TM_ECODE_FAILED;
108 }
