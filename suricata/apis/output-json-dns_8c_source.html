73 static void LogQuery(LogDnsLogThread *aft, json_t *js, DNSTransaction *tx,
74  uint64_t tx_id, DNSQueryEntry *entry)
75 {
76  MemBuffer *buffer = (MemBuffer *)aft->buffer;
77 
78  SCLogDebug("got a DNS request and now logging !!");
79 
80  json_t *djs = json_object();
81  if (djs == NULL) {
82  return;
83  }
84 
85  /* reset */
86  MemBufferReset(buffer);
87 
88  /* type */
89  json_object_set_new(djs, "type", json_string("query"));
90 
91  /* id */
92  json_object_set_new(djs, "id", json_integer(tx->tx_id));
93 
94  /* query */
95  char *c;
96  c = BytesToString((uint8_t *)((uint8_t *)entry + sizeof(DNSQueryEntry)), entry->len);
97  if (c != NULL) {
98  json_object_set_new(djs, "rrname", json_string(c));
99  SCFree(c);
100  }
101 
102  /* name */
103  char record[16] = "";
104  DNSCreateTypeString(entry->type, record, sizeof(record));
105  json_object_set_new(djs, "rrtype", json_string(record));
106 
107  /* tx id (tx counter) */
108  json_object_set_new(djs, "tx_id", json_integer(tx_id));
109 
110  /* dns */
111  json_object_set_new(js, "dns", djs);
112  OutputJSONBuffer(js, aft->dnslog_ctx->file_ctx, buffer);
113  json_object_del(js, "dns");
114 }

116 static void OutputAnswer(LogDnsLogThread *aft, json_t *djs, DNSTransaction *tx, DNSAnswerEntry *entry)
117 {
118  MemBuffer *buffer = (MemBuffer *)aft->buffer;
119  json_t *js = json_object();
120  if (js == NULL)
121  return;
122 
123  /* type */
124  json_object_set_new(js, "type", json_string("answer"));
125 
126  /* id */
127  json_object_set_new(js, "id", json_integer(tx->tx_id));
128 
129  /* rcode */
130  char rcode[16] = "";
131  DNSCreateRcodeString(tx->rcode, rcode, sizeof(rcode));
132  json_object_set_new(js, "rcode", json_string(rcode));
133 
134  /* we are logging an answer RR */
135  if (entry != NULL) {
136  /* query */
137  if (entry->fqdn_len > 0) {
138  char *c;
139  c = BytesToString((uint8_t *)((uint8_t *)entry + sizeof(DNSAnswerEntry)),
140  entry->fqdn_len);
141  if (c != NULL) {
142  json_object_set_new(js, "rrname", json_string(c));
143  SCFree(c);
144  }
145  }
146 
147  /* name */
148  char record[16] = "";
149  DNSCreateTypeString(entry->type, record, sizeof(record));
150  json_object_set_new(js, "rrtype", json_string(record));
151 
152  /* ttl */
153  json_object_set_new(js, "ttl", json_integer(entry->ttl));
154 
155  uint8_t *ptr = (uint8_t *)((uint8_t *)entry + sizeof(DNSAnswerEntry)+ entry->fqdn_len);
156  if (entry->type == DNS_RECORD_TYPE_A) {
157  char a[16] = "";
158  PrintInet(AF_INET, (const void *)ptr, a, sizeof(a));
159  json_object_set_new(js, "rdata", json_string(a));
160  } else if (entry->type == DNS_RECORD_TYPE_AAAA) {
161  char a[46] = "";
162  PrintInet(AF_INET6, (const void *)ptr, a, sizeof(a));
163  json_object_set_new(js, "rdata", json_string(a));
164  } else if (entry->data_len == 0) {
165  json_object_set_new(js, "rdata", json_string(""));
166  } else if (entry->type == DNS_RECORD_TYPE_TXT || entry->type == DNS_RECORD_TYPE_CNAME ||
167  entry->type == DNS_RECORD_TYPE_MX || entry->type == DNS_RECORD_TYPE_PTR) {
168  if (entry->data_len != 0) {
169  char buffer[256] = "";
170  uint16_t copy_len = entry->data_len < (sizeof(buffer) - 1) ?
171  entry->data_len : sizeof(buffer) - 1;
172  memcpy(buffer, ptr, copy_len);
173  buffer[copy_len] = '\0';
174  json_object_set_new(js, "rdata", json_string(buffer));
175  } else {
176  json_object_set_new(js, "rdata", json_string(""));
177  }
178  }
179  }
180 
181  /* reset */
182  MemBufferReset(buffer);
183  json_object_set_new(djs, "dns", js);
184  OutputJSONBuffer(djs, aft->dnslog_ctx->file_ctx, buffer);
185  json_object_del(djs, "dns");
186 
187  return;
188 }

190 static void OutputFailure(LogDnsLogThread *aft, json_t *djs, DNSTransaction *tx, DNSQueryEntry *entry)
191 {
192  MemBuffer *buffer = (MemBuffer *)aft->buffer;
193  json_t *js = json_object();
194  if (js == NULL)
195  return;
196 
197  /* type */
198  json_object_set_new(js, "type", json_string("answer"));
199 
200  /* id */
201  json_object_set_new(js, "id", json_integer(tx->tx_id));
202 
203  /* rcode */
204  char rcode[16] = "";
205  DNSCreateRcodeString(tx->rcode, rcode, sizeof(rcode));
206  json_object_set_new(js, "rcode", json_string(rcode));
207 
208  /* no answer RRs, use query for rname */
209  char *c;
210  c = BytesToString((uint8_t *)((uint8_t *)entry + sizeof(DNSQueryEntry)), entry->len);
211  if (c != NULL) {
212  json_object_set_new(js, "rrname", json_string(c));
213  SCFree(c);
214  }
215 
216  /* reset */
217  MemBufferReset(buffer);
218  json_object_set_new(djs, "dns", js);
219  OutputJSONBuffer(djs, aft->dnslog_ctx->file_ctx, buffer);
220  json_object_del(djs, "dns");
221 
222  return;
223 }

225 static void LogAnswers(LogDnsLogThread *aft, json_t *js, DNSTransaction *tx, uint64_t tx_id)
226 {
227 
228  SCLogDebug("got a DNS response and now logging !!");
229 
230  /* rcode != noerror */
231  if (tx->rcode) {
232  /* Most DNS servers do not support multiple queries because
233  * the rcode in response is not per-query. Multiple queries
234  * are likely to lead to FORMERR, so log this. */
235  DNSQueryEntry *query = NULL;
236  TAILQ_FOREACH(query, &tx->query_list, next) {
237  OutputFailure(aft, js, tx, query);
238  }
239  }
240 
241  DNSAnswerEntry *entry = NULL;
242  TAILQ_FOREACH(entry, &tx->answer_list, next) {
243  OutputAnswer(aft, js, tx, entry);
244  }
245 
246  entry = NULL;
247  TAILQ_FOREACH(entry, &tx->authority_list, next) {
248  OutputAnswer(aft, js, tx, entry);
249  }
250 
251 }

253 static int JsonDnsLogger(ThreadVars *tv, void *thread_data, const Packet *p, Flow *f, void *alstate, void *txptr, uint64_t tx_id)
254 {
255  SCEnter();
256 
257  LogDnsLogThread *td = (LogDnsLogThread *)thread_data;
258  DNSTransaction *tx = txptr;
259  json_t *js;
260 
261  DNSQueryEntry *query = NULL;
262  TAILQ_FOREACH(query, &tx->query_list, next) {
263  js = CreateJSONHeader((Packet *)p, 1, "dns");
264  if (unlikely(js == NULL))
265  return TM_ECODE_OK;
266 
267  LogQuery(td, js, tx, tx_id, query);
268 
269  json_decref(js);
270  }
271 
272  js = CreateJSONHeader((Packet *)p, 0, "dns");
273  if (unlikely(js == NULL))
274  return TM_ECODE_OK;
275 
276  LogAnswers(td, js, tx, tx_id);
277 
278  json_decref(js);
279 
280  SCReturnInt(TM_ECODE_OK);
281 }

284 static TmEcode LogDnsLogThreadInit(ThreadVars *t, void *initdata, void **data)
285 {
286  LogDnsLogThread *aft = SCMalloc(sizeof(LogDnsLogThread));
287  if (unlikely(aft == NULL))
288  return TM_ECODE_FAILED;
289  memset(aft, 0, sizeof(LogDnsLogThread));
290 
291  if(initdata == NULL)
292  {
293  SCLogDebug("Error getting context for DNSLog. \"initdata\" argument NULL");
294  SCFree(aft);
295  return TM_ECODE_FAILED;
296  }
297 
298  aft->buffer = MemBufferCreateNew(OUTPUT_BUFFER_SIZE);
299  if (aft->buffer == NULL) {
300  SCFree(aft);
301  return TM_ECODE_FAILED;
302  }
303 
304  /* Use the Ouptut Context (file pointer and mutex) */
305  aft->dnslog_ctx= ((OutputCtx *)initdata)->data;
306 
307  *data = (void *)aft;
308  return TM_ECODE_OK;
309 }

372 /** \brief Create a new dns log LogFileCtx.
373  * \param conf Pointer to ConfNode containing this loggers configuration.
374  * \return NULL if failure, LogFileCtx* to the file_ctx if succesful
375  * */
376 static OutputCtx *JsonDnsLogInitCtx(ConfNode *conf)
377 {
378  LogFileCtx *file_ctx = LogFileNewCtx();
379 
380  if(file_ctx == NULL) {
381  SCLogError(SC_ERR_DNS_LOG_GENERIC, "couldn't create new file_ctx");
382  return NULL;
383  }
384 
385  if (SCConfLogOpenGeneric(conf, file_ctx, DEFAULT_LOG_FILENAME, 1) < 0) {
386  LogFileFreeCtx(file_ctx);
387  return NULL;
388  }
389 
390  LogDnsFileCtx *dnslog_ctx = SCMalloc(sizeof(LogDnsFileCtx));
391  if (unlikely(dnslog_ctx == NULL)) {
392  LogFileFreeCtx(file_ctx);
393  return NULL;
394  }
395  memset(dnslog_ctx, 0x00, sizeof(LogDnsFileCtx));
396 
397  dnslog_ctx->file_ctx = file_ctx;
398 
399  OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
400  if (unlikely(output_ctx == NULL)) {
401  LogFileFreeCtx(file_ctx);
402  SCFree(dnslog_ctx);
403  return NULL;
404  }
405 
406  output_ctx->data = dnslog_ctx;
407  output_ctx->DeInit = LogDnsLogDeInitCtx;
408 
409  SCLogDebug("DNS log output initialized");
410 
411  AppLayerParserRegisterLogger(IPPROTO_UDP, ALPROTO_DNS);
412  AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_DNS);
413 
414  return output_ctx;
415 }
