58 /**
59  * \brief Handle a packet and check if needs a threshold logic
60  * Also apply rule action if necessary.
61  *
62  * \param de_ctx Detection Context
63  * \param sig Signature pointer
64  * \param p Packet structure
65  *
66  * \retval 1 alert is not suppressed
67  * \retval 0 alert is suppressed
68  */
69 static int PacketAlertHandle(DetectEngineCtx *de_ctx, DetectEngineThreadCtx *det_ctx,
70  Signature *s, Packet *p, uint16_t pos)
71 {
72  SCEnter();
73  int ret = 1;
74  DetectThresholdData *td = NULL;
75  SigMatch *sm;
76 
77  if (!(PKT_IS_IPV4(p) || PKT_IS_IPV6(p))) {
78  SCReturnInt(1);
79  }
80 
81  /* handle suppressions first */
82  if (s->sm_lists[DETECT_SM_LIST_SUPPRESS] != NULL) {
83  KEYWORD_PROFILING_SET_LIST(det_ctx, DETECT_SM_LIST_SUPPRESS);
84  sm = NULL;
85  do {
86  td = SigGetThresholdTypeIter(s, p, &sm, DETECT_SM_LIST_SUPPRESS);
87  if (td != NULL) {
88  SCLogDebug("td %p", td);
89 
90  /* PacketAlertThreshold returns 2 if the alert is suppressed but
91  * we do need to apply rule actions to the packet. */
92  KEYWORD_PROFILING_START;
93  ret = PacketAlertThreshold(de_ctx, det_ctx, td, p, s);
94  if (ret == 0 || ret == 2) {
95  KEYWORD_PROFILING_END(det_ctx, DETECT_THRESHOLD, 0);
96  /* It doesn't match threshold, remove it */
97  SCReturnInt(ret);
98  }
99  KEYWORD_PROFILING_END(det_ctx, DETECT_THRESHOLD, 1);
100  }
101  } while (sm != NULL);
102  }
103 
104  /* if we're still here, consider thresholding */
105  if (s->sm_lists[DETECT_SM_LIST_THRESHOLD] != NULL) {
106  KEYWORD_PROFILING_SET_LIST(det_ctx, DETECT_SM_LIST_THRESHOLD);
107  sm = NULL;
108  do {
109  td = SigGetThresholdTypeIter(s, p, &sm, DETECT_SM_LIST_THRESHOLD);
110  if (td != NULL) {
111  SCLogDebug("td %p", td);
112 
113  /* PacketAlertThreshold returns 2 if the alert is suppressed but
114  * we do need to apply rule actions to the packet. */
115  KEYWORD_PROFILING_START;
116  ret = PacketAlertThreshold(de_ctx, det_ctx, td, p, s);
117  if (ret == 0 || ret == 2) {
118  KEYWORD_PROFILING_END(det_ctx, DETECT_THRESHOLD ,0);
119  /* It doesn't match threshold, remove it */
120  SCReturnInt(ret);
121  }
122  KEYWORD_PROFILING_END(det_ctx, DETECT_THRESHOLD, 1);
123  }
124  } while (sm != NULL);
125  }
126  SCReturnInt(1);
127 }

130 /**
131  * \brief Check if a certain sid alerted, this is used in the test functions
132  *
133  * \param p Packet on which we want to check if the signature alerted or not
134  * \param sid Signature id of the signature that thas to be checked for a match
135  *
136  * \retval match A value > 0 on a match; 0 on no match
137  */
138 int PacketAlertCheck(Packet *p, uint32_t sid)
139 {
140  uint16_t i = 0;
141  int match = 0;
142 
143  for (i = 0; i < p->alerts.cnt; i++) {
144  if (p->alerts.alerts[i].s == NULL)
145  continue;
146 
147  if (p->alerts.alerts[i].s->id == sid)
148  match++;
149  }
150 
151  return match;
152 }

154 /**
155  * \brief Remove alert from the p->alerts.alerts array at pos
156  * \param p Pointer to the Packet
157  * \param pos Position in the array
158  * \retval 0 if the number of alerts is less than pos
159  * 1 if all goes well
160  */
161 int PacketAlertRemove(Packet *p, uint16_t pos)
162 {
163  uint16_t i = 0;
164  int match = 0;
165 
166  if (pos > p->alerts.cnt) {
167  SCLogDebug("removing %u failed, pos > cnt %u", pos, p->alerts.cnt);
168  return 0;
169  }
170 
171  for (i = pos; i <= p->alerts.cnt - 1; i++) {
172  memcpy(&p->alerts.alerts[i], &p->alerts.alerts[i + 1], sizeof(PacketAlert));
173  }
174 
175  // Update it, since we removed 1
176  p->alerts.cnt--;
177 
178  return match;
179 }

181 /** \brief append a signature match to a packet
182  *
183  * \param det_ctx thread detection engine ctx
184  * \param s the signature that matched
185  * \param p packet
186  * \param flags alert flags
187  * \param alert_msg ptr to StreamMsg object that the signature matched on
188  */
189 int PacketAlertAppend(DetectEngineThreadCtx *det_ctx, Signature *s, Packet *p, uint64_t tx_id, uint8_t flags)
190 {
191  int i = 0;
192 
193  if (p->alerts.cnt == PACKET_ALERT_MAX)
194  return 0;
195 
196  SCLogDebug("sid %"PRIu32"", s->id);
197 
198  /* It should be usually the last, so check it before iterating */
199  if (p->alerts.cnt == 0 || (p->alerts.cnt > 0 &&
200  p->alerts.alerts[p->alerts.cnt - 1].num < s->num)) {
201  /* We just add it */
202  p->alerts.alerts[p->alerts.cnt].num = s->num;
203  p->alerts.alerts[p->alerts.cnt].action = s->action;
204  p->alerts.alerts[p->alerts.cnt].flags = flags;
205  p->alerts.alerts[p->alerts.cnt].s = s;
206  p->alerts.alerts[p->alerts.cnt].tx_id = tx_id;
207  } else {
208  /* We need to make room for this s->num
209  (a bit ugly with memcpy but we are planning changes here)*/
210  for (i = p->alerts.cnt - 1; i >= 0 && p->alerts.alerts[i].num > s->num; i--) {
211  memcpy(&p->alerts.alerts[i + 1], &p->alerts.alerts[i], sizeof(PacketAlert));
212  }
213 
214  i++; /* The right place to store the alert */
215 
216  p->alerts.alerts[i].num = s->num;
217  p->alerts.alerts[i].action = s->action;
218  p->alerts.alerts[i].flags = flags;
219  p->alerts.alerts[i].s = s;
220  p->alerts.alerts[i].tx_id = tx_id;
221  }
222 
223  /* Update the count */
224  p->alerts.cnt++;
225 
226  return 0;
227 }

229 /**
230  * \brief Check the threshold of the sigs that match, set actions, break on pass action
231  * This function iterate the packet alerts array, removing those that didn't match
232  * the threshold, and those that match after a signature with the action "pass".
233  * The array is sorted by action priority/order
234  * \param de_ctx detection engine context
235  * \param det_ctx detection engine thread context
236  * \param p pointer to the packet
237  */
238 void PacketAlertFinalize(DetectEngineCtx *de_ctx, DetectEngineThreadCtx *det_ctx, Packet *p)
239 {
240  SCEnter();
241  int i = 0;
242  Signature *s = NULL;
243  SigMatch *sm = NULL;
244 
245  while (i < p->alerts.cnt) {
246  SCLogDebug("Sig->num: %"PRIu16, p->alerts.alerts[i].num);
247  s = de_ctx->sig_array[p->alerts.alerts[i].num];
248 
249  int res = PacketAlertHandle(de_ctx, det_ctx, s, p, i);
250  if (res > 0) {
251  /* Now, if we have an alert, we have to check if we want
252  * to tag this session or src/dst host */
253  KEYWORD_PROFILING_SET_LIST(det_ctx, DETECT_SM_LIST_TMATCH);
254  sm = s->sm_lists[DETECT_SM_LIST_TMATCH];
255  while (sm) {
256  /* tags are set only for alerts */
257  KEYWORD_PROFILING_START;
258  sigmatch_table[sm->type].Match(NULL, det_ctx, p, s, sm->ctx);
259  KEYWORD_PROFILING_END(det_ctx, sm->type, 1);
260  sm = sm->next;
261  }
262 
263  if (s->flags & SIG_FLAG_IPONLY) {
264  if (((p->flowflags & FLOW_PKT_TOSERVER) && !(p->flowflags & FLOW_PKT_TOSERVER_IPONLY_SET)) ||
265  ((p->flowflags & FLOW_PKT_TOCLIENT) && !(p->flowflags & FLOW_PKT_TOCLIENT_IPONLY_SET))) {
266  SCLogDebug("testing against \"ip-only\" signatures");
267 
268  if (p->flow != NULL) {
269  /* Update flow flags for iponly */
270  FLOWLOCK_WRLOCK(p->flow);
271  FlowSetIPOnlyFlagNoLock(p->flow, p->flowflags & FLOW_PKT_TOSERVER ? 1 : 0);
272 
273  if (s->action & ACTION_DROP)
274  p->flow->flags |= FLOW_ACTION_DROP;
275  if (s->action & ACTION_REJECT)
276  p->flow->flags |= FLOW_ACTION_DROP;
277  if (s->action & ACTION_REJECT_DST)
278  p->flow->flags |= FLOW_ACTION_DROP;
279  if (s->action & ACTION_REJECT_BOTH)
280  p->flow->flags |= FLOW_ACTION_DROP;
281  if (s->action & ACTION_PASS) {
282  FlowSetNoPacketInspectionFlag(p->flow);
283  }
284  FLOWLOCK_UNLOCK(p->flow);
285  }
286  }
287  }
288 
289  /* set actions on packet */
290  DetectSignatureApplyActions(p, p->alerts.alerts[i].s);
291 
292  if (PACKET_TEST_ACTION(p, ACTION_PASS)) {
293  /* Ok, reset the alert cnt to end in the previous of pass
294  * so we ignore the rest with less prio */
295  p->alerts.cnt = i;
296 
297  /* if an stream/app-layer match we enforce the pass for the flow */
298  if ((p->flow != NULL) &&
299  (p->alerts.alerts[i].flags &
300  (PACKET_ALERT_FLAG_STATE_MATCH|PACKET_ALERT_FLAG_STREAM_MATCH)))
301  {
302  FlowLockSetNoPacketInspectionFlag(p->flow);
303  }
304  break;
305 
306  /* if the signature wants to drop, check if the
307  * PACKET_ALERT_FLAG_DROP_FLOW flag is set. */
308  } else if ((PACKET_TEST_ACTION(p, ACTION_DROP)) &&
309  ((p->alerts.alerts[i].flags & PACKET_ALERT_FLAG_DROP_FLOW) ||
310  (s->flags & SIG_FLAG_APPLAYER))
311  && p->flow != NULL)
312  {
313  FLOWLOCK_WRLOCK(p->flow);
314  /* This will apply only on IPS mode (check StreamTcpPacket) */
315  p->flow->flags |= FLOW_ACTION_DROP;
316  FLOWLOCK_UNLOCK(p->flow);
317  }
318  }
319 
320  /* Thresholding removes this alert */
321  if (res == 0 || res == 2) {
322  PacketAlertRemove(p, i);
323 
324  if (p->alerts.cnt == 0)
325  break;
326  } else {
327  i++;
328  }
329  }
330 
331  /* At this point, we should have all the new alerts. Now check the tag
332  * keyword context for sessions and hosts */
333  if (!(p->flags & PKT_PSEUDO_STREAM_END))
334  TagHandlePacket(de_ctx, det_ctx, p);
335 }
