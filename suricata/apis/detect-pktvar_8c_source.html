100 static int DetectPktvarSetup (DetectEngineCtx *de_ctx, Signature *s, char *rawstr)
101 {
102  DetectPktvarData *cd = NULL;
103  SigMatch *sm = NULL;
104  char *str = rawstr;
105  char dubbed = 0;
106  uint16_t len;
107  char *varname = NULL, *varcontent = NULL;
108 #define MAX_SUBSTRINGS 30
109  int ret = 0, res = 0;
110  int ov[MAX_SUBSTRINGS];
111 
112  ret = pcre_exec(parse_regex, parse_regex_study, rawstr, strlen(rawstr), 0, 0, ov, MAX_SUBSTRINGS);
113  if (ret != 3) {
114  SCLogError(SC_ERR_PCRE_MATCH, "\"%s\" is not a valid setting for pktvar.", rawstr);
115  return -1;
116 
117  }
118 
119  const char *str_ptr;
120  res = pcre_get_substring((char *)rawstr, ov, MAX_SUBSTRINGS, 1, &str_ptr);
121  if (res < 0) {
122  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
123  return -1;
124  }
125  varname = (char *)str_ptr;
126 
127  if (ret > 2) {
128  res = pcre_get_substring((char *)rawstr, ov, MAX_SUBSTRINGS, 2, &str_ptr);
129  if (res < 0) {
130  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
131  return -1;
132  }
133  varcontent = (char *)str_ptr;
134  }
135 
136  SCLogDebug("varname %s, varcontent %s", varname, varcontent);
137 
138  if (varcontent[0] == '\"' && varcontent[strlen(varcontent)-1] == '\"') {
139  str = SCStrdup(varcontent+1);
140  if (unlikely(str == NULL)) {
141  return -1;
142  }
143  str[strlen(varcontent)-2] = '\0';
144  dubbed = 1;
145  }
146 
147  len = strlen(str);
148  if (len == 0) {
149  if (dubbed) SCFree(str);
150  return -1;
151  }
152 
153  cd = SCMalloc(sizeof(DetectPktvarData));
154  if (unlikely(cd == NULL))
155  goto error;
156 
157  char converted = 0;
158 
159  {
160  uint16_t i, x;
161  uint8_t bin = 0, binstr[3] = "", binpos = 0;
162  for (i = 0, x = 0; i < len; i++) {
163  // printf("str[%02u]: %c\n", i, str[i]);
164  if (str[i] == '|') {
165  if (bin) {
166  bin = 0;
167  } else {
168  bin = 1;
169  }
170  } else {
171  if (bin) {
172  if (isdigit((unsigned char)str[i]) ||
173  str[i] == 'A' || str[i] == 'a' ||
174  str[i] == 'B' || str[i] == 'b' ||
175  str[i] == 'C' || str[i] == 'c' ||
176  str[i] == 'D' || str[i] == 'd' ||
177  str[i] == 'E' || str[i] == 'e' ||
178  str[i] == 'F' || str[i] == 'f') {
179  // printf("part of binary: %c\n", str[i]);
180 
181  binstr[binpos] = (char)str[i];
182  binpos++;
183 
184  if (binpos == 2) {
185  uint8_t c = strtol((char *)binstr, (char **) NULL, 16) & 0xFF;
186  binpos = 0;
187  str[x] = c;
188  x++;
189  converted = 1;
190  }
191  } else if (str[i] == ' ') {
192  // printf("space as part of binary string\n");
193  }
194  } else {
195  str[x] = str[i];
196  x++;
197  }
198  }
199  }
200 #ifdef DEBUG
201  if (SCLogDebugEnabled()) {
202  for (i = 0; i < x; i++) {
203  if (isprint((unsigned char)str[i])) printf("%c", str[i]);
204  else printf("\\x%02u", str[i]);
205  }
206  printf("\n");
207  }
208 #endif
209 
210  if (converted)
211  len = x;
212  }
213 
214  cd->content = SCMalloc(len);
215  if (cd->content == NULL) {
216  SCFree(cd);
217  if (dubbed) SCFree(str);
218  return -1;
219  }
220 
221  cd->name = SCStrdup(varname);
222  if (cd->name == NULL) {
223  SCFree(cd);
224  if (dubbed) SCFree(str);
225  return -1;
226  }
227 
228  memcpy(cd->content, str, len);
229  cd->content_len = len;
230  cd->flags = 0;
231 
232  /* Okay so far so good, lets get this into a SigMatch
233  * and put it in the Signature. */
234  sm = SigMatchAlloc();
235  if (sm == NULL)
236  goto error;
237 
238  sm->type = DETECT_PKTVAR;
239  sm->ctx = (SigMatchCtx *)cd;
240 
241  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);
242 
243  if (dubbed) SCFree(str);
244  return 0;
245 
246 error:
247  if (dubbed)
248  SCFree(str);
249  if (cd) {
250  if (cd->name)
251  SCFree(cd->name);
252  SCFree(cd);
253  }
254  if (sm)
255  SCFree(sm);
256  return -1;
257 }
