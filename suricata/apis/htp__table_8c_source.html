43 static htp_status_t _htp_table_add(htp_table_t *table, const bstr *key, const void *element) {
44  // Add key.
45  if (htp_list_add(table->list, (void *)key) != HTP_OK) return HTP_ERROR;
46 
47  // Add element.
48  if (htp_list_add(table->list, (void *)element) != HTP_OK) {
49  htp_list_pop(table->list);
50  return HTP_ERROR;
51  }
52 
53  return HTP_OK;
54 }

56 htp_status_t htp_table_add(htp_table_t *table, const bstr *key, const void *element) {
57  if ((table == NULL)||(key == NULL)) return HTP_ERROR;
58 
59  // Keep track of how keys are allocated, and
60  // ensure that all invocations are consistent.
61  if (table->alloc_type == HTP_TABLE_KEYS_ALLOC_UKNOWN) {
62  table->alloc_type = HTP_TABLE_KEYS_COPIED;
63  } else {
64  if (table->alloc_type != HTP_TABLE_KEYS_COPIED) {
65  #ifdef HTP_DEBUG
66  fprintf(stderr, "# Inconsistent key management strategy. Actual %d. Attempted %d.\n",
67  table->alloc_type, HTP_TABLE_KEYS_COPIED);
68  #endif
69 
70  return HTP_ERROR;
71  }
72  }
73 
74  bstr *dupkey = bstr_dup(key);
75  if (dupkey == NULL) return HTP_ERROR;
76 
77  if (_htp_table_add(table, dupkey, element) != HTP_OK) {
78  bstr_free(dupkey);
79  return HTP_ERROR;
80  }
81 
82  return HTP_OK;
83 }

127 void htp_table_clear(htp_table_t *table) {
128  if (table == NULL) return;
129 
130  // Free the table keys, but only if we're managing them.
131  if ((table->alloc_type == HTP_TABLE_KEYS_COPIED)||(table->alloc_type == HTP_TABLE_KEYS_ADOPTED)) {
132  bstr *key = NULL;
133  for (size_t i = 0, n = htp_list_size(table->list); i < n; i += 2) {
134  key = htp_list_get(table->list, i);
135  bstr_free(key);
136  }
137  }
138 
139  htp_list_clear(table->list);
140 }

150 htp_table_t *htp_table_create(size_t size) {
151  if (size == 0) return NULL;
152 
153  htp_table_t *table = calloc(1, sizeof (htp_table_t));
154  if (table == NULL) return NULL;
155 
156  table->alloc_type = HTP_TABLE_KEYS_ALLOC_UKNOWN;
157 
158  // Use a list behind the scenes.
159  table->list = htp_list_array_create(size * 2);
160  if (table->list == NULL) {
161  free(table);
162  return NULL;
163  }
164

189 void *htp_table_get(const htp_table_t *table, const bstr *key) {
190  if ((table == NULL)||(key == NULL)) return NULL;
191 
192  // Iterate through the list, comparing
193  // keys with the parameter, return data if found.
194  for (size_t i = 0, n = htp_list_size(table->list); i < n; i += 2) {
195  bstr *key_candidate = htp_list_get(table->list, i);
196  void *element = htp_list_get(table->list, i + 1);
197  if (bstr_cmp_nocase(key_candidate, key) == 0) {
198  return element;
199  }
200  }
201 
202  return NULL;
203 }
