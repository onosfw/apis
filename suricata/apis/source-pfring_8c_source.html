199 static inline void PfringDumpCounters(PfringThreadVars *ptv)
200 {
201  pfring_stat pfring_s;
202  if (likely((pfring_stats(ptv->pd, &pfring_s) >= 0))) {
203  /* pfring counter is per socket and is not cleared after read.
204  * So to get the number of packet on the interface we can add
205  * the newly seen packets and drops for this thread and add it
206  * to the interface counter */
207  uint64_t th_pkts = StatsGetLocalCounterValue(ptv->tv, ptv->capture_kernel_packets);
208  uint64_t th_drops = StatsGetLocalCounterValue(ptv->tv, ptv->capture_kernel_drops);
209  SC_ATOMIC_ADD(ptv->livedev->pkts, pfring_s.recv - th_pkts);
210  SC_ATOMIC_ADD(ptv->livedev->drop, pfring_s.drop - th_drops);
211  StatsSetUI64(ptv->tv, ptv->capture_kernel_packets, pfring_s.recv);
212  StatsSetUI64(ptv->tv, ptv->capture_kernel_drops, pfring_s.drop);
213  }
214 }

216 /**
217  * \brief Pfring Packet Process function.
218  *
219  * This function fills in our packet structure from libpfring.
220  * From here the packets are picked up by the DecodePfring thread.
221  *
222  * \param user pointer to PfringThreadVars
223  * \param h pointer to pfring packet header
224  * \param p pointer to the current packet
225  */
226 static inline void PfringProcessPacket(void *user, struct pfring_pkthdr *h, Packet *p)
227 {
228 
229  PfringThreadVars *ptv = (PfringThreadVars *)user;
230 
231  ptv->bytes += h->caplen;
232  ptv->pkts++;
233  p->livedev = ptv->livedev;
234 
235  /* PF_RING may fail to set timestamp */
236  if (h->ts.tv_sec == 0) {
237  gettimeofday((struct timeval *)&h->ts, NULL);
238  }
239 
240  p->ts.tv_sec = h->ts.tv_sec;
241  p->ts.tv_usec = h->ts.tv_usec;
242 
243  /* PF_RING all packets are marked as a link type of ethernet
244  * so that is what we do here. */
245  p->datalink = LINKTYPE_ETHERNET;
246 
247  /* get vlan id from header. Check on vlan_id not null even if comment in
248  * header announce NO_VLAN is used when there is no VLAN. But NO_VLAN
249  * is not defined nor used in PF_RING code. And vlan_id is set to 0
250  * in PF_RING kernel code when there is no VLAN. */
251  if ((!ptv->vlan_disabled) && h->extended_hdr.parsed_pkt.vlan_id) {
252  p->vlan_id[0] = h->extended_hdr.parsed_pkt.vlan_id;
253  p->vlan_idx = 1;
254  p->vlanh[0] = NULL;
255  }
256 
257  switch (ptv->checksum_mode) {
258  case CHECKSUM_VALIDATION_RXONLY:
259  if (h->extended_hdr.rx_direction == 0) {
260  p->flags |= PKT_IGNORE_CHECKSUM;
261  }
262  break;
263  case CHECKSUM_VALIDATION_DISABLE:
264  p->flags |= PKT_IGNORE_CHECKSUM;
265  break;
266  case CHECKSUM_VALIDATION_AUTO:
267  if (ptv->livedev->ignore_checksum) {
268  p->flags |= PKT_IGNORE_CHECKSUM;
269  } else if (ChecksumAutoModeCheck(ptv->pkts,
270  SC_ATOMIC_GET(ptv->livedev->pkts),
271  SC_ATOMIC_GET(ptv->livedev->invalid_checksums))) {
272  ptv->livedev->ignore_checksum = 1;
273  p->flags |= PKT_IGNORE_CHECKSUM;
274  }
275  break;
276  default:
277  break;
278  }
279 
280  SET_PKT_LEN(p, h->caplen);
281 }

283 /**
284  * \brief Recieves packets from an interface via libpfring.
285  *
286  * This function recieves packets from an interface and passes
287  * the packet on to the pfring callback function.
288  *
289  * \param tv pointer to ThreadVars
290  * \param data pointer that gets cast into PfringThreadVars for ptv
291  * \param slot slot containing task information
292  * \retval TM_ECODE_OK on success
293  * \retval TM_ECODE_FAILED on failure
294  */
295 TmEcode ReceivePfringLoop(ThreadVars *tv, void *data, void *slot)
296 {
297  SCEnter();
298 
299  PfringThreadVars *ptv = (PfringThreadVars *)data;
300  Packet *p = NULL;
301  struct pfring_pkthdr hdr;
302  TmSlot *s = (TmSlot *)slot;
303  time_t last_dump = 0;
304  u_int buffer_size;
305  u_char *pkt_buffer;
306 
307  ptv->slot = s->slot_next;
308 
309  /* we have to enable the ring here as we need to do it after all
310  * the threads have called pfring_set_cluster(). */
311  int rc = pfring_enable_ring(ptv->pd);
312  if (rc != 0) {
313  SCLogError(SC_ERR_PF_RING_OPEN, "pfring_enable_ring failed returned %d ", rc);
314  SCReturnInt(TM_ECODE_FAILED);
315  }
316 
317  while(1) {
318  if (suricata_ctl_flags & (SURICATA_STOP | SURICATA_KILL)) {
319  SCReturnInt(TM_ECODE_OK);
320  }
321 
322  /* make sure we have at least one packet in the packet pool, to prevent
323  * us from alloc'ing packets at line rate */
324  PacketPoolWait();
325 
326  p = PacketGetFromQueueOrAlloc();
327  if (p == NULL) {
328  SCReturnInt(TM_ECODE_FAILED);
329  }
330  PKT_SET_SRC(p, PKT_SRC_WIRE);
331 
332  /* Some flavours of PF_RING may fail to set timestamp - see PF-RING-enabled libpcap code*/
333  hdr.ts.tv_sec = hdr.ts.tv_usec = 0;
334 
335  /* Check for Zero-copy mode */
336  if (ptv->flags & PFRING_FLAGS_ZERO_COPY) {
337  buffer_size = 0;
338  pkt_buffer = NULL;
339  } else {
340  buffer_size = GET_PKT_DIRECT_MAX_SIZE(p);
341  pkt_buffer = GET_PKT_DIRECT_DATA(p);
342  }
343 
344  /* Depending on what compile time options are used for pfring we either return 0 or -1 on error and always 1 for success */
345  int r = pfring_recv(ptv->pd, &pkt_buffer,
346  buffer_size,
347  &hdr,
348  LIBPFRING_WAIT_FOR_INCOMING);
349 
350  /* Check for Zero-copy mode */
351  if (ptv->flags & PFRING_FLAGS_ZERO_COPY) {
352  PacketSetData(p, pkt_buffer, hdr.caplen);
353  }
354 
355  if (r == 1) {
356  //printf("RecievePfring src %" PRIu32 " sport %" PRIu32 " dst %" PRIu32 " dstport %" PRIu32 "\n",
357  // hdr.parsed_pkt.ipv4_src,hdr.parsed_pkt.l4_src_port, hdr.parsed_pkt.ipv4_dst,hdr.parsed_pkt.l4_dst_port);
358 
359  PfringProcessPacket(ptv, &hdr, p);
360 
361  if (TmThreadsSlotProcessPkt(ptv->tv, ptv->slot, p) != TM_ECODE_OK) {
362  TmqhOutputPacketpool(ptv->tv, p);
363  SCReturnInt(TM_ECODE_FAILED);
364  }
365 
366  /* Trigger one dump of stats every second */
367  if (p->ts.tv_sec != last_dump) {
368  PfringDumpCounters(ptv);
369  last_dump = p->ts.tv_sec;
370  }
371  } else {
372  SCLogError(SC_ERR_PF_RING_RECV,"pfring_recv error %" PRId32 "", r);
373  TmqhOutputPacketpool(ptv->tv, p);
374  SCReturnInt(TM_ECODE_FAILED);
375  }
376  StatsSyncCountersIfSignalled(tv);
377  }
378 
379  return TM_ECODE_OK;
380 }

382 /**
383  * \brief Init function for RecievePfring.
384  *
385  * This is a setup function for recieving packets
386  * via libpfring.
387  *
388  * \param tv pointer to ThreadVars
389  * \param initdata pointer to the interface passed from the user
390  * \param data pointer gets populated with PfringThreadVars
391  * \todo add a config option for setting cluster id
392  * \todo Create a general pfring setup function.
393  * \retval TM_ECODE_OK on success
394  * \retval TM_ECODE_FAILED on error
395  */
396 TmEcode ReceivePfringThreadInit(ThreadVars *tv, void *initdata, void **data)
397 {
398  int rc;
399  u_int32_t version = 0;
400  PfringIfaceConfig *pfconf = (PfringIfaceConfig *) initdata;
401  unsigned int opflag;
402  char const *active_runmode = RunmodeGetActive();
403 
404  if (pfconf == NULL)
405  return TM_ECODE_FAILED;
406 
407  PfringThreadVars *ptv = SCMalloc(sizeof(PfringThreadVars));
408  if (unlikely(ptv == NULL)) {
409  pfconf->DerefFunc(pfconf);
410  return TM_ECODE_FAILED;
411  }
412  memset(ptv, 0, sizeof(PfringThreadVars));
413 
414  ptv->tv = tv;
415  ptv->threads = 1;
416 
417  ptv->interface = SCStrdup(pfconf->iface);
418  if (unlikely(ptv->interface == NULL)) {
419  SCLogError(SC_ERR_MEM_ALLOC, "Unable to allocate device string");
420  SCFree(ptv);
421  SCReturnInt(TM_ECODE_FAILED);
422  }
423 
424  ptv->livedev = LiveGetDevice(pfconf->iface);
425  if (ptv->livedev == NULL) {
426  SCLogError(SC_ERR_INVALID_VALUE, "Unable to find Live device");
427  SCFree(ptv);
428  SCReturnInt(TM_ECODE_FAILED);
429  }
430 
431  /* enable zero-copy mode for workers runmode */
432  if (active_runmode && strcmp("workers", active_runmode) == 0) {
433  ptv->flags |= PFRING_FLAGS_ZERO_COPY;
434  SCLogInfo("Enabling zero-copy for %s", ptv->interface);
435  }
436 
437  ptv->checksum_mode = pfconf->checksum_mode;
438 
439  opflag = PF_RING_PROMISC;
440 
441  /* if suri uses VLAN and if we have a recent kernel, we need
442  * to use parsed_pkt to get VLAN info */
443  if ((! ptv->vlan_disabled) && SCKernelVersionIsAtLeast(3, 0)) {
444  opflag |= PF_RING_LONG_HEADER;
445  }
446 
447  if (ptv->checksum_mode == CHECKSUM_VALIDATION_RXONLY) {
448  if (strncmp(ptv->interface, "dna", 3) == 0) {
449  SCLogWarning(SC_ERR_INVALID_VALUE,
450  "Can't use rxonly checksum-checks on DNA interface,"
451  " resetting to auto");
452  ptv->checksum_mode = CHECKSUM_VALIDATION_AUTO;
453  } else {
454  opflag |= PF_RING_LONG_HEADER;
455  }
456  }
457 
458  ptv->pd = pfring_open(ptv->interface, (uint32_t)default_packet_size, opflag);
459  if (ptv->pd == NULL) {
460  SCLogError(SC_ERR_PF_RING_OPEN,"Failed to open %s: pfring_open error."
461  " Check if %s exists and pf_ring module is loaded.",
462  ptv->interface,
463  ptv->interface);
464  pfconf->DerefFunc(pfconf);
465  SCFree(ptv);
466  return TM_ECODE_FAILED;
467  } else {
468  pfring_set_application_name(ptv->pd, PROG_NAME);
469  pfring_version(ptv->pd, &version);
470  }
471 
472  /* We only set cluster info if the number of pfring threads is greater than 1 */
473  ptv->threads = pfconf->threads;
474 
475  ptv->cluster_id = pfconf->cluster_id;
476 
477  if ((ptv->threads == 1) && (strncmp(ptv->interface, "dna", 3) == 0)) {
478  SCLogInfo("DNA interface detected, not adding thread to cluster");
479  } else if (strncmp(ptv->interface, "zc", 2) == 0) {
480  SCLogInfo("ZC interface detected, not adding thread to cluster");
481  } else {
482  ptv->ctype = pfconf->ctype;
483  rc = pfring_set_cluster(ptv->pd, ptv->cluster_id, ptv->ctype);
484 
485  if (rc != 0) {
486  SCLogError(SC_ERR_PF_RING_SET_CLUSTER_FAILED, "pfring_set_cluster "
487  "returned %d for cluster-id: %d", rc, ptv->cluster_id);
488  if (rc != PF_RING_ERROR_NOT_SUPPORTED || (pfconf->flags & PFRING_CONF_FLAGS_CLUSTER)) {
489  /* cluster is mandatory as explicitly specified in the configuration */
490  pfconf->DerefFunc(pfconf);
491  return TM_ECODE_FAILED;
492  }
493  }
494  }
495 
496  if (ptv->threads > 1) {
497  SCLogInfo("(%s) Using PF_RING v.%d.%d.%d, interface %s, cluster-id %d",
498  tv->name, (version & 0xFFFF0000) >> 16, (version & 0x0000FF00) >> 8,
499  version & 0x000000FF, ptv->interface, ptv->cluster_id);
500  } else {
501  SCLogInfo("(%s) Using PF_RING v.%d.%d.%d, interface %s, cluster-id %d, single-pfring-thread",
502  tv->name, (version & 0xFFFF0000) >> 16, (version & 0x0000FF00) >> 8,
503  version & 0x000000FF, ptv->interface, ptv->cluster_id);
504  }
505 
506  if (pfconf->bpf_filter) {
507  ptv->bpf_filter = SCStrdup(pfconf->bpf_filter);
508  if (unlikely(ptv->bpf_filter == NULL)) {
509  SCLogError(SC_ERR_MEM_ALLOC, "Set PF_RING bpf filter failed.");
510  } else {
511  SCMutexLock(&pfring_bpf_set_filter_lock);
512  rc = pfring_set_bpf_filter(ptv->pd, ptv->bpf_filter);
513  SCMutexUnlock(&pfring_bpf_set_filter_lock);
514 
515  if (rc < 0) {
516  SCLogInfo("Set PF_RING bpf filter \"%s\" failed.",
517  ptv->bpf_filter);
518  }
519  }
520  }
521 
522  ptv->capture_kernel_packets = StatsRegisterCounter("capture.kernel_packets",
523  ptv->tv);
524  ptv->capture_kernel_drops = StatsRegisterCounter("capture.kernel_drops",
525  ptv->tv);
526 
527  /* A bit strange to have this here but we only have vlan information
528  * during reading so we need to know if we want to keep vlan during
529  * the capture phase */
530  int vlanbool = 0;
531  if ((ConfGetBool("vlan.use-for-tracking", &vlanbool)) == 1 && vlanbool == 0) {
532  ptv->vlan_disabled = 1;
533  }
534 
535  /* If kernel is older than 3.8, VLAN is not stripped so we don't
536  * get the info from packt extended header but we will use a standard
537  * parsing */
538  if (! SCKernelVersionIsAtLeast(3, 0)) {
539  ptv->vlan_disabled = 1;
540  }
541 
542  /* If VLAN tracking is disabled, set cluster type to 5-tuple or in case of a
543  * ZC interface, do nothing */
544  if (ptv->vlan_disabled && ptv->ctype == CLUSTER_FLOW &&
545  strncmp(ptv->interface, "zc", 2) != 0) {
546  SCLogInfo("VLAN disabled, setting cluster type to CLUSTER_FLOW_5_TUPLE");
547  rc = pfring_set_cluster(ptv->pd, ptv->cluster_id, CLUSTER_FLOW_5_TUPLE);
548 
549  if (rc != 0) {
550  SCLogError(SC_ERR_PF_RING_SET_CLUSTER_FAILED, "pfring_set_cluster "
551  "returned %d for cluster-id: %d", rc, ptv->cluster_id);
552  pfconf->DerefFunc(pfconf);
553  return TM_ECODE_FAILED;
554  }
555  }
556 
557  *data = (void *)ptv;
558  pfconf->DerefFunc(pfconf);
559 
560  return TM_ECODE_OK;
561 }

564  * \brief This function prints stats to the screen at exit.
565  * \param tv pointer to ThreadVars
