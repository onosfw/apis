140 /**
141  * \brief NFLOG callback function
142  * This function setup a packet from a nflog message
143  */
144 static int NFLOGCallback(struct nflog_g_handle *gh, struct nfgenmsg *msg,
145  struct nflog_data *nfa, void *data)
146 {
147  NFLOGThreadVars *ntv = (NFLOGThreadVars *) data;
148  struct nfulnl_msg_packet_hdr *ph;
149  char *payload;
150  int ret;
151 
152  /* grab a packet*/
153  Packet *p = PacketGetFromQueueOrAlloc();
154  if (p == NULL)
155  return -1;
156 
157  PKT_SET_SRC(p, PKT_SRC_WIRE);
158 
159  ph = nflog_get_msg_packet_hdr(nfa);
160  if (ph != NULL) {
161  p->nflog_v.hw_protocol = ph->hw_protocol;
162  }
163 
164  p->nflog_v.ifi = nflog_get_indev(nfa);
165  p->nflog_v.ifo = nflog_get_outdev(nfa);
166 
167  ret = nflog_get_payload(nfa, &payload);
168 
169  if (ret > 0) {
170  if (ret > 65536) {
171  SCLogWarning(SC_ERR_INVALID_ARGUMENTS, "NFLOG sent too big packet");
172  SET_PKT_LEN(p, 0);
173  } else if (runmode_workers)
174  PacketSetData(p, (uint8_t *)payload, ret);
175  else
176  PacketCopyData(p, (uint8_t *)payload, ret);
177  } else if (ret == -1)
178  SET_PKT_LEN(p, 0);
179 
180  ret = nflog_get_timestamp(nfa, &p->ts);
181  if (ret != 0) {
182  memset(&p->ts, 0, sizeof(struct timeval));
183  gettimeofday(&p->ts, NULL);
184  }
185 
186  p->datalink = DLT_RAW;
187 
188 #ifdef COUNTERS
189  ntv->pkts++;
190  ntv->bytes += GET_PKT_LEN(p);
191 #endif
192  (void) SC_ATOMIC_ADD(ntv->livedev->pkts, 1);
193 
194  if (TmThreadsSlotProcessPkt(ntv->tv, ntv->slot, p) != TM_ECODE_OK) {
195  TmqhOutputPacketpool(ntv->tv, p);
196  return -1;
197  }
198 
199  return 0;
200 }

202 /**
203  * \brief Receives packet from a nflog group via libnetfilter_log
204  * This is a setup function for recieving packets via libnetfilter_log.
205  * \param tv pointer to ThreadVars
206  * \param initdata pointer to the group passed from the user
207  * \param data pointer gets populated with NFLOGThreadVars
208  * \retvalTM_ECODE_OK on success
209  * \retval TM_ECODE_FAILED on error
210  */
211 TmEcode ReceiveNFLOGThreadInit(ThreadVars *tv, void *initdata, void **data)
212 {
213  NflogGroupConfig *nflconfig = initdata;
214 
215  if (initdata == NULL) {
216  SCLogError(SC_ERR_INVALID_ARGUMENT, "initdata == NULL");
217  SCReturnInt(TM_ECODE_FAILED);
218  }
219 
220  NFLOGThreadVars *ntv = SCMalloc(sizeof(NFLOGThreadVars));
221  if (unlikely(ntv == NULL)) {
222  nflconfig->DerefFunc(nflconfig);
223  SCReturnInt(TM_ECODE_FAILED);
224  }
225  memset(ntv, 0, sizeof(NFLOGThreadVars));
226 
227  ntv->tv = tv;
228  ntv->group = nflconfig->group;
229  ntv->nlbufsiz = nflconfig->nlbufsiz;
230  ntv->nlbufsiz_max = nflconfig->nlbufsiz_max;
231  ntv->qthreshold = nflconfig->qthreshold;
232  ntv->qtimeout = nflconfig->qtimeout;
233  ntv->nful_overrun_warned = nflconfig->nful_overrun_warned;
234 
235  ntv->h = nflog_open();
236  if (ntv->h == NULL) {
237  SCLogError(SC_ERR_NFLOG_OPEN, "nflog_open() failed");
238  SCFree(ntv);
239  return TM_ECODE_FAILED;
240  }
241 
242  SCLogDebug("binding netfilter_log as nflog handler for AF_INET and AF_INET6");
243 
244  if (nflog_bind_pf(ntv->h, AF_INET) < 0) {
245  SCLogError(SC_ERR_NFLOG_BIND, "nflog_bind_pf() for AF_INET failed");
246  exit(EXIT_FAILURE);
247  }
248  if (nflog_bind_pf(ntv->h, AF_INET6) < 0) {
249  SCLogError(SC_ERR_NFLOG_BIND, "nflog_bind_pf() for AF_INET6 failed");
250  exit(EXIT_FAILURE);
251  }
252 
253  ntv->gh = nflog_bind_group(ntv->h, ntv->group);
254  if (!ntv->gh) {
255  SCLogError(SC_ERR_NFLOG_OPEN, "nflog_bind_group() failed");
256  SCFree(ntv);
257  return TM_ECODE_FAILED;
258  }
259 
260  if (nflog_set_mode(ntv->gh, NFULNL_COPY_PACKET, 0xFFFF) < 0) {
261  SCLogError(SC_ERR_NFLOG_SET_MODE, "can't set packet_copy mode");
262  SCFree(ntv);
263  return TM_ECODE_FAILED;
264  }
265 
266  nflog_callback_register(ntv->gh, &NFLOGCallback, (void *)ntv);
267 
268  if (ntv->nlbufsiz < ntv->nlbufsiz_max)
269  ntv->nlbufsiz = nfnl_rcvbufsiz(nflog_nfnlh(ntv->h), ntv->nlbufsiz);
270  else {
271  SCLogError(SC_ERR_NFLOG_MAX_BUFSIZ, "Maximum buffer size (%d) in NFLOG "
272  "has been reached", ntv->nlbufsiz);
273  return TM_ECODE_FAILED;
274  }
275 
276  if (nflog_set_qthresh(ntv->gh, ntv->qthreshold) >= 0)
277  SCLogDebug("NFLOG netlink queue threshold has been set to %d",
278  ntv->qthreshold);
279  else
280  SCLogDebug("NFLOG netlink queue threshold can't be set to %d",
281  ntv->qthreshold);
282 
283  if (nflog_set_timeout(ntv->gh, ntv->qtimeout) >= 0)
284  SCLogDebug("NFLOG netlink queue timeout has been set to %d",
285  ntv->qtimeout);
286  else
287  SCLogDebug("NFLOG netlink queue timeout can't be set to %d",
288  ntv->qtimeout);
289 
290  ntv->livedev = LiveGetDevice(nflconfig->numgroup);
291  if (ntv->livedev == NULL) {
292  SCLogError(SC_ERR_INVALID_VALUE, "Unable to find Live device");
293  SCFree(ntv);
294  SCReturnInt(TM_ECODE_FAILED);
295  }
296 
297  /* set a timeout to the socket so we can check for a signal
298  * in case we don't get packets for a longer period. */
299  struct timeval timev;
300  timev.tv_sec = 1;
301  timev.tv_usec = 0;
302 
303  int fd = nflog_fd(ntv->h);
304  if (setsockopt(fd, SOL_SOCKET, SO_RCVTIMEO, &timev, sizeof(timev)) == -1) {
305  SCLogWarning(SC_WARN_NFLOG_SETSOCKOPT, "can't set socket "
306  "timeout: %s", strerror(errno));
307  }
308 
309 #ifdef PACKET_STATISTICS
310  ntv->capture_kernel_packets = StatsRegisterCounter("capture.kernel_packets",
311  ntv->tv);
312  ntv->capture_kernel_drops = StatsRegisterCounter("capture.kernel_drops",
313  ntv->tv);
314 #endif
315 
316  char *active_runmode = RunmodeGetActive();
317  if (active_runmode && !strcmp("workers", active_runmode))
318  runmode_workers = 1;
319  else
320  runmode_workers = 0;
321 
322 #define T_DATA_SIZE 70000
323  ntv->data = SCMalloc(T_DATA_SIZE);
324  if (ntv->data == NULL) {
325  nflconfig->DerefFunc(nflconfig);
326  SCFree(ntv);
327  SCReturnInt(TM_ECODE_FAILED);
328  }
329 
330  ntv->datalen = T_DATA_SIZE;
331 #undef T_DATA_SIZE
332 
333  *data = (void *)ntv;
334 
335  nflconfig->DerefFunc(nflconfig);
336  SCReturnInt(TM_ECODE_OK);
337 }

401 /**
402  * \brief Recieves packets from a group via libnetfilter_log.
403  *
404  * This function recieves packets from a group and passes
405  * the packet on to the nflog callback function.
406  *
407  * \param tv pointer to ThreadVars
408  * \param data pointer that gets cast into NFLOGThreadVars
409  * \param slot slot containing task information
410  * \retval TM_ECODE_OK on success
411  * \retval TM_ECODE_FAILED on failure
412  */
413 TmEcode ReceiveNFLOGLoop(ThreadVars *tv, void *data, void *slot)
414 {
415  SCEnter();
416  NFLOGThreadVars *ntv = (NFLOGThreadVars *)data;
417  int rv, fd;
418  int ret = -1;
419 
420  ntv->slot = ((TmSlot *) slot)->slot_next;
421 
422  fd = nflog_fd(ntv->h);
423  if (fd < 0) {
424  SCLogError(SC_ERR_NFLOG_FD, "Can't obtain a file descriptor");
425  SCReturnInt(TM_ECODE_FAILED);
426  }
427 
428  while (1) {
429  if (suricata_ctl_flags != 0)
430  break;
431 
432  rv = recv(fd, ntv->data, ntv->datalen, 0);
433  if (rv < 0) {
434  /*We received an error on socket read */
435  if (errno == EINTR || errno == EWOULDBLOCK) {
436  /*Nothing for us to process */
437  continue;
438  } else if (errno == ENOBUFS) {
439  if (!ntv->nful_overrun_warned) {
440  int s = ntv->nlbufsiz * 2;
441  if (NFLOGSetnlbufsiz((void *)ntv, s)) {
442  SCLogWarning(SC_WARN_NFLOG_LOSING_EVENTS,
443  "We are losing events, "
444  "increasing buffer size "
445  "to %d", ntv->nlbufsiz);
446  } else {
447  ntv->nful_overrun_warned = 1;
448  }
449  }
450  continue;
451  } else {
452  SCLogWarning(SC_WARN_NFLOG_RECV,
453  "Read from NFLOG fd failed: %s",
454  strerror(errno));
455  SCReturnInt(TM_ECODE_FAILED);
456  }
457  }
458 
459  ret = nflog_handle_packet(ntv->h, ntv->data, rv);
460  if (ret != 0)
461  SCLogWarning(SC_ERR_NFLOG_HANDLE_PKT,
462  "nflog_handle_packet error %" PRId32 "", ret);
463 
464  StatsSyncCountersIfSignalled(tv);
465  }
466 
467  SCReturnInt(TM_ECODE_OK);
468 }
