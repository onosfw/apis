73 void htp_conn_destroy(htp_conn_t *conn) {
74  if (conn == NULL) return;
75 
76  if (conn->transactions != NULL) {
77  // Destroy individual transactions. Do note that iterating
78  // using the iterator does not work here because some of the
79  // list element may be NULL (and with the iterator it is impossible
80  // to distinguish a NULL element from the end of the list).
81  for (size_t i = 0, n = htp_list_size(conn->transactions); i < n; i++) {
82  htp_tx_t *tx = htp_list_get(conn->transactions, i);
83  if (tx != NULL) {
84  htp_tx_destroy_incomplete(tx);
85  }
86  }
87 
88  htp_list_destroy(conn->transactions);
89  conn->transactions = NULL;
90  }
91 
92  if (conn->messages != NULL) {
93  // Destroy individual messages.
94  for (size_t i = 0, n = htp_list_size(conn->messages); i < n; i++) {
95  htp_log_t *l = htp_list_get(conn->messages, i);
96  free((void *) l->msg);
97  free(l);
98  }
99 
100  htp_list_destroy(conn->messages);
101  conn->messages = NULL;
102  }
103 
104  if (conn->server_addr != NULL) {
105  free(conn->server_addr);
106  }
107 
108  if (conn->client_addr != NULL) {
109  free(conn->client_addr);
110  }
111 
112  free(conn);
113 }

115 htp_status_t htp_conn_open(htp_conn_t *conn, const char *client_addr, int client_port,

126 
127  if (server_addr != NULL) {
128  conn->server_addr = strdup(server_addr);
129  if (conn->server_addr == NULL) {

131  free(conn->client_addr);
132  }
133 
134  return HTP_ERROR;
135  }
136  }
137 
138  conn->server_port = server_port;
139 
140  // Remember when the connection was opened.
141  if (timestamp != NULL) {
142  memcpy(&(conn->open_timestamp), timestamp, sizeof(*timestamp));
143  }
144 
145  return HTP_OK;
146 }
