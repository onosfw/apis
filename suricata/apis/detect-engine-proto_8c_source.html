77 /**
78  * \brief Parses a protocol sent as a string.
79  *
80  * \param dp Pointer to the DetectProto instance which will be updated with the
81  * incoming protocol information.
82  * \param str Pointer to the string containing the protocol name.
83  *
84  * \retval >=0 If proto is detected, -1 otherwise.
85  */
86 int DetectProtoParse(DetectProto *dp, char *str)
87 {
88  if (strcasecmp(str, "tcp") == 0) {
89  dp->proto[IPPROTO_TCP / 8] |= 1 << (IPPROTO_TCP % 8);
90  SCLogDebug("TCP protocol detected");
91  } else if (strcasecmp(str, "tcp-pkt") == 0) {
92  dp->proto[IPPROTO_TCP / 8] |= 1 << (IPPROTO_TCP % 8);
93  SCLogDebug("TCP protocol detected, packets only");
94  dp->flags |= DETECT_PROTO_ONLY_PKT;
95  } else if (strcasecmp(str, "tcp-stream") == 0) {
96  dp->proto[IPPROTO_TCP / 8] |= 1 << (IPPROTO_TCP % 8);
97  SCLogDebug("TCP protocol detected, stream only");
98  dp->flags |= DETECT_PROTO_ONLY_STREAM;
99  } else if (strcasecmp(str, "udp") == 0) {
100  dp->proto[IPPROTO_UDP / 8] |= 1 << (IPPROTO_UDP % 8);
101  SCLogDebug("UDP protocol detected");
102  } else if (strcasecmp(str, "icmpv4") == 0) {
103  dp->proto[IPPROTO_ICMP / 8] |= 1 << (IPPROTO_ICMP % 8);
104  SCLogDebug("ICMPv4 protocol detected");
105  } else if (strcasecmp(str, "icmpv6") == 0) {
106  dp->proto[IPPROTO_ICMPV6 / 8] |= 1 << (IPPROTO_ICMPV6 % 8);
107  SCLogDebug("ICMPv6 protocol detected");
108  } else if (strcasecmp(str, "icmp") == 0) {
109  dp->proto[IPPROTO_ICMP / 8] |= 1 << (IPPROTO_ICMP % 8);
110  dp->proto[IPPROTO_ICMPV6 / 8] |= 1 << (IPPROTO_ICMPV6 % 8);
111  SCLogDebug("ICMP protocol detected, sig applies both to ICMPv4 and ICMPv6");
112  } else if (strcasecmp(str, "sctp") == 0) {
113  dp->proto[IPPROTO_SCTP / 8] |= 1 << (IPPROTO_SCTP % 8);
114  SCLogDebug("SCTP protocol detected");
115  } else if (strcasecmp(str,"ipv4") == 0 ||
116  strcasecmp(str,"ip4") == 0 ) {
117  dp->flags |= (DETECT_PROTO_IPV4 | DETECT_PROTO_ANY);
118  memset(dp->proto, 0xff, sizeof(dp->proto));
119  SCLogDebug("IPv4 protocol detected");
120  } else if (strcasecmp(str,"ipv6") == 0 ||
121  strcasecmp(str,"ip6") == 0 ) {
122  dp->flags |= (DETECT_PROTO_IPV6 | DETECT_PROTO_ANY);
123  memset(dp->proto, 0xff, sizeof(dp->proto));
124  SCLogDebug("IPv6 protocol detected");
125  } else if (strcasecmp(str,"ip") == 0 ||
126  strcasecmp(str,"pkthdr") == 0) {
127  /* Proto "ip" is treated as an "any" */
128  dp->flags |= DETECT_PROTO_ANY;
129  memset(dp->proto, 0xff, sizeof(dp->proto));
130  SCLogDebug("IP protocol detected");
131  } else {
132  goto error;
133 
134  /** \todo are numeric protocols even valid? */
135 #if 0
136  uint8_t proto_u8; /* Used to avoid sign extension */
137 
138  /* Extract out a 0-256 value with validation checks */
139  if (ByteExtractStringUint8(&proto_u8, 10, 0, str) == -1) {
140  // XXX
141  SCLogDebug("DetectProtoParse: Error in extracting byte string");
142  goto error;
143  }
144  proto = (int)proto_u8;
145 
146  /* Proto 0 is the same as "ip" above */
147  if (proto == IPPROTO_IP) {
148  dp->flags |= DETECT_PROTO_ANY;
149  } else {
150  dp->proto[proto / 8] |= 1<<(proto % 8);
151  }
152 #endif
153  }
154 
155  return 0;
156 error:
157  return -1;
158 }

160 /** \brief see if a DetectProto contains a certain proto
161  * \param dp detect proto to inspect
162  * \param proto protocol (such as IPPROTO_TCP) to look for
163  * \retval 0 protocol not in the set
164  * \retval 1 protocol is in the set */
165 int DetectProtoContainsProto(DetectProto *dp, int proto)
166 {
167  if (dp->flags & DETECT_PROTO_ANY)
168  return 1;
169 
170  if (dp->proto[proto / 8] & (1<<(proto % 8)))
171  return 1;
172 
173  return 0;
174 }

491 static int DetectProtoTestSig01(void)
492 {
493  Packet *p = NULL;
494  Signature *s = NULL;
495  ThreadVars th_v;
496  DetectEngineThreadCtx *det_ctx;
497  int result = 0;
498  Flow f;
499 
500  memset(&f, 0, sizeof(Flow));
501  memset(&th_v, 0, sizeof(th_v));
502 
503  FLOW_INITIALIZE(&f);
504 
505  p = UTHBuildPacket(NULL, 0, IPPROTO_TCP);
506 
507  p->flow = &f;
508  p->flowflags |= FLOW_PKT_TOSERVER;
509  p->flags |= PKT_HAS_FLOW;
510 
511  DetectEngineCtx *de_ctx = DetectEngineCtxInit();
512  if (de_ctx == NULL) {
513  goto end;
514  }
515 
516  de_ctx->flags |= DE_QUIET;
517 
518  s = de_ctx->sig_list = SigInit(de_ctx,"alert udp any any -> any any "
519  "(msg:\"Not tcp\"; flow:to_server; sid:1;)");
520 
521  if (s == NULL)
522  goto end;
523 
524  s = s->next = SigInit(de_ctx,"alert ip any any -> any any "
525  "(msg:\"IP\"; flow:to_server; sid:2;)");
526 
527  if (s == NULL)
528  goto end;
529 
530  s = s->next = SigInit(de_ctx,"alert tcp any any -> any any "
531  "(msg:\"TCP\"; flow:to_server; sid:3;)");
532 
533  if (s == NULL)
534  goto end;
535 
536  SigGroupBuild(de_ctx);
537  DetectEngineThreadCtxInit(&th_v, (void *)de_ctx, (void *)&det_ctx);
538 
539  SigMatchSignatures(&th_v, de_ctx, det_ctx, p);
540  if (PacketAlertCheck(p, 1)) {
541  printf("sid 1 alerted, but should not have: ");
542  goto cleanup;
543  } else if (PacketAlertCheck(p, 2) == 0) {
544  printf("sid 2 did not alert, but should have: ");
545  goto cleanup;
546  } else if (PacketAlertCheck(p, 3) == 0) {
547  printf("sid 3 did not alert, but should have: ");
548  goto cleanup;
549  }
550 
551  result = 1;
552 
553 cleanup:
554  FLOW_DESTROY(&f);
555 
556  SigGroupCleanup(de_ctx);
557  SigCleanSignatures(de_ctx);
558 
559  DetectEngineThreadCtxDeinit(&th_v, (void *)det_ctx);
560  DetectEngineCtxFree(de_ctx);
561 
562  UTHFreePackets(&p, 1);
563 end:
564  return result;
