215 /**
216  * \brief Function to rotate pcaplog file
217  *
218  * \param t Thread Variable containing input/output queue, cpu affinity etc.
219  * \param pl PcapLog thread variable.
220  *
221  * \retval 0 on succces
222  * \retval -1 on failure
223  */
224 static int PcapLogRotateFile(ThreadVars *t, PcapLogData *pl)
225 {
226  PcapFileName *pf;
227  PcapFileName *pfnext;
228 
229  PCAPLOG_PROFILE_START;
230 
231  if (PcapLogCloseFile(t,pl) < 0) {
232  SCLogDebug("PcapLogCloseFile failed");
233  return -1;
234  }
235 
236  if (pl->use_ringbuffer == RING_BUFFER_MODE_ENABLED && pl->file_cnt >= pl->max_files) {
237  pf = TAILQ_FIRST(&pl->pcap_file_list);
238  SCLogDebug("Removing pcap file %s", pf->filename);
239 
240  if (remove(pf->filename) != 0) {
241  // VJ remove can fail because file is already gone
242  //LogWarning(SC_ERR_PCAP_FILE_DELETE_FAILED,
243  // "failed to remove log file %s: %s",
244  // pf->filename, strerror( errno ));
245  }
246 
247  /* Remove directory if Sguil mode and no files left in sguil dir */
248  if (pl->mode == LOGMODE_SGUIL) {
249  pfnext = TAILQ_NEXT(pf,next);
250 
251  if (strcmp(pf->dirname, pfnext->dirname) == 0) {
252  SCLogDebug("Current entry dir %s and next entry %s "
253  "are equal: not removing dir",
254  pf->dirname, pfnext->dirname);
255  } else {
256  SCLogDebug("current entry %s and %s are "
257  "not equal: removing dir",
258  pf->dirname, pfnext->dirname);
259 
260  if (remove(pf->dirname) != 0) {
261  SCLogWarning(SC_ERR_PCAP_FILE_DELETE_FAILED,
262  "failed to remove sguil log %s: %s",
263  pf->dirname, strerror( errno ));
264  }
265  }
266  }
267 
268  TAILQ_REMOVE(&pl->pcap_file_list, pf, next);
269  PcapFileNameFree(pf);
270  pl->file_cnt--;
271  }
272 
273  if (PcapLogOpenFileCtx(pl) < 0) {
274  SCLogError(SC_ERR_FOPEN, "opening new pcap log file failed");
275  return -1;
276  }
277  pl->file_cnt++;
278  SCLogDebug("file_cnt %u", pl->file_cnt);
279 
280  PCAPLOG_PROFILE_END(pl->profile_rotate);
281  return 0;
282 }

286  PCAPLOG_PROFILE_START;

299  if ((pl->pcap_dumper = pcap_dump_open(pl->pcap_dead_handle,

336 /**
337  * \brief Pcap logging main function
338  *
339  * \param t threadvar
340  * \param p packet
341  * \param data thread module specific data
342  * \param pq pre-packet-queue
343  * \param postpq post-packet-queue
344  *
345  * \retval TM_ECODE_OK on succes
346  * \retval TM_ECODE_FAILED on serious error
347  */
348 static TmEcode PcapLog (ThreadVars *t, Packet *p, void *thread_data, PacketQueue *pq,
349  PacketQueue *postpq)
350 {
351  size_t len;
352  int rotate = 0;
353  int ret = 0;
354 
355  PcapLogThreadData *td = (PcapLogThreadData *)thread_data;
356  PcapLogData *pl = td->pcap_log;
357 
358  if ((p->flags & PKT_PSEUDO_STREAM_END) ||
359  ((p->flags & PKT_STREAM_NOPCAPLOG) &&
360  (pl->use_stream_depth == USE_STREAM_DEPTH_ENABLED)) ||
361  (IS_TUNNEL_PKT(p) && !IS_TUNNEL_ROOT_PKT(p)) ||
362  (pl->honor_pass_rules && (p->flags & PKT_NOPACKET_INSPECTION)))
363  {
364  return TM_ECODE_OK;
365  }
366 
367  PcapLogLock(pl);
368 
369  pl->pkt_cnt++;
370  pl->h->ts.tv_sec = p->ts.tv_sec;
371  pl->h->ts.tv_usec = p->ts.tv_usec;
372  pl->h->caplen = GET_PKT_LEN(p);
373  pl->h->len = GET_PKT_LEN(p);
374  len = sizeof(*pl->h) + GET_PKT_LEN(p);
375 
376  if (pl->filename == NULL) {
377  ret = PcapLogOpenFileCtx(pl);
378  if (ret < 0) {
379  PcapLogUnlock(pl);
380  return TM_ECODE_FAILED;
381  }
382  SCLogDebug("Opening PCAP log file %s", pl->filename);
383  }
384 
385  if (pl->mode == LOGMODE_SGUIL) {
386  struct tm local_tm;
387  struct tm *tms = SCLocalTime(p->ts.tv_sec, &local_tm);
388  if (tms->tm_mday != pl->prev_day) {
389  rotate = 1;
390  pl->prev_day = tms->tm_mday;
391  }
392  }
393 
394  if ((pl->size_current + len) > pl->size_limit || rotate) {
395  if (PcapLogRotateFile(t,pl) < 0) {
396  PcapLogUnlock(pl);
397  SCLogDebug("rotation of pcap failed");
398  return TM_ECODE_FAILED;
399  }
400  }
401 
402  /* XXX pcap handles, nfq, pfring, can only have one link type ipfw? we do
403  * this here as we don't know the link type until we get our first packet */
404  if (pl->pcap_dead_handle == NULL || pl->pcap_dumper == NULL) {
405  if (PcapLogOpenHandles(pl, p) != TM_ECODE_OK) {
406  PcapLogUnlock(pl);
407  return TM_ECODE_FAILED;
408  }
409  }
410 
411  PCAPLOG_PROFILE_START;
412  pcap_dump((u_char *)pl->pcap_dumper, pl->h, GET_PKT_DATA(p));
413  pl->size_current += len;
414  PCAPLOG_PROFILE_END(pl->profile_write);
415  pl->profile_data_size += len;
416 
417  SCLogDebug("pl->size_current %"PRIu64", pl->size_limit %"PRIu64,
418  pl->size_current, pl->size_limit);
419 
420  PcapLogUnlock(pl);
421  return TM_ECODE_OK;
422 }

458 
459  int i;
460  for (i = 0; i < pl->filename_part_cnt && i < MAX_TOKS; i++)
461  copy->filename_parts[i] = pl->filename_parts[i];

471 static TmEcode PcapLogDataInit(ThreadVars *t, void *initdata, void **data)
472 {
473  if (initdata == NULL) {
474  SCLogDebug("Error getting context for PcapLog. \"initdata\" argument NULL");
475  return TM_ECODE_FAILED;
476  }
477 
478  PcapLogData *pl = ((OutputCtx *)initdata)->data;
479 
480  PcapLogThreadData *td = SCCalloc(1, sizeof(*td));
481  if (unlikely(td == NULL))
482  return TM_ECODE_FAILED;
483 
484  if (pl->mode == LOGMODE_MULTI)
485  td->pcap_log = PcapLogDataCopy(pl);
486  else
487  td->pcap_log = pl;
488  BUG_ON(td->pcap_log == NULL);
489 
490  PcapLogLock(td->pcap_log);
491 
492  /** Use the Ouptut Context (file pointer and mutex) */
493  td->pcap_log->pkt_cnt = 0;
494  td->pcap_log->pcap_dead_handle = NULL;
495  td->pcap_log->pcap_dumper = NULL;
496  td->pcap_log->file_cnt = 1;
497 
498  struct timeval ts;
499  memset(&ts, 0x00, sizeof(struct timeval));
500  TimeGet(&ts);
501  struct tm local_tm;
502  struct tm *tms = SCLocalTime(ts.tv_sec, &local_tm);
503  td->pcap_log->prev_day = tms->tm_mday;
504 
505  PcapLogUnlock(td->pcap_log);
506 
507  /* count threads in the global structure */
508  SCMutexLock(&pl->plog_lock);
509  pl->threads++;
510  SCMutexUnlock(&pl->plog_lock);
511 
512  *data = (void *)td;
513 
514  return TM_ECODE_OK;
515 }

578 static int ParseFilename(PcapLogData *pl, const char *filename)
579 {
580  char *toks[MAX_TOKS] = { NULL };
581  int tok = 0;
582  char str[512] = "";
583  int s = 0;
584  int i, x;
585  char *p = NULL;
586 
587  if (filename) {
588  for (i = 0; i < (int)strlen(filename); i++) {
589  if (tok >= MAX_TOKS) {
590  SCLogError(SC_ERR_INVALID_ARGUMENT,
591  "invalid filename option. Max 2 %%-sign options");
592  goto error;
593  }
594 
595  str[s++] = filename[i];
596 
597  if (filename[i] == '%') {
598  str[s-1] = '\0';
599  SCLogDebug("filename with %%-sign: %s", str);
600 
601  p = SCStrdup(str);
602  if (p == NULL)
603  goto error;
604  toks[tok++] = p;
605 
606  s = 0;
607 
608  if (i+1 < (int)strlen(filename)) {
609  if (tok >= MAX_TOKS) {
610  SCLogError(SC_ERR_INVALID_ARGUMENT,
611  "invalid filename option. Max 2 %%-sign options");
612  goto error;
613  }
614 
615  if (filename[i+1] != 'n' && filename[i+1] != 't' && filename[i+1] != 'i') {
616  SCLogError(SC_ERR_INVALID_ARGUMENT,
617  "invalid filename option. Valid %%-sign options: %%n, %%i and %%t");
618  goto error;
619  }
620  str[0] = '%';
621  str[1] = filename[i+1];
622  str[2] = '\0';
623  p = SCStrdup(str);
624  if (p == NULL)
625  goto error;
626  toks[tok++] = p;
627  i++;
628  }
629  }
630  }
631  if (s) {
632  if (tok >= MAX_TOKS) {
633  SCLogError(SC_ERR_INVALID_ARGUMENT,
634  "invalid filename option. Max 3 %%-sign options");
635  goto error;
636 
637  }
638  str[s++] = '\0';
639  p = SCStrdup(str);
640  if (p == NULL)
641  goto error;
642  toks[tok++] = p;
643  }
644 
645  /* finally, store tokens in the pl */
646  for (i = 0; i < tok; i++) {
647  if (toks[i] == NULL)
648  goto error;
649 
650  SCLogDebug("toks[%d] %s", i, toks[i]);
651  pl->filename_parts[i] = toks[i];
652  }
653  pl->filename_part_cnt = tok;
654  }
