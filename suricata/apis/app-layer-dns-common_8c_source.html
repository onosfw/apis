88 int DNSCheckMemcap(uint32_t want, DNSState *state)
89 {
90  if (state != NULL) {

92  SC_ATOMIC_ADD(dns_memcap_state, 1);
93  DNSSetEvent(state, DNS_DECODER_EVENT_STATE_MEMCAP_REACHED);
94  return -1;

97

181 void *DNSGetTx(void *alstate, uint64_t tx_id)
182 {
183  DNSState *dns_state = (DNSState *)alstate;
184  DNSTransaction *tx = NULL;
185 
186  /* fast track: try the current tx */
187  if (dns_state->curr && dns_state->curr->tx_num == tx_id + 1)
188  return dns_state->curr;
189 
190  /* fast track:
191  * if the prev tx_id is equal to the stored tx ptr, we can
192  * use this shortcut to get to the next. */
193  if (dns_state->iter) {
194  if (tx_id == dns_state->iter->tx_num) {
195  tx = TAILQ_NEXT(dns_state->iter, next);
196  if (tx && tx->tx_num == tx_id + 1) {
197  dns_state->iter = tx;
198  return tx;
199  }
200  }
201  }
202 
203  /* no luck with the fast tracks, do the full list walk */
204  TAILQ_FOREACH(tx, &dns_state->tx_list, next) {
205  SCLogDebug("tx->tx_num %u, tx_id %"PRIu64, tx->tx_num, (tx_id+1));
206  if ((tx_id+1) != tx->tx_num)
207  continue;
208 
209  SCLogDebug("returning tx %p", tx);
210  dns_state->iter = tx;
211  return tx;
212  }
213 
214  return NULL;
215 }

283 /** \internal
284  * \brief Free a DNS TX
285  * \param tx DNS TX to free */
286 static void DNSTransactionFree(DNSTransaction *tx, DNSState *state)
287 {
288  SCEnter();
289 
290  DNSQueryEntry *q = NULL;
291  while ((q = TAILQ_FIRST(&tx->query_list))) {
292  TAILQ_REMOVE(&tx->query_list, q, next);
293  DNSDecrMemcap((sizeof(DNSQueryEntry) + q->len), state);
294  SCFree(q);
295  }
296 
297  DNSAnswerEntry *a = NULL;
298  while ((a = TAILQ_FIRST(&tx->answer_list))) {
299  TAILQ_REMOVE(&tx->answer_list, a, next);
300  DNSDecrMemcap((sizeof(DNSAnswerEntry) + a->fqdn_len + a->data_len), state);
301  SCFree(a);
302  }
303  while ((a = TAILQ_FIRST(&tx->authority_list))) {
304  TAILQ_REMOVE(&tx->authority_list, a, next);
305  DNSDecrMemcap((sizeof(DNSAnswerEntry) + a->fqdn_len + a->data_len), state);
306  SCFree(a);
307  }
308 
309  AppLayerDecoderEventsFreeEvents(&tx->decoder_events);
310 
311  if (tx->de_state != NULL) {
312  DetectEngineStateFree(tx->de_state);
313  BUG_ON(state->tx_with_detect_state_cnt == 0);
314  state->tx_with_detect_state_cnt--;
315  }
316 
317  if (state->iter == tx)
318  state->iter = NULL;
319 
320  DNSDecrMemcap(sizeof(DNSTransaction), state);
321  SCFree(tx);
322  SCReturn;
323 }

325 /**
326  * \brief dns transaction cleanup callback
327  */
328 void DNSStateTransactionFree(void *state, uint64_t tx_id)
329 {
330  SCEnter();
331 
332  DNSState *dns_state = state;
333  DNSTransaction *tx = NULL;
334 
335  SCLogDebug("state %p, id %"PRIu64, dns_state, tx_id);
336 
337  TAILQ_FOREACH(tx, &dns_state->tx_list, next) {
338  SCLogDebug("tx %p tx->tx_num %u, tx_id %"PRIu64, tx, tx->tx_num, (tx_id+1));
339  if ((tx_id+1) < tx->tx_num)
340  break;
341  else if ((tx_id+1) > tx->tx_num)
342  continue;
343 
344  if (tx == dns_state->curr)
345  dns_state->curr = NULL;
346 
347  if (tx->decoder_events != NULL) {
348  if (tx->decoder_events->cnt <= dns_state->events)
349  dns_state->events -= tx->decoder_events->cnt;
350  else
351  dns_state->events = 0;
352  }
353 
354  TAILQ_REMOVE(&dns_state->tx_list, tx, next);
355  DNSTransactionFree(tx, state);
356  break;
357  }
358  SCReturn;
359 }

361 /** \internal
362  * \brief Find the DNS Tx in the state
363  * \param tx_id id of the tx
364  * \retval tx or NULL if not found */
365 DNSTransaction *DNSTransactionFindByTxId(const DNSState *dns_state, const uint16_t tx_id)
366 {
367  if (dns_state->curr == NULL)
368  return NULL;
369 
370  /* fast path */
371  if (dns_state->curr->tx_id == tx_id) {
372  return dns_state->curr;
373 
374  /* slow path, iterate list */
375  } else {
376  DNSTransaction *tx = NULL;
377  TAILQ_FOREACH(tx, &dns_state->tx_list, next) {
378  if (tx->tx_id == tx_id) {
379  return tx;
380  }
381  }
382  }
383  /* not found */
384  return NULL;
385 }

451 /** \brief Validation checks for DNS request header
452  *
453  * Will set decoder events if anomalies are found.
454  *
455  * \retval 0 ok
456  * \retval -1 error
457  */
458 int DNSValidateRequestHeader(DNSState *dns_state, const DNSHeader *dns_header)
459 {
460  uint16_t flags = ntohs(dns_header->flags);
461 
462  if ((flags & 0x8000) != 0) {
463  SCLogDebug("not a request 0x%04x", flags);
464  DNSSetEvent(dns_state, DNS_DECODER_EVENT_NOT_A_REQUEST);
465  goto bad_data;
466  }
467 
468  if ((flags & 0x0040) != 0) {
469  SCLogDebug("Z flag not 0, 0x%04x", flags);
470  DNSSetEvent(dns_state, DNS_DECODER_EVENT_Z_FLAG_SET);
471  goto bad_data;
472  }
473 
474  return 0;
475 bad_data:
476  return -1;
477 }

508  * \brief check the query list to see if we already have this exact query
509  * \retval bool true or false

511 static int QueryIsDuplicate(DNSTransaction *tx, const uint8_t *fqdn, const uint16_t fqdn_len,
512  const uint16_t type, const uint16_t class)
513 {
514  DNSQueryEntry *q = NULL;
515 
516  TAILQ_FOREACH(q, &tx->query_list, next) {
517  uint8_t *qfqdn = (uint8_t *)q + sizeof(DNSQueryEntry);
518 
519  if (q->len == fqdn_len && q->type == type &&
520  q->class == class &&

522  return TRUE;
523  }
524  }
525  return FALSE;

528 void DNSStoreQueryInState(DNSState *dns_state, const uint8_t *fqdn, const uint16_t fqdn_len,
529  const uint16_t type, const uint16_t class, const uint16_t tx_id)
530 {
531  /* flood protection */
532  if (dns_state->givenup)
533  return;
534 
535  /* find the tx and see if this is an exact duplicate */
536  DNSTransaction *tx = DNSTransactionFindByTxId(dns_state, tx_id);
537  if ((tx != NULL) && (QueryIsDuplicate(tx, fqdn, fqdn_len, type, class) == TRUE)) {
538  SCLogDebug("query is duplicate");
539  return;
540  }
541 
542  /* see if the last tx is unreplied */
543  if (dns_state->curr != tx && dns_state->curr != NULL &&
544  dns_state->curr->replied == 0)
545  {
546  dns_state->curr->reply_lost = 1;
547  dns_state->unreplied_cnt++;
548 
549  /* check flood limit */
550  if (dns_config.request_flood != 0 &&
551  dns_state->unreplied_cnt > dns_config.request_flood) {
552  DNSSetEvent(dns_state, DNS_DECODER_EVENT_FLOODED);
553  dns_state->givenup = 1;
554  }
555  }
556 
557  if (tx == NULL) {
558  tx = DNSTransactionAlloc(dns_state, tx_id);
559  if (tx == NULL)
560  return;
561  dns_state->transaction_max++;
562  SCLogDebug("dns_state->transaction_max updated to %"PRIu64, dns_state->transaction_max);
563  TAILQ_INSERT_TAIL(&dns_state->tx_list, tx, next);
564  dns_state->curr = tx;
565  tx->tx_num = dns_state->transaction_max;
566  SCLogDebug("new tx %u with internal id %u", tx->tx_id, tx->tx_num);
567  }
568 
569  if (DNSCheckMemcap((sizeof(DNSQueryEntry) + fqdn_len), dns_state) < 0)
570  return;
571  DNSQueryEntry *q = SCMalloc(sizeof(DNSQueryEntry) + fqdn_len);
572  if (unlikely(q == NULL))
573  return;
574  DNSIncrMemcap((sizeof(DNSQueryEntry) + fqdn_len), dns_state);
575 
576  q->type = type;
577  q->class = class;
578  q->len = fqdn_len;
579  memcpy((uint8_t *)q + sizeof(DNSQueryEntry), fqdn, fqdn_len);
580 
581  TAILQ_INSERT_TAIL(&tx->query_list, q, next);
582 
583  SCLogDebug("Query for TX %04x stored", tx_id);
584 }

586 void DNSStoreAnswerInState(DNSState *dns_state, const int rtype, const uint8_t *fqdn,
587  const uint16_t fqdn_len, const uint16_t type, const uint16_t class, const uint16_t ttl,
588  const uint8_t *data, const uint16_t data_len, const uint16_t tx_id)
589 {
590  DNSTransaction *tx = DNSTransactionFindByTxId(dns_state, tx_id);
591  if (tx == NULL) {
592  tx = DNSTransactionAlloc(dns_state, tx_id);
593  if (tx == NULL)
594  return;
595  TAILQ_INSERT_TAIL(&dns_state->tx_list, tx, next);
596  dns_state->curr = tx;
597  tx->tx_num = dns_state->transaction_max;
598  }
599 
600  if (DNSCheckMemcap((sizeof(DNSAnswerEntry) + fqdn_len + data_len), dns_state) < 0)
601  return;
602  DNSAnswerEntry *q = SCMalloc(sizeof(DNSAnswerEntry) + fqdn_len + data_len);
603  if (unlikely(q == NULL))
604  return;
605  DNSIncrMemcap((sizeof(DNSAnswerEntry) + fqdn_len + data_len), dns_state);
606 
607  q->type = type;
608  q->class = class;
609  q->ttl = ttl;
610  q->fqdn_len = fqdn_len;
611  q->data_len = data_len;
612 
613  uint8_t *ptr = (uint8_t *)q + sizeof(DNSAnswerEntry);
614  if (fqdn != NULL && fqdn_len > 0) {
615  memcpy(ptr, fqdn, fqdn_len);
616  ptr += fqdn_len;
617  }
618  if (data != NULL && data_len > 0) {
619  memcpy(ptr, data, data_len);
620  }
621 
622  if (rtype == DNS_LIST_ANSWER)
623  TAILQ_INSERT_TAIL(&tx->answer_list, q, next);
624  else if (rtype == DNS_LIST_AUTHORITY)
625  TAILQ_INSERT_TAIL(&tx->authority_list, q, next);
626  else
627  BUG_ON(1);
628 
629  SCLogDebug("Answer for TX %04x stored", tx_id);
630 
631  /* mark tx is as replied so we can log it */
632  tx->replied = 1;
633 
634  /* reset unreplied counter */
635  dns_state->unreplied_cnt = 0;
636 }

638 /** \internal
639  * \brief get domain name from dns packet
640  *
641  * In case of compressed name storage this function follows the ptrs to
642  * create the full domain name.
643  *
644  * The length bytes are converted into dots, e.g. |03|com|00| becomes
645  * .com
646  * The trailing . is not stored.
647  *
648  * \param input input buffer (complete dns record)
649  * \param input_len lenght of input buffer
650  * \param offset offset into @input where dns name starts
651  * \param fqdn buffer to store result
652  * \param fqdn_size size of @fqdn buffer
653  * \retval 0 on error/no buffer
654  * \retval size size of fqdn
655  */
656 static uint16_t DNSResponseGetNameByOffset(const uint8_t * const input, const uint32_t input_len,
657  const uint16_t offset, uint8_t *fqdn, const size_t fqdn_size)
658 {
659  if (input + input_len < input + offset + 1) {
660  SCLogDebug("input buffer too small for domain of len %u", offset);
661  goto insufficient_data;
662  }
663 
664  int steps = 0;
665  uint16_t fqdn_offset = 0;
666  uint8_t length = *(input + offset);
667  const uint8_t *qdata = input + offset;
668  SCLogDebug("qry length %u", length);
669 
670  if (length == 0) {
671  memcpy(fqdn, "<root>", 6);
672  SCReturnUInt(6U);
673  }
674 
675  while (length != 0) {
676  int cnt = 0;
677  while (length & 0xc0) {
678  uint16_t offset = ((length & 0x3f) << 8) + *(qdata+1);
679  qdata = (const uint8_t *)input + offset;
680 
681  if (input + input_len < qdata + 1) {
682  SCLogDebug("input buffer too small");
683  goto insufficient_data;
684  }
685 
686  length = *qdata;
687  SCLogDebug("qry length %u", length);
688 
689  if (cnt++ == 100) {
690  SCLogDebug("too many pointer iterations, loop?");
691  goto bad_data;
692  }
693  }
694  qdata++;
695 
696  if (length == 0) {
697  break;
698  }
699 
700  if (input + input_len < qdata + length) {
701  SCLogDebug("input buffer too small for domain of len %u", length);
702  goto insufficient_data;
703  }
704  //PrintRawDataFp(stdout, qdata, length);
705 
706  if ((size_t)(fqdn_offset + length + 1) < fqdn_size) {
707  memcpy(fqdn + fqdn_offset, qdata, length);
708  fqdn_offset += length;
709  fqdn[fqdn_offset++] = '.';
710  }
711  qdata += length;
712 
713  if (input + input_len < qdata + 1) {
714  SCLogDebug("input buffer too small for len field");
715  goto insufficient_data;
716  }
717 
718  length = *qdata;
719  SCLogDebug("qry length %u", length);
720  steps++;
721  if (steps >= 255)
722  goto bad_data;
723  }
724  if (fqdn_offset) {
725  fqdn_offset--;
726  }
727  //PrintRawDataFp(stdout, fqdn, fqdn_offset);
728  SCReturnUInt(fqdn_offset);
729 bad_data:
730 insufficient_data:
731  SCReturnUInt(0U);
732 }

734 /** \internal
735  * \brief skip past domain name field
736  *
737  * Skip the domain at position data. We don't care about following compressed names
738  * as we only want to know when the next part of the buffer starts
739  *
740  * \param input input buffer (complete dns record)
741  * \param input_len lenght of input buffer
742  * \param data current position
743  *
744  * \retval NULL on out of bounds data
745  * \retval sdata ptr to position in buffer past the name
746  */
747 static const uint8_t *SkipDomain(const uint8_t * const input,
748  const uint32_t input_len, const uint8_t *data)
749 {
750  const uint8_t *sdata = data;
751  while (*sdata != 0x00) {
752  if (*sdata & 0xc0) {
753  sdata++;
754  break;
755  } else {
756  sdata += ((*sdata) + 1);
757  }
758  if (input + input_len < sdata) {
759  SCLogDebug("input buffer too small for data of len");
760  goto insufficient_data;
761  }
762  }
763  sdata++;
764  if (input + input_len < sdata) {
765  SCLogDebug("input buffer too small for data of len");
766  goto insufficient_data;
767  }
768  return sdata;
769 insufficient_data:
770  return NULL;
771 }
