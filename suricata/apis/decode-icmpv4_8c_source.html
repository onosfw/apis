47 /**
48  * Note, this is the IP header, plus a bit of the original packet, not the whole thing!
49  */
50 void DecodePartialIPV4( Packet* p, uint8_t* partial_packet, uint16_t len )
51 {
52  /** Check the sizes, the header must fit at least */
53  if (len < IPV4_HEADER_LEN) {
54  SCLogDebug("DecodePartialIPV4: ICMPV4_IPV4_TRUNC_PKT");
55  ENGINE_SET_INVALID_EVENT(p, ICMPV4_IPV4_TRUNC_PKT);
56  return;
57  }
58 
59  IPV4Hdr *icmp4_ip4h = (IPV4Hdr*)partial_packet;
60 
61  /** Check the embedded version */
62  if (IPV4_GET_RAW_VER(icmp4_ip4h) != 4) {
63  /** Check the embedded version */
64  SCLogDebug("DecodePartialIPV4: ICMPv4 contains Unknown IPV4 version "
65  "ICMPV4_IPV4_UNKNOWN_VER");
66  ENGINE_SET_INVALID_EVENT(p, ICMPV4_IPV4_UNKNOWN_VER);
67  return;
68  }
69 
70  /** We need to fill icmpv4vars */
71  p->icmpv4vars.emb_ipv4h = icmp4_ip4h;
72 
73  /** Get the IP address from the contained packet */
74  p->icmpv4vars.emb_ip4_src = IPV4_GET_RAW_IPSRC(icmp4_ip4h);
75  p->icmpv4vars.emb_ip4_dst = IPV4_GET_RAW_IPDST(icmp4_ip4h);
76 
77  p->icmpv4vars.emb_ip4_hlen=IPV4_GET_RAW_HLEN(icmp4_ip4h) << 2;
78 
79  switch (IPV4_GET_RAW_IPPROTO(icmp4_ip4h)) {
80  case IPPROTO_TCP:
81  if (len >= IPV4_HEADER_LEN + TCP_HEADER_LEN ) {
82  p->icmpv4vars.emb_tcph = (TCPHdr*)(partial_packet + IPV4_HEADER_LEN);
83  p->icmpv4vars.emb_sport = ntohs(p->icmpv4vars.emb_tcph->th_sport);
84  p->icmpv4vars.emb_dport = ntohs(p->icmpv4vars.emb_tcph->th_dport);
85  p->icmpv4vars.emb_ip4_proto = IPPROTO_TCP;
86 
87  SCLogDebug("DecodePartialIPV4: ICMPV4->IPV4->TCP header sport: "
88  "%"PRIu8" dport %"PRIu8"", p->icmpv4vars.emb_sport,
89  p->icmpv4vars.emb_dport);
90  } else if (len >= IPV4_HEADER_LEN + 4) {
91  /* only access th_sport and th_dport */
92  TCPHdr *emb_tcph = (TCPHdr*)(partial_packet + IPV4_HEADER_LEN);
93 
94  p->icmpv4vars.emb_tcph = NULL;
95  p->icmpv4vars.emb_sport = ntohs(emb_tcph->th_sport);
96  p->icmpv4vars.emb_dport = ntohs(emb_tcph->th_dport);
97  p->icmpv4vars.emb_ip4_proto = IPPROTO_TCP;
98  SCLogDebug("DecodePartialIPV4: ICMPV4->IPV4->TCP partial header sport: "
99  "%"PRIu8" dport %"PRIu8"", p->icmpv4vars.emb_sport,
100  p->icmpv4vars.emb_dport);
101  } else {
102  SCLogDebug("DecodePartialIPV4: Warning, ICMPV4->IPV4->TCP "
103  "header Didn't fit in the packet!");
104  p->icmpv4vars.emb_sport = 0;
105  p->icmpv4vars.emb_dport = 0;
106  }
107 
108  break;
109  case IPPROTO_UDP:
110  if (len >= IPV4_HEADER_LEN + UDP_HEADER_LEN ) {
111  p->icmpv4vars.emb_udph = (UDPHdr*)(partial_packet + IPV4_HEADER_LEN);
112  p->icmpv4vars.emb_sport = ntohs(p->icmpv4vars.emb_udph->uh_sport);
113  p->icmpv4vars.emb_dport = ntohs(p->icmpv4vars.emb_udph->uh_dport);
114  p->icmpv4vars.emb_ip4_proto = IPPROTO_UDP;
115 
116  SCLogDebug("DecodePartialIPV4: ICMPV4->IPV4->UDP header sport: "
117  "%"PRIu8" dport %"PRIu8"", p->icmpv4vars.emb_sport,
118  p->icmpv4vars.emb_dport);
119  } else {
120  SCLogDebug("DecodePartialIPV4: Warning, ICMPV4->IPV4->UDP "
121  "header Didn't fit in the packet!");
122  p->icmpv4vars.emb_sport = 0;
123  p->icmpv4vars.emb_dport = 0;
124  }
125 
126  break;
127  case IPPROTO_ICMP:
128  p->icmpv4vars.emb_icmpv4h = (ICMPV4Hdr*)(partial_packet + IPV4_HEADER_LEN);
129  p->icmpv4vars.emb_sport = 0;
130  p->icmpv4vars.emb_dport = 0;
131  p->icmpv4vars.emb_ip4_proto = IPPROTO_ICMP;
132 
133  SCLogDebug("DecodePartialIPV4: ICMPV4->IPV4->ICMP header");
134 
135  break;
136  }
137 
138  /* debug print */
139 #ifdef DEBUG
140  char s[16], d[16];
141  PrintInet(AF_INET, &(p->icmpv4vars.emb_ip4_src), s, sizeof(s));
142  PrintInet(AF_INET, &(p->icmpv4vars.emb_ip4_dst), d, sizeof(d));
143  SCLogDebug("ICMPv4 embedding IPV4 %s->%s - PROTO: %" PRIu32 " ID: %" PRIu32 "", s,d,
144  IPV4_GET_RAW_IPPROTO(icmp4_ip4h), IPV4_GET_RAW_IPID(icmp4_ip4h));
145 #endif
146 
147  return;
148 
149 }

154 int DecodeICMPV4(ThreadVars *tv, DecodeThreadVars *dtv, Packet *p, uint8_t *pkt, uint16_t len, PacketQueue *pq)
155 {
156  StatsIncr(tv, dtv->counter_icmpv4);
157 
158  if (len < ICMPV4_HEADER_LEN) {
159  ENGINE_SET_INVALID_EVENT(p, ICMPV4_PKT_TOO_SMALL);
160  return TM_ECODE_FAILED;
161  }
162 
163  p->icmpv4h = (ICMPV4Hdr *)pkt;
164 
165  SCLogDebug("ICMPV4 TYPE %" PRIu32 " CODE %" PRIu32 "", p->icmpv4h->type, p->icmpv4h->code);
166 
167  p->proto = IPPROTO_ICMP;
168  p->type = p->icmpv4h->type;
169  p->code = p->icmpv4h->code;
170  p->payload = pkt + ICMPV4_HEADER_LEN;
171  p->payload_len = len - ICMPV4_HEADER_LEN;
172 
173  ICMPV4ExtHdr* icmp4eh = (ICMPV4ExtHdr*) p->icmpv4h;
174 
175  switch (p->icmpv4h->type)
176  {
177  case ICMP_ECHOREPLY:
178  p->icmpv4vars.id=icmp4eh->id;
179  p->icmpv4vars.seq=icmp4eh->seq;
180  if (p->icmpv4h->code!=0) {
181  ENGINE_SET_EVENT(p,ICMPV4_UNKNOWN_CODE);
182  }
183  break;
184 
185  case ICMP_DEST_UNREACH:
186  if (p->icmpv4h->code > NR_ICMP_UNREACH) {
187  ENGINE_SET_EVENT(p,ICMPV4_UNKNOWN_CODE);
188  } else {
189  /* parse IP header plus 64 bytes */
190  if (len > ICMPV4_HEADER_PKT_OFFSET) {
191  DecodePartialIPV4( p, (uint8_t *)(pkt + ICMPV4_HEADER_PKT_OFFSET), len - ICMPV4_HEADER_PKT_OFFSET );
192 
193  /* ICMP ICMP_DEST_UNREACH influence TCP/UDP flows */
194  if (ICMPV4_DEST_UNREACH_IS_VALID(p)) {
195  FlowHandlePacket(tv, dtv, p);
196  }
197  }
198  }
199 
200 
201  break;
202 
203  case ICMP_SOURCE_QUENCH:
204  if (p->icmpv4h->code!=0) {
205  ENGINE_SET_EVENT(p,ICMPV4_UNKNOWN_CODE);
206  } else {
207  // parse IP header plus 64 bytes
208  if (len >= ICMPV4_HEADER_PKT_OFFSET)
209  DecodePartialIPV4( p, (uint8_t*) (pkt + ICMPV4_HEADER_PKT_OFFSET), len - ICMPV4_HEADER_PKT_OFFSET );
210  }
211  break;
212 
213  case ICMP_REDIRECT:
214  if (p->icmpv4h->code>ICMP_REDIR_HOSTTOS) {
215  ENGINE_SET_EVENT(p,ICMPV4_UNKNOWN_CODE);
216  } else {
217  // parse IP header plus 64 bytes
218  if (len > ICMPV4_HEADER_PKT_OFFSET)
219  DecodePartialIPV4( p, (uint8_t*) (pkt + ICMPV4_HEADER_PKT_OFFSET), len - ICMPV4_HEADER_PKT_OFFSET );
220  }
221  break;
222 
223  case ICMP_ECHO:
224  p->icmpv4vars.id=icmp4eh->id;
225  p->icmpv4vars.seq=icmp4eh->seq;
226  if (p->icmpv4h->code!=0) {
227  ENGINE_SET_EVENT(p,ICMPV4_UNKNOWN_CODE);
228  }
229  break;
230 
231  case ICMP_TIME_EXCEEDED:
232  if (p->icmpv4h->code>ICMP_EXC_FRAGTIME) {
233  ENGINE_SET_EVENT(p,ICMPV4_UNKNOWN_CODE);
234  } else {
235  // parse IP header plus 64 bytes
236  if (len > ICMPV4_HEADER_PKT_OFFSET)
237  DecodePartialIPV4( p, (uint8_t*) (pkt + ICMPV4_HEADER_PKT_OFFSET), len - ICMPV4_HEADER_PKT_OFFSET );
238  }
239  break;
240 
241  case ICMP_PARAMETERPROB:
242  if (p->icmpv4h->code!=0) {
243  ENGINE_SET_EVENT(p,ICMPV4_UNKNOWN_CODE);
244  } else {
245  // parse IP header plus 64 bytes
246  if (len > ICMPV4_HEADER_PKT_OFFSET)
247  DecodePartialIPV4( p, (uint8_t*) (pkt + ICMPV4_HEADER_PKT_OFFSET), len - ICMPV4_HEADER_PKT_OFFSET );
248  }
249  break;
250 
251  case ICMP_TIMESTAMP:
252  p->icmpv4vars.id=icmp4eh->id;
253  p->icmpv4vars.seq=icmp4eh->seq;
254  if (p->icmpv4h->code!=0) {
255  ENGINE_SET_EVENT(p,ICMPV4_UNKNOWN_CODE);
256  }
257  break;
258 
259  case ICMP_TIMESTAMPREPLY:
260  p->icmpv4vars.id=icmp4eh->id;
261  p->icmpv4vars.seq=icmp4eh->seq;
262  if (p->icmpv4h->code!=0) {
263  ENGINE_SET_EVENT(p,ICMPV4_UNKNOWN_CODE);
264  }
265  break;
266 
267  case ICMP_INFO_REQUEST:
268  p->icmpv4vars.id=icmp4eh->id;
269  p->icmpv4vars.seq=icmp4eh->seq;
270  if (p->icmpv4h->code!=0) {
271  ENGINE_SET_EVENT(p,ICMPV4_UNKNOWN_CODE);
272  }
273  break;
274 
275  case ICMP_INFO_REPLY:
276  p->icmpv4vars.id=icmp4eh->id;
277  p->icmpv4vars.seq=icmp4eh->seq;
278  if (p->icmpv4h->code!=0) {
279  ENGINE_SET_EVENT(p,ICMPV4_UNKNOWN_CODE);
280  }
281  break;
282 
283  case ICMP_ADDRESS:
284  p->icmpv4vars.id=icmp4eh->id;
285  p->icmpv4vars.seq=icmp4eh->seq;
286  if (p->icmpv4h->code!=0) {
287  ENGINE_SET_EVENT(p,ICMPV4_UNKNOWN_CODE);
288  }
289  break;
290 
291  case ICMP_ADDRESSREPLY:
292  p->icmpv4vars.id=icmp4eh->id;
293  p->icmpv4vars.seq=icmp4eh->seq;
294  if (p->icmpv4h->code!=0) {
295  ENGINE_SET_EVENT(p,ICMPV4_UNKNOWN_CODE);
296  }
297  break;
298 
299  default:
300  ENGINE_SET_EVENT(p,ICMPV4_UNKNOWN_TYPE);
301 
302  }
303 
304  return TM_ECODE_OK;
305 }
