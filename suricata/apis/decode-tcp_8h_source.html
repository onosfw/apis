171 /**
172  * \brief Calculates the checksum for the TCP packet
173  *
174  * \param shdr Pointer to source address field from the IP packet. Used as a
175  * part of the pseudoheader for computing the checksum
176  * \param pkt Pointer to the start of the TCP packet
177  * \param tlen Total length of the TCP packet(header + payload)
178  *
179  * \retval csum Checksum for the TCP packet
180  */
181 static inline uint16_t TCPCalculateChecksum(uint16_t *shdr, uint16_t *pkt,
182  uint16_t tlen)
183 {
184  uint16_t pad = 0;
185  uint32_t csum = shdr[0];
186 
187  csum += shdr[1] + shdr[2] + shdr[3] + htons(6) + htons(tlen);
188 
189  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5] + pkt[6] +
190  pkt[7] + pkt[9];
191 
192  tlen -= 20;
193  pkt += 10;
194 
195  while (tlen >= 32) {
196  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5] + pkt[6] +
197  pkt[7] + pkt[8] + pkt[9] + pkt[10] + pkt[11] + pkt[12] + pkt[13] +
198  pkt[14] + pkt[15];
199  tlen -= 32;
200  pkt += 16;
201  }
202 
203  while(tlen >= 8) {
204  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3];
205  tlen -= 8;
206  pkt += 4;
207  }
208 
209  while(tlen >= 4) {
210  csum += pkt[0] + pkt[1];
211  tlen -= 4;
212  pkt += 2;
213  }
214 
215  while (tlen > 1) {
216  csum += pkt[0];
217  pkt += 1;
218  tlen -= 2;
219  }
220 
221  if (tlen == 1) {
222  *(uint8_t *)(&pad) = (*(uint8_t *)pkt);
223  csum += pad;
224  }
225 
226  csum = (csum >> 16) + (csum & 0x0000FFFF);
227  csum += (csum >> 16);
228 
229  return (uint16_t)~csum;
230 }
