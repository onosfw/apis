94 /**
95  * \brief Match function for tos keyword.
96  *
97  * \param tv ThreadVars instance.
98  * \param det_ctx Pointer to the detection thread ctx.
99  * \param p Pointer to the packet.
100  * \param m Pointer to the SigMatch containing the tos data.
101  *
102  * \retval 0 no match
103  * \retval 1 match
104  */
105 int DetectTosMatch(ThreadVars *tv, DetectEngineThreadCtx *det_ctx, Packet *p,
106  Signature *s, const SigMatchCtx *ctx)
107 {
108  const DetectTosData *tosd = (const DetectTosData *)ctx;
109  int result = 0;
110 
111  if (!PKT_IS_IPV4(p) || PKT_IS_PSEUDOPKT(p)) {
112  return 0;
113  }
114 
115  if (tosd->tos == IPV4_GET_IPTOS(p)) {
116  SCLogDebug("tos match found for %d\n", tosd->tos);
117  result = 1;
118  }
119 
120  return (tosd->negated ^ result);
121 }

123 DetectTosData *DetectTosParse(char *arg)
124 {
125  DetectTosData *tosd = NULL;
126 #define MAX_SUBSTRINGS 30
127  int ret = 0, res = 0;
128  int ov[MAX_SUBSTRINGS];
129 
130  ret = pcre_exec(parse_regex, parse_regex_study, arg, strlen(arg), 0, 0,
131  ov, MAX_SUBSTRINGS);
132 
133  if (ret != 2) {
134  SCLogError(SC_ERR_PCRE_MATCH, "invalid tos option - %s. "
135  "The tos option value must be in the range "
136  "%u - %u", arg, DETECT_IPTOS_MIN, DETECT_IPTOS_MAX);
137  goto error;
138  }
139 
140  const char *str_ptr;
141  res = pcre_get_substring((char *)arg, ov, MAX_SUBSTRINGS, 1,
142  &str_ptr);
143  if (res < 0) {
144  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
145  goto error;
146  }
147 
148  int64_t tos = 0;
149  int negated = 0;
150 
151  if (*str_ptr == '!') {
152  str_ptr++;
153  negated = 1;
154  }
155 
156  while (isspace((unsigned char)*str_ptr))
157  str_ptr++;
158 
159  if (*str_ptr == 'x' || *str_ptr == 'X') {
160  int r = ByteExtractStringSigned(&tos, 16, 0, str_ptr + 1);
161  if (r < 0) {
162  goto error;
163  }
164  } else {
165  int r = ByteExtractStringSigned(&tos, 10, 0, str_ptr);
166  if (r < 0) {
167  goto error;
168  }
169  }
170  if (!(tos >= DETECT_IPTOS_MIN && tos <= DETECT_IPTOS_MAX)) {
171  SCLogError(SC_ERR_INVALID_SIGNATURE, "Invalid tos argument - "
172  "%s. The tos option value must be in the range "
173  "%u - %u", str_ptr, DETECT_IPTOS_MIN, DETECT_IPTOS_MAX);
174  goto error;
175  }
176 
177  tosd = SCMalloc(sizeof(DetectTosData));
178  if (unlikely(tosd == NULL))
179  goto error;
180  tosd->tos = (uint8_t)tos;
181  tosd->negated = negated;
182 
183  return tosd;
184 
185 error:
186  if (tosd != NULL)
187  DetectTosFree(tosd);
188  return NULL;
189 }

191 /**
192  * \brief Setup function for tos argument. Parse the argument and
193  * add it into the sig.
194  *
195  * \param de_ctx Detection Engine Context instance.
196  * \param s Pointer to the signature.
197  * \param arg Argument to be parsed.
198  *
199  * \retval 0 on Success.
200  * \retval -1 on Failure.
201  */
202 int DetectTosSetup(DetectEngineCtx *de_ctx, Signature *s, char *arg)
203 {
204  DetectTosData *tosd;
205  SigMatch *sm;
206 
207  tosd = DetectTosParse(arg);
208  if (tosd == NULL)
209  goto error;
210 
211  /* Okay so far so good, lets get this into a SigMatch
212  * and put it in the Signature. */
213  sm = SigMatchAlloc();
214  if (sm == NULL)
215  goto error;
216 
217  sm->type = DETECT_TOS;
218  sm->ctx = (SigMatchCtx *)tosd;
219 
220  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);
221  s->flags |= SIG_FLAG_REQUIRE_PACKET;
222 
223  return 0;
224 
225 error:
226  return -1;
227 }
