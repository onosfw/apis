43 /**
44  * Generic response line parser.
45  *
46  * @param[in] connp
47  * @return HTP status
48  */
49 htp_status_t htp_parse_response_line_generic(htp_connp_t *connp) {
50  htp_tx_t *tx = connp->out_tx;
51  unsigned char *data = bstr_ptr(tx->response_line);
52  size_t len = bstr_len(tx->response_line);
53  size_t pos = 0;
54 
55  tx->response_protocol = NULL;
56  tx->response_protocol_number = HTP_PROTOCOL_INVALID;
57  tx->response_status = NULL;
58  tx->response_status_number = HTP_STATUS_INVALID;
59  tx->response_message = NULL;
60 
61  // Ignore whitespace at the beginning of the line.
62  while ((pos < len) && (htp_is_space(data[pos]))) pos++;
63 
64  size_t start = pos;
65 
66  // Find the end of the protocol string.
67  while ((pos < len) && (!htp_is_space(data[pos]))) pos++;
68  if (pos - start == 0) return HTP_OK;
69 
70  tx->response_protocol = bstr_dup_mem(data + start, pos - start);
71  if (tx->response_protocol == NULL) return HTP_ERROR;
72 
73  tx->response_protocol_number = htp_parse_protocol(tx->response_protocol);
74 
75  #ifdef HTP_DEBUG
76  fprint_raw_data(stderr, "Response protocol", bstr_ptr(tx->response_protocol), bstr_len(tx->response_protocol));
77  fprintf(stderr, "Response protocol number: %d\n", tx->response_protocol_number);
78  #endif
79 
80  // Ignore whitespace after the response protocol.
81  while ((pos < len) && (htp_is_space(data[pos]))) pos++;
82  if (pos == len) return HTP_OK;
83 
84  start = pos;
85 
86  // Find the next whitespace character.
87  while ((pos < len) && (!htp_is_space(data[pos]))) pos++;
88  if (pos - start == 0) return HTP_OK;
89 
90  tx->response_status = bstr_dup_mem(data + start, pos - start);
91  if (tx->response_status == NULL) return HTP_ERROR;
92 
93  tx->response_status_number = htp_parse_status(tx->response_status);
94 
95  #ifdef HTP_DEBUG
96  fprint_raw_data(stderr, "Response status (as text)", bstr_ptr(tx->response_status), bstr_len(tx->response_status));
97  fprintf(stderr, "Response status number: %d\n", tx->response_status_number);
98  #endif
99 
100  // Ignore whitespace that follows the status code.
101  while ((pos < len) && (isspace(data[pos]))) pos++;
102  if (pos == len) return HTP_OK;
103 
104  // Assume the message stretches until the end of the line.
105  tx->response_message = bstr_dup_mem(data + pos, len - pos);
106  if (tx->response_message == NULL) return HTP_ERROR;
107 
108  #ifdef HTP_DEBUG
109  fprint_raw_data(stderr, "Response status message", bstr_ptr(tx->response_message), bstr_len(tx->response_message));
110  #endif
111 
112  return HTP_OK;
113 }

115 /**
116  * Generic response header parser.
117  *
118  * @param[in] connp
119  * @param[in] h
120  * @param[in] data
121  * @param[in] len
122  * @return HTP status
123  */
124 htp_status_t htp_parse_response_header_generic(htp_connp_t *connp, htp_header_t *h, unsigned char *data, size_t len) {
125  size_t name_start, name_end;
126  size_t value_start, value_end;
127  size_t prev;
128 
129  htp_chomp(data, &len);
130 
131  name_start = 0;
132 
133  // Look for the first colon.
134  size_t colon_pos = 0;
135  while ((colon_pos < len) && (data[colon_pos] != ':')) colon_pos++;
136 
137  if (colon_pos == len) {
138  // Header line with a missing colon.
139 
140  h->flags |= HTP_FIELD_UNPARSEABLE;
141  h->flags |= HTP_FIELD_INVALID;
142 
143  if (!(connp->out_tx->flags & HTP_FIELD_UNPARSEABLE)) {
144  // Only once per transaction.
145  connp->out_tx->flags |= HTP_FIELD_UNPARSEABLE;
146  connp->out_tx->flags |= HTP_FIELD_INVALID;
147  htp_log(connp, HTP_LOG_MARK, HTP_LOG_WARNING, 0, "Response field invalid: missing colon.");
148  }
149 
150  // Reset the position. We're going to treat this invalid header
151  // as a header with an empty name. That will increase the probability
152  // that the content will be inspected.
153  colon_pos = 0;
154  name_end = 0;
155  value_start = 0;
156  } else {
157  // Header line with a colon.
158 
159  if (colon_pos == 0) {
160  // Empty header name.
161 
162  h->flags |= HTP_FIELD_INVALID;
163 
164  if (!(connp->out_tx->flags & HTP_FIELD_INVALID)) {
165  // Only once per transaction.
166  connp->out_tx->flags |= HTP_FIELD_INVALID;
167  htp_log(connp, HTP_LOG_MARK, HTP_LOG_WARNING, 0, "Response field invalid: empty name.");
168  }
169  }
170 
171  name_end = colon_pos;
172 
173  // Ignore LWS after field-name.
174  prev = name_end;
175  while ((prev > name_start) && (htp_is_lws(data[prev - 1]))) {
176  prev--;
177  name_end--;
178 
179  h->flags |= HTP_FIELD_INVALID;
180 
181  if (!(connp->out_tx->flags & HTP_FIELD_INVALID)) {
182  // Only once per transaction.
183  connp->out_tx->flags |= HTP_FIELD_INVALID;
184  htp_log(connp, HTP_LOG_MARK, HTP_LOG_WARNING, 0, "Response field invalid: LWS after name.");
185  }
186  }
187 
188  value_start = colon_pos + 1;
189  }
190 
191  // Header value.
192 
193  // Ignore LWS before field-content.
194  while ((value_start < len) && (htp_is_lws(data[value_start]))) {
195  value_start++;
196  }
197 
198  // Look for the end of field-content.
199  value_end = len;
200 
201  // Check that the header name is a token.
202  size_t i = name_start;
203  while (i < name_end) {
204  if (!htp_is_token(data[i])) {
205  h->flags |= HTP_FIELD_INVALID;
206 
207  if (!(connp->out_tx->flags & HTP_FIELD_INVALID)) {
208  connp->out_tx->flags |= HTP_FIELD_INVALID;
209  htp_log(connp, HTP_LOG_MARK, HTP_LOG_WARNING, 0, "Response header name is not a token.");
210  }
211 
212  break;
213  }
214 
215  i++;
216  }
217 
218  // Now extract the name and the value.
219  h->name = bstr_dup_mem(data + name_start, name_end - name_start);
220  h->value = bstr_dup_mem(data + value_start, value_end - value_start);
221  if ((h->name == NULL) || (h->value == NULL)) {
222  bstr_free(h->name);
223  bstr_free(h->value);
224  return HTP_ERROR;
225  }
226 
227  return HTP_OK;
228 }

230 /**
231  * Generic response header line(s) processor, which assembles folded lines
232  * into a single buffer before invoking the parsing function.
233  *
234  * @param[in] connp
235  * @param[in] data
236  * @param[in] len
237  * @return HTP status
238  */
239 htp_status_t htp_process_response_header_generic(htp_connp_t *connp, unsigned char *data, size_t len) {
240  // Create a new header structure.
241  htp_header_t *h = calloc(1, sizeof (htp_header_t));
242  if (h == NULL) return HTP_ERROR;
243 
244  if (htp_parse_response_header_generic(connp, h, data, len) != HTP_OK) {
245  free(h);
246  return HTP_ERROR;
247  }
248 
249  #ifdef HTP_DEBUG
250  fprint_bstr(stderr, "Header name", h->name);
251  fprint_bstr(stderr, "Header value", h->value);
252  #endif
253 
254  // Do we already have a header with the same name?
255  htp_header_t *h_existing = htp_table_get(connp->out_tx->response_headers, h->name);
256  if (h_existing != NULL) {
257  // Keep track of repeated same-name headers.
258  h_existing->flags |= HTP_FIELD_REPEATED;
259 
260  // Having multiple C-L headers is against the RFC but many
261  // browsers ignore the subsequent headers if the values are the same.
262  if (bstr_cmp_c_nocase(h->name, "Content-Length") == 0) {
263  // Don't use string comparison here because we want to
264  // ignore small formatting differences.
265 
266  int64_t existing_cl, new_cl;
267 
268  existing_cl = htp_parse_content_length(h_existing->value);
269  new_cl = htp_parse_content_length(h->value);
270  if ((existing_cl == -1) || (new_cl == -1) || (existing_cl != new_cl)) {
271  // Ambiguous response C-L value.
272  htp_log(connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0, "Ambiguous response C-L value");
273 
274  bstr_free(h->name);
275  bstr_free(h->value);
276  free(h);
277 
278  return HTP_ERROR;
279  }
280 
281  // Ignoring the new C-L header that has the same value as the previous ones.
282  } else {
283  // Add to the existing header.
284 
285  bstr *new_value = bstr_expand(h_existing->value, bstr_len(h_existing->value) + 2 + bstr_len(h->value));
286  if (new_value == NULL) {
287  bstr_free(h->name);
288  bstr_free(h->value);
289  free(h);
290  return HTP_ERROR;
291  }
292 
293  h_existing->value = new_value;
294  bstr_add_mem_noex(h_existing->value, (unsigned char *) ", ", 2);
295  bstr_add_noex(h_existing->value, h->value);
296  }
297 
298  // The new header structure is no longer needed.
299  bstr_free(h->name);
300  bstr_free(h->value);
301  free(h);
302  } else {
303  // Add as a new header.
304  if (htp_table_add(connp->out_tx->response_headers, h->name, h) != HTP_OK) {
305  bstr_free(h->name);
306  bstr_free(h->value);
307  free(h);
308  return HTP_ERROR;
309  }
310  }
311 
312  return HTP_OK;
313 }
