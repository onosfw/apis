48 static void StreamMsgEnqueue (StreamMsgQueue *q, StreamMsg *s)
49 {
50  SCEnter();
51  SCLogDebug("s %p", s);
52  /* more packets in queue */
53  if (q->top != NULL) {
54  s->next = q->top;
55  q->top->prev = s;
56  q->top = s;
57  /* only packet */
58  } else {
59  q->top = s;
60  q->bot = s;
61  }
62  q->len++;
63 #ifdef DBG_PERF
64  if (q->len > q->dbg_maxlen)
65  q->dbg_maxlen = q->len;
66 #endif /* DBG_PERF */
67  SCReturn;
68 }

70 static StreamMsg *StreamMsgDequeue (StreamMsgQueue *q)
71 {
72  SCEnter();
73 
74  /* if the queue is empty there are no packets left.
75  * In that case we sleep and try again. */
76  if (q->len == 0) {
77  SCReturnPtr(NULL, "StreamMsg");
78  }
79 
80  /* pull the bottom packet from the queue */
81  StreamMsg *s = q->bot;
82 
83  /* more packets in queue */
84  if (q->bot->prev != NULL) {
85  q->bot = q->bot->prev;
86  q->bot->next = NULL;
87  /* just the one we remove, so now empty */
88  } else {
89  q->top = NULL;
90  q->bot = NULL;
91  }
92  q->len--;
93 
94  s->next = NULL;
95  s->prev = NULL;
96  SCReturnPtr(s, "StreamMsg");
97 }

255 /** \brief Return a list of smsgs to the pool */
256 void StreamMsgReturnListToPool(void *list)
257 {
258  /* if we have (a) smsg(s), return to the pool */
259  StreamMsg *smsg = (StreamMsg *)list;
260  while (smsg != NULL) {
261  StreamMsg *smsg_next = smsg->next;
262  SCLogDebug("returning smsg %p to pool", smsg);
263  smsg->next = NULL;
264  smsg->prev = NULL;
265  StreamMsgReturnToPool(smsg);
266  smsg = smsg_next;
267  }
