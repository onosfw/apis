189 /** \brief Convert time_t into Year, month, day, hour and minutes.
190  * \param timep Time in seconds since defined date.
191  * \param result The structure into which the broken down time it put.
192  *
193  * To convert a time in seconds into year, month, day, hours, minutes
194  * and seconds, call localtime_r(), which uses the current time zone
195  * to compute these values. Note, glibc's localtime_r() aquires a lock
196  * each time it is called, which limits parallelism. To call
197  * localtime_r() less often, the values returned are cached for the
198  * current and previous minute and then seconds are adjusted to
199  * compute the returned result. This is valid as long as the
200  * difference between the start of the current minute and the current
201  * time is less than 60 seconds. Once the minute value changes, all
202  * the other values could change.
203  *
204  * Two values are cached to prevent thrashing when changing from one
205  * minute to the next. The two cached minutes are independent and are
206  * not required to be M and M+1. If more than two minutes are
207  * requested, the least-recently-used cached value is updated more
208  * often, the results are still correct, but performance will be closer
209  * to previous performance.
210  */
211 struct tm *SCLocalTime(time_t timep, struct tm *result)
212 {
213  /* Only get a new local time when the time crosses into a new
214  * minute. */
215  int mru = mru_tm_slot;
216  int lru = 1 - mru;
217  int mru_seconds = timep - cached_minute_start[mru];
218  int lru_seconds = timep - cached_minute_start[lru];
219  int new_seconds;
220  if (mru_seconds >= 0 && mru_seconds <= 59) {
221  /* Use most-recently cached time, adjusting the seconds. */
222  new_seconds = mru_seconds;
223  } else if (lru_seconds >= 0 && lru_seconds <= 59) {
224  /* Use least-recently cached time, update to most recently used. */
225  new_seconds = lru_seconds;
226  mru = lru;
227  mru_tm_slot = mru;
228  } else {
229  /* Update least-recent cached time. */
230  if (localtime_r(&timep, &cached_local_tm[lru]) == NULL)
231  return NULL;
232 
233  /* Subtract seconds to get back to the start of the minute. */
234  new_seconds = cached_local_tm[lru].tm_sec;
235  cached_minute_start[lru] = timep - new_seconds;
236  mru = lru;
237  mru_tm_slot = mru;
238  }
239  memcpy(result, &cached_local_tm[mru], sizeof(struct tm));
240  result->tm_sec = new_seconds;
241 
242  return result;
243 }

245 /* Update the cached time string in cache index N, for the current minute. */
246 static int UpdateCachedTime(int n, time_t time)
247 {
248  struct tm local_tm;
249  struct tm *t = (struct tm *)SCLocalTime(time, &local_tm);
250  int cached_len = snprintf(cached_local_time[n], MAX_LOCAL_TIME_STRING,
251  "%02d/%02d/%02d-%02d:%02d:",
252  t->tm_mon + 1, t->tm_mday, t->tm_year + 1900,
253  t->tm_hour, t->tm_min);
254  cached_local_time_len[n] = cached_len;
255  /* Store the time of the beginning of the minute. */
256  last_local_time[n] = time - t->tm_sec;
257  mru_time_slot = n;
258 
259  return t->tm_sec;
260 }

262 /** \brief Return a formatted string for the provided time.
263  *
264  * Cache the Month/Day/Year - Hours:Min part of the time string for
265  * the current minute. Copy that result into the the return string and
266  * then only print the seconds for each call.
267  */
268 void CreateTimeString (const struct timeval *ts, char *str, size_t size)
269 {
270  time_t time = ts->tv_sec;
271  int seconds;
272 
273  /* Only get a new local time when the time crosses into a new
274  * minute */
275  int mru = mru_time_slot;
276  int lru = 1 - mru;
277  int mru_seconds = time - last_local_time[mru];
278  int lru_seconds = time - last_local_time[lru];
279  if (mru_seconds >= 0 && mru_seconds <= 59) {
280  /* Use most-recently cached time. */
281  seconds = mru_seconds;
282  } else if (lru_seconds >= 0 && lru_seconds <= 59) {
283  /* Use least-recently cached time. Change this slot to Most-recent */
284  seconds = lru_seconds;
285  mru_time_slot = lru;
286  } else {
287  /* Update least-recent cached time. Lock accessing local time
288  * function because it keeps any internal non-spin lock. */
289  seconds = UpdateCachedTime(lru, time);
290  }
291 
292  /* Copy the string up to the current minute then print the seconds
293  into the return string buffer. */
294  char *cached_str = cached_local_time[mru_time_slot];
295  int cached_len = cached_local_time_len[mru_time_slot];
296  if (cached_len >= (int)size)
297  cached_len = size;
298  memcpy(str, cached_str, cached_len);
299  snprintf(str + cached_len, size - cached_len,
300  "%02d.%06u",
301  seconds, (uint32_t) ts->tv_usec);
302 }
