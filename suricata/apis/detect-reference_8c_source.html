104 /**
105  * \internal
106  * \brief This function is used to parse reference options passed via reference: keyword
107  *
108  * \param rawstr Pointer to the user provided reference options.
109  *
110  * \retval ref Pointer to signature reference on success.
111  * \retval NULL On failure.
112  */
113 static DetectReference *DetectReferenceParse(char *rawstr, DetectEngineCtx *de_ctx)
114 {
115  SCEnter();
116 
117  DetectReference *ref = NULL;
118 #define MAX_SUBSTRINGS 30
119  int ret = 0, res = 0;
120  int ov[MAX_SUBSTRINGS];
121  char key[64] = "";
122  char content[1024] = "";
123 
124  ret = pcre_exec(parse_regex, parse_regex_study, rawstr, strlen(rawstr),
125  0, 0, ov, MAX_SUBSTRINGS);
126  if (ret < 2) {
127  SCLogError(SC_ERR_INVALID_SIGNATURE, "Unable to parse \"reference\" "
128  "keyword argument - \"%s\". Invalid argument.", rawstr);
129  goto error;
130  }
131 
132  ref = SCMalloc(sizeof(DetectReference));
133  if (unlikely(ref == NULL)) {
134  goto error;
135  }
136  memset(ref, 0, sizeof(DetectReference));
137 
138  res = pcre_copy_substring((char *)rawstr, ov, MAX_SUBSTRINGS, 1, key, sizeof(key));
139  if (res < 0) {
140  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_copy_substring failed");
141  goto error;
142  }
143 
144  res = pcre_copy_substring((char *)rawstr, ov, MAX_SUBSTRINGS, 2, content, sizeof(content));
145  if (res < 0) {
146  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
147  goto error;
148  }
149 
150  if (strlen(key) == 0 || strlen(content) == 0)
151  goto error;
152 
153  SCRConfReference *lookup_ref_conf = SCRConfGetReference(key, de_ctx);
154  if (lookup_ref_conf != NULL) {
155  ref->key = lookup_ref_conf->url;
156  } else {
157  SCLogError(SC_ERR_REFERENCE_UNKNOWN, "unknown reference key \"%s\". "
158  "Supported keys are defined in reference.config file. Please "
159  "have a look at the conf param \"reference-config-file\"", key);
160  goto error;
161  }
162 
163  /* make a copy so we can free pcre's substring */
164  ref->reference = SCStrdup((char *)content);
165  if (ref->reference == NULL) {
166  SCLogError(SC_ERR_MEM_ALLOC, "strdup failed: %s", strerror(errno));
167  goto error;
168  }
169 
170  /* free the substrings */
171  SCReturnPtr(ref, "Reference");
172 
173 error:
174  if (ref != NULL)
175  DetectReferenceFree(ref);
176 
177  SCReturnPtr(NULL, "Reference");
178 }

180 /**
181  * \internal
182  * \brief Used to add the parsed reference into the current signature.
183  *
184  * \param de_ctx Pointer to the Detection Engine Context.
185  * \param s Pointer to the Current Signature.
186  * \param m Pointer to the Current SigMatch.
187  * \param rawstr Pointer to the user provided reference options.
188  *
189  * \retval 0 On Success.
190  * \retval -1 On Failure.
191  */
192 static int DetectReferenceSetup(DetectEngineCtx *de_ctx, Signature *s,
193  char *rawstr)
194 {
195  SCEnter();
196 
197  DetectReference *ref = NULL;
198  DetectReference *sig_refs = NULL;
199 
200  ref = DetectReferenceParse(rawstr, de_ctx);
201  if (ref == NULL)
202  goto error;
203 
204  SCLogDebug("ref %s %s", ref->key, ref->reference);
205 
206  if (s->references == NULL) {
207  s->references = ref;
208  } else {
209  sig_refs = s->references;
210  while (sig_refs->next != NULL) {
211  sig_refs = sig_refs->next;
212  }
213  sig_refs->next = ref;
214  ref->next = NULL;
215  }
216 
217  SCReturnInt(0);
218 
219 error:
220  SCReturnInt(-1);
221 }
