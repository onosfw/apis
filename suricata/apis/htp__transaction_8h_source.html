48 /**
49  * Enumerate possible data handling strategies in hybrid parsing
50  * mode. The two possibilities are to make copies of all data and
51  * use bstr instances to wrap already available data.
52  */
53 enum htp_alloc_strategy_t {
54  /**
55  * Make copies of all data. This strategy should be used when
56  * the supplied buffers are transient and will go away after
57  * the invoked function returns.
58  */
59  HTP_ALLOC_COPY = 1,

62  * Reuse buffers, without a change of ownership. We assume the
63  * buffers will continue to be available until the transaction
64  * is deleted by the container.
65  */
66  HTP_ALLOC_REUSE = 2
67 };

69 /**

71  * to the next state when the processing activities associated with that state

74  */

77  HTP_REQUEST_LINE = 1,
78  HTP_REQUEST_HEADERS = 2,
79  HTP_REQUEST_BODY = 3,
80  HTP_REQUEST_TRAILER = 4,
81  HTP_REQUEST_COMPLETE = 5
82 };

303  * htp_txh_set_req_protocol(), because it will overwrite the previously
