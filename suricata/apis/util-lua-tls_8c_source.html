60 static int GetCertInfo(lua_State *luastate, const Flow *f, int direction)
61 {
62  void *state = FlowGetAppState(f);
63  if (state == NULL)
64  return LuaCallbackError(luastate, "error: no app layer state");
65 
66  SSLState *ssl_state = (SSLState *)state;
67  SSLStateConnp *connp = NULL;
68 
69  if (direction) {
70  connp = &ssl_state->client_connp;
71  } else {
72  connp = &ssl_state->server_connp;
73  }
74 
75  if (connp->cert0_subject == NULL)
76  return LuaCallbackError(luastate, "error: no cert");
77 
78  /* tls.version */
79  char ssl_version[32] = "";
80  switch (ssl_state->server_connp.version) {
81  case TLS_VERSION_UNKNOWN:
82  snprintf(ssl_version, sizeof(ssl_version), "UNDETERMINED");
83  break;
84  case SSL_VERSION_2:
85  snprintf(ssl_version, sizeof(ssl_version), "SSLv2");
86  break;
87  case SSL_VERSION_3:
88  snprintf(ssl_version, sizeof(ssl_version), "SSLv3");
89  break;
90  case TLS_VERSION_10:
91  snprintf(ssl_version, sizeof(ssl_version), "TLSv1");
92  break;
93  case TLS_VERSION_11:
94  snprintf(ssl_version, sizeof(ssl_version), "TLS 1.1");
95  break;
96  case TLS_VERSION_12:
97  snprintf(ssl_version, sizeof(ssl_version), "TLS 1.2");
98  break;
99  default:
100  snprintf(ssl_version, sizeof(ssl_version), "0x%04x",
101  ssl_state->server_connp.version);
102  break;
103  }
104 
105  int r = LuaPushStringBuffer(luastate, (uint8_t *)ssl_version, strlen(ssl_version));
106  r += LuaPushStringBuffer(luastate, (uint8_t *)connp->cert0_subject, strlen(connp->cert0_subject));
107  r += LuaPushStringBuffer(luastate, (uint8_t *)connp->cert0_issuerdn, strlen(connp->cert0_issuerdn));
108  r += LuaPushStringBuffer(luastate, (uint8_t *)connp->cert0_fingerprint, strlen(connp->cert0_fingerprint));
109  return r;
110 }

112 static int TlsGetCertInfo(lua_State *luastate)
113 {
114  int r;
115 
116  if (!(LuaStateNeedProto(luastate, ALPROTO_TLS)))
117  return LuaCallbackError(luastate, "error: protocol not tls");
118 
119  int direction = LuaStateGetDirection(luastate);
120 
121  int lock_hint = 0;
122  Flow *f = LuaStateGetFlow(luastate, &lock_hint);
123  if (f == NULL)
124  return LuaCallbackError(luastate, "internal error: no flow");
125 
126  if (lock_hint == LUA_FLOW_NOT_LOCKED_BY_PARENT) {
127  FLOWLOCK_RDLOCK(f);
128  r = GetCertInfo(luastate, f, direction);
129  FLOWLOCK_UNLOCK(f);
130  } else {
131  r = GetCertInfo(luastate, f, direction);
132  }
133  return r;
134 }
