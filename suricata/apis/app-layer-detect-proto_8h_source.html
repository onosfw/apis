112 /**
113  * \brief Registers a protocol for protocol detection phase.
114  *
115  * This is the first function to be called after calling the
116  * setup function, AppLayerProtoDetectSetup(), before calling any other
117  * app layer functions, AppLayerParser or AppLayerProtoDetect, alike.
118  * With this function you are associating/registering a string
119  * that can be used by users to write rules, i.e.
120  * you register the http protocol for protocol detection using
121  * AppLayerProtoDetectRegisterProtocol(ctx, ALPROTO_HTTP, "http"),
122  * following which you can write rules like -
123  * alert http any any -> any any (sid:1;)
124  * which basically matches on the HTTP protocol.
125  *
126  * \param alproto The protocol.
127  * \param alproto_str The string to associate with the above "alproto".
128  * Please send a static string that won't be destroyed
129  * post making this call, since this function won't
130  * create a copy of the received argument.
131  *
132  * \retval 0 On success;
133  * -1 On failure.
134  */
135 void AppLayerProtoDetectRegisterProtocol(AppProto alproto, char *alproto_name);
