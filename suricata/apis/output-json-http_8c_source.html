181 void JsonHttpLogJSONBasic(json_t *js, htp_tx_t *tx)
182 {
183  char *c;
184 
185  /* hostname */
186  if (tx->request_hostname != NULL)
187  {
188  c = bstr_util_strdup_to_c(tx->request_hostname);
189  if (c != NULL) {
190  json_object_set_new(js, "hostname", json_string(c));
191  SCFree(c);
192  }
193  }
194 
195  /* uri */
196  if (tx->request_uri != NULL)
197  {
198  c = bstr_util_strdup_to_c(tx->request_uri);
199  if (c != NULL) {
200  json_object_set_new(js, "url", json_string(c));
201  SCFree(c);
202  }
203  }
204 
205  /* user agent */
206  htp_header_t *h_user_agent = NULL;
207  if (tx->request_headers != NULL) {
208  h_user_agent = htp_table_get_c(tx->request_headers, "user-agent");
209  }
210  if (h_user_agent != NULL) {
211  c = bstr_util_strdup_to_c(h_user_agent->value);
212  if (c != NULL) {
213  json_object_set_new(js, "http_user_agent", json_string(c));
214  SCFree(c);
215  }
216  }
217 
218  /* x-forwarded-for */
219  htp_header_t *h_x_forwarded_for = NULL;
220  if (tx->request_headers != NULL) {
221  h_x_forwarded_for = htp_table_get_c(tx->request_headers, "x-forwarded-for");
222  }
223  if (h_x_forwarded_for != NULL) {
224  c = bstr_util_strdup_to_c(h_x_forwarded_for->value);
225  if (c != NULL) {
226  json_object_set_new(js, "xff", json_string(c));
227  SCFree(c);
228  }
229  }
230 
231  /* content-type */
232  htp_header_t *h_content_type = NULL;
233  if (tx->response_headers != NULL) {
234  h_content_type = htp_table_get_c(tx->response_headers, "content-type");
235  }
236  if (h_content_type != NULL) {
237  char *p;
238  c = bstr_util_strdup_to_c(h_content_type->value);
239  if (c != NULL) {
240  p = strchr(c, ';');
241  if (p != NULL)
242  *p = '\0';
243  json_object_set_new(js, "http_content_type", json_string(c));
244  SCFree(c);
245  }
246  }

249 static void JsonHttpLogJSONCustom(LogHttpFileCtx *http_ctx, json_t *js, htp_tx_t *tx)
250 {
251  char *c;
252  HttpField f;
253 
254  for (f = HTTP_FIELD_ACCEPT; f < HTTP_FIELD_SIZE; f++)
255  {
256  if ((http_ctx->fields & (1ULL<<f)) != 0)
257  {
258  /* prevent logging a field twice if extended logging is
259  enabled */
260  if (((http_ctx->flags & LOG_HTTP_EXTENDED) == 0) ||
261  ((http_ctx->flags & LOG_HTTP_EXTENDED) !=
262  (http_fields[f].flags & LOG_HTTP_EXTENDED)))
263  {
264  htp_header_t *h_field = NULL;
265  if ((http_fields[f].flags & LOG_HTTP_REQUEST) != 0)
266  {
267  if (tx->request_headers != NULL) {
268  h_field = htp_table_get_c(tx->request_headers,
269  http_fields[f].htp_field);
270  }
271  } else {
272  if (tx->response_headers != NULL) {
273  h_field = htp_table_get_c(tx->response_headers,
274  http_fields[f].htp_field);
275  }
276  }
277  if (h_field != NULL) {
278  c = bstr_util_strdup_to_c(h_field->value);
279  if (c != NULL) {
280  json_object_set_new(js,
281  http_fields[f].config_field,
282  json_string(c));
283  SCFree(c);
284  }
285  }
286  }
287  }
288  }

350 static void JsonHttpLogJSON(JsonHttpLogThread *aft, json_t *js, htp_tx_t *tx, uint64_t tx_id)
351 {
352  LogHttpFileCtx *http_ctx = aft->httplog_ctx;
353  json_t *hjs = json_object();
354  if (hjs == NULL) {
355  return;
356  }
357 
358  JsonHttpLogJSONBasic(hjs, tx);
359  /* log custom fields if configured */
360  if (http_ctx->fields != 0)
361  JsonHttpLogJSONCustom(http_ctx, hjs, tx);
362  if (http_ctx->flags & LOG_HTTP_EXTENDED)
363  JsonHttpLogJSONExtended(hjs, tx);
364 
365  json_object_set_new(js, "http", hjs);
366 }

389 
390  json_object_clear(js);
391  json_decref(js);
392 
393  SCReturnInt(TM_ECODE_OK);

427 OutputCtx *OutputHttpLogInit(ConfNode *conf)
428 {
429  LogFileCtx *file_ctx = LogFileNewCtx();
430  if(file_ctx == NULL) {
431  SCLogError(SC_ERR_HTTP_LOG_GENERIC, "couldn't create new file_ctx");
432  return NULL;
433  }
434 
435  if (SCConfLogOpenGeneric(conf, file_ctx, DEFAULT_LOG_FILENAME, 1) < 0) {
436  LogFileFreeCtx(file_ctx);
437  return NULL;
438  }
439 
440  LogHttpFileCtx *http_ctx = SCMalloc(sizeof(LogHttpFileCtx));
441  if (unlikely(http_ctx == NULL)) {
442  LogFileFreeCtx(file_ctx);
443  return NULL;
444  }
445 
446  OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
447  if (unlikely(output_ctx == NULL)) {
448  LogFileFreeCtx(file_ctx);
449  SCFree(http_ctx);
450  return NULL;
451  }
452 
453  http_ctx->file_ctx = file_ctx;
454  http_ctx->flags = LOG_HTTP_DEFAULT;
455 
456  if (conf) {
457  const char *extended = ConfNodeLookupChildValue(conf, "extended");
458 
459  if (extended != NULL) {
460  if (ConfValIsTrue(extended)) {
461  http_ctx->flags = LOG_HTTP_EXTENDED;
462  }
463  }
464  }
465  output_ctx->data = http_ctx;
466  output_ctx->DeInit = OutputHttpLogDeinit;
467 
468  /* enable the logger for the app layer */
469  AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_HTTP);
470 
471  return output_ctx;
472 }

481 OutputCtx *OutputHttpLogInitSub(ConfNode *conf, OutputCtx *parent_ctx)
482 {
483  OutputJsonCtx *ojc = parent_ctx->data;
484 
485  LogHttpFileCtx *http_ctx = SCMalloc(sizeof(LogHttpFileCtx));
486  if (unlikely(http_ctx == NULL))
487  return NULL;
488  memset(http_ctx, 0x00, sizeof(*http_ctx));
489 
490  OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
491  if (unlikely(output_ctx == NULL)) {
492  SCFree(http_ctx);
493  return NULL;
494  }
495 
496  http_ctx->file_ctx = ojc->file_ctx;
497  http_ctx->flags = LOG_HTTP_DEFAULT;
498 
499  if (conf) {
500  const char *extended = ConfNodeLookupChildValue(conf, "extended");
501 
502  if (extended != NULL) {
503  if (ConfValIsTrue(extended)) {
504  http_ctx->flags = LOG_HTTP_EXTENDED;
505  }
506  }
507 
508  ConfNode *custom;
509  if ((custom = ConfNodeLookupChild(conf, "custom")) != NULL) {
510  ConfNode *field;
511  TAILQ_FOREACH(field, &custom->head, next)
512  {
513  if (field != NULL)
514  {
515  HttpField f;
516  for (f = HTTP_FIELD_ACCEPT; f < HTTP_FIELD_SIZE; f++)
517  {
518  if ((strcmp(http_fields[f].config_field,
519  field->val) == 0) ||
520  (strcasecmp(http_fields[f].htp_field,
521  field->val) == 0))
522  {
523  http_ctx->fields |= (1ULL<<f);
524  break;
525  }
526  }
527  }
528  }
529  }
530  }
531  output_ctx->data = http_ctx;
532  output_ctx->DeInit = OutputHttpLogDeinitSub;
533 
534  /* enable the logger for the app layer */
535  AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_HTTP);
536 
537  return output_ctx;
538 }
