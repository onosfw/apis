71 static int CreateFileName(const Packet *p, SSLState *state, char *filename)
72 {
73 #define FILELEN 64 //filename len + extention + ending path / + some space
74 
75  int filenamelen = FILELEN + strlen(tls_logfile_base_dir);
76  int file_id = SC_ATOMIC_ADD(cert_id, 1);
77 
78  if (filenamelen + 1 > PATH_MAX) {
79  return 0;
80  }
81 
82  /* Use format : packet time + incremental ID
83  * When running on same pcap it will overwrite
84  * On a live device, we will not be able to overwrite */
85  snprintf(filename, filenamelen, "%s/%ld.%ld-%d.pem",
86  tls_logfile_base_dir,
87  (long int)p->ts.tv_sec,
88  (long int)p->ts.tv_usec,
89  file_id);
90  return 1;
91 }

93 static void LogTlsLogPem(LogTlsStoreLogThread *aft, const Packet *p, SSLState *state, int ipproto)
94 {
95 #define PEMHEADER "-----BEGIN CERTIFICATE-----\n"
96 #define PEMFOOTER "-----END CERTIFICATE-----\n"
97  //Logging pem certificate
98  char filename[PATH_MAX] = "";
99  FILE* fp = NULL;
100  FILE* fpmeta = NULL;
101  unsigned long pemlen;
102  unsigned char* pembase64ptr = NULL;
103  int ret;
104  uint8_t *ptmp;
105  SSLCertsChain *cert;
106 
107  if ((state->server_connp.cert_input == NULL) || (state->server_connp.cert_input_len == 0))
108  SCReturn;
109 
110  CreateFileName(p, state, filename);
111  if (strlen(filename) == 0) {
112  SCLogWarning(SC_ERR_FOPEN, "Can't create PEM filename");
113  SCReturn;
114  }
115 
116  fp = fopen(filename, "w");
117  if (fp == NULL) {
118  if (logging_dir_not_writable < LOGGING_WRITE_ISSUE_LIMIT) {
119  SCLogWarning(SC_ERR_FOPEN,
120  "Can't create PEM file '%s' in '%s' directory",
121  filename, tls_logfile_base_dir);
122  logging_dir_not_writable++;
123  }
124  SCReturn;
125  }
126 
127  TAILQ_FOREACH(cert, &state->server_connp.certs, next) {
128  pemlen = (4 * (cert->cert_len + 2) / 3) +1;
129  if (pemlen > aft->enc_buf_len) {
130  ptmp = (uint8_t*) SCRealloc(aft->enc_buf, sizeof(uint8_t) * pemlen);
131  if (ptmp == NULL) {
132  SCFree(aft->enc_buf);
133  aft->enc_buf = NULL;
134  SCLogWarning(SC_ERR_MEM_ALLOC, "Can't allocate data for base64 encoding");
135  goto end_fp;
136  }
137  aft->enc_buf = ptmp;
138  aft->enc_buf_len = pemlen;
139  }
140 
141  memset(aft->enc_buf, 0, aft->enc_buf_len);
142 
143  ret = Base64Encode((unsigned char*) cert->cert_data, cert->cert_len, aft->enc_buf, &pemlen);
144  if (ret != SC_BASE64_OK) {
145  SCLogWarning(SC_ERR_INVALID_ARGUMENTS, "Invalid return of Base64Encode function");
146  goto end_fwrite_fp;
147  }
148 
149  if (fprintf(fp, PEMHEADER) < 0)
150  goto end_fwrite_fp;
151 
152  pembase64ptr = aft->enc_buf;
153  while (pemlen > 0) {
154  size_t loffset = pemlen >= 64 ? 64 : pemlen;
155  if (fwrite(pembase64ptr, 1, loffset, fp) != loffset)
156  goto end_fwrite_fp;
157  if (fwrite("\n", 1, 1, fp) != 1)
158  goto end_fwrite_fp;
159  pembase64ptr += 64;
160  if (pemlen < 64)
161  break;
162  pemlen -= 64;
163  }
164 
165  if (fprintf(fp, PEMFOOTER) < 0)
166  goto end_fwrite_fp;
167  }
168  fclose(fp);
169 
170  //Logging certificate informations
171  memcpy(filename + (strlen(filename) - 3), "meta", 4);
172  fpmeta = fopen(filename, "w");
173  if (fpmeta != NULL) {
174  #define PRINT_BUF_LEN 46
175  char srcip[PRINT_BUF_LEN], dstip[PRINT_BUF_LEN];
176  char timebuf[64];
177  Port sp, dp;
178  CreateTimeString(&p->ts, timebuf, sizeof(timebuf));
179  if (!TLSGetIPInformations(p, srcip, PRINT_BUF_LEN, &sp, dstip, PRINT_BUF_LEN, &dp, ipproto))
180  goto end_fwrite_fpmeta;
181  if (fprintf(fpmeta, "TIME: %s\n", timebuf) < 0)
182  goto end_fwrite_fpmeta;
183  if (p->pcap_cnt > 0) {
184  if (fprintf(fpmeta, "PCAP PKT NUM: %"PRIu64"\n", p->pcap_cnt) < 0)
185  goto end_fwrite_fpmeta;
186  }
187  if (fprintf(fpmeta, "SRC IP: %s\n", srcip) < 0)
188  goto end_fwrite_fpmeta;
189  if (fprintf(fpmeta, "DST IP: %s\n", dstip) < 0)
190  goto end_fwrite_fpmeta;
191  if (fprintf(fpmeta, "PROTO: %" PRIu32 "\n", p->proto) < 0)
192  goto end_fwrite_fpmeta;
193  if (PKT_IS_TCP(p) || PKT_IS_UDP(p)) {
194  if (fprintf(fpmeta, "SRC PORT: %" PRIu16 "\n", sp) < 0)
195  goto end_fwrite_fpmeta;
196  if (fprintf(fpmeta, "DST PORT: %" PRIu16 "\n", dp) < 0)
197  goto end_fwrite_fpmeta;
198  }
199 
200  if (fprintf(fpmeta, "TLS SUBJECT: %s\n"
201  "TLS ISSUERDN: %s\n"
202  "TLS FINGERPRINT: %s\n",
203  state->server_connp.cert0_subject,
204  state->server_connp.cert0_issuerdn,
205  state->server_connp.cert0_fingerprint) < 0)
206  goto end_fwrite_fpmeta;
207 
208  fclose(fpmeta);
209  } else {
210  if (logging_dir_not_writable < LOGGING_WRITE_ISSUE_LIMIT) {
211  SCLogWarning(SC_ERR_FOPEN,
212  "Can't create meta file '%s' in '%s' directory",
213  filename, tls_logfile_base_dir);
214  logging_dir_not_writable++;
215  }
216  SCReturn;
217  }
218 
219  /* Reset the store flag */
220  state->server_connp.cert_log_flag &= ~SSL_TLS_LOG_PEM;
221  SCReturn;
222 
223 end_fwrite_fp:
224  fclose(fp);
225  if (logging_dir_not_writable < LOGGING_WRITE_ISSUE_LIMIT) {
226  SCLogWarning(SC_ERR_FWRITE, "Unable to write certificate");
227  logging_dir_not_writable++;
228  }
229 end_fwrite_fpmeta:
230  if (fpmeta) {
231  fclose(fpmeta);
232  if (logging_dir_not_writable < LOGGING_WRITE_ISSUE_LIMIT) {
233  SCLogWarning(SC_ERR_FWRITE, "Unable to write certificate metafile");
234  logging_dir_not_writable++;
235  }
236  }
237  SCReturn;
238 end_fp:
239  fclose(fp);
240  SCReturn;
241 }

243 /** \internal
244  * \brief Condition function for TLS logger
245  * \retval bool true or false -- log now?
246  */
247 static int LogTlsStoreCondition(ThreadVars *tv, const Packet *p)
248 {
249  if (p->flow == NULL) {
250  return FALSE;
251  }
252 
253  if (!(PKT_IS_TCP(p))) {
254  return FALSE;
255  }
256 
257  FLOWLOCK_RDLOCK(p->flow);
258  uint16_t proto = FlowGetAppProtocol(p->flow);
259  if (proto != ALPROTO_TLS)
260  goto dontlog;
261 
262  SSLState *ssl_state = (SSLState *)FlowGetAppState(p->flow);
263  if (ssl_state == NULL) {
264  SCLogDebug("no tls state, so no request logging");
265  goto dontlog;
266  }
267 
268  /* we only log the state once if we don't have to write
269  * the cert due to tls.store keyword. */
270  if (!(ssl_state->server_connp.cert_log_flag & SSL_TLS_LOG_PEM) &&
271  (ssl_state->flags & SSL_AL_FLAG_STATE_STORED))
272  goto dontlog;
273 
274  if (ssl_state->server_connp.cert0_issuerdn == NULL ||
275  ssl_state->server_connp.cert0_subject == NULL)
276  goto dontlog;
277 
278  FLOWLOCK_UNLOCK(p->flow);
279  return TRUE;
280 dontlog:
281  FLOWLOCK_UNLOCK(p->flow);
282  return FALSE;
283 }

285 static int LogTlsStoreLogger(ThreadVars *tv, void *thread_data, const Packet *p)
286 {
287  LogTlsStoreLogThread *aft = (LogTlsStoreLogThread *)thread_data;
288  int ipproto = (PKT_IS_IPV4(p)) ? AF_INET : AF_INET6;
289  /* check if we have TLS state or not */
290  FLOWLOCK_WRLOCK(p->flow);
291  uint16_t proto = FlowGetAppProtocol(p->flow);
292  if (proto != ALPROTO_TLS)
293  goto end;
294 
295  SSLState *ssl_state = (SSLState *)FlowGetAppState(p->flow);
296  if (unlikely(ssl_state == NULL)) {
297  goto end;
298  }
299 
300  if (ssl_state->server_connp.cert_log_flag & SSL_TLS_LOG_PEM) {
301  LogTlsLogPem(aft, p, ssl_state, ipproto);
302  }
303 
304  /* we only store the state once */
305  ssl_state->flags |= SSL_AL_FLAG_STATE_STORED;
306 end:
307  FLOWLOCK_UNLOCK(p->flow);
308  return 0;
309 }

311 static TmEcode LogTlsStoreLogThreadInit(ThreadVars *t, void *initdata, void **data)
312 {
313  LogTlsStoreLogThread *aft = SCMalloc(sizeof(LogTlsStoreLogThread));
314  if (unlikely(aft == NULL))
315  return TM_ECODE_FAILED;
316  memset(aft, 0, sizeof(LogTlsStoreLogThread));
317 
318  if (initdata == NULL) {
319  SCLogDebug("Error getting context for LogTlsStore. \"initdata\" argument NULL");
320  SCFree(aft);
321  return TM_ECODE_FAILED;
322  }
323 
324  struct stat stat_buf;
325  if (stat(tls_logfile_base_dir, &stat_buf) != 0) {
326  int ret;
327  ret = mkdir(tls_logfile_base_dir, S_IRWXU|S_IXGRP|S_IRGRP);
328  if (ret != 0) {
329  int err = errno;
330  if (err != EEXIST) {
331  SCLogError(SC_ERR_LOGDIR_CONFIG,
332  "Cannot create certs drop directory %s: %s",
333  tls_logfile_base_dir, strerror(err));
334  exit(EXIT_FAILURE);
335  }
336  } else {
337  SCLogInfo("Created certs drop directory %s",
338  tls_logfile_base_dir);
339  }
340 
341  }
342 
343  *data = (void *)aft;
344  return TM_ECODE_OK;
345 }
