108 /**
109  * \brief ERF file reading loop.
110  */
111 TmEcode ReceiveErfFileLoop(ThreadVars *tv, void *data, void *slot)
112 {
113  Packet *p = NULL;
114  ErfFileThreadVars *etv = (ErfFileThreadVars *)data;
115 
116  etv->slot = ((TmSlot *)slot)->slot_next;
117 
118  while (1) {
119  if (suricata_ctl_flags & (SURICATA_STOP | SURICATA_KILL)) {
120  SCReturnInt(TM_ECODE_OK);
121  }
122 
123  /* Make sure we have at least one packet in the packet pool,
124  * to prevent us from alloc'ing packets at line rate. */
125  PacketPoolWait();
126 
127  p = PacketGetFromQueueOrAlloc();
128  if (unlikely(p == NULL)) {
129  SCLogError(SC_ERR_MEM_ALLOC, "Failed to allocate a packet.");
130  EngineStop();
131  SCReturnInt(TM_ECODE_FAILED);
132  }
133  PKT_SET_SRC(p, PKT_SRC_WIRE);
134 
135  if (ReadErfRecord(tv, p, data) != TM_ECODE_OK) {
136  TmqhOutputPacketpool(etv->tv, p);
137  EngineStop();
138  SCReturnInt(TM_ECODE_FAILED);
139  }
140 
141  if (TmThreadsSlotProcessPkt(etv->tv, etv->slot, p) != TM_ECODE_OK) {
142  EngineStop();
143  SCReturnInt(TM_ECODE_FAILED);
144  }
145  }
146  SCReturnInt(TM_ECODE_FAILED);
147 }

149 static inline TmEcode ReadErfRecord(ThreadVars *tv, Packet *p, void *data)
150 {
151  SCEnter();
152 
153  ErfFileThreadVars *etv = (ErfFileThreadVars *)data;
154  DagRecord dr;
155 
156  int r = fread(&dr, sizeof(DagRecord), 1, etv->erf);
157  if (r < 1) {
158  if (feof(etv->erf)) {
159  SCLogInfo("End of ERF file reached");
160  }
161  else {
162  SCLogInfo("Error reading ERF record");
163  }
164  SCReturnInt(TM_ECODE_FAILED);
165  }
166  int rlen = ntohs(dr.rlen);
167  int wlen = ntohs(dr.wlen);
168  r = fread(GET_PKT_DATA(p), rlen - sizeof(DagRecord), 1, etv->erf);
169  if (r < 1) {
170  if (feof(etv->erf)) {
171  SCLogInfo("End of ERF file reached");
172  }
173  else {
174  SCLogInfo("Error reading ERF record");
175  }
176  SCReturnInt(TM_ECODE_FAILED);
177  }
178 
179  /* Only support ethernet at this time. */
180  if (dr.type != DAG_TYPE_ETH) {
181  SCLogError(SC_ERR_UNIMPLEMENTED,
182  "DAG record type %d not implemented.", dr.type);
183  SCReturnInt(TM_ECODE_FAILED);
184  }
185 
186  GET_PKT_LEN(p) = wlen;
187  p->datalink = LINKTYPE_ETHERNET;
188 
189  /* Convert ERF time to timeval - from libpcap. */
190  uint64_t ts = dr.ts;
191  p->ts.tv_sec = ts >> 32;
192  ts = (ts & 0xffffffffULL) * 1000000;
193  ts += 0x80000000; /* rounding */
194  p->ts.tv_usec = ts >> 32;
195  if (p->ts.tv_usec >= 1000000) {
196  p->ts.tv_usec -= 1000000;
197  p->ts.tv_sec++;
198  }
199 
200  etv->pkts++;
201  etv->bytes += wlen;
202 
203  SCReturnInt(TM_ECODE_OK);
204 }

209 TmEcode
210 ReceiveErfFileThreadInit(ThreadVars *tv, void *initdata, void **data)
211 {
212  SCEnter();
213 
214  if (initdata == NULL) {
215  SCLogError(SC_ERR_INVALID_ARGUMENT, "Error: No filename provided.");
216  SCReturnInt(TM_ECODE_FAILED);
217  }
218 
219  FILE *erf = fopen((const char *)initdata, "r");
220  if (erf == NULL) {
221  SCLogError(SC_ERR_FOPEN, "Failed to open %s: %s", (char *)initdata,
222  strerror(errno));
223  exit(EXIT_FAILURE);
224  }
225 
226  ErfFileThreadVars *etv = SCMalloc(sizeof(ErfFileThreadVars));
227  if (unlikely(etv == NULL)) {
228  SCLogError(SC_ERR_MEM_ALLOC, "Failed to allocate memory for ERF file thread vars.");
229  fclose(erf);
230  SCReturnInt(TM_ECODE_FAILED);
231  }
232  memset(etv, 0, sizeof(*etv));
233  etv->erf = erf;
234  etv->tv = tv;
235  *data = (void *)etv;
236 
237  SCLogInfo("Processing ERF file %s", (char *)initdata);
238 
239  SCReturnInt(TM_ECODE_OK);
240 }

269 /**
270  * \brief Decode the ERF file.
271  *
272  * This function ups the decoder counters and then passes the packet
273  * off to the ethernet decoder.
274  */
275 TmEcode
276 DecodeErfFile(ThreadVars *tv, Packet *p, void *data, PacketQueue *pq, PacketQueue *postpq)
277 {
278  SCEnter();
279  DecodeThreadVars *dtv = (DecodeThreadVars *)data;
280 
281  /* XXX HACK: flow timeout can call us for injected pseudo packets
282  * see bug: https://redmine.openinfosecfoundation.org/issues/1107 */
283  if (p->flags & PKT_PSEUDO_STREAM_END)
284  return TM_ECODE_OK;
285 
286  /* Update counters. */
287  DecodeUpdatePacketCounters(tv, dtv, p);
288 
289  DecodeEthernet(tv, dtv, p, GET_PKT_DATA(p), GET_PKT_LEN(p), pq);
290 
291  PacketDecodeFinalize(tv, dtv, p);
292 
293  SCReturnInt(TM_ECODE_OK);
294 }
