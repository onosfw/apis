18 /**
19  * \file
20  *
21  * \author Ken Steele <suricata@tilera.com>
22  * \author Anoop Saldanha <anoopsaldanha@gmail.com>
23  *
24  * Aho-corasick MPM optimized for the Tilera Tile-Gx architecture.
25  *
26  * Efficient String Matching: An Aid to Bibliographic Search
27  * Alfred V. Aho and Margaret J. Corasick
28  *
29  * - Started with util-mpm-ac.c:
30  * - Uses the delta table for calculating transitions,
31  * instead of having separate goto and failure
32  * transitions.
33  * - If we cross 2 ** 16 states, we use 4 bytes in the
34  * transition table to hold each state, otherwise we use
35  * 2 bytes.
36  * - This version of the MPM is heavy on memory, but it
37  * performs well. If you can fit the ruleset with this
38  * mpm on your box without hitting swap, this is the MPM
39  * to go for.
40  *
41  * - Added these optimizations:
42  * - Compress the input alphabet from 256 characters down
43  * to the actual characters used in the patterns, plus
44  * one character for all the unused characters.
45  * - Reduce the size of the delta table so that each state
46  * is the smallest power of two that is larger than the
47  * size of the compressed alphabet.
48  * - Specialized the search function based on state count
49  * (small for 8-bit large for 16-bit) and the size of
50  * the alphabet, so that it is constant inside the
51  * function for better optimization.
52  *
53  * \todo - Do a proper analyis of our existing MPMs and suggest a good
54  * one based on the pattern distribution and the expected
55  * traffic(say http).
56 
57  * - Irrespective of whether we cross 2 ** 16 states or
58  * not,shift to using uint32_t for state type, so that we can
59  * integrate it's status as a final state or not in the
60  * topmost byte. We are already doing it if state_count is >
61  * 2 ** 16.
62  * - Test case-senstive patterns if they have any ascii chars.
63  * If they don't treat them as nocase.
64  * - Reorder the compressed alphabet to put the most common characters
65  * first.
66  */

191 static inline int SCACTileCmpPattern(SCACTilePattern *p, uint8_t *pat,
192  uint16_t patlen, char flags)
193 {
194  if (p->len != patlen)
195  return 0;
196 
197  if (p->flags != flags)
198  return 0;
199 
200  if (flags & MPM_PATTERN_FLAG_NOCASE) {
201  // Case insensitive
202  if (SCMemcmpLowercase(p->cs, pat, patlen) != 0)
203  return 0;
204 
205  } else {
206  // Case sensitive
207  if (SCMemcmp(p->cs, pat, patlen) != 0)
208  return 0;
209  }
210 
211  return 1;
212 }

333 static inline int SCACTileInitHashAdd(SCACTileCtx *ctx, SCACTilePattern *p)
334 {
335  uint32_t hash = SCACTileInitHash(p);
336 
337  if (ctx->init_hash == NULL) {
338  return 0;
339  }
340 
341  if (ctx->init_hash[hash] == NULL) {
342  ctx->init_hash[hash] = p;
343  return 0;

345 
346  SCACTilePattern *tt = NULL;
347  SCACTilePattern *t = ctx->init_hash[hash];

356

375 /* Use Alpahbet Histogram to create compressed alphabet.
376  */
377 static void SCACTileInitTranslateTable(SCACTileCtx *ctx)
378 {
379  /* Count the number of ASCII values actually appearing in any
380  * pattern. Create compressed mapping table with unused
381  * characters mapping to zero.
382  */
383  for (int i = 0; i < 256; i++) {
384  /* Move all upper case counts to lower case */
385  if (i >= 'A' && i <= 'Z') {
386  ctx->alpha_hist[i - 'A' + 'a'] += ctx->alpha_hist[i];
387  ctx->alpha_hist[i] = 0;
388  }
389  if (ctx->alpha_hist[i]) {
390  ctx->alphabet_size++;
391  ctx->translate_table[i] = ctx->alphabet_size;
392  } else
393  ctx->translate_table[i] = 0;
394  }
395  /* Fix up translation table for uppercase */
396  for (int i = 'A'; i <= 'Z'; i++)
397  ctx->translate_table[i] = ctx->translate_table[i - 'A' + 'a'];
398 
399  SCLogDebug(" Alphabet size %d", ctx->alphabet_size);
400 
401  /* Round alphabet size up to next power-of-two Leave one extra
402  * space For the unused-chararacters = 0 mapping.
403  */
404  ctx->alphabet_size += 1; /* Extra space for unused-character */
405  if (ctx->alphabet_size <= 8) {
406  ctx->alphabet_storage = 8;
407  } else if (ctx->alphabet_size <= 16) {
408  ctx->alphabet_storage = 16;
409  } else if (ctx->alphabet_size <= 32) {
410  ctx->alphabet_storage = 32;
411  } else if (ctx->alphabet_size <= 64) {
412  ctx->alphabet_storage = 64;
413  } else if (ctx->alphabet_size <= 128) {
414  ctx->alphabet_storage = 128;
415  } else
416  ctx->alphabet_storage = 256;
417 }

419 /**
420  * \internal
421  * \brief Add a pattern to the mpm-ac context.
422  *
423  * \param mpm_ctx Mpm context.
424  * \param pat Pointer to the pattern.
425  * \param patlen Length of the pattern.
426  * \param pid Pattern id
427  * \param sid Signature id (internal id).
428  * \param flags Pattern's MPM_PATTERN_* flags.
429  *
430  * \retval 0 On success.
431  * \retval -1 On failure.
432  */
433 static int SCACTileAddPattern(MpmCtx *mpm_ctx, uint8_t *pat, uint16_t patlen,
434  uint16_t offset, uint16_t depth, uint32_t pid,
435  uint32_t sid, uint8_t flags)
436 {
437  SCACTileSearchCtx *search_ctx = (SCACTileSearchCtx *)mpm_ctx->ctx;
438  SCACTileCtx *ctx = search_ctx->init_ctx;
439 
440  SCLogDebug("Adding pattern for ctx %p, patlen %"PRIu16" and pid %" PRIu32,
441  ctx, patlen, pid);
442 
443  if (patlen == 0) {
444  SCLogWarning(SC_ERR_INVALID_ARGUMENTS, "pattern length 0");
445  return 0;
446  }
447 
448  /* Check if we have already inserted this pattern. */
449  SCACTilePattern *p = SCACTileInitHashLookup(ctx, pat, patlen, flags, pid);
450  if (p == NULL) {
451  SCLogDebug("Allocing new pattern");
452 
453  /* p will never be NULL */
454  p = SCACTileAllocPattern(mpm_ctx);
455 
456  p->len = patlen;
457  p->flags = flags;
458  p->id = pid;
459 
460  p->original_pat = SCMalloc(patlen);
461  if (p->original_pat == NULL)
462  goto error;
463  mpm_ctx->memory_cnt++;
464  mpm_ctx->memory_size += patlen;
465  memcpy(p->original_pat, pat, patlen);
466 
467  p->ci = SCMalloc(patlen);
468  if (p->ci == NULL)
469  goto error;
470  mpm_ctx->memory_cnt++;
471  mpm_ctx->memory_size += patlen;
472  memcpy_tolower(p->ci, pat, patlen);
473 
474  /* setup the case sensitive part of the pattern */
475  if (p->flags & MPM_PATTERN_FLAG_NOCASE) {
476  /* nocase means no difference between cs and ci */
477  p->cs = p->ci;
478  } else {
479  if (memcmp(p->ci, pat, p->len) == 0) {
480  /* no diff between cs and ci: pat is lowercase */
481  p->cs = p->ci;
482  } else {
483  p->cs = SCMalloc(patlen);
484  if (p->cs == NULL)
485  goto error;
486  mpm_ctx->memory_cnt++;
487  mpm_ctx->memory_size += patlen;
488  memcpy(p->cs, pat, patlen);
489  }
490  }
491 
492  /* put in the pattern hash */
493  SCACTileInitHashAdd(ctx, p);
494  /* Count alphabet usages */
495  SCACTileHistogramAlphabet(ctx, p);
496 
497  //if (mpm_ctx->pattern_cnt == 65535) {
498  // SCLogError(SC_ERR_AHO_CORASICK, "Max search words reached. Can't "
499  // "insert anymore. Exiting");
500  // exit(EXIT_FAILURE);
501  //}
502  mpm_ctx->pattern_cnt++;
503 
504  if (mpm_ctx->maxlen < patlen)
505  mpm_ctx->maxlen = patlen;
506 
507  if (mpm_ctx->minlen == 0) {
508  mpm_ctx->minlen = patlen;
509  } else {
510  if (mpm_ctx->minlen > patlen)
511  mpm_ctx->minlen = patlen;
512  }
513 
514  p->sids_size = 1;
515  p->sids = SCMalloc(p->sids_size * sizeof(SigIntId));
516  BUG_ON(p->sids == NULL);
517  p->sids[0] = sid;
518  //SCLogInfo("MPM added %u:%u", pid, sid);
519  } else {
520  /* TODO figure out how we can be called multiple times for the same CTX with the same sid */
521 
522  int found = 0;
523  uint32_t x = 0;
524  for (x = 0; x < p->sids_size; x++) {
525  if (p->sids[x] == sid) {
526  found = 1;
527  break;
528  }
529  }
530  if (!found) {
531  SigIntId *sids = SCRealloc(p->sids, (sizeof(SigIntId) * (p->sids_size + 1)));
532  BUG_ON(sids == NULL);
533  p->sids = sids;
534  p->sids[p->sids_size] = sid;
535  p->sids_size++;
536  //SCLogInfo("p->sids_size %u", p->sids_size);
537  //SCLogInfo("MPM added %u:%u (append)", pid, sid);
538  } else {
539  //SCLogInfo("rule %u already part of pid %u", sid, pid);
540  }
541  }
542 
543  return 0;
544 
545 error:
546  SCACTileFreePattern(mpm_ctx, p);
547  return -1;
548 }

592  SCACTileCtx *ctx = search_ctx->init_ctx;
593  int aa = 0;
594

596  if (ctx->allocated_state_count < ctx->state_count + 1) {

602  SCACTileReallocState(ctx, ctx->allocated_state_count);

604

606  for (aa = 0; aa < ctx->alphabet_size; aa++) {
607  ctx->goto_table[ctx->state_count][aa] = SC_AC_TILE_FAIL;
608  }
609 
610  memset(ctx->output_table + ctx->state_count, 0,
611  sizeof(SCACTileOutputTable));
612 
613  return ctx->state_count++;

822  *

825 static void SCACTileCreateFailureTable(MpmCtx *mpm_ctx)
826 {
827  SCACTileSearchCtx *search_ctx = (SCACTileSearchCtx *)mpm_ctx->ctx;
828  SCACTileCtx *ctx = search_ctx->init_ctx;
829 
830  int aa = 0;

832  int32_t r_state = 0;
833 
834  StateQueue q;
835  memset(&q, 0, sizeof(StateQueue));
836 
837  /* Allocate space for the failure table. A failure entry in the table for
838  * every state(SCACTileCtx->state_count) */
839  ctx->failure_table = SCMalloc(ctx->state_count * sizeof(int32_t));
840  if (ctx->failure_table == NULL) {
841  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");

847  * transition from the 0th state to the queue for further processing
848  * of failure states */
849  for (aa = 0; aa < ctx->alphabet_size; aa++) {
850  int32_t temp_state = ctx->goto_table[0][aa];
851  if (temp_state != 0) {
852  SCACTileEnqueue(&q, temp_state);
853  ctx->failure_table[temp_state] = 0;
854  }
855  }
856 
857  while (!SCACTileStateQueueIsEmpty(&q)) {
858  /* pick up every state from the queue and add failure transitions */
859  r_state = SCACTileDequeue(&q);

861  int32_t temp_state = ctx->goto_table[r_state][aa];
862  if (temp_state == SC_AC_TILE_FAIL)
863  continue;
864  SCACTileEnqueue(&q, temp_state);
865  state = ctx->failure_table[r_state];
866 
867  while(ctx->goto_table[state][aa] == SC_AC_TILE_FAIL)
868  state = ctx->failure_table[state];
869  ctx->failure_table[temp_state] = ctx->goto_table[state][aa];
870  SCACTileClubOutputStates(temp_state, ctx->failure_table[temp_state],
871  mpm_ctx);
872  }

1250  }

1252  mpm_ctx->memory_size += mem_size;
1253  /* Split the allocated block into pattern list array and string space. */
1254  ctx->pattern_list = mem_block;
1255  uint8_t *string_space = mem_block + pattern_list_size;
1256 
1257  /* Now make the copies of the no-case strings. */
1258  for (i = 0; i < mpm_ctx->pattern_cnt; i++) {
1259  if (!(ctx->parray[i]->flags & MPM_PATTERN_FLAG_NOCASE)) {
1260  uint32_t len = ctx->parray[i]->len;
1261  uint32_t space = ((len + 7) / 8) * 8;
1262  memcpy(string_space, ctx->parray[i]->original_pat, len);
1263  ctx->pattern_list[i].cs = string_space;
1264  ctx->pattern_list[i].patlen = len;
1265  string_space += space;
1266  }
1267  ctx->pattern_list[i].pid = ctx->parray[i]->id;
1268 
1269  /* ACPatternList now owns this memory */
1270  ctx->pattern_list[i].sids_size = ctx->parray[i]->sids_size;
1271  ctx->pattern_list[i].sids = ctx->parray[i]->sids;
1272  }
1273

1275  SCACTilePrepareStateTable(mpm_ctx);
1276 
1277  /* Convert to the Search Context structure */
1278  SCACTilePrepareSearch(mpm_ctx);
1279

1281 
1282 error:
1283  return -1;
1284 }

1456 /*
1457  * Heavily optimized pattern matching routine for TILE-Gx.
1458  */
1459 
1460 #define SCHECK(x) ((x) > 0)
1461 #define BTYPE int32_t
1462 // Extract byte N=0,1,2,3 from x
1463 #define BYTE0(x) __insn_bfextu(x, 0, 7)
1464 #define BYTE1(x) __insn_bfextu(x, 8, 15)
1465 #define BYTE2(x) __insn_bfextu(x, 16, 23)
1466 #define BYTE3(x) __insn_bfextu(x, 24, 31)
1467 
1468 int CheckMatch(SCACTileSearchCtx *ctx, PatternMatcherQueue *pmq,
1469  uint8_t *buf, uint16_t buflen,
1470  uint16_t state, int i, int matches,
1471  uint8_t *mpm_bitarray)
1472 {
1473  SCACTilePatternList *pattern_list = ctx->pattern_list;
1474  uint8_t *buf_offset = buf + i + 1; // Lift out of loop
1475  uint32_t no_of_entries = ctx->output_table[state].no_of_entries;
1476  MpmPatternIndex *patterns = ctx->output_table[state].patterns;
1477  uint8_t *pmq_bitarray = pmq->pattern_id_bitarray;
1478  uint32_t k;
1479 
1480  for (k = 0; k < no_of_entries; k++) {
1481  MpmPatternIndex pindex = patterns[k] & 0x0FFFFFFF;
1482  if (mpm_bitarray[pindex / 8] & (1 << (pindex % 8))) {
1483  /* Pattern already seen by this MPM. */
1484  /* NOTE: This is faster then rechecking if it is a case-sensitive match
1485  * since we know this pattern has already been seen, but imcrementing
1486  * matches here could over report matches. For example if the case-sensitive
1487  * pattern is "Foo" and the string is "Foo bar foo", matches would be reported
1488  * as 2, when it should really be 1, since "foo" is not a true match.
1489  */
1490  matches++;
1491  continue;
1492  }
1493  uint32_t pid = pattern_list[pindex].pid;
1494  /* Double check case-sensitve match now. */
1495  if (patterns[k] >> 31) {
1496  uint16_t patlen = pattern_list[pindex].patlen;
1497  if (SCMemcmpNZ(pattern_list[pindex].cs, buf_offset - patlen, patlen) != 0) {
1498  /* Case-sensitive match failed. */
1499  continue;
1500  }
1501  }
1502  /* New match found */
1503  mpm_bitarray[pindex / 8] |= (1 << (pindex % 8));
1504 
1505  if ((pmq_bitarray[pid / 8] & (1 << (pid % 8))) == 0) {
1506  pmq_bitarray[(pid) / 8] |= (1 << ((pid) % 8));
1507  MpmAddPid(pmq, pid);
1508  }
1509 
1510  /* Always add the Signature IDs, since they could be different in the current MPM
1511  * than in a previous MPM on the same PMQ when finding the same pattern.
1512  */
1513  MpmAddSids(pmq, pattern_list[pindex].sids,
1514  pattern_list[pindex].sids_size);
1515  matches++;
1516  }
1517 
1518  return matches;
1519 }
