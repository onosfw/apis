195 /**
196  * \brief Calculates the checksum for the IP packet
197  *
198  * \param pkt Pointer to the start of the IP packet
199  * \param hlen Length of the IP header
200  *
201  * \retval csum Checksum for the IP packet
202  */
203 static inline uint16_t IPV4CalculateChecksum(uint16_t *pkt, uint16_t hlen)
204 {
205  uint32_t csum = pkt[0];
206 
207  csum += pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[6] + pkt[7] + pkt[8] +
208  pkt[9];
209 
210  hlen -= 20;
211  pkt += 10;
212 
213  if (hlen == 0) {
214  ;
215  } else if (hlen == 4) {
216  csum += pkt[0] + pkt[1];
217  } else if (hlen == 8) {
218  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3];
219  } else if (hlen == 12) {
220  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5];
221  } else if (hlen == 16) {
222  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5] + pkt[6] +
223  pkt[7];
224  } else if (hlen == 20) {
225  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5] + pkt[6] +
226  pkt[7] + pkt[8] + pkt[9];
227  } else if (hlen == 24) {
228  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5] + pkt[6] +
229  pkt[7] + pkt[8] + pkt[9] + pkt[10] + pkt[11];
230  } else if (hlen == 28) {
231  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5] + pkt[6] +
232  pkt[7] + pkt[8] + pkt[9] + pkt[10] + pkt[11] + pkt[12] + pkt[13];
233  } else if (hlen == 32) {
234  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5] + pkt[6] +
235  pkt[7] + pkt[8] + pkt[9] + pkt[10] + pkt[11] + pkt[12] + pkt[13] +
236  pkt[14] + pkt[15];
237  } else if (hlen == 36) {
238  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5] + pkt[6] +
239  pkt[7] + pkt[8] + pkt[9] + pkt[10] + pkt[11] + pkt[12] + pkt[13] +
240  pkt[14] + pkt[15] + pkt[16] + pkt[17];
241  } else if (hlen == 40) {
242  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5] + pkt[6] +
243  pkt[7] + pkt[8] + pkt[9] + pkt[10] + pkt[11] + pkt[12] + pkt[13] +
244  pkt[14] + pkt[15] + pkt[16] + pkt[17] + pkt[18] + pkt[19];
245  }
246 
247  csum = (csum >> 16) + (csum & 0x0000FFFF);
248  csum += (csum >> 16);
249 
250  return (uint16_t) ~csum;
251 }
