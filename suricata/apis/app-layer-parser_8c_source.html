549 void AppLayerParserSetTransactionInspectId(AppLayerParserState *pstate,
550  const uint8_t ipproto, const AppProto alproto,
551  void *alstate, const uint8_t flags)
552 {
553  SCEnter();
554 
555  int direction = (flags & STREAM_TOSERVER) ? 0 : 1;
556  uint64_t total_txs = AppLayerParserGetTxCnt(ipproto, alproto, alstate);
557  uint64_t idx = AppLayerParserGetTransactionInspectId(pstate, flags);
558  int state_done_progress = AppLayerParserGetStateProgressCompletionStatus(ipproto, alproto, flags);
559  void *tx;
560  int state_progress;
561 
562  for (; idx < total_txs; idx++) {
563  tx = AppLayerParserGetTx(ipproto, alproto, alstate, idx);
564  if (tx == NULL)
565  continue;
566  state_progress = AppLayerParserGetStateProgress(ipproto, alproto, tx, flags);
567  if (state_progress >= state_done_progress)
568  continue;
569  else
570  break;
571  }
572  pstate->inspect_id[direction] = idx;
573 
574  SCReturn;
575 }

645 /** \brief active TX retrieval for logging only: so NO detection
646  *
647  * If the logger is enabled, we simply return the log_id here.
648  *
649  * Otherwise, we go look for the tx id. There probably is no point
650  * in running this function in that case though. With no detection
651  * and no logging, why run a parser in the first place?
652  **/
653 uint64_t AppLayerTransactionGetActiveLogOnly(Flow *f, uint8_t flags)
654 {
655  AppLayerParserProtoCtx *p = &alp_ctx.ctxs[f->protomap][f->alproto];
656 
657  if (p->logger == TRUE) {
658  uint64_t log_id = f->alparser->log_id;
659  SCLogDebug("returning %"PRIu64, log_id);
660  return log_id;
661  }
662 
663  /* logger is disabled, return highest 'complete' tx id */
664  uint64_t total_txs = AppLayerParserGetTxCnt(f->proto, f->alproto, f->alstate);
665  uint64_t idx = AppLayerParserGetTransactionInspectId(f->alparser, flags);
666  int state_done_progress = AppLayerParserGetStateProgressCompletionStatus(f->proto, f->alproto, flags);
667  void *tx;
668  int state_progress;
669 
670  for (; idx < total_txs; idx++) {
671  tx = AppLayerParserGetTx(f->proto, f->alproto, f->alstate, idx);
672  if (tx == NULL)
673  continue;
674  state_progress = AppLayerParserGetStateProgress(f->proto, f->alproto, tx, flags);
675  if (state_progress >= state_done_progress)
676  continue;
677  else
678  break;
679  }
680  SCLogDebug("returning %"PRIu64, idx);
681  return idx;
682 }

854 int AppLayerParserParse(AppLayerParserThreadCtx *alp_tctx, Flow *f, AppProto alproto,
855  uint8_t flags, uint8_t *input, uint32_t input_len)
856 {
857  SCEnter();
858 #ifdef DEBUG_VALIDATION
859  BUG_ON(f->protomap != FlowGetProtoMapping(f->proto));
860 #endif
861  AppLayerParserState *pstate = NULL;
862  AppLayerParserProtoCtx *p = &alp_ctx.ctxs[f->protomap][alproto];
863  void *alstate = NULL;
864 
865  /* we don't have the parser registered for this protocol */
866  if (p->StateAlloc == NULL)
867  goto end;
868 
869  /* Do this check before calling AppLayerParse */
870  if (flags & STREAM_GAP) {
871  SCLogDebug("stream gap detected (missing packets), "
872  "this is not yet supported.");
873 
874  if (f->alstate != NULL)
875  AppLayerParserStreamTruncated(f->proto, alproto, f->alstate, flags);
876  goto error;
877  }
878 
879  /* Get the parser state (if any) */
880  pstate = f->alparser;
881  if (pstate == NULL) {
882  f->alparser = pstate = AppLayerParserStateAlloc();
883  if (pstate == NULL)
884  goto error;
885  }
886  pstate->version++;
887  SCLogDebug("app layer parser state version incremented to %"PRIu8,
888  pstate->version);
889 
890  if (flags & STREAM_EOF)
891  AppLayerParserStateSetFlag(pstate, APP_LAYER_PARSER_EOF);
892 
893  alstate = f->alstate;
894  if (alstate == NULL) {
895  f->alstate = alstate = p->StateAlloc();
896  if (alstate == NULL)
897  goto error;
898  SCLogDebug("alloced new app layer state %p (name %s)",
899  alstate, AppLayerGetProtoName(f->alproto));
900  } else {
901  SCLogDebug("using existing app layer state %p (name %s))",
902  alstate, AppLayerGetProtoName(f->alproto));
903  }
904 
905  /* invoke the recursive parser, but only on data. We may get empty msgs on EOF */
906  if (input_len > 0 || (flags & STREAM_EOF)) {
907  /* invoke the parser */
908  if (p->Parser[(flags & STREAM_TOSERVER) ? 0 : 1](f, alstate, pstate,
909  input, input_len,
910  alp_tctx->alproto_local_storage[f->protomap][alproto]) < 0)
911  {
912  goto error;
913  }
914  }
915 
916  /* set the packets to no inspection and reassembly if required */
917  if (pstate->flags & APP_LAYER_PARSER_NO_INSPECTION) {
918  AppLayerParserSetEOF(pstate);
919  FlowSetNoPayloadInspectionFlag(f);
920 
921  if (f->proto == IPPROTO_TCP) {
922  StreamTcpDisableAppLayer(f);
923 
924  /* Set the no reassembly flag for both the stream in this TcpSession */
925  if (pstate->flags & APP_LAYER_PARSER_NO_REASSEMBLY) {
926  /* Used only if it's TCP */
927  TcpSession *ssn = f->protoctx;
928  if (ssn != NULL) {
929  StreamTcpSetSessionNoReassemblyFlag(ssn,
930  flags & STREAM_TOCLIENT ? 1 : 0);
931  StreamTcpSetSessionNoReassemblyFlag(ssn,
932  flags & STREAM_TOSERVER ? 1 : 0);
933  }
934  }
935  }
936  }
937 
938  /* In cases like HeartBleed for TLS we need to inspect AppLayer but not Payload */
939  if (!(f->flags & FLOW_NOPAYLOAD_INSPECTION) && pstate->flags & APP_LAYER_PARSER_NO_INSPECTION_PAYLOAD) {
940  FlowSetNoPayloadInspectionFlag(f);
941  /* Set the no reassembly flag for both the stream in this TcpSession */
942  if (f->proto == IPPROTO_TCP) {
943  /* Used only if it's TCP */
944  TcpSession *ssn = f->protoctx;
945  if (ssn != NULL) {
946  StreamTcpSetDisableRawReassemblyFlag(ssn, 0);
947  StreamTcpSetDisableRawReassemblyFlag(ssn, 1);
948  }
949  }
950  }
951 
952  /* next, see if we can get rid of transactions now */
953  AppLayerParserTransactionsCleanup(f);
954 
955  /* stream truncated, inform app layer */
956  if (flags & STREAM_DEPTH)
957  AppLayerParserStreamTruncated(f->proto, alproto, alstate, flags);
958 
959  end:
960  SCReturnInt(0);
961  error:
962  /* Set the no app layer inspection flag for both
963  * the stream in this Flow */
964  if (f->proto == IPPROTO_TCP) {
965  StreamTcpDisableAppLayer(f);
966  }
967  AppLayerParserSetEOF(pstate);
968  SCReturnInt(-1);
969 }
