62 // Below are all htp_status_t return codes used by LibHTP. Enum is not
63 // used here to allow applications to define their own codes.

65 /**
66  * The lowest htp_status_t value LibHTP will use internally.
67  */
68 #define HTP_ERROR_RESERVED -1000
69 
70 /** General-purpose error code. */
71 #define HTP_ERROR -1
72 
73 /**
74  * No processing or work was done. This is typically used by callbacks
75  * to indicate that they were not interested in doing any work in the
76  * given context.
77  */
78 #define HTP_DECLINED 0
79 
80 /** Returned by a function when its work was successfully completed. */
81 #define HTP_OK 1
82 
83 /**
84  * Returned when processing a connection stream, after consuming all
85  * provided data. The caller should call again with more data.
86  */
87 #define HTP_DATA 2
88 
89 /**
90  * Returned when processing a connection stream, after encountering
91  * a situation where processing needs to continue on the alternate
92  * stream (e.g., the inbound parser needs to observe some outbound
93  * data). The data provided was not completely consumed. On the next
94  * invocation the caller should supply only the data that has not
95  * been processed already. Use htp_connp_req_data_consumed() and
96  * htp_connp_res_data_consumed() to determine how much of the most
97  * recent data chunk was consumed.
98  */
99 #define HTP_DATA_OTHER 3
100 
101 /**
102  * Used by callbacks to indicate that the processing should stop. For example,
103  * returning HTP_STOP from a connection callback indicates that LibHTP should
104  * stop following that particular connection.
105  */
106 #define HTP_STOP 4
107 
108 /**
109  * Same as HTP_DATA, but indicates that any non-consumed part of the
110  * data chunk should be preserved (buffered) for later.
111  */
112 #define HTP_DATA_BUFFER 5

188 // Various flag bits. Even though we have a flag field in several places

192 // that contains the header. Both uses are useful.
