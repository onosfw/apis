167 static int PcapTryReopen(PcapThreadVars *ptv)
168 {
169  int pcap_activate_r;
170 
171  ptv->pcap_state = PCAP_STATE_DOWN;
172  pcap_activate_r = pcap_activate(ptv->pcap_handle);
173  if (pcap_activate_r != 0) {
174  return pcap_activate_r;
175  }
176  /* set bpf filter if we have one */
177  if (ptv->bpf_filter != NULL) {
178  if(pcap_compile(ptv->pcap_handle,&ptv->filter,ptv->bpf_filter,1,0) < 0) {
179  SCLogError(SC_ERR_BPF,"bpf compilation error %s",pcap_geterr(ptv->pcap_handle));
180  return -1;
181  }
182 
183  if(pcap_setfilter(ptv->pcap_handle,&ptv->filter) < 0) {
184  SCLogError(SC_ERR_BPF,"could not set bpf filter %s",pcap_geterr(ptv->pcap_handle));
185  return -1;
186  }
187  }
188 
189  SCLogInfo("Recovering interface listening");
190  ptv->pcap_state = PCAP_STATE_UP;
191  return 0;
192 }

230 void PcapCallbackLoop(char *user, struct pcap_pkthdr *h, u_char *pkt)
231 {
232  SCEnter();
233 
234  PcapThreadVars *ptv = (PcapThreadVars *)user;
235  Packet *p = PacketGetFromQueueOrAlloc();
236  struct timeval current_time;
237 
238  if (unlikely(p == NULL)) {
239  SCReturn;
240  }
241 
242  PKT_SET_SRC(p, PKT_SRC_WIRE);
243  p->ts.tv_sec = h->ts.tv_sec;
244  p->ts.tv_usec = h->ts.tv_usec;
245  SCLogDebug("p->ts.tv_sec %"PRIuMAX"", (uintmax_t)p->ts.tv_sec);
246  p->datalink = ptv->datalink;
247 
248  ptv->pkts++;
249  ptv->bytes += h->caplen;
250  (void) SC_ATOMIC_ADD(ptv->livedev->pkts, 1);
251  p->livedev = ptv->livedev;
252 
253  if (unlikely(PacketCopyData(p, pkt, h->caplen))) {
254  TmqhOutputPacketpool(ptv->tv, p);
255  SCReturn;
256  }
257 
258  switch (ptv->checksum_mode) {
259  case CHECKSUM_VALIDATION_AUTO:
260  if (ptv->livedev->ignore_checksum) {
261  p->flags |= PKT_IGNORE_CHECKSUM;
262  } else if (ChecksumAutoModeCheck(ptv->pkts,
263  SC_ATOMIC_GET(ptv->livedev->pkts),
264  SC_ATOMIC_GET(ptv->livedev->invalid_checksums))) {
265  ptv->livedev->ignore_checksum = 1;
266  p->flags |= PKT_IGNORE_CHECKSUM;
267  }
268  break;
269  case CHECKSUM_VALIDATION_DISABLE:
270  p->flags |= PKT_IGNORE_CHECKSUM;
271  break;
272  default:
273  break;
274  }
275 
276  if (TmThreadsSlotProcessPkt(ptv->tv, ptv->slot, p) != TM_ECODE_OK) {
277  pcap_breakloop(ptv->pcap_handle);
278  ptv->cb_result = TM_ECODE_FAILED;
279  }
280 
281  /* Trigger one dump of stats every second */
282  TimeGet(&current_time);
283  if (current_time.tv_sec != ptv->last_stats_dump) {
284  PcapDumpCounters(ptv);
285  ptv->last_stats_dump = current_time.tv_sec;
286  }
287 
288  SCReturn;
289 }

291 /**
292  * \brief Main PCAP reading Loop function
293  */
294 TmEcode ReceivePcapLoop(ThreadVars *tv, void *data, void *slot)
295 {
296  SCEnter();
297 
298  int packet_q_len = 64;
299  PcapThreadVars *ptv = (PcapThreadVars *)data;
300  int r;
301  TmSlot *s = (TmSlot *)slot;
302 
303  ptv->slot = s->slot_next;
304  ptv->cb_result = TM_ECODE_OK;
305 
306  while (1) {
307  if (suricata_ctl_flags & (SURICATA_STOP | SURICATA_KILL)) {
308  SCReturnInt(TM_ECODE_OK);
309  }
310 
311  /* make sure we have at least one packet in the packet pool, to prevent
312  * us from alloc'ing packets at line rate */
313  PacketPoolWait();
314 
315  /* Right now we just support reading packets one at a time. */
316  r = pcap_dispatch(ptv->pcap_handle, packet_q_len,
317  (pcap_handler)PcapCallbackLoop, (u_char *)ptv);
318  if (unlikely(r < 0)) {
319  int dbreak = 0;
320  SCLogError(SC_ERR_PCAP_DISPATCH, "error code %" PRId32 " %s",
321  r, pcap_geterr(ptv->pcap_handle));
322 #ifdef PCAP_ERROR_BREAK
323  if (r == PCAP_ERROR_BREAK) {
324  SCReturnInt(ptv->cb_result);
325  }
326 #endif
327  do {
328  usleep(PCAP_RECONNECT_TIMEOUT);
329  if (suricata_ctl_flags != 0) {
330  dbreak = 1;
331  break;
332  }
333  r = PcapTryReopen(ptv);
334  } while (r < 0);
335  if (dbreak) {
336  break;
337  }
338  } else if (ptv->cb_result == TM_ECODE_FAILED) {
339  SCLogError(SC_ERR_PCAP_DISPATCH, "Pcap callback PcapCallbackLoop failed");
340  SCReturnInt(TM_ECODE_FAILED);
341  }
342 
343  StatsSyncCountersIfSignalled(tv);
344  }
345 
346  PcapDumpCounters(ptv);
347  StatsSyncCountersIfSignalled(tv);
348  SCReturnInt(TM_ECODE_OK);
349 }

351 /**
352  * \brief Init function for ReceivePcap.
353  *
354  * This is a setup function for recieving packets
355  * via libpcap. There are two versions of this function
356  * depending on the major version of libpcap used.
357  * For versions prior to 1.x we use open_pcap_live,
358  * for versions 1.x and greater we use pcap_create + pcap_activate.
359  *
360  * \param tv pointer to ThreadVars
361  * \param initdata pointer to the interface passed from the user
362  * \param data pointer gets populated with PcapThreadVars
363  *
364  * \todo Create a general pcap setup function.
365  */
366 #if LIBPCAP_VERSION_MAJOR == 1
367 TmEcode ReceivePcapThreadInit(ThreadVars *tv, void *initdata, void **data)
368 {
369  SCEnter();
370  PcapIfaceConfig *pcapconfig = initdata;
371 
372  if (initdata == NULL) {
373  SCLogError(SC_ERR_INVALID_ARGUMENT, "initdata == NULL");
374  SCReturnInt(TM_ECODE_FAILED);
375  }
376 
377  PcapThreadVars *ptv = SCMalloc(sizeof(PcapThreadVars));
378  if (unlikely(ptv == NULL)) {
379  pcapconfig->DerefFunc(pcapconfig);
380  SCReturnInt(TM_ECODE_FAILED);
381  }
382  memset(ptv, 0, sizeof(PcapThreadVars));
383 
384  ptv->tv = tv;
385 
386  ptv->livedev = LiveGetDevice(pcapconfig->iface);
387  if (ptv->livedev == NULL) {
388  SCLogError(SC_ERR_INVALID_VALUE, "Unable to find Live device");
389  SCFree(ptv);
390  SCReturnInt(TM_ECODE_FAILED);
391  }
392 
393  SCLogInfo("using interface %s", (char *)pcapconfig->iface);
394 
395  ptv->checksum_mode = pcapconfig->checksum_mode;
396  if (ptv->checksum_mode == CHECKSUM_VALIDATION_AUTO) {
397  SCLogInfo("Running in 'auto' checksum mode. Detection of interface state will require "
398  xstr(CHECKSUM_SAMPLE_COUNT) " packets.");
399  }
400 
401  /* XXX create a general pcap setup function */
402  char errbuf[PCAP_ERRBUF_SIZE];
403  ptv->pcap_handle = pcap_create((char *)pcapconfig->iface, errbuf);
404  if (ptv->pcap_handle == NULL) {
405  if (strlen(errbuf)) {
406  SCLogError(SC_ERR_PCAP_CREATE, "Couldn't create a new pcap handler for %s, error %s",
407  (char *)pcapconfig->iface, errbuf);
408  } else {
409  SCLogError(SC_ERR_PCAP_CREATE, "Couldn't create a new pcap handler for %s",
410  (char *)pcapconfig->iface);
411  }
412  SCFree(ptv);
413  pcapconfig->DerefFunc(pcapconfig);
414  SCReturnInt(TM_ECODE_FAILED);
415  }
416 
417  if (pcapconfig->snaplen == 0) {
418  /* We set snaplen if we can get the MTU */
419  ptv->pcap_snaplen = GetIfaceMaxPacketSize(pcapconfig->iface);
420  } else {
421  ptv->pcap_snaplen = pcapconfig->snaplen;
422  }
423  if (ptv->pcap_snaplen > 0) {
424  /* set Snaplen. Must be called before pcap_activate */
425  int pcap_set_snaplen_r = pcap_set_snaplen(ptv->pcap_handle, ptv->pcap_snaplen);
426  if (pcap_set_snaplen_r != 0) {
427  SCLogError(SC_ERR_PCAP_SET_SNAPLEN, "Couldn't set snaplen, error: %s", pcap_geterr(ptv->pcap_handle));
428  SCFree(ptv);
429  pcapconfig->DerefFunc(pcapconfig);
430  SCReturnInt(TM_ECODE_FAILED);
431  }
432  SCLogInfo("Set snaplen to %d for '%s'", ptv->pcap_snaplen,
433  pcapconfig->iface);
434  }
435 
436  /* set Promisc, and Timeout. Must be called before pcap_activate */
437  int pcap_set_promisc_r = pcap_set_promisc(ptv->pcap_handle, pcapconfig->promisc);
438  //printf("ReceivePcapThreadInit: pcap_set_promisc(%p) returned %" PRId32 "\n", ptv->pcap_handle, pcap_set_promisc_r);
439  if (pcap_set_promisc_r != 0) {
440  SCLogError(SC_ERR_PCAP_SET_PROMISC, "Couldn't set promisc mode, error %s", pcap_geterr(ptv->pcap_handle));
441  SCFree(ptv);
442  pcapconfig->DerefFunc(pcapconfig);
443  SCReturnInt(TM_ECODE_FAILED);
444  }
445 
446  int pcap_set_timeout_r = pcap_set_timeout(ptv->pcap_handle,LIBPCAP_COPYWAIT);
447  //printf("ReceivePcapThreadInit: pcap_set_timeout(%p) returned %" PRId32 "\n", ptv->pcap_handle, pcap_set_timeout_r);
448  if (pcap_set_timeout_r != 0) {
449  SCLogError(SC_ERR_PCAP_SET_TIMEOUT, "Problems setting timeout, error %s", pcap_geterr(ptv->pcap_handle));
450  SCFree(ptv);
451  pcapconfig->DerefFunc(pcapconfig);
452  SCReturnInt(TM_ECODE_FAILED);
453  }
454 #ifdef HAVE_PCAP_SET_BUFF
455  ptv->pcap_buffer_size = pcapconfig->buffer_size;
456  if (ptv->pcap_buffer_size >= 0 && ptv->pcap_buffer_size <= INT_MAX) {
457  if (ptv->pcap_buffer_size > 0)
458  SCLogInfo("Going to use pcap buffer size of %" PRId32 "", ptv->pcap_buffer_size);
459 
460  int pcap_set_buffer_size_r = pcap_set_buffer_size(ptv->pcap_handle,ptv->pcap_buffer_size);
461  //printf("ReceivePcapThreadInit: pcap_set_timeout(%p) returned %" PRId32 "\n", ptv->pcap_handle, pcap_set_buffer_size_r);
462  if (pcap_set_buffer_size_r != 0) {
463  SCLogError(SC_ERR_PCAP_SET_BUFF_SIZE, "Problems setting pcap buffer size, error %s", pcap_geterr(ptv->pcap_handle));
464  SCFree(ptv);
465  pcapconfig->DerefFunc(pcapconfig);
466  SCReturnInt(TM_ECODE_FAILED);
467  }
468  }
469 #endif /* HAVE_PCAP_SET_BUFF */
470 
471  /* activate the handle */
472  int pcap_activate_r = pcap_activate(ptv->pcap_handle);
473  //printf("ReceivePcapThreadInit: pcap_activate(%p) returned %" PRId32 "\n", ptv->pcap_handle, pcap_activate_r);
474  if (pcap_activate_r != 0) {
475  SCLogError(SC_ERR_PCAP_ACTIVATE_HANDLE, "Couldn't activate the pcap handler, error %s", pcap_geterr(ptv->pcap_handle));
476  SCFree(ptv);
477  pcapconfig->DerefFunc(pcapconfig);
478  SCReturnInt(TM_ECODE_FAILED);
479  } else {
480  ptv->pcap_state = PCAP_STATE_UP;
481  }
482 
483  /* set bpf filter if we have one */
484  if (pcapconfig->bpf_filter) {
485  SCMutexLock(&pcap_bpf_compile_lock);
486 
487  ptv->bpf_filter = pcapconfig->bpf_filter;
488 
489  if (pcap_compile(ptv->pcap_handle,&ptv->filter,ptv->bpf_filter,1,0) < 0) {
490  SCLogError(SC_ERR_BPF, "bpf compilation error %s", pcap_geterr(ptv->pcap_handle));
491 
492  SCMutexUnlock(&pcap_bpf_compile_lock);
493  SCFree(ptv);
494  pcapconfig->DerefFunc(pcapconfig);
495  return TM_ECODE_FAILED;
496  }
497 
498  if (pcap_setfilter(ptv->pcap_handle,&ptv->filter) < 0) {
499  SCLogError(SC_ERR_BPF, "could not set bpf filter %s", pcap_geterr(ptv->pcap_handle));
500 
501  SCMutexUnlock(&pcap_bpf_compile_lock);
502  SCFree(ptv);
503  pcapconfig->DerefFunc(pcapconfig);
504  return TM_ECODE_FAILED;
505  }
506 
507  SCMutexUnlock(&pcap_bpf_compile_lock);
508  }
509 
510  /* Making it conditional to Linux even if GetIfaceOffloading return 0
511  * for non Linux. */
512 #ifdef HAVE_LINUX_ETHTOOL_H
513  if (GetIfaceOffloading(pcapconfig->iface) == 1) {
514  SCLogWarning(SC_ERR_PCAP_CREATE,
515  "Using Pcap capture with GRO or LRO activated can lead to "
516  "capture problems.");
517  }
518 #endif /* HAVE_LINUX_ETHTOOL_H */
519 
520  ptv->datalink = pcap_datalink(ptv->pcap_handle);
521 
522  pcapconfig->DerefFunc(pcapconfig);
523 
524  ptv->capture_kernel_packets = StatsRegisterCounter("capture.kernel_packets",
525  ptv->tv);
526  ptv->capture_kernel_drops = StatsRegisterCounter("capture.kernel_drops",
527  ptv->tv);
528  ptv->capture_kernel_ifdrops = StatsRegisterCounter("capture.kernel_ifdrops",
529  ptv->tv);
530 
531  *data = (void *)ptv;
532  SCReturnInt(TM_ECODE_OK);
533 }

765 void PcapTranslateIPToDevice(char *pcap_dev, size_t len)
766 {
767  char errbuf[PCAP_ERRBUF_SIZE];
768  pcap_if_t *alldevsp = NULL;
769  pcap_if_t *devsp = NULL;
770 
771  struct addrinfo aiHints;
772  struct addrinfo *aiList = NULL;
773  int retVal = 0;
774 
775  memset(&aiHints, 0, sizeof(aiHints));
776  aiHints.ai_family = AF_UNSPEC;
777  aiHints.ai_flags = AI_NUMERICHOST;
778 
779  /* try to translate IP */
780  if ((retVal = getaddrinfo(pcap_dev, NULL, &aiHints, &aiList)) != 0) {
781  return;
782  }
783 
784  if (pcap_findalldevs(&alldevsp, errbuf)) {
785  freeaddrinfo(aiList);
786  return;
787  }
788 
789  for (devsp = alldevsp; devsp ; devsp = devsp->next) {
790  pcap_addr_t *ip = NULL;
791 
792  for (ip = devsp->addresses; ip ; ip = ip->next) {
793 
794  if (aiList->ai_family != ip->addr->sa_family) {
795  continue;
796  }
797 
798  if (ip->addr->sa_family == AF_INET) {
799  if (memcmp(&((struct sockaddr_in*)aiList->ai_addr)->sin_addr, &((struct sockaddr_in*)ip->addr)->sin_addr, sizeof(struct in_addr))) {
800  continue;
801  }
802  } else if (ip->addr->sa_family == AF_INET6) {
803  if (memcmp(&((struct sockaddr_in6*)aiList->ai_addr)->sin6_addr, &((struct sockaddr_in6*)ip->addr)->sin6_addr, sizeof(struct in6_addr))) {
804  continue;
805  }
806  } else {
807  continue;
808  }
809 
810  freeaddrinfo(aiList);
811 
812  memset(pcap_dev, 0, len);
813  strlcpy(pcap_dev, devsp->name, len);
814 
815  pcap_freealldevs(alldevsp);
816  return;
817  }
818  }
819 
820  freeaddrinfo(aiList);
821 
822  pcap_freealldevs(alldevsp);
823 }
