66 /**
67  * \brief Append a chunk of body to the HtpBody struct
68  *
69  * \param body pointer to the HtpBody holding the list
70  * \param data pointer to the data of the chunk
71  * \param len length of the chunk pointed by data
72  *
73  * \retval 0 ok
74  * \retval -1 error
75  */
76 int HtpBodyAppendChunk(HtpTxUserData *htud, HtpBody *body, uint8_t *data, uint32_t len)
77 {
78  SCEnter();
79 
80  HtpBodyChunk *bd = NULL;
81 
82  if (len == 0 || data == NULL) {
83  SCReturnInt(0);
84  }
85 
86  if (body->first == NULL) {
87  /* New chunk */
88  bd = (HtpBodyChunk *)HTPMalloc(sizeof(HtpBodyChunk));
89  if (bd == NULL)
90  goto error;
91 
92  bd->len = len;
93  bd->stream_offset = 0;
94  bd->next = NULL;
95  bd->logged = 0;
96 
97  bd->data = HTPMalloc(len);
98  if (bd->data == NULL) {
99  goto error;
100  }
101  memcpy(bd->data, data, len);
102 
103  body->first = body->last = bd;
104 
105  body->content_len_so_far = len;
106  } else {
107  bd = (HtpBodyChunk *)HTPMalloc(sizeof(HtpBodyChunk));
108  if (bd == NULL)
109  goto error;
110 
111  bd->len = len;
112  bd->stream_offset = body->content_len_so_far;
113  bd->next = NULL;
114  bd->logged = 0;
115 
116  bd->data = HTPMalloc(len);
117  if (bd->data == NULL) {
118  goto error;
119  }
120  memcpy(bd->data, data, len);
121 
122  body->last->next = bd;
123  body->last = bd;
124 
125  body->content_len_so_far += len;
126  }
127  SCLogDebug("Body %p; data %p, len %"PRIu32, body, bd->data, (uint32_t)bd->len);
128 
129  SCReturnInt(0);
130 
131 error:
132  if (bd != NULL) {
133  if (bd->data != NULL) {
134  HTPFree(bd->data, bd->len);
135  }
136  HTPFree(bd, sizeof(HtpBodyChunk));
137  }
138  SCReturnInt(-1);
139 }

166 /**
167  * \brief Free the information held in the request body
168  * \param body pointer to the HtpBody holding the list
169  * \retval none
170  */
171 void HtpBodyFree(HtpBody *body)
172 {
173  SCEnter();
174 
175  if (body->first == NULL)
176  return;
177 
178  SCLogDebug("Removing chunks of Body %p; data %p, len %"PRIu32, body,
179  body->last->data, (uint32_t)body->last->len);
180 
181  HtpBodyChunk *cur = NULL;
182  HtpBodyChunk *prev = NULL;
183 
184  prev = body->first;
185  while (prev != NULL) {
186  cur = prev->next;
187  if (prev->data != NULL)
188  HTPFree(prev->data, prev->len);
189  HTPFree(prev, sizeof(HtpBodyChunk));
190  prev = cur;
191  }
192  body->first = body->last = NULL;
193 }

195 /**
196  * \brief Free request body chunks that are already fully parsed.
197  *
198  * \param state htp_state, with reference to our config
199  * \param body the body to prune
200  * \param direction STREAM_TOSERVER (request), STREAM_TOCLIENT (response)
201  *
202  * \retval none
203  */
204 void HtpBodyPrune(HtpState *state, HtpBody *body, int direction)
205 {
206  SCEnter();
207 
208  if (body == NULL || body->first == NULL) {
209  SCReturn;
210  }
211 
212  if (body->body_parsed == 0) {
213  SCReturn;
214  }
215 
216  /* get the configured inspect sizes. Default to response values */
217  uint32_t min_size = state->cfg->response_inspect_min_size;
218  uint32_t window = state->cfg->response_inspect_window;
219 
220  if (direction == STREAM_TOSERVER) {
221  min_size = state->cfg->request_inspect_min_size;
222  window = state->cfg->request_inspect_window;
223  }
224 
225  if (body->body_inspected < (min_size > window) ? min_size : window) {
226  SCReturn;
227  }
228 
229  SCLogDebug("Pruning chunks of Body %p; data %p, len %"PRIu32, body,
230  body->last->data, (uint32_t)body->last->len);
231 
232  HtpBodyChunk *cur = body->first;
233  while (cur != NULL) {
234  HtpBodyChunk *next = cur->next;
235 
236  SCLogDebug("cur->stream_offset %"PRIu64" + cur->len %u = %"PRIu64", "
237  "body->body_parsed %"PRIu64, cur->stream_offset, cur->len,
238  cur->stream_offset + cur->len, body->body_parsed);
239 
240  uint64_t left_edge = body->body_inspected;
241  if (left_edge <= min_size || left_edge <= window)
242  left_edge = 0;
243  if (left_edge)
244  left_edge -= window;
245 
246  if (cur->stream_offset + cur->len > left_edge) {
247  break;
248  }
249 
250  body->first = next;
251  if (body->last == cur) {
252  body->last = next;
253  }
254 
255  if (cur->data != NULL) {
256  HTPFree(cur->data, cur->len);
257  }
258  HTPFree(cur, sizeof(HtpBodyChunk));
259 
260  cur = next;
261  }
262 
263  SCReturn;
264 }
