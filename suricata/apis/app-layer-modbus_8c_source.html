264 static ModbusTransaction *ModbusTxFindByTransaction(const ModbusState *modbus,
265  const uint16_t transactionId) {
266  ModbusTransaction *tx = NULL;
267 
268  if (modbus->curr == NULL)
269  return NULL;
270 
271  /* fast path */
272  if ((modbus->curr->transactionId == transactionId) &&
273  !(modbus->curr->replied)) {
274  return modbus->curr;
275  /* slow path, iterate list */
276  } else {
277  TAILQ_FOREACH(tx, &modbus->tx_list, next) {
278  if ((tx->transactionId == transactionId) &&

281  }
282  }
283  /* not found */

295 static ModbusTransaction *ModbusTxAlloc(ModbusState *modbus) {
296  ModbusTransaction *tx;
297 
298  tx = (ModbusTransaction *) SCCalloc(1, sizeof(ModbusTransaction));
299  if (unlikely(tx == NULL))
300  return NULL;
301 
302  modbus->transaction_max++;
303  modbus->unreplied_cnt++;
304 
305  /* Check flood limit */
306  if ((request_flood != 0) && (modbus->unreplied_cnt > request_flood)) {
307  ModbusSetEvent(modbus, MODBUS_DECODER_EVENT_FLOODED);
308  modbus->givenup = 1;
309  }
310 
311  modbus->curr = tx;
312 
313  SCLogDebug("modbus->transaction_max updated to %"PRIu64, modbus->transaction_max);
314 
315  TAILQ_INSERT_TAIL(&modbus->tx_list, tx, next);
316 
317  tx->modbus = modbus;
318  tx->tx_num = modbus->transaction_max;
319 
320  return tx;
321 }

343  * \brief Modbus transaction cleanup callback
344  */
345 void ModbusStateTxFree(void *state, uint64_t tx_id) {
346  SCEnter();
347  ModbusState *modbus = (ModbusState *) state;
348  ModbusTransaction *tx = NULL, *ttx;
349 
350  SCLogDebug("state %p, id %"PRIu64, modbus, tx_id);
351 
352  TAILQ_FOREACH_SAFE(tx, &modbus->tx_list, next, ttx) {
353  SCLogDebug("tx %p tx->tx_num %"PRIu64", tx_id %"PRIu64, tx, tx->tx_num, (tx_id+1));
354 
355  if (tx->tx_num != (tx_id+1))
356  continue;
357 
358  if (tx == modbus->curr)
359  modbus->curr = NULL;
360 
361  if (tx->decoder_events != NULL) {
362  if (tx->decoder_events->cnt <= modbus->events)
363  modbus->events -= tx->decoder_events->cnt;
364  else
365  modbus->events = 0;
366  }
367 
368  modbus->unreplied_cnt--;
369 
370  /* Check flood limit */
371  if ((modbus->givenup == 1) &&
372  (request_flood != 0) &&
373  (modbus->unreplied_cnt < request_flood) )
374  modbus->givenup = 0;
375 
376  TAILQ_REMOVE(&modbus->tx_list, tx, next);
377  ModbusTxFree(tx);
378  break;
379  }
380  SCReturn;
381 }

450  * \brief Check Modbus header
451  *
452  * \param tx Pointer to Modbus Transaction structure
453  * \param modbus Pointer to Modbus state structure
454  * \param header Pointer to Modbus header state in which the value to be stored

456 static void ModbusCheckHeader(ModbusState *modbus,
457  ModbusHeader *header)
458 {
459  SCEnter();
460  /* MODBUS protocol is identified by the value 0. */
461  if (header->protocolId != MODBUS_PROTOCOL_VER)
462  ModbusSetEvent(modbus, MODBUS_DECODER_EVENT_INVALID_PROTOCOL_ID);
463 
464  /* Check Length field that is a byte count of the following fields */

466  (header->length > MODBUS_MAX_ADU_LEN) )
467  ModbusSetEvent(modbus, MODBUS_DECODER_EVENT_INVALID_LENGTH);

477 /** \internal
478  * \brief Parse Exception Response and verify protocol compliance.
479  *
480  * \param tx Pointer to Modbus Transaction structure
481  * \param modbus Pointer to Modbus state structure
482  * \param input Pointer the received input data
483  * \param input_len Length of the received input data
484  * \param offset Offset of the received input data pointer
485  */
486 static void ModbusExceptionResponse(ModbusTransaction *tx,
487  ModbusState *modbus,
488  uint8_t *input,
489  uint32_t input_len,
490  uint16_t *offset)
491 {
492  SCEnter();
493  uint8_t exception;
494 
495  /* Exception code (1 byte) */
496  if (ModbusExtractUint8(modbus, &exception, input, input_len, offset))
497  SCReturn;
498 
499  switch (exception) {
500  case MODBUS_ERROR_CODE_ILLEGAL_FUNCTION:
501  case MODBUS_ERROR_CODE_SERVER_DEVICE_FAILURE:
502  break;
503  case MODBUS_ERROR_CODE_ILLEGAL_DATA_VALUE:
504  if (tx->function == MODBUS_FUNC_DIAGNOSTIC) {
505  break;
506  }
507  /* Fallthrough */
508  case MODBUS_ERROR_CODE_ILLEGAL_DATA_ADDRESS:
509  if ( (tx->type & MODBUS_TYP_ACCESS_FUNCTION_MASK) ||
510  (tx->function == MODBUS_FUNC_READFIFOQUEUE) ||
511  (tx->function == MODBUS_FUNC_ENCAPINTTRANS)) {
512  break;
513  }
514  /* Fallthrough */
515  case MODBUS_ERROR_CODE_MEMORY_PARITY_ERROR:
516  if ( (tx->function == MODBUS_FUNC_READFILERECORD) ||
517  (tx->function == MODBUS_FUNC_WRITEFILERECORD) ) {
518  break;
519  }
520  /* Fallthrough */
521  default:
522  ModbusSetEvent(modbus, MODBUS_DECODER_EVENT_INVALID_EXCEPTION_CODE);
523  break;
524  }
525 
526  SCReturn;
527 }

529 /** \internal
530  * \brief Parse Read data Request, complete Transaction structure
531  * and verify protocol compliance.
532  *
533  * \param tx Pointer to Modbus Transaction structure
534  * \param modbus Pointer to Modbus state structure
535  * \param input Pointer the received input data
536  * \param input_len Length of the received input data
537  * \param offset Offset of the received input data pointer
538  */
539 static void ModbusParseReadRequest(ModbusTransaction *tx,
540  ModbusState *modbus,
541  uint8_t *input,
542  uint32_t input_len,
543  uint16_t *offset)
544 {
545  SCEnter();
546  uint16_t quantity;
547  uint8_t type = tx->type;
548 
549  /* Starting Address (2 bytes) */
550  if (ModbusExtractUint16(modbus, &(tx->read.address), input, input_len, offset))
551  goto end;
552 
553  /* Quantity (2 bytes) */
554  if (ModbusExtractUint16(modbus, &(tx->read.quantity), input, input_len, offset))
555  goto end;
556  quantity = tx->read.quantity;
557 
558  /* Check Quantity range */
559  if (type & MODBUS_TYP_BIT_ACCESS_MASK) {
560  if ((quantity == MODBUS_MIN_QUANTITY) ||
561  (quantity > MODBUS_MAX_QUANTITY_IN_BIT_ACCESS))
562  goto error;
563  } else {
564  if ((quantity == MODBUS_MIN_QUANTITY) ||
565  (quantity > MODBUS_MAX_QUANTITY_IN_WORD_ACCESS))
566  goto error;
567  }
568 
569  if (~type & MODBUS_TYP_WRITE)
570  /* Except from Read/Write Multiple Registers function (code 23) */
571  /* The length of all Read Data function requests is 6 bytes */
572  /* Modbus Application Protocol Specification V1.1b3 from 6.1 to 6.4 */
573  ModbusCheckHeaderLength(modbus, tx->length, 6);
574 
575  goto end;
576 
577 error:
578  ModbusSetEvent(modbus, MODBUS_DECODER_EVENT_INVALID_VALUE);
579 end:
580  SCReturn;
581 }

583 /** \internal
584  * \brief Parse Read data Response and verify protocol compliance
585  *
586  * \param tx Pointer to Modbus Transaction structure
587  * \param modbus Pointer to Modbus state structure
588  * \param input Pointer the received input data
589  * \param input_len Length of the received input data
590  * \param offset Offset of the received input data pointer
591  */
592 static void ModbusParseReadResponse(ModbusTransaction *tx,
593  ModbusState *modbus,
594  uint8_t *input,
595  uint32_t input_len,
596  uint16_t *offset)
597 {
598  SCEnter();
599  uint8_t count;
600 
601  /* Count (1 bytes) */
602  if (ModbusExtractUint8(modbus, &count, input, input_len, offset))
603  goto end;
604 
605  /* Check Count range and value according to the request */
606  if ((tx->type) & MODBUS_TYP_BIT_ACCESS_MASK) {
607  if ( (count < MODBUS_MIN_COUNT) ||
608  (count > MODBUS_MAX_COUNT) ||
609  (count != CEIL(tx->read.quantity)))
610  goto error;
611  } else {
612  if ( (count == MODBUS_MIN_COUNT) ||
613  (count > MODBUS_MAX_COUNT) ||
614  (count != (2 * (tx->read.quantity))))
615  goto error;
616  }
617 
618  /* Except from Read/Write Multiple Registers function (code 23) */
619  /* The length of all Read Data function responses is (3 bytes + count) */
620  /* Modbus Application Protocol Specification V1.1b3 from 6.1 to 6.4 */
621  ModbusCheckHeaderLength(modbus, tx->length, 3 + count);
622  goto end;
623 
624 error:
625  ModbusSetEvent(modbus, MODBUS_DECODER_EVENT_VALUE_MISMATCH);
626 end:
627  SCReturn;
628 }

635  * \param modbus Pointer to Modbus state structure
636  * \param input Pointer the received input data
637  * \param input_len Length of the received input data
638  * \param offset Offset of the received input data pointer
639  *
640  * \retval On success returns 0 or on failure returns -1.
641  */
642 static int ModbusParseWriteRequest(ModbusTransaction *tx,
643  ModbusState *modbus,
644  uint8_t *input,
645  uint32_t input_len,
646  uint16_t *offset)
647 {
648  SCEnter();
649  uint16_t quantity = 1, word;
650  uint8_t byte, count = 1, type = tx->type;
651 
652  int i = 0;
653 
654  /* Starting/Output/Register Address (2 bytes) */
655  if (ModbusExtractUint16(modbus, &(tx->write.address), input, input_len, offset))
656  goto end;
657 
658  if (type & MODBUS_TYP_SINGLE) {
659  /* The length of Write Single Coil (code 5) and */
660  /* Write Single Register (code 6) requests is 6 bytes */
661  /* Modbus Application Protocol Specification V1.1b3 6.5 and 6.6 */
662  if (ModbusCheckHeaderLength(modbus, tx->length, 6))
663  goto end;
664  } else if (type & MODBUS_TYP_MULTIPLE) {
665  /* Quantity (2 bytes) */
666  if (ModbusExtractUint16(modbus, &quantity, input, input_len, offset))
667  goto end;
668  tx->write.quantity = quantity;
669 
670  /* Count (1 bytes) */
671  if (ModbusExtractUint8(modbus, &count, input, input_len, offset))
672  goto end;
673  tx->write.count = count;
674 
675  if (type & MODBUS_TYP_BIT_ACCESS_MASK) {
676  /* Check Quantity range and conversion in byte (count) */
677  if ((quantity == MODBUS_MIN_QUANTITY) ||
678  (quantity > MODBUS_MAX_QUANTITY_IN_BIT_ACCESS) ||
679  (quantity != CEIL(count)))
680  goto error;
681 
682  /* The length of Write Multiple Coils (code 15) request is (7 + count) */
683  /* Modbus Application Protocol Specification V1.1b3 6.11 */
684  if (ModbusCheckHeaderLength(modbus, tx->length, 7 + count))
685  goto end;
686  } else {
687  /* Check Quantity range and conversion in byte (count) */
688  if ((quantity == MODBUS_MIN_QUANTITY) ||
689  (quantity > MODBUS_MAX_QUANTITY_IN_WORD_ACCESS) ||
690  (count != (2 * quantity)))
691  goto error;
692 
693  if (type & MODBUS_TYP_READ) {
694  /* The length of Read/Write Multiple Registers function (code 23) */
695  /* request is (11 bytes + count) */
696  /* Modbus Application Protocol Specification V1.1b3 6.17 */
697  if (ModbusCheckHeaderLength(modbus, tx->length, 11 + count))
698  goto end;
699  } else {
700  /* The length of Write Multiple Coils (code 15) and */
701  /* Write Multiple Registers (code 16) functions requests is (7 bytes + count) */
702  /* Modbus Application Protocol Specification V1.1b3 from 6.11 and 6.12 */
703  if (ModbusCheckHeaderLength(modbus, tx->length, 7 + count))
704  goto end;
705  }
706  }
707  } else {
708  /* Mask Write Register function (And_Mask and Or_Mask) */
709  quantity = 2;
710 
711  /* The length of Mask Write Register (code 22) function request is 8 */
712  /* Modbus Application Protocol Specification V1.1b3 6.16 */
713  if (ModbusCheckHeaderLength(modbus, tx->length, 8))
714  goto end;
715  }
716 
717  if (type & MODBUS_TYP_COILS) {
718  /* Output value (data block) unit is count */
719  tx->data = (uint16_t *) SCCalloc(1, count * sizeof(uint16_t));
720  if (unlikely(tx->data == NULL))
721  SCReturnInt(-1);
722 
723  if (type & MODBUS_TYP_SINGLE) {
724  /* Outputs value (2 bytes) */
725  if (ModbusExtractUint16(modbus, &word, input, input_len, offset))
726  goto end;
727  tx->data[i] = word;
728 
729  if ((word != 0x00) && (word != 0xFF00))
730  goto error;
731  } else {
732  for (i = 0; i < count; i++) {
733  /* Outputs value (1 byte) */
734  if (ModbusExtractUint8(modbus, &byte, input, input_len, offset))
735  goto end;
736  tx->data[i] = (uint16_t) byte;
737  }
738  }
739  } else {
740  /* Registers value (data block) unit is quantity */
741  tx->data = (uint16_t *) SCCalloc(1, quantity * sizeof(uint16_t));
742  if (unlikely(tx->data == NULL))
743  SCReturnInt(-1);
744 
745  for (i = 0; i < quantity; i++) {
746  /* Outputs/Registers value (2 bytes) */
747  if (ModbusExtractUint16(modbus, &word, input, input_len, offset))
748  goto end;
749  tx->data[i] = word;
750  }
751  }
752  goto end;
753 
754 error:
755  ModbusSetEvent(modbus, MODBUS_DECODER_EVENT_INVALID_VALUE);
756 end:
757  SCReturnInt(0);
758 }

766  * \param input_len Length of the received input data

793  goto error;

798

800  if (type & MODBUS_TYP_BIT_ACCESS_MASK) {

806  (quantity > MODBUS_MAX_QUANTITY_IN_BIT_ACCESS))

808  }

824

830  /* Modbus Application Protocol Specification V1.1b3 6.16 */

1097
