46 int DecodeRaw(ThreadVars *tv, DecodeThreadVars *dtv, Packet *p, uint8_t *pkt, uint16_t len, PacketQueue *pq)
47 {
48  StatsIncr(tv, dtv->counter_raw);
49 
50  /* If it is ipv4 or ipv6 it should at least be the size of ipv4 */
51  if (unlikely(len < IPV4_HEADER_LEN)) {
52  ENGINE_SET_INVALID_EVENT(p, IPV4_PKT_TOO_SMALL);
53  return TM_ECODE_FAILED;
54  }
55 
56  if (IP_GET_RAW_VER(pkt) == 4) {
57  SCLogDebug("IPV4 Packet");
58  DecodeIPV4(tv, dtv, p, GET_PKT_DATA(p), GET_PKT_LEN(p), pq);
59  } else if (IP_GET_RAW_VER(pkt) == 6) {
60  SCLogDebug("IPV6 Packet");
61  DecodeIPV6(tv, dtv, p, GET_PKT_DATA(p), GET_PKT_LEN(p), pq);
62  } else {
63  SCLogDebug("Unknown ip version %" PRIu8 "", IP_GET_RAW_VER(pkt));
64  ENGINE_SET_EVENT(p,IPRAW_INVALID_IPV);
65  }
66  return TM_ECODE_OK;
67 }
