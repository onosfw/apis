61 /**
62  * \internal
63  * \brief Setup l3_proto keyword.
64  *
65  * \param de_ctx Detection engine context
66  * \param s Signature
67  * \param optstr Options string
68  *
69  * \return Non-zero on error
70  */
71 static int DetectL3ProtoSetup(DetectEngineCtx *de_ctx, Signature *s, char *optstr)
72 {
73  char *str = optstr;
74  char dubbed = 0;
75 
76  /* strip "'s */
77  if (optstr[0] == '\"' && optstr[strlen(optstr) - 1] == '\"') {
78  str = SCStrdup(optstr + 1);
79  if (unlikely(str == NULL))
80  goto error;
81  str[strlen(optstr) - 2] = '\0';
82  dubbed = 1;
83  }
84 
85  /* reset possible any value */
86  if (s->proto.flags & DETECT_PROTO_ANY) {
87  s->proto.flags &= ~DETECT_PROTO_ANY;
88  }
89 
90  /* authorized value, ip, any, ip4, ipv4, ip6, ipv6 */
91  if (strcasecmp(str,"ipv4") == 0 ||
92  strcasecmp(str,"ip4") == 0 ) {
93  if (s->proto.flags & DETECT_PROTO_IPV6) {
94  SCLogError(SC_ERR_INVALID_SIGNATURE, "Conflicting l3 proto specified");
95  goto error;
96  }
97  s->proto.flags |= DETECT_PROTO_IPV4;
98  SCLogDebug("IPv4 protocol detected");
99  } else if (strcasecmp(str,"ipv6") == 0 ||
100  strcasecmp(str,"ip6") == 0 ) {
101  if (s->proto.flags & DETECT_PROTO_IPV6) {
102  SCLogError(SC_ERR_INVALID_SIGNATURE, "Conflicting l3 proto specified");
103  goto error;
104  }
105  s->proto.flags |= DETECT_PROTO_IPV6;
106  SCLogDebug("IPv6 protocol detected");
107  } else {
108  SCLogError(SC_ERR_INVALID_SIGNATURE, "Invalid l3 proto: \"%s\"", str);
109  goto error;
110  }
111 
112  if (dubbed)
113  SCFree(str);
114  return 0;
115 error:
116  if (dubbed)
117  SCFree(str);
118  return -1;
119 }
