92 static int LogTcpDataLoggerDir(ThreadVars *tv, void *thread_data, const Flow *f,
93  const uint8_t *data, uint32_t data_len, uint64_t tx_id, uint8_t flags)
94 {
95  SCEnter();
96  LogTcpDataLogThread *aft = thread_data;
97  LogTcpDataFileCtx *td = aft->tcpdatalog_ctx;
98  char *mode = "a";
99 
100  if (flags & OUTPUT_STREAMING_FLAG_OPEN)
101  mode = "w";
102 
103  if (data && data_len) {
104  char srcip[46] = "", dstip[46] = "";
105  if (FLOW_IS_IPV4(f)) {
106  PrintInet(AF_INET, (const void *)&f->src.addr_data32[0], srcip, sizeof(srcip));
107  PrintInet(AF_INET, (const void *)&f->dst.addr_data32[0], dstip, sizeof(dstip));
108  } else if (FLOW_IS_IPV6(f)) {
109  PrintInet(AF_INET6, (const void *)f->src.addr_data32, srcip, sizeof(srcip));
110  PrintInet(AF_INET6, (const void *)f->dst.addr_data32, dstip, sizeof(dstip));
111  }
112 
113  char name[PATH_MAX];
114 
115  char tx[64] = "";
116  if (flags & OUTPUT_STREAMING_FLAG_TRANSACTION)
117  snprintf(tx, sizeof(tx), "%"PRIu64, tx_id);
118 
119  snprintf(name, sizeof(name), "%s/%s/%s_%u-%s_%u-%s-%s.data",
120  td->log_dir,
121  td->type == STREAMING_HTTP_BODIES ? "http" : "tcp",
122  srcip, f->sp, dstip, f->dp, tx,
123  flags & OUTPUT_STREAMING_FLAG_TOSERVER ? "ts" : "tc");
124 
125  FILE *fp = fopen(name, mode);
126  BUG_ON(fp == NULL);
127 
128  // PrintRawDataFp(stdout, (uint8_t *)data, data_len);
129  fwrite(data, data_len, 1, fp);
130 
131  fclose(fp);
132  }
133  SCReturnInt(TM_ECODE_OK);
134 }

136 static int LogTcpDataLoggerFile(ThreadVars *tv, void *thread_data, const Flow *f,
137  const uint8_t *data, uint32_t data_len, uint64_t tx_id, uint8_t flags)
138 {
139  SCEnter();
140  LogTcpDataLogThread *aft = thread_data;
141  LogTcpDataFileCtx *td = aft->tcpdatalog_ctx;
142 
143  if (data && data_len) {
144  MemBufferReset(aft->buffer);
145 
146  char srcip[46] = "", dstip[46] = "";
147  if (FLOW_IS_IPV4(f)) {
148  PrintInet(AF_INET, (const void *)&f->src.addr_data32[0], srcip, sizeof(srcip));
149  PrintInet(AF_INET, (const void *)&f->dst.addr_data32[0], dstip, sizeof(dstip));
150  } else if (FLOW_IS_IPV6(f)) {
151  PrintInet(AF_INET6, (const void *)f->src.addr_data32, srcip, sizeof(srcip));
152  PrintInet(AF_INET6, (const void *)f->dst.addr_data32, dstip, sizeof(dstip));
153  }
154 
155  char name[PATH_MAX];
156  snprintf(name, sizeof(name), "%s_%u-%s_%u-%s:",
157  srcip, f->sp, dstip, f->dp,
158  flags & OUTPUT_STREAMING_FLAG_TOSERVER ? "ts" : "tc");
159 
160  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset,
161  aft->buffer->size, (uint8_t *)name,strlen(name));
162  MemBufferWriteString(aft->buffer, "\n");
163 
164  PrintRawDataToBuffer(aft->buffer->buffer, &aft->buffer->offset,
165  aft->buffer->size, (uint8_t *)data,data_len);
166 
167  SCMutexLock(&td->file_ctx->fp_mutex);
168  td->file_ctx->Write((const char *)MEMBUFFER_BUFFER(aft->buffer),
169  MEMBUFFER_OFFSET(aft->buffer), td->file_ctx);
170  SCMutexUnlock(&td->file_ctx->fp_mutex);
171  }
172  SCReturnInt(TM_ECODE_OK);
173 }

175 int LogTcpDataLogger(ThreadVars *tv, void *thread_data, const Flow *f,
176  const uint8_t *data, uint32_t data_len, uint64_t tx_id, uint8_t flags)
177 {
178  SCEnter();
179  LogTcpDataLogThread *aft = thread_data;
180  LogTcpDataFileCtx *td = aft->tcpdatalog_ctx;
181 
182  if (td->dir == 1)
183  LogTcpDataLoggerDir(tv, thread_data, f, data, data_len, tx_id, flags);
184  if (td->file == 1)
185  LogTcpDataLoggerFile(tv, thread_data, f, data, data_len, tx_id, flags);
186 
187  SCReturnInt(TM_ECODE_OK);
188 }

190 TmEcode LogTcpDataLogThreadInit(ThreadVars *t, void *initdata, void **data)
191 {
192  LogTcpDataLogThread *aft = SCMalloc(sizeof(LogTcpDataLogThread));
193  if (unlikely(aft == NULL))
194  return TM_ECODE_FAILED;
195  memset(aft, 0, sizeof(LogTcpDataLogThread));
196 
197  if(initdata == NULL)
198  {
199  SCLogDebug("Error getting context. \"initdata\" argument NULL");
200  SCFree(aft);
201  return TM_ECODE_FAILED;
202  }
203 
204  aft->buffer = MemBufferCreateNew(OUTPUT_BUFFER_SIZE);
205  if (aft->buffer == NULL) {
206  SCFree(aft);
207  return TM_ECODE_FAILED;
208  }
209 
210  /* Use the Ouptut Context (file pointer and mutex) */
211  aft->tcpdatalog_ctx= ((OutputCtx *)initdata)->data;
212 
213  *data = (void *)aft;
214  return TM_ECODE_OK;
215 }

239 /** \brief Create a new http log LogFileCtx.
240  * \param conf Pointer to ConfNode containing this loggers configuration.
241  * \return NULL if failure, LogFileCtx* to the file_ctx if succesful
242  * */
243 OutputCtx *LogTcpDataLogInitCtx(ConfNode *conf)
244 {
245  char filename[PATH_MAX] = "";
246  char dirname[32] = "";
247  strlcpy(filename, DEFAULT_LOG_FILENAME, sizeof(filename));
248 
249  LogFileCtx *file_ctx = LogFileNewCtx();
250  if(file_ctx == NULL) {
251  SCLogError(SC_ERR_HTTP_LOG_GENERIC, "couldn't create new file_ctx");
252  return NULL;
253  }
254 
255  LogTcpDataFileCtx *tcpdatalog_ctx = SCMalloc(sizeof(LogTcpDataFileCtx));
256  if (unlikely(tcpdatalog_ctx == NULL)) {
257  LogFileFreeCtx(file_ctx);
258  return NULL;
259  }
260  memset(tcpdatalog_ctx, 0x00, sizeof(LogTcpDataFileCtx));
261 
262  tcpdatalog_ctx->file_ctx = file_ctx;
263 
264  if (conf) {
265  if (conf->name) {
266  if (strcmp(conf->name, "tcp-data") == 0) {
267  tcpdatalog_ctx->type = STREAMING_TCP_DATA;
268  snprintf(filename, sizeof(filename), "%s.log", conf->name);
269  strlcpy(dirname, "tcp", sizeof(dirname));
270  } else if (strcmp(conf->name, "http-body-data") == 0) {
271  tcpdatalog_ctx->type = STREAMING_HTTP_BODIES;
272  snprintf(filename, sizeof(filename), "%s.log", conf->name);
273  strlcpy(dirname, "http", sizeof(dirname));
274  }
275  }
276 
277  const char *logtype = ConfNodeLookupChildValue(conf, "type");
278  if (logtype == NULL)
279  logtype = "file";
280 
281  if (strcmp(logtype, "file") == 0) {
282  tcpdatalog_ctx->file = 1;
283  } else if (strcmp(logtype, "dir") == 0) {
284  tcpdatalog_ctx->dir = 1;
285  } else if (strcmp(logtype, "both") == 0) {
286  tcpdatalog_ctx->file = 1;
287  tcpdatalog_ctx->dir = 1;
288  }
289  } else {
290  tcpdatalog_ctx->file = 1;
291  tcpdatalog_ctx->dir = 0;
292  }
293 
294  if (tcpdatalog_ctx->file == 1) {
295  SCLogInfo("opening logfile");
296  if (SCConfLogOpenGeneric(conf, file_ctx, filename, 1) < 0) {
297  LogFileFreeCtx(file_ctx);
298  SCFree(tcpdatalog_ctx);
299  return NULL;
300  }
301  }
302 
303  if (tcpdatalog_ctx->dir == 1) {
304  tcpdatalog_ctx->log_dir = ConfigGetLogDirectory();
305  char dirfull[PATH_MAX];
306 
307  /* create the filename to use */
308  snprintf(dirfull, PATH_MAX, "%s/%s", tcpdatalog_ctx->log_dir, dirname);
309 
310  SCLogInfo("using directory %s", dirfull);
311 
312  /* if mkdir fails file open will fail, so deal with errors there */
313 #ifndef OS_WIN32
314  (void)mkdir(dirfull, 0700);
315 #else
316  (void)mkdir(dirfull);
317 #endif
318  }
319 
320  OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
321  if (unlikely(output_ctx == NULL)) {
322  goto parsererror;
323  }
324 
325  output_ctx->data = tcpdatalog_ctx;
326  output_ctx->DeInit = LogTcpDataLogDeInitCtx;
327 
328  SCLogDebug("Streaming log output initialized");
329  return output_ctx;
330 
331 parsererror:
332  LogFileFreeCtx(file_ctx);
333  SCFree(tcpdatalog_ctx);
334  SCLogError(SC_ERR_INVALID_ARGUMENT,"Syntax error in custom http log format string.");
335  return NULL;
336 
337 }
