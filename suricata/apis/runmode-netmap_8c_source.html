96 /**
97 * \brief extract information from config file
98 *
99 * The returned structure will be freed by the thread init function.
100 * This is thus necessary to or copy the structure before giving it
101 * to thread or to reparse the file for each thread (and thus have
102 * new structure.
103 *
104 * \return a NetmapIfaceConfig corresponding to the interface name
105 */
106 static void *ParseNetmapConfig(const char *iface_name)
107 {
108  char *threadsstr = NULL;
109  ConfNode *if_root;
110  ConfNode *if_default = NULL;
111  ConfNode *netmap_node;
112  NetmapIfaceConfig *aconf = SCMalloc(sizeof(*aconf));
113  char *tmpctype;
114  char *copymodestr;
115  int boolval;
116  char *bpf_filter = NULL;
117  char *out_iface = NULL;
118 
119  if (unlikely(aconf == NULL)) {
120  return NULL;
121  }
122 
123  if (iface_name == NULL) {
124  SCFree(aconf);
125  return NULL;
126  }
127 
128  memset(aconf, 0, sizeof(*aconf));
129  aconf->DerefFunc = NetmapDerefConfig;
130  aconf->threads = 1;
131  aconf->promisc = 1;
132  aconf->checksum_mode = CHECKSUM_VALIDATION_AUTO;
133  aconf->copy_mode = NETMAP_COPY_MODE_NONE;
134  strlcpy(aconf->iface_name, iface_name, sizeof(aconf->iface_name));
135  SC_ATOMIC_INIT(aconf->ref);
136  (void) SC_ATOMIC_ADD(aconf->ref, 1);
137 
138  strlcpy(aconf->iface, aconf->iface_name, sizeof(aconf->iface));
139  if (aconf->iface[0]) {
140  size_t len = strlen(aconf->iface);
141  if (aconf->iface[len-1] == '+') {
142  aconf->iface[len-1] = '\0';
143  aconf->iface_sw = 1;
144  }
145  }
146 
147  if (ConfGet("bpf-filter", &bpf_filter) == 1) {
148  if (strlen(bpf_filter) > 0) {
149  aconf->bpf_filter = bpf_filter;
150  SCLogInfo("Going to use command-line provided bpf filter '%s'",
151  aconf->bpf_filter);
152  }
153  }
154 
155  /* Find initial node */
156  netmap_node = ConfGetNode("netmap");
157  if (netmap_node == NULL) {
158  SCLogInfo("Unable to find netmap config using default value");
159  return aconf;
160  }
161 
162  if_root = ConfNodeLookupKeyValue(netmap_node, "interface", aconf->iface_name);
163 
164  if_default = ConfNodeLookupKeyValue(netmap_node, "interface", "default");
165 
166  if (if_root == NULL && if_default == NULL) {
167  SCLogInfo("Unable to find netmap config for "
168  "interface \"%s\" or \"default\", using default value",
169  aconf->iface_name);
170  return aconf;
171  }
172 
173  /* If there is no setting for current interface use default one as main iface */
174  if (if_root == NULL) {
175  if_root = if_default;
176  if_default = NULL;
177  }
178 
179  if (ConfGetChildValueWithDefault(if_root, if_default, "threads", &threadsstr) != 1) {
180  aconf->threads = 1;
181  } else {
182  if (strcmp(threadsstr, "auto") == 0) {
183  aconf->threads = GetIfaceRSSQueuesNum(aconf->iface);
184  } else {
185  aconf->threads = (uint8_t)atoi(threadsstr);
186  }
187  }
188 
189  if (aconf->threads <= 0) {
190  aconf->threads = 1;
191  }
192  if (aconf->threads) {
193  SCLogInfo("Using %d threads for interface %s", aconf->threads,
194  aconf->iface_name);
195  }
196 
197  if (ConfGetChildValueWithDefault(if_root, if_default, "copy-iface", &out_iface) == 1) {
198  if (strlen(out_iface) > 0) {
199  aconf->out_iface_name = out_iface;
200  }
201  }
202 
203  if (ConfGetChildValueWithDefault(if_root, if_default, "copy-mode", &copymodestr) == 1) {
204  if (aconf->out_iface_name == NULL) {
205  SCLogInfo("Copy mode activated but no destination"
206  " iface. Disabling feature");
207  } else if (strlen(copymodestr) <= 0) {
208  aconf->out_iface_name = NULL;
209  } else if (strcmp(copymodestr, "ips") == 0) {
210  SCLogInfo("Netmap IPS mode activated %s->%s",
211  aconf->iface_name,
212  aconf->out_iface_name);
213  aconf->copy_mode = NETMAP_COPY_MODE_IPS;
214  } else if (strcmp(copymodestr, "tap") == 0) {
215  SCLogInfo("Netmap TAP mode activated %s->%s",
216  aconf->iface_name,
217  aconf->out_iface_name);
218  aconf->copy_mode = NETMAP_COPY_MODE_TAP;
219  } else {
220  SCLogInfo("Invalid mode (not in tap, ips)");
221  }
222  }
223 
224  if (aconf->out_iface_name && aconf->out_iface_name[0]) {
225  strlcpy(aconf->out_iface, aconf->out_iface_name,
226  sizeof(aconf->out_iface));
227  size_t len = strlen(aconf->out_iface);
228  if (aconf->out_iface[len-1] == '+') {
229  aconf->out_iface[len-1] = '\0';
230  aconf->out_iface_sw = 1;
231  }
232  }
233 
234  SC_ATOMIC_RESET(aconf->ref);
235  (void) SC_ATOMIC_ADD(aconf->ref, aconf->threads);
236 
237  /* load netmap bpf filter */
238  /* command line value has precedence */
239  if (ConfGet("bpf-filter", &bpf_filter) != 1) {
240  if (ConfGetChildValueWithDefault(if_root, if_default, "bpf-filter", &bpf_filter) == 1) {
241  if (strlen(bpf_filter) > 0) {
242  aconf->bpf_filter = bpf_filter;
243  SCLogInfo("Going to use bpf filter %s", aconf->bpf_filter);
244  }
245  }
246  }
247 
248  (void)ConfGetChildValueBoolWithDefault(if_root, if_default, "disable-promisc", (int *)&boolval);
249  if (boolval) {
250  SCLogInfo("Disabling promiscuous mode on iface %s", aconf->iface);
251  aconf->promisc = 0;
252  }
253 
254  if (ConfGetChildValueWithDefault(if_root, if_default, "checksum-checks", &tmpctype) == 1) {
255  if (strcmp(tmpctype, "auto") == 0) {
256  aconf->checksum_mode = CHECKSUM_VALIDATION_AUTO;
257  } else if (strcmp(tmpctype, "yes") == 0) {
258  aconf->checksum_mode = CHECKSUM_VALIDATION_ENABLE;
259  } else if (strcmp(tmpctype, "no") == 0) {
260  aconf->checksum_mode = CHECKSUM_VALIDATION_DISABLE;
261  } else {
262  SCLogError(SC_ERR_INVALID_ARGUMENT, "Invalid value for checksum-checks for %s", aconf->iface_name);
263  }
264  }
265 
266  return aconf;
267 }

275 int NetmapRunModeIsIPS()
276 {
277  int nlive = LiveGetDeviceCount();
278  int ldev;
279  ConfNode *if_root;
280  ConfNode *if_default = NULL;
281  ConfNode *netmap_node;
282  int has_ips = 0;
283  int has_ids = 0;
284 
285  /* Find initial node */
286  netmap_node = ConfGetNode("netmap");
287  if (netmap_node == NULL) {
288  return 0;
289  }
290 
291  if_default = ConfNodeLookupKeyValue(netmap_node, "interface", "default");
292 
293  for (ldev = 0; ldev < nlive; ldev++) {
294  char *live_dev = LiveGetDeviceName(ldev);
295  if (live_dev == NULL) {
296  SCLogError(SC_ERR_INVALID_VALUE, "Problem with config file");
297  return 0;
298  }
299  char *copymodestr = NULL;
300  if_root = ConfNodeLookupKeyValue(netmap_node, "interface", live_dev);
301 
302  if (if_root == NULL) {
303  if (if_default == NULL) {
304  SCLogError(SC_ERR_INVALID_VALUE, "Problem with config file");
305  return 0;
306  }
307  if_root = if_default;
308  }
309 
310  if (ConfGetChildValueWithDefault(if_root, if_default, "copy-mode", &copymodestr) == 1) {
311  if (strcmp(copymodestr, "ips") == 0) {
312  has_ips = 1;
313  } else {
314  has_ids = 1;
315  }
316  } else {
317  has_ids = 1;
318  }
319  }
320 
321  if (has_ids && has_ips) {
322  SCLogInfo("Netmap mode using IPS and IDS mode");
323  for (ldev = 0; ldev < nlive; ldev++) {
324  char *live_dev = LiveGetDeviceName(ldev);
325  if (live_dev == NULL) {
326  SCLogError(SC_ERR_INVALID_VALUE, "Problem with config file");
327  return 0;
328  }
329  if_root = ConfNodeLookupKeyValue(netmap_node, "interface", live_dev);
330  char *copymodestr = NULL;
331 
332  if (if_root == NULL) {
333  if (if_default == NULL) {
334  SCLogError(SC_ERR_INVALID_VALUE, "Problem with config file");
335  return 0;
336  }
337  if_root = if_default;
338  }
339 
340  if (! ((ConfGetChildValueWithDefault(if_root, if_default, "copy-mode", &copymodestr) == 1) &&
341  (strcmp(copymodestr, "ips") == 0))) {
342  SCLogError(SC_ERR_INVALID_ARGUMENT,
343  "Netmap IPS mode used and interface '%s' is in IDS or TAP mode. "
344  "Sniffing '%s' but expect bad result as stream-inline is activated.",
345  live_dev, live_dev);
346  }
347  }
348  }
349 
350  return has_ips;
351 }
