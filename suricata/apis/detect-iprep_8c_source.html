94 static uint8_t GetHostRepSrc(Packet *p, uint8_t cat, uint32_t version)
95 {
96  uint8_t val = 0;
97  Host *h = NULL;
98 
99  if (p->flags & PKT_HOST_SRC_LOOKED_UP && p->host_src == NULL) {
100  return 0;
101  } else if (p->host_src != NULL) {
102  h = (Host *)p->host_src;
103  HostLock(h);
104  } else {
105  h = HostLookupHostFromHash(&(p->src));
106 
107  p->flags |= PKT_HOST_SRC_LOOKED_UP;
108 
109  if (h == NULL)
110  return 0;
111 
112  HostReference(&p->host_src, h);
113  }
114 
115  if (h->iprep == NULL) {
116  HostRelease(h);
117  return 0;
118  }
119 
120  SReputation *r = (SReputation *)h->iprep;
121 
122  /* allow higher versions as this happens during
123  * rule reload */
124  if (r->version >= version)
125  val = r->rep[cat];
126  else
127  SCLogDebug("version mismatch %u != %u", r->version, version);
128 
129  HostRelease(h);
130  return val;
131 }

173 static inline int RepMatch(uint8_t op, uint8_t val1, uint8_t val2)
174 {
175  if (op == DETECT_IPREP_OP_GT && val1 > val2) {
176  return 1;
177  } else if (op == DETECT_IPREP_OP_LT && val1 < val2) {
178  return 1;
179  } else if (op == DETECT_IPREP_OP_EQ && val1 == val2) {
180  return 1;
181  }
182  return 0;
183 }

185 /*
186  * returns 0: no match
187  * 1: match
188  * -1: error
189  */
190 int DetectIPRepMatch (ThreadVars *t, DetectEngineThreadCtx *det_ctx, Packet *p, Signature *s, const SigMatchCtx *ctx)
191 {
192  const DetectIPRepData *rd = (const DetectIPRepData *)ctx;
193  if (rd == NULL)
194  return 0;
195 
196  uint32_t version = det_ctx->de_ctx->srep_version;
197  uint8_t val = 0;
198 
199  SCLogDebug("rd->cmd %u", rd->cmd);
200  switch(rd->cmd) {
201  case DETECT_IPREP_CMD_ANY:
202  val = GetHostRepSrc(p, rd->cat, version);
203  if (val == 0)
204  val = SRepCIDRGetIPRepSrc(det_ctx->de_ctx->srepCIDR_ctx, p, rd->cat, version);
205  if (val > 0) {
206  if (RepMatch(rd->op, val, rd->val) == 1)
207  return 1;
208  }
209  val = GetHostRepDst(p, rd->cat, version);
210  if (val == 0)
211  val = SRepCIDRGetIPRepDst(det_ctx->de_ctx->srepCIDR_ctx, p, rd->cat, version);
212  if (val > 0) {
213  return RepMatch(rd->op, val, rd->val);
214  }
215  break;
216 
217  case DETECT_IPREP_CMD_SRC:
218  val = GetHostRepSrc(p, rd->cat, version);
219  SCLogDebug("checking src -- val %u (looking for cat %u, val %u)", val, rd->cat, rd->val);
220  if (val == 0)
221  val = SRepCIDRGetIPRepSrc(det_ctx->de_ctx->srepCIDR_ctx, p, rd->cat, version);
222  if (val > 0) {
223  return RepMatch(rd->op, val, rd->val);
224  }
225  break;
226 
227  case DETECT_IPREP_CMD_DST:
228  SCLogDebug("checking dst");
229  val = GetHostRepDst(p, rd->cat, version);
230  if (val == 0)
231  val = SRepCIDRGetIPRepDst(det_ctx->de_ctx->srepCIDR_ctx, p, rd->cat, version);
232  if (val > 0) {
233  return RepMatch(rd->op, val, rd->val);
234  }
235  break;
236 
237  case DETECT_IPREP_CMD_BOTH:
238  val = GetHostRepSrc(p, rd->cat, version);
239  if (val == 0)
240  val = SRepCIDRGetIPRepSrc(det_ctx->de_ctx->srepCIDR_ctx, p, rd->cat, version);
241  if (val == 0 || RepMatch(rd->op, val, rd->val) == 0)
242  return 0;
243  val = GetHostRepDst(p, rd->cat, version);
244  if (val == 0)
245  val = SRepCIDRGetIPRepDst(det_ctx->de_ctx->srepCIDR_ctx, p, rd->cat, version);
246  if (val > 0) {
247  return RepMatch(rd->op, val, rd->val);
248  }
249  break;
250  }
251 
252  return 0;
253 }

255 int DetectIPRepSetup (DetectEngineCtx *de_ctx, Signature *s, char *rawstr)
256 {
257  DetectIPRepData *cd = NULL;
258  SigMatch *sm = NULL;
259  char *cmd_str = NULL, *name = NULL, *op_str = NULL, *value = NULL;
260  uint8_t cmd = 0;
261 #define MAX_SUBSTRINGS 30
262  int ret = 0, res = 0;
263  int ov[MAX_SUBSTRINGS];
264 
265  ret = pcre_exec(parse_regex, parse_regex_study, rawstr, strlen(rawstr), 0, 0, ov, MAX_SUBSTRINGS);
266  if (ret != 5) {
267  SCLogError(SC_ERR_PCRE_MATCH, "\"%s\" is not a valid setting for iprep", rawstr);
268  return -1;
269  }
270 
271  const char *str_ptr;
272  res = pcre_get_substring((char *)rawstr, ov, MAX_SUBSTRINGS, 1, &str_ptr);
273  if (res < 0) {
274  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
275  return -1;
276  }
277  cmd_str = (char *)str_ptr;
278 
279  res = pcre_get_substring((char *)rawstr, ov, MAX_SUBSTRINGS, 2, &str_ptr);
280  if (res < 0) {
281  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
282  goto error;
283  }
284  name = (char *)str_ptr;
285 
286  res = pcre_get_substring((char *)rawstr, ov, MAX_SUBSTRINGS, 3, &str_ptr);
287  if (res < 0) {
288  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
289  goto error;
290  }
291  op_str = (char *)str_ptr;
292 
293  res = pcre_get_substring((char *)rawstr, ov, MAX_SUBSTRINGS, 4, &str_ptr);
294  if (res < 0) {
295  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
296  goto error;
297  }
298  value = (char *)str_ptr;
299 
300  if (strcmp(cmd_str,"any") == 0) {
301  cmd = DETECT_IPREP_CMD_ANY;
302  } else if (strcmp(cmd_str,"both") == 0) {
303  cmd = DETECT_IPREP_CMD_BOTH;
304  } else if (strcmp(cmd_str,"src") == 0) {
305  cmd = DETECT_IPREP_CMD_SRC;
306  } else if (strcmp(cmd_str,"dst") == 0) {
307  cmd = DETECT_IPREP_CMD_DST;
308  } else {
309  SCLogError(SC_ERR_UNKNOWN_VALUE, "ERROR: iprep \"%s\" is not supported.", cmd_str);
310  goto error;
311  }
312 
313  //SCLogInfo("category %s", name);
314  uint8_t cat = SRepCatGetByShortname(name);
315  if (cat == 0) {
316  SCLogError(SC_ERR_UNKNOWN_VALUE, "unknown iprep category \"%s\"", name);
317  goto error;
318  }
319 
320  uint8_t op = 0;
321  uint8_t val = 0;
322 
323  if (op_str == NULL || strlen(op_str) != 1) {
324  goto error;
325  }
326 
327  switch(op_str[0]) {
328  case '<':
329  op = DETECT_IPREP_OP_LT;
330  break;
331  case '>':
332  op = DETECT_IPREP_OP_GT;
333  break;
334  case '=':
335  op = DETECT_IPREP_OP_EQ;
336  break;
337  default:
338  goto error;
339  break;
340  }
341 
342  if (value != NULL && strlen(value) > 0) {
343  int ival = atoi(value);
344  if (ival < 0 || ival > 127)
345  goto error;
346  val = (uint8_t)ival;
347  }
348 
349  cd = SCMalloc(sizeof(DetectIPRepData));
350  if (unlikely(cd == NULL))
351  goto error;
352 
353  cd->cmd = cmd;
354  cd->cat = cat;
355  cd->op = op;
356  cd->val = val;
357  SCLogDebug("cmd %u, cat %u, op %u, val %u", cd->cmd, cd->cat, cd->op, cd->val);
358 
359  pcre_free_substring(name);
360  name = NULL;
361  pcre_free_substring(cmd_str);
362  cmd_str = NULL;
363 
364  /* Okay so far so good, lets get this into a SigMatch
365  * and put it in the Signature. */
366  sm = SigMatchAlloc();
367  if (sm == NULL)
368  goto error;
369 
370  sm->type = DETECT_IPREP;
371  sm->ctx = (SigMatchCtx *)cd;
372 
373  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);
374 
375  return 0;
376 
377 error:
378  if (name != NULL)
379  pcre_free_substring(name);
380  if (cmd_str != NULL)
381  pcre_free_substring(cmd_str);
382  if (cd != NULL)
383  SCFree(cd);
384  if (sm != NULL)
385  SCFree(sm);
386  return -1;
387 }
