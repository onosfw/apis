263  if (bstr_realptr(b) != NULL) {
264  // Refuse to expand a wrapped bstring. In the future,
265  // we can change this to make a copy of the data, thus

341 int bstr_util_cmp_mem(const void *_data1, size_t len1, const void *_data2, size_t len2) {
342  const unsigned char *data1 = (const unsigned char *) _data1;
343  const unsigned char *data2 = (const unsigned char *) _data2;
344  size_t p1 = 0, p2 = 0;
345 
346  while ((p1 < len1) && (p2 < len2)) {
347  if (data1[p1] != data2[p2]) {
348  // Difference.
349  return (data1[p1] < data2[p2]) ? -1 : 1;
350  }
351 
352  p1++;
353  p2++;
354  }
355 
356  if ((p1 == len2) && (p2 == len1)) {
357  // They're identical.
358  return 0;
359  } else {
360  // One string is shorter.
361  if (p1 == len1) return -1;
362  else return 1;
363  }
364 }

391 int64_t bstr_util_mem_to_pint(const void *_data, size_t len, int base, size_t *lastlen) {
392  const unsigned char *data = (unsigned char *) _data;
393  int64_t rval = 0, tval = 0, tflag = 0;
394  size_t i = 0;
395 
396  *lastlen = i;
397 
398  for (i = 0; i < len; i++) {
399  int d = data[i];
400 
401  *lastlen = i;
402 
403  // Convert character to digit.
404  if ((d >= '0') && (d <= '9')) {
405  d -= '0';
406  } else if ((d >= 'a') && (d <= 'z')) {
407  d -= 'a' - 10;
408  } else if ((d >= 'A') && (d <= 'Z')) {
409  d -= 'A' - 10;
410  } else {
411  d = -1;
412  }
413 
414  // Check that the digit makes sense with the base we are using.
415  if ((d == -1) || (d >= base)) {
416  if (tflag) {
417  // Return what we have so far; lastlen points
418  // to the first non-digit position.
419  return rval;
420  } else {
421  // We didn't see a single digit.
422  return -1;
423  }
424  }
425 
426  if (tflag) {
427  rval *= base;
428 
429  if (tval > rval) {
430  // Overflow
431  return -2;
432  }
433 
434  rval += d;
435 
436  if (tval > rval) {
437  // Overflow
438  return -2;
439  }
440 
441  tval = rval;
442  } else {
443  tval = rval = d;
444  tflag = 1;
445  }
446  }
447 
448  *lastlen = i + 1;
449 
450  return rval;
451 }

507 void bstr_util_mem_trim(unsigned char **data, size_t *len) {
508  if ((data == NULL)||(len == NULL)) return;
509 
510  unsigned char *d = *data;
511  size_t l = *len;
512 
513  // Ignore whitespace at the beginning.
514  size_t pos = 0;
515  while ((pos < l) && isspace(d[pos])) pos++;
516  d += pos;
517  l -= pos;
518 
519  // Ignore whitespace at the end.
520  while ((l > 0)&&(isspace(d[l - 1]))) l--;
521 
522  *data = d;
523  *len = l;
524 }

526 char *bstr_util_memdup_to_c(const void *_data, size_t len) {
527  const unsigned char *data = (unsigned char *) _data;
528 
529  // Count how many NUL bytes we have in the string.
530  size_t i, nulls = 0;
531  for (i = 0; i < len; i++) {
532  if (data[i] == '\0') {
533  nulls++;
534  }
535  }
536 
537  // Now copy the string into a NUL-terminated buffer.
538 
539  char *r, *d;
540  r = d = malloc(len + nulls + 1);
541  if (d == NULL) return NULL;
542 
543  while (len--) {
544  if (*data == '\0') {
545  data++;
546  *d++ = '\\';
547  *d++ = '0';
548  } else {
549  *d++ = *data++;
550  }
551  }
552 
553  *d = '\0';
554 
555  return r;
556 }
