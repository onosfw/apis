83 /**
84  * \brief Create a command unix socket on system
85  *
86  * \retval 0 in case of error, 1 in case of success
87  */
88 int UnixNew(UnixCommand * this)
89 {
90  struct sockaddr_un addr;
91  int len;
92  int ret;
93  int on = 1;
94  char *sockettarget = NULL;
95  char *socketname;
96 
97  this->start_timestamp = time(NULL);
98  this->socket = -1;
99  this->select_max = 0;
100 
101  TAILQ_INIT(&this->commands);
102  TAILQ_INIT(&this->tasks);
103  TAILQ_INIT(&this->clients);
104 
105  if (ConfGet("unix-command.filename", &socketname) == 1) {
106  if (PathIsAbsolute(socketname)) {
107  sockettarget = SCStrdup(socketname);
108  if (unlikely(sockettarget == NULL)) {
109  SCLogError(SC_ERR_MEM_ALLOC, "Unable to allocate socket name");
110  return 0;
111  }
112  } else {
113  int socketlen = strlen(SOCKET_PATH) + strlen(socketname) + 2;
114  sockettarget = SCMalloc(socketlen);
115  if (unlikely(sockettarget == NULL)) {
116  SCLogError(SC_ERR_MEM_ALLOC, "Unable to allocate socket name");
117  return 0;
118  }
119  snprintf(sockettarget, socketlen, "%s/%s", SOCKET_PATH, socketname);
120 
121  /* Create socket dir */
122  ret = mkdir(SOCKET_PATH, S_IRWXU|S_IXGRP|S_IRGRP);
123  if ( ret != 0 ) {
124  int err = errno;
125  if (err != EEXIST) {
126  SCFree(sockettarget);
127  SCLogError(SC_ERR_OPENING_FILE,
128  "Cannot create socket directory %s: %s", SOCKET_PATH, strerror(err));
129  return 0;
130  }
131  }
132 
133  }
134  SCLogInfo("Using unix socket file '%s'", sockettarget);
135  }
136  if (sockettarget == NULL) {
137  sockettarget = SCStrdup(SOCKET_TARGET);
138  if (unlikely(sockettarget == NULL)) {
139  SCLogError(SC_ERR_MEM_ALLOC, "Unable to allocate socket name");
140  return 0;
141  }
142  }
143 
144  /* Remove socket file */
145  (void) unlink(sockettarget);
146 
147  /* set address */
148  addr.sun_family = AF_UNIX;
149  strlcpy(addr.sun_path, sockettarget, sizeof(addr.sun_path));
150  addr.sun_path[sizeof(addr.sun_path) - 1] = 0;
151  len = strlen(addr.sun_path) + sizeof(addr.sun_family);
152 
153  /* create socket */
154  this->socket = socket(AF_UNIX, SOCK_STREAM, 0);
155  if (this->socket == -1) {
156  SCLogWarning(SC_ERR_OPENING_FILE,
157  "Unix Socket: unable to create UNIX socket %s: %s",
158  addr.sun_path, strerror(errno));
159  SCFree(sockettarget);
160  return 0;
161  }
162  this->select_max = this->socket + 1;
163 
164  /* Set file mode: will not fully work on most system, the group
165  * permission is not changed on some Linux and *BSD won't do the
166  * chmod. */
167  ret = fchmod(this->socket, S_IRUSR|S_IWUSR|S_IRGRP|S_IWGRP);
168  if (ret == -1) {
169  int err = errno;
170  SCLogWarning(SC_ERR_INITIALIZATION,
171  "Unable to change permission on socket: %s (%d)",
172  strerror(err),
173  err);
174  }
175  /* set reuse option */
176  ret = setsockopt(this->socket, SOL_SOCKET, SO_REUSEADDR,
177  (char *) &on, sizeof(on));
178  if ( ret != 0 ) {
179  SCLogWarning(SC_ERR_INITIALIZATION,
180  "Cannot set sockets options: %s.", strerror(errno));
181  }
182 
183  /* bind socket */
184  ret = bind(this->socket, (struct sockaddr *) &addr, len);
185  if (ret == -1) {
186  SCLogWarning(SC_ERR_INITIALIZATION,
187  "Unix socket: UNIX socket bind(%s) error: %s",
188  sockettarget, strerror(errno));
189  SCFree(sockettarget);
190  return 0;
191  }
192 
193  /* listen */
194  if (listen(this->socket, 1) == -1) {
195  SCLogWarning(SC_ERR_INITIALIZATION,
196  "Command server: UNIX socket listen() error: %s",
197  strerror(errno));
198  SCFree(sockettarget);
199  return 0;
200  }
201  SCFree(sockettarget);
202  return 1;
203 }

267 /**
268  * \brief Accept a new client on unix socket
269  *
270  * The function is called when a new user is detected
271  * in UnixMain(). It does the initial protocol negotiation
272  * with client.
273  *
274  * \retval 0 in case of error, 1 in case of success
275  */
276 int UnixCommandAccept(UnixCommand *this)
277 {
278  char buffer[UNIX_PROTO_VERSION_LENGTH + 1];
279  json_t *client_msg;
280  json_t *server_msg;
281  json_t *version;
282  json_error_t jerror;
283  int client;
284  int ret;
285  UnixClient *uclient = NULL;
286 
287  /* accept client socket */
288  socklen_t len = sizeof(this->client_addr);
289  client = accept(this->socket, (struct sockaddr *) &this->client_addr,
290  &len);
291  if (client < 0) {
292  SCLogInfo("Unix socket: accept() error: %s",
293  strerror(errno));
294  return 0;
295  }
296  SCLogDebug("Unix socket: client connection");
297 
298  /* read client version */
299  buffer[sizeof(buffer)-1] = 0;
300  ret = recv(client, buffer, sizeof(buffer)-1, 0);
301  if (ret < 0) {
302  SCLogInfo("Command server: client doesn't send version");
303  close(client);
304  return 0;
305  }
306  if (ret >= (int)(sizeof(buffer)-1)) {
307  SCLogInfo("Command server: client message is too long, "
308  "disconnect him.");
309  close(client);
310  return 0;
311  }
312  buffer[ret] = 0;
313 
314  client_msg = json_loads(buffer, 0, &jerror);
315  if (client_msg == NULL) {
316  SCLogInfo("Invalid command, error on line %d: %s\n", jerror.line, jerror.text);
317  close(client);
318  return 0;
319  }
320 
321  version = json_object_get(client_msg, "version");
322  if (!json_is_string(version)) {
323  SCLogInfo("error: version is not a string");
324  close(client);
325  json_decref(client_msg);
326  return 0;
327  }
328 
329  /* check client version */
330  if (strcmp(json_string_value(version), UNIX_PROTO_VERSION) != 0) {
331  SCLogInfo("Unix socket: invalid client version: \"%s\"",
332  json_string_value(version));
333  json_decref(client_msg);
334  close(client);
335  return 0;
336  } else {
337  SCLogInfo("Unix socket: client version: \"%s\"",
338  json_string_value(version));
339  }
340 
341  json_decref(client_msg);
342  /* send answer */
343  server_msg = json_object();
344  if (server_msg == NULL) {
345  close(client);
346  return 0;
347  }
348  json_object_set_new(server_msg, "return", json_string("OK"));
349 
350  if (json_dump_callback(server_msg, UnixCommandSendCallback, &client, 0) == -1) {
351  SCLogWarning(SC_ERR_SOCKET, "Unable to send command");
352  json_decref(server_msg);
353  close(client);
354  return 0;
355  }
356  json_decref(server_msg);
357 
358  /* client connected */
359  SCLogDebug("Unix socket: client connected");
360 
361  uclient = SCMalloc(sizeof(UnixClient));
362  if (unlikely(uclient == NULL)) {
363  SCLogError(SC_ERR_MEM_ALLOC, "Can't allocate new cient");
364  return 0;
365  }
366  uclient->fd = client;
367  TAILQ_INSERT_TAIL(&this->clients, uclient, next);
368  UnixCommandSetMaxFD(this);
369  return 1;
370 }

386 /**
387  * \brief Command dispatcher
388  *
389  * \param this a UnixCommand:: structure
390  * \param command a string containing a json formatted
391  * command
392  *
393  * \retval 0 in case of error, 1 in case of success
394  */
395 int UnixCommandExecute(UnixCommand * this, char *command, UnixClient *client)
396 {
397  int ret = 1;
398  json_error_t error;
399  json_t *jsoncmd = NULL;
400  json_t *cmd = NULL;
401  json_t *server_msg = json_object();
402  const char * value;
403  int found = 0;
404  Command *lcmd;
405 
406  if (server_msg == NULL) {
407  return 0;
408  }
409 
410  jsoncmd = json_loads(command, 0, &error);
411  if (jsoncmd == NULL) {
412  SCLogInfo("Invalid command, error on line %d: %s\n", error.line, error.text);
413  goto error;
414  }
415 
416  cmd = json_object_get(jsoncmd, "command");
417  if(!json_is_string(cmd)) {
418  SCLogInfo("error: command is not a string");
419  goto error_cmd;
420  }
421  value = json_string_value(cmd);
422 
423  TAILQ_FOREACH(lcmd, &this->commands, next) {
424  if (!strcmp(value, lcmd->name)) {
425  int fret = TM_ECODE_OK;
426  found = 1;
427  if (lcmd->flags & UNIX_CMD_TAKE_ARGS) {
428  cmd = json_object_get(jsoncmd, "arguments");
429  if(!json_is_object(cmd)) {
430  SCLogInfo("error: argument is not an object");
431  goto error_cmd;
432  }
433  }
434  fret = lcmd->Func(cmd, server_msg, lcmd->data);
435  if (fret != TM_ECODE_OK) {
436  ret = 0;
437  }
438  break;
439  }
440  }
441 
442  if (found == 0) {
443  json_object_set_new(server_msg, "message", json_string("Unknown command"));
444  ret = 0;
445  }
446 
447  switch (ret) {
448  case 0:
449  json_object_set_new(server_msg, "return", json_string("NOK"));
450  break;
451  case 1:
452  json_object_set_new(server_msg, "return", json_string("OK"));
453  break;
454  }
455 
456  /* send answer */
457  if (json_dump_callback(server_msg, UnixCommandSendCallback, &client->fd, 0) == -1) {
458  SCLogWarning(SC_ERR_SOCKET, "Unable to send command");
459  goto error_cmd;
460  }
461 
462  json_decref(jsoncmd);
463  json_decref(server_msg);
464  return ret;
465 
466 error_cmd:
467  json_decref(jsoncmd);
468 error:
469  json_decref(server_msg);
470  UnixCommandClose(this, client->fd);
471  return 0;
472 }

474 void UnixCommandRun(UnixCommand * this, UnixClient *client)
475 {
476  char buffer[4096];
477  int ret;
478  ret = recv(client->fd, buffer, sizeof(buffer) - 1, 0);
479  if (ret <= 0) {
480  if (ret == 0) {
481  SCLogDebug("Unix socket: lost connection with client");
482  } else {
483  SCLogError(SC_ERR_SOCKET, "Unix socket: error on recv() from client: %s",
484  strerror(errno));
485  }
486  UnixCommandClose(this, client->fd);
487  return;
488  }
489  if (ret >= (int)(sizeof(buffer)-1)) {
490  SCLogInfo("Command server: client command is too long, "
491  "disconnect him.");
492  UnixCommandClose(this, client->fd);
493  }
494  buffer[ret] = 0;
495  UnixCommandExecute(this, buffer, client);
496 }

498 /**
499  * \brief Select function
500  *
501  * \retval 0 in case of error, 1 in case of success
502  */
503 int UnixMain(UnixCommand * this)
504 {
505  struct timeval tv;
506  int ret;
507  fd_set select_set;
508  UnixClient *uclient;
509  UnixClient *tclient;
510 
511  /* Wait activity on the socket */
512  FD_ZERO(&select_set);
513  FD_SET(this->socket, &select_set);
514  TAILQ_FOREACH(uclient, &this->clients, next) {
515  FD_SET(uclient->fd, &select_set);
516  }
517 
518  tv.tv_sec = 0;
519  tv.tv_usec = 200 * 1000;
520  ret = select(this->select_max, &select_set, NULL, NULL, &tv);
521 
522  /* catch select() error */
523  if (ret == -1) {
524  /* Signal was caught: just ignore it */
525  if (errno == EINTR) {
526  return 1;
527  }
528  SCLogError(SC_ERR_SOCKET, "Command server: select() fatal error: %s", strerror(errno));
529  return 0;
530  }
531 
532  if (suricata_ctl_flags & (SURICATA_STOP | SURICATA_KILL)) {
533  return 1;
534  }
535 
536  /* timeout: continue */
537  if (ret == 0) {
538  return 1;
539  }
540 
541  TAILQ_FOREACH_SAFE(uclient, &this->clients, next, tclient) {
542  if (FD_ISSET(uclient->fd, &select_set)) {
543  UnixCommandRun(this, uclient);
544  }
545  }
546  if (FD_ISSET(this->socket, &select_set)) {
547  if (!UnixCommandAccept(this))
548  return 1;
549  }
550 
551  return 1;
552 }

749 /**
750  * \brief Add a command to the list of commands
751  *
752  * This function adds a command to the list of commands available
753  * through the unix socket.
754  *
755  * When a command is received from user through the unix socket, the content
756  * of 'Command' field in the JSON message is match against keyword, then the
757  * Func is called. See UnixSocketAddPcapFile() for an example.
758  *
759  * \param keyword name of the command
760  * \param Func function to run when command is received
761  * \param data a pointer to data that are passed to Func when it is run
762  * \param flags a flag now used to tune the command type
763  * \retval TM_ECODE_OK in case of success, TM_ECODE_FAILED in case of failure
764  */
765 TmEcode UnixManagerRegisterCommand(const char * keyword,
766  TmEcode (*Func)(json_t *, json_t *, void *),
767  void *data, int flags)
768 {
769  SCEnter();
770  Command *cmd = NULL;
771  Command *lcmd = NULL;
772 
773  if (Func == NULL) {
774  SCLogError(SC_ERR_INVALID_ARGUMENT, "Null function");
775  SCReturnInt(TM_ECODE_FAILED);
776  }
777 
778  if (keyword == NULL) {
779  SCLogError(SC_ERR_INVALID_ARGUMENT, "Null keyword");
780  SCReturnInt(TM_ECODE_FAILED);
781  }
782 
783  TAILQ_FOREACH(lcmd, &command.commands, next) {
784  if (!strcmp(keyword, lcmd->name)) {
785  SCLogError(SC_ERR_INVALID_ARGUMENT, "%s already registered", keyword);
786  SCReturnInt(TM_ECODE_FAILED);
787  }
788  }
789 
790  cmd = SCMalloc(sizeof(Command));
791  if (unlikely(cmd == NULL)) {
792  SCLogError(SC_ERR_MEM_ALLOC, "Can't alloc cmd");
793  SCReturnInt(TM_ECODE_FAILED);
794  }
795  cmd->name = SCStrdup(keyword);
796  if (unlikely(cmd->name == NULL)) {
797  SCLogError(SC_ERR_MEM_ALLOC, "Can't alloc cmd name");
798  SCFree(cmd);
799  SCReturnInt(TM_ECODE_FAILED);
800  }
801  cmd->Func = Func;
802  cmd->data = data;
803  cmd->flags = flags;
804  /* Add it to the list */
805  TAILQ_INSERT_TAIL(&command.commands, cmd, next);
806 
807  SCReturnInt(TM_ECODE_OK);
808 }

864 static TmEcode UnixManager(ThreadVars *th_v, void *thread_data)
865 {
866  int ret;
867 
868  /* set the thread name */
869  SCLogDebug("%s started...", th_v->name);
870 
871  StatsSetupPrivate(th_v);
872 
873  if (UnixNew(&command) == 0) {
874  int failure_fatal = 0;
875  SCLogError(SC_ERR_INITIALIZATION,
876  "Unable to create unix command socket");

878  SCLogDebug("ConfGetBool could not load the value.");
879  }
880  if (failure_fatal) {
881  exit(EXIT_FAILURE);
882  } else {
883  return TM_ECODE_FAILED;
884  }

888  th_v->cap_flags = 0;
889  SCDropCaps(th_v);

892  UnixManagerRegisterCommand("shutdown", UnixManagerShutdownCommand, NULL, 0);
893  UnixManagerRegisterCommand("command-list", UnixManagerListCommand, &command, 0);
894  UnixManagerRegisterCommand("help", UnixManagerListCommand, &command, 0);
895  UnixManagerRegisterCommand("version", UnixManagerVersionCommand, &command, 0);
896  UnixManagerRegisterCommand("uptime", UnixManagerUptimeCommand, &command, 0);
897  UnixManagerRegisterCommand("running-mode", UnixManagerRunningModeCommand, &command, 0);
898  UnixManagerRegisterCommand("capture-mode", UnixManagerCaptureModeCommand, &command, 0);
899  UnixManagerRegisterCommand("conf-get", UnixManagerConfGetCommand, &command, UNIX_CMD_TAKE_ARGS);
900  UnixManagerRegisterCommand("dump-counters", StatsOutputCounterSocket, NULL, 0);
901  UnixManagerRegisterCommand("reload-rules", UnixManagerReloadRules, NULL, 0);
902  UnixManagerRegisterCommand("register-tenant-handler", UnixSocketRegisterTenantHandler, &command, UNIX_CMD_TAKE_ARGS);
903  UnixManagerRegisterCommand("unregister-tenant-handler", UnixSocketUnregisterTenantHandler, &command, UNIX_CMD_TAKE_ARGS);
904  UnixManagerRegisterCommand("register-tenant", UnixSocketRegisterTenant, &command, UNIX_CMD_TAKE_ARGS);
905  UnixManagerRegisterCommand("reload-tenant", UnixSocketReloadTenant, &command, UNIX_CMD_TAKE_ARGS);
906  UnixManagerRegisterCommand("unregister-tenant", UnixSocketUnregisterTenant, &command, UNIX_CMD_TAKE_ARGS);
907 
908 
909  TmThreadsSetFlag(th_v, THV_INIT_DONE);
910  while (1) {
911  ret = UnixMain(&command);
912  if (ret == 0) {
913  SCLogError(SC_ERR_FATAL, "Fatal error on unix socket");
914  }

916  if ((ret == 0) || (TmThreadsCheckFlag(th_v, THV_KILL))) {
917  UnixClient *item;
918  UnixClient *titem;

920  close(item->fd);
921  SCFree(item);
922  }
923  StatsSyncCounters(th_v);
924  break;
925  }
926 
927  UnixCommandBackgroundTasks(&command);
928  }
929  return TM_ECODE_OK;
930 }
