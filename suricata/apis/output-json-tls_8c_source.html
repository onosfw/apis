87 void JsonTlsLogJSONExtended(json_t *tjs, SSLState * state)
88 {
89  char ssl_version[SSL_VERSION_LENGTH + 1];
90 
91  /* tls.fingerprint */
92  json_object_set_new(tjs, "fingerprint",
93  json_string(state->server_connp.cert0_fingerprint));
94 
95  /* tls.version */
96  switch (state->server_connp.version) {
97  case TLS_VERSION_UNKNOWN:
98  snprintf(ssl_version, SSL_VERSION_LENGTH, "UNDETERMINED");
99  break;
100  case SSL_VERSION_2:
101  snprintf(ssl_version, SSL_VERSION_LENGTH, "SSLv2");
102  break;
103  case SSL_VERSION_3:
104  snprintf(ssl_version, SSL_VERSION_LENGTH, "SSLv3");
105  break;
106  case TLS_VERSION_10:
107  snprintf(ssl_version, SSL_VERSION_LENGTH, "TLSv1");
108  break;
109  case TLS_VERSION_11:
110  snprintf(ssl_version, SSL_VERSION_LENGTH, "TLS 1.1");
111  break;
112  case TLS_VERSION_12:
113  snprintf(ssl_version, SSL_VERSION_LENGTH, "TLS 1.2");
114  break;
115  default:
116  snprintf(ssl_version, SSL_VERSION_LENGTH, "0x%04x",
117  state->server_connp.version);
118  break;
119  }
120  json_object_set_new(tjs, "version", json_string(ssl_version));
121 }

123 static int JsonTlsLogger(ThreadVars *tv, void *thread_data, const Packet *p)
124 {
125  JsonTlsLogThread *aft = (JsonTlsLogThread *)thread_data;
126  MemBuffer *buffer = (MemBuffer *)aft->buffer;
127  OutputTlsCtx *tls_ctx = aft->tlslog_ctx;
128 
129  if (unlikely(p->flow == NULL)) {
130  return 0;
131  }
132 
133  /* check if we have TLS state or not */
134  FLOWLOCK_WRLOCK(p->flow);
135  uint16_t proto = FlowGetAppProtocol(p->flow);
136  if (proto != ALPROTO_TLS)
137  goto end;
138 
139  SSLState *ssl_state = (SSLState *)FlowGetAppState(p->flow);
140  if (unlikely(ssl_state == NULL)) {
141  goto end;
142  }
143 
144  if (ssl_state->server_connp.cert0_issuerdn == NULL || ssl_state->server_connp.cert0_subject == NULL)
145  goto end;
146 
147  json_t *js = CreateJSONHeader((Packet *)p, 0, "tls");//TODO
148  if (unlikely(js == NULL))
149  goto end;
150 
151  json_t *tjs = json_object();
152  if (tjs == NULL) {
153  free(js);
154  goto end;
155  }
156 
157  /* reset */
158  MemBufferReset(buffer);
159 
160  JsonTlsLogJSONBasic(tjs, ssl_state);
161 
162  if (tls_ctx->flags & LOG_TLS_EXTENDED) {
163  JsonTlsLogJSONExtended(tjs, ssl_state);
164  }
165 
166  json_object_set_new(js, "tls", tjs);
167 
168  OutputJSONBuffer(js, tls_ctx->file_ctx, buffer);
169  json_object_clear(js);
170  json_decref(js);
171 
172  /* we only log the state once */
173  ssl_state->flags |= SSL_AL_FLAG_STATE_LOGGED;
174 end:
175  FLOWLOCK_UNLOCK(p->flow);
176  return 0;
177 }

180 static TmEcode JsonTlsLogThreadInit(ThreadVars *t, void *initdata, void **data)
181 {
182  JsonTlsLogThread *aft = SCMalloc(sizeof(JsonTlsLogThread));
183  if (unlikely(aft == NULL))
184  return TM_ECODE_FAILED;
185  memset(aft, 0, sizeof(JsonTlsLogThread));
186 
187  if(initdata == NULL)
188  {
189  SCLogDebug("Error getting context for HTTPLog. \"initdata\" argument NULL");
190  SCFree(aft);
191  return TM_ECODE_FAILED;
192  }
193 
194  /* Use the Ouptut Context (file pointer and mutex) */
195  aft->tlslog_ctx = ((OutputCtx *)initdata)->data;
196 
197  aft->buffer = MemBufferCreateNew(OUTPUT_BUFFER_SIZE);
198  if (aft->buffer == NULL) {
199  SCFree(aft);
200  return TM_ECODE_FAILED;
201  }
202 
203  *data = (void *)aft;
204  return TM_ECODE_OK;
205 }

234 OutputCtx *OutputTlsLogInit(ConfNode *conf)
235 {
236  if (OutputTlsLoggerEnable() != 0) {
237  SCLogError(SC_ERR_CONF_YAML_ERROR, "only one 'tls' logger "
238  "can be enabled");
239  return NULL;
240  }
241 
242  LogFileCtx *file_ctx = LogFileNewCtx();
243  if(file_ctx == NULL) {
244  SCLogError(SC_ERR_HTTP_LOG_GENERIC, "couldn't create new file_ctx");
245  return NULL;
246  }
247 
248  if (SCConfLogOpenGeneric(conf, file_ctx, DEFAULT_LOG_FILENAME, 1) < 0) {
249  LogFileFreeCtx(file_ctx);
250  return NULL;
251  }
252 
253  OutputTlsCtx *tls_ctx = SCMalloc(sizeof(OutputTlsCtx));
254  if (unlikely(tls_ctx == NULL)) {
255  LogFileFreeCtx(file_ctx);
256  return NULL;
257  }
258 
259  OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
260  if (unlikely(output_ctx == NULL)) {
261  LogFileFreeCtx(file_ctx);
262  SCFree(tls_ctx);
263  return NULL;
264  }
265 
266  tls_ctx->file_ctx = file_ctx;
267  tls_ctx->flags = LOG_TLS_DEFAULT;
268 
269  if (conf) {
270  const char *extended = ConfNodeLookupChildValue(conf, "extended");
271 
272  if (extended != NULL) {
273  if (ConfValIsTrue(extended)) {
274  tls_ctx->flags = LOG_TLS_EXTENDED;
275  }
276  }
277  }
278  output_ctx->data = tls_ctx;
279  output_ctx->DeInit = OutputTlsLogDeinit;
280 
281  return output_ctx;
282 }

331 /** \internal
332  * \brief Condition function for TLS logger
333  * \retval bool true or false -- log now?
334  */
335 static int JsonTlsCondition(ThreadVars *tv, const Packet *p)
336 {
337  if (p->flow == NULL) {
338  return FALSE;
339  }
340 
341  if (!(PKT_IS_TCP(p))) {
342  return FALSE;
343  }
344 
345  FLOWLOCK_RDLOCK(p->flow);
346  uint16_t proto = FlowGetAppProtocol(p->flow);
347  if (proto != ALPROTO_TLS)
348  goto dontlog;
349 
350  SSLState *ssl_state = (SSLState *)FlowGetAppState(p->flow);
351  if (ssl_state == NULL) {
352  SCLogDebug("no tls state, so no request logging");
353  goto dontlog;
354  }
355 
356  /* we only log the state once */
357  if (ssl_state->flags & SSL_AL_FLAG_STATE_LOGGED)
358  goto dontlog;
359 
360  if (ssl_state->server_connp.cert0_issuerdn == NULL ||
361  ssl_state->server_connp.cert0_subject == NULL)
362  goto dontlog;
363 
364  /* todo: logic to log once */
365 
366  FLOWLOCK_UNLOCK(p->flow);
367  return TRUE;
368 dontlog:
369  FLOWLOCK_UNLOCK(p->flow);
370  return FALSE;
371 }
