44 /** \internal
45  * \brief See if we can really discard this host. Check use_cnt reference.
46  *
47  * \param h host
48  * \param ts timestamp
49  *
50  * \retval 0 not timed out just yet
51  * \retval 1 fully timed out, lets kill it
52  */
53 static int HostHostTimedOut(Host *h, struct timeval *ts)
54 {
55  int tags = 0;
56  int thresholds = 0;
57  int vars = 0;
58 
59  /** never prune a host that is used by a packet
60  * we are currently processing in one of the threads */
61  if (SC_ATOMIC_GET(h->use_cnt) > 0) {
62  return 0;
63  }
64 
65  if (h->iprep) {
66  if (SRepHostTimedOut(h) == 0)
67  return 0;
68 
69  SCLogDebug("host %p reputation timed out", h);
70  }
71 
72  if (TagHostHasTag(h) && TagTimeoutCheck(h, ts) == 0) {
73  tags = 1;
74  }
75  if (ThresholdHostHasThreshold(h) && ThresholdTimeoutCheck(h, ts) == 0) {
76  thresholds = 1;
77  }
78  if (HostHasHostBits(h) && HostBitsTimedoutCheck(h, ts) == 0) {
79  vars = 1;
80  }
81 
82  if (tags || thresholds || vars)
83  return 0;
84 
85  SCLogDebug("host %p timed out", h);
86  return 1;
87 }

89 /**
90  * \internal
91  *
92  * \brief check all hosts in a hash row for timing out
93  *
94  * \param hb host hash row *LOCKED*
95  * \param h last host in the hash row
96  * \param ts timestamp
97  *
98  * \retval cnt timed out hosts
99  */
100 static uint32_t HostHashRowTimeout(HostHashRow *hb, Host *h, struct timeval *ts)
101 {
102  uint32_t cnt = 0;
103 
104  do {
105  if (SCMutexTrylock(&h->m) != 0) {
106  h = h->hprev;
107  continue;
108  }
109 
110  Host *next_host = h->hprev;
111 
112  /* check if the host is fully timed out and
113  * ready to be discarded. */
114  if (HostHostTimedOut(h, ts) == 1) {
115  /* remove from the hash */
116  if (h->hprev != NULL)
117  h->hprev->hnext = h->hnext;
118  if (h->hnext != NULL)
119  h->hnext->hprev = h->hprev;
120  if (hb->head == h)
121  hb->head = h->hnext;
122  if (hb->tail == h)
123  hb->tail = h->hprev;
124 
125  h->hnext = NULL;
126  h->hprev = NULL;
127 
128  HostClearMemory (h);
129 
130  /* no one is referring to this host, use_cnt 0, removed from hash
131  * so we can unlock it and move it back to the spare queue. */
132  SCMutexUnlock(&h->m);
133 
134  /* move to spare list */
135  HostMoveToSpare(h);
136 
137  cnt++;
138  } else {
139  SCMutexUnlock(&h->m);
140  }
141 
142  h = next_host;
143  } while (h != NULL);
144 
145  return cnt;
146 }

148 /**
149  * \brief time out hosts from the hash
150  *
151  * \param ts timestamp
152  *
153  * \retval cnt number of timed out host
154  */
155 uint32_t HostTimeoutHash(struct timeval *ts)
156 {
157  uint32_t idx = 0;
158  uint32_t cnt = 0;
159 
160  for (idx = 0; idx < host_config.hash_size; idx++) {
161  HostHashRow *hb = &host_hash[idx];
162 
163  if (HRLOCK_TRYLOCK(hb) != 0)
164  continue;
165 
166  /* host hash bucket is now locked */
167 
168  if (hb->tail == NULL) {
169  HRLOCK_UNLOCK(hb);
170  continue;
171  }
172 
173  /* we have a host, or more than one */
174  cnt += HostHashRowTimeout(hb, hb->tail, ts);
175  HRLOCK_UNLOCK(hb);
176  }
177 
178  return cnt;
179 }
