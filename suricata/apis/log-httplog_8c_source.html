130 /* Retrieves the selected cookie value */
131 static uint32_t GetCookieValue(uint8_t *rawcookies, uint32_t rawcookies_len, char *cookiename,
132  uint8_t **cookievalue)
133 {
134  uint8_t *p = rawcookies;
135  uint8_t *cn = p; /* ptr to cookie name start */
136  uint8_t *cv = NULL; /* ptr to cookie value start */
137  while (p < rawcookies + rawcookies_len) {
138  if (cv == NULL && *p == '=') {
139  cv = p + 1;
140  } else if (cv != NULL && (*p == ';' || p == rawcookies + rawcookies_len - 1) ) {
141  /* Found end of cookie */
142  p++;
143  if (strlen(cookiename) == (unsigned int) (cv-cn-1) &&
144  strncmp(cookiename, (char *) cn, cv-cn-1) == 0) {
145  *cookievalue = cv;
146  return (uint32_t) (p-cv);
147  }
148  cv = NULL;
149  cn = p + 1;
150  }
151  p++;
152  }
153  return 0;
154 }

156 /* Custom format logging */
157 static void LogHttpLogCustom(LogHttpLogThread *aft, htp_tx_t *tx, const struct timeval *ts,
158  char *srcip, Port sp, char *dstip, Port dp)
159 {
160  LogHttpFileCtx *httplog_ctx = aft->httplog_ctx;
161  uint32_t i;
162  uint32_t datalen;
163  char buf[128];
164 
165  uint8_t *cvalue = NULL;
166  uint32_t cvalue_len = 0;
167 
168  htp_header_t *h_request_hdr;
169  htp_header_t *h_response_hdr;
170 
171  time_t time = ts->tv_sec;
172  struct tm local_tm;
173  struct tm *timestamp = SCLocalTime(time, &local_tm);
174 
175  for (i = 0; i < httplog_ctx->cf_n; i++) {
176  h_request_hdr = NULL;
177  h_response_hdr = NULL;
178  switch (httplog_ctx->cf_nodes[i]->type){
179  case LOG_HTTP_CF_LITERAL:
180  /* LITERAL */
181  MemBufferWriteString(aft->buffer, "%s", httplog_ctx->cf_nodes[i]->data);
182  break;
183  case LOG_HTTP_CF_TIMESTAMP:
184  /* TIMESTAMP */
185  if (httplog_ctx->cf_nodes[i]->data[0] == '\0') {
186  strftime(buf, 62, TIMESTAMP_DEFAULT_FORMAT, timestamp);
187  } else {
188  strftime(buf, 62, httplog_ctx->cf_nodes[i]->data, timestamp);
189  }
190  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset,
191  aft->buffer->size, (uint8_t *)buf,strlen(buf));
192  break;
193  case LOG_HTTP_CF_TIMESTAMP_U:
194  /* TIMESTAMP USECONDS */
195  snprintf(buf, 62, "%06u", (unsigned int) ts->tv_usec);
196  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset,
197  aft->buffer->size, (uint8_t *)buf,strlen(buf));
198  break;
199  case LOG_HTTP_CF_CLIENT_IP:
200  /* CLIENT IP ADDRESS */
201  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset,
202  aft->buffer->size, (uint8_t *)srcip,strlen(srcip));
203  break;
204  case LOG_HTTP_CF_SERVER_IP:
205  /* SERVER IP ADDRESS */
206  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset,
207  aft->buffer->size, (uint8_t *)dstip,strlen(dstip));
208  break;
209  case LOG_HTTP_CF_CLIENT_PORT:
210  /* CLIENT PORT */
211  MemBufferWriteString(aft->buffer, "%" PRIu16 "", sp);
212  break;
213  case LOG_HTTP_CF_SERVER_PORT:
214  /* SERVER PORT */
215  MemBufferWriteString(aft->buffer, "%" PRIu16 "", dp);
216  break;
217  case LOG_HTTP_CF_REQUEST_METHOD:
218  /* METHOD */
219  if (tx->request_method != NULL) {
220  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset,
221  aft->buffer->size, (uint8_t *)bstr_ptr(tx->request_method),
222  bstr_len(tx->request_method));
223  } else {
224  MemBufferWriteString(aft->buffer, LOG_HTTP_CF_NONE);
225  }
226  break;
227  case LOG_HTTP_CF_REQUEST_URI:
228  /* URI */
229  if (tx->request_uri != NULL) {
230  datalen = httplog_ctx->cf_nodes[i]->maxlen;
231  if (datalen == 0 || datalen > bstr_len(tx->request_uri)) {
232  datalen = bstr_len(tx->request_uri);
233  }
234  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset,
235  aft->buffer->size, (uint8_t *)bstr_ptr(tx->request_uri),
236  datalen);
237  } else {
238  MemBufferWriteString(aft->buffer, LOG_HTTP_CF_NONE);
239  }
240  break;
241  case LOG_HTTP_CF_REQUEST_HOST:
242  /* HOSTNAME */
243  if (tx->request_hostname != NULL)
244  {
245  datalen = httplog_ctx->cf_nodes[i]->maxlen;
246  if (datalen == 0 || datalen > bstr_len(tx->request_hostname)) {
247  datalen = bstr_len(tx->request_hostname);
248  }
249  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset,
250  aft->buffer->size, (uint8_t *)bstr_ptr(tx->request_hostname),
251  datalen);
252  } else {
253  MemBufferWriteString(aft->buffer, LOG_HTTP_CF_NONE);
254  }
255  break;
256  case LOG_HTTP_CF_REQUEST_PROTOCOL:
257  /* PROTOCOL */
258  if (tx->request_protocol != NULL) {
259  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset,
260  aft->buffer->size, (uint8_t *)bstr_ptr(tx->request_protocol),
261  bstr_len(tx->request_protocol));
262  } else {
263  MemBufferWriteString(aft->buffer, LOG_HTTP_CF_NONE);
264  }
265  break;
266  case LOG_HTTP_CF_REQUEST_HEADER:
267  /* REQUEST HEADER */
268  if (tx->request_headers != NULL) {
269  h_request_hdr = htp_table_get_c(tx->request_headers, httplog_ctx->cf_nodes[i]->data);
270  }
271  if (h_request_hdr != NULL) {
272  datalen = httplog_ctx->cf_nodes[i]->maxlen;
273  if (datalen == 0 || datalen > bstr_len(h_request_hdr->value)) {
274  datalen = bstr_len(h_request_hdr->value);
275  }
276  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset,
277  aft->buffer->size, (uint8_t *)bstr_ptr(h_request_hdr->value),
278  datalen);
279  } else {
280  MemBufferWriteString(aft->buffer, LOG_HTTP_CF_NONE);
281  }
282  break;
283  case LOG_HTTP_CF_REQUEST_COOKIE:
284  /* REQUEST COOKIE */
285  if (tx->request_headers != NULL) {
286  h_request_hdr = htp_table_get_c(tx->request_headers, "Cookie");
287  if (h_request_hdr != NULL) {
288  cvalue_len = GetCookieValue((uint8_t *) bstr_ptr(h_request_hdr->value),
289  bstr_len(h_request_hdr->value), (char *) httplog_ctx->cf_nodes[i]->data,
290  &cvalue);
291  }
292  }
293  if (cvalue_len > 0 && cvalue != NULL) {
294  datalen = httplog_ctx->cf_nodes[i]->maxlen;
295  if (datalen == 0 || datalen > cvalue_len) {
296  datalen = cvalue_len;
297  }
298  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset,
299  aft->buffer->size, cvalue, datalen);
300  } else {
301  MemBufferWriteString(aft->buffer, LOG_HTTP_CF_NONE);
302  }
303  break;
304  case LOG_HTTP_CF_REQUEST_LEN:
305  /* REQUEST LEN */
306  MemBufferWriteString(aft->buffer, "%"PRIuMAX"", (uintmax_t)tx->request_message_len);
307  break;
308  case LOG_HTTP_CF_RESPONSE_STATUS:
309  /* RESPONSE STATUS */
310  if (tx->response_status != NULL) {
311  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset,
312  aft->buffer->size, (uint8_t *)bstr_ptr(tx->response_status),
313  bstr_len(tx->response_status));
314  } else {
315  MemBufferWriteString(aft->buffer, LOG_HTTP_CF_NONE);
316  }
317  break;
318  case LOG_HTTP_CF_RESPONSE_HEADER:
319  /* RESPONSE HEADER */
320  if (tx->response_headers != NULL) {
321  h_response_hdr = htp_table_get_c(tx->response_headers,
322  httplog_ctx->cf_nodes[i]->data);
323  }
324  if (h_response_hdr != NULL) {
325  datalen = httplog_ctx->cf_nodes[i]->maxlen;
326  if (datalen == 0 || datalen > bstr_len(h_response_hdr->value)) {
327  datalen = bstr_len(h_response_hdr->value);
328  }
329  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset,
330  aft->buffer->size, (uint8_t *)bstr_ptr(h_response_hdr->value),
331  datalen);
332  } else {
333  MemBufferWriteString(aft->buffer, LOG_HTTP_CF_NONE);
334  }
335  break;
336  case LOG_HTTP_CF_RESPONSE_LEN:
337  /* RESPONSE LEN */
338  MemBufferWriteString(aft->buffer, "%"PRIuMAX"", (uintmax_t)tx->response_message_len);
339  break;
340  default:
341  /* NO MATCH */
342  MemBufferWriteString(aft->buffer, LOG_HTTP_CF_NONE);
343  SCLogDebug("No matching parameter %%%c for custom http log.", httplog_ctx->cf_nodes[i]->type);
344  break;
345  }
346  }
347  MemBufferWriteString(aft->buffer, "\n");
348 }

364  MemBufferWriteString(aft->buffer, "<no referer>");
365  }
366  MemBufferWriteString(aft->buffer, " [**] ");
367 
368  /* method */
369  if (tx->request_method != NULL) {
370  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset, aft->buffer->size,
371  (uint8_t *)bstr_ptr(tx->request_method),
372  bstr_len(tx->request_method));
373  }
374  MemBufferWriteString(aft->buffer, " [**] ");
375 
376  /* protocol */
377  if (tx->request_protocol != NULL) {
378  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset, aft->buffer->size,
379  (uint8_t *)bstr_ptr(tx->request_protocol),
380  bstr_len(tx->request_protocol));
381  } else {
382  MemBufferWriteString(aft->buffer, "<no protocol>");
383  }
384  MemBufferWriteString(aft->buffer, " [**] ");

395  MemBufferWriteString(aft->buffer, " => ");
396 
397  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset, aft->buffer->size,
398  (uint8_t *)bstr_ptr(h_location->value),
399  bstr_len(h_location->value));
400  }
401  }
402  } else {
403  MemBufferWriteString(aft->buffer, "<no status>");
404  }
405 
406  /* length */
407  MemBufferWriteString(aft->buffer, " [**] %"PRIuMAX" bytes", (uintmax_t)tx->response_message_len);
408 }

410 static TmEcode LogHttpLogIPWrapper(ThreadVars *tv, void *data, const Packet *p, Flow *f, HtpState *htp_state, htp_tx_t *tx, uint64_t tx_id, int ipproto)
411 {
412  SCEnter();
413 
414  LogHttpLogThread *aft = (LogHttpLogThread *)data;
415  LogHttpFileCtx *hlog = aft->httplog_ctx;
416  char timebuf[64];
417 
418  /* check if we have HTTP state or not */
419  CreateTimeString(&p->ts, timebuf, sizeof(timebuf));
420 
421  char srcip[46], dstip[46];
422  Port sp, dp;
423  if ((PKT_IS_TOSERVER(p))) {
424  switch (ipproto) {
425  case AF_INET:
426  PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p), srcip, sizeof(srcip));
427  PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p), dstip, sizeof(dstip));
428  break;
429  case AF_INET6:
430  PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p), srcip, sizeof(srcip));
431  PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p), dstip, sizeof(dstip));
432  break;
433  default:
434  goto end;
435  }
436  sp = p->sp;
437  dp = p->dp;
438  } else {
439  switch (ipproto) {
440  case AF_INET:
441  PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p), srcip, sizeof(srcip));
442  PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p), dstip, sizeof(dstip));
443  break;
444  case AF_INET6:
445  PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p), srcip, sizeof(srcip));
446  PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p), dstip, sizeof(dstip));
447  break;
448  default:
449  goto end;
450  }
451  sp = p->dp;
452  dp = p->sp;
453  }
454 
455  SCLogDebug("got a HTTP request and now logging !!");
456 
457  /* reset */
458  MemBufferReset(aft->buffer);
459 
460  if (hlog->flags & LOG_HTTP_CUSTOM) {
461  LogHttpLogCustom(aft, tx, &p->ts, srcip, sp, dstip, dp);
462  } else {
463  /* time */
464  MemBufferWriteString(aft->buffer, "%s ", timebuf);
465 
466  /* hostname */
467  if (tx->request_hostname != NULL) {
468  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset, aft->buffer->size,
469  (uint8_t *)bstr_ptr(tx->request_hostname),
470  bstr_len(tx->request_hostname));
471  } else {
472  MemBufferWriteString(aft->buffer, "<hostname unknown>");
473  }
474  MemBufferWriteString(aft->buffer, " [**] ");
475 
476  /* uri */
477  if (tx->request_uri != NULL) {
478  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset, aft->buffer->size,
479  (uint8_t *)bstr_ptr(tx->request_uri),
480  bstr_len(tx->request_uri));
481  }
482  MemBufferWriteString(aft->buffer, " [**] ");
483 
484  /* user agent */
485  htp_header_t *h_user_agent = NULL;
486  if (tx->request_headers != NULL) {
487  h_user_agent = htp_table_get_c(tx->request_headers, "user-agent");
488  }
489  if (h_user_agent != NULL) {
490  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset, aft->buffer->size,
491  (uint8_t *)bstr_ptr(h_user_agent->value),
492  bstr_len(h_user_agent->value));
493  } else {
494  MemBufferWriteString(aft->buffer, "<useragent unknown>");
495  }
496  if (hlog->flags & LOG_HTTP_EXTENDED) {
497  LogHttpLogExtended(aft, tx);
498  }
499 
500  /* ip/tcp header info */
501  MemBufferWriteString(aft->buffer,
502  " [**] %s:%" PRIu16 " -> %s:%" PRIu16 "\n",
503  srcip, sp, dstip, dp);
504  }
505 
506  aft->uri_cnt ++;
507 
508  SCMutexLock(&hlog->file_ctx->fp_mutex);
509  hlog->file_ctx->Write((const char *)MEMBUFFER_BUFFER(aft->buffer),
510  MEMBUFFER_OFFSET(aft->buffer), hlog->file_ctx);
511  SCMutexUnlock(&hlog->file_ctx->fp_mutex);
512 
513 end:
514  SCReturnInt(0);
515 
516 }

518 int LogHttpLogger(ThreadVars *tv, void *thread_data, const Packet *p, Flow *f, void *state, void *tx, uint64_t tx_id)
519 {
520  SCEnter();
521 
522  if (!(PKT_IS_TCP(p))) {
523  SCReturnInt(TM_ECODE_OK);
524  }
525 
526  int r = 0;
527  if (PKT_IS_IPV4(p)) {
528  r = LogHttpLogIPWrapper(tv, thread_data, p, f, (HtpState *)state, (htp_tx_t *)tx, tx_id, AF_INET);
529  } else if (PKT_IS_IPV6(p)) {
530  r = LogHttpLogIPWrapper(tv, thread_data, p, f, (HtpState *)state, (htp_tx_t *)tx, tx_id, AF_INET6);
531  }
532 
533  SCReturnInt(r);
534 }

536 TmEcode LogHttpLogThreadInit(ThreadVars *t, void *initdata, void **data)
537 {
538  LogHttpLogThread *aft = SCMalloc(sizeof(LogHttpLogThread));
539  if (unlikely(aft == NULL))
540  return TM_ECODE_FAILED;
541  memset(aft, 0, sizeof(LogHttpLogThread));
542 
543  if(initdata == NULL)
544  {
545  SCLogDebug("Error getting context for HTTPLog. \"initdata\" argument NULL");
546  SCFree(aft);
547  return TM_ECODE_FAILED;
548  }
549 
550  aft->buffer = MemBufferCreateNew(OUTPUT_BUFFER_SIZE);
551  if (aft->buffer == NULL) {
552  SCFree(aft);
553  return TM_ECODE_FAILED;
554  }
555 
556  /* Use the Ouptut Context (file pointer and mutex) */
557  aft->httplog_ctx= ((OutputCtx *)initdata)->data;
558

588 /** \brief Create a new http log LogFileCtx.
589  * \param conf Pointer to ConfNode containing this loggers configuration.
590  * \return NULL if failure, LogFileCtx* to the file_ctx if succesful
591  * */
592 OutputCtx *LogHttpLogInitCtx(ConfNode *conf)
593 {
594  LogFileCtx* file_ctx = LogFileNewCtx();
595  const char *p, *np;
596  uint32_t n;
597  if(file_ctx == NULL) {
598  SCLogError(SC_ERR_HTTP_LOG_GENERIC, "couldn't create new file_ctx");
599  return NULL;
600  }
601 
602  if (SCConfLogOpenGeneric(conf, file_ctx, DEFAULT_LOG_FILENAME, 1) < 0) {
603  LogFileFreeCtx(file_ctx);
604  return NULL;
605  }
606 
607  LogHttpFileCtx *httplog_ctx = SCMalloc(sizeof(LogHttpFileCtx));
608  if (unlikely(httplog_ctx == NULL)) {
609  LogFileFreeCtx(file_ctx);
610  return NULL;
611  }
612  memset(httplog_ctx, 0x00, sizeof(LogHttpFileCtx));
613 
614  httplog_ctx->file_ctx = file_ctx;
615  httplog_ctx->cf_n=0;
616 
617  const char *extended = ConfNodeLookupChildValue(conf, "extended");
618  const char *custom = ConfNodeLookupChildValue(conf, "custom");
619  const char *customformat = ConfNodeLookupChildValue(conf, "customformat");
620 
621  /* If custom logging format is selected, lets parse it */
622  if (custom != NULL && customformat != NULL && ConfValIsTrue(custom)) {
623  p=customformat;
624  httplog_ctx->flags |= LOG_HTTP_CUSTOM;
625  for (httplog_ctx->cf_n = 0; httplog_ctx->cf_n < LOG_HTTP_MAXN_NODES-1 && p && *p != '\0';
626  httplog_ctx->cf_n++){
627  httplog_ctx->cf_nodes[httplog_ctx->cf_n] = SCMalloc(sizeof(LogHttpCustomFormatNode));
628  if (httplog_ctx->cf_nodes[httplog_ctx->cf_n] == NULL) {
629  for (n = 0; n < httplog_ctx->cf_n; n++) {
630  SCFree(httplog_ctx->cf_nodes[n]);
631  }
632  LogFileFreeCtx(file_ctx);
633  SCFree(httplog_ctx);
634  return NULL;
635  }
636  httplog_ctx->cf_nodes[httplog_ctx->cf_n]->maxlen = 0;
637 
638  if (*p != '%'){
639  /* Literal found in format string */
640  httplog_ctx->cf_nodes[httplog_ctx->cf_n]->type = LOG_HTTP_CF_LITERAL;
641  np = strchr(p, '%');
642  if (np == NULL){
643  n = LOG_HTTP_NODE_STRLEN-2;
644  np = NULL; /* End */
645  }else{
646  n = np-p;
647  }
648  strlcpy(httplog_ctx->cf_nodes[httplog_ctx->cf_n]->data,p,n+1);
649  p = np;
650  } else {
651  /* Non Literal found in format string */
652  p++;
653  if (*p == '[') { /* Check if maxlength has been specified (ie: [25]) */
654  p++;
655  np = strchr(p, ']');
656  if (np != NULL) {
657  if (np-p > 0 && np-p < 10){
658  long maxlen = strtol(p,NULL,10);
659  if (maxlen > 0 && maxlen < LOG_HTTP_NODE_MAXOUTPUTLEN) {
660  httplog_ctx->cf_nodes[httplog_ctx->cf_n]->maxlen = (uint32_t) maxlen;
661  }
662  } else {
663  goto parsererror;
664  }
665  p = np + 1;
666  } else {
667  goto parsererror;
668  }
669  }
670  if (*p == '{') { /* Simple format char */
671  np = strchr(p, '}');
672  if (np != NULL && np-p > 1 && np-p < LOG_HTTP_NODE_STRLEN-2) {
673  p++;
674  n = np-p;
675  strlcpy(httplog_ctx->cf_nodes[httplog_ctx->cf_n]->data, p, n+1);
676  p = np;
677  } else {
678  goto parsererror;
679  }
680  p++;
681  } else {
682  httplog_ctx->cf_nodes[httplog_ctx->cf_n]->data[0] = '\0';
683  }
684  httplog_ctx->cf_nodes[httplog_ctx->cf_n]->type = *p;
685  if (*p == '%'){
686  httplog_ctx->cf_nodes[httplog_ctx->cf_n]->type = LOG_HTTP_CF_LITERAL;
687  strlcpy(httplog_ctx->cf_nodes[httplog_ctx->cf_n]->data, "%", 2);
688  }
689  p++;
690  }
691  }
692  } else {
693  if (extended == NULL) {
694  httplog_ctx->flags |= LOG_HTTP_DEFAULT;
695  } else {
696  if (ConfValIsTrue(extended)) {
697  httplog_ctx->flags |= LOG_HTTP_EXTENDED;
698  }
699  }
700  }
701 
702  OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
703  if (unlikely(output_ctx == NULL)) {
704  goto parsererror;
705  }
706 
707  output_ctx->data = httplog_ctx;
708  output_ctx->DeInit = LogHttpLogDeInitCtx;
709 
710  SCLogDebug("HTTP log output initialized");
711 
712  /* enable the logger for the app layer */
713  AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_HTTP);
714 
715  return output_ctx;
716 
717 parsererror:
718  for (n = 0;n < httplog_ctx->cf_n;n++) {
719  SCFree(httplog_ctx->cf_nodes[n]);
720  }
721  LogFileFreeCtx(file_ctx);
722  SCFree(httplog_ctx);
723  SCLogError(SC_ERR_INVALID_ARGUMENT,"Syntax error in custom http log format string.");
724  return NULL;
725 
726 }
