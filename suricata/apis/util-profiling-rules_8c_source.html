105 void SCProfilingRulesGlobalInit(void)
106 {
107  ConfNode *conf;
108  const char *val;
109 
110  conf = ConfGetNode("profiling.rules");
111  if (conf != NULL) {
112  if (ConfNodeChildValueIsTrue(conf, "enabled")) {
113  profiling_rules_enabled = 1;
114 
115  val = ConfNodeLookupChildValue(conf, "sort");
116  if (val != NULL) {
117  if (strcmp(val, "ticks") == 0) {
118  profiling_rules_sort_order =
119  SC_PROFILING_RULES_SORT_BY_TICKS;
120  }
121  else if (strcmp(val, "avgticks") == 0) {
122  profiling_rules_sort_order =
123  SC_PROFILING_RULES_SORT_BY_AVG_TICKS;
124  }
125  else if (strcmp(val, "avgticks_match") == 0) {
126  profiling_rules_sort_order =
127  SC_PROFILING_RULES_SORT_BY_AVG_TICKS_MATCH;
128  }
129  else if (strcmp(val, "avgticks_no_match") == 0) {
130  profiling_rules_sort_order =
131  SC_PROFILING_RULES_SORT_BY_AVG_TICKS_NO_MATCH;
132  }
133  else if (strcmp(val, "checks") == 0) {
134  profiling_rules_sort_order =
135  SC_PROFILING_RULES_SORT_BY_CHECKS;
136  }
137  else if (strcmp(val, "matches") == 0) {
138  profiling_rules_sort_order =
139  SC_PROFILING_RULES_SORT_BY_MATCHES;
140  }
141  else if (strcmp(val, "maxticks") == 0) {
142  profiling_rules_sort_order =
143  SC_PROFILING_RULES_SORT_BY_MAX_TICKS;
144  }
145  else {
146  SCLogError(SC_ERR_INVALID_ARGUMENT,
147  "Invalid profiling sort order: %s", val);
148  exit(EXIT_FAILURE);
149  }
150  }
151 
152  val = ConfNodeLookupChildValue(conf, "limit");
153  if (val != NULL) {
154  if (ByteExtractStringUint32(&profiling_rules_limit, 10,
155  (uint16_t)strlen(val), val) <= 0) {
156  SCLogError(SC_ERR_INVALID_ARGUMENT, "Invalid limit: %s", val);
157  exit(EXIT_FAILURE);
158  }
159  }
160  const char *filename = ConfNodeLookupChildValue(conf, "filename");
161  if (filename != NULL) {
162 
163  char *log_dir;
164  log_dir = ConfigGetLogDirectory();
165 
166  profiling_file_name = SCMalloc(PATH_MAX);
167  if (unlikely(profiling_file_name == NULL)) {
168  SCLogError(SC_ERR_MEM_ALLOC, "can't duplicate file name");
169  exit(EXIT_FAILURE);
170  }
171  snprintf(profiling_file_name, PATH_MAX, "%s/%s", log_dir, filename);
172 
173  const char *v = ConfNodeLookupChildValue(conf, "append");
174  if (v == NULL || ConfValIsTrue(v)) {
175  profiling_file_mode = "a";
176  } else {
177  profiling_file_mode = "w";
178  }
179 
180  profiling_output_to_file = 1;
181  }
182  if (ConfNodeChildValueIsTrue(conf, "json")) {
183 #ifdef HAVE_LIBJANSSON
184  profiling_rule_json = 1;
185 #else
186  SCLogWarning(SC_ERR_NO_JSON_SUPPORT, "no json support compiled in, using plain output");
187 #endif
188  }
189  }
190  }

293 static void DumpJson(FILE *fp, SCProfileSummary *summary, uint32_t count, uint64_t total_ticks)
294 {
295  char timebuf[64];
296  uint32_t i;
297  struct timeval tval;
298 
299  json_t *js = json_object();
300  if (js == NULL)
301  return;
302  json_t *jsa = json_array();
303  if (jsa == NULL) {
304  json_decref(js);
305  return;
306  }
307 
308  gettimeofday(&tval, NULL);
309  CreateIsoTimeString(&tval, timebuf, sizeof(timebuf));
310  json_object_set_new(js, "timestamp", json_string(timebuf));
311 
312  for (i = 0; i < count; i++) {
313  /* Stop dumping when we hit our first rule with 0 checks. Due
314  * to sorting this will be the beginning of all the rules with
315  * 0 checks. */
316  if (summary[i].checks == 0)
317  break;
318 
319  json_t *jsm = json_object();
320  if (jsm) {
321  json_object_set_new(jsm, "signature_id", json_integer(summary[i].sid));
322  json_object_set_new(jsm, "gid", json_integer(summary[i].gid));
323  json_object_set_new(jsm, "rev", json_integer(summary[i].rev));
324 
325  json_object_set_new(jsm, "checks", json_integer(summary[i].checks));
326  json_object_set_new(jsm, "matches", json_integer(summary[i].matches));
327 
328  json_object_set_new(jsm, "ticks_total", json_integer(summary[i].ticks));
329  json_object_set_new(jsm, "ticks_max", json_integer(summary[i].max));
330  json_object_set_new(jsm, "ticks_avg", json_integer(summary[i].avgticks));
331  json_object_set_new(jsm, "ticks_avg_match", json_integer(summary[i].avgticks_match));
332  json_object_set_new(jsm, "ticks_avg_nomatch", json_integer(summary[i].avgticks_no_match));
333 
334  double percent = (long double)summary[i].ticks /
335  (long double)total_ticks * 100;
336  json_object_set_new(jsm, "percent", json_integer(percent));
337  json_array_append(jsa, jsm);
338  }
339  }
340  json_object_set_new(js, "rules", jsa);
341 
342  char *js_s = json_dumps(js,
343  JSON_PRESERVE_ORDER|JSON_COMPACT|JSON_ENSURE_ASCII|
344 #ifdef JSON_ESCAPE_SLASH
345  JSON_ESCAPE_SLASH
346 #else
347  0
348 #endif
349  );
350 
351  if (unlikely(js_s == NULL))
352  return;
353  fprintf(fp, "%s", js_s);
354  free(js_s);
355  json_decref(js);
356 }

419 /**
420  * \brief Dump rule profiling information to file
421  *
422  * \param de_ctx The active DetectEngineCtx, used to get at the loaded rules.
423  */
424 void
425 SCProfilingRuleDump(SCProfileDetectCtx *rules_ctx)
426 {
427  uint32_t i;
428  FILE *fp;
429 
430  if (rules_ctx == NULL)
431  return;
432 
433  if (profiling_output_to_file == 1) {
434  fp = fopen(profiling_file_name, profiling_file_mode);
435 
436  if (fp == NULL) {
437  SCLogError(SC_ERR_FOPEN, "failed to open %s: %s", profiling_file_name,
438  strerror(errno));
439  return;
440  }
441  } else {
442  fp = stdout;
443  }
444 
445  int summary_size = sizeof(SCProfileSummary) * rules_ctx->size;
446  SCProfileSummary *summary = SCMalloc(summary_size);
447  if (unlikely(summary == NULL)) {
448  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory for profiling summary");
449  return;
450  }
451 
452  uint32_t count = rules_ctx->size;
453  uint64_t total_ticks = 0;
454 
455  SCLogInfo("Dumping profiling data for %u rules.", count);
456 
457  memset(summary, 0, summary_size);
458  for (i = 0; i < count; i++) {
459  summary[i].sid = rules_ctx->data[i].sid;
460  summary[i].rev = rules_ctx->data[i].rev;
461  summary[i].gid = rules_ctx->data[i].gid;
462 
463  summary[i].ticks = rules_ctx->data[i].ticks_match + rules_ctx->data[i].ticks_no_match;
464  summary[i].checks = rules_ctx->data[i].checks;
465 
466  if (summary[i].ticks > 0) {
467  summary[i].avgticks = (long double)summary[i].ticks / (long double)summary[i].checks;
468  }
469 
470  summary[i].matches = rules_ctx->data[i].matches;
471  summary[i].max = rules_ctx->data[i].max;
472  summary[i].ticks_match = rules_ctx->data[i].ticks_match;
473  summary[i].ticks_no_match = rules_ctx->data[i].ticks_no_match;
474  if (summary[i].ticks_match > 0) {
475  summary[i].avgticks_match = (long double)summary[i].ticks_match /
476  (long double)summary[i].matches;
477  }
478 
479  if (summary[i].ticks_no_match > 0) {
480  summary[i].avgticks_no_match = (long double)summary[i].ticks_no_match /
481  ((long double)summary[i].checks - (long double)summary[i].matches);
482  }
483  total_ticks += summary[i].ticks;
484  }
485 
486  switch (profiling_rules_sort_order) {
487  case SC_PROFILING_RULES_SORT_BY_TICKS:
488  qsort(summary, count, sizeof(SCProfileSummary),
489  SCProfileSummarySortByTicks);
490  break;
491  case SC_PROFILING_RULES_SORT_BY_AVG_TICKS:
492  qsort(summary, count, sizeof(SCProfileSummary),
493  SCProfileSummarySortByAvgTicks);
494  break;
495  case SC_PROFILING_RULES_SORT_BY_CHECKS:
496  qsort(summary, count, sizeof(SCProfileSummary),
497  SCProfileSummarySortByChecks);
498  break;
499  case SC_PROFILING_RULES_SORT_BY_MATCHES:
500  qsort(summary, count, sizeof(SCProfileSummary),
501  SCProfileSummarySortByMatches);
502  break;
503  case SC_PROFILING_RULES_SORT_BY_MAX_TICKS:
504  qsort(summary, count, sizeof(SCProfileSummary),
505  SCProfileSummarySortByMaxTicks);
506  break;
507  case SC_PROFILING_RULES_SORT_BY_AVG_TICKS_MATCH:
508  qsort(summary, count, sizeof(SCProfileSummary),
509  SCProfileSummarySortByAvgTicksMatch);
510  break;
511  case SC_PROFILING_RULES_SORT_BY_AVG_TICKS_NO_MATCH:
512  qsort(summary, count, sizeof(SCProfileSummary),
513  SCProfileSummarySortByAvgTicksNoMatch);
514  break;
515  }
516 #ifdef HAVE_LIBJANSSON
517  if (profiling_rule_json) {
518  DumpJson(fp, summary, count, total_ticks);
519  } else
520 #endif
521  {
522  DumpText(fp, summary, count, total_ticks);
523  }
524 
525  if (fp != stdout)
526  fclose(fp);
527  SCFree(summary);
528  SCLogInfo("Done dumping profiling data.");
529 }

543 /**
544  * \brief Update a rule counter.

546  * \param id The ID of this counter.
547  * \param ticks Number of CPU ticks for this rule.
548  * \param match Did the rule match?
549  */
550 void
551 SCProfilingRuleUpdateCounter(DetectEngineThreadCtx *det_ctx, uint16_t id, uint64_t ticks, int match)
552 {
553  if (det_ctx != NULL && det_ctx->rule_perf_data != NULL && det_ctx->rule_perf_data_size > id) {
554  SCProfileData *p = &det_ctx->rule_perf_data[id];
555 
556  p->checks++;
557  p->matches += match;
558  if (ticks > p->max)
559  p->max = ticks;
560  if (match == 1)
561  p->ticks_match += ticks;
562  else
563  p->ticks_no_match += ticks;
564  }

608 static void SCProfilingRuleThreadMerge(DetectEngineCtx *de_ctx, DetectEngineThreadCtx *det_ctx)
609 {
610  if (de_ctx == NULL || de_ctx->profile_ctx == NULL || de_ctx->profile_ctx->data == NULL ||
611  det_ctx == NULL || det_ctx->rule_perf_data == NULL)
612  return;
613 
614  int i;
615  for (i = 0; i < det_ctx->rule_perf_data_size; i++) {
616  de_ctx->profile_ctx->data[i].checks += det_ctx->rule_perf_data[i].checks;
617  de_ctx->profile_ctx->data[i].matches += det_ctx->rule_perf_data[i].matches;
618  de_ctx->profile_ctx->data[i].ticks_match += det_ctx->rule_perf_data[i].ticks_match;
619  de_ctx->profile_ctx->data[i].ticks_no_match += det_ctx->rule_perf_data[i].ticks_no_match;
620  if (det_ctx->rule_perf_data[i].max > de_ctx->profile_ctx->data[i].max)
621  de_ctx->profile_ctx->data[i].max = det_ctx->rule_perf_data[i].max;
622  }
