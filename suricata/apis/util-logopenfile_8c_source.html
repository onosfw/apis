40 /** \brief connect to the indicated local stream socket, logging any errors
41  * \param path filesystem path to connect to
42  * \param log_err, non-zero if connect failure should be logged.
43  * \retval FILE* on success (fdopen'd wrapper of underlying socket)
44  * \retval NULL on error
45  */
46 static FILE *
47 SCLogOpenUnixSocketFp(const char *path, int sock_type, int log_err)
48 {
49  struct sockaddr_un sun;
50  int s = -1;
51  FILE * ret = NULL;
52 
53  memset(&sun, 0x00, sizeof(sun));
54 
55  s = socket(PF_UNIX, sock_type, 0);
56  if (s < 0) goto err;
57 
58  sun.sun_family = AF_UNIX;
59  strlcpy(sun.sun_path, path, sizeof(sun.sun_path));
60 
61  if (connect(s, (const struct sockaddr *)&sun, sizeof(sun)) < 0)
62  goto err;
63 
64  ret = fdopen(s, "w");
65  if (ret == NULL)
66  goto err;
67 
68  return ret;
69 
70 err:
71  if (log_err)
72  SCLogWarning(SC_ERR_SOCKET,
73  "Error connecting to socket \"%s\": %s (will keep trying)",
74  path, strerror(errno));
75 
76  if (s >= 0)
77  close(s);
78 
79  return NULL;
80 }

82 /**
83  * \brief Attempt to reconnect a disconnected (or never-connected) Unix domain socket.
84  * \retval 1 if it is now connected; otherwise 0
85  */
86 static int SCLogUnixSocketReconnect(LogFileCtx *log_ctx)
87 {
88  int disconnected = 0;
89  if (log_ctx->fp) {
90  SCLogWarning(SC_ERR_SOCKET,
91  "Write error on Unix socket \"%s\": %s; reconnecting...",
92  log_ctx->filename, strerror(errno));
93  fclose(log_ctx->fp);
94  log_ctx->fp = NULL;
95  log_ctx->reconn_timer = 0;
96  disconnected = 1;
97  }
98 
99  struct timeval tv;
100  uint64_t now;
101  gettimeofday(&tv, NULL);
102  now = (uint64_t)tv.tv_sec * 1000;
103  now += tv.tv_usec / 1000; /* msec resolution */
104  if (log_ctx->reconn_timer != 0 &&
105  (now - log_ctx->reconn_timer) < LOGFILE_RECONN_MIN_TIME) {
106  /* Don't bother to try reconnecting too often. */
107  return 0;
108  }
109  log_ctx->reconn_timer = now;
110 
111  log_ctx->fp = SCLogOpenUnixSocketFp(log_ctx->filename, log_ctx->sock_type, 0);
112  if (log_ctx->fp) {
113  /* Connected at last (or reconnected) */
114  SCLogNotice("Reconnected socket \"%s\"", log_ctx->filename);
115  } else if (disconnected) {
116  SCLogWarning(SC_ERR_SOCKET, "Reconnect failed: %s (will keep trying)",
117  strerror(errno));
118  }
119 
120  return log_ctx->fp ? 1 : 0;
121 }

123 /**
124  * \brief Write buffer to log file.
125  * \retval 0 on failure; otherwise, the return value of fwrite (number of
126  * characters successfully written).
127  */
128 static int SCLogFileWrite(const char *buffer, int buffer_len, LogFileCtx *log_ctx)
129 {
130  /* Check for rotation. */
131  if (log_ctx->rotation_flag) {
132  log_ctx->rotation_flag = 0;
133  SCConfLogReopen(log_ctx);
134  }
135 
136  int ret = 0;
137 
138  if (log_ctx->fp == NULL && log_ctx->is_sock)
139  SCLogUnixSocketReconnect(log_ctx);
140 
141  if (log_ctx->fp) {
142  clearerr(log_ctx->fp);
143  ret = fwrite(buffer, buffer_len, 1, log_ctx->fp);
144  fflush(log_ctx->fp);
145 
146  if (ferror(log_ctx->fp) && log_ctx->is_sock) {
147  /* Error on Unix socket, maybe needs reconnect */
148  if (SCLogUnixSocketReconnect(log_ctx)) {
149  ret = fwrite(buffer, buffer_len, 1, log_ctx->fp);
150  fflush(log_ctx->fp);
151  }
152  }
153  }
154 
155  return ret;
156 }

205 /** \brief open a generic output "log file", which may be a regular file or a socket
206  * \param conf ConfNode structure for the output section in question
207  * \param log_ctx Log file context allocated by caller
208  * \param default_filename Default name of file to open, if not specified in ConfNode
209  * \param rotate Register the file for rotation in HUP.
210  * \retval 0 on success
211  * \retval -1 on error
212  */
213 int
214 SCConfLogOpenGeneric(ConfNode *conf,
215  LogFileCtx *log_ctx,
216  const char *default_filename,
217  int rotate)
218 {
219  char log_path[PATH_MAX];
220  char *log_dir;
221  const char *filename, *filetype;
222 
223  // Arg check
224  if (conf == NULL || log_ctx == NULL || default_filename == NULL) {
225  SCLogError(SC_ERR_INVALID_ARGUMENT,
226  "SCConfLogOpenGeneric(conf %p, ctx %p, default %p) "
227  "missing an argument",
228  conf, log_ctx, default_filename);
229  return -1;
230  }
231  if (log_ctx->fp != NULL) {
232  SCLogError(SC_ERR_INVALID_ARGUMENT,
233  "SCConfLogOpenGeneric: previously initialized Log CTX "
234  "encountered");
235  return -1;
236  }
237 
238  // Resolve the given config
239  filename = ConfNodeLookupChildValue(conf, "filename");
240  if (filename == NULL)
241  filename = default_filename;
242 
243  log_dir = ConfigGetLogDirectory();
244 
245  if (PathIsAbsolute(filename)) {
246  snprintf(log_path, PATH_MAX, "%s", filename);
247  } else {
248  snprintf(log_path, PATH_MAX, "%s/%s", log_dir, filename);
249  }
250 
251  filetype = ConfNodeLookupChildValue(conf, "filetype");
252  if (filetype == NULL)
253  filetype = DEFAULT_LOG_FILETYPE;
254 
255  const char *append = ConfNodeLookupChildValue(conf, "append");
256  if (append == NULL)
257  append = DEFAULT_LOG_MODE_APPEND;
258 
259  // Now, what have we been asked to open?
260  if (strcasecmp(filetype, "unix_stream") == 0) {
261  /* Don't bail. May be able to connect later. */
262  log_ctx->is_sock = 1;
263  log_ctx->sock_type = SOCK_STREAM;
264  log_ctx->fp = SCLogOpenUnixSocketFp(log_path, SOCK_STREAM, 1);
265  } else if (strcasecmp(filetype, "unix_dgram") == 0) {
266  /* Don't bail. May be able to connect later. */
267  log_ctx->is_sock = 1;
268  log_ctx->sock_type = SOCK_DGRAM;
269  log_ctx->fp = SCLogOpenUnixSocketFp(log_path, SOCK_DGRAM, 1);
270  } else if (strcasecmp(filetype, DEFAULT_LOG_FILETYPE) == 0 ||
271  strcasecmp(filetype, "file") == 0) {
272  log_ctx->fp = SCLogOpenFileFp(log_path, append);
273  if (log_ctx->fp == NULL)
274  return -1; // Error already logged by Open...Fp routine
275  log_ctx->is_regular = 1;
276  if (rotate) {
277  OutputRegisterFileRotationFlag(&log_ctx->rotation_flag);
278  }
279  } else if (strcasecmp(filetype, "pcie") == 0) {
280  log_ctx->pcie_fp = SCLogOpenPcieFp(log_ctx, log_path, append);
281  if (log_ctx->pcie_fp == NULL)
282  return -1; // Error already logged by Open...Fp routine
283  } else {
284  SCLogError(SC_ERR_INVALID_YAML_CONF_ENTRY, "Invalid entry for "
285  "%s.filetype. Expected \"regular\" (default), \"unix_stream\", "
286  "\"pcie\" "
287  "or \"unix_dgram\"",
288  conf->name);
289  }
290  log_ctx->filename = SCStrdup(log_path);
291  if (unlikely(log_ctx->filename == NULL)) {
292  SCLogError(SC_ERR_MEM_ALLOC,
293  "Failed to allocate memory for filename");
294  return -1;
295  }
296 
297  SCLogInfo("%s output device (%s) initialized: %s", conf->name, filetype,
298  filename);
299 
300  return 0;
301 }

303 /**
304  * \brief Reopen a regular log file with the side-affect of truncating it.
305  *
306  * This is useful to clear the log file and start a new one, or to
307  * re-open the file after its been moved by something external
308  * (eg. logrotate).
309  */
310 int SCConfLogReopen(LogFileCtx *log_ctx)
311 {
312  if (!log_ctx->is_regular) {
313  /* Not supported and not needed on non-regular files. */
314  return 0;
315  }
316 
317  if (log_ctx->filename == NULL) {
318  SCLogWarning(SC_ERR_INVALID_ARGUMENT,
319  "Can't re-open LogFileCtx without a filename.");
320  return -1;
321  }
322 
323  fclose(log_ctx->fp);
324 
325  /* Reopen the file. Append is forced in case the file was not
326  * moved as part of a rotation process. */
327  SCLogDebug("Reopening log file %s.", log_ctx->filename);
328  log_ctx->fp = SCLogOpenFileFp(log_ctx->filename, "yes");
329  if (log_ctx->fp == NULL) {
330  return -1; // Already logged by Open..Fp routine.
331  }
332 
333  return 0;
334 }

356 int SCConfLogOpenRedis(ConfNode *redis_node, LogFileCtx *log_ctx)
357 {
358  const char *redis_server = NULL;
359  const char *redis_port = NULL;
360  const char *redis_mode = NULL;
361  const char *redis_key = NULL;
362 
363  if (redis_node) {
364  redis_server = ConfNodeLookupChildValue(redis_node, "server");
365  redis_port = ConfNodeLookupChildValue(redis_node, "port");
366  redis_mode = ConfNodeLookupChildValue(redis_node, "mode");
367  redis_key = ConfNodeLookupChildValue(redis_node, "key");
368  }
369  if (!redis_server) {
370  redis_server = "127.0.0.1";
371  SCLogInfo("Using default redis server (127.0.0.1)");
372  }
373  if (!redis_port)
374  redis_port = "6379";
375  if (!redis_mode)
376  redis_mode = "list";
377  if (!redis_key)
378  redis_key = "suricata";
379  log_ctx->redis_setup.key = SCStrdup(redis_key);
380 
381  if (!log_ctx->redis_setup.key) {
382  SCLogError(SC_ERR_MEM_ALLOC, "Unable to allocate redis key name");
383  exit(EXIT_FAILURE);
384  }
385 
386  log_ctx->redis_setup.batch_size = 0;
387 
388  ConfNode *pipelining = ConfNodeLookupChild(redis_node, "pipelining");
389  if (pipelining) {
390  int enabled = 0;
391  int ret;
392  intmax_t val;
393  ret = ConfGetChildValueBool(pipelining, "enabled", &enabled);
394  if (ret && enabled) {
395  ret = ConfGetChildValueInt(pipelining, "batch-size", &val);
396  if (ret) {
397  log_ctx->redis_setup.batch_size = val;
398  } else {
399  log_ctx->redis_setup.batch_size = 10;
400  }
401  }
402  }
403 
404  if (!strcmp(redis_mode, "list")) {
405  log_ctx->redis_setup.command = redis_push_cmd;
406  if (!log_ctx->redis_setup.command) {
407  SCLogError(SC_ERR_MEM_ALLOC, "Unable to allocate redis key command");
408  exit(EXIT_FAILURE);
409  }
410  } else {
411  log_ctx->redis_setup.command = redis_publish_cmd;
412  if (!log_ctx->redis_setup.command) {
413  SCLogError(SC_ERR_MEM_ALLOC, "Unable to allocate redis key command");
414  exit(EXIT_FAILURE);
415  }
416  }
417  redisContext *c = redisConnect(redis_server, atoi(redis_port));
418  if (c != NULL && c->err) {
419  SCLogError(SC_ERR_SOCKET, "Error connecting to redis server: %s", c->errstr);
420  exit(EXIT_FAILURE);
421  }
422 
423  /* store server params for reconnection */
424  log_ctx->redis_setup.server = SCStrdup(redis_server);
425  if (!log_ctx->redis_setup.server) {
426  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating redis server string");
427  exit(EXIT_FAILURE);
428  }
429  log_ctx->redis_setup.port = atoi(redis_port);
430  log_ctx->redis_setup.tried = 0;
431 
432  log_ctx->redis = c;
433 
434  log_ctx->Close = SCLogFileCloseRedis;
435 
436  return 0;
437 }

439 int SCConfLogReopenRedis(LogFileCtx *log_ctx)
440 {
441  if (log_ctx->redis != NULL) {
442  redisFree(log_ctx->redis);
443  log_ctx->redis = NULL;
444  }
445 
446  /* only try to reconnect once per second */
447  if (log_ctx->redis_setup.tried >= time(NULL)) {
448  return -1;
449  }
450 
451  redisContext *c = redisConnect(log_ctx->redis_setup.server, log_ctx->redis_setup.port);
452  if (c != NULL && c->err) {
453  if (log_ctx->redis_setup.tried == 0) {
454  SCLogError(SC_ERR_SOCKET, "Error connecting to redis server: %s\n", c->errstr);
455  }
456  redisFree(c);
457  log_ctx->redis_setup.tried = time(NULL);
458  return -1;
459  }
460  log_ctx->redis = c;
461  log_ctx->redis_setup.tried = 0;
462  log_ctx->redis_setup.batch_count = 0;
463  return 0;
464 }

540 {
541  if (file_ctx->redis == NULL) {
542  SCConfLogReopenRedis(file_ctx);
543  if (file_ctx->redis == NULL) {
544  return -1;

550  if (file_ctx->redis_setup.batch_size) {
551  redisAppendCommand(file_ctx->redis, "%s %s %s",

561  freeReplyObject(reply);

574  SCLogInfo("Reconnected to redis server");
575  return 0;

579  }

581  default:
582  SCLogWarning(SC_ERR_INVALID_VALUE,
583  "Unsupported error code %d",

588  }
589  } else {
590  file_ctx->redis_setup.batch_count++;

593  redisReply *reply = redisCommand(file_ctx->redis, "%s %s %s",
594  file_ctx->redis_setup.command,

608  "Redis default triggered with %d", reply->type);
609  SCConfLogReopenRedis(file_ctx);
610  break;
611  }
612  freeReplyObject(reply);
613  }

619 {
620  if (file_ctx->type == LOGFILE_TYPE_SYSLOG) {
621  syslog(file_ctx->syslog_setup.alert_syslog_level, "%s",
622  (const char *)MEMBUFFER_BUFFER(buffer));
623  } else if (file_ctx->type == LOGFILE_TYPE_FILE ||
624  file_ctx->type == LOGFILE_TYPE_UNIX_DGRAM ||
625  file_ctx->type == LOGFILE_TYPE_UNIX_STREAM)

631  MEMBUFFER_OFFSET(buffer), file_ctx);

636  SCMutexLock(&file_ctx->fp_mutex);

638  MEMBUFFER_OFFSET(buffer));
