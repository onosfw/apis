82 {

88  *errcode = ERR_DER_MISSING_ELEMENT;

92  return NULL;
93 
94  for (offset=0; offset<seqsz; offset++) {

99  return NULL;

101  /* skip context-specific elements */

103  node = node->next;

131 int Asn1DerGetIssuerDN(const Asn1Generic *cert, char *buffer, uint32_t length, uint32_t *errcode)
132 {
133  const Asn1Generic *node_oid;
134  const Asn1Generic *node, *it;
135  const Asn1Generic *node_set;
136  const Asn1Generic *node_str;
137  const char *shortname;
138  int rc = -1;
139  const char *separator = ", ";
140 
141  if (errcode)
142  *errcode = ERR_DER_MISSING_ELEMENT;
143 
144  if (length < 10)
145  goto issuer_dn_error;
146  buffer[0] = '\0';
147 
148  node = Asn1DerGet(cert, SEQ_IDX_ISSUER, sizeof(SEQ_IDX_ISSUER), errcode);
149  if ((node == NULL) || node->type != ASN1_SEQUENCE)
150  goto issuer_dn_error;
151 
152  it = node;
153  while (it != NULL) {
154  if (it->data == NULL)
155  goto issuer_dn_error;
156  node_set = it->data;
157  if (node_set->type != ASN1_SET || node_set->data == NULL)
158  goto issuer_dn_error;
159  node = node_set->data;
160  if (node->type != ASN1_SEQUENCE || node->data == NULL)
161  goto issuer_dn_error;
162  node_oid = node->data;
163  if (node_oid->str == NULL || node_oid->type != ASN1_OID)
164  goto issuer_dn_error;
165  shortname = Oid2ShortStr(node_oid->str);
166  if (node->next == NULL)
167  goto issuer_dn_error;
168  node = node->next;
169  node_str = node->data;
170  if (node_str == NULL || node_str->str == NULL)
171  goto issuer_dn_error;
172 
173  switch (node_str->type) {
174  case ASN1_PRINTSTRING:
175  case ASN1_IA5STRING:
176  case ASN1_T61STRING:
177  case ASN1_UTF8STRING:
178  case ASN1_OCTETSTRING:
179  strlcat(buffer, shortname, length);
180  strlcat(buffer, "=", length);
181  strlcat(buffer, node_str->str, length);
182  break;
183  default:
184  if (errcode)
185  *errcode = ERR_DER_UNSUPPORTED_STRING;
186  goto issuer_dn_error;
187  }
188 
189  if (strcmp(shortname,"CN")==0)
190  separator = "/";
191  if (it->next != NULL)
192  strlcat(buffer, separator, length);
193  it = it->next;
194  }
195 
196  if (errcode)
197  *errcode = 0;
198 
199  rc = 0;
200 issuer_dn_error:
201  return rc;
202 }
