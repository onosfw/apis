43 /**
44  * This is a proof-of-concept processor that processes parameter names in
45  * a way _similar_ to PHP. Whitespace at the beginning is removed, and the
46  * remaining whitespace characters are converted to underscores. Proper
47  * research of PHP's behavior is needed before we can claim to be emulating it.
48  *
49  * @param[in,out] p
50  * @return HTP_OK on success, HTP_ERROR on failure.
51  */
52 htp_status_t htp_php_parameter_processor(htp_param_t *p) {
53  if (p == NULL) return HTP_ERROR;
54 
55  // Name transformation
56 
57  bstr *new_name = NULL;
58 
59  // Ignore whitespace characters at the beginning of parameter name.
60 
61  unsigned char *data = bstr_ptr(p->name);
62  size_t len = bstr_len(p->name);
63  size_t pos = 0;
64 
65  // Advance over any whitespace characters at the beginning of the name.
66  while ((pos < len) && (isspace(data[pos]))) pos++;
67 
68  // Have we seen any whitespace?
69  if (pos > 0) {
70  // Make a copy of the name, starting with
71  // the first non-whitespace character.
72  new_name = bstr_dup_mem(data + pos, len - pos);
73  if (new_name == NULL) return HTP_ERROR;
74  }
75 
76  // Replace remaining whitespace characters with underscores.
77 
78  size_t offset = pos;
79  pos = 0;
80 
81  // Advance to the end of name or to the first whitespace character.
82  while ((offset + pos < len)&&(!isspace(data[pos]))) pos++;
83 
84  // Are we at the end of the name?
85  if (offset + pos < len) {
86  // Seen whitespace within the string.
87 
88  // Make a copy of the name if needed (which would be the case
89  // with a parameter that does not have any whitespace in front).
90  if (new_name == NULL) {
91  new_name = bstr_dup(p->name);
92  if (new_name == NULL) return HTP_ERROR;
93  }
94 
95  // Change the pointers to the new name and ditch the offset.
96  data = bstr_ptr(new_name);
97  len = bstr_len(new_name);
98 
99  // Replace any whitespace characters in the copy with underscores.
100  while (pos < len) {
101  if (isspace(data[pos])) {
102  data[pos] = '_';
103  }
104 
105  pos++;
106  }
107  }
108 
109  // If we made any changes, free the old parameter name and put the new one in.
110  if (new_name != NULL) {
111  bstr_free(p->name);
112  p->name = new_name;
113  }
114 
115  return HTP_OK;
116 }
