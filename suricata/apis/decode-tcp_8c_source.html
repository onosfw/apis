45 static int DecodeTCPOptions(Packet *p, uint8_t *pkt, uint16_t len)
46 {
47  uint16_t plen = len;
48  while (plen)
49  {
50  /* single byte options */
51  if (*pkt == TCP_OPT_EOL) {
52  break;
53  } else if (*pkt == TCP_OPT_NOP) {
54  pkt++;
55  plen--;
56 
57  /* multibyte options */
58  } else {
59  if (plen < 2) {
60  break;
61  }
62 
63  /* we already know that the total options len is valid,
64  * so here the len of the specific option must be bad.
65  * Also check for invalid lengths 0 and 1. */
66  if (unlikely(*(pkt+1) > plen || *(pkt+1) < 2)) {
67  ENGINE_SET_INVALID_EVENT(p, TCP_OPT_INVALID_LEN);
68  return -1;
69  }
70 
71  p->TCP_OPTS[p->TCP_OPTS_CNT].type = *pkt;
72  p->TCP_OPTS[p->TCP_OPTS_CNT].len = *(pkt+1);
73  if (plen > 2)
74  p->TCP_OPTS[p->TCP_OPTS_CNT].data = (pkt+2);
75  else
76  p->TCP_OPTS[p->TCP_OPTS_CNT].data = NULL;
77 
78  /* we are parsing the most commonly used opts to prevent
79  * us from having to walk the opts list for these all the
80  * time. */
81  switch (p->TCP_OPTS[p->TCP_OPTS_CNT].type) {
82  case TCP_OPT_WS:
83  if (p->TCP_OPTS[p->TCP_OPTS_CNT].len != TCP_OPT_WS_LEN) {
84  ENGINE_SET_EVENT(p,TCP_OPT_INVALID_LEN);
85  } else {
86  if (p->tcpvars.ws != NULL) {
87  ENGINE_SET_EVENT(p,TCP_OPT_DUPLICATE);
88  } else {
89  p->tcpvars.ws = &p->TCP_OPTS[p->TCP_OPTS_CNT];
90  }
91  }
92  break;
93  case TCP_OPT_MSS:
94  if (p->TCP_OPTS[p->TCP_OPTS_CNT].len != TCP_OPT_MSS_LEN) {
95  ENGINE_SET_EVENT(p,TCP_OPT_INVALID_LEN);
96  } else {
97  if (p->tcpvars.mss != NULL) {
98  ENGINE_SET_EVENT(p,TCP_OPT_DUPLICATE);
99  } else {
100  p->tcpvars.mss = &p->TCP_OPTS[p->TCP_OPTS_CNT];
101  }
102  }
103  break;
104  case TCP_OPT_SACKOK:
105  if (p->TCP_OPTS[p->TCP_OPTS_CNT].len != TCP_OPT_SACKOK_LEN) {
106  ENGINE_SET_EVENT(p,TCP_OPT_INVALID_LEN);
107  } else {
108  if (p->tcpvars.sackok != NULL) {
109  ENGINE_SET_EVENT(p,TCP_OPT_DUPLICATE);
110  } else {
111  p->tcpvars.sackok = &p->TCP_OPTS[p->TCP_OPTS_CNT];
112  }
113  }
114  break;
115  case TCP_OPT_TS:
116  if (p->TCP_OPTS[p->TCP_OPTS_CNT].len != TCP_OPT_TS_LEN) {
117  ENGINE_SET_EVENT(p,TCP_OPT_INVALID_LEN);
118  } else {
119  if (p->tcpvars.ts != NULL) {
120  ENGINE_SET_EVENT(p,TCP_OPT_DUPLICATE);
121  } else {
122  p->tcpvars.ts = &p->TCP_OPTS[p->TCP_OPTS_CNT];
123  }
124  }
125  break;
126  case TCP_OPT_SACK:
127  SCLogDebug("SACK option, len %u", p->TCP_OPTS[p->TCP_OPTS_CNT].len);
128  if (p->TCP_OPTS[p->TCP_OPTS_CNT].len < TCP_OPT_SACK_MIN_LEN ||
129  p->TCP_OPTS[p->TCP_OPTS_CNT].len > TCP_OPT_SACK_MAX_LEN ||
130  !((p->TCP_OPTS[p->TCP_OPTS_CNT].len - 2) % 8 == 0))
131  {
132  ENGINE_SET_EVENT(p,TCP_OPT_INVALID_LEN);
133  } else {
134  if (p->tcpvars.sack != NULL) {
135  ENGINE_SET_EVENT(p,TCP_OPT_DUPLICATE);
136  } else {
137  p->tcpvars.sack = &p->TCP_OPTS[p->TCP_OPTS_CNT];
138  }
139  }
140  break;
141  }
142 
143  pkt += p->TCP_OPTS[p->TCP_OPTS_CNT].len;
144  plen -= (p->TCP_OPTS[p->TCP_OPTS_CNT].len);
145  p->TCP_OPTS_CNT++;
146  }
147  }
148  return 0;
149 }

151 static int DecodeTCPPacket(ThreadVars *tv, Packet *p, uint8_t *pkt, uint16_t len)
152 {
153  if (unlikely(len < TCP_HEADER_LEN)) {
154  ENGINE_SET_INVALID_EVENT(p, TCP_PKT_TOO_SMALL);
155  return -1;
156  }
157 
158  p->tcph = (TCPHdr *)pkt;
159 
160  uint8_t hlen = TCP_GET_HLEN(p);
161  if (unlikely(len < hlen)) {
162  ENGINE_SET_INVALID_EVENT(p, TCP_HLEN_TOO_SMALL);
163  return -1;
164  }
165 
166  uint8_t tcp_opt_len = hlen - TCP_HEADER_LEN;
167  if (unlikely(tcp_opt_len > TCP_OPTLENMAX)) {
168  ENGINE_SET_INVALID_EVENT(p, TCP_INVALID_OPTLEN);
169  return -1;
170  }
171 
172  if (likely(tcp_opt_len > 0)) {
173  DecodeTCPOptions(p, pkt + TCP_HEADER_LEN, tcp_opt_len);
174  }
175 
176  SET_TCP_SRC_PORT(p,&p->sp);
177  SET_TCP_DST_PORT(p,&p->dp);
178 
179  p->proto = IPPROTO_TCP;
180 
181  p->payload = pkt + hlen;
182  p->payload_len = len - hlen;
183 
184  return 0;
185 }

187 int DecodeTCP(ThreadVars *tv, DecodeThreadVars *dtv, Packet *p, uint8_t *pkt, uint16_t len, PacketQueue *pq)
188 {
189  StatsIncr(tv, dtv->counter_tcp);
190 
191  if (unlikely(DecodeTCPPacket(tv, p,pkt,len) < 0)) {
192  SCLogDebug("invalid TCP packet");
193  p->tcph = NULL;
194  return TM_ECODE_FAILED;
195  }
196 
197 #ifdef DEBUG
198  SCLogDebug("TCP sp: %" PRIu32 " -> dp: %" PRIu32 " - HLEN: %" PRIu32 " LEN: %" PRIu32 " %s%s%s%s%s",
199  GET_TCP_SRC_PORT(p), GET_TCP_DST_PORT(p), TCP_GET_HLEN(p), len,
200  p->tcpvars.sackok ? "SACKOK " : "", p->tcpvars.sack ? "SACK " : "",
201  p->tcpvars.ws ? "WS " : "", p->tcpvars.ts ? "TS " : "",
202  p->tcpvars.mss ? "MSS " : "");
203 #endif
204 
205  /* Flow is an integral part of us */
206  FlowHandlePacket(tv, dtv, p);
207 
208  return TM_ECODE_OK;
209 }
