93 static int JsonSshLogger(ThreadVars *tv, void *thread_data, const Packet *p)
94 {
95  JsonSshLogThread *aft = (JsonSshLogThread *)thread_data;
96  MemBuffer *buffer = (MemBuffer *)aft->buffer;
97  OutputSshCtx *ssh_ctx = aft->sshlog_ctx;
98 
99  if (unlikely(p->flow == NULL)) {
100  return 0;
101  }
102 
103  /* check if we have SSH state or not */
104  FLOWLOCK_WRLOCK(p->flow);
105  uint16_t proto = FlowGetAppProtocol(p->flow);
106  if (proto != ALPROTO_SSH)
107  goto end;
108 
109  SshState *ssh_state = (SshState *)FlowGetAppState(p->flow);
110  if (unlikely(ssh_state == NULL)) {
111  goto end;
112  }
113 
114  if (ssh_state->cli_hdr.software_version == NULL || ssh_state->srv_hdr.software_version == NULL)
115  goto end;
116 
117  json_t *js = CreateJSONHeader((Packet *)p, 1, "ssh");//TODO
118  if (unlikely(js == NULL))
119  goto end;
120 
121  json_t *tjs = json_object();
122  if (tjs == NULL) {
123  free(js);
124  goto end;
125  }
126 
127  /* reset */
128  MemBufferReset(buffer);
129 
130  JsonSshLogJSON(tjs, ssh_state);
131 
132  json_object_set_new(js, "ssh", tjs);
133 
134  OutputJSONBuffer(js, ssh_ctx->file_ctx, buffer);
135  json_object_clear(js);
136  json_decref(js);
137 
138  /* we only log the state once */
139  ssh_state->cli_hdr.flags |= SSH_FLAG_STATE_LOGGED;
140 end:
141  FLOWLOCK_UNLOCK(p->flow);
142  return 0;
143 }

146 static TmEcode JsonSshLogThreadInit(ThreadVars *t, void *initdata, void **data)
147 {
148  JsonSshLogThread *aft = SCMalloc(sizeof(JsonSshLogThread));
149  if (unlikely(aft == NULL))
150  return TM_ECODE_FAILED;
151  memset(aft, 0, sizeof(JsonSshLogThread));
152 
153  if(initdata == NULL)
154  {
155  SCLogDebug("Error getting context for HTTPLog. \"initdata\" argument NULL");
156  SCFree(aft);
157  return TM_ECODE_FAILED;
158  }
159 
160  /* Use the Ouptut Context (file pointer and mutex) */
161  aft->sshlog_ctx = ((OutputCtx *)initdata)->data;
162 
163  aft->buffer = MemBufferCreateNew(OUTPUT_BUFFER_SIZE);
164  if (aft->buffer == NULL) {
165  SCFree(aft);
166  return TM_ECODE_FAILED;
167  }
168 
169  *data = (void *)aft;
170  return TM_ECODE_OK;
171 }

199 #define DEFAULT_LOG_FILENAME "ssh.json"
200 OutputCtx *OutputSshLogInit(ConfNode *conf)
201 {
202  if (OutputSshLoggerEnable() != 0) {
203  SCLogError(SC_ERR_CONF_YAML_ERROR, "only one 'ssh' logger "
204  "can be enabled");
205  return NULL;
206  }
207 
208  LogFileCtx *file_ctx = LogFileNewCtx();
209  if(file_ctx == NULL) {
210  SCLogError(SC_ERR_HTTP_LOG_GENERIC, "couldn't create new file_ctx");
211  return NULL;
212  }
213 
214  if (SCConfLogOpenGeneric(conf, file_ctx, DEFAULT_LOG_FILENAME, 1) < 0) {
215  LogFileFreeCtx(file_ctx);
216  return NULL;
217  }
218 
219  OutputSshCtx *ssh_ctx = SCMalloc(sizeof(OutputSshCtx));
220  if (unlikely(ssh_ctx == NULL)) {
221  LogFileFreeCtx(file_ctx);
222  return NULL;
223  }
224 
225  OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
226  if (unlikely(output_ctx == NULL)) {
227  LogFileFreeCtx(file_ctx);
228  SCFree(ssh_ctx);
229  return NULL;
230  }
231 
232  ssh_ctx->file_ctx = file_ctx;
233 
234  output_ctx->data = ssh_ctx;
235  output_ctx->DeInit = OutputSshLogDeinit;
236 
237  return output_ctx;
238 }

278  * \brief Condition function for SSH logger
279  * \retval bool true or false -- log now?
280  */
281 static int JsonSshCondition(ThreadVars *tv, const Packet *p)
282 {
283  if (p->flow == NULL) {
284  return FALSE;
285  }
286 
287  if (!(PKT_IS_TCP(p))) {
288  return FALSE;
289  }
290 
291  FLOWLOCK_RDLOCK(p->flow);
292  uint16_t proto = FlowGetAppProtocol(p->flow);
293  if (proto != ALPROTO_SSH)
294  goto dontlog;
295 
296  SshState *ssh_state = (SshState *)FlowGetAppState(p->flow);
297  if (ssh_state == NULL) {
298  SCLogDebug("no ssh state, so no logging");
299  goto dontlog;
300  }
301 
302  /* we only log the state once */
303  if (ssh_state->cli_hdr.flags & SSH_FLAG_STATE_LOGGED)
304  goto dontlog;
305 
306  if (ssh_state->cli_hdr.software_version == NULL ||
307  ssh_state->srv_hdr.software_version == NULL)
308  goto dontlog;
309 
310  /* todo: logic to log once */
311 
312  FLOWLOCK_UNLOCK(p->flow);
313  return TRUE;
314 dontlog:
315  FLOWLOCK_UNLOCK(p->flow);
316  return FALSE;
317 }
