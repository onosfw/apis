216 /** \brief To initialize the NFQ global configuration data
217  *
218  * \param quiet It tells the mode of operation, if it is TRUE nothing will
219  * be get printed.

221 void NFQInitConfig(char quiet)
222 {
223  intmax_t value = 0;
224  char* nfq_mode = NULL;
225  int boolval;
226 
227  SCLogDebug("Initializing NFQ");
228 
229  memset(&nfq_config, 0, sizeof(nfq_config));
230 
231  if ((ConfGet("nfq.mode", &nfq_mode)) == 0) {
232  nfq_config.mode = NFQ_ACCEPT_MODE;
233  } else {
234  if (!strcmp("accept", nfq_mode)) {
235  nfq_config.mode = NFQ_ACCEPT_MODE;
236  } else if (!strcmp("repeat", nfq_mode)) {
237  nfq_config.mode = NFQ_REPEAT_MODE;
238  } else if (!strcmp("route", nfq_mode)) {
239  nfq_config.mode = NFQ_ROUTE_MODE;
240  } else {
241  SCLogError(SC_ERR_INVALID_ARGUMENT, "Unknown nfq.mode");
242  exit(EXIT_FAILURE);
243  }
244  }
245 
246  (void)ConfGetBool("nfq.fail-open", (int *)&boolval);
247  if (boolval) {
248 #ifdef HAVE_NFQ_SET_QUEUE_FLAGS
249  SCLogInfo("Enabling fail-open on queue");
250  nfq_config.flags |= NFQ_FLAG_FAIL_OPEN;
251 #else
252  SCLogError(SC_ERR_NFQ_NOSUPPORT,
253  "nfq.%s set but NFQ library has no support for it.", "fail-open");
254 #endif
255  }
256 
257  if ((ConfGetInt("nfq.repeat-mark", &value)) == 1) {
258  nfq_config.mark = (uint32_t)value;
259  }
260 
261  if ((ConfGetInt("nfq.repeat-mask", &value)) == 1) {
262  nfq_config.mask = (uint32_t)value;
263  }
264 
265  if ((ConfGetInt("nfq.route-queue", &value)) == 1) {
266  nfq_config.next_queue = ((uint32_t)value) << 16;
267  }
268 
269  if ((ConfGetInt("nfq.batchcount", &value)) == 1) {
270 #ifdef HAVE_NFQ_SET_VERDICT_BATCH
271  if (value > 255) {
272  SCLogWarning(SC_ERR_INVALID_ARGUMENT, "nfq.batchcount cannot exceed 255.");
273  value = 255;
274  }
275  if (value > 1)
276  nfq_config.batchcount = (uint8_t) (value - 1);
277 #else
278  SCLogWarning(SC_ERR_NFQ_NOSUPPORT,
279  "nfq.%s set but NFQ library has no support for it.", "batchcount");
280 #endif
281  }
282 
283  if (!quiet) {
284  switch (nfq_config.mode) {
285  case NFQ_ACCEPT_MODE:
286  SCLogInfo("NFQ running in standard ACCEPT/DROP mode");
287  break;
288  case NFQ_REPEAT_MODE:
289  SCLogInfo("NFQ running in REPEAT mode with mark %"PRIu32"/%"PRIu32,
290  nfq_config.mark, nfq_config.mask);
291  break;
292  case NFQ_ROUTE_MODE:
293  SCLogInfo("NFQ running in route mode with next queue %"PRIu32,
294  nfq_config.next_queue >> 16);
295  break;
296  }
297  }
298 
299 }

310 static void NFQVerdictCacheFlush(NFQQueueVars *t)
311 {
312 #ifdef HAVE_NFQ_SET_VERDICT_BATCH
313  int ret;
314  int iter = 0;
315 
316  do {
317  if (t->verdict_cache.mark_valid)
318  ret = nfq_set_verdict_batch2(t->qh,
319  t->verdict_cache.packet_id,
320  t->verdict_cache.verdict,
321  t->verdict_cache.mark);
322  else
323  ret = nfq_set_verdict_batch(t->qh,
324  t->verdict_cache.packet_id,
325  t->verdict_cache.verdict);
326  } while ((ret < 0) && (iter++ < NFQ_VERDICT_RETRY_TIME));
327 
328  if (ret < 0) {
329  SCLogWarning(SC_ERR_NFQ_SET_VERDICT, "nfq_set_verdict_batch failed: %s",
330  strerror(errno));

332  t->verdict_cache.len = 0;
333  t->verdict_cache.mark_valid = 0;
334  }

338 static int NFQVerdictCacheAdd(NFQQueueVars *t, Packet *p, uint32_t verdict)
339 {
340 #ifdef HAVE_NFQ_SET_VERDICT_BATCH
341  if (t->verdict_cache.maxlen == 0)
342  return -1;
343 
344  if (p->flags & PKT_STREAM_MODIFIED || verdict == NF_DROP)
345  goto flush;
346 
347  if (p->flags & PKT_MARK_MODIFIED) {
348  if (!t->verdict_cache.mark_valid) {
349  if (t->verdict_cache.len)
350  goto flush;
351  t->verdict_cache.mark_valid = 1;
352  t->verdict_cache.mark = p->nfq_v.mark;
353  } else if (t->verdict_cache.mark != p->nfq_v.mark) {
354  goto flush;
355  }
356  } else if (t->verdict_cache.mark_valid) {
357  goto flush;
358  }
359 
360  if (t->verdict_cache.len == 0) {
361  t->verdict_cache.verdict = verdict;
362  } else if (t->verdict_cache.verdict != verdict)
363  goto flush;
364 
365  /* same verdict, mark not set or identical -> can cache */
366  t->verdict_cache.packet_id = p->nfq_v.id;
367 
368  if (t->verdict_cache.len >= t->verdict_cache.maxlen)
369  NFQVerdictCacheFlush(t);
370  else
371  t->verdict_cache.len++;
372  return 0;
373  flush:
374  /* can't cache. Flush current cache and signal caller it should send single verdict */
375  if (NFQVerdictCacheLen(t) > 0)
376  NFQVerdictCacheFlush(t);
377 #endif
378  return -1;
379 }

406 /**
407  * \brief Read data from nfq message and setup Packet
408  *
409  * \note
410  * In case of error, this function verdict the packet
411  * to avoid skb to get stuck in kernel.
412  */
413 int NFQSetupPkt (Packet *p, struct nfq_q_handle *qh, void *data)
414 {
415  struct nfq_data *tb = (struct nfq_data *)data;
416  int ret;
417  char *pktdata;
418  struct nfqnl_msg_packet_hdr *ph;
419 
420  ph = nfq_get_msg_packet_hdr(tb);
421  if (ph != NULL) {
422  p->nfq_v.id = ntohl(ph->packet_id);
423  //p->nfq_v.hw_protocol = ntohs(p->nfq_v.ph->hw_protocol);
424  p->nfq_v.hw_protocol = ph->hw_protocol;
425  }
426  p->nfq_v.mark = nfq_get_nfmark(tb);
427  if (nfq_config.mode == NFQ_REPEAT_MODE) {
428  if ((nfq_config.mark & nfq_config.mask) ==
429  (p->nfq_v.mark & nfq_config.mask)) {
430  int iter = 0;
431  if (already_seen_warning < MAX_ALREADY_TREATED)
432  SCLogInfo("Packet seems already treated by suricata");
433  already_seen_warning++;
434  do {
435  ret = nfq_set_verdict(qh, p->nfq_v.id, NF_ACCEPT, 0, NULL);
436  } while ((ret < 0) && (iter++ < NFQ_VERDICT_RETRY_TIME));
437  if (ret < 0) {
438  SCLogWarning(SC_ERR_NFQ_SET_VERDICT,
439  "nfq_set_verdict of %p failed %" PRId32 ": %s",
440  p, ret, strerror(errno));
441  }
442  return -1 ;
443  }
444  }
445  p->nfq_v.ifi = nfq_get_indev(tb);
446  p->nfq_v.ifo = nfq_get_outdev(tb);
447  p->nfq_v.verdicted = 0;
448 
449 #ifdef NFQ_GET_PAYLOAD_SIGNED
450  ret = nfq_get_payload(tb, &pktdata);
451 #else
452  ret = nfq_get_payload(tb, (unsigned char **) &pktdata);
453 #endif /* NFQ_GET_PAYLOAD_SIGNED */
454  if (ret > 0) {
455  /* nfq_get_payload returns a pointer to a part of memory
456  * that is not preserved over the lifetime of our packet.
457  * So we need to copy it. */
458  if (ret > 65536) {
459  /* Will not be able to copy data ! Set length to 0
460  * to trigger an error in packet decoding.
461  * This is unlikely to happen */
462  SCLogWarning(SC_ERR_INVALID_ARGUMENTS, "NFQ sent too big packet");
463  SET_PKT_LEN(p, 0);
464  } else if (runmode_workers) {
465  PacketSetData(p, (uint8_t *)pktdata, ret);
466  } else {
467  PacketCopyData(p, (uint8_t *)pktdata, ret);
468  }
469  } else if (ret == -1) {
470  /* unable to get pointer to data, ensure packet length is zero.
471  * This will trigger an error in packet decoding */
472  SET_PKT_LEN(p, 0);
473  }
474 
475  ret = nfq_get_timestamp(tb, &p->ts);
476  if (ret != 0) {
477  memset (&p->ts, 0, sizeof(struct timeval));
478  gettimeofday(&p->ts, NULL);
479  }
480 
481  p->datalink = DLT_RAW;
482  return 0;
483 }

495  struct nfq_data *nfa, void *data)
496 {
497  NFQThreadVars *ntv = (NFQThreadVars *)data;
498  ThreadVars *tv = ntv->tv;
499  int ret;
500 
501  /* grab a packet */
502  Packet *p = PacketGetFromQueueOrAlloc();
503  if (p == NULL) {
504  return -1;
505  }
506  PKT_SET_SRC(p, PKT_SRC_WIRE);
507 
508  p->nfq_v.nfq_index = ntv->nfq_index;
509  ret = NFQSetupPkt(p, qh, (void *)nfa);

516 #endif /* COUNTERS */
517  /* NFQSetupPkt is issuing a verdict

519  TmqhOutputPacketpool(tv, p);
520  return 0;
521  }
522 
523  p->ReleasePacket = NFQReleasePacket;
524 
525 #ifdef COUNTERS
526  NFQQueueVars *nfq_q = NFQGetQueue(ntv->nfq_index);
527  nfq_q->pkts++;
528  nfq_q->bytes += GET_PKT_LEN(p);
529 #endif /* COUNTERS */
530 
531  if (ntv->slot) {
532  if (TmThreadsSlotProcessPkt(tv, ntv->slot, p) != TM_ECODE_OK) {
533  TmqhOutputPacketpool(ntv->tv, p);
534  return -1;
535  }
536  } else {
537  /* pass on... */
538  tv->tmqh_out(tv, p);
539  }
540 
541  return 0;
542 }

1072  case NFQ_REPEAT_MODE:

1101  case NFQ_ROUTE_MODE:

1113 #else /* fall back to old function */

1120  htonl(p->nfq_v.mark),
1121  0, NULL);
1122  }
1123 #endif /* HAVE_NFQ_SET_VERDICT2 */
