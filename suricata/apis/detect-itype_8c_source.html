70

90 /**
91  * \brief This function is used to match itype rule option set on a packet with those passed via itype:
92  *
93  * \param t pointer to thread vars
94  * \param det_ctx pointer to the pattern matcher thread
95  * \param p pointer to the current packet
96  * \param m pointer to the sigmatch that we will cast into DetectITypeData
97  *
98  * \retval 0 no match
99  * \retval 1 match
100  */
101 int DetectITypeMatch (ThreadVars *t, DetectEngineThreadCtx *det_ctx, Packet *p, Signature *s, const SigMatchCtx *ctx)
102 {
103  int ret = 0;
104  uint8_t pitype;
105  const DetectITypeData *itd = (const DetectITypeData *)ctx;
106 
107  if (PKT_IS_PSEUDOPKT(p))
108  return 0;
109 
110  if (PKT_IS_ICMPV4(p)) {
111  pitype = ICMPV4_GET_TYPE(p);
112  } else if (PKT_IS_ICMPV6(p)) {
113  pitype = ICMPV6_GET_TYPE(p);
114  } else {
115  /* Packet not ICMPv4 nor ICMPv6 */
116  return ret;
117  }
118 
119  switch(itd->mode) {
120  case DETECT_ITYPE_EQ:
121  ret = (pitype == itd->type1) ? 1 : 0;
122  break;
123  case DETECT_ITYPE_LT:
124  ret = (pitype < itd->type1) ? 1 : 0;
125  break;
126  case DETECT_ITYPE_GT:
127  ret = (pitype > itd->type1) ? 1 : 0;
128  break;
129  case DETECT_ITYPE_RN:
130  ret = (pitype > itd->type1 && pitype < itd->type2) ? 1 : 0;
131  break;
132  }
133 
134  return ret;
135 }

137 /**
138  * \brief This function is used to parse itype options passed via itype: keyword
139  *
140  * \param itypestr Pointer to the user provided itype options
141  *
142  * \retval itd pointer to DetectITypeData on success
143  * \retval NULL on failure
144  */
145 DetectITypeData *DetectITypeParse(char *itypestr)
146 {
147  DetectITypeData *itd = NULL;
148  char *args[3] = {NULL, NULL, NULL};
149 #define MAX_SUBSTRINGS 30
150  int ret = 0, res = 0;
151  int ov[MAX_SUBSTRINGS];
152 
153  ret = pcre_exec(parse_regex, parse_regex_study, itypestr, strlen(itypestr), 0, 0, ov, MAX_SUBSTRINGS);
154  if (ret < 1 || ret > 4) {
155  SCLogError(SC_ERR_PCRE_MATCH, "pcre_exec parse error, ret %" PRId32 ", string %s", ret, itypestr);
156  goto error;
157  }
158 
159  int i;
160  const char *str_ptr;
161  for (i = 1; i < ret; i++) {
162  res = pcre_get_substring((char *)itypestr, ov, MAX_SUBSTRINGS, i, &str_ptr);
163  if (res < 0) {
164  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
165  goto error;
166  }
167  args[i-1] = (char *)str_ptr;
168  }
169 
170  itd = SCMalloc(sizeof(DetectITypeData));
171  if (unlikely(itd == NULL))
172  goto error;
173  itd->type1 = 0;
174  itd->type2 = 0;
175  itd->mode = 0;
176 
177  /* we have either "<" or ">" */
178  if (args[0] != NULL && strlen(args[0]) != 0) {
179  /* we have a third part ("<> y"), therefore it's invalid */
180  if (args[2] != NULL) {
181  SCLogError(SC_ERR_INVALID_VALUE, "itype: invalid value");
182  goto error;
183  }
184  /* we have only a comparison ("<", ">") */
185  if (ByteExtractStringUint8(&itd->type1, 10, 0, args[1]) < 0) {
186  SCLogError(SC_ERR_INVALID_ARGUMENT, "specified icmp type %s is not "
187  "valid", args[1]);
188  goto error;
189  }
190  if ((strcmp(args[0], ">")) == 0) itd->mode = DETECT_ITYPE_GT;
191  else itd->mode = DETECT_ITYPE_LT;
192  } else { /* no "<", ">" */
193  /* we have a range ("<>") */
194  if (args[2] != NULL) {
195  itd->mode = (uint8_t) DETECT_ITYPE_RN;
196  if (ByteExtractStringUint8(&itd->type1, 10, 0, args[1]) < 0) {
197  SCLogError(SC_ERR_INVALID_ARGUMENT, "specified icmp type %s is not "
198  "valid", args[1]);
199  goto error;
200  }
201  if (ByteExtractStringUint8(&itd->type2, 10, 0, args[2]) < 0) {
202  SCLogError(SC_ERR_INVALID_ARGUMENT, "specified icmp type %s is not "
203  "valid", args[2]);
204  goto error;
205  }
206  /* we check that the first given value in the range is less than
207  the second, otherwise we swap them */
208  if (itd->type1 > itd->type2) {
209  uint8_t temp = itd->type1;
210  itd->type1 = itd->type2;
211  itd->type2 = temp;
212  }
213  } else { /* we have an equality */
214  itd->mode = DETECT_ITYPE_EQ;
215  if (ByteExtractStringUint8(&itd->type1, 10, 0, args[1]) < 0) {
216  SCLogError(SC_ERR_INVALID_ARGUMENT, "specified icmp type %s is not "
217  "valid", args[1]);
218  goto error;
219  }
220  }
221  }
222 
223  for (i = 0; i < (ret-1); i++) {
224  if (args[i] != NULL)
225  SCFree(args[i]);
226  }
227  return itd;
228 
229 error:
230  for (i = 0; i < (ret-1) && i < 3; i++) {
231  if (args[i] != NULL)
232  SCFree(args[i]);
233  }
234  if (itd != NULL)
235  DetectITypeFree(itd);
236  return NULL;
237 }

239 /**
240  * \brief this function is used to add the parsed itype data into the current signature
241  *
242  * \param de_ctx pointer to the Detection Engine Context
243  * \param s pointer to the Current Signature
244  * \param itypestr pointer to the user provided itype options
245  *
246  * \retval 0 on Success
247  * \retval -1 on Failure
248  */
249 static int DetectITypeSetup(DetectEngineCtx *de_ctx, Signature *s, char *itypestr)
250 {
251 
252  DetectITypeData *itd = NULL;
253  SigMatch *sm = NULL;
254 
255  itd = DetectITypeParse(itypestr);
256  if (itd == NULL) goto error;
257 
258  sm = SigMatchAlloc();
259  if (sm == NULL) goto error;
260 
261  sm->type = DETECT_ITYPE;
262  sm->ctx = (SigMatchCtx *)itd;
263 
264  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);
265  s->flags |= SIG_FLAG_REQUIRE_PACKET;
266 
267  return 0;
268 
269 error:
270  if (itd != NULL) DetectITypeFree(itd);
271  if (sm != NULL) SCFree(sm);
272  return -1;
273 }

423 /**
424  * \test DetectITypeMatchTest01 is a test for checking the working of itype
425  * keyword by creating 5 rules and matching a crafted packet against
426  * them. 4 out of 5 rules shall trigger.
427  */
428 int DetectITypeMatchTest01(void)
429 {
430 
431  Packet *p = NULL;
432  Signature *s = NULL;
433  ThreadVars th_v;
434  DetectEngineThreadCtx *det_ctx;
435  int result = 0;

451  goto end;

453 
454  s = s->next = SigInit(de_ctx,"alert icmp any any -> any any (itype:<15; sid:2;)");
455  if (s == NULL) {
456  goto end;
457  }
458 
459  s = s->next = SigInit(de_ctx,"alert icmp any any -> any any (itype:>20; sid:3;)");
460  if (s == NULL) {
461  goto end;
462  }
463 
464  s = s->next = SigInit(de_ctx,"alert icmp any any -> any any (itype:8<>20; sid:4;)");
465  if (s == NULL) {
466  goto end;
467  }
468 
469  s = s->next = SigInit(de_ctx,"alert icmp any any -> any any (itype:20<>8; sid:5;)");
470  if (s == NULL) {
471  goto end;
472  }
473 
474 
475  SigGroupBuild(de_ctx);
476  DetectEngineThreadCtxInit(&th_v, (void *)de_ctx, (void *)&det_ctx);
477 
478  SigMatchSignatures(&th_v, de_ctx, det_ctx, p);
479  if (PacketAlertCheck(p, 1) == 0) {

483  SCLogDebug("sid 2 did not alert, but should have");
484  goto cleanup;
485  } else if (PacketAlertCheck(p, 3)) {
486  SCLogDebug("sid 3 alerted, but should not have");
487  goto cleanup;
488  } else if (PacketAlertCheck(p, 4) == 0) {
489  SCLogDebug("sid 4 did not alert, but should have");
490  goto cleanup;

492  SCLogDebug("sid 5 did not alert, but should have");

494  }

496  result = 1;

500  SigCleanSignatures(de_ctx);
501 
502  DetectEngineThreadCtxDeinit(&th_v, (void *)det_ctx);
503  DetectEngineCtxFree(de_ctx);
504 
505  UTHFreePackets(&p, 1);
506 end:

508 }
