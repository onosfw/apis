18 /**
19  * \defgroup decode Packet decoding
20  *
21  * \brief Code in charge of protocol decoding
22  *
23  * The task of decoding packets is made in different files and
24  * as Suricata is supporting encapsulation there is a potential
25  * recursivity in the call.
26  *
27  * For each protocol a DecodePROTO function is provided. For
28  * example we have DecodeIPV4() for IPv4 and DecodePPP() for
29  * PPP.
30  *
31  * These functions have all a pkt and and a len argument which
32  * are respectively a pointer to the protocol data and the length
33  * of this protocol data.
34  *
35  * \attention The pkt parameter must point to the effective data because
36  * it will be used later to set per protocol pointer like Packet::tcph
37  *
38  * @{

69 int DecodeTunnel(ThreadVars *tv, DecodeThreadVars *dtv, Packet *p,
70  uint8_t *pkt, uint16_t len, PacketQueue *pq, enum DecodeTunnelProto proto)
71 {
72  switch (proto) {
73  case DECODE_TUNNEL_PPP:
74  return DecodePPP(tv, dtv, p, pkt, len, pq);
75  case DECODE_TUNNEL_IPV4:
76  return DecodeIPV4(tv, dtv, p, pkt, len, pq);
77  case DECODE_TUNNEL_IPV6:
78  return DecodeIPV6(tv, dtv, p, pkt, len, pq);
79  case DECODE_TUNNEL_VLAN:
80  return DecodeVLAN(tv, dtv, p, pkt, len, pq);
81  case DECODE_TUNNEL_ETHERNET:
82  return DecodeEthernet(tv, dtv, p, pkt, len, pq);
83  case DECODE_TUNNEL_ERSPAN:
84  return DecodeERSPAN(tv, dtv, p, pkt, len, pq);
85  default:
86  SCLogInfo("FIXME: DecodeTunnel: protocol %" PRIu32 " not supported.", proto);
87  break;
88  }
89  return TM_ECODE_OK;
90 }

190 /**
191  * \brief Copy data to Packet payload at given offset
192  *
193  * This function copies data/payload to a Packet. It uses the
194  * space allocated at Packet creation (pointed by Packet::pkt)
195  * or allocate some memory (pointed by Packet::ext_pkt) if the
196  * data size is to big to fit in initial space (of size
197  * default_packet_size).
198  *
199  * \param Pointer to the Packet to modify
200  * \param Offset of the copy relatively to payload of Packet
201  * \param Pointer to the data to copy
202  * \param Length of the data to copy
203  */
204 inline int PacketCopyDataOffset(Packet *p, int offset, uint8_t *data, int datalen)
205 {
206  if (unlikely(offset + datalen > MAX_PAYLOAD_SIZE)) {
207  /* too big */
208  return -1;
209  }
210 
211  /* Do we have already an packet with allocated data */
212  if (! p->ext_pkt) {
213  if (offset + datalen <= (int)default_packet_size) {
214  /* data will fit in memory allocated with packet */
215  memcpy(GET_PKT_DIRECT_DATA(p) + offset, data, datalen);
216  } else {
217  /* here we need a dynamic allocation */
218  p->ext_pkt = SCMalloc(MAX_PAYLOAD_SIZE);
219  if (unlikely(p->ext_pkt == NULL)) {
220  SET_PKT_LEN(p, 0);
221  return -1;
222  }
223  /* copy initial data */
224  memcpy(p->ext_pkt, GET_PKT_DIRECT_DATA(p), GET_PKT_DIRECT_MAX_SIZE(p));
225  /* copy data as asked */
226  memcpy(p->ext_pkt + offset, data, datalen);
227  }
228  } else {
229  memcpy(p->ext_pkt + offset, data, datalen);
230  }
231  return 0;
232 }

247 /**
248  * \brief Setup a pseudo packet (tunnel)
249  *
250  * \param parent parent packet for this pseudo pkt
251  * \param pkt raw packet data
252  * \param len packet data length
253  * \param proto protocol of the tunneled packet
254  *
255  * \retval p the pseudo packet or NULL if out of memory
256  */
257 Packet *PacketTunnelPktSetup(ThreadVars *tv, DecodeThreadVars *dtv, Packet *parent,
258  uint8_t *pkt, uint16_t len, enum DecodeTunnelProto proto,
259  PacketQueue *pq)
260 {
261  int ret;
262 
263  SCEnter();
264 
265  /* get us a packet */
266  Packet *p = PacketGetFromQueueOrAlloc();
267  if (unlikely(p == NULL)) {
268  SCReturnPtr(NULL, "Packet");
269  }
270 
271  /* copy packet and set lenght, proto */
272  PacketCopyData(p, pkt, len);
273  p->recursion_level = parent->recursion_level + 1;
274  p->ts.tv_sec = parent->ts.tv_sec;
275  p->ts.tv_usec = parent->ts.tv_usec;
276  p->datalink = DLT_RAW;
277  p->tenant_id = parent->tenant_id;
278 
279  /* set the root ptr to the lowest layer */
280  if (parent->root != NULL)
281  p->root = parent->root;
282  else
283  p->root = parent;
284 
285  /* tell new packet it's part of a tunnel */
286  SET_TUNNEL_PKT(p);
287 
288  ret = DecodeTunnel(tv, dtv, p, GET_PKT_DATA(p),
289  GET_PKT_LEN(p), pq, proto);
290 
291  if (unlikely(ret != TM_ECODE_OK)) {
292  /* Not a tunnel packet, just a pseudo packet */
293  p->root = NULL;
294  UNSET_TUNNEL_PKT(p);
295  TmqhOutputPacketpool(tv, p);
296  SCReturnPtr(NULL, "Packet");
297  }
298 
299 
300  /* tell parent packet it's part of a tunnel */
301  SET_TUNNEL_PKT(parent);
302 
303  /* increment tunnel packet refcnt in the root packet */
304  TUNNEL_INCR_PKT_TPR(p);
305 
306  /* disable payload (not packet) inspection on the parent, as the payload
307  * is the packet we will now run through the system separately. We do
308  * check it against the ip/port/other header checks though */
309  DecodeSetNoPayloadInspectionFlag(parent);
310  SCReturnPtr(p, "Packet");
311 }

314  * \brief Setup a pseudo packet (reassembled frags)
315  *
316  * Difference with PacketPseudoPktSetup is that this func doesn't increment
317  * the recursion level. It needs to be on the same level as the frags because
318  * we run the flow engine against this and we need to get the same flow.
319  *
320  * \param parent parent packet for this pseudo pkt

322  * \param len packet data length
323  * \param proto protocol of the tunneled packet
324  *
325  * \retval p the pseudo packet or NULL if out of memory
326  */
327 Packet *PacketDefragPktSetup(Packet *parent, uint8_t *pkt, uint16_t len, uint8_t proto)
328 {

330 
331  /* get us a packet */
332  Packet *p = PacketGetFromQueueOrAlloc();
333  if (unlikely(p == NULL)) {
334  SCReturnPtr(NULL, "Packet");
335  }
336 
337  /* set the root ptr to the lowest layer */
338  if (parent->root != NULL)
339  p->root = parent->root;
340  else
341  p->root = parent;
342 
343  /* copy packet and set lenght, proto */
