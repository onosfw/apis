60 static int DnsGetDnsRrname(lua_State *luastate)
61 {
62  if (!(LuaStateNeedProto(luastate, ALPROTO_DNS)))
63  return LuaCallbackError(luastate, "error: protocol not dns");
64 
65  DNSTransaction *tx = LuaStateGetTX(luastate);
66  if (tx == NULL)
67  return LuaCallbackError(luastate, "internal error: no tx");
68 
69  DNSQueryEntry *query = NULL;
70  TAILQ_FOREACH(query, &tx->query_list, next) {
71  char *c;
72  size_t input_len;
73  c = BytesToString((uint8_t *)((uint8_t *)query + sizeof(DNSQueryEntry)), query->len);
74  if (c != NULL) {
75  int ret;
76  input_len = strlen(c);
77  /* sanity check */
78  if (input_len > (size_t)(2 * query->len)) {
79  SCFree(c);
80  return LuaCallbackError(luastate, "invalid length");
81  }
82  ret = LuaPushStringBuffer(luastate, (uint8_t *)c, input_len);
83  SCFree(c);
84  return ret;
85  }
86  }
87 
88  return LuaCallbackError(luastate, "no query");
89 }

183 {

185  return LuaCallbackError(luastate, "error: protocol not dns");
186 
187  DNSTransaction *tx = LuaStateGetTX(luastate);

189  return LuaCallbackError(luastate, "internal error: no tx");
190 
191  uint32_t u = 0;
192  lua_newtable(luastate);
193  DNSAnswerEntry *answer = NULL;
194  TAILQ_FOREACH(answer, &tx->answer_list, next) {
195  lua_pushinteger(luastate, u++);
196 
197  lua_newtable(luastate);
198  char record[16] = "";
199  DNSCreateTypeString(answer->type, record, sizeof(record));
200  lua_pushstring(luastate, "type");
201  lua_pushstring(luastate, record);
202  lua_settable(luastate, -3);
203 
204  lua_pushstring(luastate, "ttl");
205  lua_pushinteger(luastate, answer->ttl);
206  lua_settable(luastate, -3);
207 
208  {
209  uint8_t *ptr = (uint8_t *)((uint8_t *)answer + sizeof(DNSAnswerEntry));
210  lua_pushstring(luastate, "rrname");
211  LuaPushStringBuffer(luastate, ptr, answer->fqdn_len);
212  lua_settable(luastate, -3);
213 
214  ptr = (uint8_t *)((uint8_t *)answer + sizeof(DNSAnswerEntry) + answer->fqdn_len);
215  if (answer->type == DNS_RECORD_TYPE_A) {
216  char a[16] = "";
217  PrintInet(AF_INET, (const void *)ptr, a, sizeof(a));
218  lua_pushstring(luastate, "addr");
219  LuaPushStringBuffer(luastate, (uint8_t *)a, strlen(a));
220  lua_settable(luastate, -3);
221  } else if (answer->type == DNS_RECORD_TYPE_AAAA) {
222  char a[46];
223  PrintInet(AF_INET6, (const void *)ptr, a, sizeof(a));
224  lua_pushstring(luastate, "addr");
225  LuaPushStringBuffer(luastate, (uint8_t *)a, strlen(a));
226  lua_settable(luastate, -3);
227  } else if (answer->data_len == 0) {
228  /* not setting 'addr' */
229  } else {
230  lua_pushstring(luastate, "addr");
231  LuaPushStringBuffer(luastate, (uint8_t *)ptr, answer->data_len);
232  lua_settable(luastate, -3);
233  }
234  }
235 
236  lua_settable(luastate, -3);
237  }
238 
239  return 1;
240 }
