400 static void AppendAppInspectionEngine(DetectEngineAppInspectionEngine *engine,
401  DetectEngineAppInspectionEngine *list[][ALPROTO_MAX][2])
402 {
403  /* append to the list */
404  DetectEngineAppInspectionEngine *tmp = list[FlowGetProtoMapping(engine->ipproto)][engine->alproto][engine->dir];
405  DetectEngineAppInspectionEngine *insert = NULL;
406  while (tmp != NULL) {
407  if (tmp->dir == engine->dir &&
408  (tmp->sm_list == engine->sm_list ||
409  tmp->inspect_flags == engine->inspect_flags
410  )) {
411  SCLogError(SC_ERR_DETECT_PREPARE, "App Inspection Engine already "
412  "registered for this direction(%"PRIu16") ||"
413  "sm_list(%d) || "
414  "[inspect(%"PRIu32")]_flags",
415  tmp->dir, tmp->sm_list, tmp->inspect_flags);
416  exit(EXIT_FAILURE);
417  }
418  insert = tmp;
419  tmp = tmp->next;
420  }
421  if (insert == NULL)
422  list[FlowGetProtoMapping(engine->ipproto)][engine->alproto][engine->dir] = engine;
423  else
424  insert->next = engine;
425 
426  return;
427 }

448  SCLogError(SC_ERR_INVALID_ARGUMENTS, "Invalid arguments");

452  DetectEngineAppInspectionEngine *tmp = list[FlowGetProtoMapping(ipproto)][alproto][dir];
453  while (tmp != NULL) {

455  return;
456  }
457  tmp = tmp->next;
458  }

465  new_engine->ipproto = ipproto;
466  new_engine->alproto = alproto;

540 /** \internal
541  * \brief Update detect threads with new detect engine
542  *
543  * Atomically update each detect thread with a new thread context
544  * that is associated to the new detection engine(s).
545  *
546  * If called in unix socket mode, it's possible that we don't have
547  * detect threads yet.
548  *
549  * \retval -1 error
550  * \retval 0 no detection threads
551  * \retval 1 successful reload
552  */
553 static int DetectEngineReloadThreads(DetectEngineCtx *new_de_ctx)
554 {
555  SCEnter();
556 
557  int i = 0;
558  int no_of_detect_tvs = 0;
559  ThreadVars *tv = NULL;
560 
561  /* count detect threads in use */
562  SCMutexLock(&tv_root_lock);
563  tv = tv_root[TVT_PPT];
564  while (tv) {
565  /* obtain the slots for this TV */
566  TmSlot *slots = tv->tm_slots;
567  while (slots != NULL) {
568  TmModule *tm = TmModuleGetById(slots->tm_id);
569 
570  if (suricata_ctl_flags != 0) {
571  SCLogInfo("rule reload interupted by engine shutdown");
572  SCMutexUnlock(&tv_root_lock);
573  return -1;
574  }
575 
576  if (!(tm->flags & TM_FLAG_DETECT_TM)) {
577  slots = slots->slot_next;
578  continue;
579  }
580  no_of_detect_tvs++;
581  break;
582  }
583 
584  tv = tv->next;
585  }
586  SCMutexUnlock(&tv_root_lock);
587 
588  /* can be zero in unix socket mode */
589  if (no_of_detect_tvs == 0) {
590  return 0;
591  }
592 
593  SCLogNotice("rule reload starting");
594 
595  /* prepare swap structures */
596  DetectEngineThreadCtx *old_det_ctx[no_of_detect_tvs];
597  DetectEngineThreadCtx *new_det_ctx[no_of_detect_tvs];
598  ThreadVars *detect_tvs[no_of_detect_tvs];
599  memset(old_det_ctx, 0x00, (no_of_detect_tvs * sizeof(DetectEngineThreadCtx *)));
600  memset(new_det_ctx, 0x00, (no_of_detect_tvs * sizeof(DetectEngineThreadCtx *)));
601  memset(detect_tvs, 0x00, (no_of_detect_tvs * sizeof(ThreadVars *)));
602 
603  /* start the process of swapping detect threads ctxs */
604 
605  /* get reference to tv's and setup new_det_ctx array */
606  SCMutexLock(&tv_root_lock);
607  tv = tv_root[TVT_PPT];
608  while (tv) {
609  /* obtain the slots for this TV */
610  TmSlot *slots = tv->tm_slots;
611  while (slots != NULL) {
612  TmModule *tm = TmModuleGetById(slots->tm_id);
613 
614  if (suricata_ctl_flags != 0) {
615  SCMutexUnlock(&tv_root_lock);
616  goto error;
617  }
618 
619  if (!(tm->flags & TM_FLAG_DETECT_TM)) {
620  slots = slots->slot_next;
621  continue;
622  }
623 
624  old_det_ctx[i] = SC_ATOMIC_GET(slots->slot_data);
625  detect_tvs[i] = tv;
626  if (new_de_ctx != NULL)
627  new_det_ctx[i] = DetectEngineThreadCtxInitForReload(tv, new_de_ctx);
628  else
629  new_det_ctx[i] = DetectEngineThreadCtxInitForMT(tv);
630  if (new_det_ctx[i] == NULL) {
631  SCLogError(SC_ERR_LIVE_RULE_SWAP, "Detect engine thread init "
632  "failure in live rule swap. Let's get out of here");
633  SCMutexUnlock(&tv_root_lock);
634  goto error;
635  }
636  SCLogDebug("live rule swap created new det_ctx - %p and de_ctx "
637  "- %p\n", new_det_ctx[i], new_de_ctx);
638  i++;
639  break;
640  }
641 
642  tv = tv->next;
643  }
644  BUG_ON(i != no_of_detect_tvs);
645 
646  /* atomicly replace the det_ctx data */
647  i = 0;
648  tv = tv_root[TVT_PPT];
649  while (tv) {
650  /* find the correct slot */
651  TmSlot *slots = tv->tm_slots;
652  while (slots != NULL) {
653  if (suricata_ctl_flags != 0) {
654  return -1;
655  }
656 
657  TmModule *tm = TmModuleGetById(slots->tm_id);
658  if (!(tm->flags & TM_FLAG_DETECT_TM)) {
659  slots = slots->slot_next;
660  continue;
661  }
662  SCLogDebug("swapping new det_ctx - %p with older one - %p",
663  new_det_ctx[i], SC_ATOMIC_GET(slots->slot_data));
664  (void)SC_ATOMIC_SET(slots->slot_data, new_det_ctx[i++]);
665  break;
666  }
667  tv = tv->next;
668  }
669  SCMutexUnlock(&tv_root_lock);

702  }

704  }
705  if (break_out)

707  SCLogDebug("new_det_ctx - %p used by detect engine", new_det_ctx[i]);
708  }
709 
710  /* this is to make sure that if someone initiated shutdown during a live
711  * rule swap, the live rule swap won't clean up the old det_ctx and
712  * de_ctx, till all detect threads have stopped working and sitting
713  * silently after setting RUNNING_DONE flag and while waiting for
714  * THV_DEINIT flag */
715  if (i != no_of_detect_tvs) { // not all threads we swapped

723  slots = slots->slot_next;
724  continue;
725  }
726

728  usleep(100);
729  }
730 
731  slots = slots->slot_next;
732  }
733 
734  tv = tv->next;
735  }
736  }
737 
738  /* free all the ctxs */
739  for (i = 0; i < no_of_detect_tvs; i++) {
740  SCLogDebug("Freeing old_det_ctx - %p used by detect",
741  old_det_ctx[i]);
742  DetectEngineThreadCtxDeinit(NULL, old_det_ctx[i]);
743  }
744 
745  SRepReloadComplete();
746 
747  SCLogNotice("rule reload complete");
748  return 1;
749 
750  error:
751  for (i = 0; i < no_of_detect_tvs; i++) {
752  if (new_det_ctx[i] != NULL)
753  DetectEngineThreadCtxDeinit(NULL, new_det_ctx[i]);
754  }
755  return -1;
