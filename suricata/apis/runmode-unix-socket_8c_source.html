149 /**
150  * \brief Add file to file queue
151  *
152  * \param this a UnixCommand:: structure
153  * \param filename absolute filename
154  * \param output_dir absolute name of directory where log will be put
155  *
156  * \retval 0 in case of error, 1 in case of success
157  */
158 static TmEcode UnixListAddFile(PcapCommand *this,
159  const char *filename, const char *output_dir, int tenant_id)
160 {
161  PcapFiles *cfile = NULL;
162  if (filename == NULL || this == NULL)
163  return TM_ECODE_FAILED;
164  cfile = SCMalloc(sizeof(PcapFiles));
165  if (unlikely(cfile == NULL)) {
166  SCLogError(SC_ERR_MEM_ALLOC, "Unable to allocate new file");
167  return TM_ECODE_FAILED;
168  }
169  memset(cfile, 0, sizeof(PcapFiles));
170 
171  cfile->filename = SCStrdup(filename);
172  if (unlikely(cfile->filename == NULL)) {
173  SCFree(cfile);
174  SCLogError(SC_ERR_MEM_ALLOC, "Unable to dup filename");
175  return TM_ECODE_FAILED;
176  }
177 
178  if (output_dir) {
179  cfile->output_dir = SCStrdup(output_dir);
180  if (unlikely(cfile->output_dir == NULL)) {
181  SCFree(cfile->filename);
182  SCFree(cfile);
183  SCLogError(SC_ERR_MEM_ALLOC, "Unable to dup output_dir");
184  return TM_ECODE_FAILED;
185  }
186  }
187 
188  cfile->tenant_id = tenant_id;
189 
190  TAILQ_INSERT_TAIL(&this->files, cfile, next);
191  return TM_ECODE_OK;
192 }

194 /**
195  * \brief Command to add a file to treatment list
196  *
197  * \param cmd the content of command Arguments as a json_t object
198  * \param answer the json_t object that has to be used to answer
199  * \param data pointer to data defining the context here a PcapCommand::
200  */
201 TmEcode UnixSocketAddPcapFile(json_t *cmd, json_t* answer, void *data)
202 {
203  PcapCommand *this = (PcapCommand *) data;
204  int ret;
205  const char *filename;
206  const char *output_dir;
207  int tenant_id = 0;
208 #ifdef OS_WIN32
209  struct _stat st;
210 #else
211  struct stat st;
212 #endif /* OS_WIN32 */
213 
214  json_t *jarg = json_object_get(cmd, "filename");
215  if(!json_is_string(jarg)) {
216  SCLogInfo("error: command is not a string");
217  json_object_set_new(answer, "message", json_string("command is not a string"));
218  return TM_ECODE_FAILED;
219  }
220  filename = json_string_value(jarg);
221 #ifdef OS_WIN32
222  if(_stat(filename, &st) != 0) {
223 #else
224  if(stat(filename, &st) != 0) {
225 #endif /* OS_WIN32 */
226  json_object_set_new(answer, "message", json_string("File does not exist"));
227  return TM_ECODE_FAILED;
228  }
229 
230  json_t *oarg = json_object_get(cmd, "output-dir");
231  if (oarg != NULL) {
232  if(!json_is_string(oarg)) {
233  SCLogInfo("error: output dir is not a string");
234  json_object_set_new(answer, "message", json_string("output dir is not a string"));
235  return TM_ECODE_FAILED;
236  }
237  output_dir = json_string_value(oarg);
238  } else {
239  SCLogInfo("error: can't get output-dir");
240  json_object_set_new(answer, "message", json_string("output dir param is mandatory"));
241  return TM_ECODE_FAILED;
242  }
243 
244 #ifdef OS_WIN32
245  if(_stat(output_dir, &st) != 0) {
246 #else
247  if(stat(output_dir, &st) != 0) {
248 #endif /* OS_WIN32 */
249  json_object_set_new(answer, "message", json_string("Output directory does not exist"));
250  return TM_ECODE_FAILED;
251  }
252 
253  json_t *targ = json_object_get(cmd, "tenant");
254  if (targ != NULL) {
255  if(!json_is_number(targ)) {
256  json_object_set_new(answer, "message", json_string("tenant is not a number"));
257  return TM_ECODE_FAILED;
258  }
259  tenant_id = json_number_value(targ);
260  }
261 
262  ret = UnixListAddFile(this, filename, output_dir, tenant_id);
263  switch(ret) {
264  case TM_ECODE_FAILED:
265  json_object_set_new(answer, "message", json_string("Unable to add file to list"));
266  return TM_ECODE_FAILED;
267  case TM_ECODE_OK:
268  SCLogInfo("Added file '%s' to list", filename);
269  json_object_set_new(answer, "message", json_string("Successfully added file to list"));
270  return TM_ECODE_OK;
271  }
272  return TM_ECODE_OK;
273 }

275 /**
276  * \brief Handle the file queue
277  *
278  * This function check if there is currently a file
279  * being parse. If it is not the case, it will start to
280  * work on a new file. This implies to start a new 'pcap-file'
281  * running mode after having set the file and the output dir.
282  * This function also handles the cleaning of the previous
283  * running mode.
284  *
285  * \param this a UnixCommand:: structure
286  * \retval 0 in case of error, 1 in case of success
287  */
288 TmEcode UnixSocketPcapFilesCheck(void *data)
289 {
290  PcapCommand *this = (PcapCommand *) data;
291  if (unix_manager_file_task_running == 1) {
292  return TM_ECODE_OK;
293  }
294  if ((unix_manager_file_task_failed == 1) || (this->running == 1)) {
295  if (unix_manager_file_task_failed) {
296  SCLogInfo("Preceeding task failed, cleaning the running mode");
297  }
298  unix_manager_file_task_failed = 0;
299  this->running = 0;
300  if (this->currentfile) {
301  SCFree(this->currentfile);
302  }
303  this->currentfile = NULL;
304 
305  /* needed by FlowForceReassembly */
306  PacketPoolInit();
307 
308  /* handle graceful shutdown of the flow engine, it's helper
309  * threads and the packet threads */
310  FlowDisableFlowManagerThread();
311  TmThreadDisableReceiveThreads();
312  FlowForceReassembly();
313  TmThreadDisablePacketThreads();
314  FlowDisableFlowRecyclerThread();
315 
316  /* kill the stats threads */
317  TmThreadKillThreadsFamily(TVT_MGMT);
318  TmThreadClearThreadsFamily(TVT_MGMT);
319 
320  /* kill packet threads -- already in 'disabled' state */
321  TmThreadKillThreadsFamily(TVT_PPT);
322  TmThreadClearThreadsFamily(TVT_PPT);
323 
324  PacketPoolDestroy();
325 
326  /* mgt and ppt threads killed, we can run non thread-safe
327  * shutdown functions */
328  StatsReleaseResources();
329  RunModeShutDown();
330  FlowShutdown();
331  IPPairShutdown();
332  HostCleanup();
333  StreamTcpFreeConfig(STREAM_VERBOSE);
334  DefragDestroy();
335  TmqResetQueues();
336 #ifdef PROFILING
337  if (profiling_rules_enabled)
338  SCProfilingDump();
339  SCProfilingDestroy();
340 #endif
341  }
342  if (!TAILQ_EMPTY(&this->files)) {
343  PcapFiles *cfile = TAILQ_FIRST(&this->files);
344  TAILQ_REMOVE(&this->files, cfile, next);
345  SCLogInfo("Starting run for '%s'", cfile->filename);
346  unix_manager_file_task_running = 1;
347  this->running = 1;
348  if (ConfSet("pcap-file.file", cfile->filename) != 1) {
349  SCLogInfo("Can not set working file to '%s'", cfile->filename);
350  PcapFilesFree(cfile);
351  return TM_ECODE_FAILED;
352  }
353  if (cfile->output_dir) {
354  if (ConfSet("default-log-dir", cfile->output_dir) != 1) {
355  SCLogInfo("Can not set output dir to '%s'", cfile->output_dir);
356  PcapFilesFree(cfile);
357  return TM_ECODE_FAILED;
358  }
359  }
360  if (cfile->tenant_id > 0) {
361  char tstr[16] = "";
362  snprintf(tstr, sizeof(tstr), "%d", cfile->tenant_id);
363  if (ConfSet("pcap-file.tenant-id", tstr) != 1) {
364  SCLogInfo("Can not set working tenant-id to '%s'", tstr);
365  PcapFilesFree(cfile);
366  return TM_ECODE_FAILED;
367  }
368  } else {
369  SCLogInfo("pcap-file.tenant-id not set");
370  }
371  this->currentfile = SCStrdup(cfile->filename);
372  if (unlikely(this->currentfile == NULL)) {
373  SCLogError(SC_ERR_MEM_ALLOC, "Failed file name allocation");
374  return TM_ECODE_FAILED;
375  }
376  PcapFilesFree(cfile);
377  StatsInit();
378 #ifdef PROFILING
379  SCProfilingRulesGlobalInit();
380  SCProfilingKeywordsGlobalInit();
381  SCProfilingInit();
382 #endif /* PROFILING */
383  DefragInit();
384  FlowInitConfig(FLOW_QUIET);
385  IPPairInitConfig(FLOW_QUIET);
386  StreamTcpInitConfig(STREAM_VERBOSE);
387  AppLayerRegisterGlobalCounters();
388  RunModeInitializeOutputs();
389  StatsSetupPostConfig();
390  RunModeDispatch(RUNMODE_PCAP_FILE, NULL);
391  FlowManagerThreadSpawn();
392  FlowRecyclerThreadSpawn();
393  StatsSpawnThreads();
394  /* Un-pause all the paused threads */
395  TmThreadContinueThreads();
396  }
397  return TM_ECODE_OK;
398 }
399 #endif

504  return TM_ECODE_FAILED;
505  }

507  json_object_set_new(answer, "message", json_string("handler added"));
508  return TM_ECODE_OK;

681 #else

706 #ifdef OS_WIN32

710 #endif /* OS_WIN32 */

719  SCLogInfo("prefix %s", prefix);

737  // TODO cleanup
738  return TM_ECODE_FAILED;
739  }
740 
741  json_object_set_new(answer, "message", json_string("reloading tenant succeeded"));
742  return TM_ECODE_OK;
