228 /** \internal
229  * \brief fill lua stack with header info
230  * \param luastate the lua state
231  * \param p packet
232  * \retval cnt number of data items placed on the stack
233  *
234  * Places: ipver (number), src ip (string), dst ip (string), protocol (number),
235  * sp or icmp type (number), dp or icmp code (number).
236  */
237 static int LuaCallbackTuplePushToStackFromPacket(lua_State *luastate, const Packet *p)
238 {
239  int ipver = 0;
240  if (PKT_IS_IPV4(p)) {
241  ipver = 4;
242  } else if (PKT_IS_IPV6(p)) {
243  ipver = 6;
244  }
245  lua_pushnumber (luastate, ipver);
246  if (ipver == 0)
247  return 1;
248 
249  char srcip[46] = "", dstip[46] = "";
250  if (PKT_IS_IPV4(p)) {
251  PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p), srcip, sizeof(srcip));
252  PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p), dstip, sizeof(dstip));
253  } else if (PKT_IS_IPV6(p)) {
254  PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p), srcip, sizeof(srcip));
255  PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p), dstip, sizeof(dstip));
256  }
257 
258  lua_pushstring (luastate, srcip);
259  lua_pushstring (luastate, dstip);
260 
261  /* proto and ports (or type/code) */
262  lua_pushnumber (luastate, p->proto);
263  if (p->proto == IPPROTO_TCP || p->proto == IPPROTO_UDP) {
264  lua_pushnumber (luastate, p->sp);
265  lua_pushnumber (luastate, p->dp);
266 
267  } else if (p->proto == IPPROTO_ICMP || p->proto == IPPROTO_ICMPV6) {
268  lua_pushnumber (luastate, p->type);
269  lua_pushnumber (luastate, p->code);
270  } else {
271  lua_pushnumber (luastate, 0);
272  lua_pushnumber (luastate, 0);
273  }
274 
275  return 6;
276 }
