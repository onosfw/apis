45 /* get the flowbit with idx from the flow */
46 static FlowBit *FlowBitGet(Flow *f, uint16_t idx)
47 {
48  GenericVar *gv = f->flowvar;
49  for ( ; gv != NULL; gv = gv->next) {
50  if (gv->type == DETECT_FLOWBITS && gv->idx == idx) {
51  return (FlowBit *)gv;
52  }
53  }
54 
55  return NULL;
56 }

58 /* add a flowbit to the flow */
59 static void FlowBitAdd(Flow *f, uint16_t idx)
60 {
61  FlowBit *fb = FlowBitGet(f, idx);
62  if (fb == NULL) {
63  fb = SCMalloc(sizeof(FlowBit));
64  if (unlikely(fb == NULL))
65  return;
66 
67  fb->type = DETECT_FLOWBITS;
68  fb->idx = idx;
69  fb->next = NULL;
70  GenericVarAppend(&f->flowvar, (GenericVar *)fb);
71 
72  //printf("FlowBitAdd: adding flowbit with idx %" PRIu32 "\n", idx);
73 #ifdef FLOWBITS_STATS
74  SCMutexLock(&flowbits_mutex);
75  flowbits_added++;
76  flowbits_memuse += sizeof(FlowBit);
77  if (flowbits_memuse > flowbits_memuse_max)
78  flowbits_memuse_max = flowbits_memuse;
79  SCMutexUnlock(&flowbits_mutex);
80 #endif /* FLOWBITS_STATS */
81  }
82 }

84 static void FlowBitRemove(Flow *f, uint16_t idx)
85 {
86  FlowBit *fb = FlowBitGet(f, idx);
87  if (fb == NULL)
88  return;
89 
90  GenericVarRemove(&f->flowvar, (GenericVar *)fb);
91 
92  //printf("FlowBitRemove: remove flowbit with idx %" PRIu32 "\n", idx);
93 #ifdef FLOWBITS_STATS
94  SCMutexLock(&flowbits_mutex);
95  flowbits_removed++;
96  if (flowbits_memuse >= sizeof(FlowBit))
97  flowbits_memuse -= sizeof(FlowBit);
98  else {
99  printf("ERROR: flowbits memory usage going below 0!\n");
100  flowbits_memuse = 0;
101  }
102  SCMutexUnlock(&flowbits_mutex);
103 #endif /* FLOWBITS_STATS */
104 }

107 {

136 void FlowBitToggleNoLock(Flow *f, uint16_t idx)
137 {
138  FlowBit *fb = FlowBitGet(f, idx);
139  if (fb != NULL) {

153 int FlowBitIsset(Flow *f, uint16_t idx)
154 {
155  int r = 0;
156  FLOWLOCK_RDLOCK(f);
157 
158  FlowBit *fb = FlowBitGet(f, idx);
159  if (fb != NULL) {
160  r = 1;
161  }
162 
163  FLOWLOCK_UNLOCK(f);
164  return r;
