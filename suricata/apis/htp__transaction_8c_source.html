358 static htp_status_t htp_tx_process_request_headers(htp_tx_t *tx) {
359  if (tx == NULL) return HTP_ERROR;
360 
361  // Determine if we have a request body, and how it is packaged.
362 
363  htp_status_t rc = HTP_OK;
364 
365  htp_header_t *cl = htp_table_get_c(tx->request_headers, "content-length");
366  htp_header_t *te = htp_table_get_c(tx->request_headers, "transfer-encoding");
367 
368  // Check for the Transfer-Encoding header, which would indicate a chunked request body.
369  if (te != NULL) {
370  // Make sure it contains "chunked" only.
371  // TODO The HTTP/1.1 RFC also allows the T-E header to contain "identity", which
372  // presumably should have the same effect as T-E header absence. However, Apache
373  // (2.2.22 on Ubuntu 12.04 LTS) instead errors out with "Unknown Transfer-Encoding: identity".
374  // And it behaves strangely, too, sending a 501 and proceeding to process the request
375  // (e.g., PHP is run), but without the body. It then closes the connection.
376  if (bstr_cmp_c_nocase(te->value, "chunked") != 0) {
377  // Invalid T-E header value.
378  tx->request_transfer_coding = HTP_CODING_INVALID;
379  tx->flags |= HTP_REQUEST_INVALID_T_E;
380  tx->flags |= HTP_REQUEST_INVALID;
381  } else {
382  // Chunked encoding is a HTTP/1.1 feature, so check that an earlier protocol
383  // version is not used. The flag will also be set if the protocol could not be parsed.
384  //
385  // TODO IIS 7.0, for example, would ignore the T-E header when it
386  // it is used with a protocol below HTTP 1.1. This should be a
387  // personality trait.
388  if (tx->request_protocol_number < HTP_PROTOCOL_1_1) {
389  tx->flags |= HTP_REQUEST_INVALID_T_E;
390  tx->flags |= HTP_REQUEST_SMUGGLING;
391  }
392 
393  // If the T-E header is present we are going to use it.
394  tx->request_transfer_coding = HTP_CODING_CHUNKED;
395 
396  // We are still going to check for the presence of C-L.
397  if (cl != NULL) {
398  // According to the HTTP/1.1 RFC (section 4.4):
399  //
400  // "The Content-Length header field MUST NOT be sent
401  // if these two lengths are different (i.e., if a Transfer-Encoding
402  // header field is present). If a message is received with both a
403  // Transfer-Encoding header field and a Content-Length header field,
404  // the latter MUST be ignored."
405  //
406  tx->flags |= HTP_REQUEST_SMUGGLING;
407  }
408  }
409  } else if (cl != NULL) {
410  // Check for a folded C-L header.
411  if (cl->flags & HTP_FIELD_FOLDED) {
412  tx->flags |= HTP_REQUEST_SMUGGLING;
413  }
414 
415  // Check for multiple C-L headers.
416  if (cl->flags & HTP_FIELD_REPEATED) {
417  tx->flags |= HTP_REQUEST_SMUGGLING;
418  // TODO Personality trait to determine which C-L header to parse.
419  // At the moment we're parsing the combination of all instances,
420  // which is bound to fail (because it will contain commas).
421  }
422 
423  // Get the body length.
424  tx->request_content_length = htp_parse_content_length(cl->value);
425  if (tx->request_content_length < 0) {
426  tx->request_transfer_coding = HTP_CODING_INVALID;
427  tx->flags |= HTP_REQUEST_INVALID_C_L;
428  tx->flags |= HTP_REQUEST_INVALID;
429  } else {
430  // We have a request body of known length.
431  tx->request_transfer_coding = HTP_CODING_IDENTITY;
432  }
433  } else {
434  // No body.
435  tx->request_transfer_coding = HTP_CODING_NO_BODY;
436  }
437 
438  // If we could not determine the correct body handling,
439  // consider the request invalid.
440  if (tx->request_transfer_coding == HTP_CODING_UNKNOWN) {
441  tx->request_transfer_coding = HTP_CODING_INVALID;
442  tx->flags |= HTP_REQUEST_INVALID;
443  }
444 
445  // Check for PUT requests, which we need to treat as file uploads.
446  if (tx->request_method_number == HTP_M_PUT) {
447  if (htp_tx_req_has_body(tx)) {
448  // Prepare to treat PUT request body as a file.
449 
450  tx->connp->put_file = calloc(1, sizeof (htp_file_t));
451  if (tx->connp->put_file == NULL) return HTP_ERROR;
452 
453  tx->connp->put_file->fd = -1;
454  tx->connp->put_file->source = HTP_FILE_PUT;
455  } else {
456  // TODO Warn about PUT request without a body.
457  }
458  }
459 
460  // Determine hostname.
461 
462  // Use the hostname from the URI, when available.
463  if (tx->parsed_uri->hostname != NULL) {
464  tx->request_hostname = bstr_dup(tx->parsed_uri->hostname);
465  if (tx->request_hostname == NULL) return HTP_ERROR;
466  }
467 
468  tx->request_port_number = tx->parsed_uri->port_number;
469 
470  // Examine the Host header.
471 
472  htp_header_t *h = htp_table_get_c(tx->request_headers, "host");
473  if (h == NULL) {
474  // No host information in the headers.
475 
476  // HTTP/1.1 requires host information in the headers.
477  if (tx->request_protocol_number >= HTP_PROTOCOL_1_1) {
478  tx->flags |= HTP_HOST_MISSING;
479  }
480  } else {
481  // Host information available in the headers.
482 
483  bstr *hostname;
484  int port;
485 
486  rc = htp_parse_header_hostport(h->value, &hostname, NULL, &port, &(tx->flags));
487  if (rc != HTP_OK) return rc;
488 
489  if (hostname != NULL) {
490  // The host information in the headers is valid.
491 
492  // Is there host information in the URI?
493  if (tx->request_hostname == NULL) {
494  // There is no host information in the URI. Place the
495  // hostname from the headers into the parsed_uri structure.
496  tx->request_hostname = hostname;
497  tx->request_port_number = port;
498  } else {
499  // The host information appears in the URI and in the headers. The
500  // HTTP RFC states that we should ignore the header copy.
501 
502  // Check for different hostnames.
503  if (bstr_cmp_nocase(hostname, tx->request_hostname) != 0) {
504  tx->flags |= HTP_HOST_AMBIGUOUS;
505  }
506 
507  // Check for different ports.
508  if (((tx->request_port_number != -1)&&(port != -1))&&(tx->request_port_number != port)) {
509  tx->flags |= HTP_HOST_AMBIGUOUS;
510  }
511 
512  bstr_free(hostname);
513  }
514  } else {
515  // Invalid host information in the headers.
516 
517  if (tx->request_hostname != NULL) {
518  // Raise the flag, even though the host information in the headers is invalid.
519  tx->flags |= HTP_HOST_AMBIGUOUS;
520  }
521  }
522  }
523 
524  // Determine Content-Type.
525  htp_header_t *ct = htp_table_get_c(tx->request_headers, "content-type");
526  if (ct != NULL) {
527  rc = htp_parse_ct_header(ct->value, &tx->request_content_type);
528  if (rc != HTP_OK) return rc;
529  }
530 
531  // Parse cookies.
532  if (tx->connp->cfg->parse_request_cookies) {
533  rc = htp_parse_cookies_v0(tx->connp);
534  if (rc != HTP_OK) return rc;
535  }
536 
537  // Parse authentication information.
538  if (tx->connp->cfg->parse_request_auth) {
539  rc = htp_parse_authorization(tx->connp);
540  if (rc == HTP_DECLINED) {
541  // Don't fail the stream if an authorization header is invalid, just set a flag.
542  tx->flags |= HTP_AUTH_INVALID;
543  } else {
544  if (rc != HTP_OK) return rc;
545  }
546  }
547 
548  // Finalize sending raw header data.
549  rc = htp_connp_req_receiver_finalize_clear(tx->connp);
550  if (rc != HTP_OK) return rc;
551 
552  // Run hook REQUEST_HEADERS.
553  rc = htp_hook_run_all(tx->connp->cfg->hook_request_headers, tx);
554  if (rc != HTP_OK) return rc;
555 
556  // We cannot proceed if the request is invalid.
557  if (tx->flags & HTP_REQUEST_INVALID) {
558  return HTP_ERROR;
559  }
560 
561  return HTP_OK;
562 }

775 htp_status_t htp_tx_res_process_body_data_ex(htp_tx_t *tx, const void *data, size_t len) {
776  if (tx == NULL) return HTP_ERROR;
777 
778  // NULL data is allowed in this private function; it's
779  // used to indicate the end of response body.
780 
781  #ifdef HTP_DEBUG
782  fprint_raw_data(stderr, __FUNCTION__, data, len);
783  #endif
784 
785  htp_tx_data_t d;
786 
787  d.tx = tx;
788  d.data = (unsigned char *) data;
789  d.len = len;
790  d.is_last = 0;
791 
792  // Keep track of body size before decompression.
793  tx->response_message_len += d.len;
794 
795  switch (tx->response_content_encoding_processing) {
796  case HTP_COMPRESSION_GZIP:
797  case HTP_COMPRESSION_DEFLATE:

807  tx->connp->out_decompressor->destroy(tx->connp->out_decompressor);
808  tx->connp->out_decompressor = NULL;
809  }
810  break;
811 
812  case HTP_COMPRESSION_NONE:
813  // When there's no decompression, response_entity_len.
814  // is identical to response_message_len.
815  tx->response_entity_len += d.len;
816 
817  htp_status_t rc = htp_res_run_hook_body_data(tx->connp, &d);
818  if (rc != HTP_OK) return HTP_ERROR;
819  break;
820 
821  default:
822  // Internal error.
823  htp_log(tx->connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0,

1015 htp_status_t htp_tx_state_response_complete_ex(htp_tx_t *tx, int hybrid_mode) {
1016  if (tx == NULL) return HTP_ERROR;
1017 
1018  if (tx->response_progress != HTP_RESPONSE_COMPLETE) {
1019  tx->response_progress = HTP_RESPONSE_COMPLETE;
1020 
1021  // Run the last RESPONSE_BODY_DATA HOOK, but only if there was a response body present.
1022  if (tx->response_transfer_coding != HTP_CODING_NO_BODY) {
1023  htp_tx_res_process_body_data_ex(tx, NULL, 0);
1024  }
1025 
1026  // Run hook RESPONSE_COMPLETE.
1027  htp_status_t rc = htp_hook_run_all(tx->connp->cfg->hook_response_complete, tx);
1028  if (rc != HTP_OK) return rc;
1029  }
1030 
1031  if (!hybrid_mode) {
1032  // Check if the inbound parser is waiting on us. If it is, that means that
1033  // there might be request data that the inbound parser hasn't consumed yet.
1034  // If we don't stop parsing we might encounter a response without a request,
1035  // which is why we want to return straight away before processing any data.
1036  //
1037  // This situation will occur any time the parser needs to see the server
1038  // respond to a particular situation before it can decide how to proceed. For
1039  // example, when a CONNECT is sent, different paths are used when it is accepted
1040  // and when it is not accepted.
1041  //
1042  // It is not enough to check only in_status here. Because of pipelining, it's possible
1043  // that many inbound transactions have been processed, and that the parser is
1044  // waiting on a response that we have not seen yet.
1045  if ((tx->connp->in_status == HTP_STREAM_DATA_OTHER) && (tx->connp->in_tx == tx->connp->out_tx)) {
1046  return HTP_DATA_OTHER;
1047  }
1048 
1049  // Do we have a signal to yield to inbound processing at
1050  // the end of the next transaction?
1051  if (tx->connp->out_data_other_at_tx_end) {
1052  // We do. Let's yield then.
1053  tx->connp->out_data_other_at_tx_end = 0;
1054  return HTP_DATA_OTHER;
1055  }
1056  }
1057 
1058  // Make a copy of the connection parser pointer, so that
1059  // we don't have to reference it via tx, which may be destroyed later.
1060  htp_connp_t *connp = tx->connp;
1061 
1062  // Finalize the transaction. This may call may destroy the transaction, if auto-destroy is enabled.
1063  htp_status_t rc = htp_tx_finalize(tx);
1064  if (rc != HTP_OK) return rc;
1065 
1066  // Disconnect transaction from the parser.
1067  connp->out_tx = NULL;
1068 
1069  connp->out_state = htp_connp_RES_IDLE;
1070 
1071  return HTP_OK;
1072 }

1074 htp_status_t htp_tx_state_response_headers(htp_tx_t *tx) {
1075  if (tx == NULL) return HTP_ERROR;

1078 
1079  // Determine content encoding.
1080 
1081  tx->response_content_encoding = HTP_COMPRESSION_NONE;
1082 
1083  htp_header_t *ce = htp_table_get_c(tx->response_headers, "content-encoding");
1084  if (ce != NULL) {

1086  tx->response_content_encoding = HTP_COMPRESSION_GZIP;
1087  } else if ((bstr_cmp_c_nocase(ce->value, "deflate") == 0) || (bstr_cmp_c_nocase(ce->value, "x-deflate") == 0)) {
1088  tx->response_content_encoding = HTP_COMPRESSION_DEFLATE;
1089  } else if (bstr_cmp_c_nocase(ce->value, "inflate") != 0) {
1090  htp_log(tx->connp, HTP_LOG_MARK, HTP_LOG_WARNING, 0, "Unknown response content encoding");
1091  }

1095  if (tx->connp->cfg->response_decompression_enabled) {
1096  tx->response_content_encoding_processing = tx->response_content_encoding;
1097  } else {
1098  tx->response_content_encoding_processing = HTP_COMPRESSION_NONE;
1099  }
1100 
1101  // Finalize sending raw header data.
1102  htp_status_t rc = htp_connp_res_receiver_finalize_clear(tx->connp);
1103  if (rc != HTP_OK) return rc;

1106  rc = htp_hook_run_all(tx->connp->cfg->hook_response_headers, tx);

1113  //
1114  // 2. As above, but the user disables decompression by setting response_content_encoding
1115  // to COMPRESSION_NONE.
1116  //
1117  // 3. Decompression is disabled and we do not attempt to enable it, but the user
1118  // forces decompression by setting response_content_encoding to one of the
1119  // supported algorithms.
1120  if ((tx->response_content_encoding_processing == HTP_COMPRESSION_GZIP) || (tx->response_content_encoding_processing == HTP_COMPRESSION_DEFLATE)) {
1121  if (tx->connp->out_decompressor != NULL) {
1122  tx->connp->out_decompressor->destroy(tx->connp->out_decompressor);
1123  tx->connp->out_decompressor = NULL;
1124  }
1125 
1126  tx->connp->out_decompressor = htp_gzip_decompressor_create(tx->connp, tx->response_content_encoding_processing);
1127  if (tx->connp->out_decompressor == NULL) return HTP_ERROR;
1128 
1129  tx->connp->out_decompressor->callback = htp_tx_res_process_body_data_decompressor_callback;
1130  } else if (tx->response_content_encoding_processing != HTP_COMPRESSION_NONE) {
1131  return HTP_ERROR;
1132  }
1133 
1134  return HTP_OK;
1135 }

1144  if (rc != HTP_OK) return rc;
1145

1148  if (tx->is_protocol_0_9) {
