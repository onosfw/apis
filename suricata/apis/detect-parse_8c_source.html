354  *
355  * \retval Pointer to Last sm.

357 SigMatch *SigMatchGetLastSMFromLists(Signature *s, int args, ...)
358 {
359  if (args == 0 || args % 2 != 0) {
360  SCLogError(SC_ERR_INVALID_ARGUMENTS, "You need to send an even no of args "
361  "(non zero as well) to this function, since we need a "
362  "SigMatch list for every SigMatch type(send a map of sm_type "
363  "and sm_list) sent");
364  /* as this is a bug we should abort to ease debugging */
365  BUG_ON(1);
366  }
367 
368  SigMatch *sm_last = NULL;
369  SigMatch *sm_new;
370  int i;
371 
372  va_list ap;
373  va_start(ap, args);
374 
375  for (i = 0; i < args; i += 2) {
376  int sm_type = va_arg(ap, int);
377  SigMatch *sm_list = va_arg(ap, SigMatch *);
378  sm_new = SigMatchGetLastSM(sm_list, sm_type);
379  if (sm_new == NULL)
380  continue;
381  if (sm_last == NULL || sm_new->idx > sm_last->idx)
382  sm_last = sm_new;
383  }
384 
385  va_end(ap);
386 
387  return sm_last;
388 }

480 static int SigParseOptions(DetectEngineCtx *de_ctx, Signature *s, char *optstr, char *output, size_t output_size)
481 {
482 #define MAX_SUBSTRINGS 30
483  int ov[MAX_SUBSTRINGS];
484  int ret = 0;
485  SigTableElmt *st = NULL;
486  char optname[64];
487  char optvalue[DETECT_MAX_RULE_SIZE] = "";
488 
489  ret = pcre_exec(option_pcre, option_pcre_extra, optstr, strlen(optstr), 0, 0, ov, MAX_SUBSTRINGS);
490  /* if successful, we either have:
491  * 2: keyword w/o value
492  * 3: keyword w value, final opt OR keyword w/o value, more options coming
493  * 4: keyword w value, more options coming
494  */
495  if (ret != 2 && ret != 3 && ret != 4) {
496  SCLogError(SC_ERR_PCRE_MATCH, "pcre_exec failed: ret %" PRId32 ", optstr \"%s\"", ret, optstr);
497  goto error;
498  }
499 
500  /* extract the substrings */
501  if (pcre_copy_substring(optstr, ov, MAX_SUBSTRINGS, 1, optname, sizeof(optname)) < 0) {
502  goto error;
503  }
504 
505  /* Call option parsing */
506  st = SigTableGet(optname);
507  if (st == NULL) {
508  SCLogError(SC_ERR_RULE_KEYWORD_UNKNOWN, "unknown rule keyword '%s'.", optname);
509  goto error;
510  }
511 
512  if (ret == 3) {
513  if (st->flags & SIGMATCH_NOOPT) {
514  if (pcre_copy_substring(optstr, ov, MAX_SUBSTRINGS, 2, output, output_size) < 0) {
515  goto error;
516  }
517  } else {
518  if (pcre_copy_substring(optstr, ov, MAX_SUBSTRINGS, 2, optvalue, sizeof(optvalue)) < 0) {
519  goto error;
520  }
521  }
522  } else if (ret == 4) {
523  if (pcre_copy_substring(optstr, ov, MAX_SUBSTRINGS, 2, optvalue, sizeof(optvalue)) < 0) {
524  goto error;
525  }
526  if (pcre_copy_substring(optstr, ov, MAX_SUBSTRINGS, 3, output, output_size) < 0) {
527  goto error;
528  }
529  }
530 
531  if (!(st->flags & (SIGMATCH_NOOPT|SIGMATCH_OPTIONAL_OPT))) {
532  if (strlen(optvalue) == 0) {
533  SCLogError(SC_ERR_INVALID_SIGNATURE, "invalid formatting or malformed option to %s keyword: \'%s\'",
534  optname, optstr);
535  goto error;
536  }
537  }
538 
539  /* setup may or may not add a new SigMatch to the list */
540  if (st->Setup(de_ctx, s, strlen(optvalue) ? optvalue : NULL) < 0) {
541  SCLogDebug("\"%s\" failed to setup", st->name);
542  goto error;
543  }
544 
545  if (ret == 4) {
546  return 1;
547  }
548 
549  return 0;
550 
551 error:
552  return -1;
553 }

739 /**
740  * \internal
741  * \brief split a signature string into a few blocks for further parsing
742  */
743 static int SigParseBasics(const DetectEngineCtx *de_ctx,
744  Signature *s, const char *sigstr, SignatureParser *parser, uint8_t addrs_direction)
745 {
746 #define MAX_SUBSTRINGS 30
747  int ov[MAX_SUBSTRINGS];
748  int ret = 0;
749 
750  ret = pcre_exec(config_pcre, config_pcre_extra, sigstr, strlen(sigstr), 0, 0, ov, MAX_SUBSTRINGS);
751  if (ret != 8 && ret != 9) {
752  SCLogDebug("pcre_exec failed: ret %" PRId32 ", sigstr \"%s\"", ret, sigstr);
753  goto error;
754  }
755 
756  if (pcre_copy_substring(sigstr, ov, MAX_SUBSTRINGS, 1, parser->action, sizeof(parser->action)) < 0)
757  goto error;
758  if (pcre_copy_substring(sigstr, ov, MAX_SUBSTRINGS, 2, parser->protocol, sizeof(parser->protocol)) < 0)
759  goto error;
760  if (pcre_copy_substring(sigstr, ov, MAX_SUBSTRINGS, 3, parser->src, sizeof(parser->src)) < 0)
761  goto error;
762  if (pcre_copy_substring(sigstr, ov, MAX_SUBSTRINGS, 4, parser->sp, sizeof(parser->sp)) < 0)
763  goto error;
764  if (pcre_copy_substring(sigstr, ov, MAX_SUBSTRINGS, 5, parser->direction, sizeof(parser->direction)) < 0)
765  goto error;
766  if (pcre_copy_substring(sigstr, ov, MAX_SUBSTRINGS, 6, parser->dst, sizeof(parser->dst)) < 0)
767  goto error;
768  if (pcre_copy_substring(sigstr, ov, MAX_SUBSTRINGS, 7, parser->dp, sizeof(parser->dp)) < 0)
769  goto error;
770  if (ret == 9) {
771  if (pcre_copy_substring(sigstr, ov, MAX_SUBSTRINGS, 8, parser->opts, sizeof(parser->opts)) < 0)
772  goto error;
773  }
774 
775  /* Parse Action */
776  if (SigParseAction(s, parser->action) < 0)
777  goto error;
778 
779  if (SigParseProto(s, parser->protocol) < 0)
780  goto error;
781 
782  if (strcmp(parser->direction, "<-") == 0) {
783  SCLogError(SC_ERR_INVALID_DIRECTION, "\"<-\" is not a valid direction modifier, \"->\" and \"<>\" are supported.");
784  goto error;
785  }
786  /* Check if it is bidirectional */
787  if (strcmp(parser->direction, "<>") == 0)
788  s->init_flags |= SIG_FLAG_INIT_BIDIREC;
789 
790  /* Parse Address & Ports */
791  if (SigParseAddress(de_ctx, s, parser->src, SIG_DIREC_SRC ^ addrs_direction) < 0)
792  goto error;
793 
794  if (SigParseAddress(de_ctx, s, parser->dst, SIG_DIREC_DST ^ addrs_direction) < 0)
795  goto error;
796 
797  /* For IPOnly */
798  if (IPOnlySigParseAddress(de_ctx, s, parser->src, SIG_DIREC_SRC ^ addrs_direction) < 0)
799  goto error;
800 
801  if (IPOnlySigParseAddress(de_ctx, s, parser->dst, SIG_DIREC_DST ^ addrs_direction) < 0)
802  goto error;
803 
804  /* By AWS - Traditionally we should be doing this only for tcp/udp/sctp,
805  * but we do it for regardless of ip proto, since the dns/dnstcp/dnsudp
806  * changes that we made sees to it that at this point of time we don't
807  * set the ip proto for the sig. We do it a bit later. */
808  if (SigParsePort(de_ctx, s, parser->sp, SIG_DIREC_SRC ^ addrs_direction) < 0)
809  goto error;
810  if (SigParsePort(de_ctx, s, parser->dp, SIG_DIREC_DST ^ addrs_direction) < 0)
811  goto error;
812 
813  return 0;
814 
815 error:
816  return -1;
817 }

822  * \param de_ctx detection engine ctx to add it to
823  * \param s memory structure to store the signature in
824  * \param sigstr the raw signature as a null terminated string
825  * \param addrs_direction direction (for bi-directional sigs)
826  *
827  * \param -1 parse error
828  * \param 0 ok

840  if (ret < 0) {
841  SCLogDebug("SigParseBasics failed");
842  SCReturnInt(-1);
843  }

861  }

863  } while (ret == 1);
864  }
865 
866  s->sig_str = NULL;
867 
868  DetectIPProtoRemoveAllSMs(s);
869 
870  SCReturnInt(ret);

1692

1706

1751  de_ctx->sig_list = sw_dup->s->next;
1752  }

1754  if (sw_temp.s != NULL) {
1755  sw_next = HashListTableLookup(de_ctx->dup_sig_hash_table,
1756  (void *)&sw_temp, 0);
1757  sw_next->s_prev = sw_dup->s_prev;

1764  sw_temp.s = sw_dup->s->next->next;
1765  sw_dup->s_prev->next = sw_dup->s->next->next;
1766  SigFree(sw_dup->s->next);

1769  sw_dup->s_prev->next = sw_dup->s->next;
1770  }
1771  SigDuplWrapper *sw_next = NULL;
1772  if (sw_temp.s != NULL) {
