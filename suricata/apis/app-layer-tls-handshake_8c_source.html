57 static void TLSCertificateErrCodeToWarning(SSLState *ssl_state, uint32_t errcode)
58 {
59  if (errcode == 0)
60  return;
61 
62  switch (errcode) {
63  case ERR_DER_ELEMENT_SIZE_TOO_BIG:
64  case ERR_DER_INVALID_SIZE:
65  AppLayerDecoderEventsSetEvent(ssl_state->f,
66  TLS_DECODER_EVENT_CERTIFICATE_INVALID_LENGTH);
67  break;
68  case ERR_DER_UNSUPPORTED_STRING:
69  AppLayerDecoderEventsSetEvent(ssl_state->f,
70  TLS_DECODER_EVENT_CERTIFICATE_INVALID_STRING);
71  break;
72  case ERR_DER_UNKNOWN_ELEMENT:
73  AppLayerDecoderEventsSetEvent(ssl_state->f,
74  TLS_DECODER_EVENT_CERTIFICATE_UNKNOWN_ELEMENT);
75  break;
76  case ERR_DER_MISSING_ELEMENT:
77  AppLayerDecoderEventsSetEvent(ssl_state->f,
78  TLS_DECODER_EVENT_CERTIFICATE_MISSING_ELEMENT);
79  break;
80  case ERR_DER_GENERIC:
81  default:
82  AppLayerDecoderEventsSetEvent(ssl_state->f,
83  TLS_DECODER_EVENT_INVALID_CERTIFICATE);
84  break;
85  };
86 }

88 int DecodeTLSHandshakeServerCertificate(SSLState *ssl_state, uint8_t *input, uint32_t input_len)
89 {
90  uint32_t certificates_length, cur_cert_length;
91  int i;
92  Asn1Generic *cert;
93  char buffer[256];
94  int rc;
95  int parsed;
96  uint8_t *start_data;
97  uint32_t errcode = 0;
98 
99  if (input_len < 3)
100  return 1;
101 
102  certificates_length = input[0]<<16 | input[1]<<8 | input[2];
103  /* check if the message is complete */
104  if (input_len < certificates_length + 3)
105  return 0;
106 
107  start_data = input;
108  input += 3;
109  parsed = 3;
110 
111  i = 0;
112  while (certificates_length > 0) {
113  cur_cert_length = input[0]<<16 | input[1]<<8 | input[2];
114  input += 3;
115  parsed += 3;
116 
117  if (input - start_data + cur_cert_length > input_len) {
118  AppLayerDecoderEventsSetEvent(ssl_state->f, TLS_DECODER_EVENT_INVALID_CERTIFICATE);
119  return -1;
120  }
121  cert = DecodeDer(input, cur_cert_length, &errcode);
122  if (cert == NULL) {
123  TLSCertificateErrCodeToWarning(ssl_state, errcode);
124  }
125  if (cert != NULL) {
126  rc = Asn1DerGetSubjectDN(cert, buffer, sizeof(buffer), &errcode);
127  if (rc != 0) {
128  TLSCertificateErrCodeToWarning(ssl_state, errcode);
129  } else {
130  SSLCertsChain *ncert;
131  //SCLogInfo("TLS Cert %d: %s\n", i, buffer);
132  if (i == 0) {
133  if (ssl_state->server_connp.cert0_subject == NULL)
134  ssl_state->server_connp.cert0_subject = SCStrdup(buffer);
135  if (ssl_state->server_connp.cert0_subject == NULL) {
136  DerFree(cert);
137  return -1;
138  }
139  }
140  ncert = (SSLCertsChain *)SCMalloc(sizeof(SSLCertsChain));
141  if (ncert == NULL) {
142  DerFree(cert);
143  return -1;
144  }
145  memset(ncert, 0, sizeof(*ncert));
146  ncert->cert_data = input;
147  ncert->cert_len = cur_cert_length;
148  TAILQ_INSERT_TAIL(&ssl_state->server_connp.certs, ncert, next);
149  }
150  rc = Asn1DerGetIssuerDN(cert, buffer, sizeof(buffer), &errcode);
151  if (rc != 0) {
152  TLSCertificateErrCodeToWarning(ssl_state, errcode);
153  } else {
154  //SCLogInfo("TLS IssuerDN %d: %s\n", i, buffer);
155  if (i == 0) {
156  if (ssl_state->server_connp.cert0_issuerdn == NULL)
157  ssl_state->server_connp.cert0_issuerdn = SCStrdup(buffer);
158  if (ssl_state->server_connp.cert0_issuerdn == NULL) {
159  DerFree(cert);
160  return -1;
161  }
162  }
163  }
164  DerFree(cert);
165 
166  if (i == 0 && ssl_state->server_connp.cert0_fingerprint == NULL) {
167  int msg_len = cur_cert_length;
168  int hash_len = 20;
169  int out_len = 60;
170  char out[out_len];
171  unsigned char *hash;
172  hash = ComputeSHA1((unsigned char *) input, (int) msg_len);
173  char *p = out;
174  int j = 0;
175 
176  if (hash == NULL) {
177  SCLogWarning(SC_ERR_MEM_ALLOC, "Can not allocate fingerprint string");
178  } else {
179  for (j = 0; j < hash_len; j++, p += 3) {
180  snprintf(p, 4, j == hash_len - 1 ? "%02x" : "%02x:", hash[j]);
181  }
182  SCFree(hash);
183  ssl_state->server_connp.cert0_fingerprint = SCStrdup(out);
184  if (ssl_state->server_connp.cert0_fingerprint == NULL) {
185  SCLogWarning(SC_ERR_MEM_ALLOC, "Can not allocate fingerprint string");
186  }
187  }
188 
189  ssl_state->server_connp.cert_input = input;
190  ssl_state->server_connp.cert_input_len = cur_cert_length;
191  }
192 
193  }
194 
195  i++;
196  certificates_length -= (cur_cert_length + 3);
197  parsed += cur_cert_length;
198  input += cur_cert_length;
199  }
200 
201  return parsed;
202 
203 }
