103 void DoDump(SCProfileKeywordDetectCtx *rules_ctx, FILE *fp, const char *name)
104 {
105  int i;
106  fprintf(fp, " ----------------------------------------------"
107  "------------------------------------------------------"
108  "----------------------------\n");
109  fprintf(fp, " Stats for: %s\n", name);
110  fprintf(fp, " ----------------------------------------------"
111  "------------------------------------------------------"
112  "----------------------------\n");
113  fprintf(fp, " %-16s %-15s %-15s %-15s %-15s %-15s %-15s %-15s\n", "Keyword", "Ticks", "Checks", "Matches", "Max Ticks", "Avg", "Avg Match", "Avg No Match");
114  fprintf(fp, " ---------------- "
115  "--------------- "
116  "--------------- "
117  "--------------- "
118  "--------------- "
119  "--------------- "
120  "--------------- "
121  "--------------- "
122  "\n");
123  for (i = 0; i < DETECT_TBLSIZE; i++) {
124  SCProfileKeywordData *d = &rules_ctx->data[i];
125  if (d == NULL || d->checks == 0)
126  continue;
127 
128  uint64_t ticks = d->ticks_match + d->ticks_no_match;
129  double avgticks = 0;
130  double avgticks_match = 0;
131  double avgticks_no_match = 0;
132  if (ticks && d->checks) {
133  avgticks = (ticks / d->checks);
134 
135  if (d->ticks_match && d->matches)
136  avgticks_match = (d->ticks_match / d->matches);
137  if (d->ticks_no_match && (d->checks - d->matches) != 0)
138  avgticks_no_match = (d->ticks_no_match / (d->checks - d->matches));
139  }
140 
141  fprintf(fp,
142  " %-16s %-15"PRIu64" %-15"PRIu64" %-15"PRIu64" %-15"PRIu64" %-15.2f %-15.2f %-15.2f\n",
143  sigmatch_table[i].name,
144  ticks,
145  d->checks,
146  d->matches,
147  d->max,
148  avgticks,
149  avgticks_match,
150  avgticks_no_match);
151  }

154 void
155 SCProfilingKeywordDump(DetectEngineCtx *de_ctx)
156 {
157  int i;
158  FILE *fp;
159  struct timeval tval;
160  struct tm *tms;
161  struct tm local_tm;
162 
163  if (profiling_keyword_enabled == 0)
164  return;
165 
166  gettimeofday(&tval, NULL);
167  tms = SCLocalTime(tval.tv_sec, &local_tm);
168 
169  if (profiling_keywords_output_to_file == 1) {
170  SCLogDebug("file %s mode %s", profiling_file_name, profiling_file_mode);
171 
172  fp = fopen(profiling_file_name, profiling_file_mode);
173 
174  if (fp == NULL) {
175  SCLogError(SC_ERR_FOPEN, "failed to open %s: %s", profiling_file_name,
176  strerror(errno));
177  return;
178  }
179  } else {
180  fp = stdout;
181  }
182 
183  fprintf(fp, " ----------------------------------------------"
184  "------------------------------------------------------"
185  "----------------------------\n");
186  fprintf(fp, " Date: %" PRId32 "/%" PRId32 "/%04d -- "
187  "%02d:%02d:%02d\n", tms->tm_mon + 1, tms->tm_mday, tms->tm_year + 1900,
188  tms->tm_hour,tms->tm_min, tms->tm_sec);
189 
190  /* global stats first */
191  DoDump(de_ctx->profile_keyword_ctx, fp, "total");
192  /* per buffer stats next, but only if there are stats to print */
193  for (i = 0; i < DETECT_SM_LIST_MAX; i++) {
194  int j;
195  uint64_t checks = 0;
196  for (j = 0; j < DETECT_TBLSIZE; j++) {
197  checks += de_ctx->profile_keyword_ctx_per_list[i]->data[j].checks;
198  }
199 
200  if (checks)
201  DoDump(de_ctx->profile_keyword_ctx_per_list[i], fp,
202  DetectSigmatchListEnumToString(i));
203  }
204 
205  fprintf(fp,"\n");
206  if (fp != stdout)
207  fclose(fp);
208 
209  SCLogInfo("Done dumping keyword profiling data.");
210 }

212 /**
213  * \brief Update a rule counter.
214  *
215  * \param id The ID of this counter.
216  * \param ticks Number of CPU ticks for this rule.
217  * \param match Did the rule match?
218  */
219 void
220 SCProfilingKeywordUpdateCounter(DetectEngineThreadCtx *det_ctx, int id, uint64_t ticks, int match)
221 {
222  if (det_ctx != NULL && det_ctx->keyword_perf_data != NULL && id < DETECT_TBLSIZE) {
223  SCProfileKeywordData *p = &det_ctx->keyword_perf_data[id];
224 
225  p->checks++;
226  p->matches += match;
227  if (ticks > p->max)
228  p->max = ticks;
229  if (match == 1)
230  p->ticks_match += ticks;
231  else
232  p->ticks_no_match += ticks;
233 
234  /* store per list (buffer type) as well */
235  if (det_ctx->keyword_perf_list >= 0 && det_ctx->keyword_perf_list < DETECT_SM_LIST_MAX) {
236  p = &det_ctx->keyword_perf_data_per_list[det_ctx->keyword_perf_list][id];
237  p->checks++;
238  p->matches += match;
239  if (ticks > p->max)
240  p->max = ticks;
241  if (match == 1)
242  p->ticks_match += ticks;
243  else
244  p->ticks_no_match += ticks;
245  }
246  }
247 }

310 static void SCProfilingKeywordThreadMerge(DetectEngineCtx *de_ctx, DetectEngineThreadCtx *det_ctx)
311 {
312  if (de_ctx == NULL || de_ctx->profile_keyword_ctx == NULL ||
313  de_ctx->profile_keyword_ctx->data == NULL || det_ctx == NULL ||
314  det_ctx->keyword_perf_data == NULL)
315  return;
316 
317  int i;
318  for (i = 0; i < DETECT_TBLSIZE; i++) {
319  de_ctx->profile_keyword_ctx->data[i].checks += det_ctx->keyword_perf_data[i].checks;
320  de_ctx->profile_keyword_ctx->data[i].matches += det_ctx->keyword_perf_data[i].matches;
321  de_ctx->profile_keyword_ctx->data[i].ticks_match += det_ctx->keyword_perf_data[i].ticks_match;
322  de_ctx->profile_keyword_ctx->data[i].ticks_no_match += det_ctx->keyword_perf_data[i].ticks_no_match;
323  if (det_ctx->keyword_perf_data[i].max > de_ctx->profile_keyword_ctx->data[i].max)
324  de_ctx->profile_keyword_ctx->data[i].max = det_ctx->keyword_perf_data[i].max;
325  }
326 
327  int j;
328  for (j = 0; j < DETECT_SM_LIST_MAX; j++) {
329  for (i = 0; i < DETECT_TBLSIZE; i++) {
330  de_ctx->profile_keyword_ctx_per_list[j]->data[i].checks += det_ctx->keyword_perf_data_per_list[j][i].checks;
331  de_ctx->profile_keyword_ctx_per_list[j]->data[i].matches += det_ctx->keyword_perf_data_per_list[j][i].matches;
332  de_ctx->profile_keyword_ctx_per_list[j]->data[i].ticks_match += det_ctx->keyword_perf_data_per_list[j][i].ticks_match;
333  de_ctx->profile_keyword_ctx_per_list[j]->data[i].ticks_no_match += det_ctx->keyword_perf_data_per_list[j][i].ticks_no_match;
334  if (det_ctx->keyword_perf_data_per_list[j][i].max > de_ctx->profile_keyword_ctx_per_list[j]->data[i].max)
335  de_ctx->profile_keyword_ctx_per_list[j]->data[i].max = det_ctx->keyword_perf_data_per_list[j][i].max;
336  }
337  }
338 }

340 void SCProfilingKeywordThreadCleanup(DetectEngineThreadCtx *det_ctx)
341 {
342  if (det_ctx == NULL || det_ctx->de_ctx == NULL || det_ctx->keyword_perf_data == NULL)
