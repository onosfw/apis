29  * \file
30  *
31  * \author David DIALLO <diallo@et.esiea.fr>
32  *
33  * Implements the Modbus function and access keywords
34  * You can specify a:
35  * - concrete function like Modbus:
36  * function 8, subfunction 4 (diagnostic: Force Listen Only Mode)
37  * - data (in primary table) register access (r/w) like Modbus:
38  * access read coils, address 1000 (.i.e Read coils: at address 1000)
39  * - write data value at specific address Modbus:
40  * access write, address 1500<>2000, value >2000 (Write multiple coils/register:
41  * at address between 1500 and 2000 value greater than 2000)
42  */

99 /** \internal
100  *
101  * \brief This function is used to parse Modbus parameters in access mode
102  *
103  * \param str Pointer to the user provided id option
104  *
105  * \retval Pointer to DetectModbusData on success or NULL on failure
106  */
107 static DetectModbus *DetectModbusAccessParse(char *str)
108 {
109  SCEnter();
110  DetectModbus *modbus = NULL;
111 
112  char arg[MAX_SUBSTRINGS];
113  int ov[MAX_SUBSTRINGS], ret, res;
114 
115  ret = pcre_exec(access_parse_regex, access_parse_regex_study, str, strlen(str), 0, 0, ov, MAX_SUBSTRINGS);
116 
117  if (ret < 1)
118  goto error;
119 
120  res = pcre_copy_substring(str, ov, MAX_SUBSTRINGS, 1, arg, MAX_SUBSTRINGS);
121  if (res < 0) {
122  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
123  goto error;
124  }
125 
126  /* We have a correct Modbus option */
127  modbus = (DetectModbus *) SCCalloc(1, sizeof(DetectModbus));
128  if (unlikely(modbus == NULL))
129  goto error;
130 
131  if (strcmp(arg, "read") == 0)
132  modbus->type = MODBUS_TYP_READ;
133  else if (strcmp(arg, "write") == 0)
134  modbus->type = MODBUS_TYP_WRITE;
135  else
136  goto error;
137 
138  if (ret > 2) {
139  res = pcre_copy_substring(str, ov, MAX_SUBSTRINGS, 2, arg, MAX_SUBSTRINGS);
140  if (res < 0) {
141  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
142  goto error;
143  }
144 
145  if (*arg != '\0') {
146  if (strcmp(arg, "discretes") == 0) {
147  if (modbus->type == MODBUS_TYP_WRITE)
148  /* Discrete access is only read access. */
149  goto error;
150 
151  modbus->type |= MODBUS_TYP_DISCRETES;
152  }
153  else if (strcmp(arg, "coils") == 0) {
154  modbus->type |= MODBUS_TYP_COILS;
155  }
156  else if (strcmp(arg, "input") == 0) {
157  if (modbus->type == MODBUS_TYP_WRITE) {
158  /* Input access is only read access. */
159  goto error;
160  }
161 
162  modbus->type |= MODBUS_TYP_INPUT;
163  }
164  else if (strcmp(arg, "holding") == 0) {
165  modbus->type |= MODBUS_TYP_HOLDING;
166  }
167  else
168  goto error;
169  }
170 
171  if (ret > 4) {
172  res = pcre_copy_substring(str, ov, MAX_SUBSTRINGS, 4, arg, MAX_SUBSTRINGS);
173  if (res < 0) {
174  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
175  goto error;
176  }
177 
178  /* We have a correct address option */
179  modbus->address = (DetectModbusValue *) SCCalloc(1, sizeof(DetectModbusValue));
180  if (unlikely(modbus->address == NULL))
181  goto error;
182 
183  if (arg[0] == '>') {
184  modbus->address->min = atoi((const char*) (arg+1));
185  modbus->address->mode = DETECT_MODBUS_GT;
186  } else if (arg[0] == '<') {
187  modbus->address->min = atoi((const char*) (arg+1));
188  modbus->address->mode = DETECT_MODBUS_LT;
189  } else {
190  modbus->address->min = atoi((const char*) arg);
191  }
192  SCLogDebug("and min/equal address %d", modbus->address->min);
193 
194  if (ret > 5) {
195  res = pcre_copy_substring(str, ov, MAX_SUBSTRINGS, 5, arg, MAX_SUBSTRINGS);
196  if (res < 0) {
197  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
198  goto error;
199  }
200 
201  if (*arg != '\0') {
202  modbus->address->max = atoi((const char*) (arg+2));
203  modbus->address->mode = DETECT_MODBUS_RA;
204  SCLogDebug("and max address %d", modbus->address->max);
205  }
206 
207  if (ret > 7) {
208  res = pcre_copy_substring(str, ov, MAX_SUBSTRINGS, 7, arg, MAX_SUBSTRINGS);
209  if (res < 0) {
210  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
211  goto error;
212  }
213 
214  if (modbus->address->mode != DETECT_MODBUS_EQ) {
215  SCLogError(SC_ERR_CONFLICTING_RULE_KEYWORDS, "rule contains conflicting keywords (address range and value).");
216  goto error;
217  }
218 
219  /* We have a correct address option */
220  modbus->data = (DetectModbusValue *) SCCalloc(1, sizeof(DetectModbusValue));
221  if (unlikely(modbus->data == NULL))
222  goto error;
223 
224  if (arg[0] == '>') {
225  modbus->data->min = atoi((const char*) (arg+1));
226  modbus->data->mode = DETECT_MODBUS_GT;
227  } else if (arg[0] == '<') {
228  modbus->data->min = atoi((const char*) (arg+1));
229  modbus->data->mode = DETECT_MODBUS_LT;
230  } else {
231  modbus->data->min = atoi((const char*) arg);
232  }
233  SCLogDebug("and min/equal value %d", modbus->data->min);
234 
235  if (ret > 8) {
236  res = pcre_copy_substring(str, ov, MAX_SUBSTRINGS, 8, arg, MAX_SUBSTRINGS);
237  if (res < 0) {
238  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
239  goto error;
240  }
241 
242  if (*arg != '\0') {
243  modbus->data->max = atoi((const char*) (arg+2));
244  modbus->data->mode = DETECT_MODBUS_RA;
245  SCLogDebug("and max value %d", modbus->data->max);
246  }
247  }
248  }
249  }
250  }
251  }
252 
253  SCReturnPtr(modbus, "DetectModbusAccess");
254 
255 error:
256  if (modbus != NULL)
257  DetectModbusFree(modbus);
258 
259  SCReturnPtr(NULL, "DetectModbus");
260 }

262 /** \internal
263  *
264  * \brief This function is used to parse Modbus parameters in function mode
265  *
266  * \param str Pointer to the user provided id option
267  *
268  * \retval id_d pointer to DetectModbusData on success
269  * \retval NULL on failure
270  */
271 static DetectModbus *DetectModbusFunctionParse(char *str)
272 {
273  SCEnter();
274  DetectModbus *modbus = NULL;
275 
276  char arg[MAX_SUBSTRINGS], *ptr = arg;
277  int ov[MAX_SUBSTRINGS], res, ret;
278 
279  ret = pcre_exec(function_parse_regex, function_parse_regex_study, str, strlen(str), 0, 0, ov, MAX_SUBSTRINGS);
280 
281  if (ret < 1)
282  goto error;
283 
284  res = pcre_copy_substring(str, ov, MAX_SUBSTRINGS, 1, arg, MAX_SUBSTRINGS);
285  if (res < 0) {
286  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
287  goto error;
288  }
289 
290  /* We have a correct Modbus function option */
291  modbus = (DetectModbus *) SCCalloc(1, sizeof(DetectModbus));
292  if (unlikely(modbus == NULL))
293  goto error;
294 
295  if (isdigit((unsigned char)ptr[0])) {
296  modbus->function = atoi((const char*) ptr);
297  SCLogDebug("will look for modbus function %d", modbus->function);
298 
299  if (ret > 2) {
300  res = pcre_copy_substring(str, ov, MAX_SUBSTRINGS, 3, arg, MAX_SUBSTRINGS);
301  if (res < 0) {
302  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
303  goto error;
304  }
305 
306  /* We have a correct address option */
307  modbus->subfunction =(uint16_t *) SCCalloc(1, sizeof(uint16_t));
308  if (modbus->subfunction == NULL)
309  goto error;
310 
311  *(modbus->subfunction) = atoi((const char*) arg);
312  SCLogDebug("and subfunction %d", *(modbus->subfunction));
313  }
314  } else {
315  uint8_t neg = 0;
316 
317  if (ptr[0] == '!') {
318  neg = 1;
319  ptr++;
320  }
321 
322  if (strcmp("assigned", ptr) == 0)
323  modbus->category = MODBUS_CAT_PUBLIC_ASSIGNED;
324  else if (strcmp("unassigned", ptr) == 0)
325  modbus->category = MODBUS_CAT_PUBLIC_UNASSIGNED;
326  else if (strcmp("public", ptr) == 0)
327  modbus->category = MODBUS_CAT_PUBLIC_ASSIGNED | MODBUS_CAT_PUBLIC_UNASSIGNED;
328  else if (strcmp("user", ptr) == 0)
329  modbus->category = MODBUS_CAT_USER_DEFINED;
330  else if (strcmp("reserved", ptr) == 0)
331  modbus->category = MODBUS_CAT_RESERVED;
332  else if (strcmp("all", ptr) == 0)
333  modbus->category = MODBUS_CAT_ALL;
334 
335  if (neg)
336  modbus->category = ~modbus->category;
337  SCLogDebug("will look for modbus category function %d", modbus->category);
338  }
339 
340  SCReturnPtr(modbus, "DetectModbusFunction");
341 
342 error:
343  if (modbus != NULL)
344  DetectModbusFree(modbus);
345 
346  SCReturnPtr(NULL, "DetectModbus");
347 }

349 /** \internal
350  *
351  * \brief this function is used to add the parsed "id" option into the current signature
352  *
353  * \param de_ctx Pointer to the Detection Engine Context
354  * \param s Pointer to the Current Signature
355  * \param str Pointer to the user provided "id" option
356  *
357  * \retval 0 on Success or -1 on Failure
358  */
359 static int DetectModbusSetup(DetectEngineCtx *de_ctx, Signature *s, char *str)
360 {
361  SCEnter();
362  DetectModbus *modbus = NULL;
363  SigMatch *sm = NULL;
364 
365  if (s->alproto != ALPROTO_UNKNOWN && s->alproto != ALPROTO_MODBUS) {
366  SCLogError(SC_ERR_CONFLICTING_RULE_KEYWORDS, "rule contains conflicting keywords.");
367  goto error;
368  }
369 
370  if ((modbus = DetectModbusFunctionParse(str)) == NULL) {
371  if ((modbus = DetectModbusAccessParse(str)) == NULL) {
372  SCLogError(SC_ERR_PCRE_MATCH, "invalid modbus option");
373  goto error;
374  }
375  }
376 
377  /* Okay so far so good, lets get this into a SigMatch and put it in the Signature. */
378  sm = SigMatchAlloc();
379  if (sm == NULL)
380  goto error;
381 
382  sm->type = DETECT_AL_MODBUS;
383  sm->ctx = (void *) modbus;
384 
385  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MODBUS_MATCH);
386  s->alproto = ALPROTO_MODBUS;
387 
388  SCReturnInt(0);
389 
390 error:
391  if (modbus != NULL)
392  DetectModbusFree(modbus);
393  if (sm != NULL)
394  SCFree(sm);
395  SCReturnInt(-1);
396 }

398 /**
399  * \brief Registration function for Modbus keyword
400  */
401 void DetectModbusRegister(void)
402 {
403  SCEnter();
404  sigmatch_table[DETECT_AL_MODBUS].name = "modbus";
405  sigmatch_table[DETECT_AL_MODBUS].Match = NULL;
406  sigmatch_table[DETECT_AL_MODBUS].AppLayerMatch = NULL;
407  sigmatch_table[DETECT_AL_MODBUS].alproto = ALPROTO_MODBUS;
408  sigmatch_table[DETECT_AL_MODBUS].Setup = DetectModbusSetup;
409  sigmatch_table[DETECT_AL_MODBUS].Free = DetectModbusFree;
410  sigmatch_table[DETECT_AL_MODBUS].RegisterTests = DetectModbusRegisterTests;
411 
412  const char *eb;
413  int eo, opts = 0;
414 
415  SCLogDebug("registering modbus rule option");
416 
417  /* Function PARSE_REGEX */
418  function_parse_regex = pcre_compile(PARSE_REGEX_FUNCTION, opts, &eb, &eo, NULL);
419  if (function_parse_regex == NULL) {
420  SCLogError(SC_ERR_PCRE_COMPILE, "Compile of \"%s\" failed at offset %" PRId32 ": %s",
421  PARSE_REGEX_FUNCTION, eo, eb);
422  goto error;
423  }
424 
425  function_parse_regex_study = pcre_study(function_parse_regex, 0, &eb);
426  if (eb != NULL) {
427  SCLogError(SC_ERR_PCRE_STUDY, "pcre study failed: %s", eb);
428  goto error;
429  }
430 
431  /* Access PARSE_REGEX */
432  access_parse_regex = pcre_compile(PARSE_REGEX_ACCESS, opts, &eb, &eo, NULL);
433  if (access_parse_regex == NULL) {
434  SCLogError(SC_ERR_PCRE_COMPILE, "Compile of \"%s\" failed at offset %" PRId32 ": %s",
435  PARSE_REGEX_ACCESS, eo, eb);
436  goto error;
437  }
438 
439  access_parse_regex_study = pcre_study(access_parse_regex, 0, &eb);
440  if (eb != NULL) {
441  SCLogError(SC_ERR_PCRE_STUDY, "pcre study failed: %s", eb);
442  goto error;
443  }
444 
445 error:
446  SCReturn;
447 }
