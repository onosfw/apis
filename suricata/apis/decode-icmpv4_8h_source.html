291 /**
292  * \brief Calculates the checksum for the ICMP packet
293  *
294  * \param pkt Pointer to the start of the ICMP packet
295  * \param hlen Total length of the ICMP packet(header + payload)
296  *
297  * \retval csum Checksum for the ICMP packet
298  */
299 static inline uint16_t ICMPV4CalculateChecksum(uint16_t *pkt, uint16_t tlen)
300 {
301  uint16_t pad = 0;
302  uint32_t csum = pkt[0];
303 
304  tlen -= 4;
305  pkt += 2;
306 
307  while (tlen >= 32) {
308  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5] + pkt[6] +
309  pkt[7] + pkt[8] + pkt[9] + pkt[10] + pkt[11] + pkt[12] + pkt[13] +
310  pkt[14] + pkt[15];
311  tlen -= 32;
312  pkt += 16;
313  }
314 
315  while(tlen >= 8) {
316  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3];
317  tlen -= 8;
318  pkt += 4;
319  }
320 
321  while(tlen >= 4) {
322  csum += pkt[0] + pkt[1];
323  tlen -= 4;
324  pkt += 2;
325  }
326 
327  while (tlen > 1) {
328  csum += pkt[0];
329  tlen -= 2;
330  pkt += 1;
331  }
332 
333  if (tlen == 1) {
334  *(uint8_t *)(&pad) = (*(uint8_t *)pkt);
335  csum += pad;
336  }
337 
338  csum = (csum >> 16) + (csum & 0x0000FFFF);
339  csum += (csum >> 16);
340 
341  return (uint16_t) ~csum;
342 }
