136  * \brief Initialize the Napatech receiver thread, generate a single
137  * NapatechThreadVar structure for each thread, this will
138  * contain a NtNetStreamRx_t stream handle which is used when the
139  * thread executes to acquire the packets.
140  *
141  * \param tv Thread variable to ThreadVars
142  * \param initdata Initial data to the adapter passed from the user,
143  * this is processed by the user.
144  *
145  * For now, we assume that we have only a single name for the NAPATECH
146  * adapter.
147  *
148  * \param data data pointer gets populated with
149  *
150  */
151 TmEcode NapatechStreamThreadInit(ThreadVars *tv, void *initdata, void **data)
152 {
153  SCEnter();
154  struct NapatechStreamDevConf *conf = (struct NapatechStreamDevConf *)initdata;
155  uintmax_t stream_id = conf->stream_id;
156  *data = NULL;
157 
158  SCLogInfo("Napatech Thread Stream ID:%lu", stream_id);
159 
160  NapatechThreadVars *ntv = SCMalloc(sizeof(NapatechThreadVars));
161  if (unlikely(ntv == NULL)) {
162  SCLogError(SC_ERR_MEM_ALLOC, "Failed to allocate memory for NAPATECH thread vars.");
163  exit(EXIT_FAILURE);
164  }
165 
166  memset(ntv, 0, sizeof (NapatechThreadVars));
167  ntv->stream_id = stream_id;
168  ntv->tv = tv;
169  ntv->hba = conf->hba;
170 
171  SCLogInfo("Started processing packets from NAPATECH Stream: %lu", ntv->stream_id);
172 
173  *data = (void *)ntv;
174 
175  SCReturnInt(TM_ECODE_OK);
176 }

178 /**
179  * \brief Main Napatech reading Loop function
180  */
181 TmEcode NapatechStreamLoop(ThreadVars *tv, void *data, void *slot)
182 {
183  SCEnter();
184 
185  int32_t status;
186  char errbuf[100];
187  uint64_t pkt_ts;
188  NtNetBuf_t packet_buffer;
189  NapatechThreadVars *ntv = (NapatechThreadVars *)data;
190  NtNetRx_t stat_cmd;
191 
192  SCLogInfo("Opening NAPATECH Stream: %lu for processing", ntv->stream_id);
193 
194  if ((status = NT_NetRxOpen(&(ntv->rx_stream), "SuricataStream", NT_NET_INTERFACE_PACKET, ntv->stream_id, ntv->hba)) != NT_SUCCESS) {
195  NT_ExplainError(status, errbuf, sizeof(errbuf));
196  SCLogError(SC_ERR_NAPATECH_OPEN_FAILED, "Failed to open NAPATECH Stream: %lu - %s", ntv->stream_id, errbuf);
197  SCFree(ntv);
198  SCReturnInt(TM_ECODE_FAILED);
199  }
200 
201  stat_cmd.cmd = NT_NETRX_READ_CMD_STREAM_DROP;
202 
203  SCLogInfo("Napatech Packet Stream Loop Started for Stream ID: %lu", ntv->stream_id);
204 
205  TmSlot *s = (TmSlot *)slot;
206  ntv->slot = s->slot_next;
207 
208  while (!(suricata_ctl_flags & (SURICATA_STOP | SURICATA_KILL))) {
209  /* make sure we have at least one packet in the packet pool, to prevent
210  * us from alloc'ing packets at line rate */
211  PacketPoolWait();
212 
213  /*
214  * Napatech returns packets 1 at a time
215  */
216  status = NT_NetRxGet(ntv->rx_stream, &packet_buffer, 1000);
217  if (unlikely(status == NT_STATUS_TIMEOUT || status == NT_STATUS_TRYAGAIN)) {
218  /*
219  * no frames currently available
220  */
221  continue;
222  } else if (unlikely(status != NT_SUCCESS)) {
223  SCLogError(SC_ERR_NAPATECH_STREAM_NEXT_FAILED,
224  "Failed to read from Napatech Stream: %lu",
225  ntv->stream_id);
226  SCReturnInt(TM_ECODE_FAILED);
227  }
228 
229  Packet *p = PacketGetFromQueueOrAlloc();
230  if (unlikely(p == NULL)) {
231  NT_NetRxRelease(ntv->rx_stream, packet_buffer);
232  SCReturnInt(TM_ECODE_FAILED);
233  }
234 
235  pkt_ts = NT_NET_GET_PKT_TIMESTAMP(packet_buffer);
236 
237  /*
238  * Handle the different timestamp forms that the napatech cards could use
239  * - NT_TIMESTAMP_TYPE_NATIVE is not supported due to having an base of 0 as opposed to NATIVE_UNIX which has a base of 1/1/1970
240  */
241  switch(NT_NET_GET_PKT_TIMESTAMP_TYPE(packet_buffer)) {
242  case NT_TIMESTAMP_TYPE_NATIVE_UNIX:
243  p->ts.tv_sec = pkt_ts / 100000000;
244  p->ts.tv_usec = ((pkt_ts % 100000000) / 100) + (pkt_ts % 100) > 50 ? 1 : 0;
245  break;
246  case NT_TIMESTAMP_TYPE_PCAP:
247  p->ts.tv_sec = pkt_ts >> 32;
248  p->ts.tv_usec = pkt_ts & 0xFFFFFFFF;
249  break;
250  case NT_TIMESTAMP_TYPE_PCAP_NANOTIME:
251  p->ts.tv_sec = pkt_ts >> 32;
252  p->ts.tv_usec = ((pkt_ts & 0xFFFFFFFF) / 1000) + (pkt_ts % 1000) > 500 ? 1 : 0;
253  break;
254  case NT_TIMESTAMP_TYPE_NATIVE_NDIS:
255  /* number of seconds between 1/1/1601 and 1/1/1970 */
256  p->ts.tv_sec = (pkt_ts / 100000000) - 11644473600;
257  p->ts.tv_usec = ((pkt_ts % 100000000) / 100) + (pkt_ts % 100) > 50 ? 1 : 0;
258  break;
259  default:
260  SCLogError(SC_ERR_NAPATECH_TIMESTAMP_TYPE_NOT_SUPPORTED,
261  "Packet from Napatech Stream: %lu does not have a supported timestamp format",
262  ntv->stream_id);
263  NT_NetRxRelease(ntv->rx_stream, packet_buffer);
264  SCReturnInt(TM_ECODE_FAILED);
265  }
266 
267  SCLogDebug("p->ts.tv_sec %"PRIuMAX"", (uintmax_t)p->ts.tv_sec);
268  p->datalink = LINKTYPE_ETHERNET;
269 
270  ntv->pkts++;
271  ntv->bytes += NT_NET_GET_PKT_WIRE_LENGTH(packet_buffer);
272 
273  // Update drop counter
274  if (unlikely((status = NT_NetRxRead(ntv->rx_stream, &stat_cmd)) != NT_SUCCESS))
275  {
276  NT_ExplainError(status, errbuf, sizeof(errbuf));
277  SCLogWarning(SC_ERR_NAPATECH_STAT_DROPS_FAILED, "Couldn't retrieve drop statistics from the RX stream: %lu - %s", ntv->stream_id, errbuf);
278  }
279  else
280  {
281  ntv->drops += stat_cmd.u.streamDrop.pktsDropped;
282  }
283 
284  if (unlikely(PacketCopyData(p, (uint8_t *)NT_NET_GET_PKT_L2_PTR(packet_buffer), NT_NET_GET_PKT_WIRE_LENGTH(packet_buffer)))) {
285  TmqhOutputPacketpool(ntv->tv, p);
286  NT_NetRxRelease(ntv->rx_stream, packet_buffer);
287  SCReturnInt(TM_ECODE_FAILED);
288  }
289 
290  if (unlikely(TmThreadsSlotProcessPkt(ntv->tv, ntv->slot, p) != TM_ECODE_OK)) {
291  TmqhOutputPacketpool(ntv->tv, p);
292  NT_NetRxRelease(ntv->rx_stream, packet_buffer);
293  SCReturnInt(TM_ECODE_FAILED);
294  }
295 
296  NT_NetRxRelease(ntv->rx_stream, packet_buffer);
297  StatsSyncCountersIfSignalled(tv);
298  }
299 
300  SCReturnInt(TM_ECODE_OK);
301 }

336 /**
337  * \brief This function passes off to link type decoders.
338  *
339  * NapatechDecode reads packets from the PacketQueue and passes
340  * them off to the proper link type decoder.
341  *
342  * \param t pointer to ThreadVars
343  * \param p pointer to the current packet
344  * \param data pointer that gets cast into PcapThreadVars for ptv
345  * \param pq pointer to the current PacketQueue
346  */
347 TmEcode NapatechDecode(ThreadVars *tv, Packet *p, void *data, PacketQueue *pq,
348  PacketQueue *postpq)
349 {
350  SCEnter();
351 
352  DecodeThreadVars *dtv = (DecodeThreadVars *)data;
353 
354  /* XXX HACK: flow timeout can call us for injected pseudo packets
355  * see bug: https://redmine.openinfosecfoundation.org/issues/1107 */
356  if (p->flags & PKT_PSEUDO_STREAM_END)
357  return TM_ECODE_OK;
358 
359  /* update counters */
360  DecodeUpdatePacketCounters(tv, dtv, p);
361 
362  switch (p->datalink) {
363  case LINKTYPE_ETHERNET:
364  DecodeEthernet(tv, dtv, p, GET_PKT_DATA(p), GET_PKT_LEN(p), pq);
365  break;
366  default:
367  SCLogError(SC_ERR_DATALINK_UNIMPLEMENTED,
368  "Error: datalink type %" PRId32 " not yet supported in module NapatechDecode",
369  p->datalink);
370  break;
371  }
372 
373  PacketDecodeFinalize(tv, dtv, p);
374 
375  SCReturnInt(TM_ECODE_OK);
376 }
