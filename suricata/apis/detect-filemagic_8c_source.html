84 /**
85  * \brief run the magic check
86  *
87  * \param file the file
88  *
89  * \retval -1 error
90  * \retval 0 ok
91  */
92 int FilemagicGlobalLookup(File *file)
93 {
94  if (file == NULL || file->chunks_head == NULL) {
95  SCReturnInt(-1);
96  }
97 
98  /* initial chunk already matching our requirement */
99  if (file->chunks_head->len >= FILEMAGIC_MIN_SIZE) {
100  file->magic = MagicGlobalLookup(file->chunks_head->data, FILEMAGIC_MIN_SIZE);
101  } else {
102  uint8_t *buf = SCMalloc(FILEMAGIC_MIN_SIZE);
103  uint32_t size = 0;
104 
105  if (likely(buf != NULL)) {
106  FileData *ffd = file->chunks_head;
107 
108  for ( ; ffd != NULL; ffd = ffd->next) {
109  uint32_t copy_len = ffd->len;
110  if (size + ffd->len > FILEMAGIC_MIN_SIZE)
111  copy_len = FILEMAGIC_MIN_SIZE - size;
112 
113  memcpy(buf + size, ffd->data, copy_len);
114  size += copy_len;
115 
116  if (size >= FILEMAGIC_MIN_SIZE) {
117  file->magic = MagicGlobalLookup(buf, size);
118  break;
119  }
120  /* file is done but smaller than FILEMAGIC_MIN_SIZE */
121  if (ffd->next == NULL && file->state >= FILE_STATE_CLOSED) {
122  file->magic = MagicGlobalLookup(buf, size);
123  break;
124  }
125  }
126 
127  SCFree(buf);
128  }
129  }
130 
131  SCReturnInt(0);
132 }

184 /**
185  * \brief match the specified filemagic
186  *
187  * \param t thread local vars
188  * \param det_ctx pattern matcher thread local data
189  * \param f *LOCKED* flow
190  * \param flags direction flags
191  * \param file file being inspected
192  * \param s signature being inspected
193  * \param m sigmatch that we will cast into DetectFilemagicData
194  *
195  * \retval 0 no match
196  * \retval 1 match
197  */
198 static int DetectFilemagicMatch (ThreadVars *t, DetectEngineThreadCtx *det_ctx,
199  Flow *f, uint8_t flags, File *file, Signature *s, SigMatch *m)
200 {
201  SCEnter();
202  int ret = 0;
203  DetectFilemagicData *filemagic = (DetectFilemagicData *)m->ctx;
204 
205  if (file->txid < det_ctx->tx_id)
206  SCReturnInt(0);
207 
208  if (file->txid > det_ctx->tx_id)
209  SCReturnInt(0);
210 
211  DetectFilemagicThreadData *tfilemagic = (DetectFilemagicThreadData *)DetectThreadCtxGetKeywordThreadCtx(det_ctx, filemagic->thread_ctx_id);
212  if (tfilemagic == NULL) {
213  SCReturnInt(0);
214  }
215 
216  if (file->magic == NULL) {
217  FilemagicThreadLookup(&tfilemagic->ctx, file);
218  }
219 
220  if (file->magic != NULL) {
221  SCLogDebug("magic %s", file->magic);
222 
223  /* we include the \0 in the inspection, so patterns can match on the
224  * end of the string. */
225  if (BoyerMooreNocase(filemagic->name, filemagic->len, (uint8_t *)file->magic,
226  strlen(file->magic) + 1, filemagic->bm_ctx) != NULL)
227  {
228 #ifdef DEBUG
229  if (SCLogDebugEnabled()) {
230  char *name = SCMalloc(filemagic->len + 1);
231  if (name != NULL) {
232  memcpy(name, filemagic->name, filemagic->len);
233  name[filemagic->len] = '\0';
234  SCLogDebug("will look for filemagic %s", name);
235  }
236  }
237 #endif
238 
239  if (!(filemagic->flags & DETECT_CONTENT_NEGATED)) {
240  ret = 1;
241  }
242  } else if (filemagic->flags & DETECT_CONTENT_NEGATED) {
243  SCLogDebug("negated match");
244  ret = 1;
245  }
246  }
247 
248  SCReturnInt(ret);
249 }

251 /**
252  * \brief Parse the filemagic keyword
253  *
254  * \param idstr Pointer to the user provided option
255  *
256  * \retval filemagic pointer to DetectFilemagicData on success
257  * \retval NULL on failure
258  */
259 static DetectFilemagicData *DetectFilemagicParse (char *str)
260 {
261  DetectFilemagicData *filemagic = NULL;
262 
263  /* We have a correct filemagic option */
264  filemagic = SCMalloc(sizeof(DetectFilemagicData));
265  if (unlikely(filemagic == NULL))
266  goto error;
267 
268  memset(filemagic, 0x00, sizeof(DetectFilemagicData));
269 
270  if (DetectContentDataParse ("filemagic", str, &filemagic->name, &filemagic->len, &filemagic->flags) == -1) {
271  goto error;
272  }
273 
274  filemagic->bm_ctx = BoyerMooreNocaseCtxInit(filemagic->name, filemagic->len);
275  if (filemagic->bm_ctx == NULL) {
276  goto error;
277  }
278 
279  SCLogDebug("flags %02X", filemagic->flags);
280  if (filemagic->flags & DETECT_CONTENT_NEGATED) {
281  SCLogDebug("negated filemagic");
282  }
283 
284 #ifdef DEBUG
285  if (SCLogDebugEnabled()) {
286  char *name = SCMalloc(filemagic->len + 1);
287  if (name != NULL) {
288  memcpy(name, filemagic->name, filemagic->len);
289  name[filemagic->len] = '\0';
290  SCLogDebug("will look for filemagic %s", name);
291  }
292  }
293 #endif
294 
295  return filemagic;
296 
297 error:
298  if (filemagic != NULL)
299  DetectFilemagicFree(filemagic);
300  return NULL;
301 }

303 static void *DetectFilemagicThreadInit(void *data)
304 {
305  char *filename = NULL;
306  FILE *fd = NULL;
307  DetectFilemagicData *filemagic = (DetectFilemagicData *)data;
308  BUG_ON(filemagic == NULL);
309 
310  DetectFilemagicThreadData *t = SCMalloc(sizeof(DetectFilemagicThreadData));
311  if (unlikely(t == NULL)) {
312  SCLogError(SC_ERR_MEM_ALLOC, "couldn't alloc ctx memory");
313  return NULL;
314  }
315  memset(t, 0x00, sizeof(DetectFilemagicThreadData));
316 
317  t->ctx = magic_open(0);
318  if (t->ctx == NULL) {
319  SCLogError(SC_ERR_MAGIC_OPEN, "magic_open failed: %s", magic_error(t->ctx));
320  goto error;
321  }
322 
323  (void)ConfGet("magic-file", &filename);
324  if (filename != NULL) {
325  SCLogInfo("using magic-file %s", filename);
326 
327  if ( (fd = fopen(filename, "r")) == NULL) {
328  SCLogWarning(SC_ERR_FOPEN, "Error opening file: \"%s\": %s", filename, strerror(errno));
329  goto error;
330  }
331  fclose(fd);
332  }
333 
334  if (magic_load(t->ctx, filename) != 0) {
335  SCLogError(SC_ERR_MAGIC_LOAD, "magic_load failed: %s", magic_error(t->ctx));
336  goto error;
337  }
338 
339  return (void *)t;
340 
341 error:
342  if (t->ctx)
343  magic_close(t->ctx);
344  SCFree(t);
345  return NULL;
346 }

358 /**
359  * \brief this function is used to parse filemagic options
360  * \brief into the current signature
361  *
362  * \param de_ctx pointer to the Detection Engine Context
363  * \param s pointer to the Current Signature
364  * \param str pointer to the user provided "filemagic" option
365  *
366  * \retval 0 on Success
367  * \retval -1 on Failure
368  */
369 static int DetectFilemagicSetup (DetectEngineCtx *de_ctx, Signature *s, char *str)
370 {
371  DetectFilemagicData *filemagic = NULL;
372  SigMatch *sm = NULL;
373 
374  filemagic = DetectFilemagicParse(str);
375  if (filemagic == NULL)
376  goto error;
377 
378  filemagic->thread_ctx_id = DetectRegisterThreadCtxFuncs(de_ctx, "filemagic",
379  DetectFilemagicThreadInit, (void *)filemagic,
380  DetectFilemagicThreadFree, 1);
381  if (filemagic->thread_ctx_id == -1)
382  goto error;
383 
384  /* Okay so far so good, lets get this into a SigMatch
385  * and put it in the Signature. */
386  sm = SigMatchAlloc();
387  if (sm == NULL)
388  goto error;
389 
390  sm->type = DETECT_FILEMAGIC;
391  sm->ctx = (void *)filemagic;
392 
393  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_FILEMATCH);
394 
395  if (s->alproto != ALPROTO_HTTP && s->alproto != ALPROTO_SMTP) {
396  SCLogError(SC_ERR_CONFLICTING_RULE_KEYWORDS, "rule contains conflicting keywords.");
397  goto error;
398  }
399 
400  if (s->alproto == ALPROTO_HTTP) {
401  AppLayerHtpNeedFileInspection();
402  }
403 
404  s->file_flags |= (FILE_SIG_NEED_FILE|FILE_SIG_NEED_MAGIC);
405  return 0;
406 
407 error:
408  if (filemagic != NULL)
409  DetectFilemagicFree(filemagic);
410  if (sm != NULL)
411  SCFree(sm);
412  return -1;
413 }
