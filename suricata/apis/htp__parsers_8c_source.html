43 /**
44  * Determines protocol number from a textual representation (i.e., "HTTP/1.1"). This
45  * function will only understand a properly formatted protocol information. It does
46  * not try to be flexible.
47  *
48  * @param[in] protocol
49  * @return Protocol version or PROTOCOL_UNKNOWN.
50  */
51 int htp_parse_protocol(bstr *protocol) {
52  if (protocol == NULL) return HTP_PROTOCOL_INVALID;
53 
54  // TODO This function uses a very strict approach to parsing, whereas
55  // browsers will typically be more flexible, allowing whitespace
56  // before and after the forward slash, as well as allowing leading
57  // zeroes in the numbers. We should be able to parse such malformed
58  // content correctly (but emit a warning).
59  if (bstr_len(protocol) == 8) {
60  unsigned char *ptr = bstr_ptr(protocol);
61  if ((ptr[0] == 'H') && (ptr[1] == 'T') && (ptr[2] == 'T') && (ptr[3] == 'P')
62  && (ptr[4] == '/') && (ptr[6] == '.')) {
63  // Check the version numbers
64  if (ptr[5] == '0') {
65  if (ptr[7] == '9') {
66  return HTP_PROTOCOL_0_9;
67  }
68  } else if (ptr[5] == '1') {
69  if (ptr[7] == '0') {
70  return HTP_PROTOCOL_1_0;
71  } else if (ptr[7] == '1') {
72  return HTP_PROTOCOL_1_1;
73  }
74  }
75  }
76  }
77 
78  return HTP_PROTOCOL_INVALID;
79 }

91 /**
92  * Parses Digest Authorization request header.
93  *
94  * @param[in] connp
95  * @param[in] auth_header
96  */
97 int htp_parse_authorization_digest(htp_connp_t *connp, htp_header_t *auth_header) {
98  // Extract the username
99  int i = bstr_index_of_c(auth_header->value, "username=");
100  if (i == -1) return HTP_DECLINED;
101 
102  unsigned char *data = bstr_ptr(auth_header->value);
103  size_t len = bstr_len(auth_header->value);
104  size_t pos = i + 9;
105 
106  // Ignore whitespace
107  while ((pos < len) && (isspace((int) data[pos]))) pos++;
108 
109  if (data[pos] != '"') return HTP_DECLINED;
110 
111  return htp_extract_quoted_string_as_bstr(data + pos, len - pos, &(connp->in_tx->request_auth_username), NULL);
112 }

114 /**
115  * Parses Basic Authorization request header.
116  *
117  * @param[in] connp
118  * @param[in] auth_header
119  */
120 int htp_parse_authorization_basic(htp_connp_t *connp, htp_header_t *auth_header) {
121  unsigned char *data = bstr_ptr(auth_header->value);
122  size_t len = bstr_len(auth_header->value);
123  size_t pos = 5;
124 
125  // Ignore whitespace
126  while ((pos < len) && (isspace((int) data[pos]))) pos++;
127  if (pos == len) return HTP_DECLINED;
128 
129  // Decode base64-encoded data
130  bstr *decoded = htp_base64_decode_mem(data + pos, len - pos);
131  if (decoded == NULL) return HTP_ERROR;
132 
133  // Now extract the username and password
134  int i = bstr_index_of_c(decoded, ":");
135  if (i == -1) {
136  bstr_free(decoded);
137  return HTP_DECLINED;
138  }
139 
140  connp->in_tx->request_auth_username = bstr_dup_ex(decoded, 0, i);
141  if (connp->in_tx->request_auth_username == NULL) {
142  bstr_free(decoded);
143  return HTP_ERROR;
144  }
145 
146  connp->in_tx->request_auth_password = bstr_dup_ex(decoded, i + 1, bstr_len(decoded) - i - 1);
147  if (connp->in_tx->request_auth_password == NULL) {
148  bstr_free(decoded);
149  bstr_free(connp->in_tx->request_auth_username);
150  return HTP_ERROR;
151  }
152 
153  bstr_free(decoded);
154 
155  return HTP_OK;
156 }

158 /**
159  * Parses Authorization request header.
160  *
161  * @param[in] connp
162  */
163 int htp_parse_authorization(htp_connp_t *connp) {
164  htp_header_t *auth_header = htp_table_get_c(connp->in_tx->request_headers, "authorization");
165  if (auth_header == NULL) {
166  connp->in_tx->request_auth_type = HTP_AUTH_NONE;
167  return HTP_OK;
168  }
169 
170  // TODO Need a flag to raise when failing to parse authentication headers.
171 
172  if (bstr_begins_with_c_nocase(auth_header->value, "basic")) {
173  // Basic authentication
174  connp->in_tx->request_auth_type = HTP_AUTH_BASIC;
175  return htp_parse_authorization_basic(connp, auth_header);
176  } else if (bstr_begins_with_c_nocase(auth_header->value, "digest")) {
177  // Digest authentication
178  connp->in_tx->request_auth_type = HTP_AUTH_DIGEST;
179  return htp_parse_authorization_digest(connp, auth_header);
180  } else {
181  // Unrecognized authentication method
182  connp->in_tx->request_auth_type = HTP_AUTH_UNRECOGNIZED;
183  }
184 
185  return HTP_OK;
186 }
