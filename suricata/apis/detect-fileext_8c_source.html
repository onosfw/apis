78 /**
79  * \brief match the specified file extension
80  *
81  * \param t thread local vars
82  * \param det_ctx pattern matcher thread local data
83  * \param f *LOCKED* flow
84  * \param flags direction flags
85  * \param file file being inspected
86  * \param s signature being inspected
87  * \param m sigmatch that we will cast into DetectFileextData
88  *
89  * \retval 0 no match
90  * \retval 1 match
91  */
92 static int DetectFileextMatch (ThreadVars *t, DetectEngineThreadCtx *det_ctx,
93  Flow *f, uint8_t flags, File *file, Signature *s, SigMatch *m)
94 {
95  SCEnter();
96  int ret = 0;
97 
98  DetectFileextData *fileext = (DetectFileextData *)m->ctx;
99 
100  if (file->name == NULL)
101  SCReturnInt(0);
102 
103  if (file->txid < det_ctx->tx_id)
104  SCReturnInt(0);
105 
106  if (file->txid > det_ctx->tx_id)
107  SCReturnInt(0);
108 
109  if (file->name_len <= fileext->len)
110  SCReturnInt(0);
111 
112  int offset = file->name_len - fileext->len;
113 
114  /* fileext->ext is already in lowercase, as SCMemcmpLowercase requires */
115  if (file->name[offset - 1] == '.' &&
116  SCMemcmpLowercase(fileext->ext, file->name + offset, fileext->len) == 0)
117  {
118  if (!(fileext->flags & DETECT_CONTENT_NEGATED)) {
119  ret = 1;
120  SCLogDebug("File ext found");
121  }
122  }
123 
124  if (ret == 0 && (fileext->flags & DETECT_CONTENT_NEGATED)) {
125  SCLogDebug("negated match");
126  ret = 1;
127  }
128 
129  SCReturnInt(ret);
130 }

132 /**
133  * \brief This function is used to parse fileet
134  *
135  * \param str Pointer to the fileext value string
136  *
137  * \retval pointer to DetectFileextData on success
138  * \retval NULL on failure
139  */
140 static DetectFileextData *DetectFileextParse (char *str)
141 {
142  DetectFileextData *fileext = NULL;
143 
144  /* We have a correct filename option */
145  fileext = SCMalloc(sizeof(DetectFileextData));
146  if (unlikely(fileext == NULL))
147  goto error;
148 
149  memset(fileext, 0x00, sizeof(DetectFileextData));
150 
151  if (DetectContentDataParse("fileext", str, &fileext->ext, &fileext->len, &fileext->flags) == -1) {
152  goto error;
153  }
154  uint16_t u;
155  for (u = 0; u < fileext->len; u++)
156  fileext->ext[u] = tolower(fileext->ext[u]);
157 
158  SCLogDebug("flags %02X", fileext->flags);
159  if (fileext->flags & DETECT_CONTENT_NEGATED) {
160  SCLogDebug("negated fileext");
161  }
162 
163 #ifdef DEBUG
164  if (SCLogDebugEnabled()) {
165  char *ext = SCMalloc(fileext->len + 1);
166  if (ext != NULL) {
167  memcpy(ext, fileext->ext, fileext->len);
168  ext[fileext->len] = '\0';
169  SCLogDebug("will look for fileext %s", ext);
170  }
171  }
172 #endif
173 
174  return fileext;
175 
176 error:
177  if (fileext != NULL)
178  DetectFileextFree(fileext);
179  return NULL;
180 
181 }

183 /**
184  * \brief this function is used to add the parsed "id" option
185  * into the current signature
186  *
187  * \param de_ctx pointer to the Detection Engine Context
188  * \param s pointer to the Current Signature
189  * \param idstr pointer to the user provided "id" option
190  *
191  * \retval 0 on Success
192  * \retval -1 on Failure
193  */
194 static int DetectFileextSetup (DetectEngineCtx *de_ctx, Signature *s, char *str)
195 {
196  DetectFileextData *fileext= NULL;
197  SigMatch *sm = NULL;
198 
199  fileext = DetectFileextParse(str);
200  if (fileext == NULL)
201  goto error;
202 
203  /* Okay so far so good, lets get this into a SigMatch
204  * and put it in the Signature. */
205  sm = SigMatchAlloc();
206  if (sm == NULL)
207  goto error;
208 
209  sm->type = DETECT_FILEEXT;
210  sm->ctx = (void *)fileext;
211 
212  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_FILEMATCH);
213 
214  if (s->alproto != ALPROTO_HTTP && s->alproto != ALPROTO_SMTP) {
215  SCLogError(SC_ERR_CONFLICTING_RULE_KEYWORDS, "rule contains conflicting keywords.");
216  goto error;
217  }
218 
219  if (s->alproto == ALPROTO_HTTP) {
220  AppLayerHtpNeedFileInspection();
221  }
222 
223  s->file_flags |= (FILE_SIG_NEED_FILE|FILE_SIG_NEED_FILENAME);
224  return 0;
225 
226 error:
227  if (fileext != NULL)
228  DetectFileextFree(fileext);
229  if (sm != NULL)
230  SCFree(sm);
231  return -1;
232 
233 }
