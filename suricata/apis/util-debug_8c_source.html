516 /**
517  * \brief Adds the global log_format to the outgoing buffer
518  *
519  * \param log_level log_level of the message that has to be logged
520  * \param msg Buffer containing the outgoing message
521  * \param file File_name from where the message originated
522  * \param function Function_name from where the message originated
523  * \param line Line_no from where the messaged originated
524  *
525  * \retval SC_OK on success; else an error code
526  */
527 SCError SCLogMessage(const SCLogLevel log_level, const char *file,
528  const unsigned int line, const char *function,
529  const SCError error_code, const char *message)
530 {
531  char buffer[SC_LOG_MAX_LOG_MSG_LEN] = "";
532  SCLogOPIfaceCtx *op_iface_ctx = NULL;
533 
534  if (sc_log_module_initialized != 1) {
535  printf("Logging module not initialized. Call SCLogInitLogModule() "
536  "first before using the debug API\n");
537  return SC_OK;
538  }
539 
540  /* get ts here so we log the same ts to each output */
541  struct timeval tval;
542  gettimeofday(&tval, NULL);
543 
544  op_iface_ctx = sc_log_config->op_ifaces;
545  while (op_iface_ctx != NULL) {
546  if (log_level != SC_LOG_NOTSET && log_level > op_iface_ctx->log_level) {
547  op_iface_ctx = op_iface_ctx->next;
548  continue;
549  }
550 
551  switch (op_iface_ctx->iface) {
552  case SC_LOG_OP_IFACE_CONSOLE:
553  if (SCLogMessageGetBuffer(&tval, op_iface_ctx->use_color, op_iface_ctx->type,
554  buffer, sizeof(buffer),
555  op_iface_ctx->log_format ?
556  op_iface_ctx->log_format : sc_log_config->log_format,
557  log_level, file, line, function,
558  error_code, message) == 0)
559  {
560  SCLogPrintToStream((log_level == SC_LOG_ERROR)? stderr: stdout, buffer);
561  }
562  break;
563  case SC_LOG_OP_IFACE_FILE:
564  if (SCLogMessageGetBuffer(&tval, 0, op_iface_ctx->type, buffer, sizeof(buffer),
565  op_iface_ctx->log_format ?
566  op_iface_ctx->log_format : sc_log_config->log_format,
567  log_level, file, line, function,
568  error_code, message) == 0)
569  {
570  SCLogPrintToStream(op_iface_ctx->file_d, buffer);
571  }
572  break;
573  case SC_LOG_OP_IFACE_SYSLOG:
574  if (SCLogMessageGetBuffer(&tval, 0, op_iface_ctx->type, buffer, sizeof(buffer),
575  op_iface_ctx->log_format ?
576  op_iface_ctx->log_format : sc_log_config->log_format,
577  log_level, file, line, function,
578  error_code, message) == 0)
579  {
580  SCLogPrintToSyslog(SCLogMapLogLevelToSyslogLevel(log_level), buffer);
581  }
582  break;
583  default:
584  break;
585  }
586  op_iface_ctx = op_iface_ctx->next;
587  }
588  return SC_OK;
589 }
