66 static json_t *CreateJSONHeaderFromFlow(Flow *f, char *event_type, int dir)
67 {
68  char timebuf[64];
69  char srcip[46], dstip[46];
70  Port sp, dp;
71 
72  json_t *js = json_object();
73  if (unlikely(js == NULL))
74  return NULL;
75 
76  struct timeval tv;
77  memset(&tv, 0x00, sizeof(tv));
78  TimeGet(&tv);
79 
80  CreateIsoTimeString(&tv, timebuf, sizeof(timebuf));
81 
82  srcip[0] = '\0';
83  dstip[0] = '\0';
84  if (FLOW_IS_IPV4(f)) {
85  if (dir == 0) {
86  PrintInet(AF_INET, (const void *)&(f->src.addr_data32[0]), srcip, sizeof(srcip));
87  PrintInet(AF_INET, (const void *)&(f->dst.addr_data32[0]), dstip, sizeof(dstip));
88  } else {
89  PrintInet(AF_INET, (const void *)&(f->dst.addr_data32[0]), srcip, sizeof(srcip));
90  PrintInet(AF_INET, (const void *)&(f->src.addr_data32[0]), dstip, sizeof(dstip));
91  }
92  } else if (FLOW_IS_IPV6(f)) {
93  if (dir == 0) {
94  PrintInet(AF_INET6, (const void *)&(f->src.address), srcip, sizeof(srcip));
95  PrintInet(AF_INET6, (const void *)&(f->dst.address), dstip, sizeof(dstip));
96  } else {
97  PrintInet(AF_INET6, (const void *)&(f->dst.address), srcip, sizeof(srcip));
98  PrintInet(AF_INET6, (const void *)&(f->src.address), dstip, sizeof(dstip));
99  }
100  }
101 
102  if (dir == 0) {
103  sp = f->sp;
104  dp = f->dp;
105  } else {
106  sp = f->dp;
107  dp = f->sp;
108  }
109 
110  char proto[16];
111  if (SCProtoNameValid(f->proto) == TRUE) {
112  strlcpy(proto, known_proto[f->proto], sizeof(proto));
113  } else {
114  snprintf(proto, sizeof(proto), "%03" PRIu32, f->proto);
115  }
116 
117  /* time */
118  json_object_set_new(js, "timestamp", json_string(timebuf));
119 
120  CreateJSONFlowId(js, (const Flow *)f);
121 
122 #if 0 // TODO
123  /* sensor id */
124  if (sensor_id >= 0)
125  json_object_set_new(js, "sensor_id", json_integer(sensor_id));
126 #endif
127  if (event_type) {
128  json_object_set_new(js, "event_type", json_string(event_type));
129  }
130 #if 0
131  /* vlan */
132  if (f->vlan_id[0] > 0) {
133  json_t *js_vlan;
134  switch (f->vlan_idx) {
135  case 1:
136  json_object_set_new(js, "vlan",
137  json_integer(f->vlan_id[0]));
138  break;
139  case 2:
140  js_vlan = json_array();
141  if (unlikely(js != NULL)) {
142  json_array_append_new(js_vlan,
143  json_integer(VLAN_GET_ID1(p)));
144  json_array_append_new(js_vlan,
145  json_integer(VLAN_GET_ID2(p)));
146  json_object_set_new(js, "vlan", js_vlan);
147  }
148  break;
149  default:
150  /* shouldn't get here */
151  break;
152  }
153  }
154 #endif
155  /* tuple */
156  json_object_set_new(js, "src_ip", json_string(srcip));
157  switch(f->proto) {
158  case IPPROTO_ICMP:
159  break;
160  case IPPROTO_UDP:
161  case IPPROTO_TCP:
162  case IPPROTO_SCTP:
163  json_object_set_new(js, "src_port", json_integer(sp));
164  break;
165  }
166  json_object_set_new(js, "dest_ip", json_string(dstip));
167  switch(f->proto) {
168  case IPPROTO_ICMP:
169  break;
170  case IPPROTO_UDP:
171  case IPPROTO_TCP:
172  case IPPROTO_SCTP:
173  json_object_set_new(js, "dest_port", json_integer(dp));
174  break;
175  }
176  json_object_set_new(js, "proto", json_string(proto));
177  switch (f->proto) {
178  case IPPROTO_ICMP:
179  case IPPROTO_ICMPV6:
180  json_object_set_new(js, "icmp_type",
181  json_integer(f->type));
182  json_object_set_new(js, "icmp_code",
183  json_integer(f->code));
184  break;
185  }
186  return js;
187 }

189 /* JSON format logging */
190 static void JsonNetFlowLogJSONToServer(JsonNetFlowLogThread *aft, json_t *js, Flow *f)
191 {
192  json_t *hjs = json_object();
193  if (hjs == NULL) {
194  return;
195  }
196 
197  json_object_set_new(hjs, "app_proto",
198  json_string(AppProtoToString(f->alproto_ts ? f->alproto_ts : f->alproto)));
199 
200  json_object_set_new(hjs, "pkts",
201  json_integer(f->todstpktcnt));
202  json_object_set_new(hjs, "bytes",
203  json_integer(f->todstbytecnt));
204 
205  char timebuf1[64], timebuf2[64];
206 
207  CreateIsoTimeString(&f->startts, timebuf1, sizeof(timebuf1));
208  CreateIsoTimeString(&f->lastts, timebuf2, sizeof(timebuf2));
209 
210  json_object_set_new(hjs, "start", json_string(timebuf1));
211  json_object_set_new(hjs, "end", json_string(timebuf2));
212 
213  int32_t age = f->lastts.tv_sec - f->startts.tv_sec;
214  json_object_set_new(hjs, "age",
215  json_integer(age));
216 
217  json_object_set_new(js, "netflow", hjs);
218 
219  /* TCP */
220  if (f->proto == IPPROTO_TCP) {
221  json_t *tjs = json_object();
222  if (tjs == NULL) {
223  return;
224  }
225 
226  TcpSession *ssn = f->protoctx;
227 
228  char hexflags[3] = "";
229  snprintf(hexflags, sizeof(hexflags), "%02x",
230  ssn ? ssn->client.tcp_flags : 0);
231  json_object_set_new(tjs, "tcp_flags", json_string(hexflags));
232 
233  JsonTcpFlags(ssn ? ssn->client.tcp_flags : 0, tjs);
234 
235  json_object_set_new(js, "tcp", tjs);
236  }
237 }

288 static int JsonNetFlowLogger(ThreadVars *tv, void *thread_data, Flow *f)
289 {
290  SCEnter();
291  JsonNetFlowLogThread *jhl = (JsonNetFlowLogThread *)thread_data;
292  MemBuffer *buffer = (MemBuffer *)jhl->buffer;
293 
294  /* reset */
295  MemBufferReset(buffer);
296  json_t *js = CreateJSONHeaderFromFlow(f, "netflow", 0); //TODO const
297  if (unlikely(js == NULL))
298  return TM_ECODE_OK;
299  JsonNetFlowLogJSONToServer(jhl, js, f);
300  OutputJSONBuffer(js, jhl->flowlog_ctx->file_ctx, buffer);
301  json_object_del(js, "netflow");
302  json_object_clear(js);
303  json_decref(js);
304 
305  /* reset */
306  MemBufferReset(buffer);
307  js = CreateJSONHeaderFromFlow(f, "netflow", 1); //TODO const
308  if (unlikely(js == NULL))
309  return TM_ECODE_OK;
310  JsonNetFlowLogJSONToClient(jhl, js, f);
311  OutputJSONBuffer(js, jhl->flowlog_ctx->file_ctx, buffer);
312  json_object_del(js, "netflow");
313  json_object_clear(js);
314  json_decref(js);
315 
316  SCReturnInt(TM_ECODE_OK);
317 }

329 OutputCtx *OutputNetFlowLogInit(ConfNode *conf)
330 {
331  SCLogInfo("hi");
332  LogFileCtx *file_ctx = LogFileNewCtx();
333  if(file_ctx == NULL) {
334  SCLogError(SC_ERR_HTTP_LOG_GENERIC, "couldn't create new file_ctx");
335  return NULL;
336  }
337 
338  if (SCConfLogOpenGeneric(conf, file_ctx, DEFAULT_LOG_FILENAME, 1) < 0) {
339  LogFileFreeCtx(file_ctx);
340  return NULL;
341  }
342 
343  LogJsonFileCtx *flow_ctx = SCMalloc(sizeof(LogJsonFileCtx));
344  if (unlikely(flow_ctx == NULL)) {
345  LogFileFreeCtx(file_ctx);
346  return NULL;
347  }
348 
349  OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
350  if (unlikely(output_ctx == NULL)) {
351  LogFileFreeCtx(file_ctx);
352  SCFree(flow_ctx);
353  return NULL;
354  }
355 
356  flow_ctx->file_ctx = file_ctx;
357  output_ctx->data = flow_ctx;
358  output_ctx->DeInit = OutputNetFlowLogDeinit;
359 
360  return output_ctx;
361 }
