55 int OutputRegisterFlowLogger(const char *name, FlowLogger LogFunc, OutputCtx *output_ctx)
56 {
57  int module_id = TmModuleGetIdByName(name);
58  if (module_id < 0)
59  return -1;
60 
61  OutputFlowLogger *op = SCMalloc(sizeof(*op));
62  if (op == NULL)
63  return -1;
64  memset(op, 0x00, sizeof(*op));
65 
66  op->LogFunc = LogFunc;
67  op->output_ctx = output_ctx;
68  op->name = name;
69  op->module_id = (TmmId) module_id;
70 
71  if (list == NULL)
72  list = op;
73  else {
74  OutputFlowLogger *t = list;
75  while (t->next)
76  t = t->next;
77  t->next = op;
78  }
79 
80  SCLogDebug("OutputRegisterFlowLogger happy");
81  return 0;
82 }

84 /** \brief Run flow logger(s)
85  * \note flow is already write locked
86  */
87 TmEcode OutputFlowLog(ThreadVars *tv, void *thread_data, Flow *f)
88 {
89  BUG_ON(thread_data == NULL);
90 
91  if (list == NULL)
92  return TM_ECODE_OK;
93  //BUG_ON(list == NULL);
94 
95  OutputLoggerThreadData *op_thread_data = (OutputLoggerThreadData *)thread_data;
96  OutputFlowLogger *logger = list;
97  OutputLoggerThreadStore *store = op_thread_data->store;
98 
99  BUG_ON(logger == NULL && store != NULL);
100  BUG_ON(logger != NULL && store == NULL);
101  BUG_ON(logger == NULL && store == NULL);
102 
103  logger = list;
104  store = op_thread_data->store;
105  while (logger && store) {
106  BUG_ON(logger->LogFunc == NULL);
107 
108  SCLogDebug("logger %p", logger);
109  //PACKET_PROFILING_TMM_START(p, logger->module_id);
110  logger->LogFunc(tv, store->thread_data, f);
111  //PACKET_PROFILING_TMM_END(p, logger->module_id);
112 
113  logger = logger->next;
114  store = store->next;
115 
116  BUG_ON(logger == NULL && store != NULL);
117  BUG_ON(logger != NULL && store == NULL);
118  }
119 
120  return TM_ECODE_OK;
121 }

123 /** \brief thread init for the flow logger
124  * This will run the thread init functions for the individual registered
125  * loggers */
126 TmEcode OutputFlowLogThreadInit(ThreadVars *tv, void *initdata, void **data)
127 {
128  OutputLoggerThreadData *td = SCMalloc(sizeof(*td));
129  if (td == NULL)
130  return TM_ECODE_FAILED;
131  memset(td, 0x00, sizeof(*td));
132 
133  *data = (void *)td;
134 
135  SCLogDebug("OutputFlowLogThreadInit happy (*data %p)", *data);
136 
137  OutputFlowLogger *logger = list;
138  while (logger) {
139  TmModule *tm_module = TmModuleGetByName((char *)logger->name);
140  if (tm_module == NULL) {
141  SCLogError(SC_ERR_INVALID_ARGUMENT,
142  "TmModuleGetByName for %s failed", logger->name);
143  exit(EXIT_FAILURE);
144  }
145 
146  if (tm_module->ThreadInit) {
147  void *retptr = NULL;
148  if (tm_module->ThreadInit(tv, (void *)logger->output_ctx, &retptr) == TM_ECODE_OK) {
149  OutputLoggerThreadStore *ts = SCMalloc(sizeof(*ts));
150 /* todo */ BUG_ON(ts == NULL);
151  memset(ts, 0x00, sizeof(*ts));
152 
153  /* store thread handle */
154  ts->thread_data = retptr;
155 
156  if (td->store == NULL) {
157  td->store = ts;
158  } else {
159  OutputLoggerThreadStore *tmp = td->store;
160  while (tmp->next != NULL)
161  tmp = tmp->next;
162  tmp->next = ts;
163  }
164 
165  SCLogDebug("%s is now set up", logger->name);
166  }
167  }
168 
169  logger = logger->next;
170  }
171 
172  return TM_ECODE_OK;
173 }
