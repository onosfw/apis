275 /** \brief Add array of Signature IDs to rule ID array.
276  *
277  * Checks size of the array first. Calls MpmAddSidsResize to increase
278  * The size of the array, since that is the slow path.
279  *
280  * \param pmq storage for match results
281  * \param sids pointer to array of Signature IDs
282  * \param sids_size number of Signature IDs in sids array.
283  *
284  */
285 static inline void
286 MpmAddSids(PatternMatcherQueue *pmq, SigIntId *sids, uint32_t sids_size)
287 {
288  if (sids_size == 0)
289  return;
290 
291  uint32_t new_size = pmq->rule_id_array_cnt + sids_size;
292  if (new_size > pmq->rule_id_array_size) {
293  if (MpmAddSidsResize(pmq, new_size) == 0) {
294  // Failed to allocate larger memory for all the SIDS, but
295  // keep as many as we can.
296  sids_size = pmq->rule_id_array_size - pmq->rule_id_array_cnt;
297  }
298  }
299  SCLogDebug("Adding %u sids", sids_size);
300  // Add SIDs for this pattern to the end of the array
301  SigIntId *ptr = pmq->rule_id_array + pmq->rule_id_array_cnt;
302  SigIntId *end = ptr + sids_size;
303  do {
304  *ptr++ = *sids++;
305  } while (ptr != end);
306  pmq->rule_id_array_cnt += sids_size;
307 }

312 static inline void
313 MpmAddPid(PatternMatcherQueue *pmq, uint32_t patid)
314 {
315  uint32_t new_size = pmq->pattern_id_array_cnt + 1;
316  if (new_size > pmq->pattern_id_array_size) {
317  if (MpmAddPidResize(pmq, new_size) == 0)
318  return;
319  }
320  pmq->pattern_id_array[pmq->pattern_id_array_cnt] = patid;
321  pmq->pattern_id_array_cnt = new_size;
322  SCLogDebug("pattern_id_array_cnt %u", pmq->pattern_id_array_cnt);
323 }
