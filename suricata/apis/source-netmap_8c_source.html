222 /**
223  * \brief Get interface flags.
224  * \param fd Network susbystem file descritor.
225  * \param ifname Inteface name.
226  * \return Interface flags or -1 on error
227  */
228 static int NetmapGetIfaceFlags(int fd, const char *ifname)
229 {
230  struct ifreq ifr;
231 
232  memset(&ifr, 0, sizeof(ifr));
233  strlcpy(ifr.ifr_name, ifname, sizeof(ifr.ifr_name));
234 
235  if (ioctl(fd, SIOCGIFFLAGS, &ifr) == -1) {
236  SCLogError(SC_ERR_NETMAP_CREATE,
237  "Unable to get flags for iface \"%s\": %s",
238  ifname, strerror(errno));
239  return -1;
240  }
241 
242 #ifdef OS_FREEBSD
243  int flags = (ifr.ifr_flags & 0xffff) | (ifr.ifr_flagshigh << 16);
244  return flags;
245 #else
246  return ifr.ifr_flags;
247 #endif
248 }

275  }
276 
277  return 0;

280 /**
281  * \brief Open interface in netmap mode.
282  * \param ifname Interface name.
283  * \param promisc Enable promiscuous mode.
284  * \param dev Pointer to requested netmap device instance.
285  * \param verbose Verbose error logging.
286  * \return Zero on success.
287  */
288 static int NetmapOpen(char *ifname, int promisc, NetmapDevice **pdevice, int verbose)
289 {
290  NetmapDevice *pdev = NULL;
291  struct nmreq nm_req;
292 
293  *pdevice = NULL;
294 
295  SCMutexLock(&netmap_devlist_lock);
296 
297  /* search interface in our already opened list */
298  TAILQ_FOREACH(pdev, &netmap_devlist, next) {
299  if (strcmp(ifname, pdev->ifname) == 0) {
300  *pdevice = pdev;
301  pdev->ref++;
302  SCMutexUnlock(&netmap_devlist_lock);
303  return 0;
304  }
305  }
306 
307  /* not found, create new record */
308  pdev = SCMalloc(sizeof(*pdev));
309  if (unlikely(pdev == NULL)) {
310  SCLogError(SC_ERR_MEM_ALLOC, "Memory allocation failed");
311  goto error;
312  }
313 
314  memset(pdev, 0, sizeof(*pdev));
315  SC_ATOMIC_INIT(pdev->threads_run);
316  strlcpy(pdev->ifname, ifname, sizeof(pdev->ifname));
317 
318  /* open netmap */
319  int fd = open("/dev/netmap", O_RDWR);
320  if (fd == -1) {
321  SCLogError(SC_ERR_NETMAP_CREATE,
322  "Couldn't open netmap device, error %s",
323  strerror(errno));
324  goto error_pdev;
325  }
326 
327  /* check interface is up */
328  int if_fd = socket(AF_INET, SOCK_DGRAM, 0);
329  if (if_fd < 0) {
330  SCLogError(SC_ERR_NETMAP_CREATE,
331  "Couldn't create control socket for '%s' interface",
332  ifname);
333  goto error_fd;
334  }
335  int if_flags = NetmapGetIfaceFlags(if_fd, ifname);
336  if (if_flags == -1) {
337  if (verbose) {
338  SCLogError(SC_ERR_NETMAP_CREATE,
339  "Can not access to interface '%s'",
340  ifname);
341  }
342  close(if_fd);
343  goto error_fd;
344  }
345  if ((if_flags & IFF_UP) == 0) {
346  if (verbose) {
347  SCLogError(SC_ERR_NETMAP_CREATE, "Interface '%s' is down", ifname);
348  }
349  close(if_fd);
350  goto error_fd;
351  }
352  if (promisc) {
353  if_flags |= IFF_PROMISC;
354  NetmapSetIfaceFlags(if_fd, ifname, if_flags);
355  }
356  close(if_fd);
357 
358  /* query netmap info */
359  memset(&nm_req, 0, sizeof(nm_req));
360  strlcpy(nm_req.nr_name, ifname, sizeof(nm_req.nr_name));
361  nm_req.nr_version = NETMAP_API;
362 
363  if (ioctl(fd, NIOCGINFO, &nm_req) != 0) {
364  if (verbose) {
365  SCLogError(SC_ERR_NETMAP_CREATE,
366  "Couldn't query netmap for %s, error %s",
367  ifname, strerror(errno));
368  }
369  goto error_fd;
370  };
371 
372  pdev->memsize = nm_req.nr_memsize;
373  pdev->rx_rings_cnt = nm_req.nr_rx_rings;
374  pdev->tx_rings_cnt = nm_req.nr_tx_rings;
375  pdev->rings_cnt = max(pdev->rx_rings_cnt, pdev->tx_rings_cnt);
376 
377  /* hw rings + sw ring */
378  pdev->rings = SCMalloc(sizeof(*pdev->rings) * (pdev->rings_cnt + 1));
379  if (unlikely(pdev->rings == NULL)) {
380  SCLogError(SC_ERR_MEM_ALLOC, "Memory allocation failed");
381  goto error_fd;
382  }
383  memset(pdev->rings, 0, sizeof(*pdev->rings) * (pdev->rings_cnt + 1));
384 
385  /* open individual instance for each ring */
386  int success_cnt = 0;
387  for (int i = 0; i <= pdev->rings_cnt; i++) {
388  NetmapRing *pring = &pdev->rings[i];
389  pring->fd = open("/dev/netmap", O_RDWR);
390  if (pring->fd == -1) {
391  SCLogError(SC_ERR_NETMAP_CREATE,
392  "Couldn't open netmap device: %s",
393  strerror(errno));
394  break;
395  }
396 
397  if (i < pdev->rings_cnt) {
398  nm_req.nr_flags = NR_REG_ONE_NIC;
399  nm_req.nr_ringid = i | NETMAP_NO_TX_POLL;
400  } else {
401  nm_req.nr_flags = NR_REG_SW;
402  nm_req.nr_ringid = NETMAP_NO_TX_POLL;
403  }
404  if (ioctl(pring->fd, NIOCREGIF, &nm_req) != 0) {
405  SCLogError(SC_ERR_NETMAP_CREATE,
406  "Couldn't register %s with netmap: %s",
407  ifname, strerror(errno));
408  break;
409  }
410 
411  if (pdev->mem == NULL) {
412  pdev->mem = mmap(0, pdev->memsize, PROT_WRITE | PROT_READ,
413  MAP_SHARED, pring->fd, 0);
414  if (pdev->mem == MAP_FAILED) {
415  SCLogError(SC_ERR_NETMAP_CREATE,
416  "Couldn't mmap netmap device: %s",
417  strerror(errno));
418  break;
419  }
420  pdev->nif = NETMAP_IF(pdev->mem, nm_req.nr_offset);
421  }
422 
423  if ((i < pdev->rx_rings_cnt) || (i == pdev->rings_cnt)) {
424  pring->rx = NETMAP_RXRING(pdev->nif, i);
425  }
426  if ((i < pdev->tx_rings_cnt) || (i == pdev->rings_cnt)) {
427  pring->tx = NETMAP_TXRING(pdev->nif, i);
428  }
429  SCSpinInit(&pring->tx_lock, 0);
430  success_cnt++;
431  }
432 
433  if (success_cnt != (pdev->rings_cnt + 1)) {
434  for(int i = 0; i < success_cnt; i++) {
435  close(pdev->rings[i].fd);
436  }
437  if (pdev->mem) {
438  munmap(pdev->mem, pdev->memsize);
439  }
440  SCFree(pdev->rings);
441  goto error_fd;
442  }
443 
444  close(fd);
445  *pdevice = pdev;
446 
447  TAILQ_INSERT_TAIL(&netmap_devlist, pdev, next);
448  SCMutexUnlock(&netmap_devlist_lock);
449 
450  return 0;
451 
452 error_fd:
453  close(fd);
454 error_pdev:
455  SCFree(pdev);
456 error:
457  SCMutexUnlock(&netmap_devlist_lock);
458  return -1;
459 }

461 /**
462  * \brief Close or dereference netmap device instance.
463  * \param pdev Netmap device instance.
464  * \return Zero on success.
465  */
466 static int NetmapClose(NetmapDevice *dev)
467 {
468  NetmapDevice *pdev, *tmp;
469 
470  SCMutexLock(&netmap_devlist_lock);
471 
472  TAILQ_FOREACH_SAFE(pdev, &netmap_devlist, next, tmp) {
473  if (pdev == dev) {
474  pdev->ref--;
475  if (!pdev->ref) {
476  munmap(pdev->mem, pdev->memsize);
477  for (int i = 0; i <= pdev->rings_cnt; i++) {
478  NetmapRing *pring = &pdev->rings[i];
479  close(pring->fd);
480  SCSpinDestroy(&pring->tx_lock);
481  }
482  SCFree(pdev->rings);
483  TAILQ_REMOVE(&netmap_devlist, pdev, next);
484  SCFree(pdev);
485  }
486  SCMutexUnlock(&netmap_devlist_lock);
487  return 0;
488  }
489  }
490 
491  SCMutexUnlock(&netmap_devlist_lock);
492  return -1;
493 }

509 /**
510  * \brief Init function for ReceiveNetmap.
511  * \param tv pointer to ThreadVars
512  * \param initdata pointer to the interface passed from the user
513  * \param data pointer gets populated with NetmapThreadVars
514  */
515 static TmEcode ReceiveNetmapThreadInit(ThreadVars *tv, void *initdata, void **data)
516 {
517  SCEnter();
518  NetmapIfaceConfig *aconf = initdata;
519 
520  if (initdata == NULL) {
521  SCLogError(SC_ERR_INVALID_ARGUMENT, "initdata == NULL");
522  SCReturnInt(TM_ECODE_FAILED);
523  }
524 
525  NetmapThreadVars *ntv = SCMalloc(sizeof(*ntv));
526  if (unlikely(ntv == NULL)) {
527  SCLogError(SC_ERR_MEM_ALLOC, "Memory allocation failed");
528  goto error;
529  }
530  memset(ntv, 0, sizeof(*ntv));
531 
532  ntv->tv = tv;
533  ntv->checksum_mode = aconf->checksum_mode;
534  ntv->copy_mode = aconf->copy_mode;
535 
536  ntv->livedev = LiveGetDevice(aconf->iface_name);
537  if (ntv->livedev == NULL) {
538  SCLogError(SC_ERR_INVALID_VALUE, "Unable to find Live device");
539  goto error_ntv;
540  }
541 
542  if (NetmapOpen(aconf->iface, aconf->promisc, &ntv->ifsrc, 1) != 0) {
543  goto error_ntv;
544  }
545 
546  if (unlikely(!aconf->iface_sw && !ntv->ifsrc->rx_rings_cnt)) {
547  SCLogError(SC_ERR_NETMAP_CREATE,
548  "Input interface '%s' does not have Rx rings",
549  aconf->iface_name);
550  goto error_src;
551  }
552 
553  if (unlikely(aconf->iface_sw && aconf->threads > 1)) {
554  SCLogError(SC_ERR_INVALID_VALUE,
555  "Interface '%s+'. "
556  "Thread count can't be greater than 1 for SW ring.",
557  aconf->iface_name);
558  goto error_src;
559  } else if (unlikely(aconf->threads > ntv->ifsrc->rx_rings_cnt)) {
560  SCLogError(SC_ERR_INVALID_VALUE,
561  "Thread count can't be greater than Rx ring count. "
562  "Configured %d threads for interface '%s' with %d Rx rings.",
563  aconf->threads, aconf->iface_name, ntv->ifsrc->rx_rings_cnt);
564  goto error_src;
565  }
566 
567  if (aconf->iface_sw) {
568  ntv->thread_idx = 0;
569  } else {
570  do {
571  ntv->thread_idx = SC_ATOMIC_GET(ntv->ifsrc->threads_run);
572  } while (SC_ATOMIC_CAS(&ntv->ifsrc->threads_run, ntv->thread_idx, ntv->thread_idx + 1) == 0);
573  }
574 
575  /* calculate thread rings binding */
576  if (aconf->iface_sw) {
577  ntv->src_ring_from = ntv->src_ring_to = ntv->ifsrc->rings_cnt;
578  } else {
579  int tmp = (ntv->ifsrc->rx_rings_cnt + 1) / aconf->threads;
580  ntv->src_ring_from = ntv->thread_idx * tmp;
581  ntv->src_ring_to = ntv->src_ring_from + tmp - 1;
582  if (ntv->thread_idx == (aconf->threads - 1)) {
583  ntv->src_ring_to = ntv->ifsrc->rx_rings_cnt - 1;
584  }
585  }
586  SCLogDebug("netmap: %s thread:%d rings:%d-%d", aconf->iface_name,
587  ntv->thread_idx, ntv->src_ring_from, ntv->src_ring_to);
588 
589  if (aconf->copy_mode != NETMAP_COPY_MODE_NONE) {
590  if (NetmapOpen(aconf->out_iface, 0, &ntv->ifdst, 1) != 0) {
591  goto error_src;
592  }
593 
594  if (unlikely(!aconf->out_iface_sw && !ntv->ifdst->tx_rings_cnt)) {
595  SCLogError(SC_ERR_NETMAP_CREATE,
596  "Output interface '%s' does not have Tx rings",
597  aconf->out_iface_name);
598  goto error_dst;
599  }
600 
601  /* calculate dst rings bindings */
602  for (int i = ntv->src_ring_from; i <= ntv->src_ring_to; i++) {
603  NetmapRing *ring = &ntv->ifsrc->rings[i];
604  if (aconf->out_iface_sw) {
605  ring->dst_ring_from = ring->dst_ring_to = ntv->ifdst->rings_cnt;
606  } else if (ntv->ifdst->tx_rings_cnt > ntv->ifsrc->rx_rings_cnt) {
607  int tmp = (ntv->ifdst->tx_rings_cnt + 1) / ntv->ifsrc->rx_rings_cnt;
608  ring->dst_ring_from = i * tmp;
609  ring->dst_ring_to = ring->dst_ring_from + tmp - 1;
610  if (i == (ntv->src_ring_to - 1)) {
611  ring->dst_ring_to = ntv->ifdst->tx_rings_cnt - 1;
612  }
613  } else {
614  ring->dst_ring_from = ring->dst_ring_to =
615  i % ntv->ifdst->tx_rings_cnt;
616  }
617  ring->dst_next_ring = ring->dst_ring_from;
618 
619  SCLogDebug("netmap: %s(%d)->%s(%d-%d)",
620  aconf->iface_name, i, aconf->out_iface_name,
621  ring->dst_ring_from, ring->dst_ring_to);
622  }
623  }
624 
625  /* basic counters */
626  ntv->capture_kernel_packets = StatsRegisterCounter("capture.kernel_packets",
627  ntv->tv);
628  ntv->capture_kernel_drops = StatsRegisterCounter("capture.kernel_drops",
629  ntv->tv);
630 
631  /* enable zero-copy mode for workers runmode */
632  char const *active_runmode = RunmodeGetActive();
633  if ((aconf->copy_mode != NETMAP_COPY_MODE_NONE) && active_runmode
634  && !strcmp("workers", active_runmode)) {
635  if (likely(ntv->ifsrc->mem == ntv->ifdst->mem)) {
636  ntv->flags |= NETMAP_FLAG_ZERO_COPY;
637  SCLogInfo("Enabling zero copy mode for %s->%s",
638  aconf->iface_name, aconf->out_iface_name);
639  } else {

682 /**
683  * \brief Output packet to destination interface or drop.
684  * \param ntv Thread local variables.
685  * \param p Source packet.
686  */
687 static TmEcode NetmapWritePacket(NetmapThreadVars *ntv, Packet *p)
688 {
689  if (ntv->copy_mode == NETMAP_COPY_MODE_IPS) {
690  if (PACKET_TEST_ACTION(p, ACTION_DROP)) {
691  return TM_ECODE_OK;
692  }
693  }
694 
695  /* map src ring_id to dst ring_id */
696  NetmapRing *rxring = &ntv->ifsrc->rings[p->netmap_v.ring_id];
697  NetmapRing *txring = &ntv->ifdst->rings[p->netmap_v.dst_ring_id];
698 
699  SCSpinLock(&txring->tx_lock);
700 
701  if (!nm_ring_space(txring->tx)) {
702  ntv->drops++;
703  SCSpinUnlock(&txring->tx_lock);
704  return TM_ECODE_FAILED;
705  }
706 
707  struct netmap_slot *ts = &txring->tx->slot[txring->tx->cur];
708 
709  if (ntv->flags & NETMAP_FLAG_ZERO_COPY) {
710  struct netmap_slot *rs = &rxring->rx->slot[p->netmap_v.slot_id];
711 
712  /* swap slot buffers */
713  uint32_t tmp_idx;
714  tmp_idx = ts->buf_idx;
715  ts->buf_idx = rs->buf_idx;
716  rs->buf_idx = tmp_idx;
717 
718  ts->len = rs->len;
719 
720  ts->flags |= NS_BUF_CHANGED;
721  rs->flags |= NS_BUF_CHANGED;
722  } else {
723  unsigned char *slot_data = (unsigned char *)NETMAP_BUF(txring->tx, ts->buf_idx);
724  memcpy(slot_data, GET_PKT_DATA(p), GET_PKT_LEN(p));
725  ts->len = GET_PKT_LEN(p);
726  ts->flags |= NS_BUF_CHANGED;
727  }
728 
729  txring->tx->head = txring->tx->cur = nm_ring_next(txring->tx, txring->tx->cur);
730  if ((ntv->flags & NETMAP_FLAG_ZERO_COPY) == 0) {
731  ioctl(txring->fd, NIOCTXSYNC, 0);
732  }
733 
734  SCSpinUnlock(&txring->tx_lock);
735 
736  return TM_ECODE_OK;
737 }

756 /**
757  * \brief Read packets from ring and pass them further.
758  * \param ntv Thread local variables.
759  * \param ring_id Ring id to read.
760  */
761 static int NetmapRingRead(NetmapThreadVars *ntv, int ring_id)
762 {
763  SCEnter();
764 
765  NetmapRing *ring = &ntv->ifsrc->rings[ring_id];
766  struct netmap_ring *rx = ring->rx;
767  uint32_t avail = nm_ring_space(rx);
768  uint32_t cur = rx->cur;
769 
770  while (likely(avail-- > 0)) {
771  struct netmap_slot *slot = &rx->slot[cur];
772  unsigned char *slot_data = (unsigned char *)NETMAP_BUF(rx, slot->buf_idx);
773 
774  if (ntv->bpf_prog.bf_len) {
775  struct pcap_pkthdr pkthdr = { {0, 0}, slot->len, slot->len };
776  if (pcap_offline_filter(&ntv->bpf_prog, &pkthdr, slot_data) == 0) {
777  /* rejected by bpf */
778  cur = nm_ring_next(rx, cur);
779  continue;
780  }
781  }
782 
783  Packet *p = PacketGetFromQueueOrAlloc();
784  if (unlikely(p == NULL)) {
785  SCReturnInt(NETMAP_FAILURE);
786  }
787 
788  PKT_SET_SRC(p, PKT_SRC_WIRE);
789  p->livedev = ntv->livedev;
790  p->datalink = LINKTYPE_ETHERNET;
791  p->ts = rx->ts;
792  ntv->pkts++;
793  ntv->bytes += slot->len;
794 
795  /* checksum validation */
796  if (ntv->checksum_mode == CHECKSUM_VALIDATION_DISABLE) {
797  p->flags |= PKT_IGNORE_CHECKSUM;
798  } else if (ntv->checksum_mode == CHECKSUM_VALIDATION_AUTO) {
799  if (ntv->livedev->ignore_checksum) {
800  p->flags |= PKT_IGNORE_CHECKSUM;
801  } else if (ChecksumAutoModeCheck(ntv->pkts,
802  SC_ATOMIC_GET(ntv->livedev->pkts),
803  SC_ATOMIC_GET(ntv->livedev->invalid_checksums))) {
804  ntv->livedev->ignore_checksum = 1;
805  p->flags |= PKT_IGNORE_CHECKSUM;
806  }
807  }
808 
809  if (ntv->flags & NETMAP_FLAG_ZERO_COPY) {
810  if (PacketSetData(p, slot_data, slot->len) == -1) {
811  TmqhOutputPacketpool(ntv->tv, p);
812  SCReturnInt(NETMAP_FAILURE);
813  }
814  } else {
815  if (PacketCopyData(p, slot_data, slot->len) == -1) {
816  TmqhOutputPacketpool(ntv->tv, p);
817  SCReturnInt(NETMAP_FAILURE);
818  }
819  }
820 
821  p->ReleasePacket = NetmapReleasePacket;
822  p->netmap_v.ring_id = ring_id;
823  p->netmap_v.slot_id = cur;
824  p->netmap_v.dst_ring_id = ring->dst_next_ring;
825  p->netmap_v.ntv = ntv;
826 
827  if (ring->dst_ring_from != ring->dst_ring_to) {
828  ring->dst_next_ring++;
829  if (ring->dst_next_ring == ring->dst_ring_to) {
830  ring->dst_next_ring = ring->dst_ring_from;
831  }
832  }
833 
834  SCLogDebug("pktlen: %" PRIu32 " (pkt %p, pkt data %p)",
835  GET_PKT_LEN(p), p, GET_PKT_DATA(p));
836 
837  if (TmThreadsSlotProcessPkt(ntv->tv, ntv->slot, p) != TM_ECODE_OK) {
838  TmqhOutputPacketpool(ntv->tv, p);
839  SCReturnInt(NETMAP_FAILURE);
840  }
841 
842  cur = nm_ring_next(rx, cur);
843  }
844  rx->head = rx->cur = cur;
845 
846  SCReturnInt(NETMAP_OK);
847 }

849 /**
850  * \brief Main netmap reading loop function
851  */
852 static TmEcode ReceiveNetmapLoop(ThreadVars *tv, void *data, void *slot)
853 {
854  SCEnter();
855 
856  TmSlot *s = (TmSlot *)slot;
857  NetmapThreadVars *ntv = (NetmapThreadVars *)data;
858  struct pollfd *fds;
859  int rings_count = ntv->src_ring_to - ntv->src_ring_from + 1;
860 
861  ntv->slot = s->slot_next;
862 
863  fds = SCMalloc(sizeof(*fds) * rings_count);
864  if (unlikely(fds == NULL)) {
865  SCLogError(SC_ERR_MEM_ALLOC, "Memory allocation failed");
866  SCReturnInt(TM_ECODE_FAILED);
867  }
868 
869  for (int i = 0; i < rings_count; i++) {
870  fds[i].fd = ntv->ifsrc->rings[ntv->src_ring_from + i].fd;
871  fds[i].events = POLLIN;
872  }
873 
874  for(;;) {
875  if (suricata_ctl_flags != 0) {
876  break;
877  }
878 
879  /* make sure we have at least one packet in the packet pool,
880  * to prevent us from alloc'ing packets at line rate */
881  PacketPoolWait();
882 
883  int r = poll(fds, rings_count, POLL_TIMEOUT);
884 
885  if (r < 0) {
886  /* error */
887  if(errno != EINTR)
888  SCLogError(SC_ERR_NETMAP_READ,
889  "Error polling netmap from iface '%s': (%d" PRIu32 ") %s",
890  ntv->ifsrc->ifname, errno, strerror(errno));
891  continue;
892  } else if (r == 0) {
893  /* no events, timeout */
894  SCLogDebug("(%s:%d-%d) Poll timeout", ntv->ifsrc->ifname,
895  ntv->src_ring_from, ntv->src_ring_to);
896  continue;
897  }
898 
899  for (int i = 0; i < rings_count; i++) {
900  if (fds[i].revents & POLL_EVENTS) {
901  if (fds[i].revents & POLLERR) {
902  SCLogError(SC_ERR_NETMAP_READ,
903  "Error reading data from iface '%s': (%d" PRIu32 ") %s",
904  ntv->ifsrc->ifname, errno, strerror(errno));
905  } else if (fds[i].revents & POLLNVAL) {
906  SCLogError(SC_ERR_NETMAP_READ,
907  "Invalid polling request");
908  }
909  continue;
910  }
911 
912  if (likely(fds[i].revents & POLLIN)) {
913  int src_ring_id = ntv->src_ring_from + i;
914  NetmapRingRead(ntv, src_ring_id);
915 
916  if ((ntv->copy_mode != NETMAP_COPY_MODE_NONE) &&
917  (ntv->flags & NETMAP_FLAG_ZERO_COPY)) {
918 
919  NetmapRing *src_ring = &ntv->ifsrc->rings[src_ring_id];
920 
921  /* sync dst tx rings */
922  for (int j = src_ring->dst_ring_from; j <= src_ring->dst_ring_to; j++) {
923  NetmapRing *dst_ring = &ntv->ifdst->rings[j];
924  /* if locked, another loop already do sync */
925  if (SCSpinTrylock(&dst_ring->tx_lock) == 0) {
926  ioctl(dst_ring->fd, NIOCTXSYNC, 0);
927  SCSpinUnlock(&dst_ring->tx_lock);
928  }
929  }
930  }
931  }
932  }
933 
934  NetmapDumpCounters(ntv);
935  StatsSyncCountersIfSignalled(tv);
936  }
937 
938  SCFree(fds);
939  StatsSyncCountersIfSignalled(tv);
940  SCReturnInt(TM_ECODE_OK);
941 }
