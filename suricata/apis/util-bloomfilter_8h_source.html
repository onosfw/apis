49 static inline int BloomFilterTest(BloomFilter *bf, void *data, uint16_t datalen)
50 {
51  uint8_t iter = 0;
52  uint32_t hash = 0;
53  int hit = 1;
54 
55  for (iter = 0; iter < bf->hash_iterations; iter++) {
56  hash = bf->Hash(data, datalen, iter, bf->bitarray_size);
57  if (!(bf->bitarray[hash/8] & (1<<hash%8))) {
58  hit = 0;
59  break;
60  }
61  }
62 
63  return hit;
64 }
