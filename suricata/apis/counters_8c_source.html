299  * \brief management thread. This thread is responsible for writing the stats
300  *
301  * \param arg thread var
302  *
303  * \retval NULL This is the value that is always returned
304  */
305 static void *StatsMgmtThread(void *arg)
306 {
307  /* block usr2. usr2 to be handled by the main thread only */
308  UtilSignalBlock(SIGUSR2);
309 
310  ThreadVars *tv_local = (ThreadVars *)arg;
311  uint8_t run = 1;
312  struct timespec cond_time;
313 
314  /* Set the thread name */
315  if (SCSetThreadName(tv_local->name) < 0) {
316  SCLogWarning(SC_ERR_THREAD_INIT, "Unable to set thread name");
317  }
318 
319  if (tv_local->thread_setup_flags != 0)
320  TmThreadSetupOptions(tv_local);
321 
322  /* Set the threads capability */
323  tv_local->cap_flags = 0;
324 
325  SCDropCaps(tv_local);
326 
327  if (stats_ctx == NULL) {
328  SCLogError(SC_ERR_STATS_NOT_INIT, "Stats API not init"
329  "StatsInitCounterApi() has to be called first");
330  TmThreadsSetFlag(tv_local, THV_CLOSED | THV_RUNNING_DONE);
331  return NULL;
332  }
333 
334  TmModule *tm = &tmm_modules[TMM_STATSLOGGER];
335  BUG_ON(tm->ThreadInit == NULL);
336  int r = tm->ThreadInit(tv_local, NULL, &stats_thread_data);
337  if (r != 0 || stats_thread_data == NULL) {
338  SCLogError(SC_ERR_THREAD_INIT, "Stats API "
339  "ThreadInit failed");
340  TmThreadsSetFlag(tv_local, THV_CLOSED | THV_RUNNING_DONE);
341  return NULL;
342  }
343  SCLogDebug("stats_thread_data %p", &stats_thread_data);
344 
345  TmThreadsSetFlag(tv_local, THV_INIT_DONE);
346  while (run) {
347  if (TmThreadsCheckFlag(tv_local, THV_PAUSE)) {
348  TmThreadsSetFlag(tv_local, THV_PAUSED);
349  TmThreadTestThreadUnPaused(tv_local);
350  TmThreadsUnsetFlag(tv_local, THV_PAUSED);
351  }
352 
353  cond_time.tv_sec = time(NULL) + stats_tts;
354  cond_time.tv_nsec = 0;
355 
356  /* wait for the set time, or until we are woken up by
357  * the shutdown procedure */
358  SCCtrlMutexLock(tv_local->ctrl_mutex);
359  SCCtrlCondTimedwait(tv_local->ctrl_cond, tv_local->ctrl_mutex, &cond_time);
360  SCCtrlMutexUnlock(tv_local->ctrl_mutex);
361 
362  StatsOutput(tv_local);
363 
364  if (TmThreadsCheckFlag(tv_local, THV_KILL)) {
365  run = 0;
366  }
367  }
368 
369  TmThreadsSetFlag(tv_local, THV_RUNNING_DONE);
370  TmThreadWaitForFlag(tv_local, THV_DEINIT);
371 
372  r = tm->ThreadDeinit(tv_local, stats_thread_data);
373  if (r != TM_ECODE_OK) {
374  SCLogError(SC_ERR_THREAD_DEINIT, "Stats Counter API "
375  "ThreadDeinit failed");
376  }
377 
378  TmThreadsSetFlag(tv_local, THV_CLOSED);
379  return NULL;
380 }

470  }

478  TmThreadWaitForFlag(tv_local, THV_DEINIT);

498 /**
499  * \brief Registers a counter.
500  *
501  * \param name Name of the counter, to be registered
502  * \param tm_name Thread module to which this counter belongs
503  * \param pctx StatsPublicThreadContext for this tm-tv instance
504  * \param type_q Qualifier describing the type of counter to be registered
505  *
506  * \retval the counter id for the newly registered counter, or the already
507  * present counter on success
508  * \retval 0 on failure
509  */
510 static uint16_t StatsRegisterQualifiedCounter(char *name, char *tm_name,
511  StatsPublicThreadContext *pctx,
512  int type_q, uint64_t (*Func)(void))
513 {
514  StatsCounter **head = &pctx->head;
515  StatsCounter *temp = NULL;
516  StatsCounter *prev = NULL;
517  StatsCounter *pc = NULL;
518 
519  if (name == NULL || pctx == NULL) {
520  SCLogDebug("Counter name, StatsPublicThreadContext NULL");
521  return 0;
522  }
523 
524  temp = prev = *head;
525  while (temp != NULL) {
526  prev = temp;
527 
528  if (strcmp(name, temp->name) == 0) {
529  break;
530  }
531 
532  temp = temp->next;
533  }
534 
535  /* We already have a counter registered by this name */
536  if (temp != NULL)
537  return(temp->id);
538 
539  /* if we reach this point we don't have a counter registered by this name */
540  if ( (pc = SCMalloc(sizeof(StatsCounter))) == NULL)
541  return 0;
542  memset(pc, 0, sizeof(StatsCounter));
543 
544  /* assign a unique id to this StatsCounter. The id is local to this
545  * thread context. Please note that the id start from 1, and not 0 */
546  pc->id = ++(pctx->curr_id);
547  pc->name = name;
548  pc->type = type_q;
549  pc->Func = Func;
550 
551  /* we now add the counter to the list */
552  if (prev == NULL)
553  *head = pc;
554  else
555  prev->next = pc;
556 
557  return pc->id;
558 }

577 /**
578  * \brief The output interface for the Stats API
579  */
580 static int StatsOutput(ThreadVars *tv)
581 {
582  const StatsThreadStore *sts = NULL;
583  const StatsCounter *pc = NULL;
584  void *td = stats_thread_data;
585 
586  if (counters_global_id == 0)
587  return -1;
588 
589  if (stats_table.nstats == 0) {
590  StatsThreadRegister("Global", &stats_ctx->global_counter_ctx);
591 
592  uint32_t nstats = counters_global_id;
593 
594  stats_table.nstats = nstats;
595  stats_table.stats = SCCalloc(stats_table.nstats, sizeof(StatsRecord));
596  if (stats_table.stats == NULL) {
597  stats_table.nstats = 0;
598  SCLogError(SC_ERR_MEM_ALLOC, "could not alloc memory for stats");
599  return -1;
600  }
601 
602  stats_table.ntstats = stats_ctx->sts_cnt;
603  uint32_t array_size = stats_table.nstats * sizeof(StatsRecord);
604  stats_table.tstats = SCCalloc(stats_table.ntstats, array_size);
605  if (stats_table.tstats == NULL) {
606  stats_table.ntstats = 0;
607  SCLogError(SC_ERR_MEM_ALLOC, "could not alloc memory for stats");
608  return -1;
609  }
610 
611  stats_table.start_time = stats_start_time;
612  }
613 
614  const uint16_t max_id = counters_global_id;
615  if (max_id == 0)
616  return -1;
617 
618  /** temporary local table to merge the per thread counters,
619  * especially needed for the average counters */
620  struct CountersMergeTable {
621  int type;
622  uint64_t value;
623  uint64_t updates;
624  } merge_table[max_id];
625  memset(&merge_table, 0x00,
626  max_id * sizeof(struct CountersMergeTable));
627 
628  int thread = stats_ctx->sts_cnt - 1;
629  StatsRecord *table = stats_table.stats;
630 
631  /* Loop through the thread counter stores. The global counters
632  * are in a separate store inside this list. */
633  sts = stats_ctx->sts;
634  SCLogDebug("sts %p", sts);
635  while (sts != NULL) {
636  BUG_ON(thread < 0);
637 
638  SCLogDebug("Thread %d %s ctx %p", thread, sts->name, sts->ctx);
639 
640  /* temporay table for quickly storing the counters for this
641  * thread store, so that we can post process them outside
642  * of the thread store lock */
643  struct CountersMergeTable thread_table[max_id];
644  memset(&thread_table, 0x00,
645  max_id * sizeof(struct CountersMergeTable));
646 
647  SCMutexLock(&sts->ctx->m);
648  pc = sts->ctx->head;
649  while (pc != NULL) {
650  SCLogDebug("Counter %s (%u:%u) value %"PRIu64,
651  pc->name, pc->id, pc->gid, pc->value);
652 
653  thread_table[pc->gid].type = pc->type;
654  switch (pc->type) {
655  case STATS_TYPE_FUNC:
656  if (pc->Func != NULL)
657  thread_table[pc->gid].value = pc->Func();
658  break;
659  case STATS_TYPE_AVERAGE:
660  default:
661  thread_table[pc->gid].value = pc->value;
662  break;
663  }
664  thread_table[pc->gid].updates = pc->updates;
665  table[pc->gid].name = pc->name;
666 
667  pc = pc->next;
668  }
669  SCMutexUnlock(&sts->ctx->m);
670 
671  /* update merge table */
672  uint16_t c;
673  for (c = 0; c < max_id; c++) {
674  struct CountersMergeTable *e = &thread_table[c];
675  /* thread only sets type if it has a counter
676  * of this type. */
677  if (e->type == 0)
678  continue;
679 
680  switch (e->type) {
681  case STATS_TYPE_MAXIMUM:
682  if (e->value > merge_table[c].value)
683  merge_table[c].value = e->value;
684  break;
685  case STATS_TYPE_FUNC:
686  merge_table[c].value = e->value;
687  break;
688  case STATS_TYPE_AVERAGE:
689  default:
690  merge_table[c].value += e->value;
691  break;
692  }
693  merge_table[c].updates += e->updates;
694  merge_table[c].type = e->type;
695  }
696 
697  /* update per thread stats table */
698  for (c = 0; c < max_id; c++) {
699  struct CountersMergeTable *e = &thread_table[c];
700  /* thread only sets type if it has a counter
701  * of this type. */
702  if (e->type == 0)
703  continue;
704 
705  uint32_t offset = (thread * stats_table.nstats) + c;
706  StatsRecord *r = &stats_table.tstats[offset];
707  r->name = table[c].name;
708  r->tm_name = sts->name;
709 
710  switch (e->type) {
711  case STATS_TYPE_AVERAGE:
712  if (e->value > 0 && e->updates > 0) {

723  thread--;
724  }

730  table[x].pvalue = table[x].value;
731  table[x].value = 0;
732  table[x].tm_name = "Total";
733

735  switch (m->type) {
736  case STATS_TYPE_MAXIMUM:
737  if (m->value > table[x].value)
738  table[x].value = m->value;
739  break;
740  case STATS_TYPE_AVERAGE:
741  if (m->value > 0 && m->updates > 0) {
742  table[x].value = (uint64_t)(m->value / m->updates);
743  }
744  break;
745  default:
746  table[x].value += m->value;
747  break;
748  }
749  }
