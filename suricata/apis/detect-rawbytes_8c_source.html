54 static int DetectRawbytesSetup (DetectEngineCtx *de_ctx, Signature *s, char *nullstr)
55 {
56  SCEnter();
57 
58  if (nullstr != NULL) {
59  SCLogError(SC_ERR_INVALID_VALUE, "rawbytes has no value");
60  return -1;
61  }
62 
63  if (s->list != DETECT_SM_LIST_NOTSET) {
64  SCLogError(SC_ERR_RAWBYTES_FILE_DATA, "\"rawbytes\" cannot be combined with \"file_data\"");
65  SCReturnInt(-1);
66  }
67 
68  SigMatch *pm = SigMatchGetLastSMFromLists(s, 2,
69  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_PMATCH]);
70  if (pm == NULL) {
71  SCLogError(SC_ERR_RAWBYTES_MISSING_CONTENT, "\"rawbytes\" needs a preceding content option");
72  SCReturnInt(-1);
73  }
74 
75  switch (pm->type) {
76  case DETECT_CONTENT:
77  {
78  DetectContentData *cd = (DetectContentData *)pm->ctx;
79  if (cd->flags & DETECT_CONTENT_RAWBYTES) {
80  SCLogError(SC_ERR_INVALID_SIGNATURE, "can't use multiple rawbytes modifiers for the same content. ");
81  SCReturnInt(-1);
82  }
83  cd->flags |= DETECT_CONTENT_RAWBYTES;
84  break;
85  }
86  default:
87  SCLogError(SC_ERR_RAWBYTES_MISSING_CONTENT, "\"rawbytes\" needs a preceding content option");
88  SCReturnInt(-1);
89  }
90 
91  SCReturnInt(0);
92 }
