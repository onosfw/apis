122 /** \brief initialize the configuration
123  * \warning Not thread safe */
124 void DefragInitConfig(char quiet)
125 {
126  SCLogDebug("initializing defrag engine...");
127 
128  memset(&defrag_config, 0, sizeof(defrag_config));
129  //SC_ATOMIC_INIT(flow_flags);
130  SC_ATOMIC_INIT(defragtracker_counter);
131  SC_ATOMIC_INIT(defrag_memuse);
132  SC_ATOMIC_INIT(defragtracker_prune_idx);
133  DefragTrackerQueueInit(&defragtracker_spare_q);
134 
135  unsigned int seed = RandomTimePreseed();
136  /* set defaults */
137  defrag_config.hash_rand = (int)(DEFRAG_DEFAULT_HASHSIZE * (rand_r(&seed) / RAND_MAX + 1.0));
138 
139  defrag_config.hash_size = DEFRAG_DEFAULT_HASHSIZE;
140  defrag_config.memcap = DEFRAG_DEFAULT_MEMCAP;
141  defrag_config.prealloc = DEFRAG_DEFAULT_PREALLOC;
142 
143  /* Check if we have memcap and hash_size defined at config */
144  char *conf_val;
145  uint32_t configval = 0;
146 
147  /** set config values for memcap, prealloc and hash_size */
148  if ((ConfGet("defrag.memcap", &conf_val)) == 1)
149  {
150  if (ParseSizeStringU64(conf_val, &defrag_config.memcap) < 0) {
151  SCLogError(SC_ERR_SIZE_PARSE, "Error parsing defrag.memcap "
152  "from conf file - %s. Killing engine",
153  conf_val);
154  exit(EXIT_FAILURE);
155  }
156  }
157  if ((ConfGet("defrag.hash-size", &conf_val)) == 1)
158  {
159  if (ByteExtractStringUint32(&configval, 10, strlen(conf_val),
160  conf_val) > 0) {
161  defrag_config.hash_size = configval;
162  } else {
163  WarnInvalidConfEntry("defrag.hash-size", "%"PRIu32, defrag_config.hash_size);
164  }
165  }
166 
167 
168  if ((ConfGet("defrag.trackers", &conf_val)) == 1)
169  {
170  if (ByteExtractStringUint32(&configval, 10, strlen(conf_val),
171  conf_val) > 0) {
172  defrag_config.prealloc = configval;
173  } else {
174  WarnInvalidConfEntry("defrag.trackers", "%"PRIu32, defrag_config.prealloc);
175  }
176  }
177  SCLogDebug("DefragTracker config from suricata.yaml: memcap: %"PRIu64", hash-size: "
178  "%"PRIu32", prealloc: %"PRIu32, defrag_config.memcap,
179  defrag_config.hash_size, defrag_config.prealloc);
180 
181  /* alloc hash memory */
182  uint64_t hash_size = defrag_config.hash_size * sizeof(DefragTrackerHashRow);
183  if (!(DEFRAG_CHECK_MEMCAP(hash_size))) {
184  SCLogError(SC_ERR_DEFRAG_INIT, "allocating defrag hash failed: "
185  "max defrag memcap is smaller than projected hash size. "
186  "Memcap: %"PRIu64", Hash table size %"PRIu64". Calculate "
187  "total hash size by multiplying \"defrag.hash-size\" with %"PRIuMAX", "
188  "which is the hash bucket size.", defrag_config.memcap, hash_size,
189  (uintmax_t)sizeof(DefragTrackerHashRow));
190  exit(EXIT_FAILURE);
191  }
192  defragtracker_hash = SCCalloc(defrag_config.hash_size, sizeof(DefragTrackerHashRow));
193  if (unlikely(defragtracker_hash == NULL)) {
194  SCLogError(SC_ERR_FATAL, "Fatal error encountered in DefragTrackerInitConfig. Exiting...");
195  exit(EXIT_FAILURE);
196  }
197  memset(defragtracker_hash, 0, defrag_config.hash_size * sizeof(DefragTrackerHashRow));
198 
199  uint32_t i = 0;
200  for (i = 0; i < defrag_config.hash_size; i++) {
201  DRLOCK_INIT(&defragtracker_hash[i]);
202  }
203  (void) SC_ATOMIC_ADD(defrag_memuse, (defrag_config.hash_size * sizeof(DefragTrackerHashRow)));
204 
205  if (quiet == FALSE) {
206  SCLogInfo("allocated %llu bytes of memory for the defrag hash... "
207  "%" PRIu32 " buckets of size %" PRIuMAX "",
208  SC_ATOMIC_GET(defrag_memuse), defrag_config.hash_size,
209  (uintmax_t)sizeof(DefragTrackerHashRow));
210  }
211 
212  if ((ConfGet("defrag.prealloc", &conf_val)) == 1)
213  {
214  if (ConfValIsTrue(conf_val)) {
215  /* pre allocate defrag trackers */
216  for (i = 0; i < defrag_config.prealloc; i++) {
217  if (!(DEFRAG_CHECK_MEMCAP(sizeof(DefragTracker)))) {
218  SCLogError(SC_ERR_DEFRAG_INIT, "preallocating defrag trackers failed: "
219  "max defrag memcap reached. Memcap %"PRIu64", "
220  "Memuse %"PRIu64".", defrag_config.memcap,
221  ((uint64_t)SC_ATOMIC_GET(defrag_memuse) + (uint64_t)sizeof(DefragTracker)));
222  exit(EXIT_FAILURE);
223  }
224 
225  DefragTracker *h = DefragTrackerAlloc();
226  if (h == NULL) {
227  SCLogError(SC_ERR_DEFRAG_INIT, "preallocating defrag failed: %s", strerror(errno));
228  exit(EXIT_FAILURE);
229  }
230  DefragTrackerEnqueue(&defragtracker_spare_q,h);
231  }
232  if (quiet == FALSE) {
233  SCLogInfo("preallocated %" PRIu32 " defrag trackers of size %" PRIuMAX "",
234  defragtracker_spare_q.len, (uintmax_t)sizeof(DefragTracker));
235  }
236  }
237  }
238 
239  if (quiet == FALSE) {
240  SCLogInfo("defrag memory usage: %llu bytes, maximum: %"PRIu64,
241  SC_ATOMIC_GET(defrag_memuse), defrag_config.memcap);
242  }
243 
244  return;
245 }

253 /** \brief shutdown the flow engine
254  * \warning Not thread safe */
255 void DefragHashShutdown(void)
256 {
257  DefragTracker *dt;
258  uint32_t u;
259 
260  DefragTrackerPrintStats();
261 
262  /* free spare queue */
263  while((dt = DefragTrackerDequeue(&defragtracker_spare_q))) {
264  BUG_ON(SC_ATOMIC_GET(dt->use_cnt) > 0);
265  DefragTrackerFree(dt);
266  }
267 
268  /* clear and free the hash */
269  if (defragtracker_hash != NULL) {
270  for (u = 0; u < defrag_config.hash_size; u++) {
271  dt = defragtracker_hash[u].head;
272  while (dt) {
273  DefragTracker *n = dt->hnext;
274  DefragTrackerClearMemory(dt);
275  DefragTrackerFree(dt);
276  dt = n;
277  }
278 
279  DRLOCK_DESTROY(&defragtracker_hash[u]);
280  }
281  SCFree(defragtracker_hash);
282  defragtracker_hash = NULL;
283  }
284  (void) SC_ATOMIC_SUB(defrag_memuse, defrag_config.hash_size * sizeof(DefragTrackerHashRow));
285  DefragTrackerQueueDestroy(&defragtracker_spare_q);
286 
287  SC_ATOMIC_DESTROY(defragtracker_prune_idx);
288  SC_ATOMIC_DESTROY(defrag_memuse);
289  SC_ATOMIC_DESTROY(defragtracker_counter);
290  //SC_ATOMIC_DESTROY(flow_flags);
291  return;
292 }

294 /** \brief compare two raw ipv6 addrs
295  *
296  * \note we don't care about the real ipv6 ip's, this is just
297  * to consistently fill the DefragHashKey6 struct, without all
298  * the ntohl calls.
299  *
300  * \warning do not use elsewhere unless you know what you're doing.
301  * detect-engine-address-ipv6.c's AddressIPv6GtU32 is likely
302  * what you are looking for.
303  */
304 static inline int DefragHashRawAddressIPv6GtU32(uint32_t *a, uint32_t *b)
305 {
306  int i;
307 
308  for (i = 0; i < 4; i++) {
309  if (a[i] > b[i])
310  return 1;
311  if (a[i] < b[i])
312  break;
313  }
314 
315  return 0;
316 }

340 /* calculate the hash key for this packet
341  *
342  * we're using:
343  * hash_rand -- set at init time
344  * source address
345  * destination address
346  * id
347  * vlan_id
348  */
349 static inline uint32_t DefragHashGetKey(Packet *p)
350 {
351  uint32_t key;
352 
353  if (p->ip4h != NULL) {
354  DefragHashKey4 dhk;
355  if (p->src.addr_data32[0] > p->dst.addr_data32[0]) {
356  dhk.src = p->src.addr_data32[0];
357  dhk.dst = p->dst.addr_data32[0];
358  } else {
359  dhk.src = p->dst.addr_data32[0];
360  dhk.dst = p->src.addr_data32[0];
361  }
362  dhk.id = (uint32_t)IPV4_GET_IPID(p);
363  dhk.vlan_id[0] = p->vlan_id[0];
364  dhk.vlan_id[1] = p->vlan_id[1];
365 
366  uint32_t hash = hashword(dhk.u32, 4, defrag_config.hash_rand);
367  key = hash % defrag_config.hash_size;
368  } else if (p->ip6h != NULL) {
369  DefragHashKey6 dhk;
370  if (DefragHashRawAddressIPv6GtU32(p->src.addr_data32, p->dst.addr_data32)) {
371  dhk.src[0] = p->src.addr_data32[0];
372  dhk.src[1] = p->src.addr_data32[1];
373  dhk.src[2] = p->src.addr_data32[2];
374  dhk.src[3] = p->src.addr_data32[3];
375  dhk.dst[0] = p->dst.addr_data32[0];
376  dhk.dst[1] = p->dst.addr_data32[1];
377  dhk.dst[2] = p->dst.addr_data32[2];
378  dhk.dst[3] = p->dst.addr_data32[3];
379  } else {
380  dhk.src[0] = p->dst.addr_data32[0];
381  dhk.src[1] = p->dst.addr_data32[1];
382  dhk.src[2] = p->dst.addr_data32[2];
383  dhk.src[3] = p->dst.addr_data32[3];
384  dhk.dst[0] = p->src.addr_data32[0];
385  dhk.dst[1] = p->src.addr_data32[1];
386  dhk.dst[2] = p->src.addr_data32[2];
387  dhk.dst[3] = p->src.addr_data32[3];
388  }
389  dhk.id = IPV6_EXTHDR_GET_FH_ID(p);
390  dhk.vlan_id[0] = p->vlan_id[0];
391  dhk.vlan_id[1] = p->vlan_id[1];
392 
393  uint32_t hash = hashword(dhk.u32, 10, defrag_config.hash_rand);
394  key = hash % defrag_config.hash_size;
395  } else
396  key = 0;
397 
398  return key;
399 }

477 /* DefragGetTrackerFromHash
478  *
479  * Hash retrieval function for trackers. Looks up the hash bucket containing the
480  * tracker pointer. Then compares the packet with the found tracker to see if it is
481  * the tracker we need. If it isn't, walk the list until the right tracker is found.
482  *
483  * returns a *LOCKED* tracker or NULL
484  */
485 DefragTracker *DefragGetTrackerFromHash (Packet *p)
486 {
487  DefragTracker *dt = NULL;
488 
489  /* get the key to our bucket */
490  uint32_t key = DefragHashGetKey(p);
491  /* get our hash bucket and lock it */
492  DefragTrackerHashRow *hb = &defragtracker_hash[key];
493  DRLOCK_LOCK(hb);
494 
495  /* see if the bucket already has a tracker */
496  if (hb->head == NULL) {
497  dt = DefragTrackerGetNew(p);
498  if (dt == NULL) {
499  DRLOCK_UNLOCK(hb);
500  return NULL;
501  }
502 
503  /* tracker is locked */
504  hb->head = dt;
505  hb->tail = dt;
506 
507  /* got one, now lock, initialize and return */
508  DefragTrackerInit(dt,p);
509 
510  DRLOCK_UNLOCK(hb);
511  return dt;
512  }
513 
514  /* ok, we have a tracker in the bucket. Let's find out if it is our tracker */
515  dt = hb->head;
516 
517  /* see if this is the tracker we are looking for */
518  if (dt->remove || DefragTrackerCompare(dt, p) == 0) {
519  DefragTracker *pdt = NULL; /* previous tracker */
520 
521  while (dt) {
522  pdt = dt;
523  dt = dt->hnext;
524 
525  if (dt == NULL) {
526  dt = pdt->hnext = DefragTrackerGetNew(p);
527  if (dt == NULL) {
528  DRLOCK_UNLOCK(hb);
529  return NULL;
530  }
531  hb->tail = dt;
532 
533  /* tracker is locked */
534 
535  dt->hprev = pdt;
536 
537  /* initialize and return */
538  DefragTrackerInit(dt,p);
539 
540  DRLOCK_UNLOCK(hb);
541  return dt;
542  }
543 
544  if (DefragTrackerCompare(dt, p) != 0) {
545  /* we found our tracker, lets put it on top of the
546  * hash list -- this rewards active trackers */
547  if (dt->hnext) {
548  dt->hnext->hprev = dt->hprev;
549  }
550  if (dt->hprev) {
551  dt->hprev->hnext = dt->hnext;
552  }
553  if (dt == hb->tail) {
554  hb->tail = dt->hprev;
555  }
556 
557  dt->hnext = hb->head;
558  dt->hprev = NULL;
559  hb->head->hprev = dt;
560  hb->head = dt;
561 
562  /* found our tracker, lock & return */
563  SCMutexLock(&dt->lock);
564  (void) DefragTrackerIncrUsecnt(dt);
565  DRLOCK_UNLOCK(hb);
566  return dt;
567  }
568  }
569  }
570 
571  /* lock & return */
572  SCMutexLock(&dt->lock);
573  (void) DefragTrackerIncrUsecnt(dt);
574  DRLOCK_UNLOCK(hb);
575  return dt;
576 }

578 /** \brief look up a tracker in the hash
579  *
580  * \param a address to look up
581  *
582  * \retval h *LOCKED* tracker or NULL
583  */
584 DefragTracker *DefragLookupTrackerFromHash (Packet *p)
585 {
586  DefragTracker *dt = NULL;
587 
588  /* get the key to our bucket */
589  uint32_t key = DefragHashGetKey(p);
590  /* get our hash bucket and lock it */
591  DefragTrackerHashRow *hb = &defragtracker_hash[key];
592  DRLOCK_LOCK(hb);
593 
594  /* see if the bucket already has a tracker */
595  if (hb->head == NULL) {
596  DRLOCK_UNLOCK(hb);
597  return dt;
598  }
599

601  dt = hb->head;
602 
603  /* see if this is the tracker we are looking for */
604  if (DefragTrackerCompare(dt, p) == 0) {
605  while (dt) {
606  dt = dt->hnext;
607 
608  if (dt == NULL) {
609  DRLOCK_UNLOCK(hb);
610  return dt;
611  }
612

614  /* we found our tracker, lets put it on top of the
615  * hash list -- this rewards active tracker */
616  if (dt->hnext) {
617  dt->hnext->hprev = dt->hprev;
618  }
619  if (dt->hprev) {
620  dt->hprev->hnext = dt->hnext;
621  }
622  if (dt == hb->tail) {
623  hb->tail = dt->hprev;
624  }
625 
626  dt->hnext = hb->head;
627  dt->hprev = NULL;
628  hb->head->hprev = dt;
629  hb->head = dt;
630 
631  /* found our tracker, lock & return */
632  SCMutexLock(&dt->lock);

647 /** \internal
648  * \brief Get a tracker from the hash directly.
649  *
650  * Called in conditions where the spare queue is empty and memcap is reached.
651  *
652  * Walks the hash until a tracker can be freed. "defragtracker_prune_idx" atomic int makes
653  * sure we don't start at the top each time since that would clear the top of
654  * the hash leading to longer and longer search times under high pressure (observed).
655  *
656  * \retval dt tracker or NULL
657  */
658 static DefragTracker *DefragTrackerGetUsedDefragTracker(void)
659 {
660  uint32_t idx = SC_ATOMIC_GET(defragtracker_prune_idx) % defrag_config.hash_size;
661  uint32_t cnt = defrag_config.hash_size;
662 
663  while (cnt--) {
664  if (++idx >= defrag_config.hash_size)
665  idx = 0;
666 
667  DefragTrackerHashRow *hb = &defragtracker_hash[idx];
668 
669  if (DRLOCK_TRYLOCK(hb) != 0)
670  continue;
671 
672  DefragTracker *dt = hb->tail;
673  if (dt == NULL) {
674  DRLOCK_UNLOCK(hb);
675  continue;
676  }
677 
678  if (SCMutexTrylock(&dt->lock) != 0) {
679  DRLOCK_UNLOCK(hb);
680  continue;
681  }
682 
683  /** never prune a tracker that is used by a packets
684  * we are currently processing in one of the threads */
685  if (SC_ATOMIC_GET(dt->use_cnt) > 0) {
686  DRLOCK_UNLOCK(hb);
687  SCMutexUnlock(&dt->lock);
688  continue;
689  }
690 
691  /* remove from the hash */
692  if (dt->hprev != NULL)
693  dt->hprev->hnext = dt->hnext;
694  if (dt->hnext != NULL)
695  dt->hnext->hprev = dt->hprev;
696  if (hb->head == dt)
697  hb->head = dt->hnext;
698  if (hb->tail == dt)
699  hb->tail = dt->hprev;
700 
701  dt->hnext = NULL;
702  dt->hprev = NULL;
703  DRLOCK_UNLOCK(hb);
704 
705  DefragTrackerClearMemory(dt);
706 
707  SCMutexUnlock(&dt->lock);
708 
709  (void) SC_ATOMIC_ADD(defragtracker_prune_idx, (defrag_config.hash_size - cnt));
710  return dt;
711  }
712 
713  return NULL;
714 }
