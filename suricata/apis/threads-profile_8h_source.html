51 #define SCMutexLock_profile(mut) ({ \
52  mutex_lock_cnt++; \
53  int retl = 0; \
54  int cont = 0; \
55  uint64_t mutex_lock_start = UtilCpuGetTicks(); \
56  if (pthread_mutex_trylock((mut)) != 0) { \
57  mutex_lock_contention++; \
58  cont = 1; \
59  retl = pthread_mutex_lock(mut); \
60  } \
61  uint64_t mutex_lock_end = UtilCpuGetTicks(); \
62  mutex_lock_wait_ticks += (uint64_t)(mutex_lock_end - mutex_lock_start); \
63  \
64  if (locks_idx < PROFILING_MAX_LOCKS && record_locks) { \
65  locks[locks_idx].file = (char *)__FILE__; \
66  locks[locks_idx].func = (char *)__func__; \
67  locks[locks_idx].line = (int)__LINE__; \
68  locks[locks_idx].type = LOCK_MUTEX; \
69  locks[locks_idx].cont = cont; \
70  locks[locks_idx].ticks = (uint64_t)(mutex_lock_end - mutex_lock_start); \
71  locks_idx++; \
72  } \
73  retl; \
74 })
