75 static int LuaGetFlowvar(lua_State *luastate)
76 {
77  uint16_t idx;
78  int id;
79  Flow *f;
80  FlowVar *fv;
81  DetectLuaData *ld;
82  int flow_lock = 0;
83 
84  /* need luajit data for id -> idx conversion */
85  lua_pushlightuserdata(luastate, (void *)&luaext_key_ld);
86  lua_gettable(luastate, LUA_REGISTRYINDEX);
87  ld = lua_touserdata(luastate, -1);
88  SCLogDebug("ld %p", ld);
89  if (ld == NULL) {
90  lua_pushnil(luastate);
91  lua_pushstring(luastate, "internal error: no ld");
92  return 2;
93  }
94 
95  /* need flow and lock hint */
96  f = LuaStateGetFlow(luastate, &flow_lock);
97  if (f == NULL) {
98  lua_pushnil(luastate);
99  lua_pushstring(luastate, "no flow");
100  return 2;
101  }
102 
103  /* need flowvar idx */
104  if (!lua_isnumber(luastate, 1)) {
105  lua_pushnil(luastate);
106  lua_pushstring(luastate, "1st arg not a number");
107  return 2;
108  }
109  id = lua_tonumber(luastate, 1);
110  if (id < 0 || id >= DETECT_LUAJIT_MAX_FLOWVARS) {
111  lua_pushnil(luastate);
112  lua_pushstring(luastate, "flowvar id out of range");
113  return 2;
114  }
115  idx = ld->flowvar[id];
116  if (idx == 0) {
117  lua_pushnil(luastate);
118  lua_pushstring(luastate, "flowvar id uninitialized");
119  return 2;
120  }
121 
122  /* lookup var */
123  if (flow_lock == LUA_FLOW_NOT_LOCKED_BY_PARENT)
124  FLOWLOCK_RDLOCK(f);
125 
126  fv = FlowVarGet(f, idx);
127  if (fv == NULL) {
128  if (flow_lock == LUA_FLOW_NOT_LOCKED_BY_PARENT)
129  FLOWLOCK_UNLOCK(f);
130 
131  lua_pushnil(luastate);
132  lua_pushstring(luastate, "no flow var");
133  return 2;
134  }
135 
136  LuaPushStringBuffer(luastate, (const uint8_t *)fv->data.fv_str.value,
137  (size_t)fv->data.fv_str.value_len);
138 
139  if (flow_lock == LUA_FLOW_NOT_LOCKED_BY_PARENT)
140  FLOWLOCK_UNLOCK(f);
141 
142  return 1;

565  Flow *f, int flow_locked, Packet *p, uint8_t flags)

571  lua_pushlightuserdata(lua_state, (void *)ld);

573 
574  /* detection engine thread ctx */

576  lua_pushlightuserdata(lua_state, (void *)det_ctx);
577  lua_settable(lua_state, LUA_REGISTRYINDEX);
578 
579  LuaStateSetFlow(lua_state, f, flow_locked);
580 
581  if (det_ctx->tx_id_set && flow_locked == LUA_FLOW_LOCKED_BY_PARENT) {
582  if (f && f->alstate) {
583  void *txptr = AppLayerParserGetTx(f->proto, f->alproto, f->alstate, det_ctx->tx_id);
584  if (txptr) {
585  LuaStateSetTX(lua_state, txptr);
586  }
587  }
588  }
589 
590  if (p != NULL)
591  LuaStateSetPacket(lua_state, p);
592 
593  LuaStateSetDirection(lua_state, (flags & STREAM_TOSERVER));
