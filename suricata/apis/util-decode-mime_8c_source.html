498  *
499  * \param stack The top of the stack
500  *
501  * \return pointer to the next node, otherwise NULL if no nodes remain
502  */
503 static MimeDecStackNode * PopStack(MimeDecStack *stack)
504 {
505  /* Move stack pointer to next item */
506  MimeDecStackNode *curr = stack->top;
507  if (curr != NULL) {
508  curr = curr->next;
509  }
510 
511  /* Always free alloc'd memory */
512  SCFree(stack->top->bdef);
513 
514  /* Now move head to free nodes list */
515  if (stack->free_nodes_cnt < STACK_FREE_NODES) {
516  stack->top->next = stack->free_nodes;
517  stack->free_nodes = stack->top;
518  stack->free_nodes_cnt++;
519  } else {
520  SCFree(stack->top);
521  }
522  stack->top = curr;
523 
524  /* Return a pointer to the top of the stack */
525  return curr;
526 }

692 /**
693  * \brief Get a line (CRLF or just CR or LF) from a buffer (similar to GetToken)
694  *
695  * \param buf The input buffer (not null-terminated)
696  * \param blen The length of the input buffer
697  * \param remainPtr Pointer to remaining after tokenizing iteration
698  * \param tokLen Output token length (if non-null line)
699  *
700  * \return Pointer to line
701  */
702 static uint8_t * GetLine(uint8_t *buf, uint32_t blen, uint8_t **remainPtr,
703  uint32_t *tokLen)
704 {
705  uint32_t i;
706  uint8_t *tok;
707 
708  /* So that it can be used just like strtok_r */
709  if (buf == NULL) {
710  buf = *remainPtr;
711  } else {
712  *remainPtr = buf;
713  }
714  if (buf == NULL)
715  return NULL;
716 
717  tok = buf;
718 
719  /* length must be specified */
720  for (i = 0; i < blen && buf[i] != 0; i++) {
721 
722  /* Found delimiter */
723  if (buf[i] == CR || buf[i] == LF) {
724 
725  /* Add another if we find either CRLF or LFCR */
726  *remainPtr += (i + 1);
727  if ((i + 1 < blen) && buf[i] != buf[i + 1] &&
728  (buf[i + 1] == CR || buf[i + 1] == LF)) {
729  (*remainPtr)++;
730  }
731  break;
732  }
733  }
734 
735  /* If no delimiter found, then point to end of buffer */
736  if (buf == *remainPtr) {
737  (*remainPtr) += i;
738  }
739 
740  /* Calculate token length */
741  *tokLen = (buf + i) - tok;
742 
743  return tok;
744 }

746 /**
747  * \brief Get token from buffer and return pointer to it
748  *
749  * \param buf The input buffer (not null-terminated)
750  * \param blen The length of the input buffer
751  * \param delims Character delimiters (null-terminated)
752  * \param remainPtr Pointer to remaining after tokenizing iteration
753  * \param tokLen Output token length (if non-null line)
754  *
755  * \return Pointer to token, or NULL if not found
756  */
757 static uint8_t * GetToken(uint8_t *buf, uint32_t blen, const char *delims,
758  uint8_t **remainPtr, uint32_t *tokenLen)
759 {
760  uint32_t i, j, delimFound = 0;
761  uint8_t *tok = NULL;
762 
763  /* So that it can be used just like strtok_r */
764  if (buf == NULL) {
765  buf = *remainPtr;
766  } else {
767  *remainPtr = buf;
768  }
769  if (buf == NULL)
770  return NULL;
771 
772  /* Must specify length */
773  for (i = 0; i < blen && buf[i] != 0; i++) {
774 
775  /* Look for delimiters */
776  for (j = 0; delims[j] != 0; j++) {
777  if (buf[i] == delims[j]) {
778  /* Data must be found before delimiter matters */
779  if (tok != NULL) {
780  (*remainPtr) += (i + 1);
781  }
782  delimFound = 1;
783  break;
784  }
785  }

1445  state->data_chunk_len += EOL_LEN;
1446  entity->decoded_body_len += EOL_LEN;
1447  }
1448  } else if (remaining > 1) {
1449  /* If last character handle as soft line break by ignoring,
1450  otherwise process as escaped '=' character */
1451 
1452  /* Not enough characters */
1453  if (remaining < 3) {
1454  entity->anomaly_flags |= ANOM_INVALID_QP;
1455  state->msg->anomaly_flags |= ANOM_INVALID_QP;
1456  SCLogDebug("Error: Quoted-printable decoding failed");
1457  } else {
1458  h1 = *(buf + offset + 1);
1459  res = DecodeQPChar(h1);
1460  if (res < 0) {
1461  entity->anomaly_flags |= ANOM_INVALID_QP;
1462  state->msg->anomaly_flags |= ANOM_INVALID_QP;
1463  SCLogDebug("Error: Quoted-printable decoding failed");
1464  } else {
1465  val = (res << 4); /* Shift result left */
1466  h2 = *(buf + offset + 2);
1467  res = DecodeQPChar(h2);
1468  if (res < 0) {
1469  entity->anomaly_flags |= ANOM_INVALID_QP;
1470  state->msg->anomaly_flags |= ANOM_INVALID_QP;
1471  SCLogDebug("Error: Quoted-printable decoding failed");
1472  } else {
1473  /* Decoding sequence succeeded */
1474  val += res;
1475 
1476  state->data_chunk[state->data_chunk_len] = val;
1477  state->data_chunk_len++;
1478  entity->decoded_body_len++;
1479 
1480  /* Add CRLF sequence if end of line */
1481  if (remaining == 3) {
1482  memcpy(state->data_chunk + state->data_chunk_len,
1483  CRLF, EOL_LEN);
1484  state->data_chunk_len += EOL_LEN;
1485  entity->decoded_body_len += EOL_LEN;
1486  }
1487 
1488  /* Account for extra 2 characters in 3-characted QP
1489  * sequence */
1490  remaining -= 2;
1491  offset += 2;
1492  }
1493  }
1494  }
1495  }
1496 
1497  /* Change by 1 */
1498  remaining--;
1499  offset++;
1500 
1501  /* If buffer full, then invoke callback */
1502  if (DATA_CHUNK_SIZE - state->data_chunk_len < EOL_LEN + 1) {
1503 
1504  /* Invoke pre-processor and callback */
1505  ret = ProcessDecodedDataChunk(state->data_chunk, state->data_chunk_len,
1506  state);
1507  if (ret != MIME_DEC_OK) {
1508  SCLogDebug("Error: ProcessDecodedDataChunk() function "
1509  "failed");
1510  }
1511  }
1512  }
1513 
1514  return ret;
1515 }

1523  * \param state The current parser state

1526  */

1533 
1534  SCLogDebug("Processing body line");
1535

1537  entity->body_len += len + 2; /* With CRLF */

1540  MimeDecConfig *mdcfg = MimeDecGetConfig();
1541  if (mdcfg != NULL && mdcfg->decode_base64 &&
1542  (entity->ctnt_flags & CTNT_IS_BASE64)) {
1543

1545  if (ret != MIME_DEC_OK) {
1546  SCLogDebug("Error: ProcessBase64BodyLine() function failed");

1553  SCLogDebug("Error: ProcessQuotedPrintableBodyLine() function "

1590  SCLogDebug("Error: ProcessDecodedDataChunk() function "
1591  "failed");
1592  }
1593  }
1594 
1595  remaining -= tobuf;
1596  offset += tobuf;
1597  }
1598  }
1599 
1600  return ret;

1603 /**
1604  * \brief Find the start of a header name on the current line
1605  *
1606  * \param buf The input line (not null-terminated)
1607  * \param blen The length of the input line
1608  * \param glen The output length of the header name
1609  *
1610  * \return Pointer to header name, or NULL if not found
1611  */
1612 static uint8_t * FindMimeHeaderStart(const uint8_t *buf, uint32_t blen, uint32_t *hlen)
1613 {

1615  uint8_t *hname = NULL;
1616 
1617  /* Init */
1618  *hlen = 0;
1619 
1620  /* Look for sequence of printable characters followed by ':', or
1621  CRLF then printable characters followed by ':' */
1622  for (i = 0; i < blen && buf[i] != 0; i++) {
1623 
1624  /* If ready for printable characters and found one, then increment */
1625  if (buf[i] != COLON && buf[i] >= PRINTABLE_START &&
1626  buf[i] <= PRINTABLE_END) {
1627  valid++;
1628  } else if (valid > 0 && buf[i] == COLON) {
1629  /* If ready for printable characters, found some, and found colon
1630  * delimiter, then a match is found */
1631  hname = (uint8_t *) buf + i - valid;
1632  *hlen = valid;
1633  break;
1634  } else {
1635  /* Otherwise reset and quit */
1636  break;
1637  }
1638  }
1639 
1640  return hname;

1643 /**
1644  * \brief Find full header name and value on the current line based on the
1645  * current state
1646  *
1647  * \param buf The current line (no CRLF)
1648  * \param blen The length of the current line
1649  * \param state The current state
1650  *
1651  * \return MIME_DEC_OK on success, otherwise < 0 on failure
1652  */
1653 static int FindMimeHeader(const uint8_t *buf, uint32_t blen,
1654  MimeDecParseState *state)
1655 {
1656  int ret = MIME_DEC_OK;
1657  uint8_t *hname, *hval = NULL;
1658  DataValue *dv;
1659  uint32_t hlen, vlen;
1660  int finish_header = 0, new_header = 0;
1661  MimeDecConfig *mdcfg = MimeDecGetConfig();
1662 
1663  /* Find first header */
1664  hname = FindMimeHeaderStart(buf, blen, &hlen);
1665  if (hname != NULL) {
1666 
1667  /* Warn and track but don't do anything yet */
1668  if (hlen > MAX_HEADER_NAME) {
1669  state->stack->top->data->anomaly_flags |= ANOM_LONG_HEADER_NAME;
1670  state->msg->anomaly_flags |= ANOM_LONG_HEADER_NAME;
1671  SCLogDebug("Error: Header name exceeds limit (%u > %u)",
1672  hlen, MAX_HEADER_NAME);
1673  }
1674 
1675  /* Value starts after 'header:' (normalize spaces) */
1676  hval = hname + hlen + 1;
1677  if (hval - buf >= (int)blen) {
1678  SCLogDebug("No Header value found");
1679  hval = NULL;
1680  } else {
1681  while (hval[0] == ' ') {
1682 
1683  /* If last character before end of bounds, set to NULL */
1684  if (hval - buf >= (int)blen - 1) {
1685  SCLogDebug("No Header value found");
1686  hval = NULL;
1687  break;
1688  }
1689 
1690  hval++;
1691  }
1692  }
1693 
1694  /* If new header found, then previous header is finished */
1695  if (state->state_flag == HEADER_STARTED) {
1696  finish_header = 1;
1697  }
1698 
1699  /* Now process new header */
1700  new_header = 1;
1701 
1702  /* Must wait for next line to determine if finished */
1703  state->state_flag = HEADER_STARTED;
1704  } else if (blen == 0) {
1705  /* Found body */
1706  /* No more headers */
1707  state->state_flag = HEADER_DONE;
1708 
1709  finish_header = 1;
1710 
1711  SCLogDebug("All Header processing finished");
1712  } else if (state->state_flag == HEADER_STARTED) {
1713  /* Found multi-line value (ie. Received header) */
1714  /* If max header value exceeded, flag it */
1715  vlen = blen;
1716  if ((mdcfg != NULL) && (state->hvlen + vlen > mdcfg->header_value_depth)) {
1717  SCLogDebug("Error: Header value of length (%u) is too long",
1718  state->hvlen + vlen);
1719  vlen = mdcfg->header_value_depth - state->hvlen;
1720  state->stack->top->data->anomaly_flags |= ANOM_LONG_HEADER_VALUE;
1721  state->msg->anomaly_flags |= ANOM_LONG_HEADER_VALUE;
1722  }
1723  if (vlen > 0) {
1724  dv = AddDataValue(state->hvalue);
1725  if (dv == NULL) {
1726  SCLogError(SC_ERR_MEM_ALLOC, "AddDataValue() function failed");
1727  return MIME_DEC_ERR_MEM;
1728  }
1729  if (state->hvalue == NULL) {
1730  state->hvalue = dv;
1731  }
1732 
1733  dv->value = SCMalloc(vlen);
1734  if (unlikely(dv->value == NULL)) {
1735  SCLogError(SC_ERR_MEM_ALLOC, "Memory allocation failed");
1736  return MIME_DEC_ERR_MEM;
1737  }
1738  memcpy(dv->value, buf, vlen);
1739  dv->value_len = vlen;
1740  state->hvlen += vlen;
1741  }
1742  } else {
1743  /* Likely a body without headers */
1744  SCLogDebug("No headers found");
1745 
1746  state->state_flag = BODY_STARTED;
1747 
1748  /* Flag beginning of body */
1749  state->body_begin = 1;
1750  state->body_end = 0;
1751 
1752  ret = ProcessBodyLine(buf, blen, state);
1753  if (ret != MIME_DEC_OK) {
1754  SCLogDebug("Error: ProcessBodyLine() function failed");
1755  return ret;
1756  }
1757  }
1758 
1759  /* If we need to finish a header, then do so below and then cleanup */
1760  if (finish_header) {
1761  /* Store the header value */
1762  ret = StoreMimeHeader(state);
1763  if (ret != MIME_DEC_OK) {
1764  SCLogDebug("Error: StoreMimeHeader() function failed");
1765  return ret;
1766  }
1767  }
1768 
1769  /* When next header is found, we always create a new one */
1770  if (new_header) {
1771  /* Copy name and value to state */
1772  state->hname = SCMalloc(hlen);
1773  if (unlikely(state->hname == NULL)) {
1774  SCLogError(SC_ERR_MEM_ALLOC, "Memory allocation failed");
1775  return MIME_DEC_ERR_MEM;
1776  }
1777  memcpy(state->hname, hname, hlen);
1778  state->hlen = hlen;
1779 
1780  if (state->hvalue != NULL) {
1781  SCLogDebug("Error: Parser failed due to unexpected header "
1782  "value");
1783  return MIME_DEC_ERR_DATA;
1784  }
1785 
1786  if (hval != NULL) {
1787  /* If max header value exceeded, flag it */
1788  vlen = blen - (hval - buf);
1789  if ((mdcfg != NULL) && (state->hvlen + vlen > mdcfg->header_value_depth)) {
1790  SCLogDebug("Error: Header value of length (%u) is too long",
1791  state->hvlen + vlen);
1792  vlen = mdcfg->header_value_depth - state->hvlen;
1793  state->stack->top->data->anomaly_flags |= ANOM_LONG_HEADER_VALUE;
1794  state->msg->anomaly_flags |= ANOM_LONG_HEADER_VALUE;
1795  }
1796 
1797  if (vlen > 0) {
1798  state->hvalue = AddDataValue(NULL);
1799  if (state->hvalue == NULL) {
1800  SCLogError(SC_ERR_MEM_ALLOC, "AddDataValue() function failed");
1801  return MIME_DEC_ERR_MEM;
1802  }
1803  state->hvalue->value = SCMalloc(vlen);
1804  if (unlikely(state->hvalue->value == NULL)) {
1805  SCLogError(SC_ERR_MEM_ALLOC, "Memory allocation failed");
1806  return MIME_DEC_ERR_MEM;
1807  }
1808  memcpy(state->hvalue->value, hval, vlen);
1809  state->hvalue->value_len = vlen;
1810  state->hvlen += vlen;
1811  }
1812  }
1813  }
1814 
1815  return ret;
1816 }

1849 /**
1850  * \brief Processes the current line for mime headers and also does post-processing
1851  * when all headers found
1852  *
1853  * \param buf The current line
1854  * \param len The length of the line
1855  * \param state The current parser state
1856  *
1857  * \return MIME_DEC_OK on success, otherwise < 0 on failure
1858  */
1859 static int ProcessMimeHeaders(const uint8_t *buf, uint32_t len,
1860  MimeDecParseState *state)
1861 {
1862  int ret = MIME_DEC_OK;
1863  MimeDecField *field;
1864  uint8_t *bptr = NULL, *rptr = NULL;
1865  uint32_t blen = 0;
1866  MimeDecEntity *entity = (MimeDecEntity *) state->stack->top->data;
1867 
1868  /* Look for mime header in current line */
1869  ret = FindMimeHeader(buf, len, state);
1870  if (ret != MIME_DEC_OK) {
1871  SCLogDebug("Error: FindMimeHeader() function failed: %d", ret);
1872  return ret;
1873  }
1874 
1875  /* Post-processing after all headers done */
1876  if (state->state_flag == HEADER_DONE) {
1877  /* First determine encoding by looking at Content-Transfer-Encoding */
1878  field = MimeDecFindField(entity, CTNT_TRAN_STR);
1879  if (field != NULL) {

1881  if (FindBuffer(field->value, field->value_len, (const uint8_t *)BASE64_STR, strlen(BASE64_STR))) {
1882  SCLogDebug("Base64 encoding found");
1883  entity->ctnt_flags |= CTNT_IS_BASE64;
1884  } else if (FindBuffer(field->value, field->value_len, (const uint8_t *)QP_STR, strlen(QP_STR))) {
1885  /* Look for quoted-printable */
1886  SCLogDebug("quoted-printable encoding found");
1887  entity->ctnt_flags |= CTNT_IS_QP;
1888  }
1889  }
1890

1892  field = MimeDecFindField(entity, CTNT_DISP_STR);
1893  if (field != NULL) {
1894  bptr = FindMimeHeaderToken(field, "filename=\"", TOK_END_STR, &blen);
1895  if (bptr != NULL) {
1896  SCLogDebug("File attachment found in disposition");
1897  entity->ctnt_flags |= CTNT_IS_ATTACHMENT;
1898 
1899  /* Copy over using dynamic memory */
1900  entity->filename = SCMalloc(blen);
1901  if (unlikely(entity->filename == NULL)) {
1902  SCLogError(SC_ERR_MEM_ALLOC, "memory allocation failed");
1903  return MIME_DEC_ERR_MEM;
1904  }
1905  memcpy(entity->filename, bptr, blen);
1906  entity->filename_len = blen;
1907  }
1908  }
1909

1911  field = MimeDecFindField(entity, CTNT_TYPE_STR);
1912  if (field != NULL) {
1913  /* Check if child entity boundary definition found */
1914  bptr = FindMimeHeaderToken(field, BND_START_STR, TOK_END_STR, &blen);
1915  if (bptr != NULL) {
1916  state->found_child = 1;
1917  entity->ctnt_flags |= CTNT_IS_MULTIPART;
1918 
1919  if (blen > (BOUNDARY_BUF - 2)) {
1920  state->stack->top->data->anomaly_flags |= ANOM_LONG_BOUNDARY;
1921  return MIME_DEC_ERR_PARSE;
1922  }
1923

1925  state->stack->top->bdef = SCMalloc(blen);
1926  if (unlikely(state->stack->top->bdef == NULL)) {
1927  SCLogError(SC_ERR_MEM_ALLOC, "Memory allocation failed");
1928  return MIME_DEC_ERR_MEM;
1929  }
1930  memcpy(state->stack->top->bdef, bptr, blen);
1931  state->stack->top->bdef_len = blen;
1932  }
1933 
1934  /* Look for file name (if not already found) */
1935  if (!(entity->ctnt_flags & CTNT_IS_ATTACHMENT)) {
1936  bptr = FindMimeHeaderToken(field, "name=\"", TOK_END_STR, &blen);
1937  if (bptr != NULL) {
1938  SCLogDebug("File attachment found");
1939  entity->ctnt_flags |= CTNT_IS_ATTACHMENT;
1940 
1941  /* Copy over using dynamic memory */
1942  entity->filename = SCMalloc(blen);
1943  if (unlikely(entity->filename == NULL)) {
1944  SCLogError(SC_ERR_MEM_ALLOC, "memory allocation failed");
1945  return MIME_DEC_ERR_MEM;
1946  }
1947  memcpy(entity->filename, bptr, blen);
1948  entity->filename_len = blen;
1949  }
1950  }
1951 
1952  /* Pull out short-hand content type */
1953  entity->ctnt_type = GetToken(field->value, field->value_len, " \r\n;",
1954  &rptr, &entity->ctnt_type_len);
1955  if (entity->ctnt_type != NULL) {
1956  /* Check for encapsulated message */
1957  if (FindBuffer(entity->ctnt_type, entity->ctnt_type_len,
1958  (const uint8_t *)MSG_STR, strlen(MSG_STR)))
1959  {
1960  SCLogDebug("Found encapsulated message entity");
1961 
1962  entity->ctnt_flags |= CTNT_IS_ENV;
1963 
1964  /* Create and push child to stack */
1965  MimeDecEntity *child = MimeDecAddEntity(entity);
1966  if (child == NULL)
1967  return MIME_DEC_ERR_MEM;

1969  PushStack(state->stack);
1970  state->stack->top->data = child;
1971 
1972  /* Mark as encapsulated child */
1973  state->stack->top->is_encap = 1;
1974 
1975  /* Ready to parse headers */
1976  state->state_flag = HEADER_READY;
1977  } else if (FindBuffer(entity->ctnt_type, entity->ctnt_type_len,
1978  (const uint8_t *)MULTIPART_STR, strlen(MULTIPART_STR)))
1979  {
1980  /* Check for multipart */
1981  SCLogDebug("Found multipart entity");
1982  entity->ctnt_flags |= CTNT_IS_MULTIPART;
1983  } else if (FindBuffer(entity->ctnt_type, entity->ctnt_type_len,
1984  (const uint8_t *)TXT_STR, strlen(TXT_STR)))
1985  {
1986  /* Check for plain text */
1987  SCLogDebug("Found plain text entity");
1988  entity->ctnt_flags |= CTNT_IS_TEXT;
1989  } else if (FindBuffer(entity->ctnt_type, entity->ctnt_type_len,
1990  (const uint8_t *)HTML_STR, strlen(HTML_STR)))
1991  {

1995  }
1996  }
1997  }

2000  field = MimeDecFindField(entity, MSG_ID_STR);
2001  if (field != NULL) {
2002  entity->msg_id = field->value;
2003  entity->msg_id_len = field->value_len;
2004  }
2005

2007  state->body_begin = 1;
2008  state->body_end = 0;
2009  }
2010
