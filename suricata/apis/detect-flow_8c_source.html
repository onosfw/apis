99 /**
100  * \brief This function is used to match flow flags set on a packet with those passed via flow:
101  *
102  * \param t pointer to thread vars
103  * \param det_ctx pointer to the pattern matcher thread
104  * \param p pointer to the current packet
105  * \param m pointer to the sigmatch that we will cast into DetectFlowData
106  *
107  * \retval 0 no match
108  * \retval 1 match
109  */
110 int DetectFlowMatch (ThreadVars *t, DetectEngineThreadCtx *det_ctx, Packet *p, Signature *s, const SigMatchCtx *ctx)
111 {
112  SCEnter();
113 
114  SCLogDebug("pkt %p", p);
115 
116  if (p->flowflags & FLOW_PKT_TOSERVER) {
117  SCLogDebug("FLOW_PKT_TOSERVER");
118  } else if (p->flowflags & FLOW_PKT_TOCLIENT) {
119  SCLogDebug("FLOW_PKT_TOCLIENT");
120  }
121 
122  if (p->flowflags & FLOW_PKT_ESTABLISHED) {
123  SCLogDebug("FLOW_PKT_ESTABLISHED");
124  }
125 
126  uint8_t cnt = 0;
127  const DetectFlowData *fd = (const DetectFlowData *)ctx;
128 
129  if ((fd->flags & DETECT_FLOW_FLAG_TOSERVER) && (p->flowflags & FLOW_PKT_TOSERVER)) {
130  cnt++;
131  } else if ((fd->flags & DETECT_FLOW_FLAG_TOCLIENT) && (p->flowflags & FLOW_PKT_TOCLIENT)) {
132  cnt++;
133  }
134 
135  if ((fd->flags & DETECT_FLOW_FLAG_ESTABLISHED) && (p->flowflags & FLOW_PKT_ESTABLISHED)) {
136  cnt++;
137  } else if (fd->flags & DETECT_FLOW_FLAG_STATELESS) {
138  cnt++;
139  }
140 
141  if (det_ctx->flags & DETECT_ENGINE_THREAD_CTX_STREAM_CONTENT_MATCH) {
142  if (fd->flags & DETECT_FLOW_FLAG_ONLYSTREAM)
143  cnt++;
144  } else {
145  if (fd->flags & DETECT_FLOW_FLAG_NOSTREAM)
146  cnt++;
147  }
148 
149  int ret = (fd->match_cnt == cnt) ? 1 : 0;
150  SCLogDebug("returning %" PRId32 " cnt %" PRIu8 " fd->match_cnt %" PRId32 " fd->flags 0x%02X p->flowflags 0x%02X",
151  ret, cnt, fd->match_cnt, fd->flags, p->flowflags);
152  SCReturnInt(ret);
153 }

155 /**
156  * \brief This function is used to parse flow options passed via flow: keyword
157  *
158  * \param flowstr Pointer to the user provided flow options
159  *
160  * \retval fd pointer to DetectFlowData on success
161  * \retval NULL on failure
162  */
163 DetectFlowData *DetectFlowParse (char *flowstr)
164 {
165  DetectFlowData *fd = NULL;
166  char *args[3] = {NULL,NULL,NULL};
167 #define MAX_SUBSTRINGS 30
168  int ret = 0, res = 0;
169  int ov[MAX_SUBSTRINGS];
170  char str1[16] = "", str2[16] = "", str3[16] = "";
171 
172  ret = pcre_exec(parse_regex, parse_regex_study, flowstr, strlen(flowstr), 0, 0, ov, MAX_SUBSTRINGS);
173  if (ret < 1 || ret > 4) {
174  SCLogError(SC_ERR_PCRE_MATCH, "parse error, ret %" PRId32 ", string %s", ret, flowstr);
175  goto error;
176  }
177 
178  if (ret > 1) {
179  res = pcre_copy_substring((char *)flowstr, ov, MAX_SUBSTRINGS, 1, str1, sizeof(str1));
180  if (res < 0) {
181  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_copy_substring failed");
182  goto error;
183  }
184  args[0] = (char *)str1;
185 
186  if (ret > 2) {
187  res = pcre_copy_substring((char *)flowstr, ov, MAX_SUBSTRINGS, 2, str2, sizeof(str2));
188  if (res < 0) {
189  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_copy_substring failed");
190  goto error;
191  }
192  args[1] = (char *)str2;
193  }
194  if (ret > 3) {
195  res = pcre_copy_substring((char *)flowstr, ov, MAX_SUBSTRINGS, 3, str3, sizeof(str3));
196  if (res < 0) {
197  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_copy_substring failed");
198  goto error;
199  }
200  args[2] = (char *)str3;
201  }
202  }
203 
204  fd = SCMalloc(sizeof(DetectFlowData));
205  if (unlikely(fd == NULL))
206  goto error;
207  fd->flags = 0;
208  fd->match_cnt = 0;
209 
210  int i;
211  for (i = 0; i < (ret - 1); i++) {
212  if (args[i]) {
213  /* inspect our options and set the flags */
214  if (strcasecmp(args[i], "established") == 0) {
215  if (fd->flags & DETECT_FLOW_FLAG_ESTABLISHED) {
216  SCLogError(SC_ERR_FLAGS_MODIFIER, "DETECT_FLOW_FLAG_ESTABLISHED flag is already set");
217  goto error;
218  } else if (fd->flags & DETECT_FLOW_FLAG_STATELESS) {
219  SCLogError(SC_ERR_FLAGS_MODIFIER, "DETECT_FLOW_FLAG_STATELESS already set");
220  goto error;
221  }
222  fd->flags |= DETECT_FLOW_FLAG_ESTABLISHED;
223  } else if (strcasecmp(args[i], "stateless") == 0) {
224  if (fd->flags & DETECT_FLOW_FLAG_STATELESS) {
225  SCLogError(SC_ERR_FLAGS_MODIFIER, "DETECT_FLOW_FLAG_STATELESS flag is already set");
226  goto error;
227  } else if (fd->flags & DETECT_FLOW_FLAG_ESTABLISHED) {
228  SCLogError(SC_ERR_FLAGS_MODIFIER, "cannot set DETECT_FLOW_FLAG_STATELESS, DETECT_FLOW_FLAG_ESTABLISHED already set");
229  goto error;
230  }
231  fd->flags |= DETECT_FLOW_FLAG_STATELESS;
232  } else if (strcasecmp(args[i], "to_client") == 0 || strcasecmp(args[i], "from_server") == 0) {
233  if (fd->flags & DETECT_FLOW_FLAG_TOCLIENT) {
234  SCLogError(SC_ERR_FLAGS_MODIFIER, "cannot set DETECT_FLOW_FLAG_TOCLIENT flag is already set");
235  goto error;
236  } else if (fd->flags & DETECT_FLOW_FLAG_TOSERVER) {
237  SCLogError(SC_ERR_FLAGS_MODIFIER, "cannot set to_client, DETECT_FLOW_FLAG_TOSERVER already set");
238  goto error;
239  }
240  fd->flags |= DETECT_FLOW_FLAG_TOCLIENT;
241  } else if (strcasecmp(args[i], "to_server") == 0 || strcasecmp(args[i], "from_client") == 0){
242  if (fd->flags & DETECT_FLOW_FLAG_TOSERVER) {
243  SCLogError(SC_ERR_FLAGS_MODIFIER, "cannot set DETECT_FLOW_FLAG_TOSERVER flag is already set");
244  goto error;
245  } else if (fd->flags & DETECT_FLOW_FLAG_TOCLIENT) {
246  SCLogError(SC_ERR_FLAGS_MODIFIER, "cannot set to_server, DETECT_FLOW_FLAG_TO_CLIENT flag already set");
247  goto error;
248  }
249  fd->flags |= DETECT_FLOW_FLAG_TOSERVER;
250  } else if (strcasecmp(args[i], "only_stream") == 0) {
251  if (fd->flags & DETECT_FLOW_FLAG_ONLYSTREAM) {
252  SCLogError(SC_ERR_FLAGS_MODIFIER, "cannot set only_stream flag is already set");
253  goto error;
254  } else if (fd->flags & DETECT_FLOW_FLAG_NOSTREAM) {
255  SCLogError(SC_ERR_FLAGS_MODIFIER, "cannot set only_stream flag, DETECT_FLOW_FLAG_NOSTREAM already set");
256  goto error;
257  }
258  fd->flags |= DETECT_FLOW_FLAG_ONLYSTREAM;
259  } else if (strcasecmp(args[i], "no_stream") == 0) {
260  if (fd->flags & DETECT_FLOW_FLAG_NOSTREAM) {
261  SCLogError(SC_ERR_FLAGS_MODIFIER, "cannot set no_stream flag is already set");
262  goto error;
263  } else if (fd->flags & DETECT_FLOW_FLAG_ONLYSTREAM) {
264  SCLogError(SC_ERR_FLAGS_MODIFIER, "cannot set no_stream flag, DETECT_FLOW_FLAG_ONLYSTREAM already set");
265  goto error;
266  }
267  fd->flags |= DETECT_FLOW_FLAG_NOSTREAM;
268  } else {
269  SCLogError(SC_ERR_INVALID_VALUE, "invalid flow option \"%s\"", args[i]);
270  goto error;
271  }
272 
273  fd->match_cnt++;
274  //printf("args[%" PRId32 "]: %s match_cnt: %" PRId32 " flags: 0x%02X\n", i, args[i], fd->match_cnt, fd->flags);
275  }
276  }
277  return fd;
278 
279 error:
280  if (fd != NULL)
281  DetectFlowFree(fd);
282  return NULL;
283 
284 }

286 /**
287  * \brief this function is used to add the parsed flowdata into the current signature
288  *
289  * \param de_ctx pointer to the Detection Engine Context
290  * \param s pointer to the Current Signature
291  * \param flowstr pointer to the user provided flow options
292  *
293  * \retval 0 on Success
294  * \retval -1 on Failure
295  */
296 int DetectFlowSetup (DetectEngineCtx *de_ctx, Signature *s, char *flowstr)
297 {
298  DetectFlowData *fd = NULL;
299  SigMatch *sm = NULL;
300 
301  fd = DetectFlowParse(flowstr);
302  if (fd == NULL)
303  goto error;
304 
305  /*ensure only one flow option*/
306  if (s->init_flags & SIG_FLAG_INIT_FLOW) {
307  SCLogError (SC_ERR_INVALID_SIGNATURE, "A signature may have only one flow option.");
308  goto error;
309  }
310 
311  /* Okay so far so good, lets get this into a SigMatch
312  * and put it in the Signature. */
313  sm = SigMatchAlloc();
314  if (sm == NULL)
315  goto error;
316 
317  sm->type = DETECT_FLOW;
318  sm->ctx = (SigMatchCtx *)fd;
319 
320  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);
321 
322  /* set the signature direction flags */
323  if (fd->flags & DETECT_FLOW_FLAG_TOSERVER) {
324  s->flags |= SIG_FLAG_TOSERVER;
325  } else if (fd->flags & DETECT_FLOW_FLAG_TOCLIENT) {
326  s->flags |= SIG_FLAG_TOCLIENT;
327  } else {
328  s->flags |= SIG_FLAG_TOSERVER;
329  s->flags |= SIG_FLAG_TOCLIENT;
330  }
331  if (fd->flags & DETECT_FLOW_FLAG_ONLYSTREAM) {
332  s->flags |= SIG_FLAG_REQUIRE_STREAM;
333  }
334  if (fd->flags & DETECT_FLOW_FLAG_NOSTREAM) {
335  s->flags |= SIG_FLAG_REQUIRE_PACKET;
336  } else {
337  s->init_flags |= SIG_FLAG_INIT_FLOW;
338  }
339 
340  return 0;
341 
342 error:
343  if (fd != NULL)
344  DetectFlowFree(fd);
345  if (sm != NULL)
346  SCFree(sm);
347  return -1;
348 
349 }
