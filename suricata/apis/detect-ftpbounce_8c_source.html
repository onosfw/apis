77 /**
78  * \brief This function is used to match ftpbounce attacks
79  *
80  * \param payload Payload of the PORT command
81  * \param payload_len Length of the payload
82  * \param ip_orig IP source to check the ftpbounce condition
83  * \param offset offset to the arguments of the PORT command
84  *
85  * \retval 1 if ftpbounce detected, 0 if not
86  */
87 int DetectFtpbounceMatchArgs(uint8_t *payload, uint16_t payload_len,
88  uint32_t ip_orig, uint16_t offset)
89 {
90  SCEnter();
91  SCLogDebug("Checking ftpbounce condition");
92  char *c = NULL;
93  uint16_t i = 0;
94  int octet = 0;
95  int octet_ascii_len = 0;
96  int noctet = 0;
97  uint32_t ip = 0;
98  /* PrintRawDataFp(stdout, payload, payload_len); */
99 
100  if (payload_len < 7) {
101  /* we need at least a differet ip address
102  * in the format 1,2,3,4,x,y where x,y is the port
103  * in two byte representation so let's look at
104  * least for the IP octets in comma separated */
105  return 0;
106  }
107 
108  if (offset + 7 >= payload_len)
109  return 0;
110 
111  c =(char*) payload;
112  if (c == NULL) {
113  SCLogDebug("No payload to check");
114  return 0;
115  }
116 
117  i = offset;
118  /* Search for the first IP octect(Skips "PORT ") */
119  while (i < payload_len && !isdigit((unsigned char)c[i])) i++;
120 
121  for (;i < payload_len && octet_ascii_len < 4 ;i++) {
122  if (isdigit((unsigned char)c[i])) {
123  octet =(c[i] - '0') + octet * 10;
124  octet_ascii_len++;
125  } else {
126  if (octet > 256) {
127  SCLogDebug("Octet not in ip format");
128  return 0;
129  }
130 
131  if (isspace((unsigned char)c[i]))
132  while (i < payload_len && isspace((unsigned char)c[i]) ) i++;
133 
134  if (i < payload_len && c[i] == ',') { /* we have an octet */
135  noctet++;
136  octet_ascii_len = 0;
137  ip =(ip << 8) + octet;
138  octet = 0;
139  } else {
140  SCLogDebug("Unrecognized character '%c'", c[i]);
141  return 0;
142  }
143  if (noctet == 4) {
144  /* Different IP than src, ftp bounce scan */
145  ip = SCByteSwap32(ip);
146 
147  if (ip != ip_orig) {
148  SCLogDebug("Different ip, so Matched ip:%d <-> ip_orig:%d",
149  ip, ip_orig);
150  return 1;
151  }
152  SCLogDebug("Same ip, so no match here");
153  return 0;
154  }
155  }
156  }
157  SCLogDebug("No match");
158  return 0;
159 }

161 /**
162  * \brief This function is used to check matches from the FTP App Layer Parser
163  *
164  * \param t pointer to thread vars
165  * \param det_ctx pointer to the pattern matcher thread
166  * \param p pointer to the current packet
167  * \param m pointer to the sigmatch but we don't use it since ftpbounce
168  * has no options
169  * \retval 0 no match
170  * \retval 1 match

172 int DetectFtpbounceALMatch(ThreadVars *t, DetectEngineThreadCtx *det_ctx,
173  Flow *f, uint8_t flags, void *state, Signature *s,
174  SigMatch *m)
175 {
176  SCEnter();
177  FtpState *ftp_state =(FtpState *)state;
178  if (ftp_state == NULL) {
179  SCLogDebug("no ftp state, no match");
180  SCReturnInt(0);
181  }
182 
183  int ret = 0;
184 
185  if (ftp_state->command == FTP_COMMAND_PORT) {
186  ret = DetectFtpbounceMatchArgs(ftp_state->port_line,
187  ftp_state->port_line_len, f->src.address.address_un_data32[0],
188  ftp_state->arg_offset);
189  }
190 
191  SCReturnInt(ret);
192 }

194 /**
195  * \brief this function is used to add the parsed ftpbounce
196  *
197  * \param de_ctx pointer to the Detection Engine Context
198  * \param s pointer to the Current Signature
199  * \param m pointer to the Current SigMatch
200  * \param ftpbouncestr pointer to the user provided ftpbounce options
201  * currently there are no options.
202  *
203  * \retval 0 on Success
204  * \retval -1 on Failure
205  */
206 int DetectFtpbounceSetup(DetectEngineCtx *de_ctx, Signature *s, char *ftpbouncestr)
207 {
208  SCEnter();
209 
210  SigMatch *sm = NULL;
211 
212  sm = SigMatchAlloc();
213  if (sm == NULL) {
214  goto error;;
215  }
216 
217  sm->type = DETECT_FTPBOUNCE;
218 
219  /* We don't need to allocate any data for ftpbounce here.
220  *
221  * TODO: As a suggestion, maybe we can add a flag in the flow
222  * to set the stream as "bounce detected" for fast Match.
223  * When you do a ftp bounce attack you usually use the same
224  * communication control stream to "setup" various destinations
225  * whithout breaking the connection, so I guess we can make it a bit faster
226  * with a flow flag set lookup in the Match function.
227  */
228  sm->ctx = NULL;
229 
230  if (s->alproto != ALPROTO_UNKNOWN && s->alproto != ALPROTO_FTP) {
231  SCLogError(SC_ERR_CONFLICTING_RULE_KEYWORDS, "rule contains conflicting keywords.");
232  goto error;
233  }
234 
235  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_AMATCH);
236 
237  s->alproto = ALPROTO_FTP;
238  s->flags |= SIG_FLAG_APPLAYER;
239  SCReturnInt(0);
240 
241 error:
242  if (sm != NULL) {
243  SigMatchFree(sm);
244  }
245  SCReturnInt(-1);
246 }
