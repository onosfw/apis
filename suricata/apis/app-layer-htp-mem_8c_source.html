46 void HTPParseMemcap()
47 {
48  char *conf_val;
49 
50  /** set config values for memcap, prealloc and hash_size */
51  if ((ConfGet("app-layer.protocols.http.memcap", &conf_val)) == 1)
52  {
53  if (ParseSizeStringU64(conf_val, &htp_config_memcap) < 0) {
54  SCLogError(SC_ERR_SIZE_PARSE, "Error parsing http.memcap "
55  "from conf file - %s. Killing engine",
56  conf_val);
57  exit(EXIT_FAILURE);
58  }
59  SCLogInfo("HTTP memcap: %"PRIu64, htp_config_memcap);
60  } else {
61  /* default to unlimited */
62  htp_config_memcap = 0;
63  }

66  SC_ATOMIC_INIT(htp_memcap);

93 /**
94  * \brief Check if alloc'ing "size" would mean we're over memcap
95  *
96  * \retval 1 if in bounds
97  * \retval 0 if not in bounds
98  */
99 int HTPCheckMemcap(uint64_t size)
100 {
101  if (htp_config_memcap == 0 || size + SC_ATOMIC_GET(htp_memuse) <= htp_config_memcap)
102  return 1;
103  (void) SC_ATOMIC_ADD(htp_memcap, 1);
104  return 0;
105 }

107 void *HTPMalloc(size_t size)
108 {
109  void *ptr = NULL;
110 
111  if (HTPCheckMemcap((uint32_t)size) == 0)
112  return NULL;
113 
114  ptr = SCMalloc(size);
115 
116  if (unlikely(ptr == NULL))
117  return NULL;
118 
119  HTPIncrMemuse((uint64_t)size);
120 
121  return ptr;
122 }

124 void *HTPRealloc(void *ptr, size_t orig_size, size_t size)
125 {
126  void *rptr = NULL;
127 
128  if (HTPCheckMemcap((uint32_t)(size - orig_size)) == 0)
129  return NULL;
130 
131  rptr = SCRealloc(ptr, size);
132  if (rptr == NULL)
133  return NULL;
134 
135  HTPIncrMemuse((uint64_t)(size - orig_size));
136 
137  return rptr;
138 }
