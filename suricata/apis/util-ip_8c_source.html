29 /**
30  * \brief Validates an IPV4 address and returns the network endian arranged
31  * version of the IPV4 address
32  *
33  * \param addr Pointer to a character string containing an IPV4 address. A
34  * valid IPV4 address is a character string containing a dotted
35  * format of "ddd.ddd.ddd.ddd"
36  *
37  * \retval Pointer to an in_addr instance containing the network endian format
38  * of the IPV4 address
39  * \retval NULL if the IPV4 address is invalid
40  */
41 struct in_addr *ValidateIPV4Address(const char *addr_str)
42 {
43  struct in_addr *addr = NULL;
44 
45  if ( (addr = SCMalloc(sizeof(struct in_addr))) == NULL) {
46  SCLogError(SC_ERR_FATAL, "Fatal error encountered in ValidateIPV4Address. Exiting...");
47  exit(EXIT_FAILURE);
48  }
49 
50  if (inet_pton(AF_INET, addr_str, addr) <= 0) {
51  SCFree(addr);
52  return NULL;
53  }
54 
55  return addr;
56 }

85 /**
86  * \brief Culls the non-netmask portion of the IP address. For example an IP
87  * address 192.168.240.1 would be chopped to 192.168.224.0 against a
88  * netmask value of 19.
89  *
90  * \param stream Pointer the IP address that has to be masked
91  * \param netmask The netmask value (cidr) to which the IP address has to be culled
92  * \param key_bitlen The bitlen of the stream
93  */
94 void MaskIPNetblock(uint8_t *stream, uint8_t netmask, uint16_t key_bitlen)
95 {
96  int mask = 0;
97  int i = 0;
98  int bytes = key_bitlen / 8;
99 
100  for (i = 0; i < bytes; i++) {
101  mask = -1;
102  if ( ((i + 1) * 8) > netmask) {
103  if ( ((i + 1) * 8 - netmask) < 8)
104  mask = -1 << ((i + 1) * 8 - netmask);
105  else
106  mask = 0;
107  }
108  stream[i] &= mask;
109  }
110 
111  return;
112 }
