57 static inline int SMTPCreateSpace(DetectEngineThreadCtx *det_ctx, uint16_t size)
58 {
59  void *ptmp;
60  if (size > det_ctx->smtp_buffers_size) {
61  ptmp = SCRealloc(det_ctx->smtp,
62  (det_ctx->smtp_buffers_size + BUFFER_STEP) * sizeof(FiledataReassembledBody));
63  if (ptmp == NULL) {
64  SCFree(det_ctx->smtp);
65  det_ctx->smtp = NULL;
66  det_ctx->smtp_buffers_size = 0;
67  det_ctx->smtp_buffers_list_len = 0;
68  return -1;
69  }
70  det_ctx->smtp = ptmp;
71 
72  memset(det_ctx->smtp + det_ctx->smtp_buffers_size, 0, BUFFER_STEP * sizeof(FiledataReassembledBody));
73  det_ctx->smtp_buffers_size += BUFFER_STEP;
74  }
75  for (int i = det_ctx->smtp_buffers_list_len; i < (size); i++) {
76  det_ctx->smtp[i].buffer_len = 0;
77  det_ctx->smtp[i].offset = 0;
78  }
79 
80  return 0;
81 }

83 static uint8_t *DetectEngineSMTPGetBufferForTX(uint64_t tx_id,
84  DetectEngineCtx *de_ctx,
85  DetectEngineThreadCtx *det_ctx,
86  Flow *f, File *curr_file,
87  uint8_t flags,
88  uint32_t *buffer_len,
89  uint32_t *stream_start_offset)
90 {
91  int index = 0;
92  uint8_t *buffer = NULL;
93  *buffer_len = 0;
94  *stream_start_offset = 0;
95  FileData *curr_chunk = NULL;
96 
97  if (det_ctx->smtp_buffers_list_len == 0) {
98  if (SMTPCreateSpace(det_ctx, 1) < 0)
99  goto end;
100  index = 0;
101 
102  if (det_ctx->smtp_buffers_list_len == 0) {
103  det_ctx->smtp_start_tx_id = tx_id;
104  }
105  det_ctx->smtp_buffers_list_len++;
106  } else {
107  if ((tx_id - det_ctx->smtp_start_tx_id) < det_ctx->smtp_buffers_list_len) {
108  if (det_ctx->smtp[(tx_id - det_ctx->smtp_start_tx_id)].buffer_len != 0) {
109  *buffer_len = det_ctx->smtp[(tx_id - det_ctx->smtp_start_tx_id)].buffer_len;
110  *stream_start_offset = det_ctx->smtp[(tx_id - det_ctx->smtp_start_tx_id)].offset;
111  return det_ctx->smtp[(tx_id - det_ctx->smtp_start_tx_id)].buffer;
112  }
113  } else {
114  if (SMTPCreateSpace(det_ctx, (tx_id - det_ctx->smtp_start_tx_id) + 1) < 0)
115  goto end;
116 
117  if (det_ctx->smtp_buffers_list_len == 0) {
118  det_ctx->smtp_start_tx_id = tx_id;
119  }
120  det_ctx->smtp_buffers_list_len++;
121  }
122  index = (tx_id - det_ctx->smtp_start_tx_id);
123  }
124 
125  /* no new data */
126  if (curr_file->content_inspected == curr_file->content_len_so_far) {
127  SCLogDebug("no new data");
128  goto end;
129  }
130 
131  curr_chunk = curr_file->chunks_head;
132  if (curr_chunk == NULL) {
133  SCLogDebug("no data chunks to inspect for this transaction");
134  goto end;
135  }
136 
137  if ((smtp_config.content_limit == 0 ||
138  curr_file->content_len_so_far < smtp_config.content_limit) &&
139  curr_file->content_len_so_far < smtp_config.content_inspect_min_size &&
140  !(flags & STREAM_EOF)) {
141  SCLogDebug("we still haven't seen the entire content. "
142  "Let's defer content inspection till we see the "
143  "entire content.");
144  goto end;
145  }
146 
147  int first = 1;
148  curr_chunk = curr_file->chunks_head;
149  while (curr_chunk != NULL) {
150  /* see if we can filter out chunks */
151  if (curr_file->content_inspected > 0) {
152  if (curr_chunk->stream_offset < curr_file->content_inspected) {
153  if ((curr_file->content_inspected - curr_chunk->stream_offset) > smtp_config.content_inspect_window) {
154  curr_chunk = curr_chunk->next;
155  continue;
156  } else {
157  /* include this one */
158  }
159  } else {
160  /* include this one */
161  }
162  }
163 
164  if (first) {
165  det_ctx->smtp[index].offset = curr_chunk->stream_offset;
166  first = 0;
167  }
168 
169  /* see if we need to grow the buffer */
170  if (det_ctx->smtp[index].buffer == NULL || (det_ctx->smtp[index].buffer_len + curr_chunk->len) > det_ctx->smtp[index].buffer_size) {
171  void *ptmp;
172  det_ctx->smtp[index].buffer_size += curr_chunk->len * 2;
173 
174  if ((ptmp = SCRealloc(det_ctx->smtp[index].buffer, det_ctx->smtp[index].buffer_size)) == NULL) {
175  SCFree(det_ctx->smtp[index].buffer);
176  det_ctx->smtp[index].buffer = NULL;
177  det_ctx->smtp[index].buffer_size = 0;
178  det_ctx->smtp[index].buffer_len = 0;
179  goto end;
180  }
181  det_ctx->smtp[index].buffer = ptmp;
182  }
183  memcpy(det_ctx->smtp[index].buffer + det_ctx->smtp[index].buffer_len, curr_chunk->data, curr_chunk->len);
184  det_ctx->smtp[index].buffer_len += curr_chunk->len;
185 
186  curr_chunk = curr_chunk->next;
187  }
188 
189  /* updat inspected tracker */
190  curr_file->content_inspected = curr_file->chunks_tail->stream_offset + curr_file->chunks_tail->len;
191 
192  buffer = det_ctx->smtp[index].buffer;
193  *buffer_len = det_ctx->smtp[index].buffer_len;
194  *stream_start_offset = det_ctx->smtp[index].offset;
195 end:
196  return buffer;
197 }

200  DetectEngineCtx *de_ctx,
