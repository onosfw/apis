43 /**
44  * This callback function feeds request body data to a Urlencoded parser
45  * and, later, feeds the parsed parameters to the correct structures.
46  *
47  * @param[in] d
48  * @return HTP_OK on success, HTP_ERROR on failure.
49  */
50 htp_status_t htp_ch_urlencoded_callback_request_body_data(htp_tx_data_t *d) {
51  htp_tx_t *tx = d->tx;
52 
53  // Check that we were not invoked again after the finalization.
54  if (tx->request_urlenp_body->params == NULL) return HTP_ERROR;
55 
56  if (d->data != NULL) {
57  // Process one chunk of data.
58  htp_urlenp_parse_partial(tx->request_urlenp_body, d->data, d->len);
59  } else {
60  // Finalize parsing.
61  htp_urlenp_finalize(tx->request_urlenp_body);
62 
63  // Add all parameters to the transaction.
64  bstr *name = NULL;
65  bstr *value = NULL;
66 
67  for (size_t i = 0, n = htp_table_size(tx->request_urlenp_body->params); i < n; i++) {
68  value = htp_table_get_index(tx->request_urlenp_body->params, i, &name);
69 
70  htp_param_t *param = calloc(1, sizeof (htp_param_t));
71  if (param == NULL) return HTP_ERROR;
72 
73  param->name = name;
74  param->value = value;
75  param->source = HTP_SOURCE_BODY;
76  param->parser_id = HTP_PARSER_URLENCODED;
77  param->parser_data = NULL;
78 
79  if (htp_tx_req_add_param(tx, param) != HTP_OK) {
80  free(param);
81  return HTP_ERROR;
82  }
83  }
84 
85  // All the parameter data is now owned by the transaction, and
86  // the parser table used to store it is no longer needed. The
87  // line below will destroy just the table, leaving keys intact.
88  htp_table_destroy_ex(tx->request_urlenp_body->params);
89  tx->request_urlenp_body->params = NULL;
90  }
91 
92  return HTP_OK;
93 }

95 /**
96  * Determine if the request has a Urlencoded body, and, if it does, create and
97  * attach an instance of the Urlencoded parser to the transaction.
98  *
99  * @param[in] connp
100  * @return HTP_OK if a new parser has been setup, HTP_DECLINED if the MIME type
101  * is not appropriate for this parser, and HTP_ERROR on failure.
102  */
103 htp_status_t htp_ch_urlencoded_callback_request_headers(htp_tx_t *tx) {
104  // Check the request content type to see if it matches our MIME type.
105  if ((tx->request_content_type == NULL) || (!bstr_begins_with_c(tx->request_content_type, HTP_URLENCODED_MIME_TYPE))) {
106  #ifdef HTP_DEBUG
107  fprintf(stderr, "htp_ch_urlencoded_callback_request_headers: Body not URLENCODED\n");
108  #endif
109 
110  return HTP_DECLINED;
111  }
112 
113  #ifdef HTP_DEBUG
114  fprintf(stderr, "htp_ch_urlencoded_callback_request_headers: Parsing URLENCODED body\n");
115  #endif
116 
117  // Create parser instance.
118  tx->request_urlenp_body = htp_urlenp_create(tx);
119  if (tx->request_urlenp_body == NULL) return HTP_ERROR;
120 
121  // Register a request body data callback.
122  htp_tx_register_request_body_data(tx, htp_ch_urlencoded_callback_request_body_data);
123 
124  return HTP_OK;
125 }

127 /**
128  * Parses request query string, if present.
129  *
130  * @param[in] connp
131  * @param[in] raw_data
132  * @param[in] raw_len
133  * @return HTP_OK if query string was parsed, HTP_DECLINED if there was no query
134  * string, and HTP_ERROR on failure.
135  */
136 htp_status_t htp_ch_urlencoded_callback_request_line(htp_tx_t *tx) {
137  // Proceed only if there's something for us to parse.
138  if ((tx->parsed_uri->query == NULL) || (bstr_len(tx->parsed_uri->query) == 0)) {
139  return HTP_DECLINED;
140  }
141 
142  // We have a non-zero length query string.
143 
144  tx->request_urlenp_query = htp_urlenp_create(tx);
145  if (tx->request_urlenp_query == NULL) return HTP_ERROR;
146 
147  if (htp_urlenp_parse_complete(tx->request_urlenp_query, bstr_ptr(tx->parsed_uri->query),
148  bstr_len(tx->parsed_uri->query)) != HTP_OK) {
149  htp_urlenp_destroy(tx->request_urlenp_query);
150  return HTP_ERROR;
151  }
152 
153  // Add all parameters to the transaction.
154 
155  bstr *name = NULL;
156  bstr *value = NULL;
157  for (size_t i = 0, n = htp_table_size(tx->request_urlenp_query->params); i < n; i++) {
158  value = htp_table_get_index(tx->request_urlenp_query->params, i, &name);
159 
160  htp_param_t *param = calloc(1, sizeof (htp_param_t));
161  if (param == NULL) return HTP_ERROR;
162 
163  param->name = name;
164  param->value = value;
165  param->source = HTP_SOURCE_QUERY_STRING;
166  param->parser_id = HTP_PARSER_URLENCODED;
167  param->parser_data = NULL;
168 
169  if (htp_tx_req_add_param(tx, param) != HTP_OK) {
170  free(param);
171  return HTP_ERROR;
172  }
173  }
174 
175  // All the parameter data is now owned by the transaction, and
176  // the parser table used to store it is no longer needed. The
177  // line below will destroy just the table, leaving keys intact.
178  htp_table_destroy_ex(tx->request_urlenp_query->params);
179  tx->request_urlenp_query->params = NULL;
180 
181  htp_urlenp_destroy(tx->request_urlenp_query);
182  tx->request_urlenp_query = NULL;
183 
184  return HTP_OK;
185 }

187 /**
188  * Finalize Multipart processing.
189  *
190  * @param[in] d
191  * @return HTP_OK on success, HTP_ERROR on failure.
192  */
193 htp_status_t htp_ch_multipart_callback_request_body_data(htp_tx_data_t *d) {
194  htp_tx_t *tx = d->tx;
195 
196  // Check that we were not invoked again after the finalization.
197  if (tx->request_mpartp->gave_up_data == 1) return HTP_ERROR;
198 
199  if (d->data != NULL) {
200  // Process one chunk of data.
201  htp_mpartp_parse(tx->request_mpartp, d->data, d->len);
202  } else {
203  // Finalize parsing.
204  htp_mpartp_finalize(tx->request_mpartp);
205 
206  htp_multipart_t *body = htp_mpartp_get_multipart(tx->request_mpartp);
207 
208  for (size_t i = 0, n = htp_list_size(body->parts); i < n; i++) {
209  htp_multipart_part_t *part = htp_list_get(body->parts, i);
210 
211  // Use text parameters.
212  if (part->type == MULTIPART_PART_TEXT) {
213  htp_param_t *param = calloc(1, sizeof (htp_param_t));
214  if (param == NULL) return HTP_ERROR;
215  param->name = part->name;
216  param->value = part->value;
217  param->source = HTP_SOURCE_BODY;
218  param->parser_id = HTP_PARSER_MULTIPART;
219  param->parser_data = part;
220 
221  if (htp_tx_req_add_param(tx, param) != HTP_OK) {
222  free(param);
223  return HTP_ERROR;
224  }
225  }
226  }
227 
228  // Tell the parser that it no longer owns names
229  // and values of MULTIPART_PART_TEXT parts.
230  tx->request_mpartp->gave_up_data = 1;
231  }
232 
233  return HTP_OK;
234 }

236 /**
237  * Inspect request headers and register the Multipart request data hook
238  * if it contains a multipart/form-data body.
239  *
240  * @param[in] connp
241  * @return HTP_OK if a new parser has been setup, HTP_DECLINED if the MIME type
242  * is not appropriate for this parser, and HTP_ERROR on failure.
243  */
244 htp_status_t htp_ch_multipart_callback_request_headers(htp_tx_t *tx) {
245  #ifdef HTP_DEBUG
246  fprintf(stderr, "htp_ch_multipart_callback_request_headers: Need to determine if multipart body is present\n");
247  #endif
248 
249  // The field tx->request_content_type does not contain the entire C-T
250  // value and so we cannot use it to look for a boundary, but we can
251  // use it for a quick check to determine if the C-T header exists.
252  if (tx->request_content_type == NULL) {
253  #ifdef HTP_DEBUG
254  fprintf(stderr, "htp_ch_multipart_callback_request_headers: Not multipart body (no C-T header)\n");
255  #endif
256 
257  return HTP_DECLINED;
258  }
259 
260  // Look for a boundary.
261 
262  htp_header_t *ct = htp_table_get_c(tx->request_headers, "content-type");
263  if (ct == NULL) return HTP_ERROR;
264 
265  bstr *boundary = NULL;
266  uint64_t flags = 0;
267 
268  htp_status_t rc = htp_mpartp_find_boundary(ct->value, &boundary, &flags);
269  if (rc != HTP_OK) {
270  #ifdef HTP_DEBUG
271  if (rc == HTP_DECLINED) {
272  fprintf(stderr, "htp_ch_multipart_callback_request_headers: Not multipart body\n");
273  }
274  #endif
275 
276  // No boundary (HTP_DECLINED) or error (HTP_ERROR).
277  return rc;
278  }
279 
280  if (boundary == NULL) return HTP_ERROR;
281 
282  // Create a Multipart parser instance.
283  tx->request_mpartp = htp_mpartp_create(tx->connp->cfg, boundary, flags);
284  if (tx->request_mpartp == NULL) {
285  bstr_free(boundary);
286  return HTP_ERROR;
287  }
288 
289  // Configure file extraction.
290  if (tx->cfg->extract_request_files) {
291  tx->request_mpartp->extract_files = 1;
292  tx->request_mpartp->extract_dir = tx->connp->cfg->tmpdir;
293  }
294 
295  // Register a request body data callback.
296  htp_tx_register_request_body_data(tx, htp_ch_multipart_callback_request_body_data);
297 
298  return HTP_OK;
299 }
