88  * \param dp Pointer to the DetectPort that has to be freed.

90 void DetectPortFree(DetectPort *dp)
91 {
92  if (dp == NULL)
93  return;
94 
95  /* only free the head if we have the original */
96  if (dp->sh != NULL && !(dp->flags & PORT_SIGGROUPHEAD_COPY)) {
97  SigGroupHeadFree(dp->sh);
98  }
99  dp->sh = NULL;
100 
101  if (dp->dst_ph != NULL && !(dp->flags & PORT_GROUP_PORTS_COPY)) {
102  DetectPortCleanupList(dp->dst_ph);
103  }
104  dp->dst_ph = NULL;
105 
106  //BUG_ON(dp->next != NULL);
107 
108  detect_port_memory -= sizeof(DetectPort);
109  detect_port_free_cnt++;
110  SCFree(dp);
111 }

255 /**
256  * \brief function for inserting a port group object. This also makes sure
257  * SigGroupContainer lists are handled correctly.
258  *
259  * \param de_ctx Pointer to the current detection engine context
260  * \param head Pointer to the DetectPort list head
261  * \param dp DetectPort to search in the DetectPort list
262  *
263  * \retval 1 inserted
264  * \retval 0 not inserted, memory of new is freed
265  * \retval -1 error
266  * */
267 int DetectPortInsert(DetectEngineCtx *de_ctx, DetectPort **head,
268  DetectPort *new)
269 {
270  if (new == NULL)
271  return 0;
272 
273  //BUG_ON(new->next != NULL);
274  //BUG_ON(new->prev != NULL);
275 
276  /* see if it already exists or overlaps with existing ag's */
277  if (*head != NULL) {
278  DetectPort *cur = NULL;
279  int r = 0;
280 
281  for (cur = *head; cur != NULL; cur = cur->next) {
282  r = DetectPortCmp(new,cur);
283  BUG_ON(r == PORT_ER);
284 
285  /* if so, handle that */
286  if (r == PORT_EQ) {
287  SCLogDebug("PORT_EQ %p %p", cur, new);
288  /* exact overlap/match */
289  if (cur != new) {
290  SigGroupHeadCopySigs(de_ctx, new->sh, &cur->sh);
291  cur->cnt += new->cnt;
292  DetectPortFree(new);
293  return 0;
294  }
295  return 1;
296  } else if (r == PORT_GT) {
297  SCLogDebug("PORT_GT (cur->next %p)", cur->next);
298  /* only add it now if we are bigger than the last
299  * group. Otherwise we'll handle it later. */
300  if (cur->next == NULL) {
301  SCLogDebug("adding GT");
302  /* put in the list */
303  new->prev = cur;
304  cur->next = new;
305  return 1;
306  }
307  } else if (r == PORT_LT) {
308  SCLogDebug("PORT_LT");
309 
310  /* see if we need to insert the ag anywhere */
311  /* put in the list */
312  if (cur->prev != NULL)
313  cur->prev->next = new;
314  new->prev = cur->prev;
315  new->next = cur;
316  cur->prev = new;
317 
318  /* update head if required */
319  if (*head == cur) {
320  *head = new;
321  }
322  return 1;
323 
324  /* alright, those were the simple cases,
325  * lets handle the more complex ones now */
326 
327  } else {
328  DetectPort *c = NULL;
329  r = DetectPortCut(de_ctx, cur, new, &c);
330  if (r == -1)
331  goto error;
332 
333  r = DetectPortInsert(de_ctx, head, new);
334  if (r == -1)
335  goto error;
336 
337  if (c != NULL) {
338  SCLogDebug("inserting C (%p)", c);
339  if (SCLogDebugEnabled()) {
340  DetectPortPrint(c);
341  }
342  r = DetectPortInsert(de_ctx, head, c);
343  if (r == -1)
344  goto error;
345  }
346  return 1;
347 
348  }
349  }
350 
351  /* head is NULL, so get a group and set head to it */
352  } else {
353  SCLogDebug("setting new head %p", new);
354  *head = new;
355  }
356 
357  return 1;
358 error:
359  /* XXX */
360  return -1;
361 }

363 /**
364  * \brief Function that cuts port groups and merge them
365  *
366  * \param de_ctx Pointer to the current detection engine context
367  * \param a pointer to DetectPort "a"
368  * \param b pointer to DetectPort "b"
369  * \param c pointer to DetectPort "c"
370  *
371  * \retval 0 ok
372  * \retval -1 error
373  * */
374 static int DetectPortCut(DetectEngineCtx *de_ctx, DetectPort *a,
375  DetectPort *b, DetectPort **c)
376 {
377  uint32_t a_port1 = a->port;
378  uint32_t a_port2 = a->port2;
379  uint32_t b_port1 = b->port;
380  uint32_t b_port2 = b->port2;
381  DetectPort *tmp = NULL;
382 
383  /* default to NULL */
384  *c = NULL;
385 
386  int r = DetectPortCmp(a,b);
387  BUG_ON(r != PORT_ES && r != PORT_EB && r != PORT_LE && r != PORT_GE);
388 
389  /* get a place to temporary put sigs lists */
390  tmp = DetectPortInit();
391  if (tmp == NULL) {
392  goto error;
393  }
394  memset(tmp, 0, sizeof(DetectPort));
395 
396  /**
397  * We have 3 parts: [aaa[abab]bbb]
398  * part a: a_port1 <-> b_port1 - 1
399  * part b: b_port1 <-> a_port2
400  * part c: a_port2 + 1 <-> b_port2
401  */
402  if (r == PORT_LE) {
403  SCLogDebug("cut r == PORT_LE");
404  a->port = a_port1;
405  a->port2 = b_port1 - 1;
406 
407  b->port = b_port1;
408  b->port2 = a_port2;
409 
410  DetectPort *tmp_c;
411  tmp_c = DetectPortInit();
412  if (tmp_c == NULL) {
413  goto error;
414  }
415  *c = tmp_c;
416 
417  tmp_c->port = a_port2 + 1;
418  tmp_c->port2 = b_port2;
419 
420  SigGroupHeadCopySigs(de_ctx,b->sh,&tmp_c->sh); /* copy old b to c */
421  SigGroupHeadCopySigs(de_ctx,a->sh,&b->sh); /* copy a to b */
422 
423  tmp_c->cnt += b->cnt;
424  b->cnt += a->cnt;
425 
426  /**
427  * We have 3 parts: [bbb[baba]aaa]
428  * part a: b_port1 <-> a_port1 - 1
429  * part b: a_port1 <-> b_port2
430  * part c: b_port2 + 1 <-> a_port2
431  */
432  } else if (r == PORT_GE) {
433  SCLogDebug("cut r == PORT_GE");
434  a->port = b_port1;
435  a->port2 = a_port1 - 1;
436 
437  b->port = a_port1;
438  b->port2 = b_port2;
439 
440  DetectPort *tmp_c;
441  tmp_c = DetectPortInit();
442  if (tmp_c == NULL) {
443  goto error;
444  }
445  *c = tmp_c;
446 
447  tmp_c->port = b_port2 + 1;
448  tmp_c->port2 = a_port2;
449 
450  /**
451  * 'a' gets clean and then 'b' sigs
452  * 'b' gets clean, then 'a' then 'b' sigs
453  * 'c' gets 'a' sigs
454  */
455  SigGroupHeadCopySigs(de_ctx,a->sh,&tmp->sh); /* store old a list */
456  SigGroupHeadClearSigs(a->sh); /* clean a list */
457  SigGroupHeadCopySigs(de_ctx,tmp->sh,&tmp_c->sh); /* copy old b to c */
458  SigGroupHeadCopySigs(de_ctx,b->sh,&a->sh); /* copy old b to a */
459  SigGroupHeadCopySigs(de_ctx,tmp->sh,&b->sh);/* prepend old a before b */
460 
461  SigGroupHeadClearSigs(tmp->sh); /* clean tmp list */
462 
463  tmp->cnt += a->cnt;
464  a->cnt = 0;
465  tmp_c->cnt += tmp->cnt;
466  a->cnt += b->cnt;
467  b->cnt += tmp->cnt;
468  tmp->cnt = 0;
469 
470  /**
471  * We have 2 or three parts:
472  *
473  * 2 part: [[abab]bbb] or [bbb[baba]]
474  * part a: a_port1 <-> a_port2
475  * part b: a_port2 + 1 <-> b_port2
476  *
477  * part a: b_port1 <-> a_port1 - 1
478  * part b: a_port1 <-> a_port2
479  *
480  * 3 part [bbb[aaa]bbb]
481  * becomes[aaa[bbb]ccc]
482  *
483  * part a: b_port1 <-> a_port1 - 1
484  * part b: a_port1 <-> a_port2
485  * part c: a_port2 + 1 <-> b_port2
486  */
487  } else if (r == PORT_ES) {
488  SCLogDebug("cut r == PORT_ES");
489  if (a_port1 == b_port1) {
490  SCLogDebug("1");
491  a->port = a_port1;
492  a->port2 = a_port2;
493 
494  b->port = a_port2 + 1;
495  b->port2 = b_port2;
496 
497  /** 'b' overlaps 'a' so 'a' needs the 'b' sigs */
498  SigGroupHeadCopySigs(de_ctx,b->sh,&a->sh);
499  a->cnt += b->cnt;
500 
501  } else if (a_port2 == b_port2) {
502  SCLogDebug("2");
503  a->port = b_port1;
504  a->port2 = a_port1 - 1;
505 
506  b->port = a_port1;
507  b->port2 = a_port2;
508 
509  /* [bbb[baba]] will be transformed into
510  * [aaa][bbb]
511  * steps: copy b sigs to tmp
512  * a overlaps b, so copy a to b
513  * clear a
514  * copy tmp to a */
515  SigGroupHeadCopySigs(de_ctx,b->sh,&tmp->sh); /* store old a list */
516  tmp->cnt = b->cnt;
517  SigGroupHeadCopySigs(de_ctx,a->sh,&b->sh);
518  b->cnt += a->cnt;
519  SigGroupHeadClearSigs(a->sh); /* clean a list */
520  SigGroupHeadCopySigs(de_ctx,tmp->sh,&a->sh);/* merge old a with b */
521  a->cnt = tmp->cnt;
522  SigGroupHeadClearSigs(tmp->sh); /* clean tmp list */
523  } else {
524  SCLogDebug("3");
525  a->port = b_port1;
526  a->port2 = a_port1 - 1;
527 
528  b->port = a_port1;
529  b->port2 = a_port2;
530 
531  DetectPort *tmp_c;
532  tmp_c = DetectPortInit();
533  if (tmp_c == NULL) {
534  goto error;
535  }
536  *c = tmp_c;
537 
538  tmp_c->port = a_port2 + 1;
539  tmp_c->port2 = b_port2;
540 
541  /**
542  * 'a' gets clean and then 'b' sigs
543  * 'b' gets clean, then 'a' then 'b' sigs
544  * 'c' gets 'b' sigs
545  */
546  SigGroupHeadCopySigs(de_ctx,a->sh,&tmp->sh); /* store old a list */
547  SigGroupHeadClearSigs(a->sh); /* clean a list */
548  SigGroupHeadCopySigs(de_ctx,b->sh,&tmp_c->sh); /* copy old b to c */
549  SigGroupHeadCopySigs(de_ctx,b->sh,&a->sh); /* copy old b to a */
550  SigGroupHeadCopySigs(de_ctx,tmp->sh,&b->sh);/* merge old a with b */
551 
552  SigGroupHeadClearSigs(tmp->sh); /* clean tmp list */
553 
554  tmp->cnt += a->cnt;
555  a->cnt = 0;
556  tmp_c->cnt += b->cnt;
557  a->cnt += b->cnt;
558  b->cnt += tmp->cnt;
559  tmp->cnt = 0;
560  }
561  /**
562  * We have 2 or three parts:
563  *
564  * 2 part: [[baba]aaa] or [aaa[abab]]
565  * part a: b_port1 <-> b_port2
566  * part b: b_port2 + 1 <-> a_port2
567  *
568  * part a: a_port1 <-> b_port1 - 1
569  * part b: b_port1 <-> b_port2
570  *
571  * 3 part [aaa[bbb]aaa]
572  * becomes[aaa[bbb]ccc]
573  *
574  * part a: a_port1 <-> b_port2 - 1
575  * part b: b_port1 <-> b_port2
576  * part c: b_port2 + 1 <-> a_port2
577  */
578  } else if (r == PORT_EB) {
579  SCLogDebug("cut r == PORT_EB");
580  if (a_port1 == b_port1) {
581  SCLogDebug("1");
582  a->port = b_port1;
583  a->port2 = b_port2;
584 
585  b->port = b_port2 + 1;
586  b->port2 = a_port2;
587 
588  /** 'b' overlaps 'a' so 'a' needs the 'b' sigs */
589  SigGroupHeadCopySigs(de_ctx,b->sh,&tmp->sh);
590  SigGroupHeadClearSigs(b->sh);
591  SigGroupHeadCopySigs(de_ctx,a->sh,&b->sh);
592  SigGroupHeadCopySigs(de_ctx,tmp->sh,&a->sh);
593 
594  SigGroupHeadClearSigs(tmp->sh);
595 
596  tmp->cnt += b->cnt;
597  b->cnt = 0;
598  b->cnt += a->cnt;
599  a->cnt += tmp->cnt;
600  tmp->cnt = 0;
601 
602  } else if (a_port2 == b_port2) {
603  SCLogDebug("2");
604 
605  a->port = a_port1;
606  a->port2 = b_port1 - 1;
607 
608  b->port = b_port1;
609  b->port2 = b_port2;
610 
611  /** 'a' overlaps 'b' so 'b' needs the 'a' sigs */
612  SigGroupHeadCopySigs(de_ctx,a->sh,&b->sh);
613 
614  b->cnt += a->cnt;
615 
616  } else {
617  SCLogDebug("3");
618  a->port = a_port1;
619  a->port2 = b_port1 - 1;
620 
621  b->port = b_port1;
622  b->port2 = b_port2;
623 
624  DetectPort *tmp_c;
625  tmp_c = DetectPortInit();
626  if (tmp_c == NULL) {
627  goto error;
628  }
629  *c = tmp_c;
630 
631  tmp_c->port = b_port2 + 1;
632  tmp_c->port2 = a_port2;
633 
634  SigGroupHeadCopySigs(de_ctx,a->sh,&b->sh);
635  SigGroupHeadCopySigs(de_ctx,a->sh,&tmp_c->sh);
636

638  tmp_c->cnt += a->cnt;
639  }
640  }

701 /**
702  * \brief Function that compare port groups
703  *
704  * \param a pointer to DetectPort "a"
705  * \param b pointer to DetectPort "b"
706  *
707  * \retval PORT_XX (Port enum value, XX is EQ, ES, EB, LE, etc)
708  * \retval PORT_ER on error
709  * */
710 int DetectPortCmp(DetectPort *a, DetectPort *b)
711 {
712  /* check any */
713  if ((a->flags & PORT_FLAG_ANY) && (b->flags & PORT_FLAG_ANY))
714  return PORT_EQ;
715  if ((a->flags & PORT_FLAG_ANY) && !(b->flags & PORT_FLAG_ANY))
716  return PORT_LT;
717  if (!(a->flags & PORT_FLAG_ANY) && (b->flags & PORT_FLAG_ANY))
718  return PORT_GT;
719 
720  uint16_t a_port1 = a->port;
721  uint16_t a_port2 = a->port2;
722  uint16_t b_port1 = b->port;
723  uint16_t b_port2 = b->port2;
724 
725  /* PORT_EQ */
726  if (a_port1 == b_port1 && a_port2 == b_port2) {
727  //SCLogDebug("PORT_EQ");
728  return PORT_EQ;
729  /* PORT_ES */
730  } else if (a_port1 >= b_port1 && a_port1 <= b_port2 && a_port2 <= b_port2) {
731  //SCLogDebug("PORT_ES");
732  return PORT_ES;
733  /* PORT_EB */
734  } else if (a_port1 <= b_port1 && a_port2 >= b_port2) {
735  //SCLogDebug("PORT_EB");
736  return PORT_EB;
737  } else if (a_port1 < b_port1 && a_port2 < b_port2 && a_port2 >= b_port1) {
738  //SCLogDebug("PORT_LE");
739  return PORT_LE;
740  } else if (a_port1 < b_port1 && a_port2 < b_port2) {
741  //SCLogDebug("PORT_LT");
742  return PORT_LT;
743  } else if (a_port1 > b_port1 && a_port1 <= b_port2 && a_port2 > b_port2) {
744  //SCLogDebug("PORT_GE");
745  return PORT_GE;
746  } else if (a_port1 > b_port2) {
747  //SCLogDebug("PORT_GT");
748  return PORT_GT;
749  } else {
750  /* should be unreachable */
751  BUG_ON(1);
752  }
753 
754  return PORT_ER;
755 }

1004 /**
1005  * \brief Parses a port string and updates the 2 port heads with the
1006  * port groups.
1007  *
1008  * \todo We don't seem to be handling negated cases, like [port,![!port,port]],
1009  * since we pass around negate without keeping a count of ! with depth.
1010  * Can solve this by keeping a count of the negations with depth, so that
1011  * an even no of negations would count as no negation and an odd no of
1012  * negations would count as a negation.
1013  *
1014  * \param gh Pointer to the port group head that should hold port ranges
1015  * that are not negated.
1016  * \param ghn Pointer to the port group head that should hold port ranges
1017  * that are negated.
1018  * \param s Pointer to the character string holding the port to be
1019  * parsed.
1020  * \param negate Flag that indicates if the receieved address string is negated
1021  * or not. 0 if it is not, 1 it it is.
1022  *
1023  * \retval 0 On successfully parsing.
1024  * \retval -1 On failure.
1025  */
1026 static int DetectPortParseDo(const DetectEngineCtx *de_ctx,
1027  DetectPort **head, DetectPort **nhead,
1028  char *s, int negate)
1029 {
1030  size_t u = 0;
1031  size_t x = 0;
1032  int o_set = 0, n_set = 0, d_set = 0;
1033  int range = 0;
1034  int depth = 0;
1035  size_t size = strlen(s);
1036  char address[1024] = "";
1037  char *rule_var_port = NULL;
1038  int r = 0;
1039 
1040  SCLogDebug("head %p, *head %p, negate %d", head, *head, negate);
1041 
1042  for (u = 0, x = 0; u < size && x < sizeof(address); u++) {
1043  address[x] = s[u];
1044  x++;
1045 
1046  if (s[u] == ':')
1047  range = 1;
1048 
1049  if (range == 1 && s[u] == '!') {
1050  SCLogError(SC_ERR_NEGATED_VALUE_IN_PORT_RANGE,"Can't have a negated value in a range.");
1051  return -1;
1052  } else if (!o_set && s[u] == '!') {
1053  SCLogDebug("negation encountered");
1054  n_set = 1;
1055  x--;
1056  } else if (s[u] == '[') {
1057  if (!o_set) {
1058  o_set = 1;
1059  x = 0;
1060  }
1061  depth++;
1062  } else if (s[u] == ']') {
1063  if (depth == 1) {
1064  address[x - 1] = '\0';
1065  SCLogDebug("Parsed port from DetectPortParseDo - %s", address);
1066  x = 0;
1067 
1068  r = DetectPortParseDo(de_ctx, head, nhead, address, negate? negate: n_set);
1069  if (r == -1)
1070  goto error;
1071 
1072  n_set = 0;
1073  }
1074  depth--;
1075  range = 0;
1076  } else if (depth == 0 && s[u] == ',') {
1077  if (o_set == 1) {
1078  o_set = 0;
1079  } else if (d_set == 1) {
1080  char *temp_rule_var_port = NULL,
1081  *alloc_rule_var_port = NULL;
1082 
1083  address[x - 1] = '\0';
1084 
1085  rule_var_port = SCRuleVarsGetConfVar(de_ctx, address,
1086  SC_RULE_VARS_PORT_GROUPS);
1087  if (rule_var_port == NULL)
1088  goto error;
1089  if (strlen(rule_var_port) == 0) {
1090  SCLogError(SC_ERR_INVALID_SIGNATURE, "variable %s resolved "
1091  "to nothing. This is likely a misconfiguration. "
1092  "Note that a negated port needs to be quoted, "
1093  "\"!$HTTP_PORTS\" instead of !$HTTP_PORTS. See issue #295.", s);
1094  goto error;
1095  }
1096  temp_rule_var_port = rule_var_port;
1097  if (negate == 1 || n_set == 1) {
1098  alloc_rule_var_port = SCMalloc(strlen(rule_var_port) + 3);
1099  if (unlikely(alloc_rule_var_port == NULL))
1100  goto error;
1101  snprintf(alloc_rule_var_port, strlen(rule_var_port) + 3,
1102  "[%s]", rule_var_port);
1103  temp_rule_var_port = alloc_rule_var_port;
1104  }
1105  r = DetectPortParseDo(de_ctx, head, nhead, temp_rule_var_port,
1106  (negate + n_set) % 2);//negate? negate: n_set);
1107  if (r == -1)
1108  goto error;
1109 
1110  d_set = 0;
1111  n_set = 0;
1112  if (alloc_rule_var_port != NULL)
1113  SCFree(alloc_rule_var_port);
1114  } else {
1115  address[x - 1] = '\0';
1116  SCLogDebug("Parsed port from DetectPortParseDo - %s", address);
1117 
1118  if (negate == 0 && n_set == 0) {
1119  r = DetectPortParseInsertString(head, address);
1120  } else {
1121  r = DetectPortParseInsertString(nhead, address);
1122  }
1123  if (r == -1)
1124  goto error;
1125 
1126  n_set = 0;
1127  }
1128  x = 0;
1129  range = 0;
1130  } else if (depth == 0 && s[u] == '$') {
1131  d_set = 1;
1132  } else if (depth == 0 && u == size-1) {
1133  range = 0;
1134  if (x == 1024) {
1135  address[x - 1] = '\0';
1136  } else {
1137  address[x] = '\0';
1138  }
1139  SCLogDebug("%s", address);
1140  x = 0;
1141  if (d_set == 1) {
1142  char *temp_rule_var_port = NULL,
1143  *alloc_rule_var_port = NULL;
1144 
1145  rule_var_port = SCRuleVarsGetConfVar(de_ctx, address,
1146  SC_RULE_VARS_PORT_GROUPS);
1147  if (rule_var_port == NULL)
1148  goto error;
1149  if (strlen(rule_var_port) == 0) {
1150  SCLogError(SC_ERR_INVALID_SIGNATURE, "variable %s resolved "
1151  "to nothing. This is likely a misconfiguration. "
1152  "Note that a negated port needs to be quoted, "
1153  "\"!$HTTP_PORTS\" instead of !$HTTP_PORTS. See issue #295.", s);
1154  goto error;
1155  }
1156  temp_rule_var_port = rule_var_port;
1157  if ((negate + n_set) % 2) {
1158  alloc_rule_var_port = SCMalloc(strlen(rule_var_port) + 3);
1159  if (unlikely(alloc_rule_var_port == NULL))
1160  goto error;
1161  snprintf(alloc_rule_var_port, strlen(rule_var_port) + 3,
1162  "[%s]", rule_var_port);
1163  temp_rule_var_port = alloc_rule_var_port;
1164  }
1165  r = DetectPortParseDo(de_ctx, head, nhead, temp_rule_var_port,
1166  (negate + n_set) % 2);
1167  if (r == -1)
1168  goto error;
1169 
1170  d_set = 0;
1171  if (alloc_rule_var_port != NULL)
1172  SCFree(alloc_rule_var_port);
1173  } else {
1174  if (!((negate + n_set) % 2)) {
1175  r = DetectPortParseInsertString(head,address);
1176  } else {
1177  r = DetectPortParseInsertString(nhead,address);
1178  }
1179  if (r == -1)
1180  goto error;
1181  }
1182  n_set = 0;
1183  }
1184  }
1185 
1186  if (depth > 0) {
1187  SCLogError(SC_ERR_INVALID_SIGNATURE, "not every port block was "
1188  "properly closed in \"%s\", %d missing closing brackets (]). "
1189  "Note: problem might be in a variable.", s, depth);
1190  goto error;
1191  } else if (depth < 0) {
1192  SCLogError(SC_ERR_INVALID_SIGNATURE, "not every port block was "
1193  "properly opened in \"%s\", %d missing opening brackets ([). "
1194  "Note: problem might be in a variable.", s, depth*-1);
1195  goto error;
1196  }
1197 
1198  return 0;
1199 error:
1200  return -1;
1201 }
