141 void PcapFileCallbackLoop(char *user, struct pcap_pkthdr *h, u_char *pkt)
142 {
143  SCEnter();
144 
145  PcapFileThreadVars *ptv = (PcapFileThreadVars *)user;
146  Packet *p = PacketGetFromQueueOrAlloc();
147 
148  if (unlikely(p == NULL)) {
149  SCReturn;
150  }
151  PACKET_PROFILING_TMM_START(p, TMM_RECEIVEPCAPFILE);
152 
153  PKT_SET_SRC(p, PKT_SRC_WIRE);
154  p->ts.tv_sec = h->ts.tv_sec;
155  p->ts.tv_usec = h->ts.tv_usec;
156  SCLogDebug("p->ts.tv_sec %"PRIuMAX"", (uintmax_t)p->ts.tv_sec);
157  p->datalink = pcap_g.datalink;
158  p->pcap_cnt = ++pcap_g.cnt;
159 
160  p->pcap_v.tenant_id = ptv->tenant_id;
161  ptv->pkts++;
162  ptv->bytes += h->caplen;
163 
164  if (unlikely(PacketCopyData(p, pkt, h->caplen))) {
165  TmqhOutputPacketpool(ptv->tv, p);
166  PACKET_PROFILING_TMM_END(p, TMM_RECEIVEPCAPFILE);
167  SCReturn;
168  }
169 
170  /* We only check for checksum disable */
171  if (pcap_g.checksum_mode == CHECKSUM_VALIDATION_DISABLE) {
172  p->flags |= PKT_IGNORE_CHECKSUM;
173  } else if (pcap_g.checksum_mode == CHECKSUM_VALIDATION_AUTO) {
174  if (ChecksumAutoModeCheck(ptv->pkts, p->pcap_cnt,
175  SC_ATOMIC_GET(pcap_g.invalid_checksums))) {
176  pcap_g.checksum_mode = CHECKSUM_VALIDATION_DISABLE;
177  p->flags |= PKT_IGNORE_CHECKSUM;
178  }
179  }
180 
181  PACKET_PROFILING_TMM_END(p, TMM_RECEIVEPCAPFILE);
182 
183  if (TmThreadsSlotProcessPkt(ptv->tv, ptv->slot, p) != TM_ECODE_OK) {
184  pcap_breakloop(pcap_g.pcap_handle);
185  ptv->cb_result = TM_ECODE_FAILED;
186  }
187 
188  SCReturn;
189 }

194 TmEcode ReceivePcapFileLoop(ThreadVars *tv, void *data, void *slot)
195 {
196  SCEnter();
197 
198  int packet_q_len = 64;
199  PcapFileThreadVars *ptv = (PcapFileThreadVars *)data;
200  int r;
201  TmSlot *s = (TmSlot *)slot;
202 
203  ptv->slot = s->slot_next;
204  ptv->cb_result = TM_ECODE_OK;
205 
206  while (1) {
207  if (suricata_ctl_flags & (SURICATA_STOP | SURICATA_KILL)) {
208  SCReturnInt(TM_ECODE_OK);
209  }
210 
211  /* make sure we have at least one packet in the packet pool, to prevent
212  * us from alloc'ing packets at line rate */
213  PacketPoolWait();
214 
215  /* Right now we just support reading packets one at a time. */
216  r = pcap_dispatch(pcap_g.pcap_handle, packet_q_len,
217  (pcap_handler)PcapFileCallbackLoop, (u_char *)ptv);
218  if (unlikely(r == -1)) {
219  SCLogError(SC_ERR_PCAP_DISPATCH, "error code %" PRId32 " %s",
220  r, pcap_geterr(pcap_g.pcap_handle));
221  if (! RunModeUnixSocketIsActive()) {
222  /* in the error state we just kill the engine */
223  EngineKill();
224  SCReturnInt(TM_ECODE_FAILED);
225  } else {
226  pcap_close(pcap_g.pcap_handle);
227  pcap_g.pcap_handle = NULL;
228  UnixSocketPcapFile(TM_ECODE_DONE);
229  SCReturnInt(TM_ECODE_DONE);
230  }
231  } else if (unlikely(r == 0)) {
232  SCLogInfo("pcap file end of file reached (pcap err code %" PRId32 ")", r);
233  if (! RunModeUnixSocketIsActive()) {
234  EngineStop();
235  } else {
236  pcap_close(pcap_g.pcap_handle);
237  pcap_g.pcap_handle = NULL;
238  UnixSocketPcapFile(TM_ECODE_DONE);
239  SCReturnInt(TM_ECODE_DONE);
240  }
241  break;
242  } else if (ptv->cb_result == TM_ECODE_FAILED) {
243  SCLogError(SC_ERR_PCAP_DISPATCH, "Pcap callback PcapFileCallbackLoop failed");
244  if (! RunModeUnixSocketIsActive()) {
245  EngineKill();
246  SCReturnInt(TM_ECODE_FAILED);
247  } else {
248  pcap_close(pcap_g.pcap_handle);
249  pcap_g.pcap_handle = NULL;
250  UnixSocketPcapFile(TM_ECODE_DONE);
251  SCReturnInt(TM_ECODE_DONE);
252  }
253  }
254  StatsSyncCountersIfSignalled(tv);
255  }
256 
257  SCReturnInt(TM_ECODE_OK);
258 }

260 TmEcode ReceivePcapFileThreadInit(ThreadVars *tv, void *initdata, void **data)
261 {
262  SCEnter();
263  char *tmpbpfstring = NULL;
264  char *tmpstring = NULL;
265  if (initdata == NULL) {
266  SCLogError(SC_ERR_INVALID_ARGUMENT, "error: initdata == NULL");
267  SCReturnInt(TM_ECODE_FAILED);
268  }
269 
270  SCLogInfo("reading pcap file %s", (char *)initdata);
271 
272  PcapFileThreadVars *ptv = SCMalloc(sizeof(PcapFileThreadVars));
273  if (unlikely(ptv == NULL))
274  SCReturnInt(TM_ECODE_FAILED);
275  memset(ptv, 0, sizeof(PcapFileThreadVars));
276 
277  intmax_t tenant = 0;
278  if (ConfGetInt("pcap-file.tenant-id", &tenant) == 1) {
279  if (tenant > 0 && tenant < UINT_MAX) {
280  ptv->tenant_id = (uint32_t)tenant;
281  SCLogInfo("tenant %u", ptv->tenant_id);
282  } else {
283  SCLogError(SC_ERR_INVALID_ARGUMENT, "tenant out of range");
284  }
285  }
286 
287  char errbuf[PCAP_ERRBUF_SIZE] = "";
288  pcap_g.pcap_handle = pcap_open_offline((char *)initdata, errbuf);
289  if (pcap_g.pcap_handle == NULL) {
290  SCLogError(SC_ERR_FOPEN, "%s\n", errbuf);
291  SCFree(ptv);
292  if (! RunModeUnixSocketIsActive()) {
293  return TM_ECODE_FAILED;
294  } else {
295  UnixSocketPcapFile(TM_ECODE_FAILED);
296  SCReturnInt(TM_ECODE_DONE);
297  }
298  }
299 
300  if (ConfGet("bpf-filter", &tmpbpfstring) != 1) {
301  SCLogDebug("could not get bpf or none specified");
302  } else {
303  SCLogInfo("using bpf-filter \"%s\"", tmpbpfstring);
304 
305  if(pcap_compile(pcap_g.pcap_handle,&pcap_g.filter,tmpbpfstring,1,0) < 0) {
306  SCLogError(SC_ERR_BPF,"bpf compilation error %s",pcap_geterr(pcap_g.pcap_handle));
307  SCFree(ptv);
308  return TM_ECODE_FAILED;
309  }
310 
311  if(pcap_setfilter(pcap_g.pcap_handle,&pcap_g.filter) < 0) {
312  SCLogError(SC_ERR_BPF,"could not set bpf filter %s",pcap_geterr(pcap_g.pcap_handle));
313  SCFree(ptv);
314  return TM_ECODE_FAILED;
315  }
316  }
317 
318  pcap_g.datalink = pcap_datalink(pcap_g.pcap_handle);
319  SCLogDebug("datalink %" PRId32 "", pcap_g.datalink);
320 
321  switch(pcap_g.datalink) {
322  case LINKTYPE_LINUX_SLL:
323  pcap_g.Decoder = DecodeSll;
324  break;
325  case LINKTYPE_ETHERNET:
326  pcap_g.Decoder = DecodeEthernet;
327  break;
328  case LINKTYPE_PPP:
329  pcap_g.Decoder = DecodePPP;
330  break;
331  case LINKTYPE_RAW:
332  pcap_g.Decoder = DecodeRaw;
333  break;
334  case LINKTYPE_NULL:
335  pcap_g.Decoder = DecodeNull;
336  break;
337 
338  default:
339  SCLogError(SC_ERR_UNIMPLEMENTED, "datalink type %" PRId32 " not "
340  "(yet) supported in module PcapFile.\n", pcap_g.datalink);
341  SCFree(ptv);
342  if (! RunModeUnixSocketIsActive()) {
343  SCReturnInt(TM_ECODE_FAILED);
344  } else {
345  pcap_close(pcap_g.pcap_handle);
346  pcap_g.pcap_handle = NULL;
347  UnixSocketPcapFile(TM_ECODE_DONE);
348  SCReturnInt(TM_ECODE_DONE);
349  }
350  }
351 
352  if (ConfGet("pcap-file.checksum-checks", &tmpstring) != 1) {
353  pcap_g.conf_checksum_mode = CHECKSUM_VALIDATION_AUTO;
354  } else {
355  if (strcmp(tmpstring, "auto") == 0) {
356  pcap_g.conf_checksum_mode = CHECKSUM_VALIDATION_AUTO;
357  } else if (strcmp(tmpstring, "yes") == 0) {
358  pcap_g.conf_checksum_mode = CHECKSUM_VALIDATION_ENABLE;
359  } else if (strcmp(tmpstring, "no") == 0) {
360  pcap_g.conf_checksum_mode = CHECKSUM_VALIDATION_DISABLE;
361  }
362  }
363  pcap_g.checksum_mode = pcap_g.conf_checksum_mode;
364 
365  ptv->tv = tv;
366  *data = (void *)ptv;
367 
368  SCReturnInt(TM_ECODE_OK);
369 }

371 void ReceivePcapFileThreadExitStats(ThreadVars *tv, void *data)
372 {
373  SCEnter();
374  PcapFileThreadVars *ptv = (PcapFileThreadVars *)data;
375 
376  if (pcap_g.conf_checksum_mode == CHECKSUM_VALIDATION_AUTO &&
377  pcap_g.cnt < CHECKSUM_SAMPLE_COUNT &&
378  SC_ATOMIC_GET(pcap_g.invalid_checksums)) {
379  uint64_t chrate = pcap_g.cnt / SC_ATOMIC_GET(pcap_g.invalid_checksums);
380  if (chrate < CHECKSUM_INVALID_RATIO)
381  SCLogWarning(SC_ERR_INVALID_CHECKSUM,
382  "1/%" PRIu64 "th of packets have an invalid checksum,"
383  " consider setting pcap-file.checksum-checks variable to no"
384  " or use '-k none' option on command line.",
385  chrate);
386  else
387  SCLogInfo("1/%" PRIu64 "th of packets have an invalid checksum",
388  chrate);
389  }
390  SCLogNotice("Pcap-file module read %" PRIu32 " packets, %" PRIu64 " bytes", ptv->pkts, ptv->bytes);
391  return;
392 }

406 TmEcode DecodePcapFile(ThreadVars *tv, Packet *p, void *data, PacketQueue *pq, PacketQueue *postpq)
407 {
408  SCEnter();
409  DecodeThreadVars *dtv = (DecodeThreadVars *)data;
410 
411  /* XXX HACK: flow timeout can call us for injected pseudo packets
412  * see bug: https://redmine.openinfosecfoundation.org/issues/1107 */
413  if (p->flags & PKT_PSEUDO_STREAM_END)
414  return TM_ECODE_OK;
415 
416  /* update counters */
417  DecodeUpdatePacketCounters(tv, dtv, p);
418 
419  double curr_ts = p->ts.tv_sec + p->ts.tv_usec / 1000.0;
420  if (curr_ts < prev_signaled_ts || (curr_ts - prev_signaled_ts) > 60.0) {
421  prev_signaled_ts = curr_ts;
422  FlowWakeupFlowManagerThread();
423  }
424 
425  /* update the engine time representation based on the timestamp
426  * of the packet. */
427  TimeSet(&p->ts);
428 
429  /* call the decoder */
430  pcap_g.Decoder(tv, dtv, p, GET_PKT_DATA(p), GET_PKT_LEN(p), pq);
431 
432 #ifdef DEBUG
433  BUG_ON(p->pkt_src != PKT_SRC_WIRE && p->pkt_src != PKT_SRC_FFR);
434 #endif
435 
436  PacketDecodeFinalize(tv, dtv, p);
437 
438  SCReturnInt(TM_ECODE_OK);
439 }
