78 static int DetectPrioritySetup (DetectEngineCtx *de_ctx, Signature *s, char *rawstr)
79 {
80  char copy_str[128] = "";
81 
82 #define MAX_SUBSTRINGS 30
83  int ret = 0;
84  int ov[MAX_SUBSTRINGS];
85 
86  ret = pcre_exec(regex, regex_study, rawstr, strlen(rawstr), 0, 0, ov, 30);
87  if (ret < 0) {
88  SCLogError(SC_ERR_PCRE_MATCH, "Invalid Priority in Signature "
89  "- %s", rawstr);
90  return -1;
91  }
92 
93  ret = pcre_copy_substring((char *)rawstr, ov, 30, 1, copy_str, sizeof(copy_str));
94  if (ret < 0) {
95  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_copy_substring failed");
96  return -1;
97  }
98 
99  long prio = 0;
100  char *endptr = NULL;
101  prio = strtol(copy_str, &endptr, 10);
102  if (endptr == NULL || *endptr != '\0') {
103  SCLogError(SC_ERR_INVALID_SIGNATURE, "Saw an invalid character as arg "
104  "to priority keyword");
105  return -1;
106  }
107  /* if we have reached here, we have had a valid priority. Assign it */
108  s->prio = prio;
109 
110  return 0;
111 }
