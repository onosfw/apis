94 static int LuaTxLogger(ThreadVars *tv, void *thread_data, const Packet *p, Flow *f, void *alstate, void *txptr, uint64_t tx_id)
95 {
96  SCEnter();
97 
98  LogLuaThreadCtx *td = (LogLuaThreadCtx *)thread_data;
99 
100  SCMutexLock(&td->lua_ctx->m);
101 
102  LuaStateSetThreadVars(td->lua_ctx->luastate, tv);
103  LuaStateSetPacket(td->lua_ctx->luastate, (Packet *)p);
104  LuaStateSetTX(td->lua_ctx->luastate, txptr);
105  LuaStateSetFlow(td->lua_ctx->luastate, f, /* locked */LUA_FLOW_LOCKED_BY_PARENT);
106 
107  /* prepare data to pass to script */
108  lua_getglobal(td->lua_ctx->luastate, "log");
109  lua_newtable(td->lua_ctx->luastate);
110  LuaPushTableKeyValueInt(td->lua_ctx->luastate, "tx_id", (int)(tx_id));
111 
112  int retval = lua_pcall(td->lua_ctx->luastate, 1, 0, 0);
113  if (retval != 0) {
114  SCLogInfo("failed to run script: %s", lua_tostring(td->lua_ctx->luastate, -1));
115  }
116 
117  SCMutexUnlock(&td->lua_ctx->m);
118  SCReturnInt(0);
119 }

171  *
172  * A single call to this function will run one script for a single
173  * packet. If it is called, it means that the registered condition

180 static int LuaPacketLoggerAlerts(ThreadVars *tv, void *thread_data, const Packet *p)
181 {
182  LogLuaThreadCtx *td = (LogLuaThreadCtx *)thread_data;
183 
184  char timebuf[64];
185  CreateTimeString(&p->ts, timebuf, sizeof(timebuf));
186 
187  if (!(PKT_IS_IPV4(p)) && !(PKT_IS_IPV6(p))) {
188  /* decoder event */
189  goto not_supported;
190  }
191 
192  char proto[16] = "";
193  if (SCProtoNameValid(IP_GET_IPPROTO(p)) == TRUE) {
194  strlcpy(proto, known_proto[IP_GET_IPPROTO(p)], sizeof(proto));
195  } else {
196  snprintf(proto, sizeof(proto), "PROTO:%03" PRIu32, IP_GET_IPPROTO(p));
197  }
198 
199  /* loop through alerts stored in the packet */
200  SCMutexLock(&td->lua_ctx->m);
201  uint16_t cnt;
202  for (cnt = 0; cnt < p->alerts.cnt; cnt++) {
203  const PacketAlert *pa = &p->alerts.alerts[cnt];
204  if (unlikely(pa->s == NULL)) {
205  continue;
206  }
207 
208  lua_getglobal(td->lua_ctx->luastate, "log");
209 
210  LuaStateSetThreadVars(td->lua_ctx->luastate, tv);
211  LuaStateSetPacket(td->lua_ctx->luastate, (Packet *)p);
212  LuaStateSetFlow(td->lua_ctx->luastate, p->flow, /* unlocked */LUA_FLOW_NOT_LOCKED_BY_PARENT);
213  LuaStateSetPacketAlert(td->lua_ctx->luastate, (PacketAlert *)pa);
214 
215  /* prepare data to pass to script */
216  //lua_newtable(td->lua_ctx->luastate);
217 
218  int retval = lua_pcall(td->lua_ctx->luastate, 0, 0, 0);
219  if (retval != 0) {
220  SCLogInfo("failed to run script: %s", lua_tostring(td->lua_ctx->luastate, -1));
221  }
222  }
223  SCMutexUnlock(&td->lua_ctx->m);
224 not_supported:
225  SCReturnInt(0);
226 }

275 static int LuaPacketConditionTls(ThreadVars *tv, const Packet *p)
276 {
277  if (p->flow == NULL) {
278  return FALSE;
279  }
280 
281  if (!(PKT_IS_IPV4(p)) && !(PKT_IS_IPV6(p))) {
282  return FALSE;
283  }
284 
285  if (!(PKT_IS_TCP(p))) {
286  return FALSE;
287  }
288 
289  FLOWLOCK_RDLOCK(p->flow);
290  uint16_t proto = FlowGetAppProtocol(p->flow);
291  if (proto != ALPROTO_TLS)
292  goto dontlog;
293 
294  SSLState *ssl_state = (SSLState *)FlowGetAppState(p->flow);
295  if (ssl_state == NULL) {
296  SCLogDebug("no tls state, so no request logging");
297  goto dontlog;
298  }
299 
300  if (ssl_state->server_connp.cert0_issuerdn == NULL ||
301  ssl_state->server_connp.cert0_subject == NULL)
302  goto dontlog;
303 
304  /* We only log the state once */
305  if (ssl_state->flags & SSL_AL_FLAG_STATE_LOGGED_LUA)
306  goto dontlog;
307 
308  FLOWLOCK_UNLOCK(p->flow);
309  return TRUE;
310 dontlog:
311  FLOWLOCK_UNLOCK(p->flow);
312  return FALSE;
313 }

528 static int LuaStatsLogger(ThreadVars *tv, void *thread_data, const StatsTable *st)
529 {
530  SCEnter();
531  LogLuaThreadCtx *td = (LogLuaThreadCtx *)thread_data;
532 
533  SCMutexLock(&td->lua_ctx->m);
534 
535  lua_State *luastate = td->lua_ctx->luastate;
536  /* get the lua function to call */
537  lua_getglobal(td->lua_ctx->luastate, "log");
538 
539  /* create lua array, which is really just a table. The key is an int (1-x),
540  * the value another table with named fields: name, tm_name, value, pvalue.
541  * { 1, { name=<name>, tmname=<tm_name>, value=<value>, pvalue=<pvalue>}}
542  * { 2, { name=<name>, tmname=<tm_name>, value=<value>, pvalue=<pvalue>}}
543  * etc
544  */
545  lua_newtable(luastate);
546  uint32_t u = 0;
547  for (; u < st->nstats; u++) {
548  lua_pushinteger(luastate, u + 1);
549 
550  lua_newtable(luastate);
551 
552  lua_pushstring(luastate, "name");
553  lua_pushstring(luastate, st->stats[u].name);
554  lua_settable(luastate, -3);
555 
556  lua_pushstring(luastate, "tmname");
557  lua_pushstring(luastate, st->stats[u].tm_name);
558  lua_settable(luastate, -3);
559 
560  lua_pushstring(luastate, "value");
561  lua_pushinteger(luastate, st->stats[u].value);
562  lua_settable(luastate, -3);
563 
564  lua_pushstring(luastate, "pvalue");
565  lua_pushinteger(luastate, st->stats[u].pvalue);
566  lua_settable(luastate, -3);
567 
568  lua_settable(luastate, -3);
569  }
570 
571  int retval = lua_pcall(td->lua_ctx->luastate, 1, 0, 0);
572  if (retval != 0) {
573  SCLogInfo("failed to run script: %s", lua_tostring(td->lua_ctx->luastate, -1));
574  }
575  SCMutexUnlock(&td->lua_ctx->m);
576  return 0;

621  if (status) {

861  * \brief initialize output instance for lua module
862  *
863  * Parses nested script list, primes them to find out what they
864  * inspect, then fills the OutputCtx::submodules list with the
865  * proper Logger function for the data type the script needs.
866  */
867 static OutputCtx *OutputLuaLogInit(ConfNode *conf)
868 {

870  if (dir == NULL)
871  dir = "";

874  if (scripts == NULL) {

876  SCLogInfo("scripts not defined");

879

882  if (unlikely(output_ctx == NULL)) {

896  ConfNode *script;
897  TAILQ_FOREACH(script, &scripts->head, next) {

899  LogLuaScriptOptions opts;
900  memset(&opts, 0x00, sizeof(opts));
901 
902  char path[PATH_MAX] = "";
903  snprintf(path, sizeof(path),"%s%s%s", dir, strlen(dir) ? "/" : "", script->val);
904  SCLogDebug("script full path %s", path);

908  SCLogError(SC_ERR_LUA_ERROR, "couldn't initialize scipt");

913  * on it's needs. */
914  OutputModule *om = SCCalloc(1, sizeof(*om));

916  SCLogError(SC_ERR_MEM_ALLOC, "calloc() failed");
917  goto error;
918  }

921  om->conf_name = script->val;

927  AppLayerHtpEnableRequestBodyCallback();
928  AppLayerHtpEnableResponseBodyCallback();

933  } else if (opts.alproto == ALPROTO_TLS) {

935  om->PacketConditionFunc = LuaPacketConditionTls;
936  } else if (opts.alproto == ALPROTO_DNS) {
937  om->TxLogFunc = LuaTxLogger;
938  om->alproto = ALPROTO_DNS;
939  AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_DNS);
940  AppLayerParserRegisterLogger(IPPROTO_UDP, ALPROTO_DNS);
941  } else if (opts.alproto == ALPROTO_SSH) {
942  om->PacketLogFunc = LuaPacketLoggerSsh;
943  om->PacketConditionFunc = LuaPacketConditionSsh;
944  } else if (opts.packet && opts.alerts) {
945  om->PacketLogFunc = LuaPacketLoggerAlerts;
946  om->PacketConditionFunc = LuaPacketConditionAlerts;
947  } else if (opts.packet && opts.alerts == 0) {
948  om->PacketLogFunc = LuaPacketLogger;
949  om->PacketConditionFunc = LuaPacketCondition;
950  } else if (opts.file) {
951  om->FileLogFunc = LuaFileLogger;
952  AppLayerHtpNeedFileInspection();
953  } else if (opts.streaming && opts.tcp_data) {
954  om->StreamingLogFunc = LuaStreamingLogger;
955  } else if (opts.flow) {
956  om->FlowLogFunc = LuaFlowLogger;
957  } else if (opts.stats) {
958  om->StatsLogFunc = LuaStatsLogger;
959  } else {
960  SCLogError(SC_ERR_LUA_ERROR, "failed to setup thread module");
961  SCFree(om);
962  goto error;
963  }
964 
965  TAILQ_INSERT_TAIL(&output_ctx->submodules, om, entries);
966  }
967 
968  return output_ctx;
969 
970 error:
971 
972  if (output_ctx != NULL) {
973  if (output_ctx->DeInit && output_ctx->data)
974  output_ctx->DeInit(output_ctx->data);
975  SCFree(output_ctx);
976  }
977  return NULL;
978 }
