81 static void LogTlsLogExtended(LogTlsLogThread *aft, SSLState * state)
82 {
83  if (state->server_connp.cert0_fingerprint != NULL) {
84  MemBufferWriteString(aft->buffer, " SHA1='%s'", state->server_connp.cert0_fingerprint);
85  }
86  switch (state->server_connp.version) {
87  case TLS_VERSION_UNKNOWN:
88  MemBufferWriteString(aft->buffer, " VERSION='UNDETERMINED'");
89  break;
90  case SSL_VERSION_2:
91  MemBufferWriteString(aft->buffer, " VERSION='SSLv2'");
92  break;
93  case SSL_VERSION_3:
94  MemBufferWriteString(aft->buffer, " VERSION='SSLv3'");
95  break;
96  case TLS_VERSION_10:
97  MemBufferWriteString(aft->buffer, " VERSION='TLSv1'");
98  break;
99  case TLS_VERSION_11:
100  MemBufferWriteString(aft->buffer, " VERSION='TLS 1.1'");
101  break;
102  case TLS_VERSION_12:
103  MemBufferWriteString(aft->buffer, " VERSION='TLS 1.2'");
104  break;
105  default:
106  MemBufferWriteString(aft->buffer, " VERSION='0x%04x'",
107  state->server_connp.version);
108  break;
109  }
110  MemBufferWriteString(aft->buffer, "\n");
111 }

113 int TLSGetIPInformations(const Packet *p, char* srcip, size_t srcip_len,
114  Port* sp, char* dstip, size_t dstip_len,
115  Port* dp, int ipproto)
116 {
117  if ((PKT_IS_TOSERVER(p))) {
118  switch (ipproto) {
119  case AF_INET:
120  PrintInet(AF_INET, (const void *) GET_IPV4_SRC_ADDR_PTR(p), srcip, srcip_len);
121  PrintInet(AF_INET, (const void *) GET_IPV4_DST_ADDR_PTR(p), dstip, dstip_len);
122  break;
123  case AF_INET6:
124  PrintInet(AF_INET6, (const void *) GET_IPV6_SRC_ADDR(p), srcip, srcip_len);
125  PrintInet(AF_INET6, (const void *) GET_IPV6_DST_ADDR(p), dstip, dstip_len);
126  break;
127  default:
128  return 0;
129  }
130  *sp = p->sp;
131  *dp = p->dp;
132  } else {
133  switch (ipproto) {
134  case AF_INET:
135  PrintInet(AF_INET, (const void *) GET_IPV4_DST_ADDR_PTR(p), srcip, srcip_len);
136  PrintInet(AF_INET, (const void *) GET_IPV4_SRC_ADDR_PTR(p), dstip, dstip_len);
137  break;
138  case AF_INET6:
139  PrintInet(AF_INET6, (const void *) GET_IPV6_DST_ADDR(p), srcip, srcip_len);
140  PrintInet(AF_INET6, (const void *) GET_IPV6_SRC_ADDR(p), dstip, dstip_len);
141  break;
142  default:
143  return 0;
144  }
145  *sp = p->dp;
146  *dp = p->sp;
147  }
148  return 1;
149 }

151 static TmEcode LogTlsLogThreadInit(ThreadVars *t, void *initdata, void **data)
152 {
153  LogTlsLogThread *aft = SCMalloc(sizeof(LogTlsLogThread));
154  if (unlikely(aft == NULL))
155  return TM_ECODE_FAILED;
156  memset(aft, 0, sizeof(LogTlsLogThread));
157 
158  if (initdata == NULL) {
159  SCLogDebug( "Error getting context for TLSLog. \"initdata\" argument NULL");
160  SCFree(aft);
161  return TM_ECODE_FAILED;
162  }
163 
164  aft->buffer = MemBufferCreateNew(OUTPUT_BUFFER_SIZE);
165  if (aft->buffer == NULL) {
166  SCFree(aft);
167  return TM_ECODE_FAILED;
168  }
169 
170  /* Use the Ouptut Context (file pointer and mutex) */
171  aft->tlslog_ctx = ((OutputCtx *) initdata)->data;
172 
173  *data = (void *) aft;
174  return TM_ECODE_OK;
175 }

212 /** \brief Create a new tls log LogFileCtx.
213  * \param conf Pointer to ConfNode containing this loggers configuration.
214  * \return NULL if failure, LogFileCtx* to the file_ctx if succesful
215  * */
216 static OutputCtx *LogTlsLogInitCtx(ConfNode *conf)
217 {
218  if (OutputTlsLoggerEnable() != 0) {
219  SCLogError(SC_ERR_CONF_YAML_ERROR, "only one 'tls' logger "
220  "can be enabled");
221  return NULL;
222  }
223 
224  LogFileCtx* file_ctx = LogFileNewCtx();
225 
226  if (file_ctx == NULL) {
227  SCLogError(SC_ERR_TLS_LOG_GENERIC, "LogTlsLogInitCtx: Couldn't "
228  "create new file_ctx");
229  return NULL;
230  }
231 
232  if (SCConfLogOpenGeneric(conf, file_ctx, DEFAULT_LOG_FILENAME, 1) < 0) {
233  goto filectx_error;
234  }
235 
236  LogTlsFileCtx *tlslog_ctx = SCCalloc(1, sizeof(LogTlsFileCtx));
237  if (unlikely(tlslog_ctx == NULL))
238  goto filectx_error;
239  tlslog_ctx->file_ctx = file_ctx;
240 
241  const char *extended = ConfNodeLookupChildValue(conf, "extended");
242  if (extended == NULL) {
243  tlslog_ctx->flags |= LOG_TLS_DEFAULT;
244  } else {
245  if (ConfValIsTrue(extended)) {
246  tlslog_ctx->flags |= LOG_TLS_EXTENDED;
247  }
248  }
249 
250  OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
251  if (unlikely(output_ctx == NULL))
252  goto tlslog_error;
253  output_ctx->data = tlslog_ctx;
254  output_ctx->DeInit = LogTlsLogDeInitCtx;
255 
256  SCLogDebug("TLS log output initialized");
257 
258  /* enable the logger for the app layer */
259  AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_TLS);
260 
261  return output_ctx;
262 
263 tlslog_error:
264  SCFree(tlslog_ctx);
265 filectx_error:
266  LogFileFreeCtx(file_ctx);
267  return NULL;
268 }

270 /** \internal
271  * \brief Condition function for TLS logger
272  * \retval bool true or false -- log now?
273  */
274 static int LogTlsCondition(ThreadVars *tv, const Packet *p)
275 {
276  if (p->flow == NULL) {
277  return FALSE;
278  }
279 
280  if (!(PKT_IS_TCP(p))) {
281  return FALSE;
282  }
283 
284  FLOWLOCK_RDLOCK(p->flow);
285  uint16_t proto = FlowGetAppProtocol(p->flow);
286  if (proto != ALPROTO_TLS)
287  goto dontlog;
288 
289  SSLState *ssl_state = (SSLState *)FlowGetAppState(p->flow);
290  if (ssl_state == NULL) {
291  SCLogDebug("no tls state, so no request logging");
292  goto dontlog;
293  }
294 
295  /* we only log the state once if we don't have to write
296  * the cert due to tls.store keyword. */
297  if (!(ssl_state->server_connp.cert_log_flag & SSL_TLS_LOG_PEM) &&
298  (ssl_state->flags & SSL_AL_FLAG_STATE_LOGGED))
299  goto dontlog;
300 
301  if (ssl_state->server_connp.cert0_issuerdn == NULL ||
302  ssl_state->server_connp.cert0_subject == NULL)
303  goto dontlog;
304 
305  /* todo: logic to log once */
306 
307  FLOWLOCK_UNLOCK(p->flow);
308  return TRUE;
309 dontlog:
310  FLOWLOCK_UNLOCK(p->flow);
311  return FALSE;
312 }

314 static int LogTlsLogger(ThreadVars *tv, void *thread_data, const Packet *p)
315 {
316  LogTlsLogThread *aft = (LogTlsLogThread *)thread_data;
317  LogTlsFileCtx *hlog = aft->tlslog_ctx;
318  char timebuf[64];
319  int ipproto = (PKT_IS_IPV4(p)) ? AF_INET : AF_INET6;
320 
321  if (unlikely(p->flow == NULL)) {
322  return 0;
323  }
324 
325  /* check if we have TLS state or not */
326  FLOWLOCK_WRLOCK(p->flow);
327  uint16_t proto = FlowGetAppProtocol(p->flow);
328  if (proto != ALPROTO_TLS)
329  goto end;
330 
331  SSLState *ssl_state = (SSLState *)FlowGetAppState(p->flow);
332  if (unlikely(ssl_state == NULL)) {
333  goto end;
334  }
335 
336  if (ssl_state->server_connp.cert0_issuerdn == NULL || ssl_state->server_connp.cert0_subject == NULL)
337  goto end;
338 
339  /* Don't log again the state. If we are here it was because we had
340  * to store the cert. */
341  if (ssl_state->flags & SSL_AL_FLAG_STATE_LOGGED)
342  goto end;
343 
344  CreateTimeString(&p->ts, timebuf, sizeof(timebuf));
345 #define PRINT_BUF_LEN 46
346  char srcip[PRINT_BUF_LEN], dstip[PRINT_BUF_LEN];
347  Port sp, dp;
348  if (!TLSGetIPInformations(p, srcip, PRINT_BUF_LEN,
349  &sp, dstip, PRINT_BUF_LEN, &dp, ipproto)) {
350  goto end;
351  }
352 
353  MemBufferReset(aft->buffer);
354  MemBufferWriteString(aft->buffer,
355  "%s %s:%d -> %s:%d TLS: Subject='%s' Issuerdn='%s'",
356  timebuf, srcip, sp, dstip, dp,
357  ssl_state->server_connp.cert0_subject,
358  ssl_state->server_connp.cert0_issuerdn);
359 
360  if (hlog->flags & LOG_TLS_EXTENDED) {
361  LogTlsLogExtended(aft, ssl_state);
362  } else {
363  MemBufferWriteString(aft->buffer, "\n");
364  }
365 
366  aft->tls_cnt++;
367 
368  SCMutexLock(&hlog->file_ctx->fp_mutex);
369  hlog->file_ctx->Write((const char *)MEMBUFFER_BUFFER(aft->buffer),
370  MEMBUFFER_OFFSET(aft->buffer), hlog->file_ctx);
371  SCMutexUnlock(&hlog->file_ctx->fp_mutex);
372 
373  /* we only log the state once */
374  ssl_state->flags |= SSL_AL_FLAG_STATE_LOGGED;
375 end:
376  FLOWLOCK_UNLOCK(p->flow);
377  return 0;
378 }
