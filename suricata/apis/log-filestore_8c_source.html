169 static void LogFilestoreLogCreateMetaFile(const Packet *p, const File *ff, char *filename, int ipver) {
170  char metafilename[PATH_MAX] = "";
171  snprintf(metafilename, sizeof(metafilename), "%s.meta", filename);
172  FILE *fp = fopen(metafilename, "w+");
173  if (fp != NULL) {
174  char timebuf[64];
175 
176  CreateTimeString(&p->ts, timebuf, sizeof(timebuf));
177 
178  fprintf(fp, "TIME: %s\n", timebuf);
179  if (p->pcap_cnt > 0) {
180  fprintf(fp, "PCAP PKT NUM: %"PRIu64"\n", p->pcap_cnt);
181  }
182 
183  char srcip[46], dstip[46];
184  Port sp, dp;
185  switch (ipver) {
186  case AF_INET:
187  PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p), srcip, sizeof(srcip));
188  PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p), dstip, sizeof(dstip));
189  break;
190  case AF_INET6:
191  PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p), srcip, sizeof(srcip));
192  PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p), dstip, sizeof(dstip));
193  break;
194  default:
195  strlcpy(srcip, "<unknown>", sizeof(srcip));
196  strlcpy(dstip, "<unknown>", sizeof(dstip));
197  break;
198  }
199  sp = p->sp;
200  dp = p->dp;
201 
202  fprintf(fp, "SRC IP: %s\n", srcip);
203  fprintf(fp, "DST IP: %s\n", dstip);
204  fprintf(fp, "PROTO: %" PRIu32 "\n", p->proto);
205  if (PKT_IS_TCP(p) || PKT_IS_UDP(p)) {
206  fprintf(fp, "SRC PORT: %" PRIu16 "\n", sp);
207  fprintf(fp, "DST PORT: %" PRIu16 "\n", dp);
208  }
209 
210  fprintf(fp, "APP PROTO: %s\n",
211  AppProtoToString(p->flow->alproto));
212 
213  /* Only applicable to HTTP traffic */
214  if (p->flow->alproto == ALPROTO_HTTP) {
215  fprintf(fp, "HTTP URI: ");
216  LogFilestoreMetaGetUri(fp, p, ff);
217  fprintf(fp, "\n");
218  fprintf(fp, "HTTP HOST: ");
219  LogFilestoreMetaGetHost(fp, p, ff);
220  fprintf(fp, "\n");
221  fprintf(fp, "HTTP REFERER: ");
222  LogFilestoreMetaGetReferer(fp, p, ff);
223  fprintf(fp, "\n");
224  fprintf(fp, "HTTP USER AGENT: ");
225  LogFilestoreMetaGetUserAgent(fp, p, ff);
226  fprintf(fp, "\n");
227  } else if (p->flow->alproto == ALPROTO_SMTP) {
228  /* Only applicable to SMTP */
229  LogFilestoreMetaGetSmtp(fp, p, ff);
230  }
231 
232  fprintf(fp, "FILENAME: ");
233  PrintRawUriFp(fp, ff->name, ff->name_len);
234  fprintf(fp, "\n");
235 
236  fclose(fp);
237  }
238 }

240 static void LogFilestoreLogCloseMetaFile(const File *ff)
241 {
242  char filename[PATH_MAX] = "";
243  snprintf(filename, sizeof(filename), "%s/file.%u",
244  g_logfile_base_dir, ff->file_id);
245  char metafilename[PATH_MAX] = "";
246  snprintf(metafilename, sizeof(metafilename), "%s.meta", filename);
247  FILE *fp = fopen(metafilename, "a");
248  if (fp != NULL) {
249  fprintf(fp, "MAGIC: %s\n",
250  ff->magic ? ff->magic : "<unknown>");
251 
252  switch (ff->state) {
253  case FILE_STATE_CLOSED:
254  fprintf(fp, "STATE: CLOSED\n");
255 #ifdef HAVE_NSS
256  if (ff->flags & FILE_MD5) {
257  fprintf(fp, "MD5: ");
258  size_t x;
259  for (x = 0; x < sizeof(ff->md5); x++) {
260  fprintf(fp, "%02x", ff->md5[x]);
261  }
262  fprintf(fp, "\n");
263  }
264 #endif
265  break;
266  case FILE_STATE_TRUNCATED:
267  fprintf(fp, "STATE: TRUNCATED\n");
268  break;
269  case FILE_STATE_ERROR:
270  fprintf(fp, "STATE: ERROR\n");
271  break;
272  default:
273  fprintf(fp, "STATE: UNKNOWN\n");
274  break;
275  }
276  fprintf(fp, "SIZE: %"PRIu64"\n", ff->size);
277 
278  fclose(fp);
279  } else {
280  SCLogInfo("opening %s failed: %s", metafilename, strerror(errno));
281  }
282 }

284 static int LogFilestoreLogger(ThreadVars *tv, void *thread_data, const Packet *p, const File *ff, const FileData *ffd, uint8_t flags)
285 {
286  SCEnter();
287  LogFilestoreLogThread *aft = (LogFilestoreLogThread *)thread_data;
288  char filename[PATH_MAX] = "";
289  int file_fd = -1;
290  int ipver = -1;
291 
292  /* no flow, no htp state */
293  if (p->flow == NULL) {
294  SCReturnInt(TM_ECODE_OK);
295  }
296 
297  if (PKT_IS_IPV4(p)) {
298  ipver = AF_INET;
299  } else if (PKT_IS_IPV6(p)) {
300  ipver = AF_INET6;
301  } else {
302  return 0;
303  }
304 
305  SCLogDebug("ff %p, ffd %p", ff, ffd);
306 
307  snprintf(filename, sizeof(filename), "%s/file.%u",
308  g_logfile_base_dir, ff->file_id);
309 
310  if (flags & OUTPUT_FILEDATA_FLAG_OPEN) {
311  aft->file_cnt++;
312 
313  /* create a .meta file that contains time, src/dst/sp/dp/proto */
314  LogFilestoreLogCreateMetaFile(p, ff, filename, ipver);
315 
316  file_fd = open(filename, O_CREAT | O_TRUNC | O_NOFOLLOW | O_WRONLY, 0644);
317  if (file_fd == -1) {
318  SCLogDebug("failed to create file");
319  return -1;
320  }
321  /* we can get called with a NULL ffd when we need to close */
322  } else if (ffd != NULL) {
323  file_fd = open(filename, O_APPEND | O_NOFOLLOW | O_WRONLY);
324  if (file_fd == -1) {
325  SCLogDebug("failed to open file %s: %s", filename, strerror(errno));
326  return -1;
327  }
328  }
329 
330  if (file_fd != -1) {
331  ssize_t r = write(file_fd, (const void *)ffd->data, (size_t)ffd->len);
332  if (r == -1) {
333  SCLogDebug("write failed: %s", strerror(errno));
334  }
335  close(file_fd);
336  }
337 
338  if (flags & OUTPUT_FILEDATA_FLAG_CLOSE) {
339  LogFilestoreLogCloseMetaFile(ff);
340  }
341 
342  return 0;
343 }

345 static TmEcode LogFilestoreLogThreadInit(ThreadVars *t, void *initdata, void **data)
346 {
347  LogFilestoreLogThread *aft = SCMalloc(sizeof(LogFilestoreLogThread));
348  if (unlikely(aft == NULL))
349  return TM_ECODE_FAILED;
350  memset(aft, 0, sizeof(LogFilestoreLogThread));
351 
352  if (initdata == NULL)
353  {
354  SCLogDebug("Error getting context for LogFilestore. \"initdata\" argument NULL");
355  SCFree(aft);
356  return TM_ECODE_FAILED;
357  }
358 
359  /* Use the Ouptut Context (file pointer and mutex) */
360  aft->file_ctx = ((OutputCtx *)initdata)->data;
361 
362  struct stat stat_buf;
363  if (stat(g_logfile_base_dir, &stat_buf) != 0) {
364  int ret;
365  ret = mkdir(g_logfile_base_dir, S_IRWXU|S_IXGRP|S_IRGRP);
366  if (ret != 0) {
367  int err = errno;
368  if (err != EEXIST) {
369  SCLogError(SC_ERR_LOGDIR_CONFIG,
370  "Cannot create file drop directory %s: %s",
371  g_logfile_base_dir, strerror(err));
372  exit(EXIT_FAILURE);
373  }
374  } else {
375  SCLogInfo("Created file drop directory %s",
376  g_logfile_base_dir);
377  }
378 
379  }
380 
381  *data = (void *)aft;
382  return TM_ECODE_OK;
383 }

424 /** \brief Create a new http log LogFilestoreCtx.
425  * \param conf Pointer to ConfNode containing this loggers configuration.
426  * \return NULL if failure, LogFilestoreCtx* to the file_ctx if succesful
427  * */
428 static OutputCtx *LogFilestoreLogInitCtx(ConfNode *conf)
429 {
430  LogFileCtx *logfile_ctx = LogFileNewCtx();
431  if (logfile_ctx == NULL) {
432  SCLogDebug("Could not create new LogFilestoreCtx");
433  return NULL;
434  }
435 
436  OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
437  if (unlikely(output_ctx == NULL))
438  return NULL;
439 
440  output_ctx->data = NULL;
441  output_ctx->DeInit = LogFilestoreLogDeInitCtx;
442 
443  char *s_default_log_dir = NULL;
444  s_default_log_dir = ConfigGetLogDirectory();
445 
446  const char *s_base_dir = NULL;
447  s_base_dir = ConfNodeLookupChildValue(conf, "log-dir");
448  if (s_base_dir == NULL || strlen(s_base_dir) == 0) {
449  strlcpy(g_logfile_base_dir,
450  s_default_log_dir, sizeof(g_logfile_base_dir));
451  } else {
452  if (PathIsAbsolute(s_base_dir)) {
453  strlcpy(g_logfile_base_dir,
454  s_base_dir, sizeof(g_logfile_base_dir));
455  } else {
456  snprintf(g_logfile_base_dir, sizeof(g_logfile_base_dir),
457  "%s/%s", s_default_log_dir, s_base_dir);
458  }
459  }
460 
461  const char *force_magic = ConfNodeLookupChildValue(conf, "force-magic");
462  if (force_magic != NULL && ConfValIsTrue(force_magic)) {
463  FileForceMagicEnable();
464  SCLogInfo("forcing magic lookup for stored files");
465  }
466 
467  const char *force_md5 = ConfNodeLookupChildValue(conf, "force-md5");
468  if (force_md5 != NULL && ConfValIsTrue(force_md5)) {
469 #ifdef HAVE_NSS
470  FileForceMd5Enable();
471  SCLogInfo("forcing md5 calculation for stored files");
472 #else
473  SCLogInfo("md5 calculation requires linking against libnss");
474 #endif
475  }
476  SCLogInfo("storing files in %s", g_logfile_base_dir);
477 
478  SCReturnPtr(output_ctx, "OutputCtx");
479 }
