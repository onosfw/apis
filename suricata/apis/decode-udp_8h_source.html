62 /**
63  * \brief Calculates the checksum for the UDP packet
64  *
65  * \param shdr Pointer to source address field from the IP packet. Used as a
66  * part of the psuedoheader for computing the checksum
67  * \param pkt Pointer to the start of the UDP packet
68  * \param hlen Total length of the UDP packet(header + payload)
69  *
70  * \retval csum Checksum for the UDP packet
71  */
72 static inline uint16_t UDPV4CalculateChecksum(uint16_t *shdr, uint16_t *pkt,
73  uint16_t tlen)
74 {
75  uint16_t pad = 0;
76  uint32_t csum = shdr[0];
77 
78  csum += shdr[1] + shdr[2] + shdr[3] + htons(17) + htons(tlen);
79 
80  csum += pkt[0] + pkt[1] + pkt[2];
81 
82  tlen -= 8;
83  pkt += 4;
84 
85  while (tlen >= 32) {
86  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5] + pkt[6] +
87  pkt[7] + pkt[8] + pkt[9] + pkt[10] + pkt[11] + pkt[12] + pkt[13] +
88  pkt[14] + pkt[15];
89  tlen -= 32;
90  pkt += 16;
91  }
92 
93  while(tlen >= 8) {
94  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3];
95  tlen -= 8;
96  pkt += 4;
97  }
98 
99  while(tlen >= 4) {
100  csum += pkt[0] + pkt[1];
101  tlen -= 4;
102  pkt += 2;
103  }
104 
105  while (tlen > 1) {
106  csum += pkt[0];
107  pkt += 1;
108  tlen -= 2;
109  }
110 
111  if (tlen == 1) {
112  *(uint8_t *)(&pad) = (*(uint8_t *)pkt);
113  csum += pad;
114  }
115 
116  csum = (csum >> 16) + (csum & 0x0000FFFF);
117  csum += (csum >> 16);
118

120  if (csum_u16 == 0)
121  return 0xFFFF;
