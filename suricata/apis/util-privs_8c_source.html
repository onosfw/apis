64 /**
65  * \brief Drop the previliges of the main thread
66  */
67 void SCDropMainThreadCaps(uint32_t userid, uint32_t groupid)
68 {
69  if (sc_set_caps == FALSE)
70  return;
71 
72  capng_clear(CAPNG_SELECT_BOTH);
73 
74  switch (run_mode) {
75  case RUNMODE_PCAP_DEV:
76  case RUNMODE_AFP_DEV:
77  capng_updatev(CAPNG_ADD, CAPNG_EFFECTIVE|CAPNG_PERMITTED,
78  CAP_NET_RAW, /* needed for pcap live mode */
79  -1);
80  break;
81  case RUNMODE_PFRING:
82  capng_updatev(CAPNG_ADD, CAPNG_EFFECTIVE|CAPNG_PERMITTED,
83  CAP_NET_ADMIN, CAP_NET_RAW,
84  -1);
85  break;
86  case RUNMODE_NFQ:
87  capng_updatev(CAPNG_ADD, CAPNG_EFFECTIVE|CAPNG_PERMITTED,
88  CAP_NET_ADMIN, /* needed for nfqueue inline mode */
89  -1);
90  break;
91  }
92 
93  if (capng_change_id(userid, groupid, CAPNG_DROP_SUPP_GRP |
94  CAPNG_CLEAR_BOUNDING) < 0)
95  {
96  SCLogError(SC_ERR_CHANGING_CAPS_FAILED, "capng_change_id for main thread"
97  " failed");
98  exit(EXIT_FAILURE);
99  }
100 
101  SCLogInfo("dropped the caps for main thread");
102 }

149 /**
150  * \brief Function to get the user and group ID from the specified user name
151  *
152  * \param user_name pointer to the given user name
153  * \param uid pointer to the user id in which result will be stored
154  * \param gid pointer to the group id in which result will be stored
155  *
156  * \retval upon success it return 0
157  */
158 int SCGetUserID(char *user_name, char *group_name, uint32_t *uid, uint32_t *gid)
159 {
160  uint32_t userid = 0;
161  uint32_t groupid = 0;
162  struct passwd *pw;
163 
164  /* Get the user ID */
165  if (isdigit((unsigned char)user_name[0]) != 0) {
166  userid = atoi(user_name);
167  pw = getpwuid(userid);
168  if (pw == NULL) {
169  SCLogError(SC_ERR_UID_FAILED, "unable to get the user ID, "
170  "check if user exist!!");
171  exit(EXIT_FAILURE);
172  }
173  } else {
174  pw = getpwnam(user_name);
175  if (pw == NULL) {
176  SCLogError(SC_ERR_UID_FAILED, "unable to get the user ID, "
177  "check if user exist!!");
178  exit(EXIT_FAILURE);
179  }
180  userid = pw->pw_uid;
181  }
182 
183  /* Get the group ID */
184  if (group_name != NULL) {
185  struct group *gp;
186 
187  if (isdigit((unsigned char)group_name[0]) != 0) {
188  groupid = atoi(group_name);
189  } else {
190  gp = getgrnam(group_name);
191  if (gp == NULL) {
192  SCLogError(SC_ERR_GID_FAILED, "unable to get the group"
193  " ID, check if group exist!!");
194  exit(EXIT_FAILURE);
195  }
196  groupid = gp->gr_gid;
197  }
198  } else {
199  groupid = pw->pw_gid;
200  }
201 
202  /* close the group database */
203  endgrent();
204  /* close the user database */
205  endpwent();
206 
207  *uid = userid;
208  *gid = groupid;
209 
210  return 0;
211 }
