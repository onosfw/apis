62 /**
63  * \brief Set an app layer decoder event.
64  *
65  * \param sevents Pointer to a AppLayerDecoderEvents pointer. If *sevents is NULL
66  * memory will be allocated.
67  * \param event The event to be stored.
68  */
69 void AppLayerDecoderEventsSetEventRaw(AppLayerDecoderEvents **sevents, uint8_t event)
70 {
71  if (*sevents == NULL) {
72  AppLayerDecoderEvents *new_devents = SCMalloc(sizeof(AppLayerDecoderEvents));
73  if (new_devents == NULL)
74  return;
75 
76  memset(new_devents, 0, sizeof(AppLayerDecoderEvents));
77  *sevents = new_devents;
78 
79  }
80  if ((*sevents)->cnt == UCHAR_MAX) {
81  /* we're full */
82  return;
83  }
84  if ((*sevents)->cnt == (*sevents)->events_buffer_size) {
85  int steps = DECODER_EVENTS_BUFFER_STEPS;
86  if (UCHAR_MAX - (*sevents)->cnt < steps)
87  steps = UCHAR_MAX - (*sevents)->cnt < steps;
88 
89  void *ptr = SCRealloc((*sevents)->events,
90  ((*sevents)->cnt + steps) * sizeof(uint8_t));
91  if (ptr == NULL) {
92  /* couldn't grow buffer, but no reason to free old
93  * so we keep the events that may already be here */
94  return;
95  }
96  (*sevents)->events = ptr;
97  (*sevents)->events_buffer_size += steps;
98  }
99 
100  (*sevents)->events[(*sevents)->cnt++] = event;
101 }

103 /**
104  * \brief Set an app layer decoder event.
105  *
106  * \param f Pointer to a flow containing DecoderEvents pointer head. If
107  * the head points to a DecoderEvents instance, a
108  * new instance would be created and the pointer head would
109  * would be updated with this new instance
110  * \param event The event to be stored.

112 void AppLayerDecoderEventsSetEvent(Flow *f, uint8_t event)
113 {
114  AppLayerDecoderEvents *events = AppLayerParserGetDecoderEvents(f->alparser);
115  AppLayerDecoderEvents *new = events;
116  AppLayerDecoderEventsSetEventRaw(&events, event);
117  if (events != new)
118  AppLayerParserSetDecoderEvents(f->alparser, events);
