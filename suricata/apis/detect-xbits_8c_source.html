65 void DetectXbitsRegister (void)
66 {
67  sigmatch_table[DETECT_XBITS].name = "xbits";
68  sigmatch_table[DETECT_XBITS].desc = "operate on bits";
69 // sigmatch_table[DETECT_XBITS].url = "https://redmine.openinfosecfoundation.org/projects/suricata/wiki/Flow-keywords#Flowbits";
70  sigmatch_table[DETECT_XBITS].Match = DetectXbitMatch;
71  sigmatch_table[DETECT_XBITS].Setup = DetectXbitSetup;
72  sigmatch_table[DETECT_XBITS].Free = DetectXbitFree;
73  sigmatch_table[DETECT_XBITS].RegisterTests = XBitsRegisterTests;
74  /* this is compatible to ip-only signatures */
75  sigmatch_table[DETECT_XBITS].flags |= SIGMATCH_IPONLY_COMPAT;
76 
77  const char *eb;
78  int eo;
79  int opts = 0;
80 
81  parse_regex = pcre_compile(PARSE_REGEX, opts, &eb, &eo, NULL);
82  if(parse_regex == NULL)
83  {
84  SCLogError(SC_ERR_PCRE_COMPILE, "pcre compile of \"%s\" failed at offset %" PRId32 ": %s", PARSE_REGEX, eo, eb);
85  goto error;
86  }
87 
88  parse_regex_study = pcre_study(parse_regex, 0, &eb);
89  if(eb != NULL)
90  {
91  SCLogError(SC_ERR_PCRE_STUDY, "pcre study failed: %s", eb);
92  goto error;
93  }
94 
95  return;
96 
97 error:
98  return;
99 }

204 int DetectXbitSetup (DetectEngineCtx *de_ctx, Signature *s, char *rawstr)
205 {
206  DetectXbitsData *cd = NULL;
207  SigMatch *sm = NULL;
208  uint8_t fb_cmd = 0;
209  uint8_t hb_dir = 0;
210 #define MAX_SUBSTRINGS 30
211  int ret = 0, res = 0;
212  int ov[MAX_SUBSTRINGS];
213  char fb_cmd_str[16] = "", fb_name[256] = "";
214  char hb_dir_str[16] = "";
215  enum VarTypes var_type = VAR_TYPE_NOT_SET;
216  int expire = 30;
217 
218  ret = pcre_exec(parse_regex, parse_regex_study, rawstr, strlen(rawstr), 0, 0, ov, MAX_SUBSTRINGS);
219  if (ret != 2 && ret != 3 && ret != 4 && ret != 5) {
220  SCLogError(SC_ERR_PCRE_MATCH, "\"%s\" is not a valid setting for xbits.", rawstr);
221  return -1;
222  }
223  SCLogDebug("ret %d, %s", ret, rawstr);
224  res = pcre_copy_substring((char *)rawstr, ov, MAX_SUBSTRINGS, 1, fb_cmd_str, sizeof(fb_cmd_str));
225  if (res < 0) {
226  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_copy_substring failed");
227  return -1;
228  }
229 
230  if (ret >= 3) {
231  res = pcre_copy_substring((char *)rawstr, ov, MAX_SUBSTRINGS, 2, fb_name, sizeof(fb_name));
232  if (res < 0) {
233  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_copy_substring failed");
234  goto error;
235  }
236  if (ret >= 4) {
237  res = pcre_copy_substring((char *)rawstr, ov, MAX_SUBSTRINGS, 3, hb_dir_str, sizeof(hb_dir_str));
238  if (res < 0) {
239  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_copy_substring failed");
240  goto error;
241  }
242  SCLogDebug("hb_dir_str %s", hb_dir_str);
243  if (strlen(hb_dir_str) > 0) {
244  if (strcmp(hb_dir_str, "ip_src") == 0) {
245  hb_dir = DETECT_XBITS_TRACK_IPSRC;
246  var_type = VAR_TYPE_HOST_BIT;
247  } else if (strcmp(hb_dir_str, "ip_dst") == 0) {
248  hb_dir = DETECT_XBITS_TRACK_IPDST;
249  var_type = VAR_TYPE_HOST_BIT;
250  } else if (strcmp(hb_dir_str, "ip_pair") == 0) {
251  hb_dir = DETECT_XBITS_TRACK_IPPAIR;
252  var_type = VAR_TYPE_IPPAIR_BIT;
253  } else {
254  // TODO
255  goto error;
256  }
257  }
258 
259  if (ret >= 5) {
260  char expire_str[16] = "";
261  res = pcre_copy_substring((char *)rawstr, ov, MAX_SUBSTRINGS, 4, expire_str, sizeof(expire_str));
262  if (res < 0) {
263  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_copy_substring failed");
264  goto error;
265  }
266  SCLogDebug("expire_str %s", expire_str);
267  expire = atoi(expire_str);
268  SCLogDebug("expire %d", expire);
269  }
270  }
271  }
272 
273  if (strcmp(fb_cmd_str,"noalert") == 0) {
274  fb_cmd = DETECT_XBITS_CMD_NOALERT;
275  } else if (strcmp(fb_cmd_str,"isset") == 0) {
276  fb_cmd = DETECT_XBITS_CMD_ISSET;
277  } else if (strcmp(fb_cmd_str,"isnotset") == 0) {
278  fb_cmd = DETECT_XBITS_CMD_ISNOTSET;
279  } else if (strcmp(fb_cmd_str,"set") == 0) {
280  fb_cmd = DETECT_XBITS_CMD_SET;
281  } else if (strcmp(fb_cmd_str,"unset") == 0) {
282  fb_cmd = DETECT_XBITS_CMD_UNSET;
283  } else if (strcmp(fb_cmd_str,"toggle") == 0) {
284  fb_cmd = DETECT_XBITS_CMD_TOGGLE;
285  } else {
286  SCLogError(SC_ERR_UNKNOWN_VALUE, "ERROR: flowbits action \"%s\" is not supported.", fb_cmd_str);
287  goto error;
288  }
289 
290  switch (fb_cmd) {
291  case DETECT_XBITS_CMD_NOALERT:
292  if (strlen(fb_name) != 0)
293  goto error;
294  s->flags |= SIG_FLAG_NOALERT;
295  return 0;
296  case DETECT_XBITS_CMD_ISNOTSET:
297  case DETECT_XBITS_CMD_ISSET:
298  case DETECT_XBITS_CMD_SET:
299  case DETECT_XBITS_CMD_UNSET:
300  case DETECT_XBITS_CMD_TOGGLE:
301  default:
302  if (strlen(fb_name) == 0)
303  goto error;
304  break;
305  }
306 
307  cd = SCMalloc(sizeof(DetectXbitsData));
308  if (unlikely(cd == NULL))
309  goto error;
310 
311  cd->idx = VariableNameGetIdx(de_ctx, fb_name, var_type);
312  cd->cmd = fb_cmd;
313  cd->tracker = hb_dir;
314  cd->type = var_type;
315  cd->expire = expire;
316 
317  SCLogDebug("idx %" PRIu32 ", cmd %s, name %s",
318  cd->idx, fb_cmd_str, strlen(fb_name) ? fb_name : "(none)");
319 
320  /* Okay so far so good, lets get this into a SigMatch
321  * and put it in the Signature. */
322  sm = SigMatchAlloc();
323  if (sm == NULL)
324  goto error;
325 
326  sm->type = DETECT_XBITS;
327  sm->ctx = (void *)cd;
328 
329  switch (fb_cmd) {
330  /* case DETECT_XBITS_CMD_NOALERT can't happen here */
331 
332  case DETECT_XBITS_CMD_ISNOTSET:
333  case DETECT_XBITS_CMD_ISSET:
334  /* checks, so packet list */
335  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);
336  break;
337 
338  case DETECT_XBITS_CMD_SET:
339  case DETECT_XBITS_CMD_UNSET:
340  case DETECT_XBITS_CMD_TOGGLE:
341  /* modifiers, only run when entire sig has matched */
342  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_POSTMATCH);
343  break;
344  }
345 
346  return 0;
347 
348 error:
349  if (cd != NULL)
350  SCFree(cd);
351  if (sm != NULL)
352  SCFree(sm);
353  return -1;
354 }

407

411  p->dst.family = AF_INET;
412  p->payload = buf;
413  p->payload_len = buflen;
414  p->proto = IPPROTO_TCP;
415 
416  XBitsTestSetup();
417 
418  de_ctx = DetectEngineCtxInit();

451  if (de_ctx != NULL) {
452  DetectEngineCtxFree(de_ctx);
453  }
454 
455  XBitsTestShutdown();
456 
457  SCFree(p);
