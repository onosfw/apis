110 static void FormatNumber(uint64_t num, char *str, size_t size)
111 {
112  if (num < 1000UL)
113  snprintf(str, size, "%"PRIu64, num);
114  else if (num < 1000000UL)
115  snprintf(str, size, "%3.1fk", (float)num/1000UL);
116  else if (num < 1000000000UL)
117  snprintf(str, size, "%3.1fm", (float)num/1000000UL);
118  else
119  snprintf(str, size, "%3.1fb", (float)num/1000000000UL);
120 }

126 SCProfilingInit(void)
127 {
128  ConfNode *conf;
129 
130  SC_ATOMIC_INIT(samples);
131 
132  intmax_t rate_v = 0;
133  (void)ConfGetInt("profiling.sample-rate", &rate_v);

135  rate = (int)rate_v;

142  conf = ConfGetNode("profiling.packets");
143  if (conf != NULL) {
144  if (ConfNodeChildValueIsTrue(conf, "enabled")) {
145  profiling_packets_enabled = 1;
146 
147  if (pthread_mutex_init(&packet_profile_lock, NULL) != 0) {
148  SCLogError(SC_ERR_MUTEX,
149  "Failed to initialize packet profiling mutex.");
150  exit(EXIT_FAILURE);
151  }
152  memset(&packet_profile_data4, 0, sizeof(packet_profile_data4));
153  memset(&packet_profile_data6, 0, sizeof(packet_profile_data6));
154  memset(&packet_profile_tmm_data4, 0, sizeof(packet_profile_tmm_data4));
155  memset(&packet_profile_tmm_data6, 0, sizeof(packet_profile_tmm_data6));
156  memset(&packet_profile_app_data4, 0, sizeof(packet_profile_app_data4));
157  memset(&packet_profile_app_data6, 0, sizeof(packet_profile_app_data6));

159  memset(&packet_profile_app_pd_data6, 0, sizeof(packet_profile_app_pd_data6));
160  memset(&packet_profile_detect_data4, 0, sizeof(packet_profile_detect_data4));
161  memset(&packet_profile_detect_data6, 0, sizeof(packet_profile_detect_data6));
162 
163  const char *filename = ConfNodeLookupChildValue(conf, "filename");
164  if (filename != NULL) {
165 
166  char *log_dir;
167  log_dir = ConfigGetLogDirectory();
168 
169  profiling_packets_file_name = SCMalloc(PATH_MAX);
170  if (unlikely(profiling_packets_file_name == NULL)) {
171  SCLogError(SC_ERR_MEM_ALLOC, "can't duplicate file name");
172  exit(EXIT_FAILURE);
173  }
174 
175  snprintf(profiling_packets_file_name, PATH_MAX, "%s/%s", log_dir, filename);
176 
177  const char *v = ConfNodeLookupChildValue(conf, "append");
178  if (v == NULL || ConfValIsTrue(v)) {
179  profiling_packets_file_mode = "a";
180  } else {
181  profiling_packets_file_mode = "w";
182  }
183 
184  profiling_packets_output_to_file = 1;
185  }
186  }
187 
188  conf = ConfGetNode("profiling.packets.csv");
189  if (conf != NULL) {
190  if (ConfNodeChildValueIsTrue(conf, "enabled")) {
191 
192  const char *filename = ConfNodeLookupChildValue(conf, "filename");
193  if (filename == NULL) {
194  filename = "packet_profile.csv";
195  }
196 
197  char *log_dir;
198  log_dir = ConfigGetLogDirectory();
199 
200  profiling_csv_file_name = SCMalloc(PATH_MAX);
201  if (unlikely(profiling_csv_file_name == NULL)) {
202  SCLogError(SC_ERR_MEM_ALLOC, "out of memory");
203  exit(EXIT_FAILURE);
204  }

207  packet_profile_csv_fp = fopen(profiling_csv_file_name, "w");

225 
226  profiling_packets_csv_enabled = 1;
227  }
228  }
229  }
230 
231  conf = ConfGetNode("profiling.locks");
232  if (conf != NULL) {

234 #ifndef PROFILE_LOCKING

236 #else

238 
239  LockRecordInitHash();
240 
241  const char *filename = ConfNodeLookupChildValue(conf, "filename");
242  if (filename != NULL) {
243  char *log_dir;
244  log_dir = ConfigGetLogDirectory();
245 
246  profiling_locks_file_name = SCMalloc(PATH_MAX);
247  if (unlikely(profiling_locks_file_name == NULL)) {
248  SCLogError(SC_ERR_MEM_ALLOC, "can't duplicate file name");

250  }
251 
252  snprintf(profiling_locks_file_name, PATH_MAX, "%s/%s", log_dir, filename);
253 
254  const char *v = ConfNodeLookupChildValue(conf, "append");
255  if (v == NULL || ConfValIsTrue(v)) {
256  profiling_locks_file_mode = "a";
257  } else {
258  profiling_locks_file_mode = "w";
259  }
260 
261  profiling_locks_output_to_file = 1;
262  }
263 #endif
264  }
265  }

273 SCProfilingDestroy(void)
274 {
275  if (profiling_packets_enabled) {
276  pthread_mutex_destroy(&packet_profile_lock);

278

284

286  SCFree(profiling_csv_file_name);

288

295 #endif
