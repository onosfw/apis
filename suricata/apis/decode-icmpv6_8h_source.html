185 /**
186  * \brief Calculates the checksum for the ICMPV6 packet
187  *
188  * \param shdr Pointer to source address field from the IPV6 packet. Used as a
189  * part of the psuedoheader for computing the checksum
190  * \param pkt Pointer to the start of the ICMPV6 packet
191  * \param tlen Total length of the ICMPV6 packet(header + payload)
192  *
193  * \retval csum Checksum for the ICMPV6 packet
194  */
195 static inline uint16_t ICMPV6CalculateChecksum(uint16_t *shdr, uint16_t *pkt,
196  uint16_t tlen)
197 {
198  uint16_t pad = 0;
199  uint32_t csum = shdr[0];
200 
201  csum += shdr[1] + shdr[2] + shdr[3] + shdr[4] + shdr[5] + shdr[6] +
202  shdr[7] + shdr[8] + shdr[9] + shdr[10] + shdr[11] + shdr[12] +
203  shdr[13] + shdr[14] + shdr[15] + htons(58 + tlen);
204 
205  csum += pkt[0];
206 
207  tlen -= 4;
208  pkt += 2;
209 
210  while (tlen >= 64) {
211  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5] + pkt[6] +
212  pkt[7] + pkt[8] + pkt[9] + pkt[10] + pkt[11] + pkt[12] + pkt[13] +
213  pkt[14] + pkt[15] + pkt[16] + pkt[17] + pkt[18] + pkt[19] +
214  pkt[20] + pkt[21] + pkt[22] + pkt[23] + pkt[24] + pkt[25] +
215  pkt[26] + pkt[27] + pkt[28] + pkt[29] + pkt[30] + pkt[31];
216  tlen -= 64;
217  pkt += 32;
218  }
219 
220  while (tlen >= 32) {
221  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5] + pkt[6] +
222  pkt[7] + pkt[8] + pkt[9] + pkt[10] + pkt[11] + pkt[12] + pkt[13] +
223  pkt[14] + pkt[15];
224  tlen -= 32;
225  pkt += 16;
226  }
227 
228  while(tlen >= 8) {
229  csum += pkt[0] + pkt[1] + pkt[2] + pkt[3];
230  tlen -= 8;
231  pkt += 4;
232  }
233 
234  while(tlen >= 4) {
235  csum += pkt[0] + pkt[1];
236  tlen -= 4;
237  pkt += 2;
238  }
239 
240  while (tlen > 1) {
241  csum += pkt[0];
242  tlen -= 2;
243  pkt += 1;
244  }
245 
246  if (tlen == 1) {
247  *(uint8_t *)(&pad) = (*(uint8_t *)pkt);
248  csum += pad;
249  }
250 
251  csum = (csum >> 16) + (csum & 0x0000FFFF);
252  csum += (csum >> 16);
253 
254  return (uint16_t) ~csum;
255 }
