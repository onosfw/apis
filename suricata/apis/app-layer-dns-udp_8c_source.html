54  * \brief Parse DNS request packet

56 static int DNSUDPRequestParse(Flow *f, void *dstate,
57  AppLayerParserState *pstate,
58  uint8_t *input, uint32_t input_len,
59  void *local_data)
60 {
61  DNSState *dns_state = (DNSState *)dstate;
62 
63  SCLogDebug("starting %u", input_len);
64 
65  if (input == NULL && AppLayerParserStateIssetFlag(pstate, APP_LAYER_PARSER_EOF)) {
66  SCReturnInt(1);
67  }
68 
69  /** \todo remove this when PP is fixed to enforce ipproto */
70  if (f != NULL && f->proto != IPPROTO_UDP)
71  SCReturnInt(-1);
72 
73  if (input == NULL || input_len == 0 || input_len < sizeof(DNSHeader)) {
74  SCLogDebug("ilen too small, hoped for at least %"PRIuMAX, (uintmax_t)sizeof(DNSHeader));
75  goto insufficient_data;
76  }
77 
78  DNSHeader *dns_header = (DNSHeader *)input;
79  SCLogDebug("DNS %p", dns_header);
80 
81  if (DNSValidateRequestHeader(dns_state, dns_header) < 0)
82  goto bad_data;
83 
84  uint16_t q;
85  const uint8_t *data = input + sizeof(DNSHeader);
86  for (q = 0; q < ntohs(dns_header->questions); q++) {
87  uint8_t fqdn[DNS_MAX_SIZE];
88  uint16_t fqdn_offset = 0;
89 
90  if (input + input_len < data + 1) {
91  SCLogDebug("input buffer too small for len");
92  goto insufficient_data;
93  }
94  SCLogDebug("query length %u", *data);
95 
96  while (*data != 0) {
97  if (*data > 63) {
98  /** \todo set event?*/
99  goto insufficient_data;
100  }
101  uint8_t length = *data;
102 
103  data++;
104 
105  if (length == 0) {
106  break;
107  }
108 
109  if (input + input_len < data + length) {
110  SCLogDebug("input buffer too small for domain of len %u", length);
111  goto insufficient_data;
112  }
113  //PrintRawDataFp(stdout, data, qry->length);
114 
115  if ((size_t)(fqdn_offset + length + 1) < sizeof(fqdn)) {
116  memcpy(fqdn + fqdn_offset, data, length);
117  fqdn_offset += length;
118  fqdn[fqdn_offset++] = '.';
119  } else {
120  /** \todo set event? */
121  goto insufficient_data;
122  }
123 
124  data += length;
125 
126  if (input + input_len < data + 1) {
127  SCLogDebug("input buffer too small for len(2)");
128  goto insufficient_data;
129  }
130 
131  SCLogDebug("qry length %u", *data);
132  }
133  if (fqdn_offset) {
134  fqdn_offset--;
135  }
136 
137  data++;
138  if (input + input_len < data + sizeof(DNSQueryTrailer)) {
139  SCLogDebug("input buffer too small for DNSQueryTrailer");
140  goto insufficient_data;
141  }
142  DNSQueryTrailer *trailer = (DNSQueryTrailer *)data;
143  SCLogDebug("trailer type %04x class %04x", ntohs(trailer->type), ntohs(trailer->class));
144  data += sizeof(DNSQueryTrailer);
145 
146  /* store our data */
147  if (dns_state != NULL) {
148  DNSStoreQueryInState(dns_state, fqdn, fqdn_offset,
149  ntohs(trailer->type), ntohs(trailer->class),
150  ntohs(dns_header->tx_id));
151  }
152  }
153 
154  SCReturnInt(1);
155 bad_data:
156 insufficient_data:
157  SCReturnInt(-1);
158 }

161  * \brief DNS UDP record parser, entry function

166 static int DNSUDPResponseParse(Flow *f, void *dstate,
167  AppLayerParserState *pstate,
168  uint8_t *input, uint32_t input_len,
169  void *local_data)
170 {
171  DNSState *dns_state = (DNSState *)dstate;
172 
173  SCLogDebug("starting %u", input_len);
174 
175  if (input == NULL && AppLayerParserStateIssetFlag(pstate, APP_LAYER_PARSER_EOF)) {
176  SCReturnInt(1);
177  }
178 
179  /** \todo remove this when PP is fixed to enforce ipproto */
180  if (f != NULL && f->proto != IPPROTO_UDP)
181  SCReturnInt(-1);
182 
183  if (input == NULL || input_len == 0 || input_len < sizeof(DNSHeader)) {
184  SCLogDebug("ilen too small, hoped for at least %"PRIuMAX, (uintmax_t)sizeof(DNSHeader));
185  goto insufficient_data;
186  }
187 
188  DNSHeader *dns_header = (DNSHeader *)input;
189  SCLogDebug("DNS %p %04x %04x", dns_header, ntohs(dns_header->tx_id), dns_header->flags);
190 
191  DNSTransaction *tx = NULL;
192  int found = 0;
193  if ((tx = DNSTransactionFindByTxId(dns_state, ntohs(dns_header->tx_id))) != NULL)
194  found = 1;
195 
196  if (!found) {
197  SCLogDebug("DNS_DECODER_EVENT_UNSOLLICITED_RESPONSE");
198  DNSSetEvent(dns_state, DNS_DECODER_EVENT_UNSOLLICITED_RESPONSE);
199  }
200 
201  if (DNSValidateResponseHeader(dns_state, dns_header) < 0)
202  goto bad_data;
203 
204  SCLogDebug("queries %04x", ntohs(dns_header->questions));
205 
206  uint16_t q;
207  const uint8_t *data = input + sizeof(DNSHeader);
208  for (q = 0; q < ntohs(dns_header->questions); q++) {
209  uint8_t fqdn[DNS_MAX_SIZE];
210  uint16_t fqdn_offset = 0;
211 
212  if (input + input_len < data + 1) {
213  SCLogDebug("input buffer too small for len");
214  goto insufficient_data;
215  }
216  SCLogDebug("qry length %u", *data);
217 
218  while (*data != 0) {
219  uint8_t length = *data;
220  data++;
221 
222  if (length == 0)
223  break;
224 
225  if (input + input_len < data + length) {
226  SCLogDebug("input buffer too small for domain of len %u", length);
227  goto insufficient_data;
228  }
229  //PrintRawDataFp(stdout, data, length);
230 
231  if ((size_t)(fqdn_offset + length + 1) < sizeof(fqdn)) {
232  memcpy(fqdn + fqdn_offset, data, length);
233  fqdn_offset += length;
234  fqdn[fqdn_offset++] = '.';
235  }
236 
237  data += length;
238 
239  if (input + input_len < data + 1) {
240  SCLogDebug("input buffer too small for len");
241  goto insufficient_data;
242  }
243 
244  SCLogDebug("length %u", *data);
245  }
246  if (fqdn_offset) {
247  fqdn_offset--;
248  }
249 
250  data++;
251  if (input + input_len < data + sizeof(DNSQueryTrailer)) {
252  SCLogDebug("input buffer too small for DNSQueryTrailer");
253  goto insufficient_data;
254  }
255 #if DEBUG
256  DNSQueryTrailer *trailer = (DNSQueryTrailer *)data;
257  SCLogDebug("trailer type %04x class %04x", ntohs(trailer->type), ntohs(trailer->class));
258 #endif
259  data += sizeof(DNSQueryTrailer);
260  }
261 
262  SCLogDebug("answer_rr %04x", ntohs(dns_header->answer_rr));
263  for (q = 0; q < ntohs(dns_header->answer_rr); q++) {
264  data = DNSReponseParse(dns_state, dns_header, q, DNS_LIST_ANSWER,
265  input, input_len, data);
266  if (data == NULL) {
267  goto insufficient_data;
268  }
269  }
270 
271  SCLogDebug("authority_rr %04x", ntohs(dns_header->authority_rr));
272  for (q = 0; q < ntohs(dns_header->authority_rr); q++) {
273  data = DNSReponseParse(dns_state, dns_header, q, DNS_LIST_AUTHORITY,
274  input, input_len, data);
275  if (data == NULL) {
276  goto insufficient_data;
277  }
278  }
279 
280  /* parse rcode, e.g. "noerror" or "nxdomain" */
281  uint8_t rcode = ntohs(dns_header->flags) & 0x0F;
282  if (rcode <= DNS_RCODE_NOTZONE) {
283  SCLogDebug("rcode %u", rcode);
284  if (tx != NULL)
285  tx->rcode = rcode;
286  } else {
287  /* this is not invalid, rcodes can be user defined */
288  SCLogDebug("unexpected DNS rcode %u", rcode);
289  }
290 
291  if (ntohs(dns_header->flags) & 0x0080) {
292  SCLogDebug("recursion desired");
293  if (tx != NULL)
294  tx->recursion_desired = 1;
295  }
296 
297  if (tx != NULL) {
298  tx->replied = 1;
299  }
300 
301  SCReturnInt(1);
302 
303 bad_data:
304 insufficient_data:
305  DNSSetEvent(dns_state, DNS_DECODER_EVENT_MALFORMED_DATA);
306  SCReturnInt(-1);
307 }

309 static uint16_t DNSUdpProbingParser(uint8_t *input, uint32_t ilen, uint32_t *offset)
310 {
311  if (ilen == 0 || ilen < sizeof(DNSHeader)) {
312  SCLogDebug("ilen too small, hoped for at least %"PRIuMAX, (uintmax_t)sizeof(DNSHeader));
313  return ALPROTO_UNKNOWN;
314  }
315 
316  if (DNSUDPRequestParse(NULL, NULL, NULL, input, ilen, NULL) == -1)
317  return ALPROTO_FAILED;
318 
319  return ALPROTO_DNS;
320 }

322 static void DNSUDPConfigure(void)
323 {
324  uint32_t request_flood = DNS_CONFIG_DEFAULT_REQUEST_FLOOD;
325  uint32_t state_memcap = DNS_CONFIG_DEFAULT_STATE_MEMCAP;
326  uint64_t global_memcap = DNS_CONFIG_DEFAULT_GLOBAL_MEMCAP;
327 
328  ConfNode *p = ConfGetNode("app-layer.protocols.dns.request-flood");
329  if (p != NULL) {
330  uint32_t value;
331  if (ParseSizeStringU32(p->val, &value) < 0) {
332  SCLogError(SC_ERR_DNS_CONFIG, "invalid value for request-flood %s", p->val);
333  } else {
334  request_flood = value;
335  }
336  }
337  SCLogInfo("DNS request flood protection level: %u", request_flood);
338  DNSConfigSetRequestFlood(request_flood);
339 
340  p = ConfGetNode("app-layer.protocols.dns.state-memcap");
341  if (p != NULL) {
342  uint32_t value;

344  SCLogError(SC_ERR_DNS_CONFIG, "invalid value for state-memcap %s", p->val);
345  } else {

348  }
349  SCLogInfo("DNS per flow memcap (state-memcap): %u", state_memcap);
350  DNSConfigSetStateMemcap(state_memcap);
351 
352  p = ConfGetNode("app-layer.protocols.dns.global-memcap");

365 void RegisterDNSUDPParsers(void)
366 {
367  char *proto_name = "dns";
368 
369  /** DNS */
370  if (AppLayerProtoDetectConfProtoDetectionEnabled("udp", proto_name)) {
371  AppLayerProtoDetectRegisterProtocol(ALPROTO_DNS, proto_name);
372 
373  if (RunmodeIsUnittests()) {
374  AppLayerProtoDetectPPRegister(IPPROTO_UDP,
375  "53",
376  ALPROTO_DNS,
377  0, sizeof(DNSHeader),
378  STREAM_TOSERVER,
379  DNSUdpProbingParser);
380  } else {
381  int have_cfg = AppLayerProtoDetectPPParseConfPorts("udp", IPPROTO_UDP,
382  proto_name, ALPROTO_DNS,
383  0, sizeof(DNSHeader),
384  DNSUdpProbingParser);
385  /* if we have no config, we enable the default port 53 */
386  if (!have_cfg) {
387  SCLogWarning(SC_ERR_DNS_CONFIG, "no DNS UDP config found, "
388  "enabling DNS detection on "
389  "port 53.");
390  AppLayerProtoDetectPPRegister(IPPROTO_UDP, "53",
391  ALPROTO_DNS, 0, sizeof(DNSHeader),
392  STREAM_TOSERVER, DNSUdpProbingParser);
393  }
394  }
395  } else {
396  SCLogInfo("Protocol detection and parser disabled for %s protocol.",
397  proto_name);
398  return;
399  }
400 
401  if (AppLayerParserConfParserEnabled("udp", proto_name)) {
402  AppLayerParserRegisterParser(IPPROTO_UDP, ALPROTO_DNS, STREAM_TOSERVER,
403  DNSUDPRequestParse);
404  AppLayerParserRegisterParser(IPPROTO_UDP, ALPROTO_DNS, STREAM_TOCLIENT,
405  DNSUDPResponseParse);
406  AppLayerParserRegisterStateFuncs(IPPROTO_UDP, ALPROTO_DNS, DNSStateAlloc,
407  DNSStateFree);
408  AppLayerParserRegisterTxFreeFunc(IPPROTO_UDP, ALPROTO_DNS,
409  DNSStateTransactionFree);
410 
411  AppLayerParserRegisterGetEventsFunc(IPPROTO_UDP, ALPROTO_DNS, DNSGetEvents);
412  AppLayerParserRegisterHasEventsFunc(IPPROTO_UDP, ALPROTO_DNS, DNSHasEvents);
413  AppLayerParserRegisterDetectStateFuncs(IPPROTO_UDP, ALPROTO_DNS,
414  DNSStateHasTxDetectState,
415  DNSGetTxDetectState, DNSSetTxDetectState);
416 
417  AppLayerParserRegisterGetTx(IPPROTO_UDP, ALPROTO_DNS,
418  DNSGetTx);
419  AppLayerParserRegisterGetTxCnt(IPPROTO_UDP, ALPROTO_DNS,
420  DNSGetTxCnt);
421  AppLayerParserRegisterGetStateProgressFunc(IPPROTO_UDP, ALPROTO_DNS,
422  DNSGetAlstateProgress);
423  AppLayerParserRegisterGetStateProgressCompletionStatus(IPPROTO_UDP, ALPROTO_DNS,
424  DNSGetAlstateProgressCompletionStatus);
425 
426  DNSAppLayerRegisterGetEventInfo(IPPROTO_UDP, ALPROTO_DNS);
427 
428  DNSUDPConfigure();
429  } else {
430  SCLogInfo("Parsed disabled for %s protocol. Protocol detection"
431  "still on.", proto_name);
432  }
433 #ifdef UNITTESTS
434  AppLayerParserRegisterProtocolUnittests(IPPROTO_UDP, ALPROTO_DNS, DNSUDPParserRegisterTests);
435 #endif
436 }
