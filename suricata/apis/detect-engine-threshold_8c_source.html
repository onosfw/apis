101 DetectThresholdData *SigGetThresholdTypeIter(Signature *sig, Packet *p, SigMatch **psm, int list)
102 {
103  SigMatch *sm = NULL;
104  DetectThresholdData *tsh = NULL;
105 
106  if (sig == NULL)
107  return NULL;
108 
109  if (*psm == NULL) {
110  sm = sig->sm_lists_tail[list];
111  } else {
112  /* Iteration in progress, using provided value */
113  sm = *psm;
114  }
115 
116  if (p == NULL)
117  return NULL;
118 
119  while (sm != NULL) {
120  if (sm->type == DETECT_THRESHOLD || sm->type == DETECT_DETECTION_FILTER) {
121  tsh = (DetectThresholdData *)sm->ctx;
122  *psm = sm->prev;
123  return tsh;
124  }
125 
126  sm = sm->prev;
127  }
128  *psm = NULL;
129 
130  return NULL;
131 }

140 int ThresholdTimeoutCheck(Host *host, struct timeval *tv)
141 {
142  DetectThresholdEntry *tde = NULL;
143  DetectThresholdEntry *tmp = NULL;
144  DetectThresholdEntry *prev = NULL;
145  int retval = 1;
146 
147  tmp = HostGetStorageById(host, threshold_id);
148  if (tmp == NULL)
149  return 1;
150 
151  prev = NULL;
152  while (tmp != NULL) {
153  if ((tv->tv_sec - tmp->tv_sec1) <= tmp->seconds) {
154  prev = tmp;
155  tmp = tmp->next;
156  retval = 0;
157  continue;
158  }
159 
160  /* timed out */
161 
162  if (prev != NULL) {
163  prev->next = tmp->next;
164 
165  tde = tmp;
166  tmp = tde->next;
167 
168  SCFree(tde);
169  } else {
170  HostSetStorageById(host, threshold_id, tmp->next);
171  tde = tmp;
172  tmp = tde->next;
173 
174  SCFree(tde);
175  }
176  }
177 
178  return retval;
179 }

212 int ThresholdHandlePacketSuppress(Packet *p, DetectThresholdData *td, uint32_t sid, uint32_t gid)
213 {
214  int ret = 0;
215  DetectAddress *m = NULL;
216  switch (td->track) {
217  case TRACK_DST:
218  m = DetectAddressLookupInHead(&td->addrs, &p->dst);
219  SCLogDebug("TRACK_DST");
220  break;
221  case TRACK_SRC:
222  m = DetectAddressLookupInHead(&td->addrs, &p->src);
223  SCLogDebug("TRACK_SRC");
224  break;
225  /* suppress if either src or dst is a match on the suppress
226  * address list */
227  case TRACK_EITHER:
228  m = DetectAddressLookupInHead(&td->addrs, &p->src);
229  if (m == NULL) {
230  m = DetectAddressLookupInHead(&td->addrs, &p->dst);
231  }
232  break;
233  case TRACK_RULE:
234  default:
235  SCLogError(SC_ERR_INVALID_VALUE,
236  "track mode %d is not supported", td->track);
237  break;
238  }
239  if (m == NULL)
240  ret = 1;
241  else
242  ret = 2; /* suppressed but still need actions */
243 
244  return ret;
245 }

247 /**
248  * \retval 2 silent match (no alert but apply actions)
249  * \retval 1 normal match
250  * \retval 0 no match
251  */
252 int ThresholdHandlePacketHost(Host *h, Packet *p, DetectThresholdData *td, uint32_t sid, uint32_t gid)
253 {
254  int ret = 0;
255 
256  DetectThresholdEntry *lookup_tsh = ThresholdHostLookupEntry(h, sid, gid);
257  SCLogDebug("lookup_tsh %p sid %u gid %u", lookup_tsh, sid, gid);
258 
259  switch(td->type) {
260  case TYPE_LIMIT:
261  {
262  SCLogDebug("limit");
263 
264  if (lookup_tsh != NULL) {
265  if ((p->ts.tv_sec - lookup_tsh->tv_sec1) < td->seconds) {
266  lookup_tsh->current_count++;
267 
268  if (lookup_tsh->current_count <= td->count) {
269  ret = 1;
270  } else {
271  ret = 2;
272  }
273  } else {
274  lookup_tsh->tv_sec1 = p->ts.tv_sec;
275  lookup_tsh->current_count = 1;
276 
277  ret = 1;
278  }
279  } else {
280  DetectThresholdEntry *e = DetectThresholdEntryAlloc(td, p, sid, gid);
281  if (e == NULL) {
282  break;
283  }
284 
285  e->tv_sec1 = p->ts.tv_sec;
286  e->current_count = 1;
287 
288  ret = 1;
289 
290  e->next = HostGetStorageById(h, threshold_id);
291  HostSetStorageById(h, threshold_id, e);
292  }
293  break;
294  }
295  case TYPE_THRESHOLD:
296  {
297  SCLogDebug("threshold");
298 
299  if (lookup_tsh != NULL) {
300  if ((p->ts.tv_sec - lookup_tsh->tv_sec1) < td->seconds) {
301  lookup_tsh->current_count++;
302 
303  if (lookup_tsh->current_count >= td->count) {
304  ret = 1;
305  lookup_tsh->current_count = 0;
306  }
307  } else {
308  lookup_tsh->tv_sec1 = p->ts.tv_sec;
309  lookup_tsh->current_count = 1;
310  }
311  } else {
312  if (td->count == 1) {
313  ret = 1;
314  } else {
315  DetectThresholdEntry *e = DetectThresholdEntryAlloc(td, p, sid, gid);
316  if (e == NULL) {
317  break;
318  }
319 
320  e->current_count = 1;
321  e->tv_sec1 = p->ts.tv_sec;
322 
323  e->next = HostGetStorageById(h, threshold_id);
324  HostSetStorageById(h, threshold_id, e);
325  }
326  }
327  break;
328  }
329  case TYPE_BOTH:
330  {
331  SCLogDebug("both");
332 
333  if (lookup_tsh != NULL) {
334  if ((p->ts.tv_sec - lookup_tsh->tv_sec1) < td->seconds) {
335  /* within time limit */
336 
337  lookup_tsh->current_count++;
338  if (lookup_tsh->current_count == td->count) {
339  ret = 1;
340  } else if (lookup_tsh->current_count > td->count) {
341  /* silent match */
342  ret = 2;
343  }
344  } else {
345  /* expired, so reset */
346  lookup_tsh->tv_sec1 = p->ts.tv_sec;
347  lookup_tsh->current_count = 1;
348 
349  /* if we have a limit of 1, this is a match */
350  if (lookup_tsh->current_count == td->count) {
351  ret = 1;
352  }
353  }
354  } else {
355  DetectThresholdEntry *e = DetectThresholdEntryAlloc(td, p, sid, gid);
356  if (e == NULL) {
357  break;
358  }
359 
360  e->current_count = 1;
361  e->tv_sec1 = p->ts.tv_sec;
362 
363  e->next = HostGetStorageById(h, threshold_id);
364  HostSetStorageById(h, threshold_id, e);
365 
366  /* for the first match we return 1 to
367  * indicate we should alert */
368  if (td->count == 1) {
369  ret = 1;
370  }
371  }
372  break;
373  }
374  /* detection_filter */
375  case TYPE_DETECTION:
376  {
377  SCLogDebug("detection_filter");
378 
379  if (lookup_tsh != NULL) {
380  long double time_diff = ((p->ts.tv_sec + p->ts.tv_usec/1000000.0) -
381  (lookup_tsh->tv_sec1 + lookup_tsh->tv_usec1/1000000.0));
382 
383  if (time_diff < td->seconds) {
384  /* within timeout */
385 
386  lookup_tsh->current_count++;
387  if (lookup_tsh->current_count > td->count) {
388  ret = 1;
389  }
390  } else {
391  /* expired, reset */
392 
393  lookup_tsh->tv_sec1 = p->ts.tv_sec;
394  lookup_tsh->tv_usec1 = p->ts.tv_usec;
395  lookup_tsh->current_count = 1;
396  }
397  } else {
398  DetectThresholdEntry *e = DetectThresholdEntryAlloc(td, p, sid, gid);
399  if (e == NULL) {
400  break;
401  }
402 
403  e->current_count = 1;
404  e->tv_sec1 = p->ts.tv_sec;
405  e->tv_usec1 = p->ts.tv_usec;
406 
407  e->next = HostGetStorageById(h, threshold_id);
408  HostSetStorageById(h, threshold_id, e);
409  }
410  break;
411  }
412  /* rate_filter */
413  case TYPE_RATE:
414  {
415  SCLogDebug("rate_filter");
416 
417  ret = 1;
418 
419  if (lookup_tsh != NULL) {
420  /* Check if we have a timeout enabled, if so,
421  * we still matching (and enabling the new_action) */
422  if (lookup_tsh->tv_timeout != 0) {
423  if ((p->ts.tv_sec - lookup_tsh->tv_timeout) > td->timeout) {
424  /* Ok, we are done, timeout reached */
425  lookup_tsh->tv_timeout = 0;
426  } else {
427  /* Already matching */
428  /* Take the action to perform */
429  switch (td->new_action) {
430  case TH_ACTION_ALERT:
431  PACKET_ALERT(p);
432  break;
433  case TH_ACTION_DROP:
434  PACKET_DROP(p);
435  break;
436  case TH_ACTION_REJECT:
437  PACKET_REJECT(p);
438  break;
439  case TH_ACTION_PASS:
440  PACKET_PASS(p);
441  break;
442  default:
443  /* Weird, leave the default action */
444  break;
445  }
446  ret = 1;
447  } /* else - if ((p->ts.tv_sec - lookup_tsh->tv_timeout) > td->timeout) */
448 
449  } else {
450  /* Update the matching state with the timeout interval */
451  if ( (p->ts.tv_sec - lookup_tsh->tv_sec1) < td->seconds) {
452  lookup_tsh->current_count++;
453  if (lookup_tsh->current_count > td->count) {
454  /* Then we must enable the new action by setting a
455  * timeout */
456  lookup_tsh->tv_timeout = p->ts.tv_sec;
457  /* Take the action to perform */
458  switch (td->new_action) {
459  case TH_ACTION_ALERT:
460  PACKET_ALERT(p);
461  break;
462  case TH_ACTION_DROP:
463  PACKET_DROP(p);
464  break;
465  case TH_ACTION_REJECT:
466  PACKET_REJECT(p);
467  break;
468  case TH_ACTION_PASS:
469  PACKET_PASS(p);
470  break;
471  default:
472  /* Weird, leave the default action */
473  break;
474  }
475  ret = 1;
476  }
477  } else {
478  lookup_tsh->tv_sec1 = p->ts.tv_sec;
479  lookup_tsh->current_count = 1;
480  }
481  } /* else - if (lookup_tsh->tv_timeout != 0) */
482  } else {
483  if (td->count == 1) {
484  ret = 1;
485  }
486 
487  DetectThresholdEntry *e = DetectThresholdEntryAlloc(td, p, sid, gid);
488  if (e == NULL) {
489  break;
490  }
491 
492  e->current_count = 1;
493  e->tv_sec1 = p->ts.tv_sec;
494  e->tv_timeout = 0;
495 
496  e->next = HostGetStorageById(h, threshold_id);
497  HostSetStorageById(h, threshold_id, e);
498  }
499  break;
500  }
501  /* case TYPE_SUPPRESS: is not handled here */
502  default:
503  SCLogError(SC_ERR_INVALID_VALUE, "type %d is not supported", td->type);
504  }
505 
506  return ret;
507 }

509 static int ThresholdHandlePacketRule(DetectEngineCtx *de_ctx, Packet *p, DetectThresholdData *td, Signature *s)
510 {
511  int ret = 0;
512 
513  if (td->type != TYPE_RATE)
514  return 1;
515 
516  DetectThresholdEntry* lookup_tsh = (DetectThresholdEntry *)de_ctx->ths_ctx.th_entry[s->num];
517  if (lookup_tsh != NULL) {
518  /* Check if we have a timeout enabled, if so,
519  * we still matching (and enabling the new_action) */
520  if ( (p->ts.tv_sec - lookup_tsh->tv_timeout) > td->timeout) {
521  /* Ok, we are done, timeout reached */
522  td->timeout = 0;
523  } else {
524  /* Already matching */
525  /* Take the action to perform */
526  switch (td->new_action) {
527  case TH_ACTION_ALERT:
528  PACKET_ALERT(p);
529  break;
530  case TH_ACTION_DROP:
531  PACKET_DROP(p);
532  break;
533  case TH_ACTION_REJECT:
534  PACKET_REJECT(p);
535  break;
536  case TH_ACTION_PASS:
537  PACKET_PASS(p);
538  break;
539  default:
540  /* Weird, leave the default action */
541  break;
542  }
543  ret = 1;
544  }
545 
546  /* Update the matching state with the timeout interval */
547  if ( (p->ts.tv_sec - lookup_tsh->tv_sec1) < td->seconds) {
548  lookup_tsh->current_count++;
549  if (lookup_tsh->current_count >= td->count) {
550  /* Then we must enable the new action by setting a
551  * timeout */
552  lookup_tsh->tv_timeout = p->ts.tv_sec;

595 /**
596  * \brief Make the threshold logic for signatures
597  *
598  * \param de_ctx Dectection Context
599  * \param tsh_ptr Threshold element
600  * \param p Packet structure
601  * \param s Signature structure
602  *
603  * \retval 2 silent match (no alert but apply actions)
604  * \retval 1 alert on this event
605  * \retval 0 do not alert on this event
606  */
607 int PacketAlertThreshold(DetectEngineCtx *de_ctx, DetectEngineThreadCtx *det_ctx,
608  DetectThresholdData *td, Packet *p, Signature *s)
609 {
610  SCEnter();
611 
612  int ret = 0;
613  if (td == NULL) {
614  SCReturnInt(0);
615  }
616 
617  if (td->type == TYPE_SUPPRESS) {
618  ret = ThresholdHandlePacketSuppress(p,td,s->id,s->gid);
619  } else if (td->track == TRACK_SRC) {
620  Host *src = HostGetHostFromHash(&p->src);
621  if (src) {
622  ret = ThresholdHandlePacketHost(src,p,td,s->id,s->gid);
623  HostRelease(src);
624  }
625  } else if (td->track == TRACK_DST) {
626  Host *dst = HostGetHostFromHash(&p->dst);
627  if (dst) {
628  ret = ThresholdHandlePacketHost(dst,p,td,s->id,s->gid);
629  HostRelease(dst);
630  }
631  } else if (td->track == TRACK_RULE) {
632  SCMutexLock(&de_ctx->ths_ctx.threshold_table_lock);
633  ret = ThresholdHandlePacketRule(de_ctx,p,td,s);
634  SCMutexUnlock(&de_ctx->ths_ctx.threshold_table_lock);
635  }
636 
637  SCReturnInt(ret);
638 }
