50 /** \brief get the flowvar with index 'idx' from the flow
51  * \note flow is not locked by this function, caller is
52  * responsible
53  */
54 FlowVar *FlowVarGet(Flow *f, uint16_t idx)
55 {
56  GenericVar *gv = f->flowvar;
57 
58  for ( ; gv != NULL; gv = gv->next) {
59  if (gv->type == DETECT_FLOWVAR && gv->idx == idx)
60  return (FlowVar *)gv;
61  }
62 
63  return NULL;
64 }

66 /* add a flowvar to the flow, or update it */
67 void FlowVarAddStrNoLock(Flow *f, uint16_t idx, uint8_t *value, uint16_t size)
68 {
69  FlowVar *fv = FlowVarGet(f, idx);
70  if (fv == NULL) {
71  fv = SCMalloc(sizeof(FlowVar));
72  if (unlikely(fv == NULL))
73  return;
74 
75  fv->type = DETECT_FLOWVAR;
76  fv->datatype = FLOWVAR_TYPE_STR;
77  fv->idx = idx;
78  fv->data.fv_str.value = value;
79  fv->data.fv_str.value_len = size;
80  fv->next = NULL;
81 
82  GenericVarAppend(&f->flowvar, (GenericVar *)fv);
83  } else {
84  FlowVarUpdateStr(fv, value, size);
85  }
86 }

137 void FlowVarPrint(GenericVar *gv)
138 {
139  uint16_t u;
140 
141  if (!SCLogDebugEnabled())
142  return;
143 
144  if (gv == NULL)
145  return;
146 
147  if (gv->type == DETECT_FLOWVAR || gv->type == DETECT_FLOWINT) {
148  FlowVar *fv = (FlowVar *)gv;
149 
150  if (fv->datatype == FLOWVAR_TYPE_STR) {
151  SCLogDebug("Name idx \"%" PRIu16 "\", Value \"", fv->idx);
152  for (u = 0; u < fv->data.fv_str.value_len; u++) {
153  if (isprint(fv->data.fv_str.value[u]))
154  SCLogDebug("%c", fv->data.fv_str.value[u]);
155  else
156  SCLogDebug("\\%02X", fv->data.fv_str.value[u]);
157  }
158  SCLogDebug("\", Len \"%" PRIu16 "\"\n", fv->data.fv_str.value_len);
159  } else if (fv->datatype == FLOWVAR_TYPE_INT) {
160  SCLogDebug("Name idx \"%" PRIu16 "\", Value \"%" PRIu32 "\"", fv->idx,
161  fv->data.fv_int.value);
162  } else {
163  SCLogDebug("Unknown data type at flowvars\n");
164  }
165  }
166  FlowVarPrint(gv->next);
167 }
