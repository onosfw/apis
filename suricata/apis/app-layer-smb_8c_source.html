631 static uint32_t PaddingParser(void *smb_state, AppLayerParserState *pstate,
632  uint8_t *input, uint32_t input_len)
633 {
634  SCEnter();
635 
636  SMBState *sstate = (SMBState *) smb_state;
637  uint8_t *p = input;
638 
639  /* Check for validity of dataoffset */
640  if ((uint64_t)(sstate->bytesprocessed - NBSS_HDR_LEN) > sstate->andx.dataoffset) {
641  sstate->andx.paddingparsed = 1;
642  SCReturnUInt((uint32_t)(p - input));
643  }
644  while (((uint64_t)(sstate->bytesprocessed - NBSS_HDR_LEN) + (p - input))
645  < sstate->andx.dataoffset && sstate->bytecount.bytecountleft--
646  && input_len--) {
647  SCLogDebug("0x%02x ", *p);
648  p++;
649  }
650  if (((uint64_t)(sstate->bytesprocessed - NBSS_HDR_LEN) + (p - input))
651  == sstate->andx.dataoffset) {
652  sstate->andx.paddingparsed = 1;
653  }
654  sstate->bytesprocessed += (p - input);
655  SCReturnUInt((uint32_t)(p - input));
656 }

1471 static uint16_t SMBProbingParser(uint8_t *input, uint32_t ilen, uint32_t *offset)
1472 {
1473  int32_t len;
1474  int32_t input_len = ilen;
1475 
1476  while (input_len >= SMB_PROBING_PARSER_MIN_DEPTH) {
1477  switch (input[0]) {
1478  case NBSS_SESSION_MESSAGE:
1479  if (input[4] == 0xFF && input[5] == 'S' && input[6] == 'M' &&
1480  input[7] == 'B') {
1481  return ALPROTO_SMB;
1482  }
1483 
1484  /* fall through */
1485  case NBSS_SESSION_REQUEST:
1486  case NBSS_POSITIVE_SESSION_RESPONSE:
1487  case NBSS_NEGATIVE_SESSION_RESPONSE:
1488  case NBSS_RETARGET_SESSION_RESPONSE:
1489  case NBSS_SESSION_KEEP_ALIVE:
1490  len = (input[1] & 0x01) << 16;
1491  len |= input[2] << 8;
1492  len |= input[3];
1493  break;
1494  default:
1495  /* -1 indicates a stream where the probing parser would be
1496  * unable to find nbss, even if it exists. This should
1497  * prevent the probing parser from beig invoked henceforth */
1498  return ALPROTO_FAILED;
1499  }
1500 
1501  input_len -= 4;
1502  if (len >= input_len) {
1503  return ALPROTO_UNKNOWN;
1504  }
1505 
1506  input_len -= len;
1507  input += 4 + len;
1508  }
1509 
1510  return ALPROTO_UNKNOWN;
1511 }
