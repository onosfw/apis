75 void *ParsePcapConfig(const char *iface)
76 {
77  char *threadsstr = NULL;
78  ConfNode *if_root;
79  ConfNode *if_default = NULL;
80  ConfNode *pcap_node;
81  PcapIfaceConfig *aconf = SCMalloc(sizeof(*aconf));
82  char *tmpbpf;
83  char *tmpctype;
84  intmax_t value;
85  int promisc = 0;
86  intmax_t snaplen = 0;
87 
88  if (unlikely(aconf == NULL)) {
89  return NULL;
90  }
91 
92  if (iface == NULL) {
93  SCFree(aconf);
94  return NULL;
95  }
96 
97  memset(aconf, 0x00, sizeof(*aconf));
98  strlcpy(aconf->iface, iface, sizeof(aconf->iface));
99 
100  aconf->buffer_size = 0;
101  /* If set command line option has precedence over config */
102  if ((ConfGetInt("pcap.buffer-size", &value)) == 1) {
103  SCLogInfo("Pcap will use %d buffer size", (int)value);
104  aconf->buffer_size = value;
105  }
106 
107  aconf->checksum_mode = CHECKSUM_VALIDATION_AUTO;
108  aconf->bpf_filter = NULL;
109  if ((ConfGet("bpf-filter", &tmpbpf)) == 1) {
110  aconf->bpf_filter = tmpbpf;
111  }
112 
113  SC_ATOMIC_INIT(aconf->ref);
114  aconf->DerefFunc = PcapDerefConfig;
115  aconf->threads = 1;
116 
117  /* Find initial node */
118  pcap_node = ConfGetNode("pcap");
119  if (pcap_node == NULL) {
120  SCLogInfo("Unable to find pcap config using default value");
121  return aconf;
122  }
123 
124  if_root = ConfNodeLookupKeyValue(pcap_node, "interface", iface);
125 
126  if_default = ConfNodeLookupKeyValue(pcap_node, "interface", "default");
127 
128  if (if_root == NULL && if_default == NULL) {
129  SCLogInfo("Unable to find pcap config for "
130  "interface %s, using default value",
131  iface);
132  return aconf;
133  }
134 
135  /* If there is no setting for current interface use default one as main iface */
136  if (if_root == NULL) {
137  if_root = if_default;
138  if_default = NULL;
139  }
140 
141  if (ConfGetChildValueWithDefault(if_root, if_default, "threads", &threadsstr) != 1) {
142  aconf->threads = 1;
143  } else {
144  if (threadsstr != NULL) {
145  aconf->threads = (uint8_t)atoi(threadsstr);
146  }
147  }
148  if (aconf->threads == 0) {
149  aconf->threads = 1;
150  }
151  (void) SC_ATOMIC_ADD(aconf->ref, aconf->threads);
152 
153  if (aconf->buffer_size == 0) {
154  char *s_limit = NULL;
155  int ret;
156  ret = ConfGetChildValueWithDefault(if_root, if_default, "buffer-size", &s_limit);
157  if (ret == 1 && s_limit) {
158  uint64_t bsize = 0;
159 
160  if (ParseSizeStringU64(s_limit, &bsize) < 0) {
161  SCLogError(SC_ERR_INVALID_ARGUMENT,
162  "Failed to parse pcap buffer size: %s",
163  s_limit);
164  } else {
165  /* the string 2gb returns 2147483648 which is 1 to high
166  * for a int. */
167  if (bsize == (uint64_t)((uint64_t)INT_MAX + (uint64_t)1))
168  bsize = (uint64_t)INT_MAX;
169 
170  if (bsize > INT_MAX) {
171  SCLogError(SC_ERR_INVALID_ARGUMENT,
172  "Failed to set pcap buffer size: 2gb max. %"PRIu64" > %d", bsize, INT_MAX);
173  } else {
174  aconf->buffer_size = (int)bsize;
175  }
176  }
177  }
178  }
179 
180  if (aconf->bpf_filter == NULL) {
181  /* set bpf filter if we have one */
182  if (ConfGetChildValueWithDefault(if_root, if_default, "bpf-filter", &tmpbpf) != 1) {
183  SCLogDebug("could not get bpf or none specified");
184  } else {
185  aconf->bpf_filter = tmpbpf;
186  }
187  } else {
188  SCLogInfo("BPF filter set from command line or via old 'bpf-filter' option.");
189  }
190 
191  if (ConfGetChildValueWithDefault(if_root, if_default, "checksum-checks", &tmpctype) == 1) {
192  if (strcmp(tmpctype, "auto") == 0) {
193  aconf->checksum_mode = CHECKSUM_VALIDATION_AUTO;
194  } else if (strcmp(tmpctype, "yes") == 0) {
195  aconf->checksum_mode = CHECKSUM_VALIDATION_ENABLE;
196  } else if (strcmp(tmpctype, "no") == 0) {
197  aconf->checksum_mode = CHECKSUM_VALIDATION_DISABLE;
198  } else {
199  SCLogError(SC_ERR_INVALID_ARGUMENT, "Invalid value for checksum-checks for %s", aconf->iface);
200  }
201  }
202 
203  aconf->promisc = LIBPCAP_PROMISC;
204  if (ConfGetChildValueBoolWithDefault(if_root, if_default, "promisc", &promisc) != 1) {
205  SCLogDebug("could not get promisc or none specified");
206  } else {
207  aconf->promisc = promisc;
208  }
209 
210  aconf->snaplen = 0;
211  if (ConfGetChildValueIntWithDefault(if_root, if_default, "snaplen", &snaplen) != 1) {
212  SCLogDebug("could not get snaplen or none specified");
213  } else {
214  aconf->snaplen = snaplen;
215  }
216 
217 
218  return aconf;
219 }

257 /**
258  * \brief RunModIdsPcapAutoFp set up the following thread packet handlers:
259  * - Receive thread (from pcap device)
260  * - Decode thread
261  * - Stream thread
262  * - Detect: If we have only 1 cpu, it will setup one Detect thread
263  * If we have more than one, it will setup num_cpus - 1
264  * starting from the second cpu available.
265  * - Outputs thread
266  * By default the threads will use the first cpu available
267  * except the Detection threads if we have more than one cpu.
268  *
269  * \retval 0 If all goes well. (If any problem is detected the engine will
270  * exit()).
271  */
272 int RunModeIdsPcapAutoFp(void)
273 {
274  int ret;
275  char *live_dev = NULL;
276 
277  SCEnter();
278  RunModeInitialize();
279  TimeModeSetLive();
280 
281  (void) ConfGet("pcap.single-pcap-dev", &live_dev);
282 
283  ret = RunModeSetLiveCaptureAutoFp(ParsePcapConfig,
284  PcapConfigGeThreadsCount,
285  "ReceivePcap",
286  "DecodePcap", "RxPcap",
287  live_dev);
288  if (ret != 0) {
289  SCLogError(SC_ERR_RUNMODE, "Runmode start failed");
290  exit(EXIT_FAILURE);
291  }
292 
293  SCLogInfo("RunModeIdsPcapAutoFp initialised");
294 
295  SCReturnInt(0);
296 }
