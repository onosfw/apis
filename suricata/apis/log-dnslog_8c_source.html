75 static void LogQuery(LogDnsLogThread *aft, char *timebuf, char *srcip, char *dstip, Port sp, Port dp, DNSTransaction *tx, DNSQueryEntry *entry)
76 {
77  LogDnsFileCtx *hlog = aft->dnslog_ctx;
78 
79  SCLogDebug("got a DNS request and now logging !!");
80 
81  /* reset */
82  MemBufferReset(aft->buffer);
83 
84  /* time & tx */
85  MemBufferWriteString(aft->buffer,
86  "%s [**] Query TX %04x [**] ", timebuf, tx->tx_id);
87 
88  /* query */
89  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset, aft->buffer->size,
90  (uint8_t *)((uint8_t *)entry + sizeof(DNSQueryEntry)),
91  entry->len);
92 
93  char record[16] = "";
94  DNSCreateTypeString(entry->type, record, sizeof(record));
95  MemBufferWriteString(aft->buffer,
96  " [**] %s [**] %s:%" PRIu16 " -> %s:%" PRIu16 "\n",
97  record, srcip, sp, dstip, dp);
98 
99  SCMutexLock(&hlog->file_ctx->fp_mutex);
100  hlog->file_ctx->Write((const char *)MEMBUFFER_BUFFER(aft->buffer),
101  MEMBUFFER_OFFSET(aft->buffer), hlog->file_ctx);
102  SCMutexUnlock(&hlog->file_ctx->fp_mutex);
103 }

105 static void LogAnswer(LogDnsLogThread *aft, char *timebuf, char *srcip, char *dstip, Port sp, Port dp, DNSTransaction *tx, DNSAnswerEntry *entry)
106 {
107  LogDnsFileCtx *hlog = aft->dnslog_ctx;
108 
109  SCLogDebug("got a DNS response and now logging !!");
110 
111  /* reset */
112  MemBufferReset(aft->buffer);
113  /* time & tx*/
114  MemBufferWriteString(aft->buffer,
115  "%s [**] Response TX %04x [**] ", timebuf, tx->tx_id);
116 
117  if (entry == NULL) {
118  if (tx->rcode) {
119  char rcode[16] = "";
120  DNSCreateRcodeString(tx->rcode, rcode, sizeof(rcode));
121  MemBufferWriteString(aft->buffer, "%s", rcode);
122  } else if (tx->recursion_desired) {
123  MemBufferWriteString(aft->buffer, "Recursion Desired");
124  }
125  } else {
126  /* query */
127  if (entry->fqdn_len > 0) {
128  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset, aft->buffer->size,
129  (uint8_t *)((uint8_t *)entry + sizeof(DNSAnswerEntry)),
130  entry->fqdn_len);
131  } else {
132  MemBufferWriteString(aft->buffer, "<no data>");
133  }
134 
135  char record[16] = "";
136  DNSCreateTypeString(entry->type, record, sizeof(record));
137  MemBufferWriteString(aft->buffer,
138  " [**] %s [**] TTL %u [**] ", record, entry->ttl);
139 
140  uint8_t *ptr = (uint8_t *)((uint8_t *)entry + sizeof(DNSAnswerEntry) + entry->fqdn_len);
141  if (entry->type == DNS_RECORD_TYPE_A) {
142  char a[16] = "";
143  PrintInet(AF_INET, (const void *)ptr, a, sizeof(a));
144  MemBufferWriteString(aft->buffer, "%s", a);
145  } else if (entry->type == DNS_RECORD_TYPE_AAAA) {
146  char a[46];
147  PrintInet(AF_INET6, (const void *)ptr, a, sizeof(a));
148  MemBufferWriteString(aft->buffer, "%s", a);
149  } else if (entry->data_len == 0) {
150  MemBufferWriteString(aft->buffer, "<no data>");
151  } else {
152  PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset,
153  aft->buffer->size, ptr, entry->data_len);
154  }
155  }
156 
157  /* ip/tcp header info */
158  MemBufferWriteString(aft->buffer,
159  " [**] %s:%" PRIu16 " -> %s:%" PRIu16 "\n",
160  srcip, sp, dstip, dp);
161 
162  SCMutexLock(&hlog->file_ctx->fp_mutex);
163  hlog->file_ctx->Write((const char *)MEMBUFFER_BUFFER(aft->buffer),
164  MEMBUFFER_OFFSET(aft->buffer), hlog->file_ctx);
165  SCMutexUnlock(&hlog->file_ctx->fp_mutex);
166 }

168 static int LogDnsLogger(ThreadVars *tv, void *data, const Packet *p, Flow *f,
169  void *state, void *tx, uint64_t tx_id)
170 {
171  LogDnsLogThread *aft = (LogDnsLogThread *)data;
172  DNSTransaction *dns_tx = (DNSTransaction *)tx;
173  SCLogDebug("pcap_cnt %ju", p->pcap_cnt);
174  char timebuf[64];
175  CreateTimeString(&p->ts, timebuf, sizeof(timebuf));
176 
177  int ipproto = 0;
178  if (PKT_IS_IPV4(p))
179  ipproto = AF_INET;
180  else if (PKT_IS_IPV6(p))
181  ipproto = AF_INET6;
182 
183  char srcip[46], dstip[46];
184  Port sp, dp;
185  if ((PKT_IS_TOCLIENT(p))) {
186  switch (ipproto) {
187  case AF_INET:
188  PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p), srcip, sizeof(srcip));
189  PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p), dstip, sizeof(dstip));
190  break;
191  case AF_INET6:
192  PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p), srcip, sizeof(srcip));
193  PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p), dstip, sizeof(dstip));
194  break;
195  default:
196  goto end;
197  }
198  sp = p->sp;
199  dp = p->dp;
200  } else {
201  switch (ipproto) {
202  case AF_INET:
203  PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p), srcip, sizeof(srcip));
204  PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p), dstip, sizeof(dstip));
205  break;
206  case AF_INET6:
207  PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p), srcip, sizeof(srcip));
208  PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p), dstip, sizeof(dstip));
209  break;
210  default:
211  goto end;
212  }
213  sp = p->dp;
214  dp = p->sp;
215  }
216 
217  DNSQueryEntry *query = NULL;
218  TAILQ_FOREACH(query, &dns_tx->query_list, next) {
219  LogQuery(aft, timebuf, dstip, srcip, dp, sp, dns_tx, query);
220  }
221 
222  if (dns_tx->rcode)
223  LogAnswer(aft, timebuf, srcip, dstip, sp, dp, dns_tx, NULL);
224  if (dns_tx->recursion_desired)
225  LogAnswer(aft, timebuf, srcip, dstip, sp, dp, dns_tx, NULL);
226 
227  DNSAnswerEntry *entry = NULL;
228  TAILQ_FOREACH(entry, &dns_tx->answer_list, next) {
229  LogAnswer(aft, timebuf, srcip, dstip, sp, dp, dns_tx, entry);
230  }
231 
232  entry = NULL;
233  TAILQ_FOREACH(entry, &dns_tx->authority_list, next) {
234  LogAnswer(aft, timebuf, srcip, dstip, sp, dp, dns_tx, entry);
235  }
236 
237  aft->dns_cnt++;
238 end:
239  return 0;
240 }

242 static TmEcode LogDnsLogThreadInit(ThreadVars *t, void *initdata, void **data)
243 {
244  LogDnsLogThread *aft = SCMalloc(sizeof(LogDnsLogThread));
245  if (unlikely(aft == NULL))
246  return TM_ECODE_FAILED;
247  memset(aft, 0, sizeof(LogDnsLogThread));
248 
249  if(initdata == NULL)
250  {
251  SCLogDebug("Error getting context for DNSLog. \"initdata\" argument NULL");
252  SCFree(aft);
253  return TM_ECODE_FAILED;
254  }
255 
256  aft->buffer = MemBufferCreateNew(OUTPUT_BUFFER_SIZE);
257  if (aft->buffer == NULL) {
258  SCFree(aft);
259  return TM_ECODE_FAILED;
260  }
261 
262  /* Use the Ouptut Context (file pointer and mutex) */
263  aft->dnslog_ctx= ((OutputCtx *)initdata)->data;
264 
265  *data = (void *)aft;
266  return TM_ECODE_OK;
267 }

302 /** \brief Create a new dns log LogFileCtx.
303  * \param conf Pointer to ConfNode containing this loggers configuration.
304  * \return NULL if failure, LogFileCtx* to the file_ctx if succesful
305  * */
306 static OutputCtx *LogDnsLogInitCtx(ConfNode *conf)
307 {
308  LogFileCtx* file_ctx = LogFileNewCtx();
309 
310  if(file_ctx == NULL) {
311  SCLogError(SC_ERR_DNS_LOG_GENERIC, "couldn't create new file_ctx");
312  return NULL;
313  }
314 
315  if (SCConfLogOpenGeneric(conf, file_ctx, DEFAULT_LOG_FILENAME, 1) < 0) {
316  LogFileFreeCtx(file_ctx);
317  return NULL;
318  }
319 
320  LogDnsFileCtx *dnslog_ctx = SCMalloc(sizeof(LogDnsFileCtx));
321  if (unlikely(dnslog_ctx == NULL)) {
322  LogFileFreeCtx(file_ctx);
323  return NULL;
324  }
325  memset(dnslog_ctx, 0x00, sizeof(LogDnsFileCtx));
326 
327  dnslog_ctx->file_ctx = file_ctx;
328 
329  OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
330  if (unlikely(output_ctx == NULL)) {
331  LogFileFreeCtx(file_ctx);
332  SCFree(dnslog_ctx);
333  return NULL;
334  }
335 
336  output_ctx->data = dnslog_ctx;
337  output_ctx->DeInit = LogDnsLogDeInitCtx;
338 
339  SCLogDebug("DNS log output initialized");
340 
341  AppLayerParserRegisterLogger(IPPROTO_UDP, ALPROTO_DNS);
342  AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_DNS);
343 
344  return output_ctx;
345 }
