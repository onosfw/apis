32 MemBuffer *MemBufferCreateNew(uint32_t size)
33 {
34  if (size > MAX_LIMIT) {
35  SCLogWarning(SC_ERR_MEM_BUFFER_API, "Mem buffer asked to create "
36  "buffer with size greater than API limit - %d", MAX_LIMIT);
37  return NULL;
38  }
39 
40  uint32_t total_size = size + sizeof(MemBuffer);
41 
42  MemBuffer *buffer = SCMalloc(total_size);
43  if (unlikely(buffer == NULL)) {
44  return NULL;
45  }
46  memset(buffer, 0, total_size);
47 
48  buffer->size = size;
49  buffer->buffer = (uint8_t *)buffer + sizeof(MemBuffer);
50 
51  return buffer;
52 }

54 /** \brief expand membuffer by size of 'expand_by'
55  *
56  * If expansion failed, buffer will still be valid.
57  *
58  * \retval result 0 ok, -1 expansion failed
59  */
60 int MemBufferExpand(MemBuffer **buffer, uint32_t expand_by) {
61  if (((*buffer)->size + expand_by) > MAX_LIMIT) {
62  SCLogWarning(SC_ERR_MEM_BUFFER_API, "Mem buffer asked to create "
63  "buffer with size greater than API limit - %d", MAX_LIMIT);
64  return -1;
65  }
66 
67  uint32_t total_size = (*buffer)->size + sizeof(MemBuffer) + expand_by;
68 
69  MemBuffer *tbuffer = SCRealloc(*buffer, total_size);
70  if (unlikely(tbuffer == NULL)) {
71  return -1;
72  }
73 
74  *buffer = tbuffer;
75  (*buffer)->size += expand_by;
76  (*buffer)->buffer = (uint8_t *)tbuffer + sizeof(MemBuffer);
77 
78  SCLogDebug("expanded buffer by %u, size is now %u", expand_by, (*buffer)->size);
79  return 0;
80 }
