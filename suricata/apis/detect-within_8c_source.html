63 /** \brief Setup within pattern (content/uricontent) modifier.
64  *
65  * \todo apply to uricontent
66  *
67  * \retval 0 ok
68  * \retval -1 error, sig needs to be invalidated
69  */
70 static int DetectWithinSetup(DetectEngineCtx *de_ctx, Signature *s, char *withinstr)
71 {
72  char *str = withinstr;
73  char dubbed = 0;
74  SigMatch *pm = NULL;
75  int ret = -1;
76 
77  /* strip "'s */
78  if (withinstr[0] == '\"' && withinstr[strlen(withinstr)-1] == '\"') {
79  str = SCStrdup(withinstr+1);
80  if (unlikely(str == NULL))
81  goto end;
82  str[strlen(withinstr) - 2] = '\0';
83  dubbed = 1;
84  }
85 
86  /* retrive the sm to apply the depth against */
87  if (s->list != DETECT_SM_LIST_NOTSET) {
88  pm = SigMatchGetLastSMFromLists(s, 2, DETECT_CONTENT, s->sm_lists_tail[s->list]);
89  } else {
90  pm = SigMatchGetLastSMFromLists(s, 28,
91  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_PMATCH],
92  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_UMATCH],
93  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HRUDMATCH],
94  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HCBDMATCH],
95  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_FILEDATA],
96  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HHDMATCH],
97  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HRHDMATCH],
98  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HMDMATCH],
99  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HCDMATCH],
100  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HSCDMATCH],
101  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HSMDMATCH],
102  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HUADMATCH],
103  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HHHDMATCH],
104  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HRHHDMATCH]);
105  }
106  if (pm == NULL) {
107  SCLogError(SC_ERR_OFFSET_MISSING_CONTENT, "within needs "
108  "preceding content, uricontent option, http_client_body, "
109  "http_server_body, http_header option, http_raw_header option, "
110  "http_method option, http_cookie, http_raw_uri, "
111  "http_stat_msg, http_stat_code, http_user_agent or "
112  "file_data/dce_stub_data sticky buffer option");
113  goto end;
114  }
115 
116 
117  /* verify other conditions */
118  DetectContentData *cd = (DetectContentData *)pm->ctx;
119  if (cd->flags & DETECT_CONTENT_WITHIN) {
120  SCLogError(SC_ERR_INVALID_SIGNATURE, "can't use multiple withins for the same content.");
121  goto end;
122  }
123  if ((cd->flags & DETECT_CONTENT_DEPTH) || (cd->flags & DETECT_CONTENT_OFFSET)) {
124  SCLogError(SC_ERR_INVALID_SIGNATURE, "can't use a relative "
125  "keyword like within/distance with a absolute "
126  "relative keyword like depth/offset for the same "
127  "content." );
128  goto end;
129  }
130  if (cd->flags & DETECT_CONTENT_NEGATED && cd->flags & DETECT_CONTENT_FAST_PATTERN) {
131  SCLogError(SC_ERR_INVALID_SIGNATURE, "can't have a relative "
132  "negated keyword set along with a fast_pattern");
133  goto end;
134  }
135  if (cd->flags & DETECT_CONTENT_FAST_PATTERN_ONLY) {
136  SCLogError(SC_ERR_INVALID_SIGNATURE, "can't have a relative "
137  "keyword set along with a fast_pattern:only;");
138  goto end;
139  }
140  if (str[0] != '-' && isalpha((unsigned char)str[0])) {
141  SigMatch *bed_sm = DetectByteExtractRetrieveSMVar(str, s);
142  if (bed_sm == NULL) {
143  SCLogError(SC_ERR_INVALID_SIGNATURE, "unknown byte_extract var "
144  "seen in within - %s\n", str);
145  goto end;
146  }
147  cd->within = ((DetectByteExtractData *)bed_sm->ctx)->local_id;
148  cd->flags |= DETECT_CONTENT_WITHIN_BE;
149  } else {
150  cd->within = strtol(str, NULL, 10);
151  if (cd->within < (int32_t)cd->content_len) {
152  SCLogError(SC_ERR_WITHIN_INVALID, "within argument \"%"PRIi32"\" is "
153  "less than the content length \"%"PRIu32"\" which is invalid, since "
154  "this will never match. Invalidating signature", cd->within,
155  cd->content_len);
156  goto end;
157  }
158  }
159  cd->flags |= DETECT_CONTENT_WITHIN;
160 
161  /* these are the only ones against which we set a flag. We have other
162  * relative keywords like byttest, isdataat, bytejump, but we don't
163  * set a flag against them */
164  SigMatch *prev_pm = SigMatchGetLastSMFromLists(s, 4,
165  DETECT_CONTENT, pm->prev,
166  DETECT_PCRE, pm->prev);
167  if (prev_pm == NULL) {
168  ret = 0;
169  goto end;
170  }
171  if (prev_pm->type == DETECT_CONTENT) {
172  DetectContentData *cd = (DetectContentData *)prev_pm->ctx;
173  if (cd->flags & DETECT_CONTENT_FAST_PATTERN_ONLY) {
174  SCLogError(SC_ERR_INVALID_SIGNATURE, "previous keyword "
175  "has a fast_pattern:only; set. Can't "
176  "have relative keywords around a fast_pattern "
177  "only content");
178  goto end;
179  }
180  cd->flags |= DETECT_CONTENT_RELATIVE_NEXT;
181  } else if (prev_pm->type == DETECT_PCRE) {
182  DetectPcreData *pd = (DetectPcreData *)prev_pm->ctx;
183  pd->flags |= DETECT_PCRE_RELATIVE_NEXT;
184  }
185 
186  ret = 0;
187  end:
188  if (dubbed)
189  SCFree(str);
190  return ret;
191 }
