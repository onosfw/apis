43 /** \param loader -1 for auto select
44  * \retval loader_id or negative in case of error */
45 int DetectLoaderQueueTask(int loader_id, LoaderFunc Func, void *func_ctx)
46 {
47  if (loader_id == -1) {
48  loader_id = cur_loader;
49  cur_loader++;
50  if (cur_loader >= num_loaders)
51  cur_loader = 0;
52  }
53  if (loader_id >= num_loaders || loader_id < 0) {
54  return -ERANGE;
55  }
56 
57  DetectLoaderControl *loader = &loaders[loader_id];
58 
59  DetectLoaderTask *t = SCCalloc(1, sizeof(*t));
60  if (t == NULL)
61  return -ENOMEM;
62 
63  t->Func = Func;
64  t->ctx = func_ctx;
65 
66  SCMutexLock(&loader->m);
67  TAILQ_INSERT_TAIL(&loader->task_list, t, next);
68  SCMutexUnlock(&loader->m);
69 
70  TmThreadWakeupDetectLoaderThreads();
71 
72  SCLogDebug("%d %p %p", loader_id, Func, func_ctx);
73  return loader_id;
74 }

76 /** \brief wait for loader tasks to complete
77  * \retval result 0 for ok, -1 for errors */
78 int DetectLoadersSync(void)
79 {
80  SCLogDebug("waiting");
81  int errors = 0;
82  int i;
83  for (i = 0; i < num_loaders; i++) {
84  int done = 0;
85  DetectLoaderControl *loader = &loaders[i];
86  while (!done) {
87  SCMutexLock(&loader->m);
88  if (TAILQ_EMPTY(&loader->task_list)) {
89  done = 1;
90  }
91  SCMutexUnlock(&loader->m);
92  }
93  SCMutexLock(&loader->m);
94  if (loader->result != 0) {
95  errors++;
96  loader->result = 0;
97  }
98  SCMutexUnlock(&loader->m);
99 
100  }
101  if (errors) {
102  SCLogError(SC_ERR_INITIALIZATION, "%d loaders reported errors", errors);
103  return -1;
104  }
105  SCLogDebug("done");
106  return 0;
107 }

116 void DetectLoadersInit(void)
117 {
118  intmax_t setting = NLOADERS;
119  (void)ConfGetInt("multi-detect.loaders", &setting);
120 
121  if (setting < 1 || setting > 1024) {
122  SCLogError(SC_ERR_INVALID_ARGUMENTS,
123  "invalid multi-detect.loaders setting %"PRIdMAX, setting);
124  exit(EXIT_FAILURE);
125  }
126  num_loaders = (int32_t)setting;
127 
128  SCLogInfo("using %d detect loader threads", num_loaders);
129 
130  BUG_ON(loaders != NULL);
131  loaders = SCCalloc(num_loaders, sizeof(DetectLoaderControl));
132  BUG_ON(loaders == NULL);
133 
134  int i;
135  for (i = 0; i < num_loaders; i++) {
136  DetectLoaderInit(&loaders[i]);
137  }
138 }

216 static TmEcode DetectLoader(ThreadVars *th_v, void *thread_data)
217 {
218  /* block usr2. usr2 to be handled by the main thread only */
219  UtilSignalBlock(SIGUSR2);
220 
221  DetectLoaderThreadData *ftd = (DetectLoaderThreadData *)thread_data;
222  BUG_ON(ftd == NULL);
223 
224  SCLogDebug("loader thread started");
225  while (1)
226  {
227  if (TmThreadsCheckFlag(th_v, THV_PAUSE)) {
228  TmThreadsSetFlag(th_v, THV_PAUSED);
229  TmThreadTestThreadUnPaused(th_v);
230  TmThreadsUnsetFlag(th_v, THV_PAUSED);
231  }
232 
233  /* see if we have tasks */
234 
235  DetectLoaderControl *loader = &loaders[ftd->instance];
236  SCMutexLock(&loader->m);
237 
238  DetectLoaderTask *task = NULL, *tmptask = NULL;
239  TAILQ_FOREACH_SAFE(task, &loader->task_list, next, tmptask) {
240  int r = task->Func(task->ctx, ftd->instance);
241  loader->result |= r;
242  TAILQ_REMOVE(&loader->task_list, task, next);
243  SCFree(task);
244  }
245 
246  SCMutexUnlock(&loader->m);
247 
248  if (TmThreadsCheckFlag(th_v, THV_KILL)) {
249  break;
250  }
251 
252  /* just wait until someone wakes us up */
253  SCCtrlMutexLock(th_v->ctrl_mutex);
254  SCCtrlCondWait(th_v->ctrl_cond, th_v->ctrl_mutex);
255  SCCtrlMutexUnlock(th_v->ctrl_mutex);
256 
257  SCLogDebug("woke up...");
258  }
259 
260  return TM_ECODE_OK;
261 }

264 void DetectLoaderThreadSpawn()
265 {

268  ThreadVars *tv_loader = NULL;

270  char name[32] = "";

275  BUG_ON(tv_loader == NULL);

282  printf("ERROR: TmThreadSpawn failed\n");

284  }
285  }
