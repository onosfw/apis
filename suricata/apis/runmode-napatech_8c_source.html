73 int NapatechRegisterDeviceStreams()
74 {
75  NtInfoStream_t info_stream;
76  NtInfo_t info;
77  char error_buf[100];
78  int status;
79  int i;
80  char live_dev_buf[9];
81  int use_all_streams;
82  ConfNode *ntstreams;
83  ConfNode *stream_id;
84 
85  if (ConfGetBool("napatech.use-all-streams", &use_all_streams) == 0)
86  {
87  SCLogError(SC_ERR_RUNMODE, "Failed retrieving napatech.use-all-streams from Conf");
88  exit(EXIT_FAILURE);
89  }
90 
91  if (use_all_streams)
92  {
93  SCLogInfo("Using All Napatech Streams");
94  // When using the default streams we need to query the service for a list of all configured
95  if ((status = NT_InfoOpen(&info_stream, "SuricataStreamInfo")) != NT_SUCCESS)
96  {
97  NT_ExplainError(status, error_buf, sizeof(error_buf) -1);
98  SCLogError(SC_ERR_NAPATECH_STREAMS_REGISTER_FAILED, "NT_InfoOpen failed: %s", error_buf);
99  return -1;
100  }
101 
102  info.cmd = NT_INFO_CMD_READ_STREAM;
103  if ((status = NT_InfoRead(info_stream, &info)) != NT_SUCCESS)
104  {
105  NT_ExplainError(status, error_buf, sizeof(error_buf) -1);
106  SCLogError(SC_ERR_NAPATECH_STREAMS_REGISTER_FAILED, "NT_InfoRead failed: %s", error_buf);
107  return -1;
108  }
109 
110  num_configured_streams = info.u.stream.data.count;
111  for (i = 0; i < num_configured_streams; i++)
112  {
113  // The Stream IDs do not have to be sequential
114  snprintf(live_dev_buf, sizeof(live_dev_buf), "nt%d", info.u.stream.data.streamIDList[i]);
115  LiveRegisterDevice(live_dev_buf);
116  }
117 
118  if ((status = NT_InfoClose(info_stream)) != NT_SUCCESS)
119  {
120  NT_ExplainError(status, error_buf, sizeof(error_buf) -1);
121  SCLogError(SC_ERR_NAPATECH_STREAMS_REGISTER_FAILED, "NT_InfoClose failed: %s", error_buf);
122  return -1;
123  }
124  }
125  else
126  {
127  SCLogInfo("Using Selected Napatech Streams");
128  // When not using the default streams we need to parse the array of streams from the conf
129  if ((ntstreams = ConfGetNode("napatech.streams")) == NULL)
130  {
131  SCLogError(SC_ERR_RUNMODE, "Failed retrieving napatech.streams from Conf");
132  exit(EXIT_FAILURE);
133  }
134 
135  // Loop through all stream numbers in the array and register the devices
136  TAILQ_FOREACH(stream_id, &ntstreams->head, next)
137  {
138  if (stream_id == NULL)
139  {
140  SCLogError(SC_ERR_NAPATECH_STREAMS_REGISTER_FAILED, "Couldn't Parse Stream Configuration");
141  exit(EXIT_FAILURE);
142  }
143  num_configured_streams++;
144 
145  snprintf(live_dev_buf, sizeof(live_dev_buf), "nt%d", atoi(stream_id->val));
146  LiveRegisterDevice(live_dev_buf);
147  }
148  }
149  return 0;
150 }

152 void *NapatechConfigParser(const char *device)
153 {
154  // Expect device to be of the form nt%d where %d is the stream id to use
155  int dev_len = strlen(device);
156  struct NapatechStreamDevConf *conf = SCMalloc(sizeof(struct NapatechStreamDevConf));
157  if (unlikely(conf == NULL))
158  return NULL;
159  if (dev_len < 3 || dev_len > 5)
160  {
161  SCLogError(SC_ERR_NAPATECH_PARSE_CONFIG, "Could not parse config for device: %s - invalid length", device);
162  return NULL;
163  }
164 
165  // device+5 is a pointer to the beginning of the stream id after the constant nt portion
166  conf->stream_id = atoi(device+2);
167 
168  // Set the host buffer allowance for this stream
169  // Right now we just look at the global default - there is no per-stream hba configuration
170  if (ConfGetInt("napatech.hba", &conf->hba) == 0)
171  conf->hba = -1;
172 
173  return (void *) conf;
174 }

182 static int NapatechInit(int runmode)
183 {
184  int ret;
185  char errbuf[100];
186 
187  RunModeInitialize();
188  TimeModeSetLive();
189 
190  /* Initialize the API and check version compatibility */
191  if ((ret = NT_Init(NTAPI_VERSION)) != NT_SUCCESS) {
192  NT_ExplainError(ret, errbuf, sizeof(errbuf));
193  SCLogError(SC_ERR_NAPATECH_INIT_FAILED ,"NT_Init failed. Code 0x%X = %s", ret, errbuf);
194  exit(EXIT_FAILURE);
195  }
196 
197  ret = NapatechRegisterDeviceStreams();
198  if (ret < 0 || num_configured_streams <= 0) {
199  SCLogError(SC_ERR_NAPATECH_STREAMS_REGISTER_FAILED, "Unable to setup up Napatech Streams");
200  exit(EXIT_FAILURE);
201  }
202 
203  switch(runmode) {
204  case NT_RUNMODE_AUTOFP:
205  ret = RunModeSetLiveCaptureAutoFp(NapatechConfigParser, NapatechGetThreadsCount,
206  "NapatechStream", "NapatechDecode",
207  "RxNT", NULL);
208  break;
209  case NT_RUNMODE_WORKERS:
210  ret = RunModeSetLiveCaptureWorkers(NapatechConfigParser, NapatechGetThreadsCount,
211  "NapatechStream", "NapatechDecode",
212  "RxNT", NULL);
213  break;
214  default:
215  ret = -1;
216  }
217 
218  if (ret != 0) {
219  SCLogError(SC_ERR_RUNMODE, "Runmode start failed");
220  exit(EXIT_FAILURE);
221  }
222  return 0;
223 }
