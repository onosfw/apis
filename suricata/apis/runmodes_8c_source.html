268 /**
269  */
270 void RunModeDispatch(int runmode, const char *custom_mode)
271 {
272  char *local_custom_mode = NULL;
273 
274  if (custom_mode == NULL) {
275  char *val = NULL;
276  if (ConfGet("runmode", &val) != 1) {
277  custom_mode = NULL;
278  } else {
279  custom_mode = val;
280  }
281  }
282 
283  if (custom_mode == NULL || strcmp(custom_mode, "auto") == 0) {
284  switch (runmode) {
285  case RUNMODE_PCAP_DEV:
286  custom_mode = RunModeIdsGetDefaultMode();
287  break;
288  case RUNMODE_PCAP_FILE:
289  custom_mode = RunModeFilePcapGetDefaultMode();
290  break;
291 #ifdef HAVE_PFRING
292  case RUNMODE_PFRING:
293  custom_mode = RunModeIdsPfringGetDefaultMode();
294  break;
295 #endif
296  case RUNMODE_NFQ:
297  custom_mode = RunModeIpsNFQGetDefaultMode();
298  break;
299  case RUNMODE_IPFW:
300  custom_mode = RunModeIpsIPFWGetDefaultMode();
301  break;
302  case RUNMODE_ERF_FILE:
303  custom_mode = RunModeErfFileGetDefaultMode();
304  break;
305  case RUNMODE_DAG:
306  custom_mode = RunModeErfDagGetDefaultMode();
307  break;
308  case RUNMODE_TILERA_MPIPE:
309  custom_mode = RunModeTileMpipeGetDefaultMode();
310  break;
311  case RUNMODE_NAPATECH:
312  custom_mode = RunModeNapatechGetDefaultMode();
313  break;
314  case RUNMODE_AFP_DEV:
315  custom_mode = RunModeAFPGetDefaultMode();
316  break;
317  case RUNMODE_NETMAP:
318  custom_mode = RunModeNetmapGetDefaultMode();
319  break;
320  case RUNMODE_UNIX_SOCKET:
321  custom_mode = RunModeUnixSocketGetDefaultMode();
322  break;
323  case RUNMODE_NFLOG:
324  custom_mode = RunModeIdsNflogGetDefaultMode();
325  break;
326  default:
327  SCLogError(SC_ERR_UNKNOWN_RUN_MODE, "Unknown runtime mode. Aborting");
328  exit(EXIT_FAILURE);
329  }
330  } else { /* if (custom_mode == NULL) */
331  /* Add compability with old 'worker' name */
332  if (!strcmp("worker", custom_mode)) {
333  SCLogWarning(SC_ERR_RUNMODE, "'worker' mode have been renamed "
334  "to 'workers', please modify your setup.");
335  local_custom_mode = SCStrdup("workers");
336  if (unlikely(local_custom_mode == NULL)) {
337  SCLogError(SC_ERR_MEM_ALLOC, "Unable to dup custom mode");
338  exit(EXIT_FAILURE);
339  }
340  custom_mode = local_custom_mode;
341  }
342  }
343 
344 #ifdef __SC_CUDA_SUPPORT__
345  if (PatternMatchDefaultMatcher() == MPM_AC_CUDA &&
346  strcasecmp(custom_mode, "autofp") != 0) {
347  SCLogError(SC_ERR_RUNMODE, "When using a cuda mpm, the only runmode we "
348  "support is autofp.");
349  exit(EXIT_FAILURE);
350  }
351 #endif
352 
353  RunMode *mode = RunModeGetCustomMode(runmode, custom_mode);
354  if (mode == NULL) {
355  SCLogError(SC_ERR_RUNMODE, "The custom type \"%s\" doesn't exist "
356  "for this runmode type \"%s\". Please use --list-runmodes to "
357  "see available custom types for this runmode",
358  custom_mode, RunModeTranslateModeToName(runmode));
359  exit(EXIT_FAILURE);
360  }
361 
362  /* Export the custom mode */
363  active_runmode = SCStrdup(custom_mode);
364  if (unlikely(active_runmode == NULL)) {
365  SCLogError(SC_ERR_MEM_ALLOC, "Unable to dup active mode");
366  exit(EXIT_FAILURE);
367  }
368 
369  mode->RunModeFunc();
370 
371  if (local_custom_mode != NULL)
372  SCFree(local_custom_mode);
373  return;
374 }

376 /**
377  * \brief Registers a new runmode.
378  *
379  * \param runmode Runmode type.
380  * \param name Custom mode for this specific runmode type. Within each
381  * runmode type, each custom name is a primary key.
382  * \param description Description for this runmode.
383  * \param RunModeFunc The function to be run for this runmode.
384  */
385 void RunModeRegisterNewRunMode(int runmode, const char *name,
386  const char *description,
387  int (*RunModeFunc)(void))
388 {
389  void *ptmp;
390  if (RunModeGetCustomMode(runmode, name) != NULL) {
391  SCLogError(SC_ERR_RUNMODE, "A runmode by this custom name has already "
392  "been registered. Please use an unique name");
393  return;
394  }
395 
396  ptmp = SCRealloc(runmodes[runmode].runmodes,
397  (runmodes[runmode].no_of_runmodes + 1) * sizeof(RunMode));
398  if (ptmp == NULL) {
399  SCFree(runmodes[runmode].runmodes);
400  runmodes[runmode].runmodes = NULL;
401  exit(EXIT_FAILURE);
402  }
403  runmodes[runmode].runmodes = ptmp;
404 
405  RunMode *mode = &runmodes[runmode].runmodes[runmodes[runmode].no_of_runmodes];
406  runmodes[runmode].no_of_runmodes++;
407 
408  mode->runmode = runmode;
409  mode->name = SCStrdup(name);
410  if (unlikely(mode->name == NULL)) {
411  SCLogError(SC_ERR_MEM_ALLOC, "Failed to allocate string");
412  exit(EXIT_FAILURE);
413  }
414  mode->description = SCStrdup(description);
415  if (unlikely(mode->description == NULL)) {
416  SCLogError(SC_ERR_MEM_ALLOC, "Failed to allocate string");
417  exit(EXIT_FAILURE);
418  }
419  mode->RunModeFunc = RunModeFunc;
420 
421  return;
422 }

520 static void InsertInRunModeOutputs(RunModeOutput *runmode_output)
521 {
522  RunModeOutput *r_output = NULL;
523  int output_priority = GetRunModeOutputPriority(runmode_output);
524 
525  TAILQ_FOREACH(r_output, &RunModeOutputs, entries) {
526  if (GetRunModeOutputPriority(r_output) < output_priority)
527  break;
528  }
529  if (r_output) {
530  TAILQ_INSERT_BEFORE(r_output, runmode_output, entries);
531  } else {
532  TAILQ_INSERT_TAIL(&RunModeOutputs, runmode_output, entries);
533  }

536 /** \brief Turn output into thread module */
537 static void SetupOutput(const char *name, OutputModule *module, OutputCtx *output_ctx)
538 {
539  /* flow logger doesn't run in the packet path */
540  if (module->FlowLogFunc) {
541  OutputRegisterFlowLogger(module->name, module->FlowLogFunc, output_ctx);
542  return;
543  }
544  /* stats logger doesn't run in the packet path */
545  if (module->StatsLogFunc) {
546  OutputRegisterStatsLogger(module->name, module->StatsLogFunc, output_ctx);
547  return;
548  }
549 
550  TmModule *tm_module = TmModuleGetByName(module->name);
551  if (tm_module == NULL) {
552  SCLogError(SC_ERR_INVALID_ARGUMENT,
553  "TmModuleGetByName for %s failed", module->name);
554  exit(EXIT_FAILURE);
555  }
556  if (strcmp(tmm_modules[TMM_ALERTDEBUGLOG].name, tm_module->name) == 0)
557  debuglog_enabled = 1;
558 
559  if (module->PacketLogFunc) {
560  SCLogDebug("%s is a packet logger", module->name);
561  OutputRegisterPacketLogger(module->name, module->PacketLogFunc,
562  module->PacketConditionFunc, output_ctx);
563 
564  /* need one instance of the packet logger module */
565  if (pkt_logger_module == NULL) {
566  pkt_logger_module = TmModuleGetByName("__packet_logger__");
567  if (pkt_logger_module == NULL) {
568  SCLogError(SC_ERR_INVALID_ARGUMENT,
569  "TmModuleGetByName for __packet_logger__ failed");
570  exit(EXIT_FAILURE);
571  }
572 
573  RunModeOutput *runmode_output = SCCalloc(1, sizeof(RunModeOutput));
574  if (unlikely(runmode_output == NULL))
575  return;
576  runmode_output->name = module->name;
577  runmode_output->tm_module = pkt_logger_module;
578  runmode_output->output_ctx = NULL;
579  InsertInRunModeOutputs(runmode_output);
580  SCLogDebug("__packet_logger__ added");
581  }
582  } else if (module->TxLogFunc) {
583  SCLogDebug("%s is a tx logger", module->name);
584  OutputRegisterTxLogger(module->name, module->alproto,
585  module->TxLogFunc, output_ctx);
586 
587  /* need one instance of the tx logger module */
588  if (tx_logger_module == NULL) {
589  tx_logger_module = TmModuleGetByName("__tx_logger__");
590  if (tx_logger_module == NULL) {
591  SCLogError(SC_ERR_INVALID_ARGUMENT,
592  "TmModuleGetByName for __tx_logger__ failed");
593  exit(EXIT_FAILURE);
594  }
595 
596  RunModeOutput *runmode_output = SCCalloc(1, sizeof(RunModeOutput));
597  if (unlikely(runmode_output == NULL))
598  return;
599  runmode_output->name = module->name;
600  runmode_output->tm_module = tx_logger_module;
601  runmode_output->output_ctx = NULL;
602  InsertInRunModeOutputs(runmode_output);
603  SCLogDebug("__tx_logger__ added");
604  }
605  } else if (module->FiledataLogFunc) {
606  SCLogDebug("%s is a filedata logger", module->name);
607  OutputRegisterFiledataLogger(module->name, module->FiledataLogFunc, output_ctx);
608 
609  /* need one instance of the tx logger module */
610  if (filedata_logger_module == NULL) {
611  filedata_logger_module = TmModuleGetByName("__filedata_logger__");
612  if (filedata_logger_module == NULL) {

614  "TmModuleGetByName for __filedata_logger__ failed");
615  exit(EXIT_FAILURE);
616  }
617 
618  RunModeOutput *runmode_output = SCCalloc(1, sizeof(RunModeOutput));
619  if (unlikely(runmode_output == NULL))
620  return;
621  runmode_output->name = module->name;
622  runmode_output->tm_module = filedata_logger_module;
623  runmode_output->output_ctx = NULL;
624  InsertInRunModeOutputs(runmode_output);
625  SCLogDebug("__filedata_logger__ added");
626  }
627  } else if (module->FileLogFunc) {
628  SCLogDebug("%s is a file logger", module->name);
629  OutputRegisterFileLogger(module->name, module->FileLogFunc, output_ctx);
630 
631  /* need one instance of the tx logger module */
632  if (file_logger_module == NULL) {

686  * Initialize the output modules.
687  */
688 void RunModeInitializeOutputs(void)
689 {
690  ConfNode *outputs = ConfGetNode("outputs");
691  if (outputs == NULL) {
692  /* No "outputs" section in the configuration. */
693  return;
694  }
695 
696  ConfNode *output, *output_config;
697  const char *enabled;
698  char tls_log_enabled = 0;
699  char tls_store_present = 0;
700 
701  TAILQ_FOREACH(output, &outputs->head, next) {
702 
703  output_config = ConfNodeLookupChild(output, output->val);
704  if (output_config == NULL) {
705  /* Shouldn't happen. */
706  FatalError(SC_ERR_INVALID_ARGUMENT,
707  "Failed to lookup configuration child node: %s", output->val);
708  }
709 
710  if (strcmp(output->val, "tls-store") == 0) {
711  tls_store_present = 1;
712  }
713 
714  enabled = ConfNodeLookupChildValue(output_config, "enabled");
715  if (enabled == NULL || !ConfValIsTrue(enabled)) {
716  continue;
717  }
718 
719  if (strncmp(output->val, "unified-", sizeof("unified-") - 1) == 0) {
720  SCLogWarning(SC_ERR_NOT_SUPPORTED,
721  "Unified1 is no longer supported,"
722  " use Unified2 instead "
723  "(see https://redmine.openinfosecfoundation.org/issues/353"
724  " for an explanation)");
725  continue;
726  } else if (strcmp(output->val, "alert-prelude") == 0) {
727 #ifndef PRELUDE
728  SCLogWarning(SC_ERR_NOT_SUPPORTED,
729  "Prelude support not compiled in. Reconfigure/"
730  "recompile with --enable-prelude to add Prelude "
731  "support.");
732  continue;
733 #endif
734  } else if (strcmp(output->val, "eve-log") == 0) {
735 #ifndef HAVE_LIBJANSSON
736  SCLogWarning(SC_ERR_NOT_SUPPORTED,
737  "Eve-log support not compiled in. Reconfigure/"
738  "recompile with libjansson and its development "
739  "files installed to add eve-log support.");
740  continue;
741 #endif
742  } else if (strcmp(output->val, "lua") == 0) {
743 #ifndef HAVE_LUA
744  SCLogWarning(SC_ERR_NOT_SUPPORTED,
745  "lua support not compiled in. Reconfigure/"
746  "recompile with lua(jit) and its development "
747  "files installed to add lua support.");
748  continue;
749 #endif
750  } else if (strcmp(output->val, "tls-log") == 0) {
751  tls_log_enabled = 1;
752  }
753 
754  OutputModule *module = OutputGetModuleByConfName(output->val);
755  if (module == NULL) {
756  FatalErrorOnInit(SC_ERR_INVALID_ARGUMENT,
757  "No output module named %s", output->val);
758  continue;
759  }
760 
761  OutputCtx *output_ctx = NULL;
762  if (module->InitFunc != NULL) {
763  output_ctx = module->InitFunc(output_config);
764  if (output_ctx == NULL) {
765  FatalErrorOnInit(SC_ERR_INVALID_ARGUMENT, "output module setup failed");
766  continue;
767  }
768  } else if (module->InitSubFunc != NULL) {
769  SCLogInfo("skipping submodule");
770  continue;
771  }
772 
773  // TODO if module == parent, find it's children
774  if (strcmp(output->val, "eve-log") == 0) {
775  ConfNode *types = ConfNodeLookupChild(output_config, "types");
776  SCLogDebug("types %p", types);
777  if (types != NULL) {
778  ConfNode *type = NULL;
779  TAILQ_FOREACH(type, &types->head, next) {
780  SCLogInfo("enabling 'eve-log' module '%s'", type->val);
781 
782  char subname[256];
783  snprintf(subname, sizeof(subname), "%s.%s", output->val, type->val);
784 
785  OutputModule *sub_module = OutputGetModuleByConfName(subname);
786  if (sub_module == NULL) {
787  FatalErrorOnInit(SC_ERR_INVALID_ARGUMENT,
788  "No output module named %s", subname);
789  continue;
790  }
791  if (sub_module->parent_name == NULL ||
792  strcmp(sub_module->parent_name,output->val) != 0) {
793  FatalError(SC_ERR_INVALID_ARGUMENT,
794  "bad parent for %s", subname);
795  }
796  if (sub_module->InitSubFunc == NULL) {
797  FatalError(SC_ERR_INVALID_ARGUMENT,
798  "bad sub-module for %s", subname);
799  }
800  ConfNode *sub_output_config = ConfNodeLookupChild(type, type->val);
801  // sub_output_config may be NULL if no config
802 
803  /* pass on parent output_ctx */
804  OutputCtx *sub_output_ctx =
805  sub_module->InitSubFunc(sub_output_config, output_ctx);
806  if (sub_output_ctx == NULL) {
807  continue;
808  }
809 
810  AddOutputToFreeList(sub_module, sub_output_ctx);
811  SetupOutput(sub_module->name, sub_module, sub_output_ctx);
812  }
813  }
814  /* add 'eve-log' to free list as it's the owner of the
815  * main output ctx from which the sub-modules share the
816  * LogFileCtx */
817  AddOutputToFreeList(module, output_ctx);
818 
819  } else if (strcmp(output->val, "lua") == 0) {
820  SCLogDebug("handle lua");
821 
822  ConfNode *scripts = ConfNodeLookupChild(output_config, "scripts");
823  BUG_ON(scripts == NULL); //TODO
824 
825  OutputModule *m;
826  TAILQ_FOREACH(m, &output_ctx->submodules, entries) {
827  SCLogDebug("m %p %s:%s", m, m->name, m->conf_name);
828 
829  ConfNode *script = NULL;
830  TAILQ_FOREACH(script, &scripts->head, next) {
831  SCLogDebug("script %s", script->val);
832  if (strcmp(script->val, m->conf_name) == 0) {
833  break;
834  }
835  }
836  BUG_ON(script == NULL);
837 
838  /* pass on parent output_ctx */
839  OutputCtx *sub_output_ctx =
840  m->InitSubFunc(script, output_ctx);
841  if (sub_output_ctx == NULL) {
842  SCLogInfo("sub_output_ctx NULL, skipping");
843  continue;
844  }
845 
846  SetupOutput(m->name, m, sub_output_ctx);
847  }
848 
849  } else {
850  AddOutputToFreeList(module, output_ctx);
851  SetupOutput(module->name, module, output_ctx);
852  }
853  }
854 
855  /* Backward compatibility code */
856  if (!tls_store_present && tls_log_enabled) {
857  /* old YAML with no "tls-store" in outputs. "tls-log" value needs
858  * to be started using 'tls-log' config as own config */
859  SCLogWarning(SC_ERR_CONF_YAML_ERROR,
860  "Please use 'tls-store' in YAML to configure TLS storage");
861 
862  TAILQ_FOREACH(output, &outputs->head, next) {
863  output_config = ConfNodeLookupChild(output, output->val);
864 
865  if (strcmp(output->val, "tls-log") == 0) {
866 
867  OutputModule *module = OutputGetModuleByConfName("tls-store");
868  if (module == NULL) {
869  SCLogWarning(SC_ERR_INVALID_ARGUMENT,
870  "No output module named %s, ignoring", "tls-store");
871  continue;
872  }
873 
874  OutputCtx *output_ctx = NULL;
875  if (module->InitFunc != NULL) {
876  output_ctx = module->InitFunc(output_config);
877  if (output_ctx == NULL) {
878  continue;
879  }
880  }
881 
882  AddOutputToFreeList(module, output_ctx);
883  SetupOutput(module->name, module, output_ctx);
884  }
885  }
886  }
887 
888 }
