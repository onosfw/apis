64 static int DNSTCPRequestParseProbe(uint8_t *input, uint32_t input_len)
65 {
66 #ifdef DEBUG
67  BUG_ON(input_len < sizeof(DNSTcpHeader));
68 #endif
69  SCLogDebug("starting %u", input_len);
70 
71  DNSTcpHeader *dns_tcp_header = (DNSTcpHeader *)input;
72  if (ntohs(dns_tcp_header->len) < sizeof(DNSHeader)) {
73  goto bad_data;
74  }
75  if (ntohs(dns_tcp_header->len) >= input_len) {
76  goto insufficient_data;
77  }
78 
79  input += 2;
80  input_len -= 2;
81  DNSHeader *dns_header = (DNSHeader *)input;
82 
83  uint16_t q;
84  const uint8_t *data = input + sizeof(DNSHeader);
85 
86  for (q = 0; q < ntohs(dns_header->questions); q++) {
87  uint16_t fqdn_offset = 0;
88 
89  if (input + input_len < data + 1) {
90  SCLogDebug("input buffer too small for len field");
91  goto insufficient_data;
92  }
93  SCLogDebug("query length %u", *data);
94 
95  while (*data != 0) {
96  if (*data > 63) {
97  /** \todo set event?*/
98  goto bad_data;
99  }
100  uint8_t length = *data;
101 
102  data++;
103 
104  if (length > 0) {
105  if (input + input_len < data + length) {
106  SCLogDebug("input buffer too small for domain of len %u", length);
107  goto insufficient_data;
108  }
109  //PrintRawDataFp(stdout, data, qry->length);
110 
111  if ((fqdn_offset + length + 1) < DNS_MAX_SIZE) {
112  fqdn_offset += length;
113  } else {
114  /** \todo set event? */
115  goto bad_data;
116  }
117  }
118 
119  data += length;
120 
121  if (input + input_len < data + 1) {
122  SCLogDebug("input buffer too small for new len");
123  goto insufficient_data;
124  }
125 
126  SCLogDebug("qry length %u", *data);
127  }
128  if (fqdn_offset) {
129  fqdn_offset--;
130  }
131 
132  data++;
133  if (input + input_len < data + sizeof(DNSQueryTrailer)) {
134  SCLogDebug("input buffer too small for DNSQueryTrailer");
135  goto insufficient_data;
136  }
137 #ifdef DEBUG
138  DNSQueryTrailer *trailer = (DNSQueryTrailer *)data;
139  SCLogDebug("trailer type %04x class %04x", ntohs(trailer->type), ntohs(trailer->class));
140 #endif
141  data += sizeof(DNSQueryTrailer);
142  }
143 
144  SCReturnInt(1);
145 insufficient_data:
146  SCReturnInt(0);
147 bad_data:
148  SCReturnInt(-1);
149 }

151 static int BufferData(DNSState *dns_state, uint8_t *data, uint16_t len)
152 {
153  if (dns_state->buffer == NULL) {
154  if (DNSCheckMemcap(0xffff, dns_state) < 0)
155  return -1;
156 
157  /** \todo be smarter about this, like use a pool or several pools for
158  * chunks of various sizes */
159  dns_state->buffer = SCMalloc(0xffff);
160  if (dns_state->buffer == NULL) {
161  return -1;
162  }
163  DNSIncrMemcap(0xffff, dns_state);
164  }
165 
166  if ((uint32_t)len + (uint32_t)dns_state->offset > (uint32_t)dns_state->record_len) {
167  SCLogDebug("oh my, we have more data than the max record size. What do we do. WHAT DO WE DOOOOO!");
168 #ifdef DEBUG
169  BUG_ON(1);
170 #endif
171  len = dns_state->record_len - dns_state->offset;
172  }
173 
174  memcpy(dns_state->buffer + dns_state->offset, data, len);
175  dns_state->offset += len;
176  return 0;
177 }

275 static int DNSTCPRequestParse(Flow *f, void *dstate,
276  AppLayerParserState *pstate,
277  uint8_t *input, uint32_t input_len,
278  void *local_data)
279 {
280  DNSState *dns_state = (DNSState *)dstate;
281  SCLogDebug("starting %u", input_len);
282 
283  if (input == NULL && AppLayerParserStateIssetFlag(pstate, APP_LAYER_PARSER_EOF)) {
284  SCReturnInt(1);
285  }
286 
287  /** \todo remove this when PP is fixed to enforce ipproto */
288  if (f != NULL && f->proto != IPPROTO_TCP)
289  SCReturnInt(-1);
290 
291  /* probably a rst/fin sending an eof */
292  if (input == NULL || input_len == 0) {
293  goto insufficient_data;
294  }
295 
296 next_record:
297  /* if this is the beginning of a record, we need at least the header */
298  if (dns_state->offset == 0 && input_len < sizeof(DNSTcpHeader)) {
299  SCLogDebug("ilen too small, hoped for at least %"PRIuMAX, (uintmax_t)sizeof(DNSTcpHeader));
300  goto insufficient_data;
301  }
302  SCLogDebug("input_len %u offset %u record %u",
303  input_len, dns_state->offset, dns_state->record_len);
304 
305  /* this is the first data of this record */
306  if (dns_state->offset == 0) {
307  DNSTcpHeader *dns_tcp_header = (DNSTcpHeader *)input;
308  SCLogDebug("DNS %p", dns_tcp_header);
309 
310  if (ntohs(dns_tcp_header->len) < sizeof(DNSHeader)) {
311  /* bogus len, doesn't fit even basic dns header */
312  goto bad_data;
313  } else if (ntohs(dns_tcp_header->len) == (input_len-2)) {
314  /* we have all data, so process w/o buffering */
315  if (DNSRequestParseData(f, dns_state, input+2, input_len-2) < 0)
316  goto bad_data;
317 
318  } else if ((input_len-2) > ntohs(dns_tcp_header->len)) {
319  /* we have all data, so process w/o buffering */
320  if (DNSRequestParseData(f, dns_state, input+2, ntohs(dns_tcp_header->len)) < 0)
321  goto bad_data;
322 
323  /* treat the rest of the data as a (potential) new record */
324  input += ntohs(dns_tcp_header->len);
325  input_len -= ntohs(dns_tcp_header->len);
326  goto next_record;
327  } else {
328  /* not enough data, store record length and buffer */
329  dns_state->record_len = ntohs(dns_tcp_header->len);
330  BufferData(dns_state, input+2, input_len-2);
331  }
332  } else if (input_len + dns_state->offset < dns_state->record_len) {
333  /* we don't have the full record yet, buffer */
334  BufferData(dns_state, input, input_len);
335  } else if (input_len > (uint32_t)(dns_state->record_len - dns_state->offset)) {
336  /* more data than expected, we may have another record coming up */
337  uint16_t need = (dns_state->record_len - dns_state->offset);
338  BufferData(dns_state, input, need);
339  int r = DNSRequestParseData(f, dns_state, dns_state->buffer, dns_state->record_len);
340  BufferReset(dns_state);
341  if (r < 0)
342  goto bad_data;
343 
344  /* treat the rest of the data as a (potential) new record */
345  input += need;
346  input_len -= need;
347  goto next_record;
348  } else {
349  /* implied exactly the amount of data we want
350  * add current to buffer, then inspect buffer */
351  BufferData(dns_state, input, input_len);
352  int r = DNSRequestParseData(f, dns_state, dns_state->buffer, dns_state->record_len);
353  BufferReset(dns_state);
354  if (r < 0)
355  goto bad_data;
356  }
357 
358  SCReturnInt(1);
359 insufficient_data:
360  SCReturnInt(-1);
361 bad_data:
362  SCReturnInt(-1);
363 }

365 static int DNSReponseParseData(Flow *f, DNSState *dns_state, const uint8_t *input, const uint32_t input_len)
366 {
367  DNSHeader *dns_header = (DNSHeader *)input;
368 
369  if (DNSValidateResponseHeader(dns_state, dns_header) < 0)
370  goto bad_data;
371 
372  DNSTransaction *tx = NULL;
373  int found = 0;
374  if ((tx = DNSTransactionFindByTxId(dns_state, ntohs(dns_header->tx_id))) != NULL)
375  found = 1;
376 
377  if (!found) {
378  SCLogDebug("DNS_DECODER_EVENT_UNSOLLICITED_RESPONSE");
379  DNSSetEvent(dns_state, DNS_DECODER_EVENT_UNSOLLICITED_RESPONSE);
380  }
381 
382  uint16_t q;
383  const uint8_t *data = input + sizeof(DNSHeader);
384  for (q = 0; q < ntohs(dns_header->questions); q++) {
385  uint8_t fqdn[DNS_MAX_SIZE];
386  uint16_t fqdn_offset = 0;
387 
388  if (input + input_len < data + 1) {
389  SCLogDebug("input buffer too small for len field");
390  goto insufficient_data;
391  }
392  SCLogDebug("qry length %u", *data);
393 
394  while (*data != 0) {
395  uint8_t length = *data;
396  data++;
397 
398  if (length > 0) {
399  if (input + input_len < data + length) {
400  SCLogDebug("input buffer too small for domain of len %u", length);
401  goto insufficient_data;
402  }
403  //PrintRawDataFp(stdout, data, length);
404 
405  if ((size_t)(fqdn_offset + length + 1) < sizeof(fqdn)) {
406  memcpy(fqdn + fqdn_offset, data, length);
407  fqdn_offset += length;
408  fqdn[fqdn_offset++] = '.';
409  }
410  }
411 
412  data += length;
413 
414  if (input + input_len < data + 1) {
415  SCLogDebug("input buffer too small for len field");
416  goto insufficient_data;
417  }
418 
419  SCLogDebug("length %u", *data);
420  }
421  if (fqdn_offset) {
422  fqdn_offset--;
423  }
424 
425  data++;
426  if (input + input_len < data + sizeof(DNSQueryTrailer)) {
427  SCLogDebug("input buffer too small for DNSQueryTrailer");
428  goto insufficient_data;
429  }
430 #if DEBUG
431  DNSQueryTrailer *trailer = (DNSQueryTrailer *)data;
432  SCLogDebug("trailer type %04x class %04x", ntohs(trailer->type), ntohs(trailer->class));
433 #endif
434  data += sizeof(DNSQueryTrailer);
435  }
436 
437  for (q = 0; q < ntohs(dns_header->answer_rr); q++) {
438  data = DNSReponseParse(dns_state, dns_header, q, DNS_LIST_ANSWER,
439  input, input_len, data);
440  if (data == NULL) {
441  goto insufficient_data;
442  }
443  }
444 
445  //PrintRawDataFp(stdout, (uint8_t *)data, input_len - (data - input));
446  for (q = 0; q < ntohs(dns_header->authority_rr); q++) {
447  data = DNSReponseParse(dns_state, dns_header, q, DNS_LIST_AUTHORITY,
448  input, input_len, data);
449  if (data == NULL) {
450  goto insufficient_data;
451  }
452  }
453 
454  /* parse rcode, e.g. "noerror" or "nxdomain" */
455  uint8_t rcode = ntohs(dns_header->flags) & 0x0F;
456  if (rcode <= DNS_RCODE_NOTZONE) {
457  SCLogDebug("rcode %u", rcode);
458  if (tx != NULL)
459  tx->rcode = rcode;
460  } else {
461  /* this is not invalid, rcodes can be user defined */
462  SCLogDebug("unexpected DNS rcode %u", rcode);
463  }
464 
465  if (ntohs(dns_header->flags) & 0x0080) {
466  SCLogDebug("recursion desired");
467  if (tx != NULL)
468  tx->recursion_desired = 1;
469  }
470 
471  if (tx != NULL) {
472  tx->replied = 1;
473  }
474 
475  SCReturnInt(1);
476 bad_data:
477 insufficient_data:
478  SCReturnInt(-1);
479 }

575 static uint16_t DNSTcpProbingParser(uint8_t *input, uint32_t ilen, uint32_t *offset)
576 {
577  if (ilen == 0 || ilen < sizeof(DNSTcpHeader)) {
578  SCLogDebug("ilen too small, hoped for at least %"PRIuMAX, (uintmax_t)sizeof(DNSTcpHeader));
579  return ALPROTO_UNKNOWN;
580  }
581 
582  DNSTcpHeader *dns_header = (DNSTcpHeader *)input;
583  if (ntohs(dns_header->len) < sizeof(DNSHeader)) {
584  /* length field bogus, won't even fit a minimal DNS header. */
585  return ALPROTO_FAILED;
586  } else if (ntohs(dns_header->len) > ilen) {
587  int r = DNSTCPRequestParseProbe(input, ilen);
588  if (r == -1) {
589  /* probing parser told us "bad data", so it's not
590  * DNS */
591  return ALPROTO_FAILED;
592  } else if (ilen > 512) {
593  SCLogDebug("all the parser told us was not enough data, which is expected. Lets assume it's DNS");
594  return ALPROTO_DNS;
595  }
596 
597  SCLogDebug("not yet enough info %u > %u", ntohs(dns_header->len), ilen);
598  return ALPROTO_UNKNOWN;
599  }
600 
601  int r = DNSTCPRequestParseProbe(input, ilen);
602  if (r != 1)
603  return ALPROTO_FAILED;
604 
605  SCLogDebug("ALPROTO_DNS");
606  return ALPROTO_DNS;
607 }

609 void RegisterDNSTCPParsers(void)
610 {
611  char *proto_name = "dns";
612 
613  /** DNS */
614  if (AppLayerProtoDetectConfProtoDetectionEnabled("tcp", proto_name)) {
615  AppLayerProtoDetectRegisterProtocol(ALPROTO_DNS, proto_name);
616 
617  if (RunmodeIsUnittests()) {
618  AppLayerProtoDetectPPRegister(IPPROTO_TCP,
619  "53",
620  ALPROTO_DNS,
621  0, sizeof(DNSTcpHeader),
622  STREAM_TOSERVER,
623  DNSTcpProbingParser);
624  } else {
625  int have_cfg = AppLayerProtoDetectPPParseConfPorts("tcp", IPPROTO_TCP,
626  proto_name, ALPROTO_DNS,
627  0, sizeof(DNSTcpHeader),
628  DNSTcpProbingParser);
