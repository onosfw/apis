63  tmqh_table[TMQH_RINGBUFFER_SRMW].InHandler = TmqhInputRingBufferSrMw;
64  tmqh_table[TMQH_RINGBUFFER_SRMW].InShutdownHandler = TmqhInputRingBufferShutdownHandler;
65  tmqh_table[TMQH_RINGBUFFER_SRMW].OutHandler = TmqhOutputRingBufferSrMw;

67  memset(ringbuffers, 0, sizeof(ringbuffers));
68 
69  int i = 0;
70  for (i = 0; i < 256; i++) {
71  ringbuffers[i] = RingBuffer8Init();
