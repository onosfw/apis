57 static json_t *JsonSmtpDataLogger(const Flow *f, void *state, void *vtx, uint64_t tx_id)
58 {
59  json_t *sjs = json_object();
60  SMTPTransaction *tx = vtx;
61  SMTPString *rcptto_str;
62  if (sjs == NULL) {
63  return NULL;
64  }
65  if (((SMTPState *)state)->helo) {
66  json_object_set_new(sjs, "helo",
67  json_string((const char *)((SMTPState *)state)->helo));
68  }
69  if (tx->mail_from) {
70  json_object_set_new(sjs, "mail_from",
71  json_string((const char *)tx->mail_from));
72  }
73  if (!TAILQ_EMPTY(&tx->rcpt_to_list)) {
74  json_t *js_rcptto = json_array();
75  if (likely(js_rcptto != NULL)) {
76  TAILQ_FOREACH(rcptto_str, &tx->rcpt_to_list, next) {
77  json_array_append_new(js_rcptto, json_string((char *)rcptto_str->str));
78  }
79  json_object_set_new(sjs, "rcpt_to", js_rcptto);
80  }
81  }
82 
83  return sjs;
84 }

86 static int JsonSmtpLogger(ThreadVars *tv, void *thread_data, const Packet *p, Flow *f, void *state, void *tx, uint64_t tx_id)
87 {
88  SCEnter();
89  JsonEmailLogThread *jhl = (JsonEmailLogThread *)thread_data;
90  MemBuffer *buffer = (MemBuffer *)jhl->buffer;
91 
92  json_t *sjs;
93  json_t *js = CreateJSONHeaderWithTxId((Packet *)p, 1, "smtp", tx_id);
94  if (unlikely(js == NULL))
95  return TM_ECODE_OK;
96 
97  /* reset */
98  MemBufferReset(buffer);
99 
100  sjs = JsonSmtpDataLogger(f, state, tx, tx_id);
101  if (sjs) {
102  json_object_set_new(js, "smtp", sjs);
103  }
104 
105  if (JsonEmailLogJson(jhl, js, p, f, state, tx, tx_id) == TM_ECODE_OK) {
106  OutputJSONBuffer(js, jhl->emaillog_ctx->file_ctx, buffer);
107  }
108  json_object_del(js, "email");
109  if (sjs) {
110  json_object_del(js, "smtp");
111  }
112 
113  json_object_clear(js);
114  json_decref(js);
115 
116  SCReturnInt(TM_ECODE_OK);
117 
118 }

155 OutputCtx *OutputSmtpLogInit(ConfNode *conf)
156 {
157  LogFileCtx *file_ctx = LogFileNewCtx();
158  if(file_ctx == NULL) {
159  SCLogError(SC_ERR_HTTP_LOG_GENERIC, "couldn't create new file_ctx");
160  return NULL;
161  }
162 
163  if (SCConfLogOpenGeneric(conf, file_ctx, DEFAULT_LOG_FILENAME, 1) < 0) {
164  LogFileFreeCtx(file_ctx);
165  return NULL;
166  }
167 
168  OutputJsonEmailCtx *email_ctx = SCMalloc(sizeof(OutputJsonEmailCtx));
169  if (unlikely(email_ctx == NULL)) {
170  LogFileFreeCtx(file_ctx);
171  return NULL;
172  }
173 
174  OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
175  if (unlikely(output_ctx == NULL)) {
176  LogFileFreeCtx(file_ctx);
177  SCFree(email_ctx);
178  return NULL;
179  }
180 
181  email_ctx->file_ctx = file_ctx;
182 
183  output_ctx->data = email_ctx;
184  output_ctx->DeInit = OutputSmtpLogDeInitCtx;
185 
186  /* enable the logger for the app layer */
187  AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_SMTP);
188 
189  return output_ctx;
190 }

220 static TmEcode JsonSmtpLogThreadInit(ThreadVars *t, void *initdata, void **data)
221 {
222  JsonEmailLogThread *aft = SCMalloc(sizeof(JsonEmailLogThread));
223  if (unlikely(aft == NULL))
224  return TM_ECODE_FAILED;
225  memset(aft, 0, sizeof(JsonEmailLogThread));
226 
227  if(initdata == NULL)
228  {
229  SCLogDebug("Error getting context for SMTPLog. \"initdata\" argument NULL");
230  SCFree(aft);
231  return TM_ECODE_FAILED;
232  }
233 
234  /* Use the Ouptut Context (file pointer and mutex) */
235  aft->emaillog_ctx = ((OutputCtx *)initdata)->data;
236 
237  aft->buffer = MemBufferCreateNew(OUTPUT_BUFFER_SIZE);
238  if (aft->buffer == NULL) {
239  SCFree(aft);
240  return TM_ECODE_FAILED;
241  }
242 
243  *data = (void *)aft;
244  return TM_ECODE_OK;
245 }
