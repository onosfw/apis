71  stub_data_buffer_len = &sstate->dcerpc.dcerpcresponse.stub_data_buffer_len;
72  stub_data_fresh = &sstate->dcerpc.dcerpcresponse.stub_data_fresh;
73  }
74

76 
77  if (stub_len == 0) {
78  SCReturnUInt(0);
79  }
80  /* if the frag is the the first frag irrespective of it being a part of

98

101  *stub_data_buffer_len += stub_len;
102

104  sstate->dcerpc.bytesprocessed += stub_len;
105
