58 int OutputRegisterTxLogger(const char *name, AppProto alproto, TxLogger LogFunc, OutputCtx *output_ctx)
59 {
60  int module_id = TmModuleGetIdByName(name);
61  if (module_id < 0)
62  return -1;
63 
64  OutputTxLogger *op = SCMalloc(sizeof(*op));
65  if (op == NULL)
66  return -1;
67  memset(op, 0x00, sizeof(*op));
68 
69  op->alproto = alproto;
70  op->LogFunc = LogFunc;
71  op->output_ctx = output_ctx;
72  op->name = name;
73  op->module_id = (TmmId) module_id;
74 
75  if (list == NULL)
76  list = op;
77  else {
78  OutputTxLogger *t = list;
79  while (t->next)
80  t = t->next;
81  t->next = op;
82  }
83 
84  SCLogDebug("OutputRegisterTxLogger happy");
85  return 0;
86 }

88 static TmEcode OutputTxLog(ThreadVars *tv, Packet *p, void *thread_data, PacketQueue *pq, PacketQueue *postpq)
89 {
90  BUG_ON(thread_data == NULL);
91  BUG_ON(list == NULL);
92 
93  OutputLoggerThreadData *op_thread_data = (OutputLoggerThreadData *)thread_data;
94  OutputTxLogger *logger = list;
95  OutputLoggerThreadStore *store = op_thread_data->store;
96 
97  BUG_ON(logger == NULL && store != NULL);
98  BUG_ON(logger != NULL && store == NULL);
99  BUG_ON(logger == NULL && store == NULL);
100 
101  if (p->flow == NULL)
102  return TM_ECODE_OK;
103 
104  Flow * const f = p->flow;
105 
106  FLOWLOCK_WRLOCK(f); /* WRITE lock before we updated flow logged id */
107  AppProto alproto = f->alproto;
108 
109  if (AppLayerParserProtocolIsTxAware(p->proto, alproto) == 0)
110  goto end;
111  if (AppLayerParserProtocolHasLogger(p->proto, alproto) == 0)
112  goto end;
113 
114  void *alstate = f->alstate;
115  if (alstate == NULL) {
116  SCLogDebug("no alstate");
117  goto end;
118  }
119 
120  uint64_t total_txs = AppLayerParserGetTxCnt(p->proto, alproto, alstate);
121  uint64_t tx_id = AppLayerParserGetTransactionLogId(f->alparser);
122  int tx_progress_done_value_ts =
123  AppLayerParserGetStateProgressCompletionStatus(p->proto, alproto,
124  STREAM_TOSERVER);
125  int tx_progress_done_value_tc =
126  AppLayerParserGetStateProgressCompletionStatus(p->proto, alproto,
127  STREAM_TOCLIENT);
128  int proto_logged = 0;
129 
130  for (; tx_id < total_txs; tx_id++)
131  {
132  void *tx = AppLayerParserGetTx(p->proto, alproto, alstate, tx_id);
133  if (tx == NULL) {
134  SCLogDebug("tx is NULL not logging");
135  continue;
136  }
137 
138  if (!(AppLayerParserStateIssetFlag(f->alparser, APP_LAYER_PARSER_EOF)))
139  {
140  int tx_progress = AppLayerParserGetStateProgress(p->proto, alproto,
141  tx, FlowGetDisruptionFlags(f, STREAM_TOSERVER));
142  if (tx_progress < tx_progress_done_value_ts) {
143  SCLogDebug("progress not far enough, not logging");
144  break;
145  }
146 
147  tx_progress = AppLayerParserGetStateProgress(p->proto, alproto,
148  tx, FlowGetDisruptionFlags(f, STREAM_TOCLIENT));
149  if (tx_progress < tx_progress_done_value_tc) {
150  SCLogDebug("progress not far enough, not logging");
151  break;
152  }
153  }
154 
155  // call each logger here (pseudo code)
156  logger = list;
157  store = op_thread_data->store;
158  while (logger && store) {
159  BUG_ON(logger->LogFunc == NULL);
160 
161  SCLogDebug("logger %p", logger);
162  if (logger->alproto == alproto) {
163  SCLogDebug("alproto match, logging tx_id %ju", tx_id);
164  PACKET_PROFILING_TMM_START(p, logger->module_id);
165  logger->LogFunc(tv, store->thread_data, p, f, alstate, tx, tx_id);
166  PACKET_PROFILING_TMM_END(p, logger->module_id);
167  proto_logged = 1;
168  }
169 
170  logger = logger->next;
171  store = store->next;
172 
173  BUG_ON(logger == NULL && store != NULL);
174  BUG_ON(logger != NULL && store == NULL);
175  }
176 
177  if (proto_logged) {
178  SCLogDebug("updating log tx_id %ju", tx_id);
179  AppLayerParserSetTransactionLogId(f->alparser);
180  }
181  }
182 
183 end:
184  FLOWLOCK_UNLOCK(f);
185  return TM_ECODE_OK;
186 }

188 /** \brief thread init for the tx logger
189  * This will run the thread init functions for the individual registered
190  * loggers */
191 static TmEcode OutputTxLogThreadInit(ThreadVars *tv, void *initdata, void **data)
192 {
193  OutputLoggerThreadData *td = SCMalloc(sizeof(*td));
194  if (td == NULL)
195  return TM_ECODE_FAILED;
196  memset(td, 0x00, sizeof(*td));
197 
198  *data = (void *)td;
199 
200  SCLogDebug("OutputTxLogThreadInit happy (*data %p)", *data);
201 
202  OutputTxLogger *logger = list;
203  while (logger) {
204  TmModule *tm_module = TmModuleGetByName((char *)logger->name);
205  if (tm_module == NULL) {
206  SCLogError(SC_ERR_INVALID_ARGUMENT,
207  "TmModuleGetByName for %s failed", logger->name);
208  exit(EXIT_FAILURE);
209  }
210 
211  if (tm_module->ThreadInit) {
212  void *retptr = NULL;
213  if (tm_module->ThreadInit(tv, (void *)logger->output_ctx, &retptr) == TM_ECODE_OK) {
214  OutputLoggerThreadStore *ts = SCMalloc(sizeof(*ts));
215 /* todo */ BUG_ON(ts == NULL);
216  memset(ts, 0x00, sizeof(*ts));
217 
218  /* store thread handle */
219  ts->thread_data = retptr;
220 
221  if (td->store == NULL) {
222  td->store = ts;
223  } else {
224  OutputLoggerThreadStore *tmp = td->store;
225  while (tmp->next != NULL)
226  tmp = tmp->next;
227  tmp->next = ts;
228  }
229 
230  SCLogDebug("%s is now set up", logger->name);
231  }
232  }
233 
234  logger = logger->next;
235  }
236 
237  return TM_ECODE_OK;
238 }

240 static TmEcode OutputTxLogThreadDeinit(ThreadVars *tv, void *thread_data)
241 {
242  OutputLoggerThreadData *op_thread_data = (OutputLoggerThreadData *)thread_data;
243  OutputLoggerThreadStore *store = op_thread_data->store;
244  OutputTxLogger *logger = list;
245 
246  while (logger && store) {
247  TmModule *tm_module = TmModuleGetByName((char *)logger->name);
248  if (tm_module == NULL) {
249  SCLogError(SC_ERR_INVALID_ARGUMENT,
250  "TmModuleGetByName for %s failed", logger->name);
251  exit(EXIT_FAILURE);
252  }
253 
254  if (tm_module->ThreadDeinit) {
255  tm_module->ThreadDeinit(tv, store->thread_data);
256  }
257 
258  OutputLoggerThreadStore *next_store = store->next;
259  SCFree(store);
260  store = next_store;
261  logger = logger->next;
262  }
263  return TM_ECODE_OK;
264 }
