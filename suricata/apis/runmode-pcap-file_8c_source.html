59 /**
60  * \brief Single thread version of the Pcap file processing.
61  */
62 int RunModeFilePcapSingle(void)
63 {
64  char *file = NULL;
65  if (ConfGet("pcap-file.file", &file) == 0) {
66  SCLogError(SC_ERR_RUNMODE, "Failed retrieving pcap-file from Conf");
67  exit(EXIT_FAILURE);
68  }
69 
70  RunModeInitialize();
71  TimeModeSetOffline();
72 
73  PcapFileGlobalInit();
74 
75  /* create the threads */
76  ThreadVars *tv = TmThreadCreatePacketHandler("PcapFile",
77  "packetpool", "packetpool",
78  "packetpool", "packetpool",
79  "pktacqloop");
80  if (tv == NULL) {
81  SCLogError(SC_ERR_RUNMODE, "threading setup failed");
82  exit(EXIT_FAILURE);
83  }
84 
85  TmModule *tm_module = TmModuleGetByName("ReceivePcapFile");
86  if (tm_module == NULL) {
87  SCLogError(SC_ERR_RUNMODE, "TmModuleGetByName failed for ReceivePcap");
88  exit(EXIT_FAILURE);
89  }
90  TmSlotSetFuncAppend(tv, tm_module, file);
91 
92  tm_module = TmModuleGetByName("DecodePcapFile");
93  if (tm_module == NULL) {
94  SCLogError(SC_ERR_RUNMODE, "TmModuleGetByName DecodePcap failed");
95  exit(EXIT_FAILURE);
96  }
97  TmSlotSetFuncAppend(tv, tm_module, NULL);
98 
99  tm_module = TmModuleGetByName("StreamTcp");
100  if (tm_module == NULL) {
101  SCLogError(SC_ERR_RUNMODE, "TmModuleGetByName StreamTcp failed");
102  exit(EXIT_FAILURE);
103  }
104  TmSlotSetFuncAppend(tv, tm_module, NULL);
105 
106  if (DetectEngineEnabled()) {
107  tm_module = TmModuleGetByName("Detect");
108  if (tm_module == NULL) {
109  SCLogError(SC_ERR_RUNMODE, "TmModuleGetByName Detect failed");
110  exit(EXIT_FAILURE);
111  }
112  TmSlotSetFuncAppend(tv, tm_module, NULL);
113  }
114 
115  SetupOutputs(tv);
116 
117  TmThreadSetCPU(tv, DETECT_CPU_SET);
118 
119  if (TmThreadSpawn(tv) != TM_ECODE_OK) {
120  SCLogError(SC_ERR_RUNMODE, "TmThreadSpawn failed");
121  exit(EXIT_FAILURE);
122  }
123 
124  return 0;
125 }

127 /**
128  * \brief RunModeFilePcapAutoFp set up the following thread packet handlers:
129  * - Receive thread (from pcap file)
130  * - Decode thread
131  * - Stream thread
132  * - Detect: If we have only 1 cpu, it will setup one Detect thread
133  * If we have more than one, it will setup num_cpus - 1
134  * starting from the second cpu available.
135  * - Outputs thread
136  * By default the threads will use the first cpu available
137  * except the Detection threads if we have more than one cpu.
138  *
139  * \retval 0 If all goes well. (If any problem is detected the engine will
140  * exit()).
141  */
142 int RunModeFilePcapAutoFp(void)
143 {
144  SCEnter();
145  char tname[TM_THREAD_NAME_MAX];
146  char qname[TM_QUEUE_NAME_MAX];
147  uint16_t cpu = 0;
148  char *queues = NULL;
149  int thread;
150 
151  RunModeInitialize();
152  RunmodeSetFlowStreamAsync();
153 
154  char *file = NULL;
155  if (ConfGet("pcap-file.file", &file) == 0) {
156  SCLogError(SC_ERR_RUNMODE, "Failed retrieving pcap-file from Conf");
157  exit(EXIT_FAILURE);
158  }
159  SCLogDebug("file %s", file);
160 
161  TimeModeSetOffline();
162 
163  PcapFileGlobalInit();
164 
165  /* Available cpus */
166  uint16_t ncpus = UtilCpuGetNumProcessorsOnline();
167 
168  /* start with cpu 1 so that if we're creating an odd number of detect
169  * threads we're not creating the most on CPU0. */
170  if (ncpus > 0)
171  cpu = 1;
172 
173  /* always create at least one thread */
174  int thread_max = TmThreadGetNbThreads(DETECT_CPU_SET);
175  if (thread_max == 0)
176  thread_max = ncpus * threading_detect_ratio;
177  if (thread_max < 1)
178  thread_max = 1;
179 
180  queues = RunmodeAutoFpCreatePickupQueuesString(thread_max);
181  if (queues == NULL) {
182  SCLogError(SC_ERR_RUNMODE, "RunmodeAutoFpCreatePickupQueuesString failed");
183  exit(EXIT_FAILURE);
184  }
185 
186  /* create the threads */
187  ThreadVars *tv_receivepcap =
188  TmThreadCreatePacketHandler("ReceivePcapFile",
189  "packetpool", "packetpool",
190  queues, "flow",
191  "pktacqloop");
192  SCFree(queues);
193 
194  if (tv_receivepcap == NULL) {
195  SCLogError(SC_ERR_FATAL, "threading setup failed");
196  exit(EXIT_FAILURE);
197  }
198  TmModule *tm_module = TmModuleGetByName("ReceivePcapFile");
199  if (tm_module == NULL) {
200  SCLogError(SC_ERR_RUNMODE, "TmModuleGetByName failed for ReceivePcap");
201  exit(EXIT_FAILURE);
202  }
203  TmSlotSetFuncAppend(tv_receivepcap, tm_module, file);
204 
205  tm_module = TmModuleGetByName("DecodePcapFile");
206  if (tm_module == NULL) {
207  SCLogError(SC_ERR_RUNMODE, "TmModuleGetByName DecodePcap failed");
208  exit(EXIT_FAILURE);
209  }
210  TmSlotSetFuncAppend(tv_receivepcap, tm_module, NULL);
211 
212  TmThreadSetCPU(tv_receivepcap, RECEIVE_CPU_SET);
213 
214  if (TmThreadSpawn(tv_receivepcap) != TM_ECODE_OK) {
215  SCLogError(SC_ERR_RUNMODE, "TmThreadSpawn failed");
216  exit(EXIT_FAILURE);
217  }
218 
219  for (thread = 0; thread < thread_max; thread++) {
220  snprintf(tname, sizeof(tname), "Detect%d", thread+1);
221  snprintf(qname, sizeof(qname), "pickup%d", thread+1);
222 
223  SCLogDebug("tname %s, qname %s", tname, qname);
224 
225  char *thread_name = SCStrdup(tname);
226  if (unlikely(thread_name == NULL)) {
227  SCLogError(SC_ERR_RUNMODE, "failed to strdup thread name");
228  exit(EXIT_FAILURE);
229  }
230  SCLogDebug("Assigning %s affinity to cpu %u", thread_name, cpu);
231 
232  ThreadVars *tv_detect_ncpu =
233  TmThreadCreatePacketHandler(thread_name,
234  qname, "flow",
235  "packetpool", "packetpool",
236  "varslot");
237  if (tv_detect_ncpu == NULL) {
238  SCLogError(SC_ERR_RUNMODE, "TmThreadsCreate failed");
239  exit(EXIT_FAILURE);
240  }
241  tm_module = TmModuleGetByName("StreamTcp");
242  if (tm_module == NULL) {
243  SCLogError(SC_ERR_RUNMODE, "TmModuleGetByName StreamTcp failed");
244  exit(EXIT_FAILURE);
245  }
246  TmSlotSetFuncAppend(tv_detect_ncpu, tm_module, NULL);
247 
248  if (DetectEngineEnabled()) {
249  tm_module = TmModuleGetByName("Detect");
250  if (tm_module == NULL) {
251  SCLogError(SC_ERR_RUNMODE, "TmModuleGetByName Detect failed");
252  exit(EXIT_FAILURE);
253  }
254  TmSlotSetFuncAppend(tv_detect_ncpu, tm_module, NULL);
255  }
256 
257  char *thread_group_name = SCStrdup("Detect");
258  if (unlikely(thread_group_name == NULL)) {
259  SCLogError(SC_ERR_RUNMODE, "error allocating memory");
260  exit(EXIT_FAILURE);
261  }
262  tv_detect_ncpu->thread_group_name = thread_group_name;
263 
264  /* add outputs as well */
265  SetupOutputs(tv_detect_ncpu);
266 
267  TmThreadSetCPU(tv_detect_ncpu, DETECT_CPU_SET);
268 
269  if (TmThreadSpawn(tv_detect_ncpu) != TM_ECODE_OK) {
270  SCLogError(SC_ERR_RUNMODE, "TmThreadSpawn failed");
271  exit(EXIT_FAILURE);
272  }
273 
274  if ((cpu + 1) == ncpus)
275  cpu = 0;
276  else
277  cpu++;
278  }
279 
280  return 0;
281 }
