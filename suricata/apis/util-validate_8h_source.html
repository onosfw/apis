36 /** \brief test if a flow is locked.
37  *
38  * If trylock returns 0 it got a lock. Which means
39  * the flow was previously unlocked.
40  */
41 #define DEBUG_ASSERT_FLOW_LOCKED(f) do { \
42  if ((f) != NULL) { \
43  int r = SCMutexTrylock(&(f)->m); \
44  if (r == 0) { \
45  BUG_ON(1); \
46  } \
47  } \
48 } while(0)

50 /** \brief validate the integrity of the flow
51  *
52  * BUG_ON's on problems
53  */
54 #define DEBUG_VALIDATE_FLOW(f) do { \
55  if ((f) != NULL) { \
56  SCMutexLock(&(f)->m); \
57  BUG_ON((f)->flags & FLOW_IPV4 && \
58  (f)->flags & FLOW_IPV6); \
59  if ((f)->proto == IPPROTO_TCP) { \
60  BUG_ON((f)->alstate != NULL && \
61  (f)->alparser == NULL); \
62  } \
63  SCMutexUnlock(&(f)->m); \
64  } \
65 } while(0)
66 
67 /** \brief validate the integrity of the packet
68  *
69  * BUG_ON's on problems
70  */
71 #define DEBUG_VALIDATE_PACKET(p) do { \
72  if ((p) != NULL) { \
73  if ((p)->flow != NULL) { \
74  DEBUG_VALIDATE_FLOW((p)->flow); \
75  } \
76  if (!((p)->flags & (PKT_IS_FRAGMENT|PKT_IS_INVALID))) { \
77  if ((p)->proto == IPPROTO_TCP) { \
78  BUG_ON((p)->tcph == NULL); \
79  } else if ((p)->proto == IPPROTO_UDP) { \
80  BUG_ON((p)->udph == NULL); \
81  } else if ((p)->proto == IPPROTO_ICMP) { \
82  BUG_ON((p)->icmpv4h == NULL); \
83  } else if ((p)->proto == IPPROTO_SCTP) { \
84  BUG_ON((p)->sctph == NULL); \
85  } else if ((p)->proto == IPPROTO_ICMPV6) { \
86  BUG_ON((p)->icmpv6h == NULL); \
87  } \
88  } \
89  if ((p)->payload_len > 0) { \
90  BUG_ON((p)->payload == NULL); \
91  } \
92  BUG_ON((p)->ip4h != NULL && (p)->ip6h != NULL); \
93  BUG_ON((p)->flowflags != 0 && (p)->flow == NULL); \
94  BUG_ON((p)->flowflags & FLOW_PKT_TOSERVER &&\
95  (p)->flowflags & FLOW_PKT_TOCLIENT); \
96  } \
97 } while(0)
