94 static int CallLoggers(ThreadVars *tv, OutputLoggerThreadStore *store_list,
95  Packet *p, const File *ff, const FileData *ffd, uint8_t flags)
96 {
97  OutputFiledataLogger *logger = list;
98  OutputLoggerThreadStore *store = store_list;
99  int file_logged = 0;
100 
101  while (logger && store) {
102  BUG_ON(logger->LogFunc == NULL);
103 
104  SCLogDebug("logger %p", logger);
105  PACKET_PROFILING_TMM_START(p, logger->module_id);
106  logger->LogFunc(tv, store->thread_data, (const Packet *)p, ff, ffd, flags);
107  PACKET_PROFILING_TMM_END(p, logger->module_id);
108 
109  file_logged = 1;
110 
111  logger = logger->next;
112  store = store->next;
113 
114  BUG_ON(logger == NULL && store != NULL);
115  BUG_ON(logger != NULL && store == NULL);
116  }
117  return file_logged;
118 }

120 static TmEcode OutputFiledataLog(ThreadVars *tv, Packet *p, void *thread_data, PacketQueue *pq, PacketQueue *postpq)
121 {
122  BUG_ON(thread_data == NULL);
123  BUG_ON(list == NULL);
124 
125  OutputLoggerThreadData *op_thread_data = (OutputLoggerThreadData *)thread_data;
126  OutputFiledataLogger *logger = list;
127  OutputLoggerThreadStore *store = op_thread_data->store;
128 
129  BUG_ON(logger == NULL && store != NULL);
130  BUG_ON(logger != NULL && store == NULL);
131  BUG_ON(logger == NULL && store == NULL);
132 
133  uint8_t flags = 0;
134  Flow * const f = p->flow;
135 
136  /* no flow, no files */
137  if (f == NULL) {
138  SCReturnInt(TM_ECODE_OK);
139  }
140 
141  if (p->flowflags & FLOW_PKT_TOCLIENT)
142  flags |= STREAM_TOCLIENT;
143  else
144  flags |= STREAM_TOSERVER;
145 
146  int file_close = (p->flags & PKT_PSEUDO_STREAM_END) ? 1 : 0;
147  int file_trunc = 0;
148 
149  FLOWLOCK_WRLOCK(f); // < need write lock for FiledataPrune below
150  file_trunc = StreamTcpReassembleDepthReached(p);
151 
152  FileContainer *ffc = AppLayerParserGetFiles(p->proto, f->alproto,
153  f->alstate, flags);
154  SCLogDebug("ffc %p", ffc);
155  if (ffc != NULL) {
156  File *ff;
157  for (ff = ffc->head; ff != NULL; ff = ff->next) {
158  if (FileForceMagic() && ff->magic == NULL) {
159  FilemagicGlobalLookup(ff);
160  }
161 
162  SCLogDebug("ff %p", ff);
163  if (ff->flags & FILE_STORED) {
164  SCLogDebug("stored flag set");
165  continue;
166  }
167 
168  if (!(ff->flags & FILE_STORE)) {
169  SCLogDebug("ff FILE_STORE not set");
170  continue;
171  }
172 
173  /* if we have no data chunks left to log, we should still
174  * close the logger(s) */
175  if (ff->chunks_head == NULL && (file_trunc || file_close)) {
176  CallLoggers(tv, store, p, ff, NULL, OUTPUT_FILEDATA_FLAG_CLOSE);
177  ff->flags |= FILE_STORED;
178  continue;
179  }
180 
181  FileData *ffd;
182  for (ffd = ff->chunks_head; ffd != NULL; ffd = ffd->next) {
183  uint8_t flags = 0;
184  int file_logged = 0;
185  FileData *write_ffd = ffd;
186 
187  SCLogDebug("ffd %p", ffd);
188 
189  /* special case: on stream end we may inform the
190  * loggers that the file is truncated. In this
191  * case we already logged the current ffd, which
192  * is the last in our list. */
193  if (ffd->stored == 1) {
194  if (!(file_close == 1 || ffd->next == NULL)) {
195  continue;
196  }
197 
198  // call writer with NULL ffd, so it can 'close'
199  // so really a 'close' msg
200  write_ffd = NULL;
201  flags |= OUTPUT_FILEDATA_FLAG_CLOSE;
202  }
203 
204  /* store */
205 
206  /* if file_id == 0, this is the first store of this file */
207  if (ff->file_id == 0) {
208  /* new file */
209  ff->file_id = SC_ATOMIC_ADD(file_id, 1);
210  flags |= OUTPUT_FILEDATA_FLAG_OPEN;
211  } else {
212  /* existing file */
213  }
214 
215  /* if file needs to be closed or truncated, inform
216  * loggers */
217  if ((file_close || file_trunc) && ff->state < FILE_STATE_CLOSED) {
218  ff->state = FILE_STATE_TRUNCATED;
219  }
220 
221  /* for the last data chunk we have, also tell the logger
222  * we're closing up */
223  if (ffd->next == NULL && ff->state >= FILE_STATE_CLOSED)
224  flags |= OUTPUT_FILEDATA_FLAG_CLOSE;
225 
226  /* do the actual logging */
227  file_logged = CallLoggers(tv, store, p, ff, write_ffd, flags);
228 
229  if (file_logged) {
230  ffd->stored = 1;
231 
232  /* all done */
233  if (flags & OUTPUT_FILEDATA_FLAG_CLOSE) {
234  ff->flags |= FILE_STORED;
235  break;
236  }
237  }
238  }
239  }
240 
241  FilePrune(ffc);
242  }
243 
244  FLOWLOCK_UNLOCK(f);
245  return TM_ECODE_OK;
246 }

251  * \brief Open the waldo file (if available) and load the file_id
252  *
253  * \param path full path for the waldo file
254  */
255 static void LogFiledataLogLoadWaldo(const char *path)
256 {
257  char line[16] = "";
258  unsigned int id = 0;
259 
260  FILE *fp = fopen(path, "r");
261  if (fp == NULL) {
262  SCLogInfo("couldn't open waldo: %s", strerror(errno));
263  SCReturn;
264  }
265 
266  if (fgets(line, (int)sizeof(line), fp) != NULL) {
267  if (sscanf(line, "%10u", &id) == 1) {
268  SCLogInfo("id %u", id);
269  (void) SC_ATOMIC_CAS(&file_id, 0, id);
270  }
271  }
272  fclose(fp);
273 }

275 /**
276  * \internal
277  *
278  * \brief Store the waldo file based on the file_id
279  *
280  * \param path full path for the waldo file
281  */
282 static void LogFiledataLogStoreWaldo(const char *path)
283 {
284  char line[16] = "";
285 
286  if (SC_ATOMIC_GET(file_id) == 0) {
287  SCReturn;
288  }
289 
290  FILE *fp = fopen(path, "w");
291  if (fp == NULL) {
292  SCLogInfo("couldn't open waldo: %s", strerror(errno));
293  SCReturn;
294  }
295 
296  snprintf(line, sizeof(line), "%u\n", SC_ATOMIC_GET(file_id));
297  if (fwrite(line, strlen(line), 1, fp) != 1) {
298  SCLogError(SC_ERR_FWRITE, "fwrite failed: %s", strerror(errno));
299  }
300  fclose(fp);
301 }

303 /** \brief thread init for the tx logger
304  * This will run the thread init functions for the individual registered
305  * loggers */
306 static TmEcode OutputFiledataLogThreadInit(ThreadVars *tv, void *initdata, void **data)
307 {
308  OutputLoggerThreadData *td = SCMalloc(sizeof(*td));
309  if (td == NULL)
310  return TM_ECODE_FAILED;
311  memset(td, 0x00, sizeof(*td));
312 
313  *data = (void *)td;
314 
315  SCLogDebug("OutputFiledataLogThreadInit happy (*data %p)", *data);
316 
317  OutputFiledataLogger *logger = list;
318  while (logger) {
319  TmModule *tm_module = TmModuleGetByName((char *)logger->name);
320  if (tm_module == NULL) {
321  SCLogError(SC_ERR_INVALID_ARGUMENT,
322  "TmModuleGetByName for %s failed", logger->name);
323  exit(EXIT_FAILURE);
324  }
325 
326  if (tm_module->ThreadInit) {
327  void *retptr = NULL;
328  if (tm_module->ThreadInit(tv, (void *)logger->output_ctx, &retptr) == TM_ECODE_OK) {
329  OutputLoggerThreadStore *ts = SCMalloc(sizeof(*ts));
330 /* todo */ BUG_ON(ts == NULL);
331  memset(ts, 0x00, sizeof(*ts));
332 
333  /* store thread handle */
334  ts->thread_data = retptr;
335 
336  if (td->store == NULL) {
337  td->store = ts;
338  } else {
339  OutputLoggerThreadStore *tmp = td->store;
340  while (tmp->next != NULL)
341  tmp = tmp->next;
342  tmp->next = ts;
343  }
344 
345  SCLogDebug("%s is now set up", logger->name);
346  }
347  }
348 
349  logger = logger->next;
350  }
351 
352  SCMutexLock(&g_waldo_mutex);
353  if (g_waldo_init == 0) {
354  ConfNode *node = ConfGetNode("file-store-waldo");
355  if (node == NULL) {
356  ConfNode *outputs = ConfGetNode("outputs");
357  if (outputs) {
358  ConfNode *output;
359  TAILQ_FOREACH(output, &outputs->head, next) {
360  /* we only care about file and file-store */
361  if (!(strcmp(output->val, "file") == 0 || strcmp(output->val, "file-store") == 0))
362  continue;
363 
364  ConfNode *file = ConfNodeLookupChild(output, output->val);
365  BUG_ON(file == NULL);
366  if (file == NULL) {
367  SCLogDebug("file-store failed, lets try 'file'");
368  file = ConfNodeLookupChild(outputs, "file");
369  if (file == NULL)
370  SCLogDebug("file failed as well, giving up");
371  }
372 
373  if (file != NULL) {
374  node = ConfNodeLookupChild(file, "waldo");
375  if (node == NULL)
376  SCLogDebug("no waldo node");
377  }
378  }
379  }
380  }
381  if (node != NULL) {
382  char *s_default_log_dir = NULL;
383  s_default_log_dir = ConfigGetLogDirectory();
384 
385  const char *waldo = node->val;
386  SCLogDebug("loading waldo %s", waldo);
387  if (waldo != NULL && strlen(waldo) > 0) {
388  if (PathIsAbsolute(waldo)) {
389  snprintf(g_waldo, sizeof(g_waldo), "%s", waldo);
390  } else {
391  snprintf(g_waldo, sizeof(g_waldo), "%s/%s", s_default_log_dir, waldo);
392  }
393 
394  SCLogDebug("loading waldo file %s", g_waldo);
395  LogFiledataLogLoadWaldo(g_waldo);
396  }
397  }
398  g_waldo_init = 1;
399  }
400  SCMutexUnlock(&g_waldo_mutex);
401  return TM_ECODE_OK;
402 }

421 
422  OutputLoggerThreadStore *next_store = store->next;
423  SCFree(store);
424  store = next_store;
425  logger = logger->next;
426  }
427 
428  SCMutexLock(&g_waldo_mutex);
429  if (g_waldo_deinit == 0) {
430  if (strlen(g_waldo) > 0) {
431  SCLogDebug("we have a waldo at %s", g_waldo);
432  LogFiledataLogStoreWaldo(g_waldo);
433  }
434  g_waldo_deinit = 1;
435  }
436  SCMutexUnlock(&g_waldo_mutex);
437  return TM_ECODE_OK;
438 }
