65 /**
66  * \internal
67  * \brief Pseudo packet setup for flow forced reassembly.
68  *
69  * \param direction Direction of the packet. 0 indicates toserver and 1
70  * indicates toclient.
71  * \param f Pointer to the flow.
72  * \param ssn Pointer to the tcp session.
73  * \param dummy Indicates to create a dummy pseudo packet. Not all pseudo
74  * packets need to force reassembly, in which case we just
75  * set dummy ack/seq values.
76  */
77 static inline Packet *FlowForceReassemblyPseudoPacketSetup(Packet *p,
78  int direction,
79  Flow *f,
80  TcpSession *ssn,
81  int dummy)
82 {
83  p->tenant_id = f->tenant_id;
84  p->datalink = DLT_RAW;
85  p->proto = IPPROTO_TCP;
86  FlowReference(&p->flow, f);
87  p->flags |= PKT_STREAM_EST;
88  p->flags |= PKT_STREAM_EOF;
89  p->flags |= PKT_HAS_FLOW;
90  p->flags |= PKT_PSEUDO_STREAM_END;
91 
92  if (f->flags & FLOW_NOPACKET_INSPECTION) {
93  DecodeSetNoPacketInspectionFlag(p);
94  }
95  if (f->flags & FLOW_NOPAYLOAD_INSPECTION) {
96  DecodeSetNoPayloadInspectionFlag(p);
97  }
98 
99  if (direction == 0)
100  p->flowflags |= FLOW_PKT_TOSERVER;
101  else
102  p->flowflags |= FLOW_PKT_TOCLIENT;
103  p->flowflags |= FLOW_PKT_ESTABLISHED;
104  p->payload = NULL;
105  p->payload_len = 0;
106 
107  if (FLOW_IS_IPV4(f)) {
108  if (direction == 0) {
109  FLOW_COPY_IPV4_ADDR_TO_PACKET(&f->src, &p->src);
110  FLOW_COPY_IPV4_ADDR_TO_PACKET(&f->dst, &p->dst);
111  p->sp = f->sp;
112  p->dp = f->dp;
113  } else {
114  FLOW_COPY_IPV4_ADDR_TO_PACKET(&f->src, &p->dst);
115  FLOW_COPY_IPV4_ADDR_TO_PACKET(&f->dst, &p->src);
116  p->sp = f->dp;
117  p->dp = f->sp;
118  }
119 
120  /* Check if we have enough room in direct data. We need ipv4 hdr + tcp hdr.
121  * Force an allocation if it is not the case.
122  */
123  if (GET_PKT_DIRECT_MAX_SIZE(p) < 40) {
124  if (PacketCallocExtPkt(p, 40) == -1) {
125  return NULL;
126  }
127  }
128  /* set the ip header */
129  p->ip4h = (IPV4Hdr *)GET_PKT_DATA(p);
130  /* version 4 and length 20 bytes for the tcp header */
131  p->ip4h->ip_verhl = 0x45;
132  p->ip4h->ip_tos = 0;
133  p->ip4h->ip_len = htons(40);
134  p->ip4h->ip_id = 0;
135  p->ip4h->ip_off = 0;
136  p->ip4h->ip_ttl = 64;
137  p->ip4h->ip_proto = IPPROTO_TCP;
138  //p->ip4h->ip_csum =
139  if (direction == 0) {
140  p->ip4h->s_ip_src.s_addr = f->src.addr_data32[0];
141  p->ip4h->s_ip_dst.s_addr = f->dst.addr_data32[0];
142  } else {
143  p->ip4h->s_ip_src.s_addr = f->dst.addr_data32[0];
144  p->ip4h->s_ip_dst.s_addr = f->src.addr_data32[0];
145  }
146 
147  /* set the tcp header */
148  p->tcph = (TCPHdr *)((uint8_t *)GET_PKT_DATA(p) + 20);
149 
150  SET_PKT_LEN(p, 40); /* ipv4 hdr + tcp hdr */
151 
152  } else if (FLOW_IS_IPV6(f)) {
153  if (direction == 0) {
154  FLOW_COPY_IPV6_ADDR_TO_PACKET(&f->src, &p->src);
155  FLOW_COPY_IPV6_ADDR_TO_PACKET(&f->dst, &p->dst);
156  p->sp = f->sp;
157  p->dp = f->dp;
158  } else {
159  FLOW_COPY_IPV6_ADDR_TO_PACKET(&f->src, &p->dst);
160  FLOW_COPY_IPV6_ADDR_TO_PACKET(&f->dst, &p->src);
161  p->sp = f->dp;
162  p->dp = f->sp;
163  }
164 
165  /* Check if we have enough room in direct data. We need ipv6 hdr + tcp hdr.
166  * Force an allocation if it is not the case.
167  */
168  if (GET_PKT_DIRECT_MAX_SIZE(p) < 60) {
169  if (PacketCallocExtPkt(p, 60) == -1) {
170  return NULL;
171  }
172  }
173  /* set the ip header */
174  p->ip6h = (IPV6Hdr *)GET_PKT_DATA(p);
175  /* version 6 */
176  p->ip6h->s_ip6_vfc = 0x60;
177  p->ip6h->s_ip6_flow = 0;
178  p->ip6h->s_ip6_nxt = IPPROTO_TCP;
179  p->ip6h->s_ip6_plen = htons(20);
180  p->ip6h->s_ip6_hlim = 64;
181  if (direction == 0) {
182  p->ip6h->s_ip6_src[0] = f->src.addr_data32[0];
183  p->ip6h->s_ip6_src[1] = f->src.addr_data32[1];
184  p->ip6h->s_ip6_src[2] = f->src.addr_data32[2];
185  p->ip6h->s_ip6_src[3] = f->src.addr_data32[3];
186  p->ip6h->s_ip6_dst[0] = f->dst.addr_data32[0];
187  p->ip6h->s_ip6_dst[1] = f->dst.addr_data32[1];
188  p->ip6h->s_ip6_dst[2] = f->dst.addr_data32[2];
189  p->ip6h->s_ip6_dst[3] = f->dst.addr_data32[3];
190  } else {
191  p->ip6h->s_ip6_src[0] = f->dst.addr_data32[0];
192  p->ip6h->s_ip6_src[1] = f->dst.addr_data32[1];
193  p->ip6h->s_ip6_src[2] = f->dst.addr_data32[2];
194  p->ip6h->s_ip6_src[3] = f->dst.addr_data32[3];
195  p->ip6h->s_ip6_dst[0] = f->src.addr_data32[0];
196  p->ip6h->s_ip6_dst[1] = f->src.addr_data32[1];
197  p->ip6h->s_ip6_dst[2] = f->src.addr_data32[2];
198  p->ip6h->s_ip6_dst[3] = f->src.addr_data32[3];
199  }
200 
201  /* set the tcp header */
202  p->tcph = (TCPHdr *)((uint8_t *)GET_PKT_DATA(p) + 40);
203 
204  SET_PKT_LEN(p, 60); /* ipv6 hdr + tcp hdr */
205  }
206 
207  p->tcph->th_offx2 = 0x50;
208  p->tcph->th_flags |= TH_ACK;
209  p->tcph->th_win = 10;
210  p->tcph->th_urp = 0;
211 
212  /* to server */
213  if (direction == 0) {
214  p->tcph->th_sport = htons(f->sp);
215  p->tcph->th_dport = htons(f->dp);
216 
217  if (dummy) {
218  p->tcph->th_seq = htonl(ssn->client.next_seq);
219  p->tcph->th_ack = htonl(ssn->server.last_ack);
220  } else {
221  p->tcph->th_seq = htonl(ssn->client.next_seq);
222  p->tcph->th_ack = htonl(ssn->server.seg_list_tail->seq +
223  ssn->server.seg_list_tail->payload_len);
224  }
225 
226  /* to client */
227  } else {
228  p->tcph->th_sport = htons(f->dp);
229  p->tcph->th_dport = htons(f->sp);
230 
231  if (dummy) {
232  p->tcph->th_seq = htonl(ssn->server.next_seq);
233  p->tcph->th_ack = htonl(ssn->client.last_ack);
234  } else {
235  p->tcph->th_seq = htonl(ssn->server.next_seq);
236  p->tcph->th_ack = htonl(ssn->client.seg_list_tail->seq +
237  ssn->client.seg_list_tail->payload_len);
238  }
239  }
240 
241  if (FLOW_IS_IPV4(f)) {
242  p->tcph->th_sum = TCPCalculateChecksum(p->ip4h->s_ip_addrs,
243  (uint16_t *)p->tcph, 20);
244  /* calc ipv4 csum as we may log it and barnyard might reject
245  * a wrong checksum */
246  p->ip4h->ip_csum = IPV4CalculateChecksum((uint16_t *)p->ip4h,
247  IPV4_GET_RAW_HLEN(p->ip4h));
248  } else if (FLOW_IS_IPV6(f)) {
249  p->tcph->th_sum = TCPCalculateChecksum(p->ip6h->s_ip6_addrs,
250  (uint16_t *)p->tcph, 20);
251  }
252 
253  memset(&p->ts, 0, sizeof(struct timeval));
254  TimeGet(&p->ts);
255 
256  AppLayerParserSetEOF(f->alparser);
257 
258  return p;
259 }

277 /**
278  * \brief Check if a flow needs forced reassembly, or any other processing
279  *
280  * \param f *LOCKED* flow
281  * \param server ptr to int that should be set to 1 or 2 if we return 1
282  * \param client ptr to int that should be set to 1 or 2 if we return 1
283  *
284  * \retval 0 no
285  * \retval 1 yes
286  */
287 int FlowForceReassemblyNeedReassembly(Flow *f, int *server, int *client)
288 {
289  TcpSession *ssn;
290 
291  if (f == NULL) {
292  *server = *client = STREAM_HAS_UNPROCESSED_SEGMENTS_NONE;
293  SCReturnInt(0);
294  }
295 
296  /* Get the tcp session for the flow */
297  ssn = (TcpSession *)f->protoctx;
298  if (ssn == NULL) {
299  *server = *client = STREAM_HAS_UNPROCESSED_SEGMENTS_NONE;
300  SCReturnInt(0);
301  }
302 
303  *client = StreamNeedsReassembly(ssn, 0);
304  *server = StreamNeedsReassembly(ssn, 1);
305 
306  /* if state is not fully closed we assume that we haven't fully
307  * inspected the app layer state yet */
308  if (ssn->state >= TCP_ESTABLISHED && ssn->state != TCP_CLOSED)
309  {
310  if (*client != STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_REASSEMBLY)
311  *client = STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_ONLY_DETECTION;
312 
313  if (*server != STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_REASSEMBLY)
314  *server = STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_ONLY_DETECTION;
315  }
316 
317  /* if app layer still needs some love, push through */
318  if (f->alproto != ALPROTO_UNKNOWN && f->alstate != NULL &&
319  AppLayerParserProtocolSupportsTxs(f->proto, f->alproto))
320  {
321  uint64_t total_txs = AppLayerParserGetTxCnt(f->proto, f->alproto, f->alstate);
322 
323  if (AppLayerParserGetTransactionActive(f->proto, f->alproto,
324  f->alparser, STREAM_TOCLIENT) < total_txs)
325  {
326  if (*server != STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_REASSEMBLY)
327  *server = STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_ONLY_DETECTION;
328  }
329  if (AppLayerParserGetTransactionActive(f->proto, f->alproto,
330  f->alparser, STREAM_TOSERVER) < total_txs)
331  {
332  if (*client != STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_REASSEMBLY)
333  *client = STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_ONLY_DETECTION;
334  }
335  }
336 
337  /* nothing to do */
338  if (*client == STREAM_HAS_UNPROCESSED_SEGMENTS_NONE &&
339  *server == STREAM_HAS_UNPROCESSED_SEGMENTS_NONE) {
340  SCReturnInt(0);
341  }
342 
343  SCReturnInt(1);
344 }

346 /**
347  * \internal
348  * \brief Forces reassembly for flow if it needs it.
349  *
350  * The function requires flow to be locked beforehand.
351  *
352  * \param f Pointer to the flow.
353  * \param server action required for server: 1 or 2
354  * \param client action required for client: 1 or 2
355  *
356  * \retval 0 This flow doesn't need any reassembly processing; 1 otherwise.
357  */
358 int FlowForceReassemblyForFlow(Flow *f, int server, int client)
359 {
360  Packet *p1 = NULL, *p2 = NULL, *p3 = NULL;
361  TcpSession *ssn;
362 
363  /* looks like we have no flows in this queue */
364  if (f == NULL) {
365  return 0;
366  }
367 
368  /* Get the tcp session for the flow */
369  ssn = (TcpSession *)f->protoctx;
370  if (ssn == NULL) {
371  return 0;
372  }
373 
374  /* The packets we use are based on what segments in what direction are
375  * unprocessed.
376  * p1 if we have client segments for reassembly purpose only. If we
377  * have no server segments p2 can be a toserver packet with dummy
378  * seq/ack, and if we have server segments p2 has to carry out reassembly
379  * for server segment as well, in which case we will also need a p3 in the
380  * toclient which is now dummy since all we need it for is detection */
381 
382  /* insert a pseudo packet in the toserver direction */
383  if (client == STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_REASSEMBLY) {
384  p1 = FlowForceReassemblyPseudoPacketGet(1, f, ssn, 0);
385  if (p1 == NULL) {
386  goto done;
387  }
388  PKT_SET_SRC(p1, PKT_SRC_FFR);
389 
390  if (server == STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_REASSEMBLY) {
391  p2 = FlowForceReassemblyPseudoPacketGet(0, f, ssn, 0);
392  if (p2 == NULL) {
393  FlowDeReference(&p1->flow);
394  TmqhOutputPacketpool(NULL, p1);
395  goto done;
396  }
397  PKT_SET_SRC(p2, PKT_SRC_FFR);
398 
399  p3 = FlowForceReassemblyPseudoPacketGet(1, f, ssn, 1);
400  if (p3 == NULL) {
401  FlowDeReference(&p1->flow);
402  TmqhOutputPacketpool(NULL, p1);
403  FlowDeReference(&p2->flow);
404  TmqhOutputPacketpool(NULL, p2);
405  goto done;
406  }
407  PKT_SET_SRC(p3, PKT_SRC_FFR);
408  } else {
409  p2 = FlowForceReassemblyPseudoPacketGet(0, f, ssn, 1);
410  if (p2 == NULL) {
411  FlowDeReference(&p1->flow);
412  TmqhOutputPacketpool(NULL, p1);
413  goto done;
414  }
415  PKT_SET_SRC(p2, PKT_SRC_FFR);
416  }
417 
418  } else if (client == STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_ONLY_DETECTION) {
419  if (server == STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_REASSEMBLY) {
420  p1 = FlowForceReassemblyPseudoPacketGet(0, f, ssn, 0);
421  if (p1 == NULL) {
422  goto done;
423  }
424  PKT_SET_SRC(p1, PKT_SRC_FFR);
425 
426  p2 = FlowForceReassemblyPseudoPacketGet(1, f, ssn, 1);
427  if (p2 == NULL) {
428  FlowDeReference(&p1->flow);
429  TmqhOutputPacketpool(NULL, p1);
430  goto done;
431  }
432  PKT_SET_SRC(p2, PKT_SRC_FFR);
433  } else {
434  p1 = FlowForceReassemblyPseudoPacketGet(0, f, ssn, 1);
435  if (p1 == NULL) {
436  goto done;
437  }
438  PKT_SET_SRC(p1, PKT_SRC_FFR);
439 
440  if (server == STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_ONLY_DETECTION) {
441  p2 = FlowForceReassemblyPseudoPacketGet(1, f, ssn, 1);
442  if (p2 == NULL) {
443  FlowDeReference(&p1->flow);
444  TmqhOutputPacketpool(NULL, p1);
445  goto done;
446  }
447  PKT_SET_SRC(p2, PKT_SRC_FFR);
448  }
449  }
450 
451  } else {
452  if (server == STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_REASSEMBLY) {
453  p1 = FlowForceReassemblyPseudoPacketGet(0, f, ssn, 0);
454  if (p1 == NULL) {
455  goto done;
456  }
457  PKT_SET_SRC(p1, PKT_SRC_FFR);
458 
459  p2 = FlowForceReassemblyPseudoPacketGet(1, f, ssn, 1);
460  if (p2 == NULL) {
461  FlowDeReference(&p1->flow);
462  TmqhOutputPacketpool(NULL, p1);
463  goto done;
464  }
465  PKT_SET_SRC(p2, PKT_SRC_FFR);
466  } else if (server == STREAM_HAS_UNPROCESSED_SEGMENTS_NEED_ONLY_DETECTION) {
467  p1 = FlowForceReassemblyPseudoPacketGet(1, f, ssn, 1);
468  if (p1 == NULL) {
469  goto done;
470  }
471  PKT_SET_SRC(p1, PKT_SRC_FFR);
472  } else {
473  /* impossible */
474  BUG_ON(1);
475  }
476  }
477 
478  /* inject the packet(s) into the appropriate thread */
479  int thread_id = (int)f->thread_id;
480  Packet *packets[4] = { p1, p2 ? p2 : p3, p2 ? p3 : NULL, NULL }; /**< null terminated array of packets */
481  if (unlikely(!(TmThreadsInjectPacketsById(packets, thread_id)))) {
482  FlowDeReference(&p1->flow);
483  TmqhOutputPacketpool(NULL, p1);
484  if (p2) {
485  FlowDeReference(&p2->flow);
486  TmqhOutputPacketpool(NULL, p2);
487  }
488  if (p3) {
489  FlowDeReference(&p3->flow);
490  TmqhOutputPacketpool(NULL, p3);
491  }
492  }
493 
494  /* done, in case of error (no packet) we still tag flow as complete
495  * as we're probably resource stress if we couldn't get packets */
496 done:
497  f->flags |= FLOW_TIMEOUT_REASSEMBLY_DONE;
498  return 1;
499 }

501 /**
502  * \internal
503  * \brief Forces reassembly for flows that need it.
504  *
505  * When this function is called we're running in virtually dead engine,
506  * so locking the flows is not strictly required. The reasons it is still
507  * done are:
508  * - code consistency
509  * - silence complaining profilers
510  * - allow us to aggressively check using debug valdation assertions
511  * - be robust in case of future changes
512  * - locking overhead if neglectable when no other thread fights us
513  *
514  * \param q The queue to process flows from.
515  */
516 static inline void FlowForceReassemblyForHash(void)
517 {
518  Flow *f;
519  TcpSession *ssn;
520  int client_ok = 0;
521  int server_ok = 0;
522  uint32_t idx = 0;
523 
524  for (idx = 0; idx < flow_config.hash_size; idx++) {
525  FlowBucket *fb = &flow_hash[idx];
526 
527  PacketPoolWaitForN(9);
528  FBLOCK_LOCK(fb);
529 
530  /* get the topmost flow from the QUEUE */
531  f = fb->head;
532 
533  /* we need to loop through all the flows in the queue */
534  while (f != NULL) {
535  PacketPoolWaitForN(3);
536 
537  FLOWLOCK_WRLOCK(f);
538 
539  /* Get the tcp session for the flow */
540  ssn = (TcpSession *)f->protoctx;
541 
542  /* \todo Also skip flows that shouldn't be inspected */
543  if (ssn == NULL) {
544  FLOWLOCK_UNLOCK(f);
545  f = f->hnext;
546  continue;
547  }
548 
549  if (FlowForceReassemblyNeedReassembly(f, &server_ok, &client_ok) == 1) {
550  FlowForceReassemblyForFlow(f, server_ok, client_ok);
551  }
552 
553  FLOWLOCK_UNLOCK(f);
554 
555  /* next flow in the queue */
556  f = f->hnext;
557  }
558  FBLOCK_UNLOCK(fb);
559  }
560  return;
561 }
