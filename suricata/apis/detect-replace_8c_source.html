76 int DetectReplaceSetup(DetectEngineCtx *de_ctx, Signature *s, char *replacestr)
77 {
78  uint8_t *content = NULL;
79  uint16_t len = 0;
80  uint32_t flags = 0;
81  SigMatch *pm = NULL;
82  DetectContentData *ud = NULL;
83 
84  int ret = DetectContentDataParse("replace", replacestr, &content, &len, &flags);
85  if (ret == -1)
86  goto error;
87 
88  if (flags & DETECT_CONTENT_NEGATED) {
89  SCLogError(SC_ERR_INVALID_VALUE, "Can't negate replacement string: %s",
90  replacestr);
91  goto error;
92  }
93 
94  switch (run_mode) {
95  case RUNMODE_NFQ:
96  case RUNMODE_IPFW:
97  break;
98  default:
99  SCLogWarning(SC_ERR_RUNMODE,
100  "Can't use 'replace' keyword in non IPS mode: %s",
101  s->sig_str);
102  /* this is a success, having the alert is interesting */
103  return 0;
104  }
105 
106  /* add to the latest "content" keyword from either dmatch or pmatch */
107  pm = SigMatchGetLastSMFromLists(s, 2,
108  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_PMATCH]);
109  if (pm == NULL) {
110  SCLogError(SC_ERR_WITHIN_MISSING_CONTENT, "replace needs"
111  "preceding content option for raw sig");
112  SCFree(content);
113  return -1;
114  }
115 
116  /* we can remove this switch now with the unified structure */
117  ud = (DetectContentData *)pm->ctx;
118  if (ud == NULL) {
119  SCLogError(SC_ERR_INVALID_ARGUMENT, "invalid argument");
120  SCFree(content);
121  return -1;
122  }
123  if (ud->flags & DETECT_CONTENT_NEGATED) {
124  SCLogError(SC_ERR_INVALID_SIGNATURE, "can't have a relative "
125  "negated keyword set along with a replacement");
126  goto error;
127  }
128  if (ud->content_len != len) {
129  SCLogError(SC_ERR_INVALID_SIGNATURE, "can't have a content "
130  "length different from replace length");
131  goto error;
132  }
133 
134  ud->replace = SCMalloc(len);
135  if (ud->replace == NULL) {
136  goto error;
137  }
138  memcpy(ud->replace, content, len);
139  ud->replace_len = len;
140  ud->flags |= DETECT_CONTENT_REPLACE;
141  /* want packet matching only won't be able to replace data with
142  * a flow.
143  */
144  s->flags |= SIG_FLAG_REQUIRE_PACKET;
145  SCFree(content);
146 
147  return 0;
148 
149 error:
150  SCFree(content);
151  return -1;
152 }
153 
154 /* Add to the head of the replace-list.
155  *
156  * The first to add to the replace-list has the highest priority. So,
157  * adding the the head of the list results in the newest modifications
158  * of content being applied first, so later changes can over ride
159  * earlier changes. Thus the highest priority modifications should be
160  * applied last.
161  */
162 DetectReplaceList *DetectReplaceAddToList(DetectReplaceList *replist,
163  uint8_t *found,
164  DetectContentData *cd)
165 {
166  DetectReplaceList *newlist;
167 
168  if (cd->content_len != cd->replace_len)
169  return NULL;
170  SCLogDebug("replace: Adding match");
171 
172  newlist = SCMalloc(sizeof(DetectReplaceList));
173  if (unlikely(newlist == NULL))
174  return replist;
175  newlist->found = found;
176  newlist->cd = cd;
177  /* Push new value onto the front of the list. */
178  newlist->next = replist;
179 
180  return newlist;
181 }

184 void DetectReplaceExecuteInternal(Packet *p, DetectReplaceList *replist)
185 {
186  DetectReplaceList *tlist = NULL;
187 
188  SCLogDebug("replace: Executing match");
189  while (replist) {
190  memcpy(replist->found, replist->cd->replace, replist->cd->replace_len);
191  SCLogDebug("replace: injecting '%s'", replist->cd->replace);
192  p->flags |= PKT_STREAM_MODIFIED;
193  ReCalculateChecksum(p);
194  tlist = replist;
195  replist = replist->next;
196  SCFree(tlist);
197  }
198 }

214 /**
215  * \test Test packet Matches
216  * \param raw_eth_pkt pointer to the ethernet packet
217  * \param pktsize size of the packet
218  * \param sig pointer to the signature to test
219  * \param sid sid number of the signature
220  * \retval return 1 if match
221  * \retval return 0 if not
222  */
223 static
224 int DetectReplaceLongPatternMatchTest(uint8_t *raw_eth_pkt, uint16_t pktsize,
225  char *sig, uint32_t sid, uint8_t *pp,
226  uint16_t *len)
227 {
228  int result = 0;
229 
230  Packet *p = NULL;
231  p = PacketGetFromAlloc();
232  if (unlikely(p == NULL))
233  return 0;
234 
235  DecodeThreadVars dtv;
236 
237  ThreadVars th_v;
238  DetectEngineThreadCtx *det_ctx = NULL;
239 
240  if (pp == NULL) {
241  SCLogDebug("replace: looks like a second run");
242  }
243 
244  PacketCopyData(p, raw_eth_pkt, pktsize);
245  memset(&dtv, 0, sizeof(DecodeThreadVars));
246  memset(&th_v, 0, sizeof(th_v));
247  dtv.app_tctx = AppLayerGetCtxThread(&th_v);
248 
249  FlowInitConfig(FLOW_QUIET);
250  DecodeEthernet(&th_v, &dtv, p, GET_PKT_DATA(p), pktsize, NULL);
251 
252  DetectEngineCtx *de_ctx = DetectEngineCtxInit();
253  if (de_ctx == NULL) {
254  goto end;
255  }
256  de_ctx->flags |= DE_QUIET;
257 
258  de_ctx->sig_list = SigInit(de_ctx, sig);
259  if (de_ctx->sig_list == NULL) {
260  goto end;
261  }
262  de_ctx->sig_list->next = NULL;
263 
264  if (de_ctx->sig_list->sm_lists_tail[DETECT_SM_LIST_PMATCH]->type == DETECT_CONTENT) {
265  DetectContentData *co = (DetectContentData *)de_ctx->sig_list->sm_lists_tail[DETECT_SM_LIST_PMATCH]->ctx;
266  if (co->flags & DETECT_CONTENT_RELATIVE_NEXT) {
267  printf("relative next flag set on final match which is content: ");
268  goto end;
269  }
270  }
271 
272  SigGroupBuild(de_ctx);
273  DetectEngineAddToMaster(de_ctx);
274  DetectEngineThreadCtxInit(&th_v, NULL, (void *)&det_ctx);
275 
276  SigMatchSignatures(&th_v, de_ctx, det_ctx, p);
277  DetectEngineMoveToFreeList(de_ctx);
278 
279  if (PacketAlertCheck(p, sid) != 1) {
280  SCLogDebug("replace: no alert on sig %d", sid);
281  goto end;
282  }
283 
284  if (pp) {
285  memcpy(pp, GET_PKT_DATA(p), GET_PKT_LEN(p));
286  *len = pktsize;
287  SCLogDebug("replace: copying %d on %p", *len, pp);
288  }
289 
290 
291  result = 1;
292 end:
293  if (dtv.app_tctx != NULL)
294  AppLayerDestroyCtxThread(dtv.app_tctx);
295  if (det_ctx != NULL)
296  DetectEngineThreadCtxDeinit(&th_v, (void *)det_ctx);
297  DetectEnginePruneFreeList();
298  PACKET_RECYCLE(p);
299  FlowShutdown();
300  SCFree(p);
301 
302 
303  return result;
304 }
