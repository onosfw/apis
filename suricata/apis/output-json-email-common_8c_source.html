90 static inline char *SkipWhiteSpaceTill(char *p, char *savep)
91 {
92  char *sp = p;
93  if (unlikely(p == NULL)) {
94  return NULL;
95  }
96  while (((*sp == '\t') || (*sp == ' ')) && (sp < savep)) {
97  sp++;
98  }
99  return sp;

102 static json_t* JsonEmailJsonArrayFromCommaList(const uint8_t *val, size_t len)
103 {
104  json_t *ajs = json_array();
105  if (likely(ajs != NULL)) {
106  char *savep = NULL;
107  char *p;
108  char *sp;
109  char *to_line = BytesToString((uint8_t *)val, len);
110  if (likely(to_line != NULL)) {
111  p = strtok_r(to_line, ",", &savep);
112  if (p == NULL) {
113  json_decref(ajs);
114  SCFree(to_line);
115  return NULL;
116  }
117  sp = SkipWhiteSpaceTill(p, savep);
118  json_array_append_new(ajs, json_string(sp));
119  while ((p = strtok_r(NULL, ",", &savep)) != NULL) {
120  sp = SkipWhiteSpaceTill(p, savep);
121  json_array_append_new(ajs, json_string(sp));
122  }
123  }
124  SCFree(to_line);
125  }
126 
127  return ajs;
128 }

132 static void JsonEmailLogJSONMd5(OutputJsonEmailCtx *email_ctx, json_t *js, SMTPTransaction *tx)
133 {
134  if (email_ctx->flags & LOG_EMAIL_SUBJECT_MD5) {
135  MimeDecField *field;
136  MimeDecEntity *entity = tx->msg_tail;
137  if (entity == NULL) {
138  return;
139  }
140  field = MimeDecFindField(entity, "subject");
141  if (field != NULL) {
142  unsigned char md5[MD5_LENGTH];
143  char smd5[256];
144  char *value = BytesToString((uint8_t *)field->value , field->value_len);
145  if (value) {
146  size_t i,x;
147  HASH_HashBuf(HASH_AlgMD5, md5, (unsigned char *)value, strlen(value));
148  for (i = 0, x = 0; x < sizeof(md5); x++) {
149  i += snprintf(smd5 + i, 255 - i, "%02x", md5[x]);
150  }
151  json_object_set_new(js, "subject_md5", json_string(smd5));
152  SCFree(value);
153  }
154  }
155  }
156 
157  if (email_ctx->flags & LOG_EMAIL_BODY_MD5) {
158  MimeDecParseState *mime_state = tx->mime_state;
159  if (mime_state && mime_state->md5_ctx && (mime_state->state_flag == PARSE_DONE)) {
160  size_t x;
161  int i;
162  char s[256];
163  if (likely(s != NULL)) {
164  for (i = 0, x = 0; x < sizeof(mime_state->md5); x++) {
165  i += snprintf(s + i, 255-i, "%02x", mime_state->md5[x]);
166  }
167  json_object_set_new(js, "body_md5", json_string(s));
168  }
169  }
170  }
171 }

186 static void JsonEmailLogJSONCustom(OutputJsonEmailCtx *email_ctx, json_t *js, SMTPTransaction *tx)
187 {
188  int f = 0;
189  MimeDecField *field;
190  MimeDecEntity *entity = tx->msg_tail;
191  if (entity == NULL) {
192  return;
193  }
194 
195  while(email_fields[f].config_field) {
196  if (((email_ctx->fields & (1ULL<<f)) != 0)
197  ||
198  ((email_ctx->flags & LOG_EMAIL_EXTENDED) && (email_fields[f].flags & LOG_EMAIL_EXTENDED))
199  ) {
200  if (email_fields[f].flags & LOG_EMAIL_ARRAY) {
201  json_t *ajs = json_array();
202  if (ajs) {
203  int found = MimeDecFindFieldsForEach(entity, email_fields[f].email_field, JsonEmailAddToJsonArray, ajs);
204  if (found > 0) {
205  json_object_set_new(js, email_fields[f].config_field, ajs);
206  } else {
207  json_decref(ajs);
208  }
209  }
210  } else if (email_fields[f].flags & LOG_EMAIL_COMMA) {
211  field = MimeDecFindField(entity, email_fields[f].email_field);
212  if (field) {
213  json_t *ajs = JsonEmailJsonArrayFromCommaList(field->value, field->value_len);
214  if (ajs) {
215  json_object_set_new(js, email_fields[f].config_field, ajs);
216  }
217  }
218  } else {
219  field = MimeDecFindField(entity, email_fields[f].email_field);
220  if (field != NULL) {
221  char *s = BytesToString((uint8_t *)field->value,
222  (size_t)field->value_len);
223  if (likely(s != NULL)) {
224  json_object_set_new(js, email_fields[f].config_field, json_string(s));
225  SCFree(s);
226  }
227  }
228  }
229 
230  }
231  f++;
232  }
233 }

235 /* JSON format logging */
236 json_t *JsonEmailLogJsonData(const Flow *f, void *state, void *vtx, uint64_t tx_id)
237 {
238  SMTPState *smtp_state;
239  MimeDecParseState *mime_state;
240  MimeDecEntity *entity;
241 
242  json_t *sjs = json_object();
243  if (sjs == NULL) {
244  SCReturnPtr(NULL, "json_t");
245  }
246 
247  /* check if we have SMTP state or not */
248  AppProto proto = FlowGetAppProtocol(f);
249  switch (proto) {
250  case ALPROTO_SMTP:
251  smtp_state = (SMTPState *)state;
252  if (smtp_state == NULL) {
253  SCLogDebug("no smtp state, so no request logging");
254  SCReturnPtr(NULL, "json_t");
255  }
256  SMTPTransaction *tx = vtx;
257  mime_state = tx->mime_state;
258  entity = tx->msg_tail;
259  SCLogDebug("lets go mime_state %p, entity %p, state_flag %u", mime_state, entity, mime_state ? mime_state->state_flag : 0);
260  break;
261  default:
262  /* don't know how we got here */
263  SCReturnPtr(NULL, "json_t");
264  }
265  if ((mime_state != NULL)) {
266  if (entity == NULL) {
267  SCReturnPtr(NULL, "json_t");
268  }
269 
270  json_object_set_new(sjs, "status",
271  json_string(MimeDecParseStateGetStatus(mime_state)));
272 
273  MimeDecField *field;
274 
275  /* From: */
276  field = MimeDecFindField(entity, "from");
277  if (field != NULL) {
278  char *s = BytesToString((uint8_t *)field->value,
279  (size_t)field->value_len);
280  if (likely(s != NULL)) {
281  //printf("From: \"%s\"\n", s);
282  char * sp = SkipWhiteSpaceTill(s, s + strlen(s));
283  json_object_set_new(sjs, "from", json_string(sp));
284  SCFree(s);
285  }
286  }
287 
288  /* To: */
289  field = MimeDecFindField(entity, "to");
290  if (field != NULL) {
291  json_t *ajs = JsonEmailJsonArrayFromCommaList(field->value, field->value_len);
292  if (ajs) {
293  json_object_set_new(sjs, "to", ajs);
294  }
295  }
296 
297  /* Cc: */
298  field = MimeDecFindField(entity, "cc");
299  if (field != NULL) {
300  json_t *ajs = JsonEmailJsonArrayFromCommaList(field->value, field->value_len);
301  if (ajs) {
302  json_object_set_new(sjs, "cc", ajs);
303  }
304  }
305 
306  if (mime_state->stack == NULL || mime_state->stack->top == NULL || mime_state->stack->top->data == NULL)
307  SCReturnPtr(NULL, "json_t");
308 
309  entity = (MimeDecEntity *)mime_state->stack->top->data;
310  int attch_cnt = 0;
311  int url_cnt = 0;
312  json_t *js_attch = json_array();
313  json_t *js_url = json_array();
314  if (entity->url_list != NULL) {
315  MimeDecUrl *url;
316  for (url = entity->url_list; url != NULL; url = url->next) {
317  char *s = BytesToString((uint8_t *)url->url,
318  (size_t)url->url_len);
319  if (s != NULL) {
320  json_array_append_new(js_url,
321  json_string(s));
322  SCFree(s);
323  url_cnt += 1;
324  }
325  }
326  }
327  for (entity = entity->child; entity != NULL; entity = entity->next) {
328  if (entity->ctnt_flags & CTNT_IS_ATTACHMENT) {
329 
330  char *s = BytesToString((uint8_t *)entity->filename,
331  (size_t)entity->filename_len);
332  json_array_append_new(js_attch,
333  json_string(s));
334  SCFree(s);
335  attch_cnt += 1;
336  }
337  if (entity->url_list != NULL) {
338  MimeDecUrl *url;
339  for (url = entity->url_list; url != NULL; url = url->next) {
340  char *s = BytesToString((uint8_t *)url->url,
341  (size_t)url->url_len);
342  if (s != NULL) {
343  json_array_append_new(js_url,
344  json_string(s));
345  SCFree(s);
346  url_cnt += 1;
347  }
348  }
349  }
350  }
351  if (attch_cnt > 0) {
352  json_object_set_new(sjs, "attachment", js_attch);
353  } else {
354  json_decref(js_attch);
355  }
356  if (url_cnt > 0) {
357  json_object_set_new(sjs, "url", js_url);
358  } else {
359  json_decref(js_url);
360  }
361  SCReturnPtr(sjs, "json_t");
362  }
363 
364  json_decref(sjs);
365  SCReturnPtr(NULL, "json_t");
366 }

404 void OutputEmailInitConf(ConfNode *conf, OutputJsonEmailCtx *email_ctx)
405 {
406  if (conf) {
407  const char *extended = ConfNodeLookupChildValue(conf, "extended");
408 
409  if (extended != NULL) {
410  if (ConfValIsTrue(extended)) {
411  email_ctx->flags = LOG_EMAIL_EXTENDED;
412  }
413  }
414 
415  email_ctx->fields = 0;
416  ConfNode *custom;
417  if ((custom = ConfNodeLookupChild(conf, "custom")) != NULL) {
418  ConfNode *field;
419  TAILQ_FOREACH(field, &custom->head, next) {
420  if (field != NULL) {
421  int f = 0;
422  while(email_fields[f].config_field) {
423  if ((strcmp(email_fields[f].config_field,
424  field->val) == 0) ||
425  (strcasecmp(email_fields[f].email_field,
426  field->val) == 0))
427  {
428  email_ctx->fields |= (1ULL<<f);
429  break;
430  }
431  f++;
432  }
433  }
434  }
435  }
436 
437  email_ctx->flags = 0;
438  ConfNode *md5_conf;
439  if ((md5_conf = ConfNodeLookupChild(conf, "md5")) != NULL) {
440  ConfNode *field;
441  TAILQ_FOREACH(field, &md5_conf->head, next) {
442  if (field != NULL) {
443  if (strcmp("body", field->val) == 0) {
444  SCLogInfo("Going to log the md5 sum of email body");
445  email_ctx->flags |= LOG_EMAIL_BODY_MD5;
446  }
447  if (strcmp("subject", field->val) == 0) {
448  SCLogInfo("Going to log the md5 sum of email subject");
449  email_ctx->flags |= LOG_EMAIL_SUBJECT_MD5;
450  }
451  }
452  }
453  }
454  }
455  return;
456 }
