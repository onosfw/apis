47  * Regex
48  * fragbits: [!+*](MDR)
49  */
50 #define PARSE_REGEX "^\\s*(?:([\\+\\*!]))?\\s*([MDR]+)"

109 /**
110  * \internal
111  * \brief This function is used to match fragbits on a packet with those passed via fragbits:
112  *
113  * \param t pointer to thread vars
114  * \param det_ctx pointer to the pattern matcher thread
115  * \param p pointer to the current packet
116  * \param s pointer to the Signature
117  * \param m pointer to the sigmatch
118  *
119  * \retval 0 no match
120  * \retval 1 match
121  */
122 static int DetectFragBitsMatch (ThreadVars *t, DetectEngineThreadCtx *det_ctx, Packet *p, Signature *s, const SigMatchCtx *ctx)
123 {
124  int ret = 0;
125  uint16_t fragbits = 0;
126  const DetectFragBitsData *de = (const DetectFragBitsData *)ctx;
127 
128  if (!de || !PKT_IS_IPV4(p) || !p || PKT_IS_PSEUDOPKT(p))
129  return ret;
130 
131  if(IPV4_GET_MF(p))
132  fragbits |= FRAGBITS_HAVE_MF;
133  if(IPV4_GET_DF(p))
134  fragbits |= FRAGBITS_HAVE_DF;
135  if(IPV4_GET_RF(p))
136  fragbits |= FRAGBITS_HAVE_RF;
137 
138  switch(de->modifier) {
139  case MODIFIER_ANY:
140  if((fragbits & de->fragbits) > 0)
141  return 1;
142  return ret;
143  case MODIFIER_PLUS:
144  if(((fragbits & de->fragbits) == de->fragbits) && (((fragbits - de->fragbits) > 0)))
145  return 1;
146  return ret;
147  case MODIFIER_NOT:
148  if((fragbits & de->fragbits) != de->fragbits)
149  return 1;
150  return ret;
151  default:
152  if(fragbits == de->fragbits)
153  return 1;
154  }
155 
156  return ret;
157 }

159 /**
160  * \internal
161  * \brief This function is used to parse fragbits options passed via fragbits: keyword
162  *
163  * \param rawstr Pointer to the user provided fragbits options
164  *
165  * \retval de pointer to DetectFragBitsData on success
166  * \retval NULL on failure
167  */
168 static DetectFragBitsData *DetectFragBitsParse (char *rawstr)
169 {
170  DetectFragBitsData *de = NULL;
171 #define MAX_SUBSTRINGS 30
172  int ret = 0, found = 0, res = 0;
173  int ov[MAX_SUBSTRINGS];
174  const char *str_ptr = NULL;
175  char *args[2] = { NULL, NULL};
176  char *ptr;
177  int i;
178 
179  ret = pcre_exec(parse_regex, parse_regex_study, rawstr, strlen(rawstr), 0, 0, ov, MAX_SUBSTRINGS);
180 
181  if (ret < 1) {
182  SCLogError(SC_ERR_PCRE_MATCH, "pcre_exec parse error, ret %" PRId32 ", string %s", ret, rawstr);
183  goto error;
184  }
185 
186  for (i = 0; i < (ret - 1); i++) {
187 
188  res = pcre_get_substring((char *)rawstr, ov, MAX_SUBSTRINGS,i + 1, &str_ptr);
189 
190  if (res < 0) {
191  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
192  goto error;
193  }
194 
195  args[i] = (char *)str_ptr;
196  }
197 
198  if(args[1] == NULL) {
199  SCLogError(SC_ERR_INVALID_VALUE, "invalid value");
200  goto error;
201  }
202 
203  de = SCMalloc(sizeof(DetectFragBitsData));
204  if (unlikely(de == NULL))
205  goto error;
206 
207  memset(de,0,sizeof(DetectFragBitsData));
208 
209  /** First parse args[0] */
210 
211  if(args[0]) {
212 
213  ptr = args[0];
214 
215  while (*ptr != '\0') {
216  switch (*ptr) {
217  case '!':
218  de->modifier = MODIFIER_NOT;
219  break;
220  case '+':
221  de->modifier = MODIFIER_PLUS;
222  break;
223  case '*':
224  de->modifier = MODIFIER_ANY;
225  break;
226  }
227  ptr++;
228  }
229 
230  }
231 
232  /** Second parse first set of fragbits */
233 
234  ptr = args[1];
235 
236  while (*ptr != '\0') {
237  switch (*ptr) {
238  case 'M':
239  case 'm':
240  de->fragbits |= FRAGBITS_HAVE_MF;
241  found++;
242  break;
243  case 'D':
244  case 'd':
245  de->fragbits |= FRAGBITS_HAVE_DF;
246  found++;
247  break;
248  case 'R':
249  case 'r':
250  de->fragbits |= FRAGBITS_HAVE_RF;
251  found++;
252  break;
253  default:
254  found = 0;
255  break;
256  }
257  ptr++;
258  }
259 
260  if(found == 0)
261  goto error;
262 
263  for (i = 0; i < 2; i++) {
264  if (args[i] != NULL)
265  SCFree(args[i]);
266  }
267  return de;
268 
269 error:
270  for (i = 0; i < 2; i++) {
271  if (args[i] != NULL)
272  SCFree(args[i]);
273  }
274  if (de != NULL)
275  SCFree(de);
276  return NULL;
277 }

282  *
283  * \param de_ctx pointer to the Detection Engine Context
284  * \param s pointer to the Current Signature

289  * \retval -1 on Failure

291 static int DetectFragBitsSetup (DetectEngineCtx *de_ctx, Signature *s, char *rawstr)
292 {
293  DetectFragBitsData *de = NULL;
294  SigMatch *sm = NULL;
295 
296  de = DetectFragBitsParse(rawstr);
297  if (de == NULL)

300  sm = SigMatchAlloc();

302  goto error;
303

305  sm->ctx = (SigMatchCtx *)de;
306 
307  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);

309 
310  return 0;
311 
312 error:

314  if (sm) SCFree(sm);
315  return -1;
316 }
