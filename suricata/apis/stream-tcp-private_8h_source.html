123 /** 'Raw' reassembly is disabled for this ssn. */
124 #define STREAMTCP_FLAG_DISABLE_RAW 0x0020
125 /** Flag to indicate that the session is handling asynchronous stream.*/
126 #define STREAMTCP_FLAG_ASYNC 0x0040
127 /** Flag to indicate we're dealing with 4WHS: SYN, SYN, SYN/ACK, ACK
128  * (http://www.breakingpointsystems.com/community/blog/tcp-portals-the-three-way-handshake-is-a-lie) */
129 #define STREAMTCP_FLAG_4WHS 0x0080
130 /** Flag to indicate that this session is possible trying to evade the detection
131  * (http://www.packetstan.com/2010/06/recently-ive-been-on-campaign-to-make.html) */
132 #define STREAMTCP_FLAG_DETECTION_EVASION_ATTEMPT 0x0100
133 /** Flag to indicate the client (SYN pkt) permits SACK */
134 #define STREAMTCP_FLAG_CLIENT_SACKOK 0x0200
135 /** Flag to indicate both sides of the session permit SACK (SYN + SYN/ACK) */
136 #define STREAMTCP_FLAG_SACKOK 0x0400
137 /** Flag for triggering RAW reassembly before the size limit is reached or
138  the stream reaches EOF. */
139 #define STREAMTCP_FLAG_TRIGGER_RAW_REASSEMBLY 0x0800
140 /** 3WHS confirmed by server -- if suri sees 3whs ACK but server doesn't (pkt
141  * is lost on the way to server), SYN/ACK is retransmitted. If server sends
142  * normal packet we assume 3whs to be completed. Only used for SYN/ACK resend
143  * event. */
144 #define STREAMTCP_FLAG_3WHS_CONFIRMED 0x1000
145 /** App Layer tracking/reassembly is disabled */
