133 int DetectSslStateMatch(ThreadVars *t, DetectEngineThreadCtx *det_ctx,
134  Flow *f, uint8_t flags, void *alstate, Signature *s,
135  SigMatch *m)
136 {
137  int result = 1;
138  DetectSslStateData *ssd = (DetectSslStateData *)m->ctx;
139  SSLState *ssl_state = (SSLState *)alstate;
140  if (ssl_state == NULL) {
141  SCLogDebug("no app state, no match");
142  return 0;
143  }
144 
145  if ((ssd->flags & SSL_AL_FLAG_STATE_CLIENT_HELLO) &&
146  !(ssl_state->flags & SSL_AL_FLAG_STATE_CLIENT_HELLO)) {
147  result = 0;
148  goto end;
149  }
150  if ((ssd->flags & SSL_AL_FLAG_STATE_SERVER_HELLO) &&
151  !(ssl_state->flags & SSL_AL_FLAG_STATE_SERVER_HELLO)) {
152  result = 0;
153  goto end;
154  }
155  if ((ssd->flags & SSL_AL_FLAG_STATE_CLIENT_KEYX) &&
156  !(ssl_state->flags & SSL_AL_FLAG_STATE_CLIENT_KEYX)) {
157  result = 0;
158  goto end;
159  }
160  if ((ssd->flags & SSL_AL_FLAG_STATE_SERVER_KEYX) &&
161  !(ssl_state->flags & SSL_AL_FLAG_STATE_SERVER_KEYX)) {
162  result = 0;
163  goto end;
164  }
165 
166  end:
167  return result;
168 }

179 DetectSslStateData *DetectSslStateParse(char *arg)
180 {
181 #define MAX_SUBSTRINGS 30
182  int ret = 0, res = 0;
183  int ov1[MAX_SUBSTRINGS];
184  int ov2[MAX_SUBSTRINGS];
185  const char *str1;
186  const char *str2;
187  uint32_t flags = 0;
188  DetectSslStateData *ssd = NULL;
189 
190  ret = pcre_exec(parse_regex1, parse_regex1_study, arg, strlen(arg), 0, 0,
191  ov1, MAX_SUBSTRINGS);
192  if (ret < 1) {
193  SCLogError(SC_ERR_INVALID_SIGNATURE, "Invalid arg \"%s\" supplied to "
194  "ssl_state keyword.", arg);
195  goto error;
196  }
197  res = pcre_get_substring((char *)arg, ov1, MAX_SUBSTRINGS, 1, &str1);
198  if (res < 0) {
199  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
200  goto error;
201  }
202 
203  if (strcmp("client_hello", str1) == 0) {
204  flags |= DETECT_SSL_STATE_CLIENT_HELLO;
205  } else if (strcmp("server_hello", str1) == 0) {
206  flags |= DETECT_SSL_STATE_SERVER_HELLO;
207  } else if (strcmp("client_keyx", str1) == 0) {
208  flags |= DETECT_SSL_STATE_CLIENT_KEYX;
209  } else if (strcmp("server_keyx", str1) == 0) {
210  flags |= DETECT_SSL_STATE_SERVER_KEYX;
211  } else if (strcmp("unknown", str1) == 0) {
212  flags |= DETECT_SSL_STATE_UNKNOWN;
213  } else {
214  SCLogError(SC_ERR_INVALID_SIGNATURE, "Found invalid option \"%s\" "
215  "in ssl_state keyword.", str1);
216  goto error;
217  }
218 
219  pcre_free_substring(str1);
220 
221  res = pcre_get_substring((char *)arg, ov1, MAX_SUBSTRINGS, 2, &str1);
222  if (res < 0) {
223  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
224  goto error;
225  }
226  while (res > 0) {
227  ret = pcre_exec(parse_regex2, parse_regex2_study, str1, strlen(str1), 0, 0,
228  ov2, MAX_SUBSTRINGS);
229  if (ret < 1) {
230  SCLogError(SC_ERR_INVALID_SIGNATURE, "Invalid arg \"%s\" supplied to "
231  "ssl_state keyword.", arg);
232  goto error;
233  }
234  res = pcre_get_substring((char *)str1, ov2, MAX_SUBSTRINGS, 1, &str2);
235  if (res <= 0) {
236  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
237  goto error;
238  }
239  if (strcmp("client_hello", str2) == 0) {
240  flags |= DETECT_SSL_STATE_CLIENT_HELLO;
241  } else if (strcmp("server_hello", str2) == 0) {
242  flags |= DETECT_SSL_STATE_SERVER_HELLO;
243  } else if (strcmp("client_keyx", str2) == 0) {
244  flags |= DETECT_SSL_STATE_CLIENT_KEYX;
245  } else if (strcmp("server_keyx", str2) == 0) {
246  flags |= DETECT_SSL_STATE_SERVER_KEYX;
247  } else if (strcmp("unknown", str2) == 0) {
248  flags |= DETECT_SSL_STATE_UNKNOWN;
249  } else {
250  SCLogError(SC_ERR_INVALID_SIGNATURE, "Found invalid option \"%s\" "
251  "in ssl_state keyword.", str2);
252  goto error;
253  }
254 
255  res = pcre_get_substring((char *)str1, ov2, MAX_SUBSTRINGS, 2, &str2);
256  if (res < 0) {
257  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
258  goto error;
259  }
260 
261  pcre_free_substring(str1);
262  str1 = str2;
263  }
264 
265  if ( (ssd = SCMalloc(sizeof(DetectSslStateData))) == NULL) {
266  goto error;
267  }
268  ssd->flags = flags;
269 
270  return ssd;
271 
272 error:
273  if (ssd != NULL)
274  DetectSslStateFree(ssd);
275  return NULL;
276 }

278  /**
279  * \internal
280  * \brief Setup function for ssl_state keyword.
281  *
282  * \param de_ctx Pointer to the Detection Engine Context.
283  * \param s Pointer to the Current Signature
284  * \param arg String holding the arg.
285  *
286  * \retval 0 On success.
287  * \retval -1 On failure.
288  */
289 int DetectSslStateSetup(DetectEngineCtx *de_ctx, Signature *s, char *arg)
290 {
291  DetectSslStateData *ssd = NULL;
292  SigMatch *sm = NULL;
293 
294  ssd = DetectSslStateParse(arg);
295  if (ssd == NULL)
296  goto error;
297 
298  sm = SigMatchAlloc();
299  if (sm == NULL)
300  goto error;
301 
302  sm->type = DETECT_AL_SSL_STATE;
303  sm->ctx = (SigMatchCtx*)ssd;
304 
305  if (s->alproto != ALPROTO_UNKNOWN && s->alproto != ALPROTO_TLS) {
306  SCLogError(SC_ERR_CONFLICTING_RULE_KEYWORDS,
307  "Rule contains conflicting keywords. Have non-tls alproto "
308  "set for a rule containing \"ssl_state\" keyword");
309  goto error;
310  }
311  s->alproto = ALPROTO_TLS;
312 
313  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_AMATCH);
314 
315  return 0;
316 
317 error:
318  if (ssd != NULL)
319  DetectSslStateFree(ssd);
320  if (sm != NULL)
321  SCFree(sm);
322  return -1;
323 }
