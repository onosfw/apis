184 /** \brief Populate lua states pool
185  *
186  * \param num keyword instances
187  * \param reloads bool indicating we have rule reloads enabled
188  */
189 int DetectLuajitSetupStatesPool(int num, int reloads)
190 {
191  int retval = 0;
192  pthread_mutex_lock(&luajit_states_lock);
193 
194  if (luajit_states == NULL) {
195  intmax_t cnt = 0;
196  ConfNode *denode = NULL;
197  ConfNode *decnf = ConfGetNode("detect-engine");
198  if (decnf != NULL) {
199  TAILQ_FOREACH(denode, &decnf->head, next) {
200  if (strcmp(denode->val, "luajit-states") == 0) {
201  ConfGetChildValueInt(denode, "luajit-states", &cnt);
202  }
203  }
204  }
205 
206  if (cnt == 0) {
207  int cpus = UtilCpuGetNumProcessorsOnline();
208  if (cpus == 0) {
209  cpus = 10;
210  }
211  cnt = num * cpus;
212  cnt *= 3; /* assume 3 threads per core */
213 
214  /* alloc a bunch extra so reload can add new rules/instances */
215  if (reloads)
216  cnt *= 5;
217  }
218 
219  luajit_states = PoolInit(0, cnt, 0, LuaStatePoolAlloc, NULL, NULL, NULL, LuaStatePoolFree);
220  if (luajit_states == NULL) {
221  SCLogError(SC_ERR_LUA_ERROR, "luastate pool init failed, lua/luajit keywords won't work");
222  retval = -1;
223  }
224  }
225 
226  pthread_mutex_unlock(&luajit_states_lock);
227  return retval;
228 }

294 int DetectLuaMatchBuffer(DetectEngineThreadCtx *det_ctx, Signature *s, SigMatch *sm,
295  uint8_t *buffer, uint32_t buffer_len, uint32_t offset,
296  Flow *f, int flow_lock)
297 {
298  SCEnter();
299  int ret = 0;
300 
301  if (buffer == NULL || buffer_len == 0)
302  SCReturnInt(0);
303 
304  DetectLuaData *luajit = (DetectLuaData *)sm->ctx;
305  if (luajit == NULL)
306  SCReturnInt(0);
307 
308  DetectLuaThreadData *tluajit = (DetectLuaThreadData *)DetectThreadCtxGetKeywordThreadCtx(det_ctx, luajit->thread_ctx_id);
309  if (tluajit == NULL)
310  SCReturnInt(0);
311 
312  /* setup extension data for use in lua c functions */
313  LuaExtensionsMatchSetup(tluajit->luastate, luajit, det_ctx,
314  f, flow_lock, /* no packet in the ctx */NULL, 0);
315 
316  /* prepare data to pass to script */
317  lua_getglobal(tluajit->luastate, "match");
318  lua_newtable(tluajit->luastate); /* stack at -1 */
319 
320  lua_pushliteral (tluajit->luastate, "offset"); /* stack at -2 */
321  lua_pushnumber (tluajit->luastate, (int)(offset + 1));
322  lua_settable(tluajit->luastate, -3);
323 
324  lua_pushstring (tluajit->luastate, luajit->buffername); /* stack at -2 */
325  LuaPushStringBuffer(tluajit->luastate, (const uint8_t *)buffer, (size_t)buffer_len);
326  lua_settable(tluajit->luastate, -3);
327 
328  int retval = lua_pcall(tluajit->luastate, 1, 1, 0);
329  if (retval != 0) {
330  SCLogInfo("failed to run script: %s", lua_tostring(tluajit->luastate, -1));
331  }
332 
333  /* process returns from script */
334  if (lua_gettop(tluajit->luastate) > 0) {
335  /* script returns a number (return 1 or return 0) */
336  if (lua_type(tluajit->luastate, 1) == LUA_TNUMBER) {
337  double script_ret = lua_tonumber(tluajit->luastate, 1);
338  SCLogDebug("script_ret %f", script_ret);
339  lua_pop(tluajit->luastate, 1);
340 
341  if (script_ret == 1.0)
342  ret = 1;
343 
344  /* script returns a table */
345  } else if (lua_type(tluajit->luastate, 1) == LUA_TTABLE) {
346  lua_pushnil(tluajit->luastate);
347  const char *k, *v;
348  while (lua_next(tluajit->luastate, -2)) {
349  v = lua_tostring(tluajit->luastate, -1);
350  lua_pop(tluajit->luastate, 1);
351  k = lua_tostring(tluajit->luastate, -1);
352 
353  if (!k || !v)
354  continue;
355 
356  SCLogDebug("k='%s', v='%s'", k, v);
357 
358  if (strcmp(k, "retval") == 0) {
359  if (atoi(v) == 1)
360  ret = 1;
361  } else {
362  /* set flow var? */
363  }
364  }
365 
366  /* pop the table */
367  lua_pop(tluajit->luastate, 1);
368  }
369  } else {
370  SCLogDebug("no stack");
371  }
372 
373  /* clear the stack */
374  while (lua_gettop(tluajit->luastate) > 0) {
375  lua_pop(tluajit->luastate, 1);
376  }
377 
378  if (luajit->negated) {
379  if (ret == 1)
380  ret = 0;
381  else
382  ret = 1;
383  }
384 
385  SCReturnInt(ret);
386 
387 }

390  * \brief match the specified luajit
391  *
392  * \param t thread local vars
393  * \param det_ctx pattern matcher thread local data
394  * \param p packet
395  * \param s signature being inspected
396  * \param m sigmatch that we will cast into DetectLuaData

401 static int DetectLuaMatch (ThreadVars *tv, DetectEngineThreadCtx *det_ctx,
402  Packet *p, Signature *s, const SigMatchCtx *ctx)
403 {
404  SCEnter();
405  int ret = 0;
406  DetectLuaData *luajit = (DetectLuaData *)ctx;
407  if (luajit == NULL)
408  SCReturnInt(0);
409 
410  DetectLuaThreadData *tluajit = (DetectLuaThreadData *)DetectThreadCtxGetKeywordThreadCtx(det_ctx, luajit->thread_ctx_id);
411  if (tluajit == NULL)
412  SCReturnInt(0);
413 
414  /* setup extension data for use in lua c functions */
415  uint8_t flags = 0;
416  if (p->flowflags & FLOW_PKT_TOSERVER)
417  flags = STREAM_TOSERVER;
418  else if (p->flowflags & FLOW_PKT_TOCLIENT)
419  flags = STREAM_TOCLIENT;
420 
421  LuaExtensionsMatchSetup(tluajit->luastate, luajit, det_ctx,
422  p->flow, /* flow not locked */LUA_FLOW_NOT_LOCKED_BY_PARENT, p, flags);
423 
424  if ((tluajit->flags & DATATYPE_PAYLOAD) && p->payload_len == 0)
425  SCReturnInt(0);
426  if ((tluajit->flags & DATATYPE_PACKET) && GET_PKT_LEN(p) == 0)
427  SCReturnInt(0);
428  if (tluajit->alproto != ALPROTO_UNKNOWN) {
429  if (p->flow == NULL)
430  SCReturnInt(0);
431 
432  FLOWLOCK_RDLOCK(p->flow);
433  int alproto = p->flow->alproto;
434  FLOWLOCK_UNLOCK(p->flow);
435 
436  if (tluajit->alproto != alproto)
437  SCReturnInt(0);
438  }
439 
440  lua_getglobal(tluajit->luastate, "match");
441  lua_newtable(tluajit->luastate); /* stack at -1 */
442 
443  if ((tluajit->flags & DATATYPE_PAYLOAD) && p->payload_len) {
444  lua_pushliteral(tluajit->luastate, "payload"); /* stack at -2 */
445  LuaPushStringBuffer (tluajit->luastate, (const uint8_t *)p->payload, (size_t)p->payload_len); /* stack at -3 */
446  lua_settable(tluajit->luastate, -3);
447  }
448  if ((tluajit->flags & DATATYPE_PACKET) && GET_PKT_LEN(p)) {
449  lua_pushliteral(tluajit->luastate, "packet"); /* stack at -2 */
450  LuaPushStringBuffer (tluajit->luastate, (const uint8_t *)GET_PKT_DATA(p), (size_t)GET_PKT_LEN(p)); /* stack at -3 */
451  lua_settable(tluajit->luastate, -3);
452  }
453  if (tluajit->alproto == ALPROTO_HTTP) {
454  FLOWLOCK_RDLOCK(p->flow);
455  HtpState *htp_state = p->flow->alstate;
456  if (htp_state != NULL && htp_state->connp != NULL) {
457  htp_tx_t *tx = NULL;
458  uint64_t idx = AppLayerParserGetTransactionInspectId(p->flow->alparser,
459  STREAM_TOSERVER);
460  uint64_t total_txs= AppLayerParserGetTxCnt(IPPROTO_TCP, ALPROTO_HTTP, htp_state);
461  for ( ; idx < total_txs; idx++) {
462  tx = AppLayerParserGetTx(IPPROTO_TCP, ALPROTO_HTTP, htp_state, idx);
463  if (tx == NULL)
464  continue;
465 
466  if ((tluajit->flags & DATATYPE_HTTP_REQUEST_LINE) && tx->request_line != NULL &&
467  bstr_len(tx->request_line) > 0) {
468  lua_pushliteral(tluajit->luastate, "http.request_line"); /* stack at -2 */
469  LuaPushStringBuffer(tluajit->luastate,
470  (const uint8_t *)bstr_ptr(tx->request_line),
471  bstr_len(tx->request_line));
472  lua_settable(tluajit->luastate, -3);
473  }
474  }
475  }
476  FLOWLOCK_UNLOCK(p->flow);
477  }
478 
479  int retval = lua_pcall(tluajit->luastate, 1, 1, 0);
480  if (retval != 0) {
481  SCLogInfo("failed to run script: %s", lua_tostring(tluajit->luastate, -1));
482  }
483 
484  /* process returns from script */
485  if (lua_gettop(tluajit->luastate) > 0) {
486 
487  /* script returns a number (return 1 or return 0) */
488  if (lua_type(tluajit->luastate, 1) == LUA_TNUMBER) {
489  double script_ret = lua_tonumber(tluajit->luastate, 1);
490  SCLogDebug("script_ret %f", script_ret);
491  lua_pop(tluajit->luastate, 1);
492 
493  if (script_ret == 1.0)
494  ret = 1;
495 
496  /* script returns a table */
497  } else if (lua_type(tluajit->luastate, 1) == LUA_TTABLE) {
498  lua_pushnil(tluajit->luastate);
499  const char *k, *v;
500  while (lua_next(tluajit->luastate, -2)) {
501  v = lua_tostring(tluajit->luastate, -1);
502  lua_pop(tluajit->luastate, 1);
503  k = lua_tostring(tluajit->luastate, -1);
504 
505  if (!k || !v)
506  continue;
507 
508  SCLogDebug("k='%s', v='%s'", k, v);
509 
510  if (strcmp(k, "retval") == 0) {
511  if (atoi(v) == 1)
512  ret = 1;
513  } else {
514  /* set flow var? */
515  }
516  }
517 
518  /* pop the table */
519  lua_pop(tluajit->luastate, 1);
520  }
521  }
522  while (lua_gettop(tluajit->luastate) > 0) {
523  lua_pop(tluajit->luastate, 1);
524  }
525 
526  if (luajit->negated) {
527  if (ret == 1)
528  ret = 0;
529  else
530  ret = 1;
531  }
532 
533  SCReturnInt(ret);
534 }

804 
805  /* hackish, needed to allow unittests to pass buffers as scripts instead of files */
806 #ifdef UNITTESTS

816  if (status) {
817  SCLogError(SC_ERR_LUA_ERROR, "couldn't load file: %s", lua_tostring(luastate, -1));
818  goto error;
819  }
820 #ifdef UNITTESTS
821  }
822 #endif

825  if (lua_pcall(luastate, 0, 0, 0) != 0) {
826  SCLogError(SC_ERR_LUA_ERROR, "couldn't prime file: %s", lua_tostring(luastate, -1));
827  goto error;
828  }
829 
830  lua_getglobal(luastate, "init");
831  if (lua_type(luastate, -1) != LUA_TFUNCTION) {
832  SCLogError(SC_ERR_LUA_ERROR, "no init function in script");
833  goto error;
834  }

1041  * \brief this function is used to parse luajit options
1042  * \brief into the current signature
1043  *

1045  * \param s pointer to the Current Signature
1046  * \param str pointer to the user provided "luajit" option
1047  *
1048  * \retval 0 on Success
1049  * \retval -1 on Failure

1052 {

1066  DetectLuaThreadFree, 0);

1068  goto error;
1069 
1070  if (luajit->alproto != ALPROTO_UNKNOWN) {
1071  if (s->alproto != ALPROTO_UNKNOWN && luajit->alproto != s->alproto) {
1072  goto error;

1075  }

1078  * and put it in the Signature. */

1080  if (sm == NULL)
1081  goto error;

1088  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_PMATCH);
1089  else
1090  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);
1091  } else if (luajit->alproto == ALPROTO_HTTP) {
1092  if (luajit->flags & DATATYPE_HTTP_RESPONSE_BODY)
1093  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_FILEDATA);

1095  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_HCBDMATCH);
1096  else if (luajit->flags & DATATYPE_HTTP_URI)
1097  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_UMATCH);
1098  else if (luajit->flags & DATATYPE_HTTP_URI_RAW)
1099  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_HRUDMATCH);
1100  else if (luajit->flags & DATATYPE_HTTP_REQUEST_COOKIE)
1101  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_HCDMATCH);
1102  else if (luajit->flags & DATATYPE_HTTP_REQUEST_UA)
1103  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_HUADMATCH);
1104  else if (luajit->flags & (DATATYPE_HTTP_REQUEST_HEADERS|DATATYPE_HTTP_RESPONSE_HEADERS))
1105  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_HHDMATCH);
1106  else if (luajit->flags & (DATATYPE_HTTP_REQUEST_HEADERS_RAW|DATATYPE_HTTP_RESPONSE_HEADERS_RAW))
1107  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_HRHDMATCH);
1108  else if (luajit->flags & DATATYPE_HTTP_RESPONSE_COOKIE)
1109  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_HCDMATCH);
1110  else
1111  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_HRLMATCH);
1112  } else if (luajit->alproto == ALPROTO_DNS) {
1113  if (luajit->flags & DATATYPE_DNS_RRNAME) {
1114  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_DNSQUERYNAME_MATCH);
1115  } else if (luajit->flags & DATATYPE_DNS_REQUEST) {
1116  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_DNSREQUEST_MATCH);
1117  } else if (luajit->flags & DATATYPE_DNS_RESPONSE) {
1118  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_DNSRESPONSE_MATCH);
1119  }
1120  } else if (luajit->alproto == ALPROTO_TLS) {
1121  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_AMATCH);
1122  } else if (luajit->alproto == ALPROTO_SSH) {
1123  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_AMATCH);
1124  } else {
1125  SCLogError(SC_ERR_LUA_ERROR, "luajit can't be used with protocol %s",
1126  AppLayerGetProtoName(luajit->alproto));
1127  goto error;
1128  }
1129 
1130  de_ctx->detect_luajit_instances++;
1131  return 0;
1132 
1133 error:

1135  DetectLuaFree(luajit);
1136  if (sm != NULL)
1137  SCFree(sm);
1138  return -1;
1139 }
