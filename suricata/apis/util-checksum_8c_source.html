30 int ReCalculateChecksum(Packet *p)
31 {
32  if (PKT_IS_IPV4(p)) {
33  if (PKT_IS_TCP(p)) {
34  /* TCP */
35  p->tcph->th_sum = 0;
36  p->tcph->th_sum = TCPCalculateChecksum(p->ip4h->s_ip_addrs,
37  (uint16_t *)p->tcph, (p->payload_len + TCP_GET_HLEN(p)));
38  } else if (PKT_IS_UDP(p)) {
39  p->udph->uh_sum = 0;
40  p->udph->uh_sum = UDPV4CalculateChecksum(p->ip4h->s_ip_addrs,
41  (uint16_t *)p->udph, (p->payload_len + UDP_HEADER_LEN));
42  }
43  /* IPV4 */
44  p->ip4h->ip_csum = 0;
45  p->ip4h->ip_csum = IPV4CalculateChecksum((uint16_t *)p->ip4h,
46  IPV4_GET_RAW_HLEN(p->ip4h));
47  } else if (PKT_IS_IPV6(p)) {
48  /* just TCP for IPV6 */
49  if (PKT_IS_TCP(p)) {
50  p->tcph->th_sum = 0;
51  p->tcph->th_sum = TCPV6CalculateChecksum(p->ip6h->s_ip6_addrs,
52  (uint16_t *)p->tcph, (p->payload_len + TCP_GET_HLEN(p)));
53  } else if (PKT_IS_UDP(p)) {
54  p->udph->uh_sum = 0;
55  p->udph->uh_sum = UDPV6CalculateChecksum(p->ip6h->s_ip6_addrs,
56  (uint16_t *)p->udph, (p->payload_len + UDP_HEADER_LEN));
57  }
58  }
59 
60  return 0;

63 /**
64  * \brief Check if the number of invalid checksums indicate checksum
65  * offloading in place.
66  *
67  * \retval 1 yes, offloading in place
68  * \retval 0 no, no offloading used
69  */
70 int ChecksumAutoModeCheck(uint32_t thread_count,
71  unsigned int iface_count, unsigned int iface_fail)
72 {
73  if (thread_count == CHECKSUM_SAMPLE_COUNT) {
74  if (iface_fail != 0) {
75  if ((iface_count / iface_fail) < CHECKSUM_INVALID_RATIO) {
76  SCLogInfo("More than 1/%dth of packets have an invalid "
77  "checksum, assuming checksum offloading is used (%u/%u)",
78  CHECKSUM_INVALID_RATIO, iface_fail, iface_count);
79  return 1;
80  } else {
81  SCLogInfo("Less than 1/%dth of packets have an invalid "
82  "checksum, assuming checksum offloading is NOT used (%u/%u)",
83  CHECKSUM_INVALID_RATIO, iface_fail, iface_count);
84  }
85  } else {
86  SCLogInfo("No packets with invalid checksum, assuming checksum offloading is NOT used");
87  }
88  }
89  return 0;
90 }
