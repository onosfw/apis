84 /**
85  * \brief This function is used to match icmp_seq rule option set on a packet
86  *
87  * \param t pointer to thread vars
88  * \param det_ctx pointer to the pattern matcher thread
89  * \param p pointer to the current packet
90  * \param m pointer to the sigmatch that we will cast into DetectIcmpSeqData
91  *
92  * \retval 0 no match
93  * \retval 1 match
94  */
95 int DetectIcmpSeqMatch (ThreadVars *t, DetectEngineThreadCtx *det_ctx, Packet *p, Signature *s, const SigMatchCtx *ctx)
96 {
97  uint16_t seqn;
98  const DetectIcmpSeqData *iseq = (const DetectIcmpSeqData *)ctx;
99 
100  if (PKT_IS_PSEUDOPKT(p))
101  return 0;
102 
103  if (PKT_IS_ICMPV4(p)) {
104  switch (ICMPV4_GET_TYPE(p)){
105  case ICMP_ECHOREPLY:
106  case ICMP_ECHO:
107  case ICMP_TIMESTAMP:
108  case ICMP_TIMESTAMPREPLY:
109  case ICMP_INFO_REQUEST:
110  case ICMP_INFO_REPLY:
111  case ICMP_ADDRESS:
112  case ICMP_ADDRESSREPLY:
113  SCLogDebug("ICMPV4_GET_SEQ(p) %"PRIu16" (network byte order), "
114  "%"PRIu16" (host byte order)", ICMPV4_GET_SEQ(p),
115  ntohs(ICMPV4_GET_SEQ(p)));
116 
117  seqn = ICMPV4_GET_SEQ(p);
118  break;
119  default:
120  SCLogDebug("Packet has no seq field");
121  return 0;
122  }
123  } else if (PKT_IS_ICMPV6(p)) {
124 
125  switch (ICMPV6_GET_TYPE(p)) {
126  case ICMP6_ECHO_REQUEST:
127  case ICMP6_ECHO_REPLY:
128  SCLogDebug("ICMPV6_GET_SEQ(p) %"PRIu16" (network byte order), "
129  "%"PRIu16" (host byte order)", ICMPV6_GET_SEQ(p),
130  ntohs(ICMPV6_GET_SEQ(p)));
131 
132  seqn = ICMPV6_GET_SEQ(p);
133  break;
134  default:
135  SCLogDebug("Packet has no seq field");
136  return 0;
137  }
138  } else {
139  SCLogDebug("Packet not ICMPV4 nor ICMPV6");
140  return 0;
141  }
142 
143  if (seqn == iseq->seq)
144  return 1;
145 
146  return 0;
147 }

149 /**
150  * \brief This function is used to parse icmp_seq option passed via icmp_seq: keyword
151  *
152  * \param icmpseqstr Pointer to the user provided icmp_seq options
153  *
154  * \retval iseq pointer to DetectIcmpSeqData on success
155  * \retval NULL on failure
156  */
157 DetectIcmpSeqData *DetectIcmpSeqParse (char *icmpseqstr)
158 {
159  DetectIcmpSeqData *iseq = NULL;
160  char *substr[3] = {NULL, NULL, NULL};
161 #define MAX_SUBSTRINGS 30
162  int ret = 0, res = 0;
163  int ov[MAX_SUBSTRINGS];
164  int i;
165  const char *str_ptr;
166 
167  ret = pcre_exec(parse_regex, parse_regex_study, icmpseqstr, strlen(icmpseqstr), 0, 0, ov, MAX_SUBSTRINGS);
168  if (ret < 1 || ret > 4) {
169  SCLogError(SC_ERR_PCRE_MATCH,"Parse error %s", icmpseqstr);
170  goto error;
171  }
172 
173  for (i = 1; i < ret; i++) {
174  res = pcre_get_substring((char *)icmpseqstr, ov, MAX_SUBSTRINGS, i, &str_ptr);
175  if (res < 0) {
176  SCLogError(SC_ERR_PCRE_GET_SUBSTRING,"pcre_get_substring failed");
177  goto error;
178  }
179  substr[i-1] = (char *)str_ptr;
180  }
181 
182  iseq = SCMalloc(sizeof(DetectIcmpSeqData));
183  if (unlikely(iseq == NULL))
184  goto error;
185 
186  iseq->seq = 0;
187 
188  if (substr[0] != NULL && strlen(substr[0]) != 0) {
189  if (substr[2] == NULL) {
190  SCLogError(SC_ERR_MISSING_QUOTE,"Missing quote in input");
191  goto error;
192  }
193  } else {
194  if (substr[2] != NULL) {
195  SCLogError(SC_ERR_MISSING_QUOTE,"Missing quote in input");
196  goto error;
197  }
198  }
199 
200  uint16_t seq = 0;
201  if (ByteExtractStringUint16(&seq, 10, 0, substr[1]) < 0) {
202  SCLogError(SC_ERR_INVALID_ARGUMENT, "specified icmp seq %s is not "
203  "valid", substr[1]);
204  goto error;
205  }
206  iseq->seq = htons(seq);
207 
208  for (i = 0; i < 3; i++) {
209  if (substr[i] != NULL) SCFree(substr[i]);
210  }
211 
212  return iseq;
213 
214 error:
215  for (i = 0; i < 3; i++) {
216  if (substr[i] != NULL) SCFree(substr[i]);
217  }
218  if (iseq != NULL) DetectIcmpSeqFree(iseq);
219  return NULL;
220 
221 }

223 /**
224  * \brief this function is used to add the parsed icmp_seq data into the current signature
225  *
226  * \param de_ctx pointer to the Detection Engine Context
227  * \param s pointer to the Current Signature
228  * \param icmpseqstr pointer to the user provided icmp_seq option
229  *
230  * \retval 0 on Success
231  * \retval -1 on Failure
232  */
233 static int DetectIcmpSeqSetup (DetectEngineCtx *de_ctx, Signature *s, char *icmpseqstr)
234 {
235  DetectIcmpSeqData *iseq = NULL;
236  SigMatch *sm = NULL;
237 
238  iseq = DetectIcmpSeqParse(icmpseqstr);
239  if (iseq == NULL) goto error;
240 
241  sm = SigMatchAlloc();
242  if (sm == NULL) goto error;
243 
244  sm->type = DETECT_ICMP_SEQ;
245  sm->ctx = (SigMatchCtx *)iseq;
246 
247  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);
248 
249  return 0;
250 
251 error:
252  if (iseq != NULL) DetectIcmpSeqFree(iseq);
253  if (sm != NULL) SCFree(sm);
254  return -1;
255 
256 }
