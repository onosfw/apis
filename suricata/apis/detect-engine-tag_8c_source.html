113 /**
114  * \brief This function is used to add a tag to a session (type session)
115  * or update it if it's already installed. The number of times to
116  * allow an update is limited by DETECT_TAG_MATCH_LIMIT. This way
117  * repetitive matches to the same rule are limited of setting tags,
118  * to avoid DOS attacks
119  *
120  * \param p pointer to the current packet
121  * \param tde pointer to the new DetectTagDataEntry
122  *
123  * \retval 0 if the tde was added succesfuly
124  * \retval 1 if an entry of this sid/gid already exist and was updated
125  */
126 int TagFlowAdd(Packet *p, DetectTagDataEntry *tde)
127 {
128  uint8_t updated = 0;
129  uint16_t tag_cnt = 0;
130  DetectTagDataEntry *iter = NULL;
131 
132  if (p->flow == NULL)
133  return 1;
134 
135  FLOWLOCK_WRLOCK(p->flow);
136  iter = FlowGetStorageById(p->flow, flow_tag_id);
137  if (iter != NULL) {
138  /* First iterate installed entries searching a duplicated sid/gid */
139  for (; iter != NULL; iter = iter->next) {
140  tag_cnt++;
141 
142  if (iter->sid == tde->sid && iter->gid == tde->gid) {
143  iter->cnt_match++;
144 
145  /* If so, update data, unless the maximum MATCH limit is
146  * reached. This prevents possible DOS attacks */
147  if (iter->cnt_match < DETECT_TAG_MATCH_LIMIT) {
148  /* Reset time and counters */
149  iter->first_ts = iter->last_ts = tde->first_ts;
150  iter->packets = 0;
151  iter->bytes = 0;
152  }
153  updated = 1;
154  break;
155  }
156  }
157  }
158 
159  /* If there was no entry of this rule, prepend the new tde */
160  if (updated == 0 && tag_cnt < DETECT_TAG_MAX_TAGS) {
161  DetectTagDataEntry *new_tde = DetectTagDataCopy(tde);
162  if (new_tde != NULL) {
163  new_tde->next = FlowGetStorageById(p->flow, flow_tag_id);
164  FlowSetStorageById(p->flow, flow_tag_id, new_tde);
165  SCLogDebug("adding tag with first_ts %u", new_tde->first_ts);
166  (void) SC_ATOMIC_ADD(num_tags, 1);
167  }
168  } else if (tag_cnt == DETECT_TAG_MAX_TAGS) {
169  SCLogDebug("Max tags for sessions reached (%"PRIu16")", tag_cnt);
170  }
171 
172  FLOWLOCK_UNLOCK(p->flow);
173  return updated;
174 }

191  Host *host = NULL;
192

195  if (tde->flags & TAG_ENTRY_FLAG_DIR_SRC) {
196  host = HostGetHostFromHash(&p->src);
197  } else if (tde->flags & TAG_ENTRY_FLAG_DIR_DST) {

217  SCLogDebug("updating existing host");
218

220  DetectTagDataEntry *iter = NULL;
221

223  num_tags++;

225  iter->cnt_match++;

227  * reached. This prevents possible DOS attacks */

232  iter->bytes = 0;

235  break;
236  }
237  }
238

240  if (updated == 0 && num_tags < DETECT_TAG_MAX_TAGS) {
241  /* get a new tde as the one we have is on the stack */
242  DetectTagDataEntry *new_tde = DetectTagDataCopy(tde);
243  if (new_tde != NULL) {
244  (void) SC_ATOMIC_ADD(num_tags, 1);
245 
246  new_tde->next = tag;
247  HostSetStorageById(host, host_tag_id, new_tde);
248  }
249  } else if (num_tags == DETECT_TAG_MAX_TAGS) {
250  SCLogDebug("Max tags for sessions reached (%"PRIu16")", num_tags);
251  }
252  }
253 
254  HostRelease(host);
255  SCReturnInt(updated);

258 static void TagHandlePacketFlow(Flow *f, Packet *p)
259 {
260  if (FlowGetStorageById(f, flow_tag_id) == NULL)
261  return;
262 
263  DetectTagDataEntry *tde = NULL;
264  DetectTagDataEntry *prev = NULL;
265  DetectTagDataEntry *iter = FlowGetStorageById(f, flow_tag_id);
266  uint8_t flag_added = 0;
267 
268  while (iter != NULL) {
269  /* update counters */
270  iter->last_ts = p->ts.tv_sec;
271  switch (iter->metric) {
272  case DETECT_TAG_METRIC_PACKET:
273  iter->packets++;
274  break;
275  case DETECT_TAG_METRIC_BYTES:
276  iter->bytes += GET_PKT_LEN(p);
277  break;
278  }
279 
280  /* If this packet triggered the rule with tag, we dont need
281  * to log it (the alert will log it) */
282  if (!(iter->flags & TAG_ENTRY_FLAG_SKIPPED_FIRST)) {
283  iter->flags |= TAG_ENTRY_FLAG_SKIPPED_FIRST;
284  } else {
285  /* Update metrics; remove if tag expired; and set alerts */
286  switch (iter->metric) {
287  case DETECT_TAG_METRIC_PACKET:
288  if (iter->packets > iter->count) {
289  SCLogDebug("flow tag expired: packets %u > %u",
290  iter->packets, iter->count);
291  /* tag expired */
292  if (prev != NULL) {
293  tde = iter;
294  prev->next = iter->next;
295  iter = iter->next;
296  SCFree(tde);
297  (void) SC_ATOMIC_SUB(num_tags, 1);
298  continue;
299  } else {
300  FlowSetStorageById(p->flow, flow_tag_id, iter->next);
301  tde = iter;
302  iter = iter->next;
303  SCFree(tde);
304  (void) SC_ATOMIC_SUB(num_tags, 1);
305  continue;
306  }
307  } else if (flag_added == 0) {
308  /* It's matching the tag. Add it to be logged and
309  * update "flag_added" to add the packet once. */
310  p->flags |= PKT_HAS_TAG;
311  flag_added++;
312  }
313  break;
314  case DETECT_TAG_METRIC_BYTES:
315  if (iter->bytes > iter->count) {
316  /* tag expired */
317  SCLogDebug("flow tag expired: bytes %u > %u",
318  iter->bytes, iter->count);
319  if (prev != NULL) {
320  tde = iter;
321  prev->next = iter->next;
322  iter = iter->next;
323  SCFree(tde);
324  (void) SC_ATOMIC_SUB(num_tags, 1);
325  continue;
326  } else {
327  FlowSetStorageById(p->flow, flow_tag_id, iter->next);
328  tde = iter;
329  iter = iter->next;
330  SCFree(tde);
331  (void) SC_ATOMIC_SUB(num_tags, 1);
332  continue;
333  }
334  } else if (flag_added == 0) {
335  /* It's matching the tag. Add it to be logged and
336  * update "flag_added" to add the packet once. */
337  p->flags |= PKT_HAS_TAG;
338  flag_added++;
339  }
340  break;
341  case DETECT_TAG_METRIC_SECONDS:
342  /* last_ts handles this metric, but also a generic time based
343  * expiration to prevent dead sessions/hosts */
344  if (iter->last_ts - iter->first_ts > iter->count) {
345  SCLogDebug("flow tag expired: %u - %u = %u > %u",
346  iter->last_ts, iter->first_ts,
347  (iter->last_ts - iter->first_ts), iter->count);
348  /* tag expired */
349  if (prev != NULL) {
350  tde = iter;
351  prev->next = iter->next;
352  iter = iter->next;
353  SCFree(tde);
354  (void) SC_ATOMIC_SUB(num_tags, 1);
355  continue;
356  } else {
357  FlowSetStorageById(p->flow, flow_tag_id, iter->next);
358  tde = iter;
359  iter = iter->next;
360  SCFree(tde);
361  (void) SC_ATOMIC_SUB(num_tags, 1);
362  continue;
363  }
364  } else if (flag_added == 0) {
365  /* It's matching the tag. Add it to be logged and
366  * update "flag_added" to add the packet once. */
367  p->flags |= PKT_HAS_TAG;
368  flag_added++;
369  }
370  break;
371  }
372 
373  }
374 
375  prev = iter;
376  iter = iter->next;
377  }
378 }

513  unsigned int current_tags = SC_ATOMIC_GET(num_tags);

543  *

547  * \retval 1 no tags or tags removed -- host is free to go (from tag perspective)

550 int TagTimeoutCheck(Host *host, struct timeval *tv)
551 {
552  DetectTagDataEntry *tde = NULL;
553  DetectTagDataEntry *tmp = NULL;
554  DetectTagDataEntry *prev = NULL;
555  int retval = 1;
556 
557  tmp = HostGetStorageById(host, host_tag_id);
558  if (tmp == NULL)
559  return 1;
560 
561  prev = NULL;
562  while (tmp != NULL) {
563  if ((tv->tv_sec - tmp->last_ts) <= TAG_MAX_LAST_TIME_SEEN) {
564  prev = tmp;
565  tmp = tmp->next;
566  retval = 0;
567  continue;
568  }
569 
570  /* timed out */
571 
572  if (prev != NULL) {

579  (void) SC_ATOMIC_SUB(num_tags, 1);
