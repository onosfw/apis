104 static inline void AlertFastLogOutputAlert(AlertFastLogThread *aft, char *buffer,
105  int alert_size)
106 {
107  SCMutex *file_lock = &aft->file_ctx->fp_mutex;
108  /* Output the alert string and count alerts. Only need to lock here. */
109  SCMutexLock(file_lock);
110  aft->file_ctx->alerts++;
111  aft->file_ctx->Write(buffer, alert_size, aft->file_ctx);
112  SCMutexUnlock(file_lock);
113 }

115 int AlertFastLogger(ThreadVars *tv, void *data, const Packet *p)
116 {
117  AlertFastLogThread *aft = (AlertFastLogThread *)data;
118  int i;
119  char timebuf[64];
120  int decoder_event = 0;
121 
122  CreateTimeString(&p->ts, timebuf, sizeof(timebuf));
123 
124  char srcip[46], dstip[46];
125  if (PKT_IS_IPV4(p)) {
126  PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p), srcip, sizeof(srcip));
127  PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p), dstip, sizeof(dstip));
128  } else if (PKT_IS_IPV6(p)) {
129  PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p), srcip, sizeof(srcip));
130  PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p), dstip, sizeof(dstip));
131  } else {
132  decoder_event = 1;
133  }
134 
135  /* Buffer to store the generated alert strings. The buffer is
136  * filled with alert strings until it doesn't have room to store
137  * another full alert, only then is the buffer written. This is
138  * more efficient for multiple alerts and only slightly slower for
139  * single alerts.
140  */
141  char alert_buffer[MAX_FASTLOG_BUFFER_SIZE];
142 
143  for (i = 0; i < p->alerts.cnt; i++) {
144  const PacketAlert *pa = &p->alerts.alerts[i];
145  if (unlikely(pa->s == NULL)) {
146  continue;
147  }
148 
149  char *action = "";
150  if ((pa->action & ACTION_DROP) && EngineModeIsIPS()) {
151  action = "[Drop] ";
152  } else if (pa->action & ACTION_DROP) {
153  action = "[wDrop] ";
154  }
155 
156  char proto[16] = "";
157  if (likely(decoder_event == 0)) {
158  if (SCProtoNameValid(IP_GET_IPPROTO(p)) == TRUE) {
159  strlcpy(proto, known_proto[IP_GET_IPPROTO(p)], sizeof(proto));
160  } else {
161  snprintf(proto, sizeof(proto), "PROTO:%03" PRIu32, IP_GET_IPPROTO(p));
162  }
163  }
164 
165  /* Create the alert string without locking. */
166  int size = 0;
167  if (likely(decoder_event == 0)) {
168  PrintBufferData(alert_buffer, &size, MAX_FASTLOG_ALERT_SIZE,
169  "%s %s[**] [%" PRIu32 ":%" PRIu32 ":%"
170  PRIu32 "] %s [**] [Classification: %s] [Priority: %"PRIu32"]"
171  " {%s} %s:%" PRIu32 " -> %s:%" PRIu32 "\n", timebuf, action,
172  pa->s->gid, pa->s->id, pa->s->rev, pa->s->msg, pa->s->class_msg, pa->s->prio,
173  proto, srcip, p->sp, dstip, p->dp);
174  } else {
175  PrintBufferData(alert_buffer, &size, MAX_FASTLOG_ALERT_SIZE,
176  "%s %s[**] [%" PRIu32 ":%" PRIu32
177  ":%" PRIu32 "] %s [**] [Classification: %s] [Priority: "
178  "%" PRIu32 "] [**] [Raw pkt: ", timebuf, action, pa->s->gid,
179  pa->s->id, pa->s->rev, pa->s->msg, pa->s->class_msg, pa->s->prio);
180  PrintBufferRawLineHex(alert_buffer, &size, MAX_FASTLOG_ALERT_SIZE,
181  GET_PKT_DATA(p), GET_PKT_LEN(p) < 32 ? GET_PKT_LEN(p) : 32);
182  if (p->pcap_cnt != 0) {
183  PrintBufferData(alert_buffer, &size, MAX_FASTLOG_ALERT_SIZE,
184  "] [pcap file packet: %"PRIu64"]\n", p->pcap_cnt);
185  } else {
186  PrintBufferData(alert_buffer, &size, MAX_FASTLOG_ALERT_SIZE, "]\n");
187  }
188  }
189 
190  /* Write the alert to output file */
191  AlertFastLogOutputAlert(aft, alert_buffer, size);
192  }
193 
194  return TM_ECODE_OK;
195 }

197 TmEcode AlertFastLogThreadInit(ThreadVars *t, void *initdata, void **data)
198 {
199  AlertFastLogThread *aft = SCMalloc(sizeof(AlertFastLogThread));
200  if (unlikely(aft == NULL))
201  return TM_ECODE_FAILED;
202  memset(aft, 0, sizeof(AlertFastLogThread));
203  if(initdata == NULL)
204  {
205  SCLogDebug("Error getting context for AlertFastLog. \"initdata\" argument NULL");
206  SCFree(aft);
207  return TM_ECODE_FAILED;
208  }
209  /** Use the Ouptut Context (file pointer and mutex) */
210  aft->file_ctx = ((OutputCtx *)initdata)->data;
211 
212  *data = (void *)aft;
213  return TM_ECODE_OK;
214 }

240 /**
241  * \brief Create a new LogFileCtx for "fast" output style.
242  * \param conf The configuration node for this output.
243  * \return A LogFileCtx pointer on success, NULL on failure.
244  */
245 OutputCtx *AlertFastLogInitCtx(ConfNode *conf)
246 {
247  LogFileCtx *logfile_ctx = LogFileNewCtx();
248  if (logfile_ctx == NULL) {
249  SCLogDebug("AlertFastLogInitCtx2: Could not create new LogFileCtx");
250  return NULL;
251  }
252 
253  if (SCConfLogOpenGeneric(conf, logfile_ctx, DEFAULT_LOG_FILENAME, 1) < 0) {
254  LogFileFreeCtx(logfile_ctx);
255  return NULL;
256  }
257 
258  OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
259  if (unlikely(output_ctx == NULL))
260  return NULL;
261  output_ctx->data = logfile_ctx;
262  output_ctx->DeInit = AlertFastLogDeInitCtx;
263 
264  return output_ctx;
265 }

271  SCFree(output_ctx);
