47 int DecodeMPLS(ThreadVars *tv, DecodeThreadVars *dtv, Packet *p, uint8_t *pkt,
48  uint16_t len, PacketQueue *pq)
49 {
50  uint32_t shim;
51  int label;
52  int event = 0;
53 
54  StatsIncr(tv, dtv->counter_mpls);
55 
56  do {
57  if (len < MPLS_HEADER_LEN) {
58  ENGINE_SET_INVALID_EVENT(p, MPLS_HEADER_TOO_SMALL);
59  return TM_ECODE_FAILED;
60  }
61  shim = *(uint32_t *)pkt;
62  pkt += MPLS_HEADER_LEN;
63  len -= MPLS_HEADER_LEN;
64  } while (MPLS_BOTTOM(shim) == 0);
65 
66  label = MPLS_LABEL(shim);
67  if (label == MPLS_LABEL_IPV4) {
68  return DecodeIPV4(tv, dtv, p, pkt, len, pq);
69  }
70  else if (label == MPLS_LABEL_ROUTER_ALERT) {
71  /* Not valid at the bottom of the stack. */
72  event = MPLS_BAD_LABEL_ROUTER_ALERT;
73  }
74  else if (label == MPLS_LABEL_IPV6) {
75  return DecodeIPV6(tv, dtv, p, pkt, len, pq);
76  }
77  else if (label == MPLS_LABEL_NULL) {
78  /* Shouldn't appear on the wire. */
79  event = MPLS_BAD_LABEL_IMPLICIT_NULL;
80  }
81  else if (label < MPLS_MAX_RESERVED_LABEL) {
82  event = MPLS_BAD_LABEL_RESERVED;
83  }
84 
85  if (event) {
86  goto end;
87  }
88 
89  /* Best guess at inner packet. */
90  switch (pkt[0] >> 4) {
91  case MPLS_PROTO_IPV4:
92  DecodeIPV4(tv, dtv, p, pkt, len, pq);
93  break;
94  case MPLS_PROTO_IPV6:
95  DecodeIPV6(tv, dtv, p, pkt, len, pq);
96  break;
97  case MPLS_PROTO_ETHERNET_PW:
98  DecodeEthernet(tv, dtv, p, pkt + MPLS_PW_LEN, len - MPLS_PW_LEN,
99  pq);
100  break;
101  default:
102  ENGINE_SET_INVALID_EVENT(p, MPLS_UNKNOWN_PAYLOAD_TYPE);
103  return TM_ECODE_OK;
104  }
105 
106 end:
107  if (event) {
108  ENGINE_SET_EVENT(p, event);
109  }
110  return TM_ECODE_OK;
111 }
