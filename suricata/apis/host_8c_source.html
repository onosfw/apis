129 /** \brief initialize the configuration
130  * \warning Not thread safe */
131 void HostInitConfig(char quiet)
132 {
133  SCLogDebug("initializing host engine...");
134  if (HostStorageSize() > 0)
135  g_host_size = sizeof(Host) + HostStorageSize();
136 
137  memset(&host_config, 0, sizeof(host_config));
138  //SC_ATOMIC_INIT(flow_flags);
139  SC_ATOMIC_INIT(host_counter);
140  SC_ATOMIC_INIT(host_memuse);
141  SC_ATOMIC_INIT(host_prune_idx);
142  HostQueueInit(&host_spare_q);
143 
144  unsigned int seed = RandomTimePreseed();
145  /* set defaults */
146  host_config.hash_rand = (int)( HOST_DEFAULT_HASHSIZE * (rand_r(&seed) / RAND_MAX + 1.0));
147 
148  host_config.hash_size = HOST_DEFAULT_HASHSIZE;
149  host_config.memcap = HOST_DEFAULT_MEMCAP;
150  host_config.prealloc = HOST_DEFAULT_PREALLOC;
151 
152  /* Check if we have memcap and hash_size defined at config */
153  char *conf_val;
154  uint32_t configval = 0;
155 
156  /** set config values for memcap, prealloc and hash_size */
157  if ((ConfGet("host.memcap", &conf_val)) == 1)
158  {
159  if (ParseSizeStringU64(conf_val, &host_config.memcap) < 0) {
160  SCLogError(SC_ERR_SIZE_PARSE, "Error parsing host.memcap "
161  "from conf file - %s. Killing engine",
162  conf_val);
163  exit(EXIT_FAILURE);
164  }
165  }
166  if ((ConfGet("host.hash-size", &conf_val)) == 1)
167  {
168  if (ByteExtractStringUint32(&configval, 10, strlen(conf_val),
169  conf_val) > 0) {
170  host_config.hash_size = configval;
171  }
172  }
173 
174  if ((ConfGet("host.prealloc", &conf_val)) == 1)
175  {
176  if (ByteExtractStringUint32(&configval, 10, strlen(conf_val),
177  conf_val) > 0) {
178  host_config.prealloc = configval;
179  } else {
180  WarnInvalidConfEntry("host.prealloc", "%"PRIu32, host_config.prealloc);
181  }
182  }
183  SCLogDebug("Host config from suricata.yaml: memcap: %"PRIu64", hash-size: "
184  "%"PRIu32", prealloc: %"PRIu32, host_config.memcap,
185  host_config.hash_size, host_config.prealloc);
186 
187  /* alloc hash memory */
188  uint64_t hash_size = host_config.hash_size * sizeof(HostHashRow);
189  if (!(HOST_CHECK_MEMCAP(hash_size))) {
190  SCLogError(SC_ERR_HOST_INIT, "allocating host hash failed: "
191  "max host memcap is smaller than projected hash size. "
192  "Memcap: %"PRIu64", Hash table size %"PRIu64". Calculate "
193  "total hash size by multiplying \"host.hash-size\" with %"PRIuMAX", "
194  "which is the hash bucket size.", host_config.memcap, hash_size,
195  (uintmax_t)sizeof(HostHashRow));
196  exit(EXIT_FAILURE);
197  }
198  host_hash = SCCalloc(host_config.hash_size, sizeof(HostHashRow));
199  if (unlikely(host_hash == NULL)) {
200  SCLogError(SC_ERR_FATAL, "Fatal error encountered in HostInitConfig. Exiting...");
201  exit(EXIT_FAILURE);
202  }
203  memset(host_hash, 0, host_config.hash_size * sizeof(HostHashRow));
204 
205  uint32_t i = 0;
206  for (i = 0; i < host_config.hash_size; i++) {
207  HRLOCK_INIT(&host_hash[i]);
208  }
209  (void) SC_ATOMIC_ADD(host_memuse, (host_config.hash_size * sizeof(HostHashRow)));
210 
211  if (quiet == FALSE) {
212  SCLogInfo("allocated %llu bytes of memory for the host hash... "
213  "%" PRIu32 " buckets of size %" PRIuMAX "",
214  SC_ATOMIC_GET(host_memuse), host_config.hash_size,
215  (uintmax_t)sizeof(HostHashRow));
216  }
217 
218  /* pre allocate hosts */
219  for (i = 0; i < host_config.prealloc; i++) {
220  if (!(HOST_CHECK_MEMCAP(g_host_size))) {
221  SCLogError(SC_ERR_HOST_INIT, "preallocating hosts failed: "
222  "max host memcap reached. Memcap %"PRIu64", "
223  "Memuse %"PRIu64".", host_config.memcap,
224  ((uint64_t)SC_ATOMIC_GET(host_memuse) + g_host_size));
225  exit(EXIT_FAILURE);
226  }
227 
228  Host *h = HostAlloc();
229  if (h == NULL) {
230  SCLogError(SC_ERR_HOST_INIT, "preallocating host failed: %s", strerror(errno));
231  exit(EXIT_FAILURE);
232  }
233  HostEnqueue(&host_spare_q,h);
234  }
235 
236  if (quiet == FALSE) {
237  SCLogInfo("preallocated %" PRIu32 " hosts of size %" PRIu16 "",
238  host_spare_q.len, g_host_size);
239  SCLogInfo("host memory usage: %llu bytes, maximum: %"PRIu64,
240  SC_ATOMIC_GET(host_memuse), host_config.memcap);
241  }
242 
243  return;
244 }

259 /** \brief shutdown the flow engine
260  * \warning Not thread safe */
261 void HostShutdown(void)
262 {
263  Host *h;
264  uint32_t u;
265 
266  HostPrintStats();
267 
268  /* free spare queue */
269  while((h = HostDequeue(&host_spare_q))) {
270  BUG_ON(SC_ATOMIC_GET(h->use_cnt) > 0);
271  HostFree(h);
272  }
273 
274  /* clear and free the hash */
275  if (host_hash != NULL) {
276  for (u = 0; u < host_config.hash_size; u++) {
277  Host *h = host_hash[u].head;
278  while (h) {
279  Host *n = h->hnext;
280  HostFree(h);
281  h = n;
282  }
283 
284  HRLOCK_DESTROY(&host_hash[u]);
285  }
286  SCFree(host_hash);
287  host_hash = NULL;
288  }
289  (void) SC_ATOMIC_SUB(host_memuse, host_config.hash_size * sizeof(HostHashRow));
290  HostQueueDestroy(&host_spare_q);
291 
292  SC_ATOMIC_DESTROY(host_prune_idx);
293  SC_ATOMIC_DESTROY(host_memuse);
294  SC_ATOMIC_DESTROY(host_counter);
295  //SC_ATOMIC_DESTROY(flow_flags);
296  return;
297 }

299 /** \brief Cleanup the host engine
300  *
301  * Cleanup the host engine from tag and threshold.
302  *
303  */
304 void HostCleanup(void)
305 {
306  Host *h;
307  uint32_t u;
308 
309  if (host_hash != NULL) {
310  for (u = 0; u < host_config.hash_size; u++) {
311  h = host_hash[u].head;
312  HostHashRow *hb = &host_hash[u];
313  HRLOCK_LOCK(hb);
314  while (h) {
315  if ((SC_ATOMIC_GET(h->use_cnt) > 0) && (h->iprep != NULL)) {
316  /* iprep is attached to host only clear local storage */
317  HostFreeStorage(h);
318  h = h->hnext;
319  } else {
320  Host *n = h->hnext;
321  /* remove from the hash */
322  if (h->hprev != NULL)
323  h->hprev->hnext = h->hnext;
324  if (h->hnext != NULL)
325  h->hnext->hprev = h->hprev;
326  if (hb->head == h)
327  hb->head = h->hnext;
328  if (hb->tail == h)
329  hb->tail = h->hprev;
330  h->hnext = NULL;
331  h->hprev = NULL;
332  HostClearMemory(h);
333  HostMoveToSpare(h);
334  h = n;
335  }
336  }
337  HRLOCK_UNLOCK(hb);
338  }
339  }
340 
341  return;
342 }

344 /* calculate the hash key for this packet
345  *
346  * we're using:
347  * hash_rand -- set at init time
348  * source address
349  */
350 uint32_t HostGetKey(Address *a)
351 {
352  uint32_t key;
353 
354  if (a->family == AF_INET) {
355  uint32_t hash = hashword(&a->addr_data32[0], 1, host_config.hash_rand);
356  key = hash % host_config.hash_size;
357  } else if (a->family == AF_INET6) {
358  uint32_t hash = hashword(a->addr_data32, 4, host_config.hash_rand);
359  key = hash % host_config.hash_size;
360  } else
361  key = 0;
362 
363  return key;
364 }

378  *
379  * Get a new host. We're checking memcap first and will try to make room
380  * if the memcap is reached.
381  *
382  * \retval h *LOCKED* host on succes, NULL on error.
383  */
384 static Host *HostGetNew(Address *a)
385 {
386  Host *h = NULL;
387 
388  /* get a host from the spare queue */
389  h = HostDequeue(&host_spare_q);
390  if (h == NULL) {
391  /* If we reached the max memcap, we get a used host */
392  if (!(HOST_CHECK_MEMCAP(g_host_size))) {
393  /* declare state of emergency */
394  //if (!(SC_ATOMIC_GET(host_flags) & HOST_EMERGENCY)) {
395  // SC_ATOMIC_OR(host_flags, HOST_EMERGENCY);
396 
397  /* under high load, waking up the flow mgr each time leads
398  * to high cpu usage. Flows are not timed out much faster if
399  * we check a 1000 times a second. */
400  // FlowWakeupFlowManagerThread();
401  //}
402 
403  h = HostGetUsedHost();
404  if (h == NULL) {
405  return NULL;
406  }

414  }

416  /* host is initialized but *unlocked* */

452 /* HostGetHostFromHash
453  *
454  * Hash retrieval function for hosts. Looks up the hash bucket containing the
455  * host pointer. Then compares the packet with the found host to see if it is
456  * the host we need. If it isn't, walk the list until the right host is found.
457  *
458  * returns a *LOCKED* host or NULL
459  */
460 Host *HostGetHostFromHash (Address *a)
461 {
462  Host *h = NULL;
463 
464  /* get the key to our bucket */
465  uint32_t key = HostGetKey(a);
466  /* get our hash bucket and lock it */
467  HostHashRow *hb = &host_hash[key];
468  HRLOCK_LOCK(hb);
469 
470  /* see if the bucket already has a host */
471  if (hb->head == NULL) {
472  h = HostGetNew(a);
473  if (h == NULL) {
474  HRLOCK_UNLOCK(hb);
475  return NULL;
476  }
477 
478  /* host is locked */
479  hb->head = h;
480  hb->tail = h;
481 
482  /* got one, now lock, initialize and return */
483  HostInit(h,a);
484 
485  HRLOCK_UNLOCK(hb);
486  return h;
487  }
488 
489  /* ok, we have a host in the bucket. Let's find out if it is our host */
490  h = hb->head;
491 
492  /* see if this is the host we are looking for */
493  if (HostCompare(h, a) == 0) {
494  Host *ph = NULL; /* previous host */
495 
496  while (h) {
497  ph = h;
498  h = h->hnext;
499 
500  if (h == NULL) {
501  h = ph->hnext = HostGetNew(a);
502  if (h == NULL) {
503  HRLOCK_UNLOCK(hb);
504  return NULL;
505  }
506  hb->tail = h;
507 
508  /* host is locked */
509 
510  h->hprev = ph;
511 
512  /* initialize and return */
513  HostInit(h,a);
514 
515  HRLOCK_UNLOCK(hb);
516  return h;
517  }
518 
519  if (HostCompare(h, a) != 0) {
520  /* we found our host, lets put it on top of the
521  * hash list -- this rewards active hosts */
522  if (h->hnext) {
523  h->hnext->hprev = h->hprev;
524  }
525  if (h->hprev) {
526  h->hprev->hnext = h->hnext;
527  }
528  if (h == hb->tail) {
529  hb->tail = h->hprev;
530  }
531 
532  h->hnext = hb->head;
533  h->hprev = NULL;
534  hb->head->hprev = h;
535  hb->head = h;
536 
537  /* found our host, lock & return */
538  SCMutexLock(&h->m);
539  (void) HostIncrUsecnt(h);
540  HRLOCK_UNLOCK(hb);
541  return h;
542  }
543  }
544  }
545 
546  /* lock & return */
547  SCMutexLock(&h->m);
548  (void) HostIncrUsecnt(h);
549  HRLOCK_UNLOCK(hb);
550  return h;
551 }

622 /** \internal
623  * \brief Get a host from the hash directly.
624  *
625  * Called in conditions where the spare queue is empty and memcap is reached.
626  *
627  * Walks the hash until a host can be freed. "host_prune_idx" atomic int makes
628  * sure we don't start at the top each time since that would clear the top of
629  * the hash leading to longer and longer search times under high pressure (observed).
630  *
631  * \retval h host or NULL
632  */
633 static Host *HostGetUsedHost(void)
634 {
635  uint32_t idx = SC_ATOMIC_GET(host_prune_idx) % host_config.hash_size;
636  uint32_t cnt = host_config.hash_size;
637 
638  while (cnt--) {
639  if (++idx >= host_config.hash_size)
640  idx = 0;
641 
642  HostHashRow *hb = &host_hash[idx];
643 
644  if (HRLOCK_TRYLOCK(hb) != 0)
645  continue;
646 
647  Host *h = hb->tail;
648  if (h == NULL) {
649  HRLOCK_UNLOCK(hb);
650  continue;
651  }
652 
653  if (SCMutexTrylock(&h->m) != 0) {
654  HRLOCK_UNLOCK(hb);
655  continue;
656  }
657 
658  /** never prune a host that is used by a packets
659  * we are currently processing in one of the threads */
660  if (SC_ATOMIC_GET(h->use_cnt) > 0) {
661  HRLOCK_UNLOCK(hb);
662  SCMutexUnlock(&h->m);
663  continue;
664  }
665 
666  /* remove from the hash */
667  if (h->hprev != NULL)
668  h->hprev->hnext = h->hnext;
669  if (h->hnext != NULL)
670  h->hnext->hprev = h->hprev;
671  if (hb->head == h)
672  hb->head = h->hnext;
673  if (hb->tail == h)
674  hb->tail = h->hprev;
675 
676  h->hnext = NULL;
677  h->hprev = NULL;
678  HRLOCK_UNLOCK(hb);
679 
680  HostClearMemory (h);
681 
682  SCMutexUnlock(&h->m);
683 
684  (void) SC_ATOMIC_ADD(host_prune_idx, (host_config.hash_size - cnt));
685  return h;
686  }
687 
688  return NULL;
689 }
