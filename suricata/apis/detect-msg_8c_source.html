50 static int DetectMsgSetup (DetectEngineCtx *de_ctx, Signature *s, char *msgstr)
51 {
52  char *str = NULL;
53  uint16_t len;
54 
55  if (strlen(msgstr) == 0)
56  goto error;
57 
58  /* strip "'s */
59  if (msgstr[0] == '\"' && msgstr[strlen(msgstr)-1] == '\"') {
60  str = SCStrdup(msgstr+1);
61  if (unlikely(str == NULL))
62  goto error;
63  str[strlen(msgstr)-2] = '\0';
64  } else if (msgstr[1] == '\"' && msgstr[strlen(msgstr)-1] == '\"') {
65  /* XXX do this parsing in a better way */
66  str = SCStrdup(msgstr+2);
67  if (unlikely(str == NULL))
68  goto error;
69  str[strlen(msgstr)-3] = '\0';
70  //printf("DetectMsgSetup: format hack applied: \'%s\'\n", str);
71  } else {
72  SCLogError(SC_ERR_INVALID_VALUE, "format error \'%s\'", msgstr);
73  goto error;
74  }
75 
76  len = strlen(str);
77  if (len == 0)
78  goto error;
79 
80  char converted = 0;
81 
82  {
83  uint16_t i, x;
84  uint8_t escape = 0;
85 
86  /* it doesn't matter if we need to escape or not we remove the extra "\" to mimic snort */
87  for (i = 0, x = 0; i < len; i++) {
88  //printf("str[%02u]: %c\n", i, str[i]);
89  if(!escape && str[i] == '\\') {
90  escape = 1;
91  } else if (escape) {
92  if (str[i] != ':' &&
93  str[i] != ';' &&
94  str[i] != '\\' &&
95  str[i] != '\"')
96  {
97  SCLogDebug("character \"%c\" does not need to be escaped but is" ,str[i]);
98  }
99  escape = 0;
100  converted = 1;
101 
102  str[x] = str[i];
103  x++;
104  }else{
105  str[x] = str[i];
106  x++;
107  }
108 
109  }
110 #if 0 //def DEBUG
111  if (SCLogDebugEnabled()) {
112  for (i = 0; i < x; i++) {
113  printf("%c", str[i]);
114  }
115  printf("\n");
116  }
117 #endif
118 
119  if (converted) {
120  len = x;
121  str[len] = '\0';
122  }
123  }
124 
125  s->msg = SCMalloc(len + 1);
126  if (s->msg == NULL)
127  goto error;
128 
129  strlcpy(s->msg, str, len + 1);
130 
131  SCFree(str);
132  return 0;
133 
134 error:
135  SCFree(str);
136  return -1;
137 }
