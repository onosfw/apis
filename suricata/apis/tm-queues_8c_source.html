49 Tmq* TmqCreateQueue(char *name)
50 {
51  if (tmq_id >= TMQ_MAX_QUEUES)
52  goto error;
53 
54  Tmq *q = &tmqs[tmq_id];
55  q->name = name;
56  q->id = tmq_id++;
57  /* for cuda purposes */
58  q->q_type = 0;
59 
60  SCLogDebug("created queue \'%s\', %p", name, q);
61  return q;
62 
63 error:

66 }

97 /**
98  * \brief Checks if all the queues allocated so far have at least one reader
99  * and writer.
100  */
101 void TmValidateQueueState(void)
102 {
103  int i = 0;
104  char err = FALSE;
105 
106  for (i = 0; i < tmq_id; i++) {
107  SCMutexLock(&trans_q[tmqs[i].id].mutex_q);
108  if (tmqs[i].reader_cnt == 0) {
109  SCLogError(SC_ERR_THREAD_QUEUE, "queue \"%s\" doesn't have a reader (id %d, max %u)", tmqs[i].name, i, tmq_id);
110  err = TRUE;
111  } else if (tmqs[i].writer_cnt == 0) {
112  SCLogError(SC_ERR_THREAD_QUEUE, "queue \"%s\" doesn't have a writer (id %d, max %u)", tmqs[i].name, i, tmq_id);
113  err = TRUE;
114  }
115  SCMutexUnlock(&trans_q[tmqs[i].id].mutex_q);
116 
117  if (err == TRUE)
118  goto error;
119  }
120 
121  return;
122 
123 error:
124  SCLogError(SC_ERR_FATAL, "fatal error during threading setup");
125  exit(EXIT_FAILURE);
126 }
