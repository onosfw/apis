50 void TmqhFlowRegister(void)
51 {
52  tmqh_table[TMQH_FLOW].name = "flow";
53  tmqh_table[TMQH_FLOW].InHandler = TmqhInputFlow;
54  tmqh_table[TMQH_FLOW].OutHandlerCtxSetup = TmqhOutputFlowSetupCtx;
55  tmqh_table[TMQH_FLOW].OutHandlerCtxFree = TmqhOutputFlowFreeCtx;
56  tmqh_table[TMQH_FLOW].RegisterTests = TmqhFlowRegisterTests;
57 
58  char *scheduler = NULL;
59  if (ConfGet("autofp-scheduler", &scheduler) == 1) {
60  if (strcasecmp(scheduler, "round-robin") == 0) {
61  SCLogInfo("AutoFP mode using \"Round Robin\" flow load balancer");
62  tmqh_table[TMQH_FLOW].OutHandler = TmqhOutputFlowRoundRobin;
63  } else if (strcasecmp(scheduler, "active-packets") == 0) {
64  SCLogInfo("AutoFP mode using \"Active Packets\" flow load balancer");
65  tmqh_table[TMQH_FLOW].OutHandler = TmqhOutputFlowActivePackets;
66  } else if (strcasecmp(scheduler, "hash") == 0) {
67  SCLogInfo("AutoFP mode using \"Hash\" flow load balancer");
68  tmqh_table[TMQH_FLOW].OutHandler = TmqhOutputFlowHash;
69  } else {
70  SCLogError(SC_ERR_INVALID_YAML_CONF_ENTRY, "Invalid entry \"%s\" "
71  "for autofp-scheduler in conf. Killing engine.",
72  scheduler);
73  exit(EXIT_FAILURE);
74  }
75  } else {
76  SCLogInfo("AutoFP mode using default \"Active Packets\" flow load balancer");
77  tmqh_table[TMQH_FLOW].OutHandler = TmqhOutputFlowActivePackets;
78  }
79 
80  return;
81 }

84 Packet *TmqhInputFlow(ThreadVars *tv)
85 {
86  PacketQueue *q = &trans_q[tv->inq->id];
87 
88  StatsSyncCountersIfSignalled(tv);
89 
90  SCMutexLock(&q->mutex_q);
91  if (q->len == 0) {
92  /* if we have no packets in queue, wait... */
93  SCCondWait(&q->cond_q, &q->mutex_q);
94  }
95 
96  if (q->len > 0) {
97  Packet *p = PacketDequeue(q);
98  SCMutexUnlock(&q->mutex_q);
99  return p;
100  } else {
101  /* return NULL if we have no pkt. Should only happen on signals. */
102  SCMutexUnlock(&q->mutex_q);
103  return NULL;
104  }
105 }

107 static int StoreQueueId(TmqhFlowCtx *ctx, char *name)
108 {
109  void *ptmp;
110  Tmq *tmq = TmqGetQueueByName(name);
111  if (tmq == NULL) {
112  tmq = TmqCreateQueue(SCStrdup(name));
113  if (tmq == NULL)
114  return -1;
115  }
116  tmq->writer_cnt++;
117 
118  uint16_t id = tmq->id;
119 
120  if (ctx->queues == NULL) {
121  ctx->size = 1;
122  ctx->queues = SCMalloc(ctx->size * sizeof(TmqhFlowMode));
123  if (ctx->queues == NULL) {
124  return -1;
125  }
126  memset(ctx->queues, 0, ctx->size * sizeof(TmqhFlowMode));
127  } else {
128  ctx->size++;
129  ptmp = SCRealloc(ctx->queues, ctx->size * sizeof(TmqhFlowMode));
130  if (ptmp == NULL) {
131  SCFree(ctx->queues);
132  ctx->queues = NULL;
133  return -1;
134  }
135  ctx->queues = ptmp;
136 
137  memset(ctx->queues + (ctx->size - 1), 0, sizeof(TmqhFlowMode));
138  }
139  ctx->queues[ctx->size - 1].q = &trans_q[id];
140  SC_ATOMIC_INIT(ctx->queues[ctx->size - 1].total_packets);
141  SC_ATOMIC_INIT(ctx->queues[ctx->size - 1].total_flows);
142 
143  return 0;
144 }

146 /**
147  * \brief setup the queue handlers ctx
148  *
149  * Parses a comma separated string "queuename1,queuename2,etc"
150  * and sets the ctx up to devide flows over these queue's.
151  *
152  * \param queue_str comma separated string with output queue names
153  *
154  * \retval ctx queues handlers ctx or NULL in error
155  */
156 void *TmqhOutputFlowSetupCtx(char *queue_str)
157 {
158  if (queue_str == NULL || strlen(queue_str) == 0)
159  return NULL;
160 
161  SCLogDebug("queue_str %s", queue_str);
162 
163  TmqhFlowCtx *ctx = SCMalloc(sizeof(TmqhFlowCtx));
164  if (unlikely(ctx == NULL))
165  return NULL;
166  memset(ctx,0x00,sizeof(TmqhFlowCtx));
167 
168  char *str = SCStrdup(queue_str);
169  if (unlikely(str == NULL)) {
170  goto error;
171  }
172  char *tstr = str;
173 
174  /* parse the comma separated string */
175  do {
176  char *comma = strchr(tstr,',');
177  if (comma != NULL) {
178  *comma = '\0';
179  char *qname = tstr;
180  int r = StoreQueueId(ctx,qname);
181  if (r < 0)
182  goto error;
183  } else {
184  char *qname = tstr;
185  int r = StoreQueueId(ctx,qname);
186  if (r < 0)
187  goto error;
188  }
189  tstr = comma ? (comma + 1) : comma;
190  } while (tstr != NULL);
191 
192  SC_ATOMIC_INIT(ctx->round_robin_idx);
193 
194  SCFree(str);
195  return (void *)ctx;
196 
197 error:
198  SCFree(ctx);
199  if (str != NULL)
200  SCFree(str);
201  return NULL;
202 }

204 void TmqhOutputFlowFreeCtx(void *ctx)
205 {
206  int i;
207  TmqhFlowCtx *fctx = (TmqhFlowCtx *)ctx;
208 
209  SCLogInfo("AutoFP - Total flow handler queues - %" PRIu16,
210  fctx->size);
211  for (i = 0; i < fctx->size; i++) {
212  SCLogInfo("AutoFP - Queue %-2"PRIu32 " - pkts: %-12"PRIu64" flows: %-12"PRIu64, i,
213  SC_ATOMIC_GET(fctx->queues[i].total_packets),
214  SC_ATOMIC_GET(fctx->queues[i].total_flows));
215  SC_ATOMIC_DESTROY(fctx->queues[i].total_packets);
216  SC_ATOMIC_DESTROY(fctx->queues[i].total_flows);
217  }
218 
219  SCFree(fctx->queues);
220 
221  return;
222 }

224 /**
225  * \brief select the queue to output in a round robin fashion.
226  *
227  * \param tv thread vars
228  * \param p packet
229  */
230 void TmqhOutputFlowRoundRobin(ThreadVars *tv, Packet *p)
231 {
232  int16_t qid = 0;
233 
234  TmqhFlowCtx *ctx = (TmqhFlowCtx *)tv->outctx;
235 
236  /* if no flow we use the first queue,
237  * should be rare */
238  if (p->flow != NULL) {
239  qid = SC_ATOMIC_GET(p->flow->autofp_tmqh_flow_qid);
240  if (qid == -1) {
241  qid = SC_ATOMIC_ADD(ctx->round_robin_idx, 1);
242  if (qid >= ctx->size) {
243  SC_ATOMIC_RESET(ctx->round_robin_idx);
244  qid = 0;
245  }
246  (void) SC_ATOMIC_ADD(ctx->queues[qid].total_flows, 1);
247  (void) SC_ATOMIC_SET(p->flow->autofp_tmqh_flow_qid, qid);
248  }
249  } else {
250  qid = ctx->last++;
251 
252  if (ctx->last == ctx->size)
253  ctx->last = 0;
254  }
255  (void) SC_ATOMIC_ADD(ctx->queues[qid].total_packets, 1);
256 
257  PacketQueue *q = ctx->queues[qid].q;
258  SCMutexLock(&q->mutex_q);
259  PacketEnqueue(q, p);
260  SCCondSignal(&q->cond_q);
261  SCMutexUnlock(&q->mutex_q);
262 
263  return;
264 }

267  * \brief select the queue to output to based on queue lengths.
268  *
269  * \param tv thread vars
270  * \param p packet

272 void TmqhOutputFlowActivePackets(ThreadVars *tv, Packet *p)
273 {
274  int16_t qid = 0;
275 
276  TmqhFlowCtx *ctx = (TmqhFlowCtx *)tv->outctx;
277 
278  /* if no flow we use the first queue,
279  * should be rare */
280  if (p->flow != NULL) {
281  qid = SC_ATOMIC_GET(p->flow->autofp_tmqh_flow_qid);
282  if (qid == -1) {
283  uint16_t i = 0;
284  int lowest_id = 0;
285  TmqhFlowMode *queues = ctx->queues;
286  uint32_t lowest = queues[i].q->len;
287  for (i = 1; i < ctx->size; i++) {
288  if (queues[i].q->len < lowest) {
289  lowest = queues[i].q->len;
290  lowest_id = i;
291  }
292  }
293  qid = lowest_id;
294  (void) SC_ATOMIC_SET(p->flow->autofp_tmqh_flow_qid, lowest_id);
295  (void) SC_ATOMIC_ADD(ctx->queues[qid].total_flows, 1);
296  }
297  } else {
298  qid = ctx->last++;
299 
300  if (ctx->last == ctx->size)
301  ctx->last = 0;
302  }
303  (void) SC_ATOMIC_ADD(ctx->queues[qid].total_packets, 1);
304 
305  PacketQueue *q = ctx->queues[qid].q;
306  SCMutexLock(&q->mutex_q);
307  PacketEnqueue(q, p);
308  SCCondSignal(&q->cond_q);
309  SCMutexUnlock(&q->mutex_q);
310 
311  return;
312 }

314 /**
315  * \brief select the queue to output based on address hash.
316  *
317  * \param tv thread vars.
318  * \param p packet.
319  */
320 void TmqhOutputFlowHash(ThreadVars *tv, Packet *p)
321 {
322  int16_t qid = 0;
323 
324  TmqhFlowCtx *ctx = (TmqhFlowCtx *)tv->outctx;
325 
326  /* if no flow we use the first queue,
327  * should be rare */
328  if (p->flow != NULL) {
329  qid = SC_ATOMIC_GET(p->flow->autofp_tmqh_flow_qid);
330  if (qid == -1) {
331 #if __WORDSIZE == 64
332  uint64_t addr = (uint64_t)p->flow;
333 #else
334  uint32_t addr = (uint32_t)p->flow;
335 #endif
336  addr >>= 7;
337 
338  /* we don't have to worry about possible overflow, since
339  * ctx->size will be lesser than 2 ** 31 for sure */
340  qid = addr % ctx->size;
341  (void) SC_ATOMIC_SET(p->flow->autofp_tmqh_flow_qid, qid);
342  (void) SC_ATOMIC_ADD(ctx->queues[qid].total_flows, 1);
343  }
344  } else {
345  qid = ctx->last++;
346 
347  if (ctx->last == ctx->size)
348  ctx->last = 0;
349  }
350  (void) SC_ATOMIC_ADD(ctx->queues[qid].total_packets, 1);
351 
352  PacketQueue *q = ctx->queues[qid].q;
353  SCMutexLock(&q->mutex_q);
354  PacketEnqueue(q, p);
355  SCCondSignal(&q->cond_q);
356  SCMutexUnlock(&q->mutex_q);
357 
358  return;
359 }
