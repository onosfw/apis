116 /**
117  * \internal
118  * \brief This function is used to parse threshold options passed via threshold: keyword
119  *
120  * \param rawstr Pointer to the user provided threshold options
121  *
122  * \retval de pointer to DetectThresholdData on success
123  * \retval NULL on failure
124  */
125 static DetectThresholdData *DetectThresholdParse(char *rawstr)
126 {
127  DetectThresholdData *de = NULL;
128 #define MAX_SUBSTRINGS 30
129  int ret = 0, res = 0;
130  int ov[MAX_SUBSTRINGS];
131  const char *str_ptr = NULL;
132  char *args[9] = { NULL, NULL, NULL, NULL, NULL, NULL, NULL, NULL, NULL };
133  char *copy_str = NULL, *threshold_opt = NULL;
134  int second_found = 0, count_found = 0;
135  int type_found = 0, track_found = 0;
136  int second_pos = 0, count_pos = 0;
137  uint16_t pos = 0;
138  int i = 0;
139 
140  copy_str = SCStrdup(rawstr);
141  if (unlikely(copy_str == NULL)) {
142  goto error;
143  }
144 
145  char *saveptr = NULL;
146  for (pos = 0, threshold_opt = strtok_r(copy_str,",", &saveptr);
147  pos < strlen(copy_str) && threshold_opt != NULL;
148  pos++, threshold_opt = strtok_r(NULL,"," , &saveptr))
149  {
150  if(strstr(threshold_opt,"count"))
151  count_found++;
152  if(strstr(threshold_opt,"second"))
153  second_found++;
154  if(strstr(threshold_opt,"type"))
155  type_found++;
156  if(strstr(threshold_opt,"track"))
157  track_found++;
158  }
159  SCFree(copy_str);
160  copy_str = NULL;
161 
162  if(count_found != 1 || second_found != 1 || type_found != 1 || track_found != 1)
163  goto error;
164 
165  ret = pcre_exec(parse_regex, parse_regex_study, rawstr, strlen(rawstr), 0, 0, ov, MAX_SUBSTRINGS);
166 
167  if (ret < 5) {
168  SCLogError(SC_ERR_PCRE_MATCH, "pcre_exec parse error, ret %" PRId32 ", string %s", ret, rawstr);
169  goto error;
170  }
171 
172  de = SCMalloc(sizeof(DetectThresholdData));
173  if (unlikely(de == NULL))
174  goto error;
175 
176  memset(de,0,sizeof(DetectThresholdData));
177 
178  for (i = 0; i < (ret - 1); i++) {
179 
180  res = pcre_get_substring((char *)rawstr, ov, MAX_SUBSTRINGS,i + 1, &str_ptr);
181 
182  if (res < 0) {
183  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
184  goto error;
185  }
186 
187  args[i] = (char *)str_ptr;
188 
189  if (strncasecmp(args[i],"limit",strlen("limit")) == 0)
190  de->type = TYPE_LIMIT;
191  if (strncasecmp(args[i],"both",strlen("both")) == 0)
192  de->type = TYPE_BOTH;
193  if (strncasecmp(args[i],"threshold",strlen("threshold")) == 0)
194  de->type = TYPE_THRESHOLD;
195  if (strncasecmp(args[i],"by_dst",strlen("by_dst")) == 0)
196  de->track = TRACK_DST;
197  if (strncasecmp(args[i],"by_src",strlen("by_src")) == 0)
198  de->track = TRACK_SRC;
199  if (strncasecmp(args[i],"count",strlen("count")) == 0)
200  count_pos = i+1;
201  if (strncasecmp(args[i],"seconds",strlen("seconds")) == 0)
202  second_pos = i+1;
203  }
204 
205  if (args[count_pos] == NULL || args[second_pos] == NULL) {
206  goto error;
207  }
208 
209  if (ByteExtractStringUint32(&de->count, 10, strlen(args[count_pos]),
210  args[count_pos]) <= 0) {
211  goto error;
212  }
213 
214  if (ByteExtractStringUint32(&de->seconds, 10, strlen(args[second_pos]),
215  args[second_pos]) <= 0) {
216  goto error;
217  }
218 
219  for (i = 0; i < (ret - 1); i++){
220  if (args[i] != NULL) SCFree(args[i]);
221  }
222  return de;
223 
224 error:
225  for (i = 0; i < (ret - 1); i++){
226  if (args[i] != NULL) SCFree(args[i]);
227  }
228  if (de != NULL)
229  SCFree(de);
230  return NULL;
231 }

244 static int DetectThresholdSetup(DetectEngineCtx *de_ctx, Signature *s, char *rawstr)
245 {

253  if (tmpm != NULL) {

272 
273 error:
