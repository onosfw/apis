73 
74  regex_study = pcre_study(regex, 0, &eb);
75  if (eb != NULL) {
76  SCLogError(SC_ERR_PCRE_STUDY, "pcre study failed: %s", eb);

84 /**
85  * \brief Parses the raw string supplied with the "Classtype" keyword.
86  *
87  * \param Pointer to the string to be parsed.
88  *
89  * \retval bool success or failure.
90  */
91 static int DetectClasstypeParseRawString(char *rawstr, char *out, size_t outsize)
92 {
93 #define MAX_SUBSTRINGS 30
94  int ret = 0;
95  int ov[MAX_SUBSTRINGS];
96  size_t len = strlen(rawstr);
97 
98  /* get rid of the double quotes if present */
99  if (rawstr[0] == '\"' && rawstr[strlen(rawstr) - 1] == '\"') {
100  rawstr++;
101  rawstr[strlen(rawstr) - 1] = '\0';
102  len -= 2;
103  }
104 
105  ret = pcre_exec(regex, regex_study, rawstr, len, 0, 0, ov, 30);
106  if (ret < 0) {
107  SCLogError(SC_ERR_PCRE_MATCH, "Invalid Classtype in Signature");
108  goto end;
109  }
110 
111  ret = pcre_copy_substring((char *)rawstr, ov, 30, 1, out, outsize);
112  if (ret < 0) {
113  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_copy_substring failed");
114  goto end;
115  }
116 
117  return 0;
118  end:
119  return -1;
120 }

122 /**
123  * \brief The setup function that would be called when the Signature parsing
124  * module encounters the "Classtype" keyword.
125  *
126  * \param de_ctx Pointer to the Detection Engine Context.
127  * \param s Pointer the current Signature instance that is being parsed.
128  * \param rawstr Pointer to the argument supplied to the classtype keyword.
129  *
130  * \retval 0 On success
131  * \retval -1 On failure
132  */
133 static int DetectClasstypeSetup(DetectEngineCtx *de_ctx, Signature *s, char *rawstr)
134 {
135  char parsed_ct_name[1024] = "";
136  SCClassConfClasstype *ct = NULL;
137 
138  if (DetectClasstypeParseRawString(rawstr, parsed_ct_name, sizeof(parsed_ct_name)) < -1) {
139  SCLogError(SC_ERR_PCRE_PARSE, "Error parsing classtype argument supplied with the "
140  "classtype keyword");
141  goto error;
142  }
143 
144  ct = SCClassConfGetClasstype(parsed_ct_name, de_ctx);
145  if (ct == NULL) {
146  SCLogError(SC_ERR_UNKNOWN_VALUE, "Unknown Classtype: \"%s\". Invalidating the Signature",
147  parsed_ct_name);
148  goto error;
149  }
150 
151  /* if we have retrieved the classtype, assign the message to be displayed
152  * for this Signature by fast.log, if a Packet matches this Signature */
153  s->class = ct->classtype_id;
154  s->class_msg = ct->classtype_desc;
155 
156  /* if a priority keyword has appeared before the classtype, s->prio would
157  * hold a value which is != -1, in which case we don't overwrite the value.
158  * Otherwise, overwrite the value */
159  if (s->prio == -1)
160  s->prio = ct->priority;
161 
162  return 0;
163 
164  error:
165  return -1;
166 }
