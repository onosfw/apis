43 /**
44  * Decompress a chunk of gzip-compressed data.
45  *
46  * @param[in] drec
47  * @param[in] d
48  * @return HTP_OK on success, HTP_ERROR or some other negative integer on failure.
49  */
50 static htp_status_t htp_gzip_decompressor_decompress(htp_decompressor_gzip_t *drec, htp_tx_data_t *d) {
51  size_t consumed = 0;
52 
53  // Pass-through the NULL chunk, which indicates the end of the stream.
54 
55  if (d->data == NULL) {
56  // Prepare data for callback.
57  htp_tx_data_t dout;
58  dout.tx = d->tx;
59  dout.data = NULL;
60  dout.len = 0;
61 
62  // Send decompressed data to the callback.
63  htp_status_t callback_rc = drec->super.callback(&dout);
64  if (callback_rc != HTP_OK) {
65  inflateEnd(&drec->stream);
66  drec->zlib_initialized = 0;
67 
68  return callback_rc;
69  }
70 
71  return HTP_OK;
72  }
73 
74  // Decompress data.
75  int rc = 0;
76  drec->stream.next_in = (unsigned char *) (d->data + consumed);
77  drec->stream.avail_in = d->len - consumed;
78 
79  while (drec->stream.avail_in != 0) {
80  // If there's no more data left in the
81  // buffer, send that information out.
82  if (drec->stream.avail_out == 0) {
83  drec->crc = crc32(drec->crc, drec->buffer, GZIP_BUF_SIZE);
84 
85  // Prepare data for callback.
86  htp_tx_data_t d2;
87  d2.tx = d->tx;
88  d2.data = drec->buffer;
89  d2.len = GZIP_BUF_SIZE;
90 
91  // Send decompressed data to callback.
92  htp_status_t callback_rc = drec->super.callback(&d2);
93  if (callback_rc != HTP_OK) {
94  inflateEnd(&drec->stream);
95  drec->zlib_initialized = 0;
96 
97  return callback_rc;
98  }
99 
100  drec->stream.next_out = drec->buffer;
101  drec->stream.avail_out = GZIP_BUF_SIZE;
102  }
103 
104  rc = inflate(&drec->stream, Z_NO_FLUSH);
105 
106  if (rc == Z_STREAM_END) {
107  // How many bytes do we have?
108  size_t len = GZIP_BUF_SIZE - drec->stream.avail_out;
109 
110  // Update CRC
111  drec->crc = crc32(drec->crc, drec->buffer, len);
112 
113  // Prepare data for the callback.
114  htp_tx_data_t d2;
115  d2.tx = d->tx;
116  d2.data = drec->buffer;
117  d2.len = len;
118 
119  // Send decompressed data to the callback.
120  htp_status_t callback_rc = drec->super.callback(&d2);
121  if (callback_rc != HTP_OK) {
122  inflateEnd(&drec->stream);
123  drec->zlib_initialized = 0;
124 
125  return callback_rc;
126  }
127 
128  // TODO Handle trailer.
129 
130  return HTP_OK;
131  }
132 
133  if (rc != Z_OK) {
134  htp_log(d->tx->connp, HTP_LOG_MARK, HTP_LOG_WARNING, 0, "GZip decompressor: inflate failed with %d", rc);
135 
136  inflateEnd(&drec->stream);
137  drec->zlib_initialized = 0;
138 
139  return HTP_ERROR;
140  }
141  }
142 
143  return HTP_OK;
144 }

163 /**
164  * Create a new decompressor instance.
165  *
166  * @param[in] connp
167  * @param[in] format
168  * @return New htp_decompressor_t instance on success, or NULL on failure.
169  */
170 htp_decompressor_t *htp_gzip_decompressor_create(htp_connp_t *connp, enum htp_content_encoding_t format) {
171  htp_decompressor_gzip_t *drec = calloc(1, sizeof (htp_decompressor_gzip_t));
172  if (drec == NULL) return NULL;
173 
174  drec->super.decompress = (int (*)(htp_decompressor_t *, htp_tx_data_t *))htp_gzip_decompressor_decompress;
175  drec->super.destroy = (void (*)(htp_decompressor_t *))htp_gzip_decompressor_destroy;
176 
177  drec->buffer = malloc(GZIP_BUF_SIZE);
178  if (drec->buffer == NULL) {
179  free(drec);
180  return NULL;
181  }
182 
183  // Initialize zlib.
184  int rc;
185 
186  if (format == HTP_COMPRESSION_DEFLATE) {
187  // Negative values activate raw processing,
188  // which is what we need for deflate.
189  rc = inflateInit2(&drec->stream, -15);
190  } else {
191  // Increased windows size activates gzip header processing.
192  rc = inflateInit2(&drec->stream, 15 + 32);
193  }
194 
195  if (rc != Z_OK) {
196  htp_log(connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0, "GZip decompressor: inflateInit2 failed with code %d", rc);
197 
198  inflateEnd(&drec->stream);
199  free(drec->buffer);
200  free(drec);
201 
202  return NULL;
203  }
204 
205  drec->zlib_initialized = 1;
206  drec->stream.avail_out = GZIP_BUF_SIZE;
207  drec->stream.next_out = drec->buffer;
208 
209  #if 0
210  if (format == COMPRESSION_DEFLATE) {
211  drec->initialized = 1;
212  }
213  #endif
214 
215  return (htp_decompressor_t *) drec;
216 }
