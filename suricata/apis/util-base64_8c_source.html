66  * \brief Decodes a 4-byte base64-encoded block into a 3-byte ascii-encoded block

69  * \param b64 the 4-byte base64 input block

71  * \return none

73 static inline void DecodeBase64Block(uint8_t ascii[ASCII_BLOCK], uint8_t b64[B64_BLOCK]) {
74 
75  ascii[0] = (uint8_t) (b64[0] << 2) | (b64[1] >> 4);
76  ascii[1] = (uint8_t) (b64[1] << 4) | (b64[2] >> 2);
77  ascii[2] = (uint8_t) (b64[2] << 6) | (b64[3]);
78 }

80 /**
81  * \brief Decodes a base64-encoded string buffer into an ascii-encoded byte buffer
82  *
83  * \param dest The destination byte buffer
84  * \param src The source string
85  * \param len The length of the source string
86  *
87  * \return Number of bytes decoded, or 0 if no data is decoded or it fails
88  */
89 uint32_t DecodeBase64(uint8_t *dest, const uint8_t *src, uint32_t len) {
90 
91  int val;
92  uint32_t padding = 0, numDecoded = 0, bbidx = 0, valid = 1, i;
93  uint8_t *dptr = dest;
94  uint8_t b64[B64_BLOCK] = { 0,0,0,0 };
95 
96  /* Traverse through each alpha-numeric letter in the source array */
97  for(i = 0; i < len && src[i] != 0; i++) {
98 
99  /* Get decimal representation */
100  val = GetBase64Value(src[i]);
101  if (val < 0) {
102 
103  /* Invalid character found, so decoding fails */
104  if (src[i] != '=') {
105  valid = 0;
106  numDecoded = 0;
107  break;
108  }
109  padding++;
110  }
111 
112  /* For each alpha-numeric letter in the source array, find the numeric
113  * value */
114  b64[bbidx++] = (val > 0 ? val : 0);
115 
116  /* Decode every 4 base64 bytes into 3 ascii bytes */
117  if (bbidx == B64_BLOCK) {
118 
119  /* For every 4 bytes, add 3 bytes but deduct the '=' padded blocks */
120  numDecoded += ASCII_BLOCK - (padding < B64_BLOCK ?
121  padding : ASCII_BLOCK);
122 
123  /* Decode base-64 block into ascii block and move pointer */
124  DecodeBase64Block(dptr, b64);
125  dptr += ASCII_BLOCK;
126 
127  /* Reset base-64 block and index */
128  bbidx = 0;
129  padding = 0;
130  }
131  }
132 
133  /* Finish remaining b64 bytes by padding */
134  if (valid && bbidx > 0) {
135 
136  /* Decode remaining */
137  numDecoded += ASCII_BLOCK - (B64_BLOCK - bbidx);
138  DecodeBase64Block(dptr, b64);
139  }
140 
141  if (numDecoded == 0) {
142  SCLogDebug("base64 decoding failed");
143  }
144 
145  return numDecoded;
146 }
