69 /**
70  * \brief Function to log the FlowVars in to alert-debug.log
71  *
72  * \param aft Pointer to AltertDebugLog Thread
73  * \param p Pointer to the packet
74  *
75  */
76 static void AlertDebugLogFlowVars(AlertDebugLogThread *aft, const Packet *p)
77 {
78  const GenericVar *gv = p->flow->flowvar;
79  uint16_t i;
80  while (gv != NULL) {
81  if (gv->type == DETECT_FLOWVAR || gv->type == DETECT_FLOWINT) {
82  FlowVar *fv = (FlowVar *) gv;
83 
84  if (fv->datatype == FLOWVAR_TYPE_STR) {
85  MemBufferWriteString(aft->buffer, "FLOWVAR idx(%"PRIu32"): ",
86  fv->idx);
87  for (i = 0; i < fv->data.fv_str.value_len; i++) {
88  if (isprint(fv->data.fv_str.value[i])) {
89  MemBufferWriteString(aft->buffer, "%c",
90  fv->data.fv_str.value[i]);
91  } else {
92  MemBufferWriteString(aft->buffer, "\\%02X",
93  fv->data.fv_str.value[i]);
94  }
95  }
96  } else if (fv->datatype == FLOWVAR_TYPE_INT) {
97  MemBufferWriteString(aft->buffer, "FLOWVAR idx(%"PRIu32"): "
98  " %" PRIu32 "\"", fv->idx, fv->data.fv_int.value);
99  }
100  }
101  gv = gv->next;
102  }
103 }

165 static TmEcode AlertDebugLogger(ThreadVars *tv, const Packet *p, void *thread_data)
166 {
167  AlertDebugLogThread *aft = (AlertDebugLogThread *)thread_data;
168  int i;
169  char timebuf[64];
170  const char *pkt_src_str = NULL;
171 
172  if (p->alerts.cnt == 0)
173  return TM_ECODE_OK;
174 
175  MemBufferReset(aft->buffer);
176 
177  CreateTimeString(&p->ts, timebuf, sizeof(timebuf));
178 
179  MemBufferWriteString(aft->buffer, "+================\n"
180  "TIME: %s\n", timebuf);
181  if (p->pcap_cnt > 0) {
182  MemBufferWriteString(aft->buffer, "PCAP PKT NUM: %"PRIu64"\n", p->pcap_cnt);
183  }
184  pkt_src_str = PktSrcToString(p->pkt_src);
185  MemBufferWriteString(aft->buffer, "PKT SRC: %s\n", pkt_src_str);
186 
187  char srcip[46], dstip[46];
188  if (PKT_IS_IPV4(p)) {
189  PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p), srcip, sizeof(srcip));
190  PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p), dstip, sizeof(dstip));
191  } else if (PKT_IS_IPV6(p)) {
192  PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p), srcip, sizeof(srcip));
193  PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p), dstip, sizeof(dstip));
194  }
195 
196  MemBufferWriteString(aft->buffer, "SRC IP: %s\n"
197  "DST IP: %s\n"
198  "PROTO: %" PRIu32 "\n",
199  srcip, dstip, p->proto);
200  if (PKT_IS_TCP(p) || PKT_IS_UDP(p)) {
201  MemBufferWriteString(aft->buffer, "SRC PORT: %" PRIu32 "\n"
202  "DST PORT: %" PRIu32 "\n",
203  p->sp, p->dp);
204  if (PKT_IS_TCP(p)) {
205  MemBufferWriteString(aft->buffer, "TCP SEQ: %"PRIu32"\n"
206  "TCP ACK: %"PRIu32"\n",
207  TCP_GET_SEQ(p), TCP_GET_ACK(p));
208  }
209  }
210 
211  /* flow stuff */
212  MemBufferWriteString(aft->buffer, "FLOW: to_server: %s, "
213  "to_client: %s\n",
214  p->flowflags & FLOW_PKT_TOSERVER ? "TRUE" : "FALSE",
215  p->flowflags & FLOW_PKT_TOCLIENT ? "TRUE" : "FALSE");
216 
217  if (p->flow != NULL) {
218  int applayer = 0;
219  FLOWLOCK_RDLOCK(p->flow);
220  applayer = StreamTcpAppLayerIsDisabled(p->flow);
221  CreateTimeString(&p->flow->startts, timebuf, sizeof(timebuf));
222  MemBufferWriteString(aft->buffer, "FLOW Start TS: %s\n", timebuf);
223  MemBufferWriteString(aft->buffer, "FLOW PKTS TODST: %"PRIu32"\n"
224  "FLOW PKTS TOSRC: %"PRIu32"\n"
225  "FLOW Total Bytes: %"PRIu64"\n",
226  p->flow->todstpktcnt, p->flow->tosrcpktcnt,
227  p->flow->todstbytecnt + p->flow->tosrcbytecnt);
228  MemBufferWriteString(aft->buffer,
229  "FLOW IPONLY SET: TOSERVER: %s, TOCLIENT: %s\n"
230  "FLOW ACTION: DROP: %s\n"
231  "FLOW NOINSPECTION: PACKET: %s, PAYLOAD: %s, APP_LAYER: %s\n"
232  "FLOW APP_LAYER: DETECTED: %s, PROTO %"PRIu16"\n",
233  p->flow->flags & FLOW_TOSERVER_IPONLY_SET ? "TRUE" : "FALSE",
234  p->flow->flags & FLOW_TOCLIENT_IPONLY_SET ? "TRUE" : "FALSE",
235  p->flow->flags & FLOW_ACTION_DROP ? "TRUE" : "FALSE",
236  p->flow->flags & FLOW_NOPACKET_INSPECTION ? "TRUE" : "FALSE",
237  p->flow->flags & FLOW_NOPAYLOAD_INSPECTION ? "TRUE" : "FALSE",
238  applayer ? "TRUE" : "FALSE",
239  (p->flow->alproto != ALPROTO_UNKNOWN) ? "TRUE" : "FALSE", p->flow->alproto);
240  AlertDebugLogFlowVars(aft, p);
241  AlertDebugLogFlowBits(aft, (Packet *)p); /* < no const */
242  FLOWLOCK_UNLOCK(p->flow);
243  }
244 
245  AlertDebugLogPktVars(aft, p);
246 
247 /* any stuff */
248 /* Sig details? */
249 
250  MemBufferWriteString(aft->buffer,
251  "PACKET LEN: %" PRIu32 "\n"
252  "PACKET:\n",
253  GET_PKT_LEN(p));
254  PrintRawDataToBuffer(aft->buffer->buffer, &aft->buffer->offset, aft->buffer->size,
255  GET_PKT_DATA(p), GET_PKT_LEN(p));
256 
257  MemBufferWriteString(aft->buffer, "ALERT CNT: %" PRIu32 "\n",
258  p->alerts.cnt);
259 
260  for (i = 0; i < p->alerts.cnt; i++) {
261  const PacketAlert *pa = &p->alerts.alerts[i];
262  if (unlikely(pa->s == NULL)) {
263  continue;
264  }
265 
266  MemBufferWriteString(aft->buffer,
267  "ALERT MSG [%02d]: %s\n"
268  "ALERT GID [%02d]: %" PRIu32 "\n"
269  "ALERT SID [%02d]: %" PRIu32 "\n"
270  "ALERT REV [%02d]: %" PRIu32 "\n"
271  "ALERT CLASS [%02d]: %s\n"
272  "ALERT PRIO [%02d]: %" PRIu32 "\n"
273  "ALERT FOUND IN [%02d]: %s\n",
274  i, pa->s->msg,
275  i, pa->s->gid,
276  i, pa->s->id,
277  i, pa->s->rev,
278  i, pa->s->class_msg ? pa->s->class_msg : "<none>",
279  i, pa->s->prio,
280  i,
281  pa->flags & PACKET_ALERT_FLAG_STREAM_MATCH ? "STREAM" :
282  (pa->flags & PACKET_ALERT_FLAG_STATE_MATCH ? "STATE" : "PACKET"));
283  if (pa->flags & PACKET_ALERT_FLAG_TX) {
284  MemBufferWriteString(aft->buffer,
285  "ALERT IN TX [%02d]: %"PRIu64"\n", i, pa->tx_id);
286  } else {
287  MemBufferWriteString(aft->buffer,
288  "ALERT IN TX [%02d]: N/A\n", i);
289  }
290  if (p->payload_len > 0) {
291  MemBufferWriteString(aft->buffer,
292  "PAYLOAD LEN: %" PRIu32 "\n"
293  "PAYLOAD:\n",
294  p->payload_len);
295  PrintRawDataToBuffer(aft->buffer->buffer, &aft->buffer->offset, aft->buffer->size,
296  p->payload, p->payload_len);
297  }
298  if ((pa->flags & PACKET_ALERT_FLAG_STATE_MATCH) ||
299  (pa->flags & PACKET_ALERT_FLAG_STREAM_MATCH)) {
300  /* This is an app layer or stream alert */
301  int ret;
302  uint8_t flag;
303  if (!(PKT_IS_TCP(p)) || p->flow == NULL ||
304  p->flow->protoctx == NULL) {
305  return TM_ECODE_OK;
306  }
307  /* IDS mode reverse the data */
308  /** \todo improve the order selection policy */
309  if (p->flowflags & FLOW_PKT_TOSERVER) {
310  flag = FLOW_PKT_TOCLIENT;
311  } else {
312  flag = FLOW_PKT_TOSERVER;
313  }
314  ret = StreamSegmentForEach((const Packet *)p, flag,
315  AlertDebugPrintStreamSegmentCallback,
316  (void *)aft);
317  if (ret < 0) {
318  return TM_ECODE_FAILED;
319  }
320  }
321  }
322 
323  SCMutexLock(&aft->file_ctx->fp_mutex);
324  (void)MemBufferPrintToFPAsString(aft->buffer, aft->file_ctx->fp);
325  fflush(aft->file_ctx->fp);
326  aft->file_ctx->alerts += p->alerts.cnt;
327  SCMutexUnlock(&aft->file_ctx->fp_mutex);
328 
329  return TM_ECODE_OK;
330 }

395 static TmEcode AlertDebugLogThreadInit(ThreadVars *t, void *initdata, void **data)
396 {
397  AlertDebugLogThread *aft = SCMalloc(sizeof(AlertDebugLogThread));
398  if (unlikely(aft == NULL))
399  return TM_ECODE_FAILED;
400  memset(aft, 0, sizeof(AlertDebugLogThread));
401 
402  if(initdata == NULL)
403  {
404  SCLogDebug("Error getting context for DebugLog. \"initdata\" argument NULL");
405  SCFree(aft);
406  return TM_ECODE_FAILED;
407  }
408  /** Use the Ouptut Context (file pointer and mutex) */
409  aft->file_ctx = ((OutputCtx *)initdata)->data;
410 
411  /* 1 mb seems sufficient enough */
412  aft->buffer = MemBufferCreateNew(1 * 1024 * 1024);
413  if (aft->buffer == NULL) {
414  SCFree(aft);
415  return TM_ECODE_FAILED;
416  }
417 
418  *data = (void *)aft;
419  return TM_ECODE_OK;
420 }

458 /**
459  * \brief Create a new LogFileCtx for alert debug logging.
460  *
461  * \param ConfNode containing configuration for this logger.
462  *
463  * \return output_ctx if succesful, NULL otherwise
464  */
465 static OutputCtx *AlertDebugLogInitCtx(ConfNode *conf)
466 {
467  LogFileCtx *file_ctx = NULL;
468 
469  file_ctx = LogFileNewCtx();
470  if (file_ctx == NULL) {
471  SCLogDebug("couldn't create new file_ctx");
472  goto error;
473  }
474 
475  if (SCConfLogOpenGeneric(conf, file_ctx, DEFAULT_LOG_FILENAME, 1) < 0) {
476  goto error;
477  }
478 
479  OutputCtx *output_ctx = SCMalloc(sizeof(OutputCtx));
480  if (unlikely(output_ctx == NULL))
481  goto error;
482 
483  memset(output_ctx, 0x00, sizeof(OutputCtx));
484  output_ctx->data = file_ctx;
485  output_ctx->DeInit = AlertDebugLogDeInitCtx;
486 
487  SCLogDebug("Alert debug log output initialized");
488  return output_ctx;
489 
490 error:
491  if (file_ctx != NULL) {
492  LogFileFreeCtx(file_ctx);
493  }
494 
495  return NULL;
496 }
