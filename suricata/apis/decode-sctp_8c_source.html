42 static int DecodeSCTPPacket(ThreadVars *tv, Packet *p, uint8_t *pkt, uint16_t len)
43 {
44  if (unlikely(len < SCTP_HEADER_LEN)) {
45  ENGINE_SET_INVALID_EVENT(p, SCTP_PKT_TOO_SMALL);
46  return -1;
47  }
48 
49  p->sctph = (SCTPHdr *)pkt;
50 
51  SET_SCTP_SRC_PORT(p,&p->sp);
52  SET_SCTP_DST_PORT(p,&p->dp);
53 
54  p->payload = pkt + sizeof(SCTPHdr);
55  p->payload_len = len - sizeof(SCTPHdr);
56 
57  p->proto = IPPROTO_SCTP;
58 
59  return 0;
60 }

62 int DecodeSCTP(ThreadVars *tv, DecodeThreadVars *dtv, Packet *p, uint8_t *pkt, uint16_t len, PacketQueue *pq)
63 {
64  StatsIncr(tv, dtv->counter_sctp);
65 
66  if (unlikely(DecodeSCTPPacket(tv, p,pkt,len) < 0)) {
67  p->sctph = NULL;
68  return TM_ECODE_FAILED;
69  }
70 
71 #ifdef DEBUG
72  SCLogDebug("SCTP sp: %" PRIu32 " -> dp: %" PRIu32,
73  SCTP_GET_SRC_PORT(p), SCTP_GET_DST_PORT(p));
74 #endif
75 
76  /* Flow is an integral part of us */
77  FlowHandlePacket(tv, dtv, p);
78 
79  return TM_ECODE_OK;
80 }
