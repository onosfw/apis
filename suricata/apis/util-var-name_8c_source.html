39 {

101 /** \brief Initialize the Name idx hash.
102  * \param de_ctx Ptr to the detection engine ctx.
103  * \retval -1 in case of error
104  * \retval 0 in case of success
105  */
106 int VariableNameInitHash(DetectEngineCtx *de_ctx)
107 {
108  de_ctx->variable_names = HashListTableInit(4096, VariableNameHash, VariableNameCompare, VariableNameFree);
109  if (de_ctx->variable_names == NULL)
110  return -1;
111 
112  de_ctx->variable_idxs = HashListTableInit(4096, VariableIdxHash, VariableIdxCompare, NULL);
113  if (de_ctx->variable_idxs == NULL)
114  return -1;
115 
116  de_ctx->variable_names_idx = 0;
117  return 0;
118 }

132 /** \brief Get a name idx for a name. If the name is already used reuse the idx.
133  * \param name nul terminated string with the name
134  * \param type variable type
135  * \retval 0 in case of error
136  * \retval idx the idx or 0
137  */
138 uint16_t VariableNameGetIdx(DetectEngineCtx *de_ctx, char *name, enum VarTypes type)
139 {
140  uint16_t idx = 0;
141 
142  VariableName *fn = SCMalloc(sizeof(VariableName));
143  if (unlikely(fn == NULL))
144  goto error;
145 
146  memset(fn, 0, sizeof(VariableName));
147 
148  fn->type = type;
149  fn->name = SCStrdup(name);
150  if (fn->name == NULL)
151  goto error;
152 
153  VariableName *lookup_fn = (VariableName *)HashListTableLookup(de_ctx->variable_names, (void *)fn, 0);
154  if (lookup_fn == NULL) {
155  de_ctx->variable_names_idx++;
156 
157  idx = fn->idx = de_ctx->variable_names_idx;
158  HashListTableAdd(de_ctx->variable_names, (void *)fn, 0);
159  HashListTableAdd(de_ctx->variable_idxs, (void *)fn, 0);
160  } else {
161  idx = lookup_fn->idx;
162  VariableNameFree(fn);
163  }
164 
165  return idx;
166 error:
167  VariableNameFree(fn);
168  return 0;
169 }
