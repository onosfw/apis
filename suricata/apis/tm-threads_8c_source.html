113 /**
114  * \brief Separate run function so we can call it recursively.
115  *
116  * \todo Deal with post_pq for slots beyond the first.
117  */
118 TmEcode TmThreadsSlotVarRun(ThreadVars *tv, Packet *p,
119  TmSlot *slot)
120 {
121  TmEcode r;
122  TmSlot *s;
123  Packet *extra_p;
124 
125  for (s = slot; s != NULL; s = s->slot_next) {
126  TmSlotFunc SlotFunc = SC_ATOMIC_GET(s->SlotFunc);
127  PACKET_PROFILING_TMM_START(p, s->tm_id);
128 
129  if (unlikely(s->id == 0)) {
130  r = SlotFunc(tv, p, SC_ATOMIC_GET(s->slot_data), &s->slot_pre_pq, &s->slot_post_pq);
131  } else {
132  r = SlotFunc(tv, p, SC_ATOMIC_GET(s->slot_data), &s->slot_pre_pq, NULL);
133  }
134 
135  PACKET_PROFILING_TMM_END(p, s->tm_id);
136 
137  /* handle error */
138  if (unlikely(r == TM_ECODE_FAILED)) {
139  /* Encountered error. Return packets to packetpool and return */
140  TmqhReleasePacketsToPacketPool(&s->slot_pre_pq);
141 
142  SCMutexLock(&s->slot_post_pq.mutex_q);
143  TmqhReleasePacketsToPacketPool(&s->slot_post_pq);
144  SCMutexUnlock(&s->slot_post_pq.mutex_q);
145 
146  TmThreadsSetFlag(tv, THV_FAILED);
147  return TM_ECODE_FAILED;
148  }
149 
150  /* handle new packets */
151  while (s->slot_pre_pq.top != NULL) {
152  extra_p = PacketDequeue(&s->slot_pre_pq);
153  if (unlikely(extra_p == NULL))
154  continue;
155 
156  /* see if we need to process the packet */
157  if (s->slot_next != NULL) {
158  r = TmThreadsSlotVarRun(tv, extra_p, s->slot_next);
159  if (unlikely(r == TM_ECODE_FAILED)) {
160  TmqhReleasePacketsToPacketPool(&s->slot_pre_pq);
161 
162  SCMutexLock(&s->slot_post_pq.mutex_q);
163  TmqhReleasePacketsToPacketPool(&s->slot_post_pq);
164  SCMutexUnlock(&s->slot_post_pq.mutex_q);
165 
166  TmqhOutputPacketpool(tv, extra_p);
167  TmThreadsSetFlag(tv, THV_FAILED);
168  return TM_ECODE_FAILED;
169  }
170  }
171  tv->tmqh_out(tv, extra_p);
172  }
173  }
174 
175  return TM_ECODE_OK;
176 }

178 /** \internal
179  *
180  * \brief Process flow timeout packets
181  *
182  * Process flow timeout pseudo packets. During shutdown this loop
183  * is run until the flow engine kills the thread and the queue is
184  * empty.
185  */
186 static int TmThreadTimeoutLoop(ThreadVars *tv, TmSlot *s)
187 {
188  TmSlot *stream_slot = NULL, *slot = NULL;
189  int run = 1;
190  int r = TM_ECODE_OK;
191 
192  for (slot = s; slot != NULL; slot = slot->slot_next) {
193  if (slot->tm_id == TMM_STREAMTCP) {
194  stream_slot = slot;
195  break;
196  }
197  }
198 
199  if (tv->stream_pq == NULL || stream_slot == NULL)
200  return r;
201 
202  SCLogDebug("flow end loop starting");
203  while(run) {
204  Packet *p;
205  if (tv->stream_pq->len != 0) {
206  SCMutexLock(&tv->stream_pq->mutex_q);
207  p = PacketDequeue(tv->stream_pq);
208  SCMutexUnlock(&tv->stream_pq->mutex_q);
209  BUG_ON(p == NULL);
210 
211  if ((r = TmThreadsSlotProcessPkt(tv, stream_slot, p) != TM_ECODE_OK)) {
212  if (r == TM_ECODE_FAILED)
213  run = 0;
214  }
215  } else {
216  usleep(1);
217  }
218 
219  if (tv->stream_pq->len == 0 && TmThreadsCheckFlag(tv, THV_KILL)) {
220  run = 0;
221  }
222  }
223  SCLogDebug("flow end loop complete");
224 
225  return r;
226 }

255 void *TmThreadsSlotPktAcqLoop(void *td)
256 {
257  /* block usr2. usr2 to be handled by the main thread only */
258  UtilSignalBlock(SIGUSR2);
259 
260  ThreadVars *tv = (ThreadVars *)td;
261  TmSlot *s = tv->tm_slots;
262  char run = 1;
263  TmEcode r = TM_ECODE_OK;
264  TmSlot *slot = NULL;
265 
266  /* Set the thread name */
267  if (SCSetThreadName(tv->name) < 0) {
268  SCLogWarning(SC_ERR_THREAD_INIT, "Unable to set thread name");
269  }
270 
271  if (tv->thread_setup_flags != 0)
272  TmThreadSetupOptions(tv);
273 
274  /* Drop the capabilities for this thread */
275  SCDropCaps(tv);
276 
277  PacketPoolInit();
278 
279  /* check if we are setup properly */
280  if (s == NULL || s->PktAcqLoop == NULL || tv->tmqh_in == NULL || tv->tmqh_out == NULL) {
281  SCLogError(SC_ERR_FATAL, "TmSlot or ThreadVars badly setup: s=%p,"
282  " PktAcqLoop=%p, tmqh_in=%p,"
283  " tmqh_out=%p",
284  s, s ? s->PktAcqLoop : NULL, tv->tmqh_in, tv->tmqh_out);
285  EngineKill();
286 
287  TmThreadsSetFlag(tv, THV_CLOSED | THV_RUNNING_DONE);
288  pthread_exit((void *) -1);
289  return NULL;
290  }
291 
292  for (slot = s; slot != NULL; slot = slot->slot_next) {
293  if (slot->SlotThreadInit != NULL) {
294  void *slot_data = NULL;
295  r = slot->SlotThreadInit(tv, slot->slot_initdata, &slot_data);
296  if (r != TM_ECODE_OK) {
297  if (r == TM_ECODE_DONE) {
298  EngineDone();
299  TmThreadsSetFlag(tv, THV_CLOSED | THV_INIT_DONE | THV_RUNNING_DONE);
300  goto error;
301  } else {
302  EngineKill();
303  TmThreadsSetFlag(tv, THV_CLOSED | THV_RUNNING_DONE);
304  goto error;
305  }
306  }
307  (void)SC_ATOMIC_SET(slot->slot_data, slot_data);
308  }
309  memset(&slot->slot_pre_pq, 0, sizeof(PacketQueue));
310  SCMutexInit(&slot->slot_pre_pq.mutex_q, NULL);
311  memset(&slot->slot_post_pq, 0, sizeof(PacketQueue));
312  SCMutexInit(&slot->slot_post_pq.mutex_q, NULL);
313 
314  /* get the 'pre qeueue' from module before the stream module */
315  if (slot->slot_next != NULL && slot->slot_next->tm_id == TMM_STREAMTCP) {
316  SCLogDebug("pre-stream packetqueue %p (postq)", &s->slot_post_pq);
317  tv->stream_pq = &slot->slot_post_pq;
318  /* if the stream module is the first, get the threads input queue */
319  } else if (slot == (TmSlot *)tv->tm_slots && slot->tm_id == TMM_STREAMTCP) {
320  tv->stream_pq = &trans_q[tv->inq->id];
321  SCLogDebug("pre-stream packetqueue %p (inq)", &slot->slot_pre_pq);
322  }
323  }
324 
325  StatsSetupPrivate(tv);
326 
327  TmThreadsSetFlag(tv, THV_INIT_DONE);
328 
329  while(run) {
330  if (TmThreadsCheckFlag(tv, THV_PAUSE)) {
331  TmThreadsSetFlag(tv, THV_PAUSED);
332  TmThreadTestThreadUnPaused(tv);
333  TmThreadsUnsetFlag(tv, THV_PAUSED);
334  }
335 
336  r = s->PktAcqLoop(tv, SC_ATOMIC_GET(s->slot_data), s);
337 
338  if (r == TM_ECODE_FAILED || TmThreadsCheckFlag(tv, THV_KILL_PKTACQ)
339  || suricata_ctl_flags) {
340  run = 0;
341  }
342  if (r == TM_ECODE_DONE) {
343  run = 0;
344  }
345  }
346  StatsSyncCounters(tv);
347 
348  TmThreadsSetFlag(tv, THV_FLOW_LOOP);
349 
350  /* process all pseudo packets the flow timeout may throw at us */
351  TmThreadTimeoutLoop(tv, s);
352 
353  TmThreadsSetFlag(tv, THV_RUNNING_DONE);
354  TmThreadWaitForFlag(tv, THV_DEINIT);
355 
356  PacketPoolDestroy();
357 
358  for (slot = s; slot != NULL; slot = slot->slot_next) {
359  if (slot->SlotThreadExitPrintStats != NULL) {
360  slot->SlotThreadExitPrintStats(tv, SC_ATOMIC_GET(slot->slot_data));
361  }
362 
363  if (slot->SlotThreadDeinit != NULL) {
364  r = slot->SlotThreadDeinit(tv, SC_ATOMIC_GET(slot->slot_data));
365  if (r != TM_ECODE_OK) {
366  TmThreadsSetFlag(tv, THV_CLOSED);
367  goto error;
368  }
369  }
370 
371  BUG_ON(slot->slot_pre_pq.len);
372  BUG_ON(slot->slot_post_pq.len);
373  }
374 
375  tv->stream_pq = NULL;
376  SCLogDebug("%s ending", tv->name);
377  TmThreadsSetFlag(tv, THV_CLOSED);
378  pthread_exit((void *) 0);
379  return NULL;
380 
381 error:
382  tv->stream_pq = NULL;
383  pthread_exit((void *) -1);
384  return NULL;
385 }

389  * \todo Only the first "slot" currently makes the "post_pq" available
390  * to the thread module.
391  */
392 void *TmThreadsSlotVar(void *td)
393 {
394  /* block usr2. usr2 to be handled by the main thread only */
395  UtilSignalBlock(SIGUSR2);
396 
397  ThreadVars *tv = (ThreadVars *)td;
398  TmSlot *s = (TmSlot *)tv->tm_slots;
399  Packet *p = NULL;
400  char run = 1;
401  TmEcode r = TM_ECODE_OK;
402 
403  PacketPoolInitEmpty();
404 
405  /* Set the thread name */
406  if (SCSetThreadName(tv->name) < 0) {
407  SCLogWarning(SC_ERR_THREAD_INIT, "Unable to set thread name");
408  }
409 
410  if (tv->thread_setup_flags != 0)
411  TmThreadSetupOptions(tv);
412 
413  /* Drop the capabilities for this thread */
414  SCDropCaps(tv);
415 
416  /* check if we are setup properly */
417  if (s == NULL || tv->tmqh_in == NULL || tv->tmqh_out == NULL) {
418  EngineKill();
419 
420  TmThreadsSetFlag(tv, THV_CLOSED | THV_RUNNING_DONE);
421  pthread_exit((void *) -1);
422  return NULL;
423  }
424 
425  for (; s != NULL; s = s->slot_next) {
426  if (s->SlotThreadInit != NULL) {
427  void *slot_data = NULL;
428  r = s->SlotThreadInit(tv, s->slot_initdata, &slot_data);
429  if (r != TM_ECODE_OK) {
430  EngineKill();
431 
432  TmThreadsSetFlag(tv, THV_CLOSED | THV_RUNNING_DONE);
433  goto error;
434  }
435  (void)SC_ATOMIC_SET(s->slot_data, slot_data);
436  }
437  memset(&s->slot_pre_pq, 0, sizeof(PacketQueue));
438  SCMutexInit(&s->slot_pre_pq.mutex_q, NULL);
439  memset(&s->slot_post_pq, 0, sizeof(PacketQueue));
440  SCMutexInit(&s->slot_post_pq.mutex_q, NULL);
441 
442  /* special case: we need to access the stream queue
443  * from the flow timeout code */
444 
445  /* get the 'pre qeueue' from module before the stream module */
446  if (s->slot_next != NULL && s->slot_next->tm_id == TMM_STREAMTCP) {
447  SCLogDebug("pre-stream packetqueue %p (preq)", &s->slot_pre_pq);
448  tv->stream_pq = &s->slot_pre_pq;
449  /* if the stream module is the first, get the threads input queue */
450  } else if (s == (TmSlot *)tv->tm_slots && s->tm_id == TMM_STREAMTCP) {
451  tv->stream_pq = &trans_q[tv->inq->id];
452  SCLogDebug("pre-stream packetqueue %p (inq)", &s->slot_pre_pq);
453  }
454  }
455 
456  StatsSetupPrivate(tv);
457 
458  TmThreadsSetFlag(tv, THV_INIT_DONE);
459 
460  s = (TmSlot *)tv->tm_slots;
461 
462  while (run) {
463  if (TmThreadsCheckFlag(tv, THV_PAUSE)) {
464  TmThreadsSetFlag(tv, THV_PAUSED);
465  TmThreadTestThreadUnPaused(tv);
466  TmThreadsUnsetFlag(tv, THV_PAUSED);
467  }
468 
469  /* input a packet */
470  p = tv->tmqh_in(tv);
471 
472  if (p != NULL) {
473  /* run the thread module(s) */
474  r = TmThreadsSlotVarRun(tv, p, s);
475  if (r == TM_ECODE_FAILED) {
476  TmqhOutputPacketpool(tv, p);
477  TmThreadsSetFlag(tv, THV_FAILED);
478  break;
479  }
480 
481  /* output the packet */
482  tv->tmqh_out(tv, p);
483 
484  } /* if (p != NULL) */
485 
486  /* now handle the post_pq packets */
487  TmSlot *slot;
488  for (slot = s; slot != NULL; slot = slot->slot_next) {
489  if (slot->slot_post_pq.top != NULL) {
490  while (1) {
491  SCMutexLock(&slot->slot_post_pq.mutex_q);
492  Packet *extra_p = PacketDequeue(&slot->slot_post_pq);
493  SCMutexUnlock(&slot->slot_post_pq.mutex_q);
494 
495  if (extra_p == NULL)
496  break;
497 
498  if (slot->slot_next != NULL) {
499  r = TmThreadsSlotVarRun(tv, extra_p, slot->slot_next);
500  if (r == TM_ECODE_FAILED) {
501  SCMutexLock(&slot->slot_post_pq.mutex_q);
502  TmqhReleasePacketsToPacketPool(&slot->slot_post_pq);
503  SCMutexUnlock(&slot->slot_post_pq.mutex_q);
504 
505  TmqhOutputPacketpool(tv, extra_p);
506  TmThreadsSetFlag(tv, THV_FAILED);
507  break;
508  }
509  }
510  /* output the packet */
511  tv->tmqh_out(tv, extra_p);
512  } /* while */
513  } /* if */
514  } /* for */
515 
516  if (TmThreadsCheckFlag(tv, THV_KILL)) {
517  run = 0;
518  }
519  } /* while (run) */
520  StatsSyncCounters(tv);
521 
522  TmThreadsSetFlag(tv, THV_RUNNING_DONE);
523  TmThreadWaitForFlag(tv, THV_DEINIT);
524 
525  PacketPoolDestroy();
526 
527  s = (TmSlot *)tv->tm_slots;
528 
529  for ( ; s != NULL; s = s->slot_next) {
530  if (s->SlotThreadExitPrintStats != NULL) {
531  s->SlotThreadExitPrintStats(tv, SC_ATOMIC_GET(s->slot_data));
532  }
533 
534  if (s->SlotThreadDeinit != NULL) {
535  r = s->SlotThreadDeinit(tv, SC_ATOMIC_GET(s->slot_data));
536  if (r != TM_ECODE_OK) {
537  TmThreadsSetFlag(tv, THV_CLOSED);
538  goto error;
539  }
540  }
541  BUG_ON(s->slot_pre_pq.len);
542  BUG_ON(s->slot_post_pq.len);
543  }
544 
545  SCLogDebug("%s ending", tv->name);
546  tv->stream_pq = NULL;
547  TmThreadsSetFlag(tv, THV_CLOSED);
548  pthread_exit((void *) 0);
549  return NULL;
550 
551 error:
552  tv->stream_pq = NULL;
553  pthread_exit((void *) -1);
554  return NULL;
555 }
