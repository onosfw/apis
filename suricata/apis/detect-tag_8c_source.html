101 /**
102  * \brief This function is used to setup a tag for session/host
103  *
104  * \param t pointer to thread vars
105  * \param det_ctx pointer to the pattern matcher thread
106  * \param p pointer to the current packet
107  * \param m pointer to the sigmatch that we will cast into DetectTagData
108  *
109  * \retval 0 no match
110  * \retval 1 match
111  */
112 int DetectTagMatch(ThreadVars *t, DetectEngineThreadCtx *det_ctx, Packet *p, Signature *s, const SigMatchCtx *ctx)
113 {
114  const DetectTagData *td = (const DetectTagData *)ctx;
115  DetectTagDataEntry tde;
116  memset(&tde, 0, sizeof(DetectTagDataEntry));
117 
118  switch (td->type) {
119  case DETECT_TAG_TYPE_HOST:
120 #ifdef DEBUG
121  BUG_ON(!(td->direction == DETECT_TAG_DIR_SRC || td->direction == DETECT_TAG_DIR_DST));
122 #endif
123 
124  tde.sid = s->id;
125  tde.gid = s->gid;
126  tde.last_ts = tde.first_ts = p->ts.tv_sec;
127  tde.metric = td->metric;
128  tde.count = td->count;
129  if (td->direction == DETECT_TAG_DIR_SRC)
130  tde.flags |= TAG_ENTRY_FLAG_DIR_SRC;
131  else if (td->direction == DETECT_TAG_DIR_DST)
132  tde.flags |= TAG_ENTRY_FLAG_DIR_DST;
133 
134  SCLogDebug("Tagging Host with sid %"PRIu32":%"PRIu32"", s->id, s->gid);
135  TagHashAddTag(&tde, p);
136  break;
137  case DETECT_TAG_TYPE_SESSION:
138  if (p->flow != NULL) {
139  SCLogDebug("Setting up tag for flow");
140  /* If it already exists it will be updated */
141  tde.sid = s->id;
142  tde.gid = s->gid;
143  tde.last_ts = tde.first_ts = p->ts.tv_sec;
144  tde.metric = td->metric;
145  tde.count = td->count;
146 
147  SCLogDebug("Adding to or updating flow; first_ts %u count %u",
148  tde.first_ts, tde.count);
149  TagFlowAdd(p, &tde);
150  } else {
151  SCLogDebug("No flow to append the session tag");
152  }
153  break;
154 #ifdef DEBUG
155  default:
156  SCLogDebug("unknown type of a tag keyword (not session nor host)");
157  BUG_ON(1);
158  break;
159 #endif
160  }
161 
162  return 1;
163 }

165 /**
166  * \brief This function is used to parse tag options passed to tag keyword
167  *
168  * \param tagstr Pointer to the user provided tag options
169  *
170  * \retval td pointer to DetectTagData on success
171  * \retval NULL on failure
172  */
173 DetectTagData *DetectTagParse(char *tagstr)
174 {
175  DetectTagData td;
176 #define MAX_SUBSTRINGS 30
177  int ret = 0, res = 0;
178  int ov[MAX_SUBSTRINGS];
179  const char *str_ptr = NULL;
180 
181  ret = pcre_exec(parse_regex, parse_regex_study, tagstr, strlen(tagstr), 0, 0, ov, MAX_SUBSTRINGS);
182  if (ret < 1) {
183  SCLogError(SC_ERR_PCRE_MATCH, "parse error, ret %" PRId32 ", string %s", ret, tagstr);
184  goto error;
185  }
186 
187  res = pcre_get_substring((char *)tagstr, ov, MAX_SUBSTRINGS, 1, &str_ptr);
188  if (res < 0 || str_ptr == NULL) {
189  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
190  goto error;
191  }
192 
193  /* Type */
194  if (strcasecmp("session", str_ptr) == 0) {
195  td.type = DETECT_TAG_TYPE_SESSION;
196  } else if (strcasecmp("host", str_ptr) == 0) {
197  td.type = DETECT_TAG_TYPE_HOST;
198  } else {
199  SCLogError(SC_ERR_INVALID_VALUE, "Invalid argument type. Must be session or host (%s)", tagstr);
200  goto error;
201  }
202  pcre_free_substring(str_ptr);
203  str_ptr = NULL;
204 
205  /* default tag is 256 packets from session or dst host */
206  td.count = DETECT_TAG_MAX_PKTS;
207  td.metric = DETECT_TAG_METRIC_PACKET;
208  td.direction = DETECT_TAG_DIR_DST;
209 
210  if (ret > 4) {
211  res = pcre_get_substring((char *)tagstr, ov, MAX_SUBSTRINGS, 3, &str_ptr);
212  if (res < 0 || str_ptr == NULL) {
213  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
214  goto error;
215  }
216 
217  /* count */
218  if (ByteExtractStringUint32(&td.count, 10, strlen(str_ptr),
219  str_ptr) <= 0) {
220  SCLogError(SC_ERR_INVALID_VALUE, "Invalid argument for count. Must be a value in the range of 0 to %"PRIu32" (%s)", UINT32_MAX, tagstr);
221  goto error;
222  }
223 
224  pcre_free_substring(str_ptr);
225  str_ptr = NULL;
226 
227  res = pcre_get_substring((char *)tagstr, ov, MAX_SUBSTRINGS, 4, &str_ptr);
228  if (res < 0 || str_ptr == NULL) {
229  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
230  goto error;
231  }
232 
233  /* metric */
234  if (strcasecmp("packets", str_ptr) == 0) {
235  td.metric = DETECT_TAG_METRIC_PACKET;
236  if (DETECT_TAG_MAX_PKTS > 0 && td.count > DETECT_TAG_MAX_PKTS)
237  td.count = DETECT_TAG_MAX_PKTS;
238  /* TODO: load DETECT_TAG_MAX_PKTS from config */
239  } else if (strcasecmp("seconds", str_ptr) == 0) {
240  td.metric = DETECT_TAG_METRIC_SECONDS;
241  } else if (strcasecmp("bytes", str_ptr) == 0) {
242  td.metric = DETECT_TAG_METRIC_BYTES;
243  } else {
244  SCLogError(SC_ERR_INVALID_VALUE, "Invalid argument metric. Must be one of \"seconds\", \"packets\" or \"bytes\" (%s)", tagstr);
245  goto error;
246  }
247 
248  pcre_free_substring(str_ptr);
249  str_ptr = NULL;
250 
251  /* if specified, overwrite it */
252  if (ret == 7) {
253  res = pcre_get_substring((char *)tagstr, ov, MAX_SUBSTRINGS, 6, &str_ptr);
254  if (res < 0 || str_ptr == NULL) {
255  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
256  goto error;
257  }
258 
259  /* metric */
260  if (strcasecmp("src", str_ptr) == 0) {
261  td.direction = DETECT_TAG_DIR_SRC;
262  } else if (strcasecmp("dst", str_ptr) == 0) {
263  td.direction = DETECT_TAG_DIR_DST;
264  } else {
265  SCLogError(SC_ERR_INVALID_VALUE, "Invalid argument direction. Must be one of \"src\" or \"dst\" (only valid for tag host type, not sessions) (%s)", tagstr);
266  goto error;
267  }
268 
269  if (td.type != DETECT_TAG_TYPE_HOST) {
270  SCLogWarning(SC_ERR_INVALID_VALUE, "Argument direction doesn't make sense for type \"session\" (%s [%"PRIu8"])", tagstr, td.type);
271  }
272 
273  pcre_free_substring(str_ptr);
274  str_ptr = NULL;
275  }
276  }
277 
278  DetectTagData *real_td = SCMalloc(sizeof(DetectTagData));
279  if (unlikely(real_td == NULL)) {
280  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");
281  goto error;
282  }
283 
284  memcpy(real_td, &td, sizeof(DetectTagData));
285  return real_td;
286 
287 error:
288  if (str_ptr != NULL)
289  pcre_free_substring(str_ptr);
290  return NULL;
291 }

293 /**
294  * \brief this function is used to add the parsed tag data into the current signature
295  *
296  * \param de_ctx pointer to the Detection Engine Context
297  * \param s pointer to the Current Signature
298  * \param tagstr pointer to the user provided tag options
299  *
300  * \retval 0 on Success
301  * \retval -1 on Failure
302  */
303 int DetectTagSetup(DetectEngineCtx *de_ctx, Signature *s, char *tagstr)
304 {
305  DetectTagData *td = NULL;
306  SigMatch *sm = NULL;
307 
308  td = DetectTagParse(tagstr);
309  if (td == NULL) goto error;
310 
311  sm = SigMatchAlloc();
312  if (sm == NULL)
313  goto error;
314 
315  sm->type = DETECT_TAG;
316  sm->ctx = (SigMatchCtx *)td;
317 
318  /* Append it to the list of tags */
319  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_TMATCH);
320 
321  return 0;
322 
323 error:
324  if (td != NULL) DetectTagDataFree(td);
325  if (sm != NULL) SCFree(sm);
326  return -1;
327 
328 }
