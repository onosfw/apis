126 static inline void CudaBufferPacket(CudaThreadVars *ctv, Packet *p)
127 {
128  if (p->cuda_pkt_vars.cuda_mpm_enabled) {
129  while (!p->cuda_pkt_vars.cuda_done) {
130  SCMutexLock(&p->cuda_pkt_vars.cuda_mutex);
131  if (p->cuda_pkt_vars.cuda_done) {
132  SCMutexUnlock(&p->cuda_pkt_vars.cuda_mutex);
133  break;
134  } else {
135  SCCondWait(&p->cuda_pkt_vars.cuda_cond, &p->cuda_pkt_vars.cuda_mutex);
136  SCMutexUnlock(&p->cuda_pkt_vars.cuda_mutex);
137  }
138  }
139  }
140  p->cuda_pkt_vars.cuda_done = 0;
141 
142  if (p->payload_len == 0 ||
143  (p->flags & (PKT_NOPAYLOAD_INSPECTION & PKT_NOPACKET_INSPECTION)) ||
144  (p->flags & PKT_ALLOC) ||
145  (ctv->data_buffer_size_min_limit != 0 && p->payload_len < ctv->data_buffer_size_min_limit) ||
146  (p->payload_len > ctv->data_buffer_size_max_limit && ctv->data_buffer_size_max_limit != 0) ) {
147  p->cuda_pkt_vars.cuda_mpm_enabled = 0;
148  return;
149  }
150 
151  MpmCtx *mpm_ctx = NULL;
152  if (p->proto == IPPROTO_TCP) {
153  if (p->flowflags & FLOW_PKT_TOSERVER)
154  mpm_ctx = ctv->mpm_proto_tcp_ctx_ts;
155  else
156  mpm_ctx = ctv->mpm_proto_tcp_ctx_tc;
157  } else if (p->proto == IPPROTO_UDP) {
158  if (p->flowflags & FLOW_PKT_TOSERVER)
159  mpm_ctx = ctv->mpm_proto_udp_ctx_ts;
160  else
161  mpm_ctx = ctv->mpm_proto_udp_ctx_tc;
162  } else {
163  mpm_ctx = ctv->mpm_proto_other_ctx;
164  }
165  if (mpm_ctx == NULL || mpm_ctx->pattern_cnt == 0) {
166  p->cuda_pkt_vars.cuda_mpm_enabled = 0;
167  return;
168  }
169 
170 #if __WORDSIZE==64
171  CudaBufferSlice *slice = CudaBufferGetSlice(ctv->cuda_ac_cb,
172  p->payload_len + sizeof(uint64_t) + sizeof(CUdeviceptr),
173  (void *)p);
174  if (slice == NULL) {
175  SCLogError(SC_ERR_FATAL, "Error retrieving slice. Please report "
176  "this to dev.");
177  p->cuda_pkt_vars.cuda_mpm_enabled = 0;
178  return;
179  }
180  *((uint64_t *)(slice->buffer + slice->start_offset)) = p->payload_len;
181  *((CUdeviceptr *)(slice->buffer + slice->start_offset + sizeof(uint64_t))) = ((SCACCtx *)(mpm_ctx->ctx))->state_table_u32_cuda;
182  memcpy(slice->buffer + slice->start_offset + sizeof(uint64_t) + sizeof(CUdeviceptr), p->payload, p->payload_len);
183 #else
184  CudaBufferSlice *slice = CudaBufferGetSlice(ctv->cuda_ac_cb,
185  p->payload_len + sizeof(uint32_t) + sizeof(CUdeviceptr),
186  (void *)p);
187  if (slice == NULL) {
188  SCLogError(SC_ERR_FATAL, "Error retrieving slice. Please report "
189  "this to dev.");
190  p->cuda_pkt_vars.cuda_mpm_enabled = 0;
191  return;
192  }
193  *((uint32_t *)(slice->buffer + slice->start_offset)) = p->payload_len;
194  *((CUdeviceptr *)(slice->buffer + slice->start_offset + sizeof(uint32_t))) = ((SCACCtx *)(mpm_ctx->ctx))->state_table_u32_cuda;
195  memcpy(slice->buffer + slice->start_offset + sizeof(uint32_t) + sizeof(CUdeviceptr), p->payload, p->payload_len);
196 #endif
197  p->cuda_pkt_vars.cuda_mpm_enabled = 1;
198  SC_ATOMIC_SET(slice->done, 1);
199 
200  SCLogDebug("cuda ac buffering packet %p, payload_len - %"PRIu16" and deviceptr - %"PRIu64"\n",
201  p, p->payload_len, (unsigned long)((SCACCtx *)(mpm_ctx->ctx))->state_table_u32_cuda);
202 
203  return;
204 }
