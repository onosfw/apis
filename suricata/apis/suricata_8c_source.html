396 static int SetBpfString(int optind, char *argv[])
397 {
398  char *bpf_filter = NULL;
399  uint32_t bpf_len = 0;
400  int tmpindex = 0;
401 
402  /* attempt to parse remaining args as bpf filter */
403  tmpindex = optind;
404  while(argv[tmpindex] != NULL) {
405  bpf_len+=strlen(argv[tmpindex]) + 1;
406  tmpindex++;
407  }
408 
409  if (bpf_len == 0)
410  return TM_ECODE_OK;
411 
412  if (EngineModeIsIPS()) {
413  SCLogError(SC_ERR_NOT_SUPPORTED,
414  "BPF filter not available in IPS mode."
415  " Use firewall filtering if possible.");
416  return TM_ECODE_FAILED;
417  }
418 
419  bpf_filter = SCMalloc(bpf_len);
420  if (unlikely(bpf_filter == NULL))
421  return TM_ECODE_OK;
422  memset(bpf_filter, 0x00, bpf_len);
423 
424  tmpindex = optind;
425  while(argv[tmpindex] != NULL) {
426  strlcat(bpf_filter, argv[tmpindex],bpf_len);
427  if(argv[tmpindex + 1] != NULL) {
428  strlcat(bpf_filter," ", bpf_len);
429  }
430  tmpindex++;
431  }
432 
433  if(strlen(bpf_filter) > 0) {
434  if (ConfSetFinal("bpf-filter", bpf_filter) != 1) {
435  SCLogError(SC_ERR_FATAL, "Failed to set bpf filter.");
436  return TM_ECODE_FAILED;
437  }
438  }
439  SCFree(bpf_filter);
440 
441  return TM_ECODE_OK;
442 }

444 static void SetBpfStringFromFile(char *filename)
445 {
446  char *bpf_filter = NULL;
447  char *bpf_comment_tmp = NULL;
448  char *bpf_comment_start = NULL;
449  uint32_t bpf_len = 0;
450 #ifdef OS_WIN32
451  struct _stat st;
452 #else
453  struct stat st;
454 #endif /* OS_WIN32 */
455  FILE *fp = NULL;
456  size_t nm = 0;
457 
458  if (EngineModeIsIPS()) {
459  SCLogError(SC_ERR_NOT_SUPPORTED,
460  "BPF filter not available in IPS mode."
461  " Use firewall filtering if possible.");
462  exit(EXIT_FAILURE);
463  }
464 
465 #ifdef OS_WIN32
466  if(_stat(filename, &st) != 0) {
467 #else
468  if(stat(filename, &st) != 0) {
469 #endif /* OS_WIN32 */
470  SCLogError(SC_ERR_FOPEN, "Failed to stat file %s", filename);
471  exit(EXIT_FAILURE);
472  }
473  bpf_len = st.st_size + 1;
474 
475  fp = fopen(filename,"r");
476  if (fp == NULL) {
477  SCLogError(SC_ERR_FOPEN, "Failed to open file %s", filename);
478  exit(EXIT_FAILURE);
479  }
480 
481  bpf_filter = SCMalloc(bpf_len * sizeof(char));
482  if (unlikely(bpf_filter == NULL)) {
483  SCLogError(SC_ERR_MEM_ALLOC, "Failed to allocate buffer for bpf filter in file %s", filename);
484  exit(EXIT_FAILURE);
485  }
486  memset(bpf_filter, 0x00, bpf_len);
487 
488  nm = fread(bpf_filter, bpf_len - 1, 1, fp);
489  if((ferror(fp) != 0)||( nm != 1)) {
490  *bpf_filter='\0';
491  }
492  fclose(fp);
493 
494  if(strlen(bpf_filter) > 0) {
495  /*replace comments with space*/
496  bpf_comment_start = bpf_filter;
497  while((bpf_comment_tmp = strchr(bpf_comment_start, '#')) != NULL) {
498  while((*bpf_comment_tmp !='\0') &&
499  (*bpf_comment_tmp != '\r') && (*bpf_comment_tmp != '\n'))
500  {
501  *bpf_comment_tmp++ = ' ';
502  }
503  bpf_comment_start = bpf_comment_tmp;
504  }
505  /*remove remaining '\r' and '\n' */
506  while((bpf_comment_tmp = strchr(bpf_filter, '\r')) != NULL) {
507  *bpf_comment_tmp = ' ';
508  }
509  while((bpf_comment_tmp = strchr(bpf_filter, '\n')) != NULL) {
510  *bpf_comment_tmp = ' ';
511  }
512  if(ConfSetFinal("bpf-filter", bpf_filter) != 1) {
513  SCLogError(SC_ERR_FOPEN, "ERROR: Failed to set bpf filter!");
514  SCFree(bpf_filter);
515  exit(EXIT_FAILURE);
516  }
517  }
518  SCFree(bpf_filter);
519 }
