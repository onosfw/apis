50 /**
51  * \brief Decode and check the identifier information of the
52  * current node that is in extended format
53  *
54  * \param ac pointer to the ASN1 Context data
55  *
56  * \retval byte of the status of the parser
57  */
58 uint8_t SCAsn1GetHighTagNumber(Asn1Ctx *ac)
59 {
60  uint8_t ret = 0;
61  uint32_t tag_num = 0;
62 
63  /* If we have a high tag num, skip the id octet */
64  ac->iter++;
65 
66  Asn1Node *node = ASN1CTX_CUR_NODE(ac);
67 
68  ret = SCAsn1CheckBounds(ac);
69  if (ret == ASN1_PARSER_ERR) {
70  ac->parser_status |= ASN1_STATUS_INVALID | ASN1_STATUS_OOB;
71  return ret;
72  }
73 
74  uint8_t raw_id = *ac->iter;
75 
76  tag_num += ASN1_BER_GET_HIGH_TAG_NUM(raw_id);
77 
78  if (ASN1_BER_GET_HIGH_TAG_NUM(raw_id) == 0) {
79  /* Set event, invalid id */
80  node->flags |= ASN1_BER_EVENT_INVALID_ID;
81  ac->parser_status |= ASN1_STATUS_INVALID;
82  return ASN1_PARSER_ERR;
83  }
84 
85  ac->iter++;
86  if (!ASN1_BER_IS_HIGH_TAG_END(raw_id)) {
87  do {
88  ret = SCAsn1CheckBounds(ac);
89  if (ret == ASN1_PARSER_ERR) {
90  ac->parser_status |= ASN1_STATUS_INVALID | ASN1_STATUS_OOB;
91  return ret;
92  }
93 
94  raw_id = *ac->iter;
95 
96  if ((uint64_t) ((uint64_t)tag_num +
97  (uint64_t)ASN1_BER_GET_HIGH_TAG_NUM(raw_id)) > UINT32_MAX)
98  {
99  node->flags |= ASN1_BER_EVENT_ID_TOO_LONG;
100  ac->parser_status |= ASN1_STATUS_INVALID;
101  return ASN1_PARSER_ERR;
102  }
103 
104  tag_num += ASN1_BER_GET_HIGH_TAG_NUM(raw_id);
105  ac->iter++;
106  } while (!ASN1_BER_IS_HIGH_TAG_END(raw_id));
107  }
108  node->id.tag_num = tag_num;
109 
110  return ASN1_PARSER_OK;
111 }

113 /**
114  * \brief Decode and check the length, of the current node
115  * in definite but extended format, that we are parsing,
116  * checking invalid opts
117  *
118  * \param ac pointer to the ASN1 Context data
119  *
120  * \retval byte of the status of the parser
121  */
122 uint32_t SCAsn1GetLengthLongForm(Asn1Ctx *ac)
123 {
124  uint8_t raw_len = *ac->iter;
125  uint8_t ret = 0;
126  uint32_t content_len = 0;
127  uint8_t oct_len = ASN1_BER_GET_LONG_LEN_OCTETS(raw_len);
128  uint8_t i = 0;
129 
130  Asn1Node *node = ASN1CTX_CUR_NODE(ac);
131 
132  for (; i < oct_len; i++) {
133  ac->iter++;
134 
135  ret = SCAsn1CheckBounds(ac);
136  if (ret == ASN1_PARSER_ERR) {
137  ac->parser_status |= ASN1_STATUS_INVALID | ASN1_STATUS_OOB;
138  return ASN1_PARSER_ERR;
139  }
140 
141  raw_len = *ac->iter;
142  if (raw_len == 0xFF && ac->iter == node->len.ptr + 1) {
143  /* 8.1.3.5, 0xFF shall not be used */
144  node->flags |= ASN1_BER_EVENT_INVALID_LEN;
145  ac->parser_status = ASN1_STATUS_INVALID;
146  return ASN1_PARSER_ERR;
147  }
148 
149  if ((uint64_t) ((uint64_t)content_len +
150  (uint64_t) ASN1_BER_GET_HIGH_TAG_NUM(raw_len)) > UINT32_MAX)
151  {
152  node->flags |= ASN1_BER_EVENT_LEN_TOO_LONG;
153  ac->parser_status = ASN1_STATUS_INVALID;
154  return ASN1_PARSER_ERR;
155  }
156 
157  content_len += raw_len;
158  }
159 
160  ac->iter++;
161 
162  node->len.len = content_len;
163  return ASN1_PARSER_OK;
164 }

167 /**
168  * \brief Check the content length and perform other inspections
169  * and decodings if necessary
170  *
171  * \param ac pointer to the ASN1 Context data
172  *
173  * \retval byte of the status of the parser
174  */
175 uint8_t SCAsn1DecodeContent(Asn1Ctx *ac)
176 {
177 
178  Asn1Node *node = ASN1CTX_CUR_NODE(ac);
179 
180  /* Uops, if we are done, we break here */
181  if (node->flags & ASN1_NODE_IS_EOC)
182  return ASN1_PARSER_OK;
183 
184  /* First check the form of length (BER, DER, CER)
185  * and if we are on a zero length */
186  if (node->len.form != ASN1_BER_LEN_INDEFINITE &&
187  node->len.len == 0)
188  {
189  node->data.len = 0;
190  return ASN1_PARSER_OK;
191  }
192 
193  node->data.ptr = ac->iter;
194  /* If we have a complete length, check that
195  * it is in bounds */
196  if (ac->iter + node->len.len > ac->end) {
197  /* We do not have all the content octets! */
198  node->data.len = ac->end - ac->iter;
199  } else {
200  /* We have all the content octets */
201  node->data.len = node->len.len;
202  }
203 
204  return ASN1_PARSER_OK;
205 }

207 /**
208  * \brief Decode and check the length, of the current node
209  * that we are parsing, also check invalid opts
210  *
211  * \param ac pointer to the ASN1 Context data
212  *
213  * \retval byte of the status of the parser
214  */
215 uint8_t SCAsn1DecodeLength(Asn1Ctx *ac)
216 {
217  uint8_t ret = 0;
218  ret = SCAsn1CheckBounds(ac);
219  if (ret == ASN1_PARSER_ERR) {
220  ac->parser_status |= ASN1_STATUS_INVALID | ASN1_STATUS_OOB;
221  return ASN1_PARSER_ERR;
222  }
223 
224  Asn1Node *node = ASN1CTX_CUR_NODE(ac);
225  /* Store the position */
226  node->len.ptr = ac->iter;
227 
228  uint8_t len_byte = *ac->iter;
229 
230  //SCPrintByteBin(len_byte);
231 
232  if (*node->id.ptr == 0 && len_byte == 0) {
233  node->flags |= ASN1_NODE_IS_EOC;
234  ac->iter++;
235  return ASN1_PARSER_OK;
236  }
237 
238  if (ASN1_BER_IS_INDEFINITE_LEN(len_byte)) {
239  node->len.form = ASN1_BER_LEN_INDEFINITE;
240  node->len.len = 0;
241  ac->iter++;
242 
243  uint8_t *tmp_iter = ac->iter;
244 
245  /* Check that e-o-c is in bounds */
246  for (; tmp_iter < ac->end - 1; tmp_iter++) {
247  if (ASN1_BER_IS_EOC(tmp_iter)) {
248  node->data.len = tmp_iter - ac->iter;
249  node->len.len = tmp_iter - ac->iter;
250  return ASN1_PARSER_OK;
251  }
252  }
253 
254  /* EOC Not found */
255  ac->parser_status |= ASN1_STATUS_INVALID;
256  node->flags |= ASN1_BER_EVENT_EOC_NOT_FOUND;
257 
258  return ASN1_PARSER_ERR;
259 
260  } else {
261  /* Look which form we get (and if it apply to the id type) */
262  if (ASN1_BER_IS_SHORT_LEN(len_byte)) {
263  node->len.form = ASN1_BER_LEN_SHORT;
264  node->len.len = ASN1_BER_GET_SHORT_LEN(len_byte);
265  ac->iter++;
266  } else {
267  node->len.form = ASN1_BER_LEN_LONG;
268 
269  /* Ok, let's parse the long form */
270  return SCAsn1GetLengthLongForm(ac);
271  }
272 
273  }
274  return ASN1_PARSER_OK;
275 }

277 /**
278  * \brief Decode and check the identifier information of the
279  * current node that we are parsing, also check invalid opts
280  *
281  * \param ac pointer to the ASN1 Context data
282  *
283  * \retval byte of the status of the parser
284  */
285 uint8_t SCAsn1DecodeIdentifier(Asn1Ctx *ac)
286 {
287  uint8_t ret = 0;
288  ret = SCAsn1CheckBounds(ac);
289  if (ret == ASN1_PARSER_ERR) {
290  ac->parser_status |= ASN1_STATUS_INVALID | ASN1_STATUS_OOB;
291  return ret;
292  }
293 
294  Asn1Node *node = ASN1CTX_CUR_NODE(ac);
295  /* Store the position */
296  node->id.ptr = ac->iter;
297 
298  //SCPrintByteBin(*ac->iter);
299 
300  node->id.class_tag = ASN1_BER_GET_CLASS_TAG(*ac->iter);
301  node->id.tag_type = ASN1_BER_IS_CONSTRUCTED(*ac->iter);
302 
303  if (ASN1_BER_IS_HIGH_TAG(*ac->iter)) {
304  return SCAsn1GetHighTagNumber(ac);
305  } else {
306  node->id.tag_num = ASN1_BER_GET_LOW_TAG_NUM(*ac->iter);
307  ac->iter++;
308  }
309 
310  return ASN1_PARSER_OK;
311 }

342  * \brief Create a new ASN1 Parsing context
343  *
344  * \retval Asn1Ctx pointer to the new ctx
345  */
346 Asn1Ctx *SCAsn1CtxNew(void)
347 {
348  Asn1Ctx *ac = SCMalloc(sizeof(Asn1Ctx));
349 
350  if (unlikely(ac == NULL)) {
351  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");
352  return NULL;
353  }
354  memset(ac, 0, sizeof(Asn1Ctx));
355 
356  ac->asn1_stack = SCMalloc(sizeof(Asn1Node *) * asn1_max_frames_config);
357  if (ac->asn1_stack == NULL) {
358  SCFree(ac);
359  return NULL;
360  }
361  memset(ac->asn1_stack, 0, sizeof(Asn1Node *) * asn1_max_frames_config);
362 
363  return ac;
364 }

378  Asn1Node *node = ASN1CTX_GET_NODE(ac, i);

380  SCFree(node);
381  }
382  }
383  SCFree(ac);

429 /**
430  * \brief Decode the nodes/frames located at certain position/level
431  *
432  * \param ac pointer to the ASN1 ctx
433  * \param node_id node index at the asn1 stack of the ctx
434  *
435  * \retval byte of parser status
436  */
437 uint8_t SCAsn1Decode(Asn1Ctx *ac, uint16_t node_id)
438 {
439  Asn1Node *node = NULL;
440  uint8_t ret = 0;
441 
442  /* while remaining data, and no fatal error, or end, or max stack frames */
443  while (ac->iter < ac->end
444  && !(ac->parser_status & ASN1_STATUS_DONE)
445  && ac->cur_frame < asn1_max_frames_config)
446  {
447  /* Prepare a new frame */
448  if (SCAsn1CtxNewFrame(ac, node_id) == NULL)
449  break;
450 
451  ac->cur_frame = node_id;
452  node = ASN1CTX_GET_NODE(ac, node_id);
453 
454  SCLogDebug("ASN1 Getting ID, cur:%x remaining %"PRIu32, (uint8_t)*ac->iter, (uint32_t)(ac->end - ac->iter));
455 
456  /* Get identifier/tag */
457  ret = SCAsn1DecodeIdentifier(ac);
458  if (ret == ASN1_PARSER_ERR) {
459  SCLogDebug("Error parsing identifier");
460 
461  node->flags |= ASN1_BER_EVENT_INVALID_ID;
462  ac->ctx_flags |= node->flags;
463 
464  break;
465  }
466 
467  SCLogDebug("ASN1 Getting LEN");
468 
469  /* Get length of content */
470  ret = SCAsn1DecodeLength(ac);
471  if (ret == ASN1_PARSER_ERR) {
472  SCLogDebug("Error parsing length");
473 
474  node->flags |= ASN1_BER_EVENT_INVALID_LEN;
475  ac->ctx_flags |= node->flags;
476 
477  break;
478  }
479 
480  if ( !(node->flags & ASN1_NODE_IS_EOC)) {
481  SCLogDebug("ASN1 Getting CONTENT");
482 
483  /* Inspect content */
484  ret = SCAsn1DecodeContent(ac);
485  if (ret == ASN1_PARSER_ERR) {
486  SCLogDebug("Error parsing content");
487 
488  break;
489  }
490 
491  /* Skip to the next record (if any) */
492  if (node->id.tag_type != ASN1_TAG_TYPE_CONSTRUCTED)
493  /* Is primitive, skip it all (no need to decode it)*/
494  ac->iter += node->data.len;
495  }
496 
497  /* Check if we are done with data */
498  ret = SCAsn1CheckBounds(ac);
499  if (ret == ASN1_PARSER_ERR) {
500 
501  ac->parser_status |= ASN1_STATUS_DONE;
502  /* There's no more data available */
503  ret = ASN1_PARSER_OK;
504 
505  break;
506  }
507 #if 0
508  printf("Tag Num: %"PRIu32", Tag Type: %"PRIu8", Class:%"PRIu8", Length: %"PRIu32"\n", node->id.tag_num, node->id.tag_type, node->id.class_tag, node->len.len);
509  printf("Data: \n");
510  PrintRawDataFp(stdout, node->data.ptr, node->len.len);
511  printf(" -- EOD --\n");
512 #endif
513 
514  /* Stack flags/events here, so we have the resume at the ctx flags */
515  ac->ctx_flags |= node->flags;
516 
517  /* Check if it's not a primitive type,
518  * then we need to decode contents */
519  if (node->id.tag_type == ASN1_TAG_TYPE_CONSTRUCTED) {
520  ret = SCAsn1Decode(ac, node_id + 1);
521  } /* Else we have reached a primitive type and stop the recursion,
522  * look if we have other branches at the same level */
523 
524  /* But first check if it's a constructed node, and the sum of child
525  * lengths was more than the length of this frame
526  * this would mean that we have an overflow at the attributes */
527  if (ac->iter > node->data.ptr + node->data.len + 1) {
528  /* We decoded more length on this frame */
529  }
530 
531  node_id = ac->cur_frame + 1;
532  }
533 
534  return ret;
535 }
