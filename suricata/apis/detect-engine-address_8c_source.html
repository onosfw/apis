94 void DetectAddressFree(DetectAddress *ag)
95 {
96  if (ag == NULL)
97  return;
98 
99  SCLogDebug("ag %p, sh %p", ag, ag->sh);
100 
101  /* only free the head if we have the original */
102  if (ag->sh != NULL && !(ag->flags & ADDRESS_SIGGROUPHEAD_COPY)) {
103  SCLogDebug("- ag %p, sh %p not a copy, so call SigGroupHeadFree", ag,
104  ag->sh);
105  SigGroupHeadFree(ag->sh);
106  }
107  ag->sh = NULL;
108 
109  if (!(ag->flags & ADDRESS_HAVEPORT)) {
110  SCLogDebug("- ag %p dst_gh %p", ag, ag->dst_gh);
111 
112  if (ag->dst_gh != NULL)
113  DetectAddressHeadFree(ag->dst_gh);
114  ag->dst_gh = NULL;
115  } else {
116  SCLogDebug("- ag %p port %p", ag, ag->port);
117 
118  if (ag->port != NULL && !(ag->flags & ADDRESS_PORTS_COPY)) {
119  SCLogDebug("- ag %p port %p, not a copy so call DetectPortCleanupList",
120  ag, ag->port);
121  DetectPortCleanupList(ag->port);
122  }
123  ag->port = NULL;
124  }
125 #ifdef DEBUG
126  detect_address_group_memory -= sizeof(DetectAddress);
127  detect_address_group_free_cnt++;
128 #endif
129  SCFree(ag);
130 
131  return;

285

307  } else {
308  *head = ag;

367 /**
368  * \brief Same as DetectAddressInsert, but then for inserting a address group
369  * object. This also makes sure SigGroupContainer lists are handled
370  * correctly.
371  *
372  * \param de_ctx Pointer to the detection engine context.
373  * \param gh Pointer to the DetectAddressHead list to which it has to
374  * be inserted.
375  * \param new Pointer to the DetectAddress, that has to be inserted.
376  *
377  * \retval 1 On successfully inserting it.
378  * \retval -1 On error.
379  * \retval 0 Not inserted, memory of new is freed.
380  */
381 int DetectAddressInsert(DetectEngineCtx *de_ctx, DetectAddressHead *gh,
382  DetectAddress *new)
383 {
384  DetectAddress *head = NULL;
385  DetectAddress *cur = NULL;
386  DetectAddress *c = NULL;
387  int r = 0;
388 
389  if (new == NULL)
390  return 0;
391 
392  BUG_ON(new->ip.family == 0 && !(new->flags & ADDRESS_FLAG_ANY));
393 
394  /* get our head ptr based on the address we want to insert */
395  head = GetHeadPtr(gh, new);
396 
397  /* see if it already exists or overlaps with existing ag's */
398  if (head != NULL) {
399  cur = NULL;
400 
401  for (cur = head; cur != NULL; cur = cur->next) {
402  r = DetectAddressCmp(new, cur);
403  BUG_ON(r == ADDRESS_ER);
404 
405  /* if so, handle that */
406  if (r == ADDRESS_EQ) {
407  /* exact overlap/match */
408  if (cur != new) {
409  DetectPort *port = new->port;
410  for ( ; port != NULL; port = port->next)
411  DetectPortInsertCopy(de_ctx, &cur->port, port);
412  SigGroupHeadCopySigs(de_ctx, new->sh, &cur->sh);
413  cur->cnt += new->cnt;
414  DetectAddressFree(new);
415 
416  return 0;
417  }
418 
419  return 1;
420  } else if (r == ADDRESS_GT) {
421  /* only add it now if we are bigger than the last group.
422  * Otherwise we'll handle it later. */
423  if (cur->next == NULL) {
424  /* put in the list */
425  new->prev = cur;
426  cur->next = new;
427 
428  return 1;
429  }
430  } else if (r == ADDRESS_LT) {
431  /* see if we need to insert the ag anywhere put in the list */
432  if (cur->prev != NULL)
433  cur->prev->next = new;
434  new->prev = cur->prev;
435  new->next = cur;
436  cur->prev = new;
437 
438  /* update head if required */
439  if (head == cur) {
440  head = new;
441 
442  if (SetHeadPtr(gh, head) < 0)
443  goto error;
444  }
445 
446  return 1;
447  /* alright, those were the simple cases, lets handle the more
448  * complex ones now */
449  } else if (r == ADDRESS_ES) {
450  c = NULL;
451  r = DetectAddressCut(de_ctx, cur, new, &c);
452  if (r == -1)
453  goto error;
454 
455  DetectAddressInsert(de_ctx, gh, new);
456  if (c != NULL)
457  DetectAddressInsert(de_ctx, gh, c);
458 
459  return 1;
460  } else if (r == ADDRESS_EB) {
461  c = NULL;
462  r = DetectAddressCut(de_ctx, cur, new, &c);
463  if (r == -1)
464  goto error;
465 
466  DetectAddressInsert(de_ctx, gh, new);
467  if (c != NULL)
468  DetectAddressInsert(de_ctx, gh, c);
469 
470  return 1;
471  } else if (r == ADDRESS_LE) {
472  c = NULL;
473  r = DetectAddressCut(de_ctx, cur, new, &c);
474  if (r == -1)
475  goto error;
476 
477  DetectAddressInsert(de_ctx, gh, new);
478  if (c != NULL)
479  DetectAddressInsert(de_ctx, gh, c);
480 
481  return 1;
482  } else if (r == ADDRESS_GE) {
483  c = NULL;
484  r = DetectAddressCut(de_ctx, cur,new,&c);
485  if (r == -1)
486  goto error;
487 
488  DetectAddressInsert(de_ctx, gh, new);
489  if (c != NULL)
490  DetectAddressInsert(de_ctx, gh, c);
491 
492  return 1;
493  }
494  }
495 
496  /* head is NULL, so get a group and set head to it */
497  } else {
498  head = new;
499  if (SetHeadPtr(gh, head) < 0) {
500  SCLogDebug("SetHeadPtr failed");
501  goto error;
502  }
503  }
504 
505  return 1;
506 
507 error:
508  /* XXX */
509  return -1;
510 }

549  * \internal
550  * \brief Creates a cidr ipv6 netblock, based on the cidr netblock value.
551  *
552  * For example if we send a cidr of 7 as argument, an ipv6 address
553  * mask of the value FE:00:00:00:00:00:00:00 is created and updated
554  * in the argument struct in6_addr *in6.
555  *
556  * \todo I think for the final section: while (cidr > 0), we can simply
557  * replace it with a
558  * if (cidr > 0) {
559  * in6->s6_addr[i] = -1 << (8 - cidr);
560  *
561  * \param cidr The value of the cidr.
562  * \param in6 Pointer to an ipv6 address structure(struct in6_addr) which will
563  * hold the cidr netblock result.
564  */
565 static void DetectAddressParseIPv6CIDR(int cidr, struct in6_addr *in6)
566 {
567  int i = 0;
568 
569  memset(in6, 0, sizeof(struct in6_addr));
570 
571  while (cidr > 8) {
572  in6->s6_addr[i] = 0xff;
573  cidr -= 8;
574  i++;
575  }
576 
577  while (cidr > 0) {
578  in6->s6_addr[i] |= 0x80;
579  if (--cidr > 0)
580  in6->s6_addr[i] = in6->s6_addr[i] >> 1;
581  }
582 
583  return;
584 }

586 /**
587  * \internal
588  * \brief Parses an ipv4/ipv6 address string and updates the result into the
589  * DetectAddress instance sent as the argument.
590  *
591  * \param dd Pointer to the DetectAddress instance which should be updated with
592  * the address range details from the parsed ip string.
593  * \param str Pointer to address string that has to be parsed.
594  *
595  * \retval 0 On successfully parsing the address string.
596  * \retval -1 On failure.
597  */
598 int DetectAddressParseString(DetectAddress *dd, char *str)
599 {
600  char *ip = NULL;
601  char *ip2 = NULL;
602  char *mask = NULL;
603  int r = 0;
604  char ipstr[256];
605 
606  while (*str != '\0' && *str == ' ')
607  str++;
608 
609  /* first handle 'any' */
610  if (strcasecmp(str, "any") == 0) {
611  dd->flags |= ADDRESS_FLAG_ANY;
612  SCLogDebug("address is \'any\'");
613  return 0;
614  }
615 
616  strlcpy(ipstr, str, sizeof(ipstr));
617  SCLogDebug("str %s", str);
618 
619  /* we work with a copy so that we can put a
620  * nul-termination in it later */
621  ip = ipstr;
622 
623  /* handle the negation case */
624  if (ip[0] == '!') {
625  dd->flags |= ADDRESS_FLAG_NOT;
626  ip++;
627  }
628 
629  /* see if the address is an ipv4 or ipv6 address */
630  if ((strchr(str, ':')) == NULL) {
631  /* IPv4 Address */
632  struct in_addr in;
633 
634  dd->ip.family = AF_INET;
635 
636  if ((mask = strchr(ip, '/')) != NULL) {
637  /* 1.2.3.4/xxx format (either dotted or cidr notation */
638  ip[mask - ip] = '\0';
639  mask++;
640  uint32_t ip4addr = 0;
641  uint32_t netmask = 0;
642  size_t u = 0;
643 
644  if ((strchr (mask, '.')) == NULL) {
645  /* 1.2.3.4/24 format */
646 
647  for (u = 0; u < strlen(mask); u++) {
648  if(!isdigit((unsigned char)mask[u]))
649  goto error;
650  }
651 
652  int cidr = atoi(mask);
653  if (cidr < 0 || cidr > 32)
654  goto error;
655 
656  netmask = CIDRGet(cidr);
657  } else {
658  /* 1.2.3.4/255.255.255.0 format */
659  r = inet_pton(AF_INET, mask, &in);
660  if (r <= 0)
661  goto error;
662 
663  netmask = in.s_addr;
664  }
665 
666  r = inet_pton(AF_INET, ip, &in);
667  if (r <= 0)
668  goto error;
669 
670  ip4addr = in.s_addr;
671 
672  dd->ip.addr_data32[0] = dd->ip2.addr_data32[0] = ip4addr & netmask;
673  dd->ip2.addr_data32[0] |=~ netmask;
674  } else if ((ip2 = strchr(ip, '-')) != NULL) {
675  /* 1.2.3.4-1.2.3.6 range format */
676  ip[ip2 - ip] = '\0';
677  ip2++;
678 
679  r = inet_pton(AF_INET, ip, &in);
680  if (r <= 0)
681  goto error;
682  dd->ip.addr_data32[0] = in.s_addr;
683 
684  r = inet_pton(AF_INET, ip2, &in);
685  if (r <= 0)
686  goto error;
687  dd->ip2.addr_data32[0] = in.s_addr;
688 
689  /* a > b is illegal, a = b is ok */
690  if (ntohl(dd->ip.addr_data32[0]) > ntohl(dd->ip2.addr_data32[0]))
691  goto error;
692  } else {
693  /* 1.2.3.4 format */
694  r = inet_pton(AF_INET, ip, &in);
695  if (r <= 0)
696  goto error;
697  /* single host */
698  dd->ip.addr_data32[0] = in.s_addr;
699  dd->ip2.addr_data32[0] = in.s_addr;
700  }
701  } else {
702  /* IPv6 Address */
703  struct in6_addr in6, mask6;
704  uint32_t ip6addr[4], netmask[4];
705 
706  dd->ip.family = AF_INET6;
707 
708  if ((mask = strchr(ip, '/')) != NULL) {
709  ip[mask - ip] = '\0';
710  mask++;
711 
712  int cidr = atoi(mask);
713  if (cidr < 0 || cidr > 128)
714  goto error;
715 
716  r = inet_pton(AF_INET6, ip, &in6);
717  if (r <= 0)
718  goto error;

930

935  "Please file a bug report on this.");

956  /* normal block */

1104  goto error;

1148 error:
