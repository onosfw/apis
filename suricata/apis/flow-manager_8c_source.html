257 /**
258  * \internal
259  *
260  * \brief check all flows in a hash row for timing out
261  *
262  * \param f last flow in the hash row
263  * \param ts timestamp
264  * \param emergency bool indicating emergency mode
265  * \param counters ptr to FlowTimeoutCounters structure
266  *
267  * \retval cnt timed out flows
268  */
269 static uint32_t FlowManagerHashRowTimeout(Flow *f, struct timeval *ts,
270  int emergency, FlowTimeoutCounters *counters)
271 {
272  uint32_t cnt = 0;
273 
274  do {
275  /* check flow timeout based on lastts and state. Both can be
276  * accessed w/o Flow lock as we do have the hash row lock (so flow
277  * can't disappear) and flow_state is atomic. lastts can only
278  * be modified when we have both the flow and hash row lock */
279 
280  int state = SC_ATOMIC_GET(f->flow_state);
281 
282  /* timeout logic goes here */
283  if (FlowManagerFlowTimeout(f, state, ts, emergency) == 0) {
284  f = f->hprev;
285  continue;
286  }
287 
288  /* before grabbing the flow lock, make sure we have at least
289  * 3 packets in the pool */
290  PacketPoolWaitForN(3);
291 
292  FLOWLOCK_WRLOCK(f);
293 
294  Flow *next_flow = f->hprev;
295 
296  /* check if the flow is fully timed out and
297  * ready to be discarded. */
298  if (FlowManagerFlowTimedOut(f, ts) == 1) {
299  /* remove from the hash */
300  if (f->hprev != NULL)
301  f->hprev->hnext = f->hnext;
302  if (f->hnext != NULL)
303  f->hnext->hprev = f->hprev;
304  if (f->fb->head == f)
305  f->fb->head = f->hnext;
306  if (f->fb->tail == f)
307  f->fb->tail = f->hprev;
308 
309  f->hnext = NULL;
310  f->hprev = NULL;
311 
312  if (f->flags & FLOW_TCP_REUSED)
313  counters->tcp_reuse++;
314 
315  if (state == FLOW_STATE_NEW)
316  f->flow_end_flags |= FLOW_END_FLAG_STATE_NEW;
317  else if (state == FLOW_STATE_ESTABLISHED)
318  f->flow_end_flags |= FLOW_END_FLAG_STATE_ESTABLISHED;
319  else if (state == FLOW_STATE_CLOSED)
320  f->flow_end_flags |= FLOW_END_FLAG_STATE_CLOSED;
321 
322  if (emergency)
323  f->flow_end_flags |= FLOW_END_FLAG_EMERGENCY;
324  f->flow_end_flags |= FLOW_END_FLAG_TIMEOUT;
325 
326 // FlowClearMemory (f, f->protomap);
327 
328  /* no one is referring to this flow, use_cnt 0, removed from hash
329  * so we can unlock it and move it back to the spare queue. */
330  FLOWLOCK_UNLOCK(f);
331  FlowEnqueue(&flow_recycle_q, f);
332  /* move to spare list */
333 // FlowMoveToSpare(f);
334 
335  cnt++;
336 
337  switch (state) {
338  case FLOW_STATE_NEW:
339  default:
340  counters->new++;
341  break;
342  case FLOW_STATE_ESTABLISHED:
343  counters->est++;
344  break;
345  case FLOW_STATE_CLOSED:
346  counters->clo++;
347  break;
348  }
349  } else {
350  FLOWLOCK_UNLOCK(f);
351  }
352 
353  f = next_flow;
354  } while (f != NULL);
355 
356  return cnt;
357 }

359 /**
360  * \brief time out flows from the hash
361  *
362  * \param ts timestamp
363  * \param try_cnt number of flows to time out max (0 is unlimited)
364  * \param hash_min min hash index to consider
365  * \param hash_max max hash index to consider
366  * \param counters ptr to FlowTimeoutCounters structure
367  *
368  * \retval cnt number of timed out flow
369  */
370 static uint32_t FlowTimeoutHash(struct timeval *ts, uint32_t try_cnt,
371  uint32_t hash_min, uint32_t hash_max,
372  FlowTimeoutCounters *counters)
373 {
374  uint32_t idx = 0;
375  uint32_t cnt = 0;
376  int emergency = 0;
377 
378  if (SC_ATOMIC_GET(flow_flags) & FLOW_EMERGENCY)
379  emergency = 1;
380 
381  for (idx = hash_min; idx < hash_max; idx++) {
382  FlowBucket *fb = &flow_hash[idx];
383 
384  /* before grabbing the row lock, make sure we have at least
385  * 9 packets in the pool */
386  PacketPoolWaitForN(9);
387 
388  if (FBLOCK_TRYLOCK(fb) != 0)
389  continue;
390 
391  /* flow hash bucket is now locked */
392 
393  if (fb->tail == NULL)
394  goto next;
395 
396  /* we have a flow, or more than one */
397  cnt += FlowManagerHashRowTimeout(fb->tail, ts, emergency, counters);
398 
399 next:
400  FBLOCK_UNLOCK(fb);
401 
402  if (try_cnt > 0 && cnt >= try_cnt)
403  break;
404  }
405 
406  return cnt;
407 }

416  * \retval cnt removed out flows

418 static uint32_t FlowManagerHashRowCleanup(Flow *f)
419 {
420  uint32_t cnt = 0;
421 
422  do {
423  FLOWLOCK_WRLOCK(f);
424 
425  Flow *next_flow = f->hprev;
426 
427  int state = SC_ATOMIC_GET(f->flow_state);
428 
429  /* remove from the hash */
430  if (f->hprev != NULL)
431  f->hprev->hnext = f->hnext;
432  if (f->hnext != NULL)
433  f->hnext->hprev = f->hprev;
434  if (f->fb->head == f)
435  f->fb->head = f->hnext;
436  if (f->fb->tail == f)
437  f->fb->tail = f->hprev;
438 
439  f->hnext = NULL;
440  f->hprev = NULL;
441 
442  if (state == FLOW_STATE_NEW)
443  f->flow_end_flags |= FLOW_END_FLAG_STATE_NEW;
444  else if (state == FLOW_STATE_ESTABLISHED)
445  f->flow_end_flags |= FLOW_END_FLAG_STATE_ESTABLISHED;
446  else if (state == FLOW_STATE_CLOSED)
447  f->flow_end_flags |= FLOW_END_FLAG_STATE_CLOSED;
448 
449  f->flow_end_flags |= FLOW_END_FLAG_SHUTDOWN;
450 
451  /* no one is referring to this flow, use_cnt 0, removed from hash
452  * so we can unlock it and move it to the recycle queue. */
453  FLOWLOCK_UNLOCK(f);
454 
455  FlowEnqueue(&flow_recycle_q, f);
456 
457  cnt++;
458 
459  f = next_flow;
460  } while (f != NULL);
461 
462  return cnt;
463 }
