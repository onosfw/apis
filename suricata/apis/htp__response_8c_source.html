141 /**
142  * Handles request parser state changes. At the moment, this function is used only
143  * to configure data receivers, which are sent raw connection data.
144  *
145  * @param[in] connp
146  * @return HTP_OK, or a value returned from a callback.
147  */
148 static htp_status_t htp_res_handle_state_change(htp_connp_t *connp) {
149  if (connp->out_state_previous == connp->out_state) return HTP_OK;
150 
151  if (connp->out_state == htp_connp_RES_HEADERS) {
152  htp_status_t rc = HTP_OK;
153 
154  switch (connp->out_tx->response_progress) {
155  case HTP_RESPONSE_HEADERS:
156  rc = htp_connp_res_receiver_set(connp, connp->out_tx->cfg->hook_response_header_data);
157  break;
158 
159  case HTP_RESPONSE_TRAILER:
160  rc = htp_connp_res_receiver_set(connp, connp->out_tx->cfg->hook_response_trailer_data);
161  break;
162 
163  default:
164  // Do nothing; receivers are currently used only for header blocks.
165  break;
166  }
167 
168  if (rc != HTP_OK) return rc;
169  }
170 
171  // Same comment as in htp_req_handle_state_change(). Below is a copy.
172 
173  // Initially, I had the finalization of raw data sending here, but that
174  // caused the last REQUEST_HEADER_DATA hook to be invoked after the
175  // REQUEST_HEADERS hook -- which I thought made no sense. For that reason,
176  // the finalization is now initiated from the request header processing code,
177  // which is less elegant but provides a better user experience. Having some
178  // (or all) hooks to be invoked on state change might work better.
179 
180  connp->out_state_previous = connp->out_state;
181 
182  return HTP_OK;

185 /**
186  * If there is any data left in the outbound data chunk, this function will preserve
187  * it for later consumption. The maximum amount accepted for buffering is controlled
188  * by htp_config_t::field_limit_hard.
189  *
190  * @param[in] connp
191  * @return HTP_OK, or HTP_ERROR on fatal failure.
192  */
193 static htp_status_t htp_connp_res_buffer(htp_connp_t *connp) {
194  if (connp->out_current_data == NULL) return HTP_OK;
195 
196  unsigned char *data = connp->out_current_data + connp->out_current_consume_offset;
197  size_t len = connp->out_current_read_offset - connp->out_current_consume_offset;
198 
199  // Check the hard (buffering) limit.
200 
201  size_t newlen = connp->out_buf_size + len;
202 
203  // When calculating the size of the buffer, take into account the
204  // space we're using for the response header buffer.
205  if (connp->out_header != NULL) {
206  newlen += bstr_len(connp->out_header);
207  }
208 
209  if (newlen > connp->out_tx->cfg->field_limit_hard) {
210  htp_log(connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0, "Response the buffer limit: size %zd limit %zd.",
211  newlen, connp->out_tx->cfg->field_limit_hard);
212  return HTP_ERROR;
213  }
214 
215  // Copy the data remaining in the buffer.
216 
217  if (connp->out_buf == NULL) {
218  connp->out_buf = malloc(len);
219  if (connp->out_buf == NULL) return HTP_ERROR;
220  memcpy(connp->out_buf, data, len);
221  connp->out_buf_size = len;
222  } else {
223  size_t newsize = connp->out_buf_size + len;
224  unsigned char *newbuf = realloc(connp->out_buf, newsize);
225  if (newbuf == NULL) return HTP_ERROR;
226  connp->out_buf = newbuf;
227  memcpy(connp->out_buf + connp->out_buf_size, data, len);
228  connp->out_buf_size = newsize;
229  }
230 
231  // Reset the consumer position.
232  connp->out_current_consume_offset = connp->out_current_read_offset;
233 
234  return HTP_OK;
235 }

237 /**
238  * Returns to the caller the memory region that should be processed next. This function
239  * hides away the buffering process from the rest of the code, allowing it to work with
240  * non-buffered data that's in the outbound chunk, or buffered data that's in our structures.
241  *
242  * @param[in] connp
243  * @param[out] data
244  * @param[out] len
245  * @return HTP_OK
246  */
247 static htp_status_t htp_connp_res_consolidate_data(htp_connp_t *connp, unsigned char **data, size_t *len) {
248  if (connp->out_buf == NULL) {
249  // We do not have any data buffered; point to the current data chunk.
250  *data = connp->out_current_data + connp->out_current_consume_offset;
251  *len = connp->out_current_read_offset - connp->out_current_consume_offset;
252  } else {
253  // We do have data in the buffer. Add data from the current
254  // chunk, and point to the consolidated buffer.
255  if (htp_connp_res_buffer(connp) != HTP_OK) {
256  return HTP_ERROR;
257  }
258 
259  *data = connp->out_buf;
260  *len = connp->out_buf_size;
261  }
262 
263  return HTP_OK;
264 }

306 /**
307  * Processes a chunk of data.
308  *
309  * @param[in] connp
310  * @returns HTP_OK on state change, HTP_ERROR on error, or HTP_DATA when more data is needed.
311  */
312 htp_status_t htp_connp_RES_BODY_CHUNKED_DATA(htp_connp_t *connp) {
313  size_t bytes_to_consume;
314 
315  // Determine how many bytes we can consume.
316  if (connp->out_current_len - connp->out_current_read_offset >= connp->out_chunked_length) {
317  bytes_to_consume = connp->out_chunked_length;
318  } else {
319  bytes_to_consume = connp->out_current_len - connp->out_current_read_offset;
320  }
321 
322  if (bytes_to_consume == 0) return HTP_DATA;
323 
324  // Consume the data.
325  htp_status_t rc = htp_tx_res_process_body_data_ex(connp->out_tx, connp->out_current_data + connp->out_current_read_offset, bytes_to_consume);
326  if (rc != HTP_OK) return rc;
327 
328  // Adjust the counters.
329  connp->out_current_read_offset += bytes_to_consume;
330  connp->out_current_consume_offset += bytes_to_consume;
331  connp->out_stream_offset += bytes_to_consume;
332  connp->out_chunked_length -= bytes_to_consume;
333 
334  // Have we seen the entire chunk?
335  if (connp->out_chunked_length == 0) {
336  connp->out_state = htp_connp_RES_BODY_CHUNKED_DATA_END;
337  return HTP_OK;
338  }
339 
340  return HTP_DATA;
341 }

343 /**
344  * Extracts chunk length.
345  *
346  * @param[in] connp
347  * @returns HTP_OK on state change, HTP_ERROR on error, or HTP_DATA when more data is needed.
348  */
349 htp_status_t htp_connp_RES_BODY_CHUNKED_LENGTH(htp_connp_t *connp) {
350  for (;;) {
351  OUT_COPY_BYTE_OR_RETURN(connp);
352 
353  // Have we reached the end of the line?
354  if (connp->out_next_byte == LF) {
355  unsigned char *data;
356  size_t len;
357 
358  if (htp_connp_res_consolidate_data(connp, &data, &len) != HTP_OK) {
359  return HTP_ERROR;
360  }
361 
362  connp->out_tx->response_message_len += len;
363 
364  #ifdef HTP_DEBUG
365  fprint_raw_data(stderr, "Chunk length line", data, len);
366  #endif
367 
368  htp_chomp(data, &len);
369 
370  connp->out_chunked_length = htp_parse_chunked_length(data, len);
371 
372  htp_connp_res_clear_buffer(connp);
373 
374  // Handle chunk length
375  if (connp->out_chunked_length > 0) {
376  // More data available
377  connp->out_state = htp_connp_RES_BODY_CHUNKED_DATA;
378  } else if (connp->out_chunked_length == 0) {
379  // End of data
380  connp->out_state = htp_connp_RES_HEADERS;
381  connp->out_tx->response_progress = HTP_RESPONSE_TRAILER;
382  } else {
383  // Invalid chunk length
384  htp_log(connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0,
385  "Response chunk encoding: Invalid chunk length: %d", connp->out_chunked_length);
386  return HTP_ERROR;
387  }
388 
389  return HTP_OK;
390  }
391  }
392 
393  return HTP_ERROR;
394 }

463 /**
464  * Determines presence (and encoding) of a response body.
465  *
466  * @param[in] connp
467  * @returns HTP_OK on state change, HTP_ERROR on error, or HTP_DATA when more data is needed.
468  */
469 htp_status_t htp_connp_RES_BODY_DETERMINE(htp_connp_t *connp) {
470  // If the request uses the CONNECT method, then not only are we
471  // to assume there's no body, but we need to ignore all
472  // subsequent data in the stream.
473  if (connp->out_tx->request_method_number == HTP_M_CONNECT) {
474  if ((connp->out_tx->response_status_number >= 200)
475  && (connp->out_tx->response_status_number <= 299)) {
476  // This is a successful CONNECT stream, which means
477  // we need to switch into tunneling mode: on the
478  // request side we'll now probe the tunnel data to see
479  // if we need to parse or ignore it. So on the response
480  // side we wrap up the tx and wait.
481  connp->out_state = htp_connp_RES_FINALIZE;
482  return HTP_OK;
483  } else {
484  // This is a failed CONNECT stream, which means that
485  // we can unblock request parsing
486  connp->in_status = HTP_STREAM_DATA;
487 
488  // We are going to continue processing this transaction,
489  // adding a note for ourselves to stop at the end (because
490  // we don't want to see the beginning of a new transaction).
491  connp->out_data_other_at_tx_end = 1;
492  }
493  }
494 
495  // Check for an interim "100 Continue" response. Ignore it if found, and revert back to RES_LINE.
496  if (connp->out_tx->response_status_number == 100) {
497  if (connp->out_tx->seen_100continue != 0) {
498  htp_log(connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0, "Already seen 100-Continue.");
499  return HTP_ERROR;
500  }
501 
502  // Ignore any response headers seen so far.
503  htp_header_t *h = NULL;
504  for (size_t i = 0, n = htp_table_size(connp->out_tx->response_headers); i < n; i++) {
505  h = htp_table_get_index(connp->out_tx->response_headers, i, NULL);
506  bstr_free(h->name);
507  bstr_free(h->value);
508  free(h);
509  }
510 
511  htp_table_clear(connp->out_tx->response_headers);
512 
513  // Expecting to see another response line next.
514  connp->out_state = htp_connp_RES_LINE;
515  connp->out_tx->response_progress = HTP_RESPONSE_LINE;
516  connp->out_tx->seen_100continue++;
517 
518  return HTP_OK;
519  }
520 
521  // 1. Any response message which MUST NOT include a message-body
522  // (such as the 1xx, 204, and 304 responses and any response to a HEAD
523  // request) is always terminated by the first empty line after the
524  // header fields, regardless of the entity-header fields present in the
525  // message.
526  if (((connp->out_tx->response_status_number >= 100) && (connp->out_tx->response_status_number <= 199))
527  || (connp->out_tx->response_status_number == 204) || (connp->out_tx->response_status_number == 304)
528  || (connp->out_tx->request_method_number == HTP_M_HEAD)) {
529  // There's no response body
530  connp->out_tx->response_transfer_coding = HTP_CODING_NO_BODY;
531  connp->out_state = htp_connp_RES_FINALIZE;
532  } else {
533  // We have a response body
534 
535  htp_header_t *ct = htp_table_get_c(connp->out_tx->response_headers, "content-type");
536  htp_header_t *cl = htp_table_get_c(connp->out_tx->response_headers, "content-length");
537  htp_header_t *te = htp_table_get_c(connp->out_tx->response_headers, "transfer-encoding");
538 
539  if (ct != NULL) {
540  connp->out_tx->response_content_type = bstr_dup_lower(ct->value);
541  if (connp->out_tx->response_content_type == NULL) return HTP_ERROR;
542 
543  // Ignore parameters
544  unsigned char *data = bstr_ptr(connp->out_tx->response_content_type);
545  size_t len = bstr_len(ct->value);
546  size_t newlen = 0;
547  while (newlen < len) {
548  // TODO Some platforms may do things differently here.
549  if (htp_is_space(data[newlen]) || (data[newlen] == ';')) {
550  bstr_adjust_len(connp->out_tx->response_content_type, newlen);
551  break;
552  }
553 
554  newlen++;
555  }
556  }
557 
558  // 2. If a Transfer-Encoding header field (section 14.40) is present and
559  // indicates that the "chunked" transfer coding has been applied, then
560  // the length is defined by the chunked encoding (section 3.6).
561  if ((te != NULL) && (bstr_cmp_c_nocase(te->value, "chunked") == 0)) {
562  // If the T-E header is present we are going to use it.
563  connp->out_tx->response_transfer_coding = HTP_CODING_CHUNKED;
564 
565  // We are still going to check for the presence of C-L
566  if (cl != NULL) {
567  // This is a violation of the RFC
568  connp->out_tx->flags |= HTP_REQUEST_SMUGGLING;
569  }
570 
571  connp->out_state = htp_connp_RES_BODY_CHUNKED_LENGTH;
572  connp->out_tx->response_progress = HTP_RESPONSE_BODY;
573  }// 3. If a Content-Length header field (section 14.14) is present, its
574  // value in bytes represents the length of the message-body.
575  else if (cl != NULL) {
576  // We know the exact length
577  connp->out_tx->response_transfer_coding = HTP_CODING_IDENTITY;
578 
579  // Check for multiple C-L headers
580  if (cl->flags & HTP_FIELD_REPEATED) {
581  connp->out_tx->flags |= HTP_REQUEST_SMUGGLING;
582  }
583 
584  // Get body length
585  connp->out_tx->response_content_length = htp_parse_content_length(cl->value);
586  if (connp->out_tx->response_content_length < 0) {
587  htp_log(connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0, "Invalid C-L field in response: %d",
588  connp->out_tx->response_content_length);
589  return HTP_ERROR;
590  } else {
591  connp->out_content_length = connp->out_tx->response_content_length;
592  connp->out_body_data_left = connp->out_content_length;
593 
594  if (connp->out_content_length != 0) {
595  connp->out_state = htp_connp_RES_BODY_IDENTITY_CL_KNOWN;
596  connp->out_tx->response_progress = HTP_RESPONSE_BODY;
597  } else {
598  connp->out_state = htp_connp_RES_FINALIZE;
599  }
600  }
601  } else {
602  // 4. If the message uses the media type "multipart/byteranges", which is
603  // self-delimiting, then that defines the length. This media type MUST
604  // NOT be used unless the sender knows that the recipient can parse it;
605  // the presence in a request of a Range header with multiple byte-range
606  // specifiers implies that the client can parse multipart/byteranges
607  // responses.
608  if (ct != NULL) {
609  // TODO Handle multipart/byteranges
610  if (bstr_index_of_c_nocase(ct->value, "multipart/byteranges") != -1) {
611  htp_log(connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0,
612  "C-T multipart/byteranges in responses not supported");
613  return HTP_ERROR;
614  }
615  }
616 
617  // 5. By the server closing the connection. (Closing the connection
618  // cannot be used to indicate the end of a request body, since that
619  // would leave no possibility for the server to send back a response.)
620  connp->out_state = htp_connp_RES_BODY_IDENTITY_STREAM_CLOSE;
621  connp->out_tx->response_transfer_coding = HTP_CODING_IDENTITY;
622  connp->out_tx->response_progress = HTP_RESPONSE_BODY;
623  connp->out_body_data_left = -1;
624  }
625  }
626 
627  // NOTE We do not need to check for short-style HTTP/0.9 requests here because
628  // that is done earlier, before response line parsing begins
629 
630  htp_status_t rc = htp_tx_state_response_headers(connp->out_tx);
631  if (rc != HTP_OK) return rc;
632 
633  return HTP_OK;

636 /**
637  * Parses response headers.
638  *
639  * @param[in] connp
640  * @returns HTP_OK on state change, HTP_ERROR on error, or HTP_DATA when more data is needed.
641  */
642 htp_status_t htp_connp_RES_HEADERS(htp_connp_t *connp) {
643  for (;;) {
644  OUT_COPY_BYTE_OR_RETURN(connp);
645 
646  // Have we reached the end of the line?
647  if (connp->out_next_byte == LF) {
648  unsigned char *data;
649  size_t len;
650 
651  if (htp_connp_res_consolidate_data(connp, &data, &len) != HTP_OK) {
652  return HTP_ERROR;
653  }
654 
655  #ifdef HTP_DEBUG
656  fprint_raw_data(stderr, __FUNCTION__, data, len);
657  #endif
658 
659  // Should we terminate headers?
660  if (htp_connp_is_line_terminator(connp, data, len)) {
661  // Parse previous header, if any.
662  if (connp->out_header != NULL) {
663  if (connp->cfg->process_response_header(connp, bstr_ptr(connp->out_header),
664  bstr_len(connp->out_header)) != HTP_OK) return HTP_ERROR;
665 
666  bstr_free(connp->out_header);
667  connp->out_header = NULL;
668  }
669 
670  htp_connp_res_clear_buffer(connp);
671 
672  // We've seen all response headers.
673  if (connp->out_tx->response_progress == HTP_RESPONSE_HEADERS) {
674  // Response headers.
675 
676  // The next step is to determine if this response has a body.
677  connp->out_state = htp_connp_RES_BODY_DETERMINE;
678  } else {
679  // Response trailer.
680 
681  // Finalize sending raw trailer data.
682  htp_status_t rc = htp_connp_res_receiver_finalize_clear(connp);
683  if (rc != HTP_OK) return rc;
684 
685  // Run hook response_TRAILER.
686  rc = htp_hook_run_all(connp->cfg->hook_response_trailer, connp->out_tx);
687  if (rc != HTP_OK) return rc;
688 
689  // The next step is to finalize this response.
690  connp->out_state = htp_connp_RES_FINALIZE;
691  }
692 
693  return HTP_OK;
694  }
695 
696  htp_chomp(data, &len);
697 
698  // Check for header folding.
699  if (htp_connp_is_line_folded(data, len) == 0) {
700  // New header line.
701 
702  // Parse previous header, if any.
703  if (connp->out_header != NULL) {
704  if (connp->cfg->process_response_header(connp, bstr_ptr(connp->out_header),
705  bstr_len(connp->out_header)) != HTP_OK) return HTP_ERROR;
706 
707  bstr_free(connp->out_header);
708  connp->out_header = NULL;
709  }
710 
711  OUT_PEEK_NEXT(connp);
712 
713  if (htp_is_folding_char(connp->out_next_byte) == 0) {
714  // Because we know this header is not folded, we can process the buffer straight away.
715  if (connp->cfg->process_response_header(connp, data, len) != HTP_OK) return HTP_ERROR;
716  } else {
717  // Keep the partial header data for parsing later.
718  connp->out_header = bstr_dup_mem(data, len);
719  if (connp->out_header == NULL) return HTP_ERROR;
720  }
721  } else {
722  // Folding; check that there's a previous header line to add to.
723  if (connp->out_header == NULL) {
724  // Invalid folding.
725 
726  // Warn only once per transaction.
727  if (!(connp->out_tx->flags & HTP_INVALID_FOLDING)) {
728  connp->out_tx->flags |= HTP_INVALID_FOLDING;
729  htp_log(connp, HTP_LOG_MARK, HTP_LOG_WARNING, 0, "Invalid response field folding");
730  }
731 
732  // Keep the header data for parsing later.
733  connp->out_header = bstr_dup_mem(data, len);
734  if (connp->out_header == NULL) return HTP_ERROR;
735  } else {
736  // Add to the existing header.
737  bstr *new_out_header = bstr_add_mem(connp->out_header, data, len);
738  if (new_out_header == NULL) return HTP_ERROR;
739  connp->out_header = new_out_header;
740  }
741  }
742 
743  htp_connp_res_clear_buffer(connp);
744  }
745  }
746 
747  return HTP_ERROR;
748 }

750 /**
751  * Parses response line.
752  *
753  * @param[in] connp
754  * @returns HTP_OK on state change, HTP_ERROR on error, or HTP_DATA when more data is needed.
755  */
756 htp_status_t htp_connp_RES_LINE(htp_connp_t *connp) {
757  for (;;) {
758  // Don't try to get more data if the stream is closed. If we do, we'll return, asking for more data.
759  if (connp->out_status != HTP_STREAM_CLOSED) {
760  // Get one byte
761  OUT_COPY_BYTE_OR_RETURN(connp);
762  }
763 
764  // Have we reached the end of the line? We treat stream closure as end of line in
765  // order to handle the case when the first line of the response is actually response body
766  // (and we wish it processed as such).
767  if ((connp->out_next_byte == LF)||(connp->out_status == HTP_STREAM_CLOSED)) {
768  unsigned char *data;
769  size_t len;
770 
771  if (htp_connp_res_consolidate_data(connp, &data, &len) != HTP_OK) {
772  return HTP_ERROR;
773  }
774 
775  #ifdef HTP_DEBUG
776  fprint_raw_data(stderr, __FUNCTION__, data, len);
777  #endif
778 
779  // Is this a line that should be ignored?
780  if (htp_connp_is_line_ignorable(connp, data, len)) {
781  // We have an empty/whitespace line, which we'll note, ignore and move on
782  connp->out_tx->response_ignored_lines++;
783 
784  // TODO How many lines are we willing to accept?
785 
786  // Start again
787  htp_connp_res_clear_buffer(connp);
788 
789  return HTP_OK;
790  }
791 
792  // Deallocate previous response line allocations, which we would have on a 100 response.
793 
794  if (connp->out_tx->response_line != NULL) {
795  bstr_free(connp->out_tx->response_line);
796  connp->out_tx->response_line = NULL;
797  }
798 
799  if (connp->out_tx->response_protocol != NULL) {
800  bstr_free(connp->out_tx->response_protocol);
801  connp->out_tx->response_protocol = NULL;
802  }
803 
804  if (connp->out_tx->response_status != NULL) {
805  bstr_free(connp->out_tx->response_status);
806  connp->out_tx->response_status = NULL;
807  }
808 
809  if (connp->out_tx->response_message != NULL) {
810  bstr_free(connp->out_tx->response_message);
811  connp->out_tx->response_message = NULL;
812  }
813 
814  // Process response line.
815 
816  int chomp_result = htp_chomp(data, &len);
817 
818  connp->out_tx->response_line = bstr_dup_mem(data, len);
819  if (connp->out_tx->response_line == NULL) return HTP_ERROR;
820 
821  if (connp->cfg->parse_response_line(connp) != HTP_OK) return HTP_ERROR;
822 
823  // If the response line is invalid, determine if it _looks_ like
824  // a response line. If it does not look like a line, process the
825  // data as a response body because that is what browsers do.
826 
827  if (htp_treat_response_line_as_body(connp->out_tx)) {
828  connp->out_tx->response_content_encoding_processing = HTP_COMPRESSION_NONE;
829 
830  htp_status_t rc = htp_tx_res_process_body_data_ex(connp->out_tx, data, len + chomp_result);
831  if (rc != HTP_OK) return rc;
832 
833  // Continue to process response body. Because we don't have
834  // any headers to parse, we assume the body continues until
835  // the end of the stream.
836  connp->out_tx->response_transfer_coding = HTP_CODING_IDENTITY;
837  connp->out_tx->response_progress = HTP_RESPONSE_BODY;
838  connp->out_state = htp_connp_RES_BODY_IDENTITY_STREAM_CLOSE;
839  connp->out_body_data_left = -1;
840 
841  return HTP_OK;
842  }
843 
844  htp_status_t rc = htp_tx_state_response_line(connp->out_tx);
845  if (rc != HTP_OK) return rc;
846 
847  htp_connp_res_clear_buffer(connp);
848 
849  // Move on to the next phase.
850  connp->out_state = htp_connp_RES_HEADERS;
851  connp->out_tx->response_progress = HTP_RESPONSE_HEADERS;
852 
853  return HTP_OK;
854  }
855  }
856 
857  return HTP_ERROR;
858 }

868 /**
869  * The response idle state will initialize response processing, as well as
870  * finalize each transactions after we are done with it.
871  *
872  * @param[in] connp
873  * @returns HTP_OK on state change, HTP_ERROR on error, or HTP_DATA when more data is needed.
874  */
875 htp_status_t htp_connp_RES_IDLE(htp_connp_t *connp) {
876  // We want to start parsing the next response (and change
877  // the state from IDLE) only if there's at least one
878  // byte of data available. Otherwise we could be creating
879  // new structures even if there's no more data on the
880  // connection.
881  OUT_TEST_NEXT_BYTE_OR_RETURN(connp);
882 
883  // Parsing a new response
884

886  connp->out_tx = htp_list_get(connp->conn->transactions, connp->out_next_tx_index);
887  if (connp->out_tx == NULL) {

891 
892  // We've used one transaction
893  connp->out_next_tx_index++;

895  // TODO Detect state mismatch
896 
897  connp->out_content_length = -1;
898  connp->out_body_data_left = -1;

906 int htp_connp_res_data(htp_connp_t *connp, const htp_time_t *timestamp, const void *data, size_t len) {
907  #ifdef HTP_DEBUG
908  fprintf(stderr, "htp_connp_res_data(connp->out_status %x)\n", connp->out_status);
909  fprint_raw_data(stderr, __FUNCTION__, data, len);
910  #endif
911 
912  // Return if the connection is in stop state
913  if (connp->out_status == HTP_STREAM_STOP) {
914  htp_log(connp, HTP_LOG_MARK, HTP_LOG_INFO, 0, "Outbound parser is in HTP_STREAM_STOP");
915 
916  return HTP_STREAM_STOP;
917  }
918 
919  // Return if the connection has had a fatal error
920  if (connp->out_status == HTP_STREAM_ERROR) {
921  htp_log(connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0, "Outbound parser is in HTP_STREAM_ERROR");
922 
923  #ifdef HTP_DEBUG
924  fprintf(stderr, "htp_connp_res_data: returning HTP_STREAM_DATA (previous error)\n");
925  #endif
926 
927  return HTP_STREAM_ERROR;
928  }
929 
930  // Sanity check: we must have a transaction pointer if the state is not IDLE (no outbound transaction)
931  if ((connp->out_tx == NULL)&&(connp->out_state != htp_connp_RES_IDLE)) {
932  connp->out_status = HTP_STREAM_ERROR;
933 
934  htp_log(connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0, "Missing outbound transaction data");
935 
936  return HTP_STREAM_ERROR;
937  }
938 
939  // If the length of the supplied data chunk is zero, proceed
940  // only if the stream has been closed. We do not allow zero-sized
941  // chunks in the API, but we use it internally to force the parsers
942  // to finalize parsing.
943  if (((data == NULL) || (len == 0)) && (connp->out_status != HTP_STREAM_CLOSED)) {
944  htp_log(connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0, "Zero-length data chunks are not allowed");
945 
946  #ifdef HTP_DEBUG
947  fprintf(stderr, "htp_connp_res_data: returning HTP_STREAM_DATA (zero-length chunk)\n");
948  #endif
949 
950  return HTP_STREAM_CLOSED;
951  }
952 
953  // Remember the timestamp of the current response data chunk
954  if (timestamp != NULL) {
955  memcpy(&connp->out_timestamp, timestamp, sizeof (*timestamp));
956  }
957 
958  // Store the current chunk information
959  connp->out_current_data = (unsigned char *) data;
960  connp->out_current_len = len;
961  connp->out_current_read_offset = 0;
962  connp->out_current_consume_offset = 0;
963  connp->out_current_receiver_offset = 0;
964 
965  htp_conn_track_outbound_data(connp->conn, len, timestamp);
966 
967  // Return without processing any data if the stream is in tunneling
968  // mode (which it would be after an initial CONNECT transaction.
969  if (connp->out_status == HTP_STREAM_TUNNEL) {
970  #ifdef HTP_DEBUG
971  fprintf(stderr, "htp_connp_res_data: returning HTP_STREAM_TUNNEL\n");
972  #endif
973 
974  return HTP_STREAM_TUNNEL;
975  }
976 
977  // Invoke a processor, in a loop, until an error
978  // occurs or until we run out of data. Many processors
979  // will process a request, each pointing to the next
980  // processor that needs to run.
981  for (;;) {
982  #ifdef HTP_DEBUG
983  fprintf(stderr, "htp_connp_res_data: out state=%s, progress=%s\n",
984  htp_connp_out_state_as_string(connp),
985  htp_tx_response_progress_as_string(connp->out_tx));
986  #endif
987 
988  // Return if there's been an error
989  // or if we've run out of data. We are relying
990  // on processors to add error messages, so we'll
991  // keep quiet here.
992  htp_status_t rc = connp->out_state(connp);
993  if (rc == HTP_OK) {
994  if (connp->out_status == HTP_STREAM_TUNNEL) {
995  #ifdef HTP_DEBUG
996  fprintf(stderr, "htp_connp_res_data: returning HTP_STREAM_TUNNEL\n");
997  #endif
998 
999  return HTP_STREAM_TUNNEL;
1000  }
1001 
1002  rc = htp_res_handle_state_change(connp);
1003  }
1004 
1005  if (rc != HTP_OK) {
1006  // Do we need more data?
1007  if ((rc == HTP_DATA) || (rc == HTP_DATA_BUFFER)) {
1008  htp_connp_res_receiver_send_data(connp, 0 /* not last */);
1009 
1010  if (rc == HTP_DATA_BUFFER) {
1011  if (htp_connp_res_buffer(connp) != HTP_OK) {
1012  connp->out_status = HTP_STREAM_ERROR;
1013  return HTP_STREAM_ERROR;
1014  }
1015  }
1016 
1017  #ifdef HTP_DEBUG
1018  fprintf(stderr, "htp_connp_res_data: returning HTP_STREAM_DATA\n");
1019  #endif
1020 
1021  connp->out_status = HTP_STREAM_DATA;
1022 
1023  return HTP_STREAM_DATA;
1024  }
1025 
1026  // Check for stop
1027  if (rc == HTP_STOP) {
1028  #ifdef HTP_DEBUG
1029  fprintf(stderr, "htp_connp_res_data: returning HTP_STREAM_STOP\n");
1030  #endif
1031 
1032  connp->out_status = HTP_STREAM_STOP;
1033 
1034  return HTP_STREAM_STOP;
1035  }
1036 
1037  // Check for suspended parsing
1038  if (rc == HTP_DATA_OTHER) {
1039  // We might have actually consumed the entire data chunk?
1040  if (connp->out_current_read_offset >= connp->out_current_len) {
1041  #ifdef HTP_DEBUG
1042  fprintf(stderr, "htp_connp_res_data: returning HTP_STREAM_DATA (suspended parsing)\n");
1043  #endif
1044 
1045  connp->out_status = HTP_STREAM_DATA;
1046 
1047  // Do not send STREAM_DATE_DATA_OTHER if we've
1048  // consumed the entire chunk
1049  return HTP_STREAM_DATA;
1050  } else {
1051  #ifdef HTP_DEBUG
1052  fprintf(stderr, "htp_connp_res_data: returning HTP_STREAM_DATA_OTHER\n");
1053  #endif
1054 
1055  connp->out_status = HTP_STREAM_DATA_OTHER;
1056 
1057  // Partial chunk consumption
1058  return HTP_STREAM_DATA_OTHER;
1059  }
1060  }
1061 
1062  #ifdef HTP_DEBUG
1063  fprintf(stderr, "htp_connp_res_data: returning HTP_STREAM_ERROR\n");
1064  #endif
1065 
1066  // Permanent stream error.
1067  connp->out_status = HTP_STREAM_ERROR;
1068 
1069  return HTP_STREAM_ERROR;
1070  }
1071  }
1072 
1073  // Permanent stream error.
1074  connp->out_status = HTP_STREAM_ERROR;
1075 
1076  return HTP_STREAM_ERROR;
