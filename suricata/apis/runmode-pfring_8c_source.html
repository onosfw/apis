94  char *threadsstr = NULL;
95  PfringIfaceConfig *pfconf = SCMalloc(sizeof(*pfconf));

97 #ifdef HAVE_PFRING

101 
102  if (unlikely(pfconf == NULL)) {
103  return NULL;
104  }
105 
106  if (iface == NULL) {
107  SCFree(pfconf);
108  return NULL;

110 
111  strlcpy(pfconf->iface, iface, sizeof(pfconf->iface));
112  pfconf->flags = 0;
113  pfconf->threads = 1;
114  pfconf->cluster_id = 1;
115 #ifdef HAVE_PFRING
116  pfconf->ctype = default_ctype;

118  pfconf->DerefFunc = PfringDerefConfig;

122 
123  /* Find initial node */
124  if (ConfGet("pfring.threads", &threadsstr) != 1) {

126  } else {

132  pfconf->threads = 1;
133  }

138  if (strncmp(pfconf->iface, "zc", 2) == 0) {

150

153  SCLogInfo("ZC interface detected, not setting cluster type for PF_RING (iface %s)",
154  pfconf->iface);
155  } else if ((pfconf->threads == 1) && (strncmp(pfconf->iface, "dna", 3) == 0)) {
156  SCLogInfo("DNA interface detected, not setting cluster type for PF_RING (iface %s)",
157  pfconf->iface);

161  SCLogInfo("Using round-robin cluster mode for PF_RING (iface %s)",

163  pfconf->ctype = (cluster_type)tmpctype;

178 /**
179  * \brief extract information from config file
180  *
181  * The returned structure will be freed by the thread init function.
182  * This is thus necessary to or copy the structure before giving it
183  * to thread or to reparse the file for each thread (and thus have
184  * new structure.
185  *
186  * If old config system is used, then return the smae parameters
187  * value for each interface.
188  *
189  * \return a PfringIfaceConfig corresponding to the interface name
190  */
191 void *ParsePfringConfig(const char *iface)
192 {
193  char *threadsstr = NULL;
194  ConfNode *if_root;
195  ConfNode *if_default = NULL;
196  ConfNode *pf_ring_node;
197  PfringIfaceConfig *pfconf = SCMalloc(sizeof(*pfconf));
198  char *tmpclusterid;
199  char *tmpctype = NULL;
200 #ifdef HAVE_PFRING
201  cluster_type default_ctype = CLUSTER_ROUND_ROBIN;
202  int getctype = 0;
203 #endif
204  char *bpf_filter = NULL;
205 
206  if (unlikely(pfconf == NULL)) {
207  return NULL;
208  }
209 
210  if (iface == NULL) {
211  SCFree(pfconf);
212  return NULL;
213  }
214 
215  memset(pfconf, 0, sizeof(PfringIfaceConfig));
216  strlcpy(pfconf->iface, iface, sizeof(pfconf->iface));
217  pfconf->threads = 1;
218  pfconf->cluster_id = 1;
219 #ifdef HAVE_PFRING
220  pfconf->ctype = (cluster_type)default_ctype;
221 #endif
222  pfconf->DerefFunc = PfringDerefConfig;
223  SC_ATOMIC_INIT(pfconf->ref);
224  (void) SC_ATOMIC_ADD(pfconf->ref, 1);
225 
226  /* Find initial node */
227  pf_ring_node = ConfGetNode("pfring");
228  if (pf_ring_node == NULL) {
229  SCLogInfo("Unable to find pfring config using default value");
230  return pfconf;
231  }
232 
233  if_root = ConfNodeLookupKeyValue(pf_ring_node, "interface", iface);
234 
235  if_default = ConfNodeLookupKeyValue(pf_ring_node, "interface", "default");
236 
237  if (if_root == NULL && if_default == NULL) {
238  SCLogInfo("Unable to find pfring config for "
239  "interface %s, using default value or 1.0 "
240  "configuration system. ",
241  iface);
242  return pfconf;
243  }
244 
245  /* If there is no setting for current interface use default one as main iface */
246  if (if_root == NULL) {
247  if_root = if_default;
248  if_default = NULL;
249  }
250 
251  if (ConfGetChildValueWithDefault(if_root, if_default, "threads", &threadsstr) != 1) {
252  pfconf->threads = 1;
253  } else {
254  if (threadsstr != NULL) {
255  pfconf->threads = (uint8_t)atoi(threadsstr);
256  }
257  }
258  if (pfconf->threads == 0) {
259  pfconf->threads = 1;
260  }
261 
262  SC_ATOMIC_RESET(pfconf->ref);
263  (void) SC_ATOMIC_ADD(pfconf->ref, pfconf->threads);
264 
265  /* command line value has precedence */
266  if (ConfGet("pfring.cluster-id", &tmpclusterid) == 1) {
267  pfconf->cluster_id = (uint16_t)atoi(tmpclusterid);
268  pfconf->flags |= PFRING_CONF_FLAGS_CLUSTER;
269  SCLogDebug("Going to use command-line provided cluster-id %" PRId32,
270  pfconf->cluster_id);
271  } else {
272 
273  if (strncmp(pfconf->iface, "zc", 2) == 0) {
274  SCLogInfo("ZC interface detected, not setting cluster-id for PF_RING (iface %s)",
275  pfconf->iface);
276  } else if ((pfconf->threads == 1) && (strncmp(pfconf->iface, "dna", 3) == 0)) {
277  SCLogInfo("DNA interface detected, not setting cluster-id for PF_RING (iface %s)",
278  pfconf->iface);
279  } else if (ConfGetChildValueWithDefault(if_root, if_default, "cluster-id", &tmpclusterid) != 1) {
280  SCLogError(SC_ERR_INVALID_ARGUMENT,
281  "Could not get cluster-id from config");
282  } else {
283  pfconf->cluster_id = (uint16_t)atoi(tmpclusterid);
284  pfconf->flags |= PFRING_CONF_FLAGS_CLUSTER;
285  SCLogDebug("Going to use cluster-id %" PRId32, pfconf->cluster_id);
286  }
287  }
288 
289  /*load pfring bpf filter*/
290  /* command line value has precedence */
291  if (ConfGet("bpf-filter", &bpf_filter) == 1) {
292  if (strlen(bpf_filter) > 0) {
293  pfconf->bpf_filter = SCStrdup(bpf_filter);
294  if (unlikely(pfconf->bpf_filter == NULL)) {
295  SCLogError(SC_ERR_MEM_ALLOC,
296  "Can't allocate BPF filter string");
297  } else {
298  SCLogDebug("Going to use command-line provided bpf filter %s",
299  pfconf->bpf_filter);
300  }
301  }
302  } else {
303  if (ConfGetChildValueWithDefault(if_root, if_default, "bpf-filter", &bpf_filter) == 1) {
304  if (strlen(bpf_filter) > 0) {
305  pfconf->bpf_filter = SCStrdup(bpf_filter);
306  if (unlikely(pfconf->bpf_filter == NULL)) {
307  SCLogError(SC_ERR_MEM_ALLOC,
308  "Can't allocate BPF filter string");
309  } else {
310  SCLogDebug("Going to use bpf filter %s",
311  pfconf->bpf_filter);
312  }
313  }
314  }
315  }
316 
317 #ifdef HAVE_PFRING
318  if (ConfGet("pfring.cluster-type", &tmpctype) == 1) {
319  SCLogDebug("Going to use command-line provided cluster-type");
320  getctype = 1;
321  } else {
322  if (strncmp(pfconf->iface, "zc", 2) == 0) {
323  SCLogInfo("ZC interface detected, not setting cluster type for PF_RING (iface %s)",
324  pfconf->iface);
325  } else if ((pfconf->threads == 1) && (strncmp(pfconf->iface, "dna", 3) == 0)) {
326  SCLogInfo("DNA interface detected, not setting cluster type for PF_RING (iface %s)",
327  pfconf->iface);
328  } else if (ConfGetChildValueWithDefault(if_root, if_default, "cluster-type", &tmpctype) != 1) {
329  SCLogError(SC_ERR_GET_CLUSTER_TYPE_FAILED,
330  "Could not get cluster-type from config");
331  } else {
332  getctype = 1;
333  }
334  }
335 
336  if (getctype) {
337  if (strcmp(tmpctype, "cluster_round_robin") == 0) {
338  SCLogInfo("Using round-robin cluster mode for PF_RING (iface %s)",
339  pfconf->iface);
340  pfconf->ctype = CLUSTER_ROUND_ROBIN;
341  } else if (strcmp(tmpctype, "cluster_flow") == 0) {
342  SCLogInfo("Using flow cluster mode for PF_RING (iface %s)",
343  pfconf->iface);
344  pfconf->ctype = CLUSTER_FLOW;
345  } else {
346  SCLogError(SC_ERR_INVALID_CLUSTER_TYPE,
347  "invalid cluster-type %s",
348  tmpctype);
349  SCFree(pfconf);
350  return NULL;
351  }
352  }
353 #endif /* HAVE_PFRING */
354  if (ConfGetChildValueWithDefault(if_root, if_default, "checksum-checks", &tmpctype) == 1) {
355  if (strcmp(tmpctype, "auto") == 0) {
356  pfconf->checksum_mode = CHECKSUM_VALIDATION_AUTO;
357  } else if (strcmp(tmpctype, "yes") == 0) {
358  pfconf->checksum_mode = CHECKSUM_VALIDATION_ENABLE;
359  } else if (strcmp(tmpctype, "no") == 0) {
360  pfconf->checksum_mode = CHECKSUM_VALIDATION_DISABLE;
361  } else if (strcmp(tmpctype, "rx-only") == 0) {
362  pfconf->checksum_mode = CHECKSUM_VALIDATION_RXONLY;
363  } else {
364  SCLogError(SC_ERR_INVALID_ARGUMENT, "Invalid value for checksum-checks for %s", pfconf->iface);
365  }
366  }
367 
368  return pfconf;
369 }
