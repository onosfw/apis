202 #define SCLog(x, ...) \
203  do { \
204  if (sc_log_global_log_level >= x && \
205  (sc_log_fg_filters_present == 0 || \
206  SCLogMatchFGFilterWL(__FILE__, __FUNCTION__, __LINE__) == 1 || \
207  SCLogMatchFGFilterBL(__FILE__, __FUNCTION__, __LINE__) == 1) && \
208  (sc_log_fd_filters_present == 0 || \
209  SCLogMatchFDFilter(__FUNCTION__) == 1)) \
210  { \
211  char _sc_log_msg[SC_LOG_MAX_LOG_MSG_LEN] = ""; \
212  \
213  snprintf(_sc_log_msg, SC_LOG_MAX_LOG_MSG_LEN, __VA_ARGS__); \
214  \
215  SCLogMessage(x, \
216  __FILE__, \
217  __LINE__, \
218  __FUNCTION__, SC_OK, _sc_log_msg); \
219  } \
220  } while(0)
