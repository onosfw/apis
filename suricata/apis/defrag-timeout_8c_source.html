38 /** \internal
39  * \brief See if we can really discard this tracker. Check use_cnt reference.
40  *
41  * \param dt tracker
42  * \param ts timestamp
43  *
44  * \retval 0 not timed out just yet
45  * \retval 1 fully timed out, lets kill it
46  */
47 static int DefragTrackerTimedOut(DefragTracker *dt, struct timeval *ts)
48 {
49  /** never prune a trackers that is used by a packet
50  * we are currently processing in one of the threads */
51  if (SC_ATOMIC_GET(dt->use_cnt) > 0) {
52  return 0;
53  }
54 
55  /* retain if remove is not set and not timed out */
56  if (!dt->remove && timercmp(&dt->timeout, ts, >))
57  return 0;
58 
59  return 1;
60 }

62 /**
63  * \internal
64  *
65  * \brief check all trackers in a hash row for timing out
66  *
67  * \param hb tracker hash row *LOCKED*
68  * \param dt last tracker in the hash row
69  * \param ts timestamp
70  *
71  * \retval cnt timed out tracker
72  */
73 static uint32_t DefragTrackerHashRowTimeout(DefragTrackerHashRow *hb, DefragTracker *dt, struct timeval *ts)
74 {
75  uint32_t cnt = 0;
76 
77  do {
78  if (SCMutexTrylock(&dt->lock) != 0) {
79  dt = dt->hprev;
80  continue;
81  }
82 
83  DefragTracker *next_dt = dt->hprev;
84 
85  /* check if the tracker is fully timed out and
86  * ready to be discarded. */
87  if (DefragTrackerTimedOut(dt, ts) == 1) {
88  /* remove from the hash */
89  if (dt->hprev != NULL)
90  dt->hprev->hnext = dt->hnext;
91  if (dt->hnext != NULL)
92  dt->hnext->hprev = dt->hprev;
93  if (hb->head == dt)
94  hb->head = dt->hnext;
95  if (hb->tail == dt)
96  hb->tail = dt->hprev;
97 
98  dt->hnext = NULL;
99  dt->hprev = NULL;
100 
101  DefragTrackerClearMemory(dt);
102 
103  /* no one is referring to this tracker, use_cnt 0, removed from hash
104  * so we can unlock it and move it back to the spare queue. */
105  SCMutexUnlock(&dt->lock);
106 
107  /* move to spare list */
108  DefragTrackerMoveToSpare(dt);
109 
110  cnt++;
111  } else {
112  SCMutexUnlock(&dt->lock);
113  }
114 
115  dt = next_dt;
116  } while (dt != NULL);
117 
118  return cnt;
119 }

121 /**
122  * \brief time out tracker from the hash
123  *
124  * \param ts timestamp
125  *
126  * \retval cnt number of timed out tracker
127  */
128 uint32_t DefragTimeoutHash(struct timeval *ts)
129 {
130  uint32_t idx = 0;
131  uint32_t cnt = 0;
132 
133  for (idx = 0; idx < defrag_config.hash_size; idx++) {
134  DefragTrackerHashRow *hb = &defragtracker_hash[idx];
135 
136  if (DRLOCK_TRYLOCK(hb) != 0)
137  continue;
138 
139  /* defrag hash bucket is now locked */
140 
141  if (hb->tail == NULL) {
142  DRLOCK_UNLOCK(hb);
143  continue;
144  }
145 
146  /* we have a tracker, or more than one */
147  cnt += DefragTrackerHashRowTimeout(hb, hb->tail, ts);
148  DRLOCK_UNLOCK(hb);
149  }
150 
151  return cnt;
152 }
