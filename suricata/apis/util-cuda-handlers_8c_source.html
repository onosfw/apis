53 void CudaHandlerAddCudaProfileFromConf(const char *name,
54  void *(*Callback)(ConfNode *node),
55  void (*Free)(void *))
56 {
57  /* we don't do data validation */
58  SCMutexLock(&mutex);
59 
60  CudaHandlerConfProfile *tmp_cp = conf_profiles;
61  while (tmp_cp != NULL && strcasecmp(name, tmp_cp->name) != 0)
62  tmp_cp = tmp_cp->next;
63 
64  if (tmp_cp != NULL) {
65  SCLogError(SC_ERR_INVALID_ARGUMENT, "We already have a cuda conf "
66  "profile by the name \"%s\" registered.", name);
67  exit(EXIT_FAILURE);
68  }
69 
70  char tmp[200];
71  int r = snprintf(tmp, sizeof(tmp), "%s%s", "cuda.", name);
72  if (r < 0) {
73  SCLogError(SC_ERR_FATAL, "snprintf failure.");
74  exit(EXIT_FAILURE);
75  } else if (r > (int)sizeof(tmp)) {
76  SCLogError(SC_ERR_FATAL, "buffer not big enough to write param.");
77  exit(EXIT_FAILURE);
78  }
79  void *ctx = Callback(ConfGetNode(tmp));
80  if (ctx == NULL) {
81  SCMutexUnlock(&mutex);
82  return;
83  }
84 
85  CudaHandlerConfProfile *new_cp = SCMalloc(sizeof(CudaHandlerConfProfile));
86  if (unlikely(new_cp == NULL))
87  exit(EXIT_FAILURE);
88  memset(new_cp, 0, sizeof(CudaHandlerConfProfile));
89  new_cp->name = SCStrdup(name);
90  if (new_cp->name == NULL)
91  exit(EXIT_FAILURE);
92  new_cp->ctx = ctx;
93  new_cp->Free = Free;
94 
95  if (conf_profiles == NULL) {
96  conf_profiles = new_cp;
97  } else {
98  new_cp->next = conf_profiles;
99  conf_profiles = new_cp;
100  }
101 
102  SCMutexUnlock(&mutex);
103  return;
104 }

143 /* we use a concept where every device on the gpu has only 1 context. If
144  * a section in the engine wants to use a device and tries to open a context

172 CUcontext CudaHandlerModuleGetContext(const char *name, int device_id)
173 {
174  void *ptmp;
175  SCMutexLock(&mutex);
176 
177  CudaHandlerModule *module = cudahl_modules;
178  while (module != NULL && strcasecmp(module->name, name) != 0)
179  module = module->next;
180  if (module != NULL) {
181  if (module->device_id != device_id) {
182  SCLogError(SC_ERR_CUDA_HANDLER_ERROR, "Module already "
183  "registered, but the new device_id is different "
184  "from the already registered device_id.");
185  exit(EXIT_FAILURE);
186  }
187  SCMutexUnlock(&mutex);
188  return module->context;
189  }
190 
191  CudaHandlerModule *new_module = SCMalloc(sizeof(CudaHandlerModule));
192  if (unlikely(new_module == NULL))
193  exit(EXIT_FAILURE);
194  memset(new_module, 0, sizeof(CudaHandlerModule));
195  new_module->device_id = device_id;
196  new_module->name = SCStrdup(name);
197  if (new_module->name == NULL)
198  exit(EXIT_FAILURE);
199  if (cudahl_modules == NULL) {
200  cudahl_modules = new_module;
201  } else {
202  new_module->next = cudahl_modules;
203  cudahl_modules = new_module;
204  }
205 
206  if (no_of_cuda_contexts <= device_id) {
207  ptmp = SCRealloc(cuda_contexts, sizeof(CUcontext) * (device_id + 1));
208  if (unlikely(ptmp == NULL)) {
209  SCFree(cuda_contexts);
210  cuda_contexts = NULL;
211  exit(EXIT_FAILURE);
212  }
213  cuda_contexts = ptmp;
214 
215  memset(cuda_contexts + no_of_cuda_contexts, 0,
216  sizeof(CUcontext) * ((device_id + 1) - no_of_cuda_contexts));
217  no_of_cuda_contexts = device_id + 1;
218  }
219 
220  if (cuda_contexts[device_id] == 0) {
221  SCCudaDevices *devices = SCCudaGetDeviceList();
222  if (SCCudaCtxCreate(&cuda_contexts[device_id], CU_CTX_SCHED_BLOCKING_SYNC,
223  devices->devices[device_id]->device) == -1) {
224  SCLogDebug("ctxcreate failure.");
225  exit(EXIT_FAILURE);
226  }
227  }
228  new_module->context = cuda_contexts[device_id];
229 
230  SCMutexUnlock(&mutex);
231  return cuda_contexts[device_id];
232 }

234 void CudaHandlerModuleStoreData(const char *module_name,
235  const char *data_name, void *data_ptr)
236 {
237  SCMutexLock(&mutex);
238 
239  CudaHandlerModule *module = cudahl_modules;
240  while (module != NULL && strcasecmp(module->name, module_name) != 0)
241  module = module->next;
242  if (module == NULL) {
243  SCLogError(SC_ERR_CUDA_HANDLER_ERROR, "Trying to retrieve data "
244  "\"%s\" from module \"%s\" that hasn't been registered "
245  "yet.", module_name, data_name);
246  exit(EXIT_FAILURE);
247  }
248 
249  CudaHandlerModuleData *data = module->module_data;
250  while (data != NULL && (strcasecmp(data_name, data->name) != 0)) {
251  data = data->next;
252  }
253  if (data != NULL) {
254  SCLogWarning(SC_ERR_CUDA_HANDLER_ERROR, "Data \"%s\" already "
255  "registered for this module \"%s\".", data_name,
256  module_name);
257  SCMutexUnlock(&mutex);
258  goto end;
259  }
260 
261  CudaHandlerModuleData *new_data = SCMalloc(sizeof(CudaHandlerModuleData));
262  if (unlikely(new_data == NULL))
263  exit(EXIT_FAILURE);
264  memset(new_data, 0, sizeof(CudaHandlerModuleData));
265  new_data->name = SCStrdup(data_name);
266  if (new_data->name == NULL)
267  exit(EXIT_FAILURE);
268  new_data->data = data_ptr;
269 
270  if (module->module_data == NULL) {
271  module->module_data = new_data;
272  } else {
273  new_data->next = module->module_data;
274  module->module_data = new_data;
275  }
276 
277  SCMutexUnlock(&mutex);
278 
279  end:
280  return;
281 }

313 int CudaHandlerGetCudaModule(CUmodule *p_module, const char *ptx_image)
314 {
315 #define CUDA_HANDLER_GET_CUDA_MODULE_BUFFER_EXTRA_SPACE 15
316 
317  int i = 0;
318 
319  /* select the ptx image based on the compute capability supported by all
320  * devices (i.e. the lowest) */
321  char *image = SCMalloc(strlen(ptx_image) + CUDA_HANDLER_GET_CUDA_MODULE_BUFFER_EXTRA_SPACE);
322  if (unlikely(image == NULL)) {
323  exit(EXIT_FAILURE);
324  }
325  memset(image, 0x00, strlen(ptx_image) + CUDA_HANDLER_GET_CUDA_MODULE_BUFFER_EXTRA_SPACE);
326 
327  int major = INT_MAX;
328  int minor = INT_MAX;
329  SCCudaDevices *devices = SCCudaGetDeviceList();
330  for (i = 0; i < devices->count; i++){
331  if (devices->devices[i]->major_rev < major){
332  major = devices->devices[i]->major_rev;
333  minor = devices->devices[i]->minor_rev;
334  }
335  if (devices->devices[i]->major_rev == major &&
336  devices->devices[i]->minor_rev < minor){
337  minor = devices->devices[i]->minor_rev;
338  }
339  }
340  snprintf(image,
341  strlen(ptx_image) + CUDA_HANDLER_GET_CUDA_MODULE_BUFFER_EXTRA_SPACE,
342  "%s_sm_%u%u",
343  ptx_image, major, minor);
344 
345  /* we don't have a cuda module associated with this module. Create a
346  * cuda module, update the module with this cuda module reference and
347  * then return the module refernce back to the calling function using
348  * the argument */
349  SCLogDebug("Loading kernel module: %s\n",image);
350  if (SCCudaModuleLoadData(p_module, (void *)SCCudaPtxDumpGetModule(image)) == -1)
351  goto error;
352  SCFree(image);
353 
354  return 0;
355  error:
356  SCFree(image);
357  return -1;
358 
359 #undef CUDA_HANDLER_GET_CUDA_MODULE_BUFFER_EXTRA_SPACE
360 }
