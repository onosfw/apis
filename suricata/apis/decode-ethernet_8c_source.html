41 int DecodeEthernet(ThreadVars *tv, DecodeThreadVars *dtv, Packet *p,
42  uint8_t *pkt, uint16_t len, PacketQueue *pq)
43 {
44  StatsIncr(tv, dtv->counter_eth);
45 
46  if (unlikely(len < ETHERNET_HEADER_LEN)) {
47  ENGINE_SET_INVALID_EVENT(p, ETHERNET_PKT_TOO_SMALL);
48  return TM_ECODE_FAILED;
49  }
50 
51  p->ethh = (EthernetHdr *)pkt;
52  if (unlikely(p->ethh == NULL))
53  return TM_ECODE_FAILED;
54 
55  SCLogDebug("p %p pkt %p ether type %04x", p, pkt, ntohs(p->ethh->eth_type));
56 
57  switch (ntohs(p->ethh->eth_type)) {
58  case ETHERNET_TYPE_IP:
59  //printf("DecodeEthernet ip4\n");
60  DecodeIPV4(tv, dtv, p, pkt + ETHERNET_HEADER_LEN,
61  len - ETHERNET_HEADER_LEN, pq);
62  break;
63  case ETHERNET_TYPE_IPV6:
64  //printf("DecodeEthernet ip6\n");
65  DecodeIPV6(tv, dtv, p, pkt + ETHERNET_HEADER_LEN,
66  len - ETHERNET_HEADER_LEN, pq);
67  break;
68  case ETHERNET_TYPE_PPPOE_SESS:
69  //printf("DecodeEthernet PPPOE Session\n");
70  DecodePPPOESession(tv, dtv, p, pkt + ETHERNET_HEADER_LEN,
71  len - ETHERNET_HEADER_LEN, pq);
72  break;
73  case ETHERNET_TYPE_PPPOE_DISC:
74  //printf("DecodeEthernet PPPOE Discovery\n");
75  DecodePPPOEDiscovery(tv, dtv, p, pkt + ETHERNET_HEADER_LEN,
76  len - ETHERNET_HEADER_LEN, pq);
77  break;
78  case ETHERNET_TYPE_VLAN:
79  case ETHERNET_TYPE_8021QINQ:
80  DecodeVLAN(tv, dtv, p, pkt + ETHERNET_HEADER_LEN,
81  len - ETHERNET_HEADER_LEN, pq);
82  break;
83  case ETHERNET_TYPE_MPLS_UNICAST:
84  case ETHERNET_TYPE_MPLS_MULTICAST:
85  DecodeMPLS(tv, dtv, p, pkt + ETHERNET_HEADER_LEN,
86  len - ETHERNET_HEADER_LEN, pq);
87  break;
88  default:
89  SCLogDebug("p %p pkt %p ether type %04x not supported", p,
90  pkt, ntohs(p->ethh->eth_type));
91  }
92 
93  return TM_ECODE_OK;
94 }
