96 /**
97  * \brief This function is used to match the window size on a packet
98  *
99  * \param t pointer to thread vars
100  * \param det_ctx pointer to the pattern matcher thread
101  * \param p pointer to the current packet
102  * \param m pointer to the sigmatch that we will cast into DetectWindowData
103  *
104  * \retval 0 no match
105  * \retval 1 match
106  */
107 int DetectWindowMatch(ThreadVars *t, DetectEngineThreadCtx *det_ctx, Packet *p, Signature *s, const SigMatchCtx *ctx)
108 {
109  const DetectWindowData *wd = (const DetectWindowData *)ctx;
110 
111  if ( !(PKT_IS_TCP(p)) || wd == NULL || PKT_IS_PSEUDOPKT(p)) {
112  return 0;
113  }
114 
115  if ( (!wd->negated && wd->size == TCP_GET_WINDOW(p)) || (wd->negated && wd->size != TCP_GET_WINDOW(p))) {
116  return 1;
117  }
118 
119  return 0;
120 }

122 /**
123  * \brief This function is used to parse window options passed via window: keyword
124  *
125  * \param windowstr Pointer to the user provided window options (negation! and size)
126  *
127  * \retval wd pointer to DetectWindowData on success
128  * \retval NULL on failure
129  */
130 DetectWindowData *DetectWindowParse(char *windowstr)
131 {
132  DetectWindowData *wd = NULL;
133 #define MAX_SUBSTRINGS 30
134  int ret = 0, res = 0;
135  int ov[MAX_SUBSTRINGS];
136 
137  ret = pcre_exec(parse_regex, parse_regex_study, windowstr, strlen(windowstr), 0, 0, ov, MAX_SUBSTRINGS);
138  if (ret < 1 || ret > 3) {
139  SCLogError(SC_ERR_PCRE_MATCH, "pcre_exec parse error, ret %" PRId32 ", string %s", ret, windowstr);
140  goto error;
141  }
142 
143  wd = SCMalloc(sizeof(DetectWindowData));
144  if (unlikely(wd == NULL))
145  goto error;
146 
147  if (ret > 1) {
148  char copy_str[128] = "";
149  res = pcre_copy_substring((char *)windowstr, ov, MAX_SUBSTRINGS, 1,
150  copy_str, sizeof(copy_str));
151  if (res < 0) {
152  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_copy_substring failed");
153  goto error;
154  }
155 
156  /* Detect if it's negated */
157  if (copy_str[0] == '!')
158  wd->negated = 1;
159  else
160  wd->negated = 0;
161 
162  if (ret > 2) {
163  res = pcre_copy_substring((char *)windowstr, ov, MAX_SUBSTRINGS, 2,
164  copy_str, sizeof(copy_str));
165  if (res < 0) {
166  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_copy_substring failed");
167  goto error;
168  }
169 
170  /* Get the window size if it's a valid value (in packets, we
171  * should alert if this doesn't happend from decode) */
172  if (-1 == ByteExtractStringUint16(&wd->size, 10, 0, copy_str)) {
173  goto error;
174  }
175  }
176  }
177 
178  return wd;
179 
180 error:
181  if (wd != NULL)
182  DetectWindowFree(wd);
183  return NULL;
184 
185 }

187 /**
188  * \brief this function is used to add the parsed window sizedata into the current signature
189  *
190  * \param de_ctx pointer to the Detection Engine Context
191  * \param s pointer to the Current Signature
192  * \param windowstr pointer to the user provided window options
193  *
194  * \retval 0 on Success
195  * \retval -1 on Failure
196  */
197 int DetectWindowSetup (DetectEngineCtx *de_ctx, Signature *s, char *windowstr)
198 {
199  DetectWindowData *wd = NULL;
200  SigMatch *sm = NULL;
201 
202  wd = DetectWindowParse(windowstr);
203  if (wd == NULL) goto error;
204 
205  /* Okay so far so good, lets get this into a SigMatch
206  * and put it in the Signature. */
207  sm = SigMatchAlloc();
208  if (sm == NULL)
209  goto error;
210 
211  sm->type = DETECT_WINDOW;
212  sm->ctx = (SigMatchCtx *)wd;
213 
214  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);
215  s->flags |= SIG_FLAG_REQUIRE_PACKET;
216 
217  return 0;
218 
219 error:
220  if (wd != NULL) DetectWindowFree(wd);
221  if (sm != NULL) SCFree(sm);
222  return -1;
223 
224 }
