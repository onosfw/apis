52 /**
53  * \brief Get the number of cpus configured in the system
54  * \retval 0 if the syscall is not available or we have an error;
55  * otherwise it will return the number of cpus configured
56  */
57 uint16_t UtilCpuGetNumProcessorsConfigured()
58 {
59 #ifdef SYSCONF_NPROCESSORS_CONF_COMPAT
60  long nprocs = -1;
61  nprocs = sysconf(_SC_NPROCESSORS_CONF);
62  if (nprocs < 1) {
63  SCLogError(SC_ERR_SYSCALL, "Couldn't retrieve the number of cpus "
64  "configured (%s)", strerror(errno));
65  return 0;
66  }
67 
68  if (nprocs > UINT16_MAX) {
69  SCLogDebug("It seems that there are more than %"PRIu16" CPUs "
70  "configured on this system. You can modify util-cpu.{c,h} "
71  "to use uint32_t to support it", UINT16_MAX);
72  return UINT16_MAX;
73  }
74 
75  return (uint16_t)nprocs;
76 #elif OS_WIN32
77  long nprocs = -1;
78  const char* envvar = getenv("NUMBER_OF_PROCESSORS");
79  nprocs = (NULL != envvar) ? atoi(envvar) : 0;
80  if (nprocs < 1) {
81  SCLogError(SC_ERR_SYSCALL, "Couldn't retrieve the number of cpus "
82  "configured from the NUMBER_OF_PROCESSORS environment variable");
83  return 0;
84  }
85  return (uint16_t)nprocs;
86 #else
87  SCLogError(SC_ERR_SYSCONF, "Couldn't retrieve the number of cpus "
88  "configured, sysconf macro unavailable");
89  return 0;
90 #endif
91 }

158 /**
159  * \brief Print a summary of CPUs detected (configured and online)
160  */
161 void UtilCpuPrintSummary()
162 {
163  uint16_t cpus_conf = UtilCpuGetNumProcessorsConfigured();
164  uint16_t cpus_online = UtilCpuGetNumProcessorsOnline();
165 
166  SCLogDebug("CPUs Summary: ");
167  if (cpus_conf > 0)
168  SCLogDebug("CPUs configured: %"PRIu16, cpus_conf);
169  if (cpus_online > 0)
170  SCLogInfo("CPUs/cores online: %"PRIu16, cpus_online);
171  if (cpus_online == 0 && cpus_conf == 0)
172  SCLogInfo("Couldn't retireve any information of CPU's, please, send your operating "
173  "system info and check util-cpu.{c,h}");

189 uint64_t UtilCpuGetTicks(void)
190 {
191  uint64_t val;
192 #if defined(__GNUC__) && (defined(__x86_64) || defined(_X86_64_) || defined(ia_64) || defined(__i386__))
193 #if defined(__x86_64) || defined(_X86_64_) || defined(ia_64)
194  __asm__ __volatile__ (
195  "xorl %%eax,%%eax\n\t"
196  "cpuid\n\t"
197  ::: "%rax", "%rbx", "%rcx", "%rdx");
198 #else
199  __asm__ __volatile__ (
200  "xorl %%eax,%%eax\n\t"
201  "pushl %%ebx\n\t"
202  "cpuid\n\t"
203  "popl %%ebx\n\t"
204  ::: "%eax", "%ecx", "%edx");
205 #endif
206  uint32_t a, d;
207  __asm__ __volatile__ ("rdtsc" : "=a" (a), "=d" (d));
208  val = ((uint64_t)a) | (((uint64_t)d) << 32);
209 #if defined(__x86_64) || defined(_X86_64_) || defined(ia_64)
210  __asm__ __volatile__ (
211  "xorl %%eax,%%eax\n\t"
212  "cpuid\n\t"
213  ::: "%rax", "%rbx", "%rcx", "%rdx");
214 #else
215  __asm__ __volatile__ (
216  "xorl %%eax,%%eax\n\t"
217  "pushl %%ebx\n\t"
218  "cpuid\n\t"
219  "popl %%ebx\n\t"
220  ::: "%eax", "%ecx", "%edx");
221 #endif
222 
223 #else /* #if defined(__GNU__) */
224 #warning Using inferior version of UtilCpuGetTicks
225  struct timeval now;
226  gettimeofday(&now, NULL);

228 #endif
