311 static int B2gAddPattern(MpmCtx *mpm_ctx, uint8_t *pat, uint16_t patlen, uint16_t offset, uint16_t depth, uint32_t pid, SigIntId sid, uint8_t flags)
312 {
313  B2gCtx *ctx = (B2gCtx *)mpm_ctx->ctx;
314 
315  SCLogDebug("ctx %p len %"PRIu16" pid %" PRIu32, ctx, patlen, pid);
316 
317  if (patlen == 0)
318  return 0;
319 
320  /* get a memory piece */
321  B2gPattern *p = B2gInitHashLookup(ctx, pat, patlen, flags, pid);
322  if (p == NULL) {
323  SCLogDebug("allocing new pattern");
324 
325  p = B2gAllocPattern(mpm_ctx);
326  if (p == NULL)
327  goto error;
328 
329  p->len = patlen;
330  p->flags = flags;
331  p->id = pid;
332 
333  p->original_pat = SCMalloc(patlen);
334  if (p->original_pat == NULL)
335  goto error;
336  mpm_ctx->memory_cnt++;
337  mpm_ctx->memory_size += patlen;
338  memcpy(p->original_pat, pat, patlen);
339 
340  /* setup the case insensitive part of the pattern */
341  p->ci = SCMalloc(patlen);
342  if (p->ci == NULL)
343  goto error;
344 
345  mpm_ctx->memory_cnt++;
346  mpm_ctx->memory_size += patlen;
347  memcpy_tolower(p->ci, pat, patlen);
348 
349  /* setup the case sensitive part of the pattern */
350  if (p->flags & MPM_PATTERN_FLAG_NOCASE) {
351  /* nocase means no difference between cs and ci */
352  p->cs = p->ci;
353  } else {
354  if (SCMemcmp(p->ci,pat,p->len) == 0) {
355  /* no diff between cs and ci: pat is lowercase */
356  p->cs = p->ci;
357  } else {
358  p->cs = SCMalloc(patlen);
359  if (p->cs == NULL)
360  goto error;
361 
362  mpm_ctx->memory_cnt++;
363  mpm_ctx->memory_size += patlen;
364  memcpy(p->cs, pat, patlen);
365  }
366  }
367 
368  p->sids_size = 1;
369  p->sids = SCMalloc(p->sids_size * sizeof(SigIntId));
370  BUG_ON(p->sids == NULL);
371  p->sids[0] = sid;
372  mpm_ctx->memory_cnt++;
373  mpm_ctx->memory_size += sizeof(SigIntId);
374 
375  //printf("B2gAddPattern: ci \""); prt(p->ci,p->len);
376  //printf("\" cs \""); prt(p->cs,p->len);
377  //printf("\"\n");
378 
379  /* put in the pattern hash */
380  B2gInitHashAdd(ctx, p);
381 
382  if (mpm_ctx->pattern_cnt == 65535) {
383  printf("Max search words reached\n");
384  exit(1);
385  }
386  mpm_ctx->pattern_cnt++;
387 
388  if (mpm_ctx->maxlen < patlen) mpm_ctx->maxlen = patlen;
389  if (mpm_ctx->minlen == 0) mpm_ctx->minlen = patlen;
390  else if (mpm_ctx->minlen > patlen) mpm_ctx->minlen = patlen;
391  } else {
392  /* Multiple sids for the same pid, so keep an array of sids. */
393 
394  /* TODO figure out how we can be called multiple times for the
395  * same CTX with the same sid */
396  int found = 0;
397  uint32_t x = 0;
398  for (x = 0; x < p->sids_size; x++) {
399  if (p->sids[x] == sid) {
400  found = 1;
401  break;
402  }
403  }
404  if (!found) {
405  SigIntId *sids = SCRealloc(p->sids, (sizeof(SigIntId) * (p->sids_size + 1)));
406  BUG_ON(sids == NULL);
407  p->sids = sids;
408  p->sids[p->sids_size] = sid;
409  p->sids_size++;
410  mpm_ctx->memory_size += sizeof(SigIntId);
411  }
412  }
413 
414  return 0;
415 
416 error:
417  B2gFreePattern(mpm_ctx, p);
418  return -1;
419 }

434 static inline uint32_t B2gBloomHash(void *data, uint16_t datalen, uint8_t iter, uint32_t hash_size)
435 {
436  uint8_t *d = (uint8_t *)data;
437  uint16_t i;
438  uint32_t hash = (uint32_t)u8_tolower(*d);
439 
440  for (i = 1; i < datalen; i++) {
441  d++;
442  hash += (u8_tolower(*d)) ^ i;
443  }
444  hash <<= (iter+1);
445 
446  hash %= hash_size;
447  return hash;
448 }

584  B2gPattern *thi = ctx->hash[idx];
585  while (thi->next) thi = thi->next;
586  thi->next = hi;
587  }
588  ctx->pat_x_cnt++;
589  }
590  }
591

593  ctx->bloom = (BloomFilter **)SCMalloc(sizeof(BloomFilter *) * ctx->hash_size);

596  memset(ctx->bloom, 0, sizeof(BloomFilter *) * ctx->hash_size);
597 
598  mpm_ctx->memory_cnt++;
599  mpm_ctx->memory_size += (sizeof(BloomFilter *) * ctx->hash_size);
600 
601  uint32_t h;
602  for (h = 0; h < ctx->hash_size; h++) {
603  B2gPattern *hi = ctx->hash[h];
604  if (hi == NULL)
605  continue;
606 
607  ctx->bloom[h] = BloomFilterInit(b2g_bloom_size, 2, B2gBloomHash);
608  if (ctx->bloom[h] == NULL)
609  continue;
610 
611  mpm_ctx->memory_cnt += BloomFilterMemoryCnt(ctx->bloom[h]);
612  mpm_ctx->memory_size += BloomFilterMemorySize(ctx->bloom[h]);
613 
614  if (ctx->pminlen[h] > 8)
615  ctx->pminlen[h] = 8;
616 
617  B2gPattern *thi = hi;
618  do {
619  SCLogDebug("adding \"%c%c\" to the bloom", thi->ci[0], thi->ci[1]);
620  BloomFilterAdd(ctx->bloom[h], thi->ci, ctx->pminlen[h]);
621  thi = thi->next;
622  } while (thi != NULL);
623  }
624 
625  return;
626 error:

1063  //if (memcmp(thi->cs, buf+j, thi->len) == 0) {
1064 #ifdef PRINTMATCH

1070  bitarray, thi->sids, thi->sids_size);

1095  SCLogDebug("pos %"PRIu32"", pos);

1102 uint32_t B2gSearch(MpmCtx *mpm_ctx, MpmThreadCtx *mpm_thread_ctx, PatternMatcherQueue *pmq, uint8_t *buf, uint16_t buflen)
1103 {
1104  B2gCtx *ctx = (B2gCtx *)mpm_ctx->ctx;
1105 #ifdef B2G_COUNTERS
1106  B2gThreadCtx *tctx = (B2gThreadCtx *)mpm_thread_ctx->ctx;
1107 #endif
1108  uint32_t pos = 0, matches = 0;
1109  B2G_TYPE d;
1110  uint32_t j;
1111 
1112  COUNT(tctx->stat_calls++);
1113  COUNT(tctx->stat_m_total+=ctx->m);
1114 
1115  if (buflen < ctx->m)
1116  return 0;
1117 
1118  uint8_t *bitarray = NULL;
1119  if (pmq) {
1120  bitarray = alloca(pmq->pattern_id_bitarray_size);
1121  memset(bitarray, 0, pmq->pattern_id_bitarray_size);
1122  }
1123 
1124  while (pos <= (buflen - ctx->m)) {
1125  j = ctx->m - 1;
1126  d = ~0;
1127 
1128  do {
1129  uint16_t h = B2G_HASH16(u8_tolower(buf[pos + j - 1]),u8_tolower(buf[pos + j]));
1130  d = ((d << 1) & ctx->B2G[h]);
1131  j = j - 1;
1132  } while (d != 0 && j != 0);
1133 
1134  /* (partial) match, move on to verification */
1135  if (d != 0) {
1136  COUNT(tctx->stat_d0++);
1137  //printf("output at pos %" PRIu32 ": ", pos); prt(buf + pos, ctx->m); printf("\n");
1138 
1139  /* get our patterns from the hash */
1140  uint16_t h = B2G_HASH16(u8_tolower(buf[pos + ctx->m - 2]),u8_tolower(buf[pos + ctx->m - 1]));
1141 
1142  if (ctx->bloom[h] != NULL) {
1143  COUNT(tctx->stat_pminlen_calls++);
