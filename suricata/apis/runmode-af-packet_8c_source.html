95 /**
96  * \brief extract information from config file
97  *
98  * The returned structure will be freed by the thread init function.
99  * This is thus necessary to or copy the structure before giving it
100  * to thread or to reparse the file for each thread (and thus have
101  * new structure.
102  *
103  * \return a AFPIfaceConfig corresponding to the interface name
104  */
105 void *ParseAFPConfig(const char *iface)
106 {
107  char *threadsstr = NULL;
108  ConfNode *if_root;
109  ConfNode *if_default = NULL;
110  ConfNode *af_packet_node;
111  AFPIfaceConfig *aconf = SCMalloc(sizeof(*aconf));
112  char *tmpclusterid;
113  char *tmpctype;
114  char *copymodestr;
115  intmax_t value;
116  int boolval;
117  char *bpf_filter = NULL;
118  char *out_iface = NULL;
119 
120  if (unlikely(aconf == NULL)) {
121  return NULL;
122  }
123 
124  if (iface == NULL) {
125  SCFree(aconf);
126  return NULL;
127  }
128 
129  strlcpy(aconf->iface, iface, sizeof(aconf->iface));
130  aconf->threads = 1;
131  SC_ATOMIC_INIT(aconf->ref);
132  (void) SC_ATOMIC_ADD(aconf->ref, 1);
133  aconf->buffer_size = 0;
134  aconf->cluster_id = 1;
135  aconf->cluster_type = PACKET_FANOUT_HASH;
136  aconf->promisc = 1;
137  aconf->checksum_mode = CHECKSUM_VALIDATION_KERNEL;
138  aconf->DerefFunc = AFPDerefConfig;
139  aconf->flags = 0;
140  aconf->bpf_filter = NULL;
141  aconf->out_iface = NULL;
142  aconf->copy_mode = AFP_COPY_MODE_NONE;
143 
144  if (ConfGet("bpf-filter", &bpf_filter) == 1) {
145  if (strlen(bpf_filter) > 0) {
146  aconf->bpf_filter = bpf_filter;
147  SCLogInfo("Going to use command-line provided bpf filter '%s'",
148  aconf->bpf_filter);
149  }
150  }
151 
152  /* Find initial node */
153  af_packet_node = ConfGetNode("af-packet");
154  if (af_packet_node == NULL) {
155  SCLogInfo("Unable to find af-packet config using default value");
156  return aconf;
157  }
158 
159  if_root = ConfNodeLookupKeyValue(af_packet_node, "interface", iface);
160 
161  if_default = ConfNodeLookupKeyValue(af_packet_node, "interface", "default");
162 
163  if (if_root == NULL && if_default == NULL) {
164  SCLogInfo("Unable to find af-packet config for "
165  "interface \"%s\" or \"default\", using default value",
166  iface);
167  return aconf;
168  }
169 
170  /* If there is no setting for current interface use default one as main iface */
171  if (if_root == NULL) {
172  if_root = if_default;
173  if_default = NULL;
174  }
175 
176  if (ConfGetChildValueWithDefault(if_root, if_default, "threads", &threadsstr) != 1) {
177  aconf->threads = 0;
178  } else {
179  if (threadsstr != NULL) {
180  if (strcmp(threadsstr, "auto") == 0) {
181  aconf->threads = 0;
182  } else {
183  aconf->threads = (uint8_t)atoi(threadsstr);
184  }
185  }
186  }
187  if (aconf->threads == 0) {
188  int rss_queues;
189  aconf->threads = (int)UtilCpuGetNumProcessorsOnline();
190  /* Get the number of RSS queues and take the min */
191  rss_queues = GetIfaceRSSQueuesNum(iface);
192  if (rss_queues > 0) {
193  if (rss_queues < aconf->threads) {
194  aconf->threads = rss_queues;
195  SCLogInfo("More core than RSS queues, using %d threads for interface %s",
196  aconf->threads, iface);
197  }
198  }
199  if (aconf->threads)
200  SCLogInfo("Using %d AF_PACKET threads for interface %s", aconf->threads, iface);
201  }
202  if (aconf->threads <= 0) {
203  aconf->threads = 1;
204  }
205 
206  if (ConfGetChildValueWithDefault(if_root, if_default, "copy-iface", &out_iface) == 1) {
207  if (strlen(out_iface) > 0) {
208  aconf->out_iface = out_iface;
209  }
210  }
211 
212  (void)ConfGetChildValueBoolWithDefault(if_root, if_default, "use-mmap", (int *)&boolval);
213  if (boolval) {
214  SCLogInfo("Enabling mmaped capture on iface %s",
215  aconf->iface);
216  aconf->flags |= AFP_RING_MODE;
217  }
218  (void)ConfGetChildValueBoolWithDefault(if_root, if_default, "use-emergency-flush", (int *)&boolval);
219  if (boolval) {
220  SCLogInfo("Enabling ring emergency flush on iface %s",
221  aconf->iface);
222  aconf->flags |= AFP_EMERGENCY_MODE;
223  }
224 
225 
226  aconf->copy_mode = AFP_COPY_MODE_NONE;
227  if (ConfGetChildValueWithDefault(if_root, if_default, "copy-mode", &copymodestr) == 1) {
228  if (aconf->out_iface == NULL) {
229  SCLogInfo("Copy mode activated but no destination"
230  " iface. Disabling feature");
231  } else if (!(aconf->flags & AFP_RING_MODE)) {
232  SCLogInfo("Copy mode activated but use-mmap "
233  "set to no. Disabling feature");
234  } else if (strlen(copymodestr) <= 0) {
235  aconf->out_iface = NULL;
236  } else if (strcmp(copymodestr, "ips") == 0) {
237  SCLogInfo("AF_PACKET IPS mode activated %s->%s",
238  iface,
239  aconf->out_iface);
240  aconf->copy_mode = AFP_COPY_MODE_IPS;
241  } else if (strcmp(copymodestr, "tap") == 0) {
242  SCLogInfo("AF_PACKET TAP mode activated %s->%s",
243  iface,
244  aconf->out_iface);
245  aconf->copy_mode = AFP_COPY_MODE_TAP;
246  } else {
247  SCLogInfo("Invalid mode (not in tap, ips)");
248  }
249  }
250 
251  SC_ATOMIC_RESET(aconf->ref);
252  (void) SC_ATOMIC_ADD(aconf->ref, aconf->threads);
253 
254  if (ConfGetChildValueWithDefault(if_root, if_default, "cluster-id", &tmpclusterid) != 1) {
255  SCLogError(SC_ERR_INVALID_ARGUMENT,"Could not get cluster-id from config");
256  } else {
257  aconf->cluster_id = (uint16_t)atoi(tmpclusterid);
258  SCLogDebug("Going to use cluster-id %" PRId32, aconf->cluster_id);
259  }
260 
261  if (ConfGetChildValueWithDefault(if_root, if_default, "cluster-type", &tmpctype) != 1) {
262  SCLogError(SC_ERR_GET_CLUSTER_TYPE_FAILED,"Could not get cluster-type from config");
263  } else if (strcmp(tmpctype, "cluster_round_robin") == 0) {
264  SCLogInfo("Using round-robin cluster mode for AF_PACKET (iface %s)",
265  aconf->iface);
266  aconf->cluster_type = PACKET_FANOUT_LB;
267  } else if (strcmp(tmpctype, "cluster_flow") == 0) {
268  /* In hash mode, we also ask for defragmentation needed to
269  * compute the hash */
270  uint16_t defrag = 0;
271  int conf_val = 0;
272  SCLogInfo("Using flow cluster mode for AF_PACKET (iface %s)",
273  aconf->iface);
274  ConfGetChildValueBoolWithDefault(if_root, if_default, "defrag", &conf_val);
275  if (conf_val) {
276  SCLogInfo("Using defrag kernel functionality for AF_PACKET (iface %s)",
277  aconf->iface);
278  defrag = PACKET_FANOUT_FLAG_DEFRAG;
279  }
280  aconf->cluster_type = PACKET_FANOUT_HASH | defrag;
281  } else if (strcmp(tmpctype, "cluster_cpu") == 0) {
282  SCLogInfo("Using cpu cluster mode for AF_PACKET (iface %s)",
283  aconf->iface);
284  aconf->cluster_type = PACKET_FANOUT_CPU;
285  } else if (strcmp(tmpctype, "cluster_qm") == 0) {
286  SCLogInfo("Using queue based cluster mode for AF_PACKET (iface %s)",
287  aconf->iface);
288  aconf->cluster_type = PACKET_FANOUT_QM;
289  } else if (strcmp(tmpctype, "cluster_random") == 0) {
290  SCLogInfo("Using random based cluster mode for AF_PACKET (iface %s)",
291  aconf->iface);
292  aconf->cluster_type = PACKET_FANOUT_RND;
293  } else if (strcmp(tmpctype, "cluster_rollover") == 0) {
294  SCLogInfo("Using rollover based cluster mode for AF_PACKET (iface %s)",
295  aconf->iface);
296  aconf->cluster_type = PACKET_FANOUT_ROLLOVER;
297 
298  } else {
299  SCLogError(SC_ERR_INVALID_CLUSTER_TYPE,"invalid cluster-type %s",tmpctype);
300  SCFree(aconf);
301  return NULL;
302  }
303 
304  int conf_val = 0;
305  ConfGetChildValueBoolWithDefault(if_root, if_default, "rollover", &conf_val);
306  if (conf_val) {
307  SCLogInfo("Using rollover kernel functionality for AF_PACKET (iface %s)",
308  aconf->iface);
309  aconf->cluster_type |= PACKET_FANOUT_FLAG_ROLLOVER;
310  }
311 
312  /*load af_packet bpf filter*/
313  /* command line value has precedence */
314  if (ConfGet("bpf-filter", &bpf_filter) != 1) {
315  if (ConfGetChildValueWithDefault(if_root, if_default, "bpf-filter", &bpf_filter) == 1) {
316  if (strlen(bpf_filter) > 0) {
317  aconf->bpf_filter = bpf_filter;
318  SCLogInfo("Going to use bpf filter %s", aconf->bpf_filter);
319  }
320  }
321  }
322 
323  if ((ConfGetChildValueIntWithDefault(if_root, if_default, "buffer-size", &value)) == 1) {
324  aconf->buffer_size = value;
325  } else {
326  aconf->buffer_size = 0;
327  }
328  if ((ConfGetChildValueIntWithDefault(if_root, if_default, "ring-size", &value)) == 1) {
329  aconf->ring_size = value;
330  if (value * aconf->threads < max_pending_packets) {
331  aconf->ring_size = max_pending_packets / aconf->threads + 1;
332  SCLogWarning(SC_ERR_AFP_CREATE, "Inefficient setup: ring-size < max_pending_packets. "
333  "Resetting to decent value %d.", aconf->ring_size);
334  /* We want at least that max_pending_packets packets can be handled by the
335  * interface. This is generous if we have multiple interfaces listening. */
336  }
337  } else {
338  /* We want that max_pending_packets packets can be handled by suricata
339  * for this interface. To take burst into account we multiply the obtained
340  * size by 2. */
341  aconf->ring_size = max_pending_packets * 2 / aconf->threads;
342  }
343 
344  (void)ConfGetChildValueBoolWithDefault(if_root, if_default, "disable-promisc", (int *)&boolval);
345  if (boolval) {
346  SCLogInfo("Disabling promiscuous mode on iface %s",
347  aconf->iface);
348  aconf->promisc = 0;
349  }
350 
351  if (ConfGetChildValueWithDefault(if_root, if_default, "checksum-checks", &tmpctype) == 1) {
352  if (strcmp(tmpctype, "auto") == 0) {
353  aconf->checksum_mode = CHECKSUM_VALIDATION_AUTO;
354  } else if (strcmp(tmpctype, "yes") == 0) {
355  aconf->checksum_mode = CHECKSUM_VALIDATION_ENABLE;
356  } else if (strcmp(tmpctype, "no") == 0) {
357  aconf->checksum_mode = CHECKSUM_VALIDATION_DISABLE;
358  } else if (strcmp(tmpctype, "kernel") == 0) {
359  aconf->checksum_mode = CHECKSUM_VALIDATION_KERNEL;
360  } else {
361  SCLogError(SC_ERR_INVALID_ARGUMENT, "Invalid value for checksum-checks for %s", aconf->iface);
362  }
363  }
364 
365  if (GetIfaceOffloading(iface) == 1) {
366  SCLogWarning(SC_ERR_AFP_CREATE,
367  "Using AF_PACKET with GRO or LRO activated can lead to capture problems");
368  }
369 
370  return aconf;
371 }

379 int AFPRunModeIsIPS()
380 {
381  int nlive = LiveGetDeviceCount();
382  int ldev;
383  ConfNode *if_root;
384  ConfNode *if_default = NULL;
385  ConfNode *af_packet_node;
386  int has_ips = 0;
387  int has_ids = 0;
388 
389  /* Find initial node */
390  af_packet_node = ConfGetNode("af-packet");
391  if (af_packet_node == NULL) {
392  return 0;
393  }
394 
395  if_default = ConfNodeLookupKeyValue(af_packet_node, "interface", "default");
396 
397  for (ldev = 0; ldev < nlive; ldev++) {
398  char *live_dev = LiveGetDeviceName(ldev);
399  if (live_dev == NULL) {
400  SCLogError(SC_ERR_INVALID_VALUE, "Problem with config file");
401  return 0;
402  }
403  char *copymodestr = NULL;
404  if_root = ConfNodeLookupKeyValue(af_packet_node, "interface", live_dev);
405 
406  if (if_root == NULL) {
407  if (if_default == NULL) {
408  SCLogError(SC_ERR_INVALID_VALUE, "Problem with config file");
409  return 0;
410  }
411  if_root = if_default;
412  }
413 
414  if (ConfGetChildValueWithDefault(if_root, if_default, "copy-mode", &copymodestr) == 1) {
415  if (strcmp(copymodestr, "ips") == 0) {
416  has_ips = 1;
417  } else {
418  has_ids = 1;
419  }
420  } else {
421  has_ids = 1;
422  }
423  }
424 
425  if (has_ids && has_ips) {
426  SCLogInfo("AF_PACKET mode using IPS and IDS mode");
427  for (ldev = 0; ldev < nlive; ldev++) {
428  char *live_dev = LiveGetDeviceName(ldev);
429  if (live_dev == NULL) {
430  SCLogError(SC_ERR_INVALID_VALUE, "Problem with config file");
431  return 0;
432  }
433  if_root = ConfNodeLookupKeyValue(af_packet_node, "interface", live_dev);
434  char *copymodestr = NULL;
435 
436  if (if_root == NULL) {
437  if (if_default == NULL) {
438  SCLogError(SC_ERR_INVALID_VALUE, "Problem with config file");
439  return 0;
440  }
441  if_root = if_default;
442  }
443 
444  if (! ((ConfGetChildValueWithDefault(if_root, if_default, "copy-mode", &copymodestr) == 1) &&
445  (strcmp(copymodestr, "ips") == 0))) {
446  SCLogError(SC_ERR_INVALID_ARGUMENT,
447  "AF_PACKET IPS mode used and interface '%s' is in IDS or TAP mode. "
448  "Sniffing '%s' but expect bad result as stream-inline is activated.",
449  live_dev, live_dev);
450  }
451  }
452  }
453 
454  return has_ips;
455 }
