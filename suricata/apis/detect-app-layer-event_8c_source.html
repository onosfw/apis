131 static int DetectAppLayerEventParseAppP2(DetectAppLayerEventData *data,
132  uint8_t *ipproto_bitarray,
133  AppLayerEventType *event_type)
134 {
135  int event_id = 0;
136  const char *p_idx;
137  uint8_t ipproto;
138  char alproto_name[50];
139  int r = 0;
140 
141  p_idx = strchr(data->arg, '.');
142  strlcpy(alproto_name, data->arg, p_idx - data->arg + 1);
143 
144  if (ipproto_bitarray[IPPROTO_TCP / 8] & 1 << (IPPROTO_TCP % 8)) {
145  ipproto = IPPROTO_TCP;
146  } else if (ipproto_bitarray[IPPROTO_UDP / 8] & 1 << (IPPROTO_UDP % 8)) {
147  ipproto = IPPROTO_UDP;
148  } else {
149  SCLogError(SC_ERR_INVALID_SIGNATURE, "protocol %s is disabled", alproto_name);
150  return -1;
151  }
152 
153  r = AppLayerParserGetEventInfo(ipproto, data->alproto,
154  p_idx + 1, &event_id, event_type);
155  if (r < 0) {
156  SCLogError(SC_ERR_INVALID_SIGNATURE, "app-layer-event keyword's "
157  "protocol \"%s\" doesn't have event \"%s\" registered",
158  alproto_name, p_idx + 1);
159  return -1;
160  }
161  data->event_id = event_id;
162 
163  return 0;
164 }

166 static DetectAppLayerEventData *DetectAppLayerEventParseAppP1(const char *arg)
167 {
168  /* period index */
169  DetectAppLayerEventData *aled;
170  AppProto alproto;
171  const char *p_idx;
172  char alproto_name[50];
173 
174  p_idx = strchr(arg, '.');
175  /* + 1 for trailing \0 */
176  strlcpy(alproto_name, arg, p_idx - arg + 1);
177 
178  alproto = AppLayerGetProtoByName(alproto_name);
179  if (alproto == ALPROTO_UNKNOWN) {
180  SCLogError(SC_ERR_INVALID_SIGNATURE, "app-layer-event keyword "
181  "supplied with unknown protocol \"%s\"",
182  alproto_name);
183  return NULL;
184  }
185 
186  aled = SCMalloc(sizeof(*aled));
187  if (unlikely(aled == NULL))
188  return NULL;
189  memset(aled, 0x00, sizeof(*aled));
190  aled->alproto = alproto;
191  aled->arg = SCStrdup(arg);
192  if (aled->arg == NULL) {
193  SCFree(aled);
194  return NULL;
195  }
196 
197  return aled;
198 }

200 static DetectAppLayerEventData *DetectAppLayerEventParse(const char *arg,
201  AppLayerEventType *event_type)
202 {
203  *event_type = 0;
204 
205  if (arg == NULL) {
206  SCLogError(SC_ERR_INVALID_SIGNATURE, "app-layer-event keyword supplied "
207  "with no arguments. This keyword needs an argument.");
208  return NULL;
209  }
210 
211  while (*arg != '\0' && isspace((unsigned char)*arg))
212  arg++;
213 
214  if (strchr(arg, '.') == NULL) {
215  return DetectAppLayerEventParsePkt(arg, event_type);
216  } else {
217  return DetectAppLayerEventParseAppP1(arg);
218  }
219 }

221 static int DetectAppLayerEventSetupP2(Signature *s,
222  SigMatch *sm)
223 {
224  AppLayerEventType event_type = 0;
225 
226  if (DetectAppLayerEventParseAppP2((DetectAppLayerEventData *)sm->ctx, s->proto.proto,
227  &event_type) < 0) {
228  /* DetectAppLayerEventParseAppP2 prints errors */
229  return -1;
230  }
231  if (event_type == APP_LAYER_EVENT_TYPE_GENERAL)
232  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_AMATCH);
233  else
234  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_APP_EVENT);
235  /* We should have set this flag already in SetupP1 */
236  s->flags |= SIG_FLAG_APPLAYER;
237 
238  return 0;
239 }

241 static int DetectAppLayerEventSetupP1(DetectEngineCtx *de_ctx, Signature *s, char *arg)
242 {
243  DetectAppLayerEventData *data = NULL;
244  SigMatch *sm = NULL;
245  AppLayerEventType event_type;
246 
247  data = DetectAppLayerEventParse(arg, &event_type);
248  if (data == NULL)
249  goto error;
250 
251  sm = SigMatchAlloc();
252  if (sm == NULL)
253  goto error;
254 
255  sm->type = DETECT_AL_APP_LAYER_EVENT;
256  sm->ctx = (SigMatchCtx *)data;
257 
258  if (s->alproto != ALPROTO_UNKNOWN) {
259  if (s->alproto != data->alproto) {
260  SCLogError(SC_ERR_CONFLICTING_RULE_KEYWORDS, "rule contains "
261  "conflicting keywords needing different alprotos");
262  goto error;
263  }
264  } else {
265  s->alproto = data->alproto;
266  }
267 
268  if (event_type == APP_LAYER_EVENT_TYPE_PACKET) {
269  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);
270  } else {
271  /* We push it to this list temporarily. We deal with
272  * these in DetectAppLayerEventPrepare(). */
273  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_APP_EVENT);
274  s->flags |= SIG_FLAG_APPLAYER;
275  }
276 
277  return 0;
278 
279 error:
280  if (data)
281  SCFree(data);
282  if (sm) {
283  sm->ctx = NULL;
284  SigMatchFree(sm);
285  }
286  return -1;
287 }

300 int DetectAppLayerEventPrepare(Signature *s)
301 {
302  SigMatch *sm = s->sm_lists[DETECT_SM_LIST_APP_EVENT];
303  s->sm_lists[DETECT_SM_LIST_APP_EVENT] = NULL;
304  s->sm_lists_tail[DETECT_SM_LIST_APP_EVENT] = NULL;
305 
306  while (sm != NULL) {
307  sm->next = sm->prev = NULL;

311  }
312 
313  return 0;
