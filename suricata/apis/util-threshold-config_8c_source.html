97 /**

106 static char *SCThresholdConfGetConfFilename(const DetectEngineCtx *de_ctx)
107 {
108  char *log_filename = NULL;
109  char config_value[256] = "";
110 
111  if (de_ctx != NULL && strlen(de_ctx->config_prefix) > 0) {
112  snprintf(config_value, sizeof(config_value),
113  "%s.threshold-file", de_ctx->config_prefix);
114 
115  /* try loading prefix setting, fall back to global if that
116  * fails. */

118  if (ConfGet("threshold-file", &log_filename) != 1) {
119  log_filename = (char *)THRESHOLD_CONF_DEF_CONF_FILEPATH;
120  }
121  }
122  } else {
123  if (ConfGet("threshold-file", &log_filename) != 1) {
124  log_filename = (char *)THRESHOLD_CONF_DEF_CONF_FILEPATH;
125  }
126  }

130 /**
131  * \brief Inits the context to be used by the Threshold Config parsing API.
132  *
133  * This function initializes the hash table to be used by the Detection
134  * Engine Context to hold the data from the threshold.config file,
135  * obtains the file desc to parse the threshold.config file, and
136  * inits the regex used to parse the lines from threshold.config
137  * file.
138  *
139  * \param de_ctx Pointer to the Detection Engine Context.
140  * \param utfd Pointer for unit test file descriptor.
141  *
142  * \retval 0 On success.
143  * \retval -1 On failure.
144  */
145 int SCThresholdConfInitContext(DetectEngineCtx *de_ctx, FILE *utfd)
146 {
147  char *filename = NULL;
148  const char *eb = NULL;
149  FILE *fd = utfd;
150  int eo;
151  int opts = 0;
152 
153  if (fd == NULL) {
154  filename = SCThresholdConfGetConfFilename(de_ctx);
155  if ( (fd = fopen(filename, "r")) == NULL) {
156  SCLogWarning(SC_ERR_FOPEN, "Error opening file: \"%s\": %s", filename, strerror(errno));
157  goto error;
158  }
159  }
160 
161  regex_base = pcre_compile(DETECT_BASE_REGEX, opts, &eb, &eo, NULL);
162  if (regex_base == NULL) {
163  SCLogError(SC_ERR_PCRE_COMPILE, "Compile of \"%s\" failed at offset %" PRId32 ": %s",DETECT_BASE_REGEX, eo, eb);
164  goto error;
165  }
166 
167  regex_base_study = pcre_study(regex_base, 0, &eb);
168  if (eb != NULL) {
169  SCLogError(SC_ERR_PCRE_STUDY, "pcre study failed: %s", eb);
170  goto error;
171  }
172 
173  regex_threshold = pcre_compile(DETECT_THRESHOLD_REGEX, opts, &eb, &eo, NULL);
174  if (regex_threshold == NULL) {
175  SCLogError(SC_ERR_PCRE_COMPILE, "Compile of \"%s\" failed at offset %" PRId32 ": %s",DETECT_THRESHOLD_REGEX, eo, eb);
176  goto error;
177  }
178 
179  regex_threshold_study = pcre_study(regex_threshold, 0, &eb);
180  if (eb != NULL) {
181  SCLogError(SC_ERR_PCRE_STUDY, "pcre study failed: %s", eb);
182  goto error;
183  }
184 
185  regex_rate = pcre_compile(DETECT_RATE_REGEX, opts, &eb, &eo, NULL);
186  if (regex_rate == NULL) {
187  SCLogError(SC_ERR_PCRE_COMPILE, "Compile of \"%s\" failed at offset %" PRId32 ": %s",DETECT_RATE_REGEX, eo, eb);
188  goto error;
189  }
190 
191  regex_rate_study = pcre_study(regex_rate, 0, &eb);
192  if (eb != NULL) {
193  SCLogError(SC_ERR_PCRE_STUDY, "pcre study failed: %s", eb);
194  goto error;
195  }
196 
197  regex_suppress = pcre_compile(DETECT_SUPPRESS_REGEX, opts, &eb, &eo, NULL);
198  if (regex_suppress == NULL) {
199  SCLogError(SC_ERR_PCRE_COMPILE, "Compile of \"%s\" failed at offset %" PRId32 ": %s",DETECT_SUPPRESS_REGEX, eo, eb);
200  goto error;
201  }
202 
203  regex_suppress_study = pcre_study(regex_suppress, 0, &eb);
204  if (eb != NULL) {
205  SCLogError(SC_ERR_PCRE_STUDY, "pcre study failed: %s", eb);
206  goto error;
207  }
208 
209  SCThresholdConfParseFile(de_ctx, fd);
210  SCThresholdConfDeInitContext(de_ctx, fd);
211 
212  SCLogDebug("Global thresholding options defined");
213  return 0;
214 
215 error:
216  SCThresholdConfDeInitContext(de_ctx, fd);
217  return -1;
218 }

275 static int SetupSuppressRule(DetectEngineCtx *de_ctx, uint32_t id, uint32_t gid,
276  uint8_t parsed_type, uint8_t parsed_track, uint32_t parsed_count,
277  uint32_t parsed_seconds, uint32_t parsed_timeout, uint8_t parsed_new_action,
278  const char *th_ip)
279 {
280  Signature *s = NULL;
281  SigMatch *sm = NULL;
282  DetectThresholdData *de = NULL;
283 
284  BUG_ON(parsed_type != TYPE_SUPPRESS);
285 
286  /* Install it */
287  if (id == 0 && gid == 0) {
288  if (parsed_track == TRACK_RULE) {
289  SCLogWarning(SC_ERR_EVENT_ENGINE, "suppressing all rules");
290  }
291 
292  /* update each sig with our suppress info */
293  for (s = de_ctx->sig_list; s != NULL; s = s->next) {
294  /* tag the rule as noalert */
295  if (parsed_track == TRACK_RULE) {
296  s->flags |= SIG_FLAG_NOALERT;
297  continue;
298  }
299 
300  de = SCMalloc(sizeof(DetectThresholdData));
301  if (unlikely(de == NULL))
302  goto error;
303  memset(de,0,sizeof(DetectThresholdData));
304 
305  de->type = TYPE_SUPPRESS;
306  de->track = parsed_track;
307  de->count = parsed_count;
308  de->seconds = parsed_seconds;
309  de->new_action = parsed_new_action;
310  de->timeout = parsed_timeout;
311 
312  if (parsed_track != TRACK_RULE) {
313  if (DetectAddressParse((const DetectEngineCtx *)de_ctx, &de->addrs, (char *)th_ip) != 0) {
314  SCLogError(SC_ERR_INVALID_IP_NETBLOCK, "failed to parse %s", th_ip);
315  goto error;
316  }
317  }
318 
319  sm = SigMatchAlloc();
320  if (sm == NULL) {
321  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating SigMatch");
322  goto error;
323  }
324 
325  sm->type = DETECT_THRESHOLD;
326  sm->ctx = (void *)de;
327  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_SUPPRESS);
328  }
329  } else if (id == 0 && gid > 0) {
330  if (parsed_track == TRACK_RULE) {
331  SCLogWarning(SC_ERR_EVENT_ENGINE, "suppressing all rules with gid %"PRIu32, gid);
332  }
333  /* set up suppression for each signature with a matching gid */
334  for (s = de_ctx->sig_list; s != NULL; s = s->next) {
335  if (s->gid != gid)
336  continue;
337 
338  /* tag the rule as noalert */
339  if (parsed_track == TRACK_RULE) {
340  s->flags |= SIG_FLAG_NOALERT;
341  continue;
342  }
343 
344  de = SCMalloc(sizeof(DetectThresholdData));
345  if (unlikely(de == NULL))
346  goto error;
347 
348  memset(de,0,sizeof(DetectThresholdData));
349 
350  de->type = TYPE_SUPPRESS;
351  de->track = parsed_track;
352  de->count = parsed_count;
353  de->seconds = parsed_seconds;
354  de->new_action = parsed_new_action;
355  de->timeout = parsed_timeout;
356 
357  if (parsed_track != TRACK_RULE) {
358  if (DetectAddressParse((const DetectEngineCtx *)de_ctx, &de->addrs, (char *)th_ip) != 0) {
359  SCLogError(SC_ERR_INVALID_IP_NETBLOCK, "failed to parse %s", th_ip);
360  goto error;
361  }
362  }
363 
364  sm = SigMatchAlloc();
365  if (sm == NULL) {
366  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating SigMatch");
367  goto error;
368  }
369 
370  sm->type = DETECT_THRESHOLD;
371  sm->ctx = (void *)de;
372 
373  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_SUPPRESS);
374  }
375  } else if (id > 0 && gid == 0) {
376  SCLogError(SC_ERR_INVALID_VALUE, "Can't use a event config that has "
377  "sid > 0 and gid == 0. Please fix this "
378  "in your threshold.conf file");
379  goto error;
380  } else {
381  s = SigFindSignatureBySidGid(de_ctx, id, gid);
382  if (s == NULL) {
383  SCLogWarning(SC_ERR_EVENT_ENGINE, "can't suppress sid "
384  "%"PRIu32", gid %"PRIu32": unknown rule", id, gid);
385  } else {
386  if (parsed_track == TRACK_RULE) {
387  s->flags |= SIG_FLAG_NOALERT;
388  goto end;
389  }
390 
391  de = SCMalloc(sizeof(DetectThresholdData));
392  if (unlikely(de == NULL))
393  goto error;
394  memset(de,0,sizeof(DetectThresholdData));
395 
396  de->type = TYPE_SUPPRESS;
397  de->track = parsed_track;
398  de->count = parsed_count;
399  de->seconds = parsed_seconds;
400  de->new_action = parsed_new_action;
401  de->timeout = parsed_timeout;
402 
403  if (DetectAddressParse((const DetectEngineCtx *)de_ctx, &de->addrs, (char *)th_ip) != 0) {
404  SCLogError(SC_ERR_INVALID_IP_NETBLOCK, "failed to parse %s", th_ip);
405  goto error;
406  }
407 
408  sm = SigMatchAlloc();
409  if (sm == NULL) {
410  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating SigMatch");
411  goto error;
412  }
413 
414  sm->type = DETECT_THRESHOLD;
415  sm->ctx = (void *)de;
416 
417  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_SUPPRESS);
418  }
419  }
420 
421 end:
422  return 0;
423 error:
424  if (de != NULL) {
425  DetectAddressHeadCleanup(&de->addrs);
426  SCFree(de);
427  }
428  return -1;
429 }

1133 {

1139  int esc_pos = 0;

1142  return;

1157  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");

1159  return;

1162  }

1165  break;

1168  (void)fgetc(fd);
1169

1184  break;
1185  }
1186  }
