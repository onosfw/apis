61 /**
62  * \brief Run the actual payload match functions
63  *
64  * The following keywords are inspected:
65  * - content, including all the http and dce modified contents
66  * - isdaatat
67  * - pcre
68  * - bytejump
69  * - bytetest
70  * - byte_extract
71  * - urilen
72  * -
73  *
74  * All keywords are evaluated against the buffer with buffer_len.
75  *
76  * For accounting the last match in relative matching the
77  * det_ctx->buffer_offset int is used.
78  *
79  * \param de_ctx Detection engine context
80  * \param det_ctx Detection engine thread context
81  * \param s Signature to inspect
82  * \param sm SigMatch to inspect
83  * \param f Flow (for pcre flowvar storage)
84  * \param buffer Ptr to the buffer to inspect
85  * \param buffer_len Length of the payload
86  * \param stream_start_offset Indicates the start of the current buffer in
87  * the whole buffer stream inspected. This
88  * applies if the current buffer is inspected
89  * in chunks.
90  * \param inspection_mode Refers to the engine inspection mode we are currently
91  * inspecting. Can be payload, stream, one of the http
92  * buffer inspection modes or dce inspection mode.
93  * \param data Used to send some custom data. For example in
94  * payload inspection mode, data contains packet ptr,
95  * and under dce inspection mode, contains dce state.
96  *
97  * \retval 0 no match
98  * \retval 1 match
99  */
100 int DetectEngineContentInspection(DetectEngineCtx *de_ctx, DetectEngineThreadCtx *det_ctx,
101  Signature *s, SigMatch *sm,
102  Flow *f,
103  uint8_t *buffer, uint32_t buffer_len,
104  uint32_t stream_start_offset,
105  uint8_t inspection_mode, void *data)
106 {
107  SCEnter();
108  KEYWORD_PROFILING_START;
109 
110  det_ctx->inspection_recursion_counter++;
111 
112  if (det_ctx->inspection_recursion_counter == de_ctx->inspection_recursion_limit) {
113  det_ctx->discontinue_matching = 1;
114  KEYWORD_PROFILING_END(det_ctx, sm->type, 0);
115  SCReturnInt(0);
116  }
117 
118  if (sm == NULL || buffer_len == 0) {
119  KEYWORD_PROFILING_END(det_ctx, sm->type, 0);
120  SCReturnInt(0);
121  }
122 
123  /* \todo unify this which is phase 2 of payload inspection unification */
124  if (sm->type == DETECT_CONTENT) {
125 
126  DetectContentData *cd = (DetectContentData *)sm->ctx;
127  SCLogDebug("inspecting content %"PRIu32" buffer_len %"PRIu32, cd->id, buffer_len);
128 
129  /* we might have already have this content matched by the mpm.
130  * (if there is any other reason why we'd want to avoid checking
131  * it here, please fill it in) */
132  //if (cd->flags & DETECT_CONTENT_NO_DOUBLE_INSPECTION_REQUIRED) {
133  // goto match;
134  //}
135 
136  /* rule parsers should take care of this */
137 #ifdef DEBUG
138  BUG_ON(cd->depth != 0 && cd->depth <= cd->offset);
139 #endif
140 
141  /* search for our pattern, checking the matches recursively.
142  * if we match we look for the next SigMatch as well */
143  uint8_t *found = NULL;
144  uint32_t offset = 0;
145  uint32_t depth = buffer_len;
146  uint32_t prev_offset = 0; /**< used in recursive searching */
147  uint32_t prev_buffer_offset = det_ctx->buffer_offset;
148 
149  do {
150  if ((cd->flags & DETECT_CONTENT_DISTANCE) ||
151  (cd->flags & DETECT_CONTENT_WITHIN)) {
152  SCLogDebug("det_ctx->buffer_offset %"PRIu32, det_ctx->buffer_offset);
153 
154  offset = prev_buffer_offset;
155  depth = buffer_len;
156 
157  int distance = cd->distance;
158  if (cd->flags & DETECT_CONTENT_DISTANCE) {
159  if (cd->flags & DETECT_CONTENT_DISTANCE_BE) {
160  distance = det_ctx->bj_values[cd->distance];
161  }
162  if (distance < 0 && (uint32_t)(abs(distance)) > offset)
163  offset = 0;
164  else
165  offset += distance;
166 
167  SCLogDebug("cd->distance %"PRIi32", offset %"PRIu32", depth %"PRIu32,
168  distance, offset, depth);
169  }
170 
171  if (cd->flags & DETECT_CONTENT_WITHIN) {
172  if (cd->flags & DETECT_CONTENT_WITHIN_BE) {
173  if ((int32_t)depth > (int32_t)(prev_buffer_offset + det_ctx->bj_values[cd->within] + distance)) {
174  depth = prev_buffer_offset + det_ctx->bj_values[cd->within] + distance;
175  }
176  } else {
177  if ((int32_t)depth > (int32_t)(prev_buffer_offset + cd->within + distance)) {
178  depth = prev_buffer_offset + cd->within + distance;
179  }
180 
181  SCLogDebug("cd->within %"PRIi32", det_ctx->buffer_offset %"PRIu32", depth %"PRIu32,
182  cd->within, prev_buffer_offset, depth);
183  }
184 
185  if (stream_start_offset != 0 && prev_buffer_offset == 0) {
186  if (depth <= stream_start_offset) {
187  goto no_match;
188  } else if (depth >= (stream_start_offset + buffer_len)) {
189  ;
190  } else {
191  depth = depth - stream_start_offset;
192  }
193  }
194  }
195 
196  if (cd->flags & DETECT_CONTENT_DEPTH_BE) {
197  if ((det_ctx->bj_values[cd->depth] + prev_buffer_offset) < depth) {
198  depth = prev_buffer_offset + det_ctx->bj_values[cd->depth];
199  }
200  } else {
201  if (cd->depth != 0) {
202  if ((cd->depth + prev_buffer_offset) < depth) {
203  depth = prev_buffer_offset + cd->depth;
204  }
205 
206  SCLogDebug("cd->depth %"PRIu32", depth %"PRIu32, cd->depth, depth);
207  }
208  }
209 
210  if (cd->flags & DETECT_CONTENT_OFFSET_BE) {
211  if (det_ctx->bj_values[cd->offset] > offset)
212  offset = det_ctx->bj_values[cd->offset];
213  } else {
214  if (cd->offset > offset) {
215  offset = cd->offset;
216  SCLogDebug("setting offset %"PRIu32, offset);
217  }
218  }
219  } else { /* implied no relative matches */
220  /* set depth */
221  if (cd->flags & DETECT_CONTENT_DEPTH_BE) {
222  depth = det_ctx->bj_values[cd->depth];
223  } else {
224  if (cd->depth != 0) {
225  depth = cd->depth;
226  }
227  }
228 
229  if (stream_start_offset != 0 && cd->flags & DETECT_CONTENT_DEPTH) {
230  if (depth <= stream_start_offset) {
231  goto no_match;
232  } else if (depth >= (stream_start_offset + buffer_len)) {
233  ;
234  } else {
235  depth = depth - stream_start_offset;
236  }
237  }
238 
239  /* set offset */
240  if (cd->flags & DETECT_CONTENT_OFFSET_BE)
241  offset = det_ctx->bj_values[cd->offset];
242  else
243  offset = cd->offset;
244  prev_buffer_offset = 0;
245  }
246 
247  /* update offset with prev_offset if we're searching for
248  * matches after the first occurence. */
249  SCLogDebug("offset %"PRIu32", prev_offset %"PRIu32, offset, prev_offset);
250  if (prev_offset != 0)
251  offset = prev_offset;
252 
253  SCLogDebug("offset %"PRIu32", depth %"PRIu32, offset, depth);
254 
255  if (depth > buffer_len)
256  depth = buffer_len;
257 
258  /* if offset is bigger than depth we can never match on a pattern.
259  * We can however, "match" on a negated pattern. */
260  if (offset > depth || depth == 0) {
261  if (cd->flags & DETECT_CONTENT_NEGATED) {
262  goto match;
263  } else {
264  goto no_match;
265  }
266  }
267 
268  uint8_t *sbuffer = buffer + offset;
269  uint32_t sbuffer_len = depth - offset;
270  uint32_t match_offset = 0;
271  SCLogDebug("sbuffer_len %"PRIu32, sbuffer_len);
272 #ifdef DEBUG
273  BUG_ON(sbuffer_len > buffer_len);
274 #endif
275 
276  /* \todo Add another optimization here. If cd->content_len is
277  * greater than sbuffer_len found is anyways NULL */
278 
279  /* do the actual search */
280  if (cd->flags & DETECT_CONTENT_NOCASE)
281  found = BoyerMooreNocase(cd->content, cd->content_len, sbuffer, sbuffer_len, cd->bm_ctx);
282  else
283  found = BoyerMoore(cd->content, cd->content_len, sbuffer, sbuffer_len, cd->bm_ctx);
284 
285  /* next we evaluate the result in combination with the
286  * negation flag. */
287  SCLogDebug("found %p cd negated %s", found, cd->flags & DETECT_CONTENT_NEGATED ? "true" : "false");
288 
289  if (found == NULL && !(cd->flags & DETECT_CONTENT_NEGATED)) {
290  goto no_match;
291  } else if (found == NULL && (cd->flags & DETECT_CONTENT_NEGATED)) {
292  goto match;
293  } else if (found != NULL && (cd->flags & DETECT_CONTENT_NEGATED)) {
294  SCLogDebug("content %"PRIu32" matched at offset %"PRIu32", but negated so no match", cd->id, match_offset);
295  /* don't bother carrying recursive matches now, for preceding
296  * relative keywords */
297  if (DETECT_CONTENT_IS_SINGLE(cd))
298  det_ctx->discontinue_matching = 1;
299  goto no_match;
300  } else {
301  match_offset = (uint32_t)((found - buffer) + cd->content_len);
302  SCLogDebug("content %"PRIu32" matched at offset %"PRIu32"", cd->id, match_offset);
303  det_ctx->buffer_offset = match_offset;
304 
305  /* Match branch, add replace to the list if needed */
306  if (cd->flags & DETECT_CONTENT_REPLACE) {
307  if (inspection_mode == DETECT_ENGINE_CONTENT_INSPECTION_MODE_PAYLOAD) {
308  /* we will need to replace content if match is confirmed */
309  det_ctx->replist = DetectReplaceAddToList(det_ctx->replist, found, cd);
310  } else {
311  SCLogWarning(SC_ERR_INVALID_VALUE, "Can't modify payload without packet");
312  }
313  }
314  if (!(cd->flags & DETECT_CONTENT_RELATIVE_NEXT)) {
315  SCLogDebug("no relative match coming up, so this is a match");
316  goto match;
317  }
318 
319  /* bail out if we have no next match. Technically this is an
320  * error, as the current cd has the DETECT_CONTENT_RELATIVE_NEXT
321  * flag set. */
322  if (sm->next == NULL) {
323  goto no_match;
324  }
325 
326  SCLogDebug("content %"PRIu32, cd->id);
327  KEYWORD_PROFILING_END(det_ctx, sm->type, 1);
328 
329  /* see if the next buffer keywords match. If not, we will
330  * search for another occurence of this content and see
331  * if the others match then until we run out of matches */
332  int r = DetectEngineContentInspection(de_ctx, det_ctx, s, sm->next, f, buffer, buffer_len, stream_start_offset, inspection_mode, data);
333  if (r == 1) {
334  SCReturnInt(1);
335  }
336 
337  if (det_ctx->discontinue_matching)
338  goto no_match;
339 
340  /* set the previous match offset to the start of this match + 1 */
341  prev_offset = (match_offset - (cd->content_len - 1));
342  SCLogDebug("trying to see if there is another match after prev_offset %"PRIu32, prev_offset);
343  }
344 
345  } while(1);
346 
347  } else if (sm->type == DETECT_ISDATAAT) {
348  SCLogDebug("inspecting isdataat");
349 
350  DetectIsdataatData *id = (DetectIsdataatData *)sm->ctx;
351  if (id->flags & ISDATAAT_RELATIVE) {
352  if (det_ctx->buffer_offset + id->dataat > buffer_len) {
353  SCLogDebug("det_ctx->buffer_offset + id->dataat %"PRIu32" > %"PRIu32, det_ctx->buffer_offset + id->dataat, buffer_len);
354  if (id->flags & ISDATAAT_NEGATED)
355  goto match;
356  goto no_match;
357  } else {
358  SCLogDebug("relative isdataat match");
359  if (id->flags & ISDATAAT_NEGATED)
360  goto no_match;
361  goto match;
362  }
363  } else {
364  if (id->dataat < buffer_len) {
365  SCLogDebug("absolute isdataat match");
366  if (id->flags & ISDATAAT_NEGATED)
367  goto no_match;
368  goto match;
369  } else {
370  SCLogDebug("absolute isdataat mismatch, id->isdataat %"PRIu32", buffer_len %"PRIu32"", id->dataat, buffer_len);
371  if (id->flags & ISDATAAT_NEGATED)
372  goto match;
373  goto no_match;
374  }
375  }
376 
377  } else if (sm->type == DETECT_PCRE) {
378  SCLogDebug("inspecting pcre");
379  DetectPcreData *pe = (DetectPcreData *)sm->ctx;
380  uint32_t prev_buffer_offset = det_ctx->buffer_offset;
381  uint32_t prev_offset = 0;
382  int r = 0;
383 
384  det_ctx->pcre_match_start_offset = 0;
385  do {
386  Packet *p = NULL;
387  if (inspection_mode == DETECT_ENGINE_CONTENT_INSPECTION_MODE_PAYLOAD)
388  p = (Packet *)data;
389  r = DetectPcrePayloadMatch(det_ctx, s, sm, p, f,
390  buffer, buffer_len);
391  if (r == 0) {
392  goto no_match;
393  }
394 
395  if (!(pe->flags & DETECT_PCRE_RELATIVE_NEXT)) {
396  SCLogDebug("no relative match coming up, so this is a match");
397  goto match;
398  }
399  KEYWORD_PROFILING_END(det_ctx, sm->type, 1);
400 
401  /* save it, in case we need to do a pcre match once again */
402  prev_offset = det_ctx->pcre_match_start_offset;
403 
404  /* see if the next payload keywords match. If not, we will
405  * search for another occurence of this pcre and see
406  * if the others match, until we run out of matches */
407  r = DetectEngineContentInspection(de_ctx, det_ctx, s, sm->next,
408  f, buffer, buffer_len, stream_start_offset, inspection_mode, data);
409  if (r == 1) {
410  SCReturnInt(1);
411  }
412 
413  if (det_ctx->discontinue_matching)
414  goto no_match;
415 
416  det_ctx->buffer_offset = prev_buffer_offset;
417  det_ctx->pcre_match_start_offset = prev_offset;
418  } while (1);
419 
420  } else if (sm->type == DETECT_BYTETEST) {
421  DetectBytetestData *btd = (DetectBytetestData *)sm->ctx;
422  uint8_t flags = btd->flags;
423  int32_t offset = btd->offset;
424  uint64_t value = btd->value;
425  if (flags & DETECT_BYTETEST_OFFSET_BE) {
426  offset = det_ctx->bj_values[offset];
427  }
428  if (flags & DETECT_BYTETEST_VALUE_BE) {
429  value = det_ctx->bj_values[value];
430  }
431 
432  /* if we have dce enabled we will have to use the endianness
433  * specified by the dce header */
434  if (flags & DETECT_BYTETEST_DCE) {
435  DCERPCState *dcerpc_state = (DCERPCState *)data;
436  /* enable the endianness flag temporarily. once we are done
437  * processing we reset the flags to the original value*/
438  flags |= ((dcerpc_state->dcerpc.dcerpchdr.packed_drep[0] & 0x10) ?
439  DETECT_BYTETEST_LITTLE: 0);
440  }
441 
442  if (DetectBytetestDoMatch(det_ctx, s, sm->ctx, buffer, buffer_len, flags,
443  offset, value) != 1) {
444  goto no_match;
445  }
446 
447  goto match;
448 
449  } else if (sm->type == DETECT_BYTEJUMP) {
450  DetectBytejumpData *bjd = (DetectBytejumpData *)sm->ctx;
451  uint8_t flags = bjd->flags;
452  int32_t offset = bjd->offset;
453 
454  if (flags & DETECT_BYTEJUMP_OFFSET_BE) {
455  offset = det_ctx->bj_values[offset];
456  }
457 
458  /* if we have dce enabled we will have to use the endianness
459  * specified by the dce header */
460  if (flags & DETECT_BYTEJUMP_DCE) {
461  DCERPCState *dcerpc_state = (DCERPCState *)data;
462  /* enable the endianness flag temporarily. once we are done
463  * processing we reset the flags to the original value*/
464  flags |= ((dcerpc_state->dcerpc.dcerpchdr.packed_drep[0] & 0x10) ?
465  DETECT_BYTEJUMP_LITTLE: 0);
466  }
467 
468  if (DetectBytejumpDoMatch(det_ctx, s, sm->ctx, buffer, buffer_len,
469  flags, offset) != 1) {
470  goto no_match;
471  }
472 
473  goto match;
474 
475  } else if (sm->type == DETECT_BYTE_EXTRACT) {
476 
477  DetectByteExtractData *bed = (DetectByteExtractData *)sm->ctx;
478  uint8_t endian = bed->endian;
479 
480  /* if we have dce enabled we will have to use the endianness
481  * specified by the dce header */
482  if ((bed->flags & DETECT_BYTE_EXTRACT_FLAG_ENDIAN) &&
483  endian == DETECT_BYTE_EXTRACT_ENDIAN_DCE) {
484 
485  DCERPCState *dcerpc_state = (DCERPCState *)data;
486  /* enable the endianness flag temporarily. once we are done
487  * processing we reset the flags to the original value*/
488  endian |= ((dcerpc_state->dcerpc.dcerpchdr.packed_drep[0] == 0x10) ?
489  DETECT_BYTE_EXTRACT_ENDIAN_LITTLE : DETECT_BYTE_EXTRACT_ENDIAN_BIG);
490  }
491 
492  if (DetectByteExtractDoMatch(det_ctx, sm, s, buffer,
493  buffer_len,
494  &det_ctx->bj_values[bed->local_id],
495  endian) != 1) {
496  goto no_match;
497  }
498 
499  goto match;
500 
501  /* we should never get here, but bail out just in case */
502  } else if (sm->type == DETECT_AL_URILEN) {
503  SCLogDebug("inspecting uri len");
504 
505  int r = 0;
506  DetectUrilenData *urilend = (DetectUrilenData *) sm->ctx;
507 
508  switch (urilend->mode) {
509  case DETECT_URILEN_EQ:
510  if (buffer_len == urilend->urilen1)
511  r = 1;
512  break;
513  case DETECT_URILEN_LT:
514  if (buffer_len < urilend->urilen1)
515  r = 1;
516  break;
517  case DETECT_URILEN_GT:
518  if (buffer_len > urilend->urilen1)
519  r = 1;
520  break;
521  case DETECT_URILEN_RA:
522  if (buffer_len > urilend->urilen1 &&
523  buffer_len < urilend->urilen2) {
524  r = 1;
525  }
526  break;
527  }
528 
529  if (r == 1) {
530  goto match;
531  }
532 
533  det_ctx->discontinue_matching = 0;
534 
535  goto no_match;
536 #ifdef HAVE_LUA
537  }
538  else if (sm->type == DETECT_LUA) {
539  SCLogDebug("lua starting");
540  /* for flowvar gets and sets we need to know the flow's lock status */
541  int flow_lock = LUA_FLOW_LOCKED_BY_PARENT;
542  if (inspection_mode <= DETECT_ENGINE_CONTENT_INSPECTION_MODE_STREAM)
543  flow_lock = LUA_FLOW_NOT_LOCKED_BY_PARENT;
544 
545  if (DetectLuaMatchBuffer(det_ctx, s, sm, buffer, buffer_len,
546  det_ctx->buffer_offset, f, flow_lock) != 1)
547  {
548  SCLogDebug("lua no_match");
549  goto no_match;
550  }
551  SCLogDebug("lua match");
552  goto match;
553 #endif /* HAVE_LUA */
554  } else {
555  SCLogDebug("sm->type %u", sm->type);
556 #ifdef DEBUG
557  BUG_ON(1);
558 #endif
559  }
560 
561 no_match:
562  KEYWORD_PROFILING_END(det_ctx, sm->type, 0);
563  SCReturnInt(0);
564 
565 match:
566  /* this sigmatch matched, inspect the next one. If it was the last,
567  * the buffer portion of the signature matched. */
568  if (sm->next != NULL) {
569  KEYWORD_PROFILING_END(det_ctx, sm->type, 1);
570  int r = DetectEngineContentInspection(de_ctx, det_ctx, s, sm->next, f, buffer, buffer_len, stream_start_offset, inspection_mode, data);
571  SCReturnInt(r);
572  } else {
573  KEYWORD_PROFILING_END(det_ctx, sm->type, 1);
574  SCReturnInt(1);
575  }
