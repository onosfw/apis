163 /** \brief Wait until we have the requested ammount of packets in the pool
164  *
165  * In some cases waiting for packets is undesirable. Especially when
166  * a wait would happen under a lock of some kind, other parts of the
167  * engine could have to wait.
168  *
169  * This function only returns when at least N packets are in our pool.
170  *
171  * \param n number of packets needed
172  */
173 void PacketPoolWaitForN(int n)
174 {
175  PktPool *my_pool = GetThreadPacketPool();
176  Packet *p = NULL;
177 
178  while (1) {
179  int i = 0;
180  PacketPoolWait();
181 
182  /* count packets in our stack */
183  p = my_pool->head;
184  while (p != NULL) {
185  if (++i == n)
186  return;
187 
188  p = p->next;
189  }
190 
191  /* continue counting in the return stack */
192  if (my_pool->return_stack.head != NULL) {
193  SCMutexLock(&my_pool->return_stack.mutex);
194  p = my_pool->return_stack.head;
195  while (p != NULL) {
196  if (++i == n) {
197  SCMutexUnlock(&my_pool->return_stack.mutex);
198  return;
199  }
200  p = p->next;
201  }
202  SCMutexUnlock(&my_pool->return_stack.mutex);
203 
204  /* or signal that we need packets and wait */
205  } else {
206  SCMutexLock(&my_pool->return_stack.mutex);
207  SC_ATOMIC_ADD(my_pool->return_stack.sync_now, 1);
208  SCCondWait(&my_pool->return_stack.cond, &my_pool->return_stack.mutex);
209  SCMutexUnlock(&my_pool->return_stack.mutex);
210  }
211  }
212 }

237 /** \brief Get a new packet from the packet pool
238  *
239  * Only allocates from the thread's local stack, or mallocs new packets.
240  * If the local stack is empty, first move all the return stack packets to
241  * the local stack.
242  * \retval Packet pointer, or NULL on failure.
243  */
244 Packet *PacketPoolGetPacket(void)
245 {
246  PktPool *pool = GetThreadPacketPool();
247 #ifdef DEBUG_VALIDATION
248  BUG_ON(pool->initialized == 0);
249  BUG_ON(pool->destroyed == 1);
250 #endif /* DEBUG_VALIDATION */
251  if (pool->head) {
252  /* Stack is not empty. */
253  Packet *p = pool->head;
254  pool->head = p->next;
255  p->pool = pool;
256  PACKET_REINIT(p);
257  return p;
258  }
259 
260  /* Local Stack is empty, so check the return stack, which requires
261  * locking. */
262  PacketPoolGetReturnedPackets(pool);
263 
264  /* Try to allocate again. Need to check for not empty again, since the
265  * return stack might have been empty too.
266  */
267  if (pool->head) {
268  /* Stack is not empty. */
269  Packet *p = pool->head;
270  pool->head = p->next;
271  p->pool = pool;
272  PACKET_REINIT(p);
273  return p;
274  }
275 
276  /* Failed to allocate a packet, so return NULL. */
277  /* Optionally, could allocate a new packet here. */
278  return NULL;
279 }

281 /** \brief Return packet to Packet pool
282  *
283  */
284 void PacketPoolReturnPacket(Packet *p)
285 {
286  PktPool *my_pool = GetThreadPacketPool();
287 
288  PACKET_RELEASE_REFS(p);
289 
290  PktPool *pool = p->pool;
291  if (pool == NULL) {
292  PacketFree(p);
293  return;
294  }
295 #ifdef DEBUG_VALIDATION
296  BUG_ON(pool->initialized == 0);
297  BUG_ON(pool->destroyed == 1);
298  BUG_ON(my_pool->initialized == 0);
299  BUG_ON(my_pool->destroyed == 1);
300 #endif /* DEBUG_VALIDATION */
301 
302  if (pool == my_pool) {
303  /* Push back onto this thread's own stack, so no locking. */
304  p->next = my_pool->head;
305  my_pool->head = p;
306  } else {
307  PktPool *pending_pool = my_pool->pending_pool;
308  if (pending_pool == NULL) {
309  /* No pending packet, so store the current packet. */
310  my_pool->pending_pool = pool;
311  my_pool->pending_head = p;
312  my_pool->pending_tail = p;
313  my_pool->pending_count = 1;
314  } else if (pending_pool == pool) {
315  /* Another packet for the pending pool list. */
316  p->next = my_pool->pending_head;
317  my_pool->pending_head = p;
318  my_pool->pending_count++;
319  if (SC_ATOMIC_GET(pool->return_stack.sync_now) || my_pool->pending_count > max_pending_return_packets) {
320  /* Return the entire list of pending packets. */
321  SCMutexLock(&pool->return_stack.mutex);
322  my_pool->pending_tail->next = pool->return_stack.head;
323  pool->return_stack.head = my_pool->pending_head;
324  SC_ATOMIC_RESET(pool->return_stack.sync_now);
325  SCMutexUnlock(&pool->return_stack.mutex);
326  SCCondSignal(&pool->return_stack.cond);
327  /* Clear the list of pending packets to return. */
328  my_pool->pending_pool = NULL;
329  my_pool->pending_head = NULL;
330  my_pool->pending_tail = NULL;
331  my_pool->pending_count = 0;
332  }
333  } else {
334  /* Push onto return stack for this pool */
335  SCMutexLock(&pool->return_stack.mutex);
336  p->next = pool->return_stack.head;
337  pool->return_stack.head = p;
338  SC_ATOMIC_RESET(pool->return_stack.sync_now);
339  SCMutexUnlock(&pool->return_stack.mutex);
340  SCCondSignal(&pool->return_stack.cond);
341  }
342  }
343 }

364 void PacketPoolInit(void)
365 {
366  extern intmax_t max_pending_packets;
367 
368 #ifndef TLS
369  TmqhPacketPoolInit();
370 #endif
371 
372  PktPool *my_pool = GetThreadPacketPool();
373 
374 #ifdef DEBUG_VALIDATION
375  BUG_ON(my_pool->initialized);
376  my_pool->initialized = 1;
377  my_pool->destroyed = 0;
378 #endif /* DEBUG_VALIDATION */
379 
380  SCMutexInit(&my_pool->return_stack.mutex, NULL);
381  SCCondInit(&my_pool->return_stack.cond, NULL);
382  SC_ATOMIC_INIT(my_pool->return_stack.sync_now);
383 
384  /* pre allocate packets */
385  SCLogDebug("preallocating packets... packet size %" PRIuMAX "",
386  (uintmax_t)SIZE_OF_PACKET);
387  int i = 0;
388  for (i = 0; i < max_pending_packets; i++) {
389  Packet *p = PacketGetFromAlloc();
390  if (unlikely(p == NULL)) {
391  SCLogError(SC_ERR_FATAL, "Fatal error encountered while allocating a packet. Exiting...");
392  exit(EXIT_FAILURE);
393  }
394  PacketPoolStorePacket(p);
395  }
396  SCLogInfo("preallocated %"PRIiMAX" packets. Total memory %"PRIuMAX"",
397  max_pending_packets, (uintmax_t)(max_pending_packets*SIZE_OF_PACKET));

401 void PacketPoolDestroy(void)
402 {
403  Packet *p = NULL;
404  PktPool *my_pool = GetThreadPacketPool();
405 
406 #ifdef DEBUG_VALIDATION
407  BUG_ON(my_pool->destroyed);
408 #endif /* DEBUG_VALIDATION */
409 
410  if (my_pool && my_pool->pending_pool != NULL) {
411  p = my_pool->pending_head;
412  while (p) {
413  Packet *next_p = p->next;
414  PacketFree(p);
415  p = next_p;
416  my_pool->pending_count--;
417  }
418 #ifdef DEBUG_VALIDATION
419  BUG_ON(my_pool->pending_count);
420 #endif /* DEBUG_VALIDATION */
421  my_pool->pending_pool = NULL;
422  my_pool->pending_head = NULL;
423  my_pool->pending_tail = NULL;
424  }
425 
426  while ((p = PacketPoolGetPacket()) != NULL) {
427  PacketFree(p);
428  }
429 
430  SC_ATOMIC_DESTROY(my_pool->return_stack.sync_now);
431 
432 #ifdef DEBUG_VALIDATION
433  my_pool->initialized = 0;
434  my_pool->destroyed = 1;
435 #endif /* DEBUG_VALIDATION */
436 }

443 void TmqhOutputPacketpool(ThreadVars *t, Packet *p)
444 {
445  int proot = 0;
446 
447  SCEnter();
448  SCLogDebug("Packet %p, p->root %p, alloced %s", p, p->root, p->flags & PKT_ALLOC ? "true" : "false");
449 
450  /** \todo make this a callback
451  * Release tcp segments. Done here after alerting can use them. */
452  if (p->flow != NULL && p->proto == IPPROTO_TCP) {
453  SCMutexLock(&p->flow->m);
454  StreamTcpPruneSession(p->flow, p->flowflags & FLOW_PKT_TOSERVER ?
455  STREAM_TOSERVER : STREAM_TOCLIENT);
456  SCMutexUnlock(&p->flow->m);
457  }
458 
459  if (IS_TUNNEL_PKT(p)) {
460  SCLogDebug("Packet %p is a tunnel packet: %s",
461  p,p->root ? "upper layer" : "tunnel root");
462 
463  /* get a lock to access root packet fields */
464  SCMutex *m = p->root ? &p->root->tunnel_mutex : &p->tunnel_mutex;
465  SCMutexLock(m);
466 
467  if (IS_TUNNEL_ROOT_PKT(p)) {
468  SCLogDebug("IS_TUNNEL_ROOT_PKT == TRUE");
469  if (TUNNEL_PKT_TPR(p) == 0) {
470  SCLogDebug("TUNNEL_PKT_TPR(p) == 0, no more tunnel packet "
471  "depending on this root");
472  /* if this packet is the root and there are no
473  * more tunnel packets, return it to the pool */
474 
475  /* fall through */
476  } else {
477  SCLogDebug("tunnel root Packet %p: TUNNEL_PKT_TPR(p) > 0, so "
478  "packets are still depending on this root, setting "
479  "p->tunnel_verdicted == 1", p);
480  /* if this is the root and there are more tunnel
481  * packets, return this to the pool. It's still referenced
482  * by the tunnel packets, and we will return it
483  * when we handle them */
484  SET_TUNNEL_PKT_VERDICTED(p);
485 
486  PACKET_PROFILING_END(p);
487  SCMutexUnlock(m);
488  SCReturn;
489  }
490  } else {
491  SCLogDebug("NOT IS_TUNNEL_ROOT_PKT, so tunnel pkt");
492 
493  /* the p->root != NULL here seems unnecessary: IS_TUNNEL_PKT checks
494  * that p->tunnel_pkt == 1, IS_TUNNEL_ROOT_PKT checks that +
495  * p->root == NULL. So when we are here p->root can only be
496  * non-NULL, right? CLANG thinks differently. May be a FP, but
497  * better safe than sorry. VJ */
498  if (p->root != NULL && IS_TUNNEL_PKT_VERDICTED(p->root) &&
499  TUNNEL_PKT_TPR(p) == 1)
500  {
501  SCLogDebug("p->root->tunnel_verdicted == 1 && TUNNEL_PKT_TPR(p) == 1");
502  /* the root is ready and we are the last tunnel packet,
503  * lets enqueue them both. */
504  TUNNEL_DECR_PKT_TPR_NOLOCK(p);
505 
506  /* handle the root */
507  SCLogDebug("setting proot = 1 for root pkt, p->root %p "
508  "(tunnel packet %p)", p->root, p);
509  proot = 1;
510 
511  /* fall through */
512  } else {
513  /* root not ready yet, so get rid of the tunnel pkt only */
514 
515  SCLogDebug("NOT p->root->tunnel_verdicted == 1 && "
516  "TUNNEL_PKT_TPR(p) == 1 (%" PRIu32 ")", TUNNEL_PKT_TPR(p));
517 
518  TUNNEL_DECR_PKT_TPR_NOLOCK(p);
519 
520  /* fall through */
521  }
522  }
523  SCMutexUnlock(m);
524 
525  SCLogDebug("tunnel stuff done, move on (proot %d)", proot);
526  }
527 
528  FlowDeReference(&p->flow);
529 
530  /* we're done with the tunnel root now as well */
531  if (proot == 1) {
532  SCLogDebug("getting rid of root pkt... alloc'd %s", p->root->flags & PKT_ALLOC ? "true" : "false");
533 
534  FlowDeReference(&p->root->flow);
535 
536  p->root->ReleasePacket(p->root);
537  p->root = NULL;
538  }
539 
540  PACKET_PROFILING_END(p);
541 
542  p->ReleasePacket(p);
543 
544  SCReturn;
545 }

570 /**
571  * \brief Set the max_pending_return_packets value
572  *
573  * Set it to the max pending packets value, devided by the number
574  * of lister threads. Normally, in autofp these are the stream/detect/log
575  * worker threads.
576  *
577  * The max_pending_return_packets value needs to stay below the packet
578  * pool size of the 'producers' (normally pkt capture threads but also
579  * flow timeout injection ) to avoid a deadlock where all the 'workers'
580  * keep packets in their return pools, while the capture thread can't
581  * continue because its pool is empty.
582  */
583 void PacketPoolPostRunmodes(void)
584 {
585  extern intmax_t max_pending_packets;
586 
587  uint32_t threads = TmThreadCountThreadsByTmmFlags(TM_FLAG_DETECT_TM);
588  if (threads == 0)
589  return;
590  if (threads > max_pending_packets)
591  return;
592 
593  uint32_t packets = (max_pending_packets / threads) - 1;
594  if (packets < max_pending_return_packets)
595  max_pending_return_packets = packets;
596 
597  SCLogDebug("detect threads %u, max packets %u, max_pending_return_packets %u",
598  threads, (uint)threads, max_pending_return_packets);
599 }
