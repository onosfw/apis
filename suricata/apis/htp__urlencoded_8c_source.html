43 /**
44  * This method is invoked whenever a piece of data, belonging to a single field (name or value)
45  * becomes available. It will either create a new parameter or store the transient information
46  * until a parameter can be created.
47  *
48  * @param[in] urlenp
49  * @param[in] data
50  * @param[in] startpos
51  * @param[in] endpos
52  * @param[in] c Should contain -1 if the reason this function is called is because the end of
53  * the current data chunk is reached.
54  */
55 static void htp_urlenp_add_field_piece(htp_urlenp_t *urlenp, const unsigned char *data, size_t startpos, size_t endpos, int last_char) {
56  // Add field if we know it ended (last_char is something other than -1)
57  // or if we know that there won't be any more input data (urlenp->_complete is true).
58  if ((last_char != -1) || (urlenp->_complete)) {
59  // Prepare the field value, assembling from multiple pieces as necessary.
60 
61  bstr *field = NULL;
62 
63  // Did we use the string builder for this field?
64  if (bstr_builder_size(urlenp->_bb) > 0) {
65  // The current field consists of more than once piece, we have to use the string builder.
66 
67  // Add current piece to string builder.
68  if ((data != NULL) && (endpos - startpos > 0)) {
69  bstr_builder_append_mem(urlenp->_bb, data + startpos, endpos - startpos);
70  }
71 
72  // Generate the field and clear the string builder.
73  field = bstr_builder_to_str(urlenp->_bb);
74  if (field == NULL) return;
75 
76  bstr_builder_clear(urlenp->_bb);
77  } else {
78  // We only have the current piece to work with, so no need to involve the string builder.
79  if ((data != NULL) && (endpos - startpos > 0)) {
80  field = bstr_dup_mem(data + startpos, endpos - startpos);
81  if (field == NULL) return;
82  }
83  }
84 
85  // Process field as key or value, as appropriate.
86 
87  if (urlenp->_state == HTP_URLENP_STATE_KEY) {
88  // Key.
89 
90  // If there is no more work left to do, then we have a single key. Add it.
91  if ((urlenp->_complete)||(last_char == urlenp->argument_separator)) {
92 
93  // Handling empty pairs is tricky. We don't want to create a pair for
94  // an entirely empty input, but in some cases it may be appropriate
95  // (e.g., /index.php?&q=2).
96  if ((field != NULL)||(last_char == urlenp->argument_separator)) {
97  // Add one pair, with an empty value and possibly empty key too.
98 
99  bstr *name = field;
100  if (name == NULL) {
101  name = bstr_dup_c("");
102  if (name == NULL) return;
103  }
104 
105  bstr *value = bstr_dup_c("");
106  if (value == NULL) {
107  bstr_free(name);
108  return;
109  }
110 
111  if (urlenp->decode_url_encoding) {
112  htp_tx_urldecode_params_inplace(urlenp->tx, name);
113  }
114 
115  htp_table_addn(urlenp->params, name, value);
116 
117  urlenp->_name = NULL;
118 
119  #ifdef HTP_DEBUG
120  fprint_raw_data(stderr, "NAME", bstr_ptr(name), bstr_len(name));
121  fprint_raw_data(stderr, "VALUE", bstr_ptr(value), bstr_len(value));
122  #endif
123  }
124  } else {
125  // This key will possibly be followed by a value, so keep it for later.
126  urlenp->_name = field;
127  }
128  } else {
129  // Value (with a key remembered from before).
130 
131  bstr *name = urlenp->_name;
132  urlenp->_name = NULL;
133 
134  if (name == NULL) {
135  name = bstr_dup_c("");
136  if (name == NULL) {
137  bstr_free(field);
138  return;
139  }
140  }
141 
142  bstr *value = field;
143  if (value == NULL) {
144  value = bstr_dup_c("");
145  if (value == NULL) {
146  bstr_free(name);
147  return;
148  }
149  }
150 
151  if (urlenp->decode_url_encoding) {
152  htp_tx_urldecode_params_inplace(urlenp->tx, name);
153  htp_tx_urldecode_params_inplace(urlenp->tx, value);
154  }
155 
156  htp_table_addn(urlenp->params, name, value);
157 
158  #ifdef HTP_DEBUG
159  fprint_raw_data(stderr, "NAME", bstr_ptr(name), bstr_len(name));
160  fprint_raw_data(stderr, "VALUE", bstr_ptr(value), bstr_len(value));
161  #endif
162  }
163  } else {
164  // The field has not ended. We'll make a copy of of the available data for later.
165  if ((data != NULL) && (endpos - startpos > 0)) {
166  bstr_builder_append_mem(urlenp->_bb, data + startpos, endpos - startpos);
167  }
168  }
169 }

171 /**
172  * Creates a new URLENCODED parser.
173  *
174  * @return New parser, or NULL on memory allocation failure.
175  */
176 htp_urlenp_t *htp_urlenp_create(htp_tx_t *tx) {
177  htp_urlenp_t *urlenp = calloc(1, sizeof (htp_urlenp_t));
178  if (urlenp == NULL) return NULL;
179 
180  urlenp->tx = tx;
181 
182  urlenp->params = htp_table_create(HTP_URLENP_DEFAULT_PARAMS_SIZE);
183  if (urlenp->params == NULL) {
184  free(urlenp);
185  return NULL;
186  }
187 
188  urlenp->_bb = bstr_builder_create();
189  if (urlenp->_bb == NULL) {
190  htp_table_destroy(urlenp->params);
191  free(urlenp);
192  return NULL;
193  }
194 
195  urlenp->argument_separator = '&';
196  urlenp->decode_url_encoding = 1;
197  urlenp->_state = HTP_URLENP_STATE_KEY;
198 
199  return urlenp;
200 }

202 /**
203  * Destroys an existing URLENCODED parser.
204  *
205  * @param[in] urlenp
206  */
207 void htp_urlenp_destroy(htp_urlenp_t *urlenp) {
208  if (urlenp == NULL) return;
209 
210  if (urlenp->_name != NULL) {
211  bstr_free(urlenp->_name);
212  }
213 
214  bstr_builder_destroy(urlenp->_bb);
215 
216  if (urlenp->params != NULL) {
217  // Destroy parameters.
218  for (size_t i = 0, n = htp_table_size(urlenp->params); i < n; i++) {
219  bstr *b = htp_table_get_index(urlenp->params, i, NULL);
220  // Parameter name will be freed by the table code.
221  bstr_free(b);
222  }
223 
224  htp_table_destroy(urlenp->params);
225  }
226 
227  free(urlenp);
228 }

259 /**
260  * Parses the provided data chunk, keeping state to allow streaming parsing, i.e., the
261  * parsing where only partial information is available at any one time. The method
262  * htp_urlenp_finalize() must be invoked at the end to finalize parsing.
263  *
264  * @param[in] urlenp
265  * @param[in] _data
266  * @param[in] len
267  * @return
268  */
269 htp_status_t htp_urlenp_parse_partial(htp_urlenp_t *urlenp, const void *_data, size_t len) {
270  unsigned char *data = (unsigned char *) _data;
271  size_t startpos = 0;
272  size_t pos = 0;
273  int c;
274 
275  if (data == NULL) len = 0;
276 
277  do {
278  // Get the next character, or use -1 to indicate end of input.
279  if (pos < len) c = data[pos];
280  else c = -1;
281 
282  switch (urlenp->_state) {
283 
284  case HTP_URLENP_STATE_KEY:
285  // Look for =, argument separator, or end of input.
286  if ((c == '=') || (c == urlenp->argument_separator) || (c == -1)) {
287  // Data from startpos to pos.
288  htp_urlenp_add_field_piece(urlenp, data, startpos, pos, c);
289 
290  // If it's not the end of input, then it must be the end of this field.
291  if (c != -1) {
292  // Next state.
293  startpos = pos + 1;
294 
295  if (c == urlenp->argument_separator) {
296  urlenp->_state = HTP_URLENP_STATE_KEY;
297  } else {
298  urlenp->_state = HTP_URLENP_STATE_VALUE;
299  }
300  }
301  }
302 
303  pos++;
304 
305  break;
306 
307  case HTP_URLENP_STATE_VALUE:
308  // Look for argument separator or end of input.
309  if ((c == urlenp->argument_separator) || (c == -1)) {
310  // Data from startpos to pos.
311  htp_urlenp_add_field_piece(urlenp, data, startpos, pos, c);
312 
313  // If it's not the end of input, then it must be the end of this field.
314  if (c != -1) {
315  // Next state.
316  startpos = pos + 1;
317  urlenp->_state = HTP_URLENP_STATE_KEY;
318  }
319  }
320 
321  pos++;
322 
323  break;
324 
325  default:
326  // Invalid state.
327  return HTP_ERROR;
328  }
329  } while (c != -1);
330 
331  return HTP_OK;
332 }
