18 /**
19  * \file
20  *
21  * \author Pablo Rincon <pablo.rincon.crespo@gmail.com>
22  *
23  * Implements the ssh.softwareversion keyword
24  * You can match over the software version string of ssh, and it will
25  * be compared from the beginning of the string so you can say for
26  * example ssh.softwareversion:"PuTTY" and it can match, or you can
27  * also specify the version, something like
28  * ssh.softwareversion:"PuTTY-Release-0.55"
29  * I find this useful to match over a known vulnerable server/client
30  * software version incombination to other checks, so you can know
31  * that the risk is higher
32  */

92  SCLogDebug("registering ssh.softwareversion rule option");

95  if (parse_regex == NULL) {

112 /**
113  * \brief match the specified version on a ssh session
114  *
115  * \param t pointer to thread vars
116  * \param det_ctx pointer to the pattern matcher thread
117  * \param p pointer to the current packet
118  * \param m pointer to the sigmatch that we will cast into DetectSshSoftwareVersionData
119  *
120  * \retval 0 no match
121  * \retval 1 match
122  */
123 int DetectSshSoftwareVersionMatch (ThreadVars *t, DetectEngineThreadCtx *det_ctx, Flow *f, uint8_t flags, void *state, Signature *s, SigMatch *m)
124 {
125  SCEnter();
126 
127  DetectSshSoftwareVersionData *ssh = (DetectSshSoftwareVersionData *)m->ctx;
128  SshState *ssh_state = (SshState *)state;
129  if (ssh_state == NULL) {
130  SCLogDebug("no ssh state, no match");
131  SCReturnInt(0);
132  }
133 
134  int ret = 0;
135  if ((flags & STREAM_TOCLIENT) && (ssh_state->srv_hdr.flags & SSH_FLAG_VERSION_PARSED)) {
136  SCLogDebug("looking for ssh server softwareversion %s length %"PRIu16" on %s", ssh->software_ver, ssh->len, ssh_state->srv_hdr.software_version);
137  ret = (strncmp((char *) ssh_state->srv_hdr.software_version, (char *) ssh->software_ver, ssh->len) == 0)? 1 : 0;
138  } else if ((flags & STREAM_TOSERVER) && (ssh_state->cli_hdr.flags & SSH_FLAG_VERSION_PARSED)) {
139  SCLogDebug("looking for ssh client softwareversion %s length %"PRIu16" on %s", ssh->software_ver, ssh->len, ssh_state->cli_hdr.software_version);
140  ret = (strncmp((char *) ssh_state->cli_hdr.software_version, (char *) ssh->software_ver, ssh->len) == 0)? 1 : 0;
141  }
142  SCReturnInt(ret);
143 }

145 /**
146  * \brief This function is used to parse IPV4 ip_id passed via keyword: "id"
147  *
148  * \param idstr Pointer to the user provided id option
149  *
150  * \retval id_d pointer to DetectSshSoftwareVersionData on success
151  * \retval NULL on failure
152  */
153 DetectSshSoftwareVersionData *DetectSshSoftwareVersionParse (char *str)
154 {
155  DetectSshSoftwareVersionData *ssh = NULL;
156 #define MAX_SUBSTRINGS 30
157  int ret = 0, res = 0;
158  int ov[MAX_SUBSTRINGS];
159 
160  ret = pcre_exec(parse_regex, parse_regex_study, str, strlen(str), 0, 0,
161  ov, MAX_SUBSTRINGS);
162 
163  if (ret < 1 || ret > 3) {
164  SCLogError(SC_ERR_PCRE_MATCH, "invalid ssh.softwareversion option");
165  goto error;
166  }
167 
168  if (ret > 1) {
169  const char *str_ptr = NULL;
170  res = pcre_get_substring((char *)str, ov, MAX_SUBSTRINGS, 1, &str_ptr);
171  if (res < 0) {
172  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
173  goto error;
174  }
175 
176  /* We have a correct id option */
177  ssh = SCMalloc(sizeof(DetectSshSoftwareVersionData));
178  if (unlikely(ssh == NULL))
179  goto error;
180 
181  ssh->software_ver = (uint8_t *)SCStrdup((char *)str_ptr);
182  if (ssh->software_ver == NULL) {
183  goto error;
184  }
185  pcre_free_substring(str_ptr);
186 
187  ssh->len = strlen((char *)ssh->software_ver);
188 
189  SCLogDebug("will look for ssh %s", ssh->software_ver);
190  }
191 
192  return ssh;
193 
194 error:
195  if (ssh != NULL)
196  DetectSshSoftwareVersionFree(ssh);
197  return NULL;
198 
199 }

201 /**
202  * \brief this function is used to add the parsed "id" option
203  * \brief into the current signature
204  *
205  * \param de_ctx pointer to the Detection Engine Context
206  * \param s pointer to the Current Signature
207  * \param idstr pointer to the user provided "id" option
208  *
209  * \retval 0 on Success
210  * \retval -1 on Failure
211  */
212 static int DetectSshSoftwareVersionSetup (DetectEngineCtx *de_ctx, Signature *s, char *str)
213 {
214  DetectSshSoftwareVersionData *ssh = NULL;
215  SigMatch *sm = NULL;
216 
217  ssh = DetectSshSoftwareVersionParse(str);
218  if (ssh == NULL)
219  goto error;
220 
221  /* Okay so far so good, lets get this into a SigMatch
222  * and put it in the Signature. */
223  sm = SigMatchAlloc();
224  if (sm == NULL)
225  goto error;
226 
227  if (s->alproto != ALPROTO_UNKNOWN && s->alproto != ALPROTO_SSH) {
228  SCLogError(SC_ERR_CONFLICTING_RULE_KEYWORDS, "rule contains conflicting keywords.");
229  goto error;
230  }
231 
232  sm->type = DETECT_AL_SSH_SOFTWAREVERSION;
233  sm->ctx = (void *)ssh;
234 
235  s->flags |= SIG_FLAG_APPLAYER;
236  s->alproto = ALPROTO_SSH;
237 
238  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_AMATCH);
239 
240  return 0;
241 
242 error:
243  if (ssh != NULL)
244  DetectSshSoftwareVersionFree(ssh);
245  if (sm != NULL)
246  SCFree(sm);
247  return -1;
248 
249 }
