47 /**
48  * \internal
49  * \brief this function is used to decode IEEE802.1q packets
50  *
51  * \param tv pointer to the thread vars
52  * \param dtv pointer code thread vars
53  * \param p pointer to the packet struct
54  * \param pkt pointer to the raw packet
55  * \param len packet len
56  * \param pq pointer to the packet queue
57  *
58  */
59 int DecodeVLAN(ThreadVars *tv, DecodeThreadVars *dtv, Packet *p, uint8_t *pkt, uint16_t len, PacketQueue *pq)
60 {
61  uint32_t proto;
62 
63  if (p->vlan_idx == 0)
64  StatsIncr(tv, dtv->counter_vlan);
65  else if (p->vlan_idx == 1)
66  StatsIncr(tv, dtv->counter_vlan_qinq);
67 
68  if(len < VLAN_HEADER_LEN) {
69  ENGINE_SET_INVALID_EVENT(p, VLAN_HEADER_TOO_SMALL);
70  return TM_ECODE_FAILED;
71  }
72  if (p->vlan_idx >= 2) {
73  ENGINE_SET_EVENT(p,VLAN_HEADER_TOO_MANY_LAYERS);
74  return TM_ECODE_FAILED;
75  }
76 
77  p->vlanh[p->vlan_idx] = (VLANHdr *)pkt;
78  if(p->vlanh[p->vlan_idx] == NULL)
79  return TM_ECODE_FAILED;
80 
81  proto = GET_VLAN_PROTO(p->vlanh[p->vlan_idx]);
82 
83  SCLogDebug("p %p pkt %p VLAN protocol %04x VLAN PRI %d VLAN CFI %d VLAN ID %d Len: %" PRId32 "",
84  p, pkt, proto, GET_VLAN_PRIORITY(p->vlanh[p->vlan_idx]),
85  GET_VLAN_CFI(p->vlanh[p->vlan_idx]), GET_VLAN_ID(p->vlanh[p->vlan_idx]), len);
86 
87  /* only store the id for flow hashing if it's not disabled. */
88  if (dtv->vlan_disabled == 0)
89  p->vlan_id[p->vlan_idx] = (uint16_t)GET_VLAN_ID(p->vlanh[p->vlan_idx]);
90 
91  p->vlan_idx++;
92 
93  switch (proto) {
94  case ETHERNET_TYPE_IP:
95  DecodeIPV4(tv, dtv, p, pkt + VLAN_HEADER_LEN,
96  len - VLAN_HEADER_LEN, pq);
97  break;
98  case ETHERNET_TYPE_IPV6:
99  DecodeIPV6(tv, dtv, p, pkt + VLAN_HEADER_LEN,
100  len - VLAN_HEADER_LEN, pq);
101  break;
102  case ETHERNET_TYPE_PPPOE_SESS:
103  DecodePPPOESession(tv, dtv, p, pkt + VLAN_HEADER_LEN,
104  len - VLAN_HEADER_LEN, pq);
105  break;
106  case ETHERNET_TYPE_PPPOE_DISC:
107  DecodePPPOEDiscovery(tv, dtv, p, pkt + VLAN_HEADER_LEN,
108  len - VLAN_HEADER_LEN, pq);
109  break;
110  case ETHERNET_TYPE_VLAN:
111  case ETHERNET_TYPE_8021AD:
112  if (p->vlan_idx >= 2) {
113  ENGINE_SET_EVENT(p,VLAN_HEADER_TOO_MANY_LAYERS);
114  return TM_ECODE_OK;
115  } else {
116  DecodeVLAN(tv, dtv, p, pkt + VLAN_HEADER_LEN,
117  len - VLAN_HEADER_LEN, pq);
118  }
119  break;
120  default:
121  SCLogDebug("unknown VLAN type: %" PRIx32 "", proto);
122  ENGINE_SET_INVALID_EVENT(p, VLAN_UNKNOWN_TYPE);
123  return TM_ECODE_OK;
124  }
125 
126  return TM_ECODE_OK;
127 }

129 uint16_t DecodeVLANGetId(const Packet *p, uint8_t layer)
130 {
131  if (unlikely(layer > 1))
132  return 0;
133 
134  if (p->vlanh[layer] == NULL && (p->vlan_idx >= (layer + 1))) {
135  return p->vlan_id[layer];
136  } else {
137  return GET_VLAN_ID(p->vlanh[layer]);
138  }
139  return 0;
140 }
