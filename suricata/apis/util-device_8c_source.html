37 /**
38  * \brief Add a pcap device for monitoring
39  *
40  * \param dev string with the device name
41  *
42  * \retval 0 on success.
43  * \retval -1 on failure.
44  */
45 int LiveRegisterDevice(char *dev)
46 {
47  LiveDevice *pd = SCMalloc(sizeof(LiveDevice));
48  if (unlikely(pd == NULL)) {
49  return -1;
50  }
51 
52  pd->dev = SCStrdup(dev);
53  if (unlikely(pd->dev == NULL)) {
54  SCFree(pd);
55  return -1;
56  }
57  SC_ATOMIC_INIT(pd->pkts);
58  SC_ATOMIC_INIT(pd->drop);
59  SC_ATOMIC_INIT(pd->invalid_checksums);
60  pd->ignore_checksum = 0;
61  TAILQ_INSERT_TAIL(&live_devices, pd, next);
62 
63  SCLogDebug("Device \"%s\" registered.", dev);
64  return 0;
65 }

144 int LiveBuildDeviceListCustom(char * runmode, char * itemname)
145 {
146  ConfNode *base = ConfGetNode(runmode);
147  ConfNode *child;
148  int i = 0;
149 
150  if (base == NULL)
151  return 0;
152 
153  TAILQ_FOREACH(child, &base->head, next) {
154  ConfNode *subchild;
155  TAILQ_FOREACH(subchild, &child->head, next) {
156  if ((!strcmp(subchild->name, itemname))) {
157  if (!strcmp(subchild->val, "default"))
158  break;
159  SCLogInfo("Adding %s %s from config file",
160  itemname, subchild->val);
161  LiveRegisterDevice(subchild->val);
162  i++;
163  }
164  }
165  }
166 
167  return i;
168 }

179 int LiveDeviceListClean()
180 {
181  SCEnter();
182  LiveDevice *pd, *tpd;
183 
184  TAILQ_FOREACH_SAFE(pd, &live_devices, next, tpd) {
185  if (live_devices_stats) {
186  SCLogNotice("Stats for '%s': pkts: %" PRIu64", drop: %" PRIu64 " (%.2f%%), invalid chksum: %" PRIu64,
187  pd->dev,
188  SC_ATOMIC_GET(pd->pkts),
189  SC_ATOMIC_GET(pd->drop),
190  100 * (SC_ATOMIC_GET(pd->drop) * 1.0) / SC_ATOMIC_GET(pd->pkts),
191  SC_ATOMIC_GET(pd->invalid_checksums));
192  }
193  if (pd->dev)
194  SCFree(pd->dev);
195  SC_ATOMIC_DESTROY(pd->pkts);
196  SC_ATOMIC_DESTROY(pd->drop);
197  SC_ATOMIC_DESTROY(pd->invalid_checksums);
198  SCFree(pd);
199  }
200 
201  SCReturnInt(TM_ECODE_OK);
202 }
