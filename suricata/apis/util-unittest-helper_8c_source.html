65  * \brief UTHBuildPacketReal is a function that create tcp/udp packets for unittests
66  * specifying ip and port sources and destinations (IPV6)
67  *
68  * \param payload pointer to the payloadd buffer
69  * \param payload_len pointer to the length of the payload
70  * \param ipproto Protocols allowed atm are IPPROTO_TCP and IPPROTO_UDP
71  * \param src pointer to a string containing the ip source
72  * \param dst pointer to a string containing the ip destination
73  * \param sport pointer to a string containing the port source
74  * \param dport pointer to a string containing the port destination
75  *
76  * \retval Packet pointer to the built in packet
77  */
78 Packet *UTHBuildPacketIPV6Real(uint8_t *payload, uint16_t payload_len,
79  uint8_t ipproto, char *src, char *dst,
80  uint16_t sport, uint16_t dport)
81 {
82  uint32_t in[4];
83 
84  Packet *p = PacketGetFromAlloc();
85  if (unlikely(p == NULL))
86  return NULL;
87 
88  TimeSet(&p->ts);
89 
90  p->src.family = AF_INET6;
91  p->dst.family = AF_INET6;
92  p->payload = payload;
93  p->payload_len = payload_len;
94  p->proto = ipproto;
95 
96  p->ip6h = SCMalloc(sizeof(IPV6Hdr));
97  if (p->ip6h == NULL)
98  goto error;
99  memset(p->ip6h, 0, sizeof(IPV6Hdr));
100  p->ip6h->s_ip6_nxt = ipproto;
101  p->ip6h->s_ip6_plen = htons(payload_len + sizeof(TCPHdr));
102 
103  if (inet_pton(AF_INET6, src, &in) != 1)
104  goto error;
105  p->src.addr_data32[0] = in[0];
106  p->src.addr_data32[1] = in[1];
107  p->src.addr_data32[2] = in[2];
108  p->src.addr_data32[3] = in[3];
109  p->sp = sport;
110  p->ip6h->s_ip6_src[0] = in[0];
111  p->ip6h->s_ip6_src[1] = in[1];
112  p->ip6h->s_ip6_src[2] = in[2];
113  p->ip6h->s_ip6_src[3] = in[3];
114 
115  if (inet_pton(AF_INET6, dst, &in) != 1)
116  goto error;
117  p->dst.addr_data32[0] = in[0];
118  p->dst.addr_data32[1] = in[1];
119  p->dst.addr_data32[2] = in[2];
120  p->dst.addr_data32[3] = in[3];
121  p->dp = dport;
122  p->ip6h->s_ip6_dst[0] = in[0];
123  p->ip6h->s_ip6_dst[1] = in[1];
124  p->ip6h->s_ip6_dst[2] = in[2];
125  p->ip6h->s_ip6_dst[3] = in[3];
126 
127  p->tcph = SCMalloc(sizeof(TCPHdr));
128  if (p->tcph == NULL)
129  goto error;
130  memset(p->tcph, 0, sizeof(TCPHdr));
131  p->tcph->th_sport = htons(sport);
132  p->tcph->th_dport = htons(dport);
133 
134  SET_PKT_LEN(p, sizeof(IPV6Hdr) + sizeof(TCPHdr) + payload_len);
135  return p;
136 
137 error:
138  if (p != NULL) {
139  if (p->ip6h != NULL) {
140  SCFree(p->ip6h);
141  }
142  if (p->tcph != NULL) {
143  SCFree(p->tcph);
144  }
145  SCFree(p);
146  }
147  return NULL;
148 }

150 /**
151  * \brief UTHBuildPacketReal is a function that create tcp/udp packets for unittests
152  * specifying ip and port sources and destinations
153  *
154  * \param payload pointer to the payloadd buffer
155  * \param payload_len pointer to the length of the payload
156  * \param ipproto Protocols allowed atm are IPPROTO_TCP and IPPROTO_UDP
157  * \param src pointer to a string containing the ip source
158  * \param dst pointer to a string containing the ip destination
159  * \param sport pointer to a string containing the port source
160  * \param dport pointer to a string containing the port destination
161  *
162  * \retval Packet pointer to the built in packet
163  */
164 Packet *UTHBuildPacketReal(uint8_t *payload, uint16_t payload_len,
165  uint8_t ipproto, char *src, char *dst,
166  uint16_t sport, uint16_t dport)
167 {
168  struct in_addr in;
169 
170  Packet *p = PacketGetFromAlloc();
171  if (unlikely(p == NULL))
172  return NULL;
173 
174  struct timeval tv;
175  TimeGet(&tv);
176  COPY_TIMESTAMP(&tv, &p->ts);
177 
178  p->src.family = AF_INET;
179  p->dst.family = AF_INET;
180  p->payload = payload;
181  p->payload_len = payload_len;
182  p->proto = ipproto;
183 
184  if (inet_pton(AF_INET, src, &in) != 1)
185  goto error;
186  p->src.addr_data32[0] = in.s_addr;
187  p->sp = sport;
188 
189  if (inet_pton(AF_INET, dst, &in) != 1)
190  goto error;
191  p->dst.addr_data32[0] = in.s_addr;
192  p->dp = dport;
193 
194  p->ip4h = (IPV4Hdr *)GET_PKT_DATA(p);
195  if (p->ip4h == NULL)
196  goto error;
197 
198  p->ip4h->s_ip_src.s_addr = p->src.addr_data32[0];
199  p->ip4h->s_ip_dst.s_addr = p->dst.addr_data32[0];
200  p->ip4h->ip_proto = ipproto;
201  p->ip4h->ip_verhl = sizeof(IPV4Hdr);
202  p->proto = ipproto;
203 
204  int hdr_offset = sizeof(IPV4Hdr);
205  switch (ipproto) {
206  case IPPROTO_UDP:
207  p->udph = (UDPHdr *)(GET_PKT_DATA(p) + sizeof(IPV4Hdr));
208  if (p->udph == NULL)
209  goto error;
210 
211  p->udph->uh_sport = sport;
212  p->udph->uh_dport = dport;
213  hdr_offset += sizeof(UDPHdr);
214  break;
215  case IPPROTO_TCP:
216  p->tcph = (TCPHdr *)(GET_PKT_DATA(p) + sizeof(IPV4Hdr));
217  if (p->tcph == NULL)
218  goto error;
219 
220  p->tcph->th_sport = htons(sport);
221  p->tcph->th_dport = htons(dport);
222  hdr_offset += sizeof(TCPHdr);
223  break;
224  case IPPROTO_ICMP:
225  p->icmpv4h = (ICMPV4Hdr *)(GET_PKT_DATA(p) + sizeof(IPV4Hdr));
226  if (p->icmpv4h == NULL)
227  goto error;
228 
229  hdr_offset += sizeof(ICMPV4Hdr);
230  break;
231  default:
232  break;
233  /* TODO: Add more protocols */
234  }
235 
236  PacketCopyDataOffset(p, hdr_offset, payload, payload_len);
237  SET_PKT_LEN(p, hdr_offset + payload_len);
238  p->payload = GET_PKT_DATA(p)+hdr_offset;
239 
240  return p;
241 
242 error:
243  SCFree(p);
244  return NULL;
245 }

491 /**
492  * \brief UTHGenericTest: function that perfom a generic check taking care of
493  * as maximum common unittest elements as possible.
494  * It will create a detection engine, append an array
495  * of signatures an check the spected results for each
496  * of them, it check matches for an array of packets
497  *
498  * \param pkt pointer to the array of packets
499  * \param numpkts number of packets to match
500  * \param sigs array of char* pointing to signatures to load
501  * \param numsigs number of signatures to load and check
502  * \param results pointer to arrays of numbers, each of them foreach packet
503  * to check if sids matches that packet as expected with
504  * that number of times or not. The size of results should be
505  * numpkts * numsigs * sizeof(uint16_t *)
506  *
507  * Example:
508  * result[1][3] would mean the number of times the pkt[1]
509  * match the sid[3]
510  *
511  * \retval int 1 if the match of all the sids is the specified has the
512  * specified results; 0 if not
513  */
514 int UTHGenericTest(Packet **pkt, int numpkts, char *sigs[], uint32_t sids[], uint32_t *results, int numsigs)
515 {
516 
517  int result = 0;
518  if (pkt == NULL || sigs == NULL || numpkts == 0
519  || sids == NULL || results == NULL || numsigs == 0) {
520  SCLogError(SC_ERR_INVALID_ARGUMENT, "Arguments invalid, that the pointer/arrays are not NULL, and the number of signatures and packets is > 0");
521  goto end;
522  }
523  DetectEngineCtx *de_ctx = DetectEngineCtxInit();
524  if (de_ctx == NULL) {
525  goto end;
526  }
527  de_ctx->flags |= DE_QUIET;
528 
529  if (UTHAppendSigs(de_ctx, sigs, numsigs) == 0)
530  goto cleanup;
531 
532  result = UTHMatchPacketsWithResults(de_ctx, pkt, numpkts, sids, results, numsigs);
533 
534 cleanup:
535  if (de_ctx != NULL) {
536  SigGroupCleanup(de_ctx);
537  SigCleanSignatures(de_ctx);
538  DetectEngineCtxFree(de_ctx);
539  }
540 end:
541  return result;
542 }

871  */
872 int CheckUTHTestPacket(Packet *p, uint8_t ipproto)
873 {
874  uint16_t sport = 41424;
875  uint16_t dport = 80;
876  uint8_t payload[] = "Payload";
877 
878  uint8_t len = sizeof(payload);
879 
880  if (p == NULL)
881  return 0;
882 
883  if (p->payload_len != len)
884  return 0;
885 
886  if (strncmp((char *)payload, (char *)p->payload, len) != 0)
887  return 0;
888 
889  if (p->src.family != AF_INET)
890  return 0;
891  if (p->dst.family != AF_INET)
892  return 0;
893  if (p->proto != ipproto)
894  return 0;
895 
896  switch(ipproto) {
897  case IPPROTO_UDP:
898  if (p->udph == NULL)
899  return 0;
900  if (p->udph->uh_sport != sport)
901  return 0;
902  if (p->udph->uh_dport != dport)
903  return 0;
904  break;
905  case IPPROTO_TCP:
906  if (p->tcph == NULL)
907  return 0;
908  if (ntohs(p->tcph->th_sport) != sport)
909  return 0;
910  if (ntohs(p->tcph->th_dport) != dport)
911  return 0;
912  break;
913  }
914  return 1;
915 }
