70 int LogStatsLogger(ThreadVars *tv, void *thread_data, const StatsTable *st)
71 {
72  SCEnter();
73  LogStatsLogThread *aft = (LogStatsLogThread *)thread_data;
74 
75  struct timeval tval;
76  struct tm *tms;
77 
78  gettimeofday(&tval, NULL);
79  struct tm local_tm;
80  tms = SCLocalTime(tval.tv_sec, &local_tm);
81 
82  /* Calculate the Engine uptime */
83  int up_time = (int)difftime(tval.tv_sec, st->start_time);
84  int sec = up_time % 60; // Seconds in a minute
85  int in_min = up_time / 60;
86  int min = in_min % 60; // Minutes in a hour
87  int in_hours = in_min / 60;
88  int hours = in_hours % 24; // Hours in a day
89  int days = in_hours / 24;
90 
91  MemBufferWriteString(aft->buffer, "----------------------------------------------"
92  "---------------------\n");
93  MemBufferWriteString(aft->buffer, "Date: %" PRId32 "/%" PRId32 "/%04d -- "
94  "%02d:%02d:%02d (uptime: %"PRId32"d, %02dh %02dm %02ds)\n",
95  tms->tm_mon + 1, tms->tm_mday, tms->tm_year + 1900, tms->tm_hour,
96  tms->tm_min, tms->tm_sec, days, hours, min, sec);
97  MemBufferWriteString(aft->buffer, "----------------------------------------------"
98  "---------------------\n");
99  MemBufferWriteString(aft->buffer, "%-25s | %-25s | %-s\n", "Counter", "TM Name",
100  "Value");
101  MemBufferWriteString(aft->buffer, "----------------------------------------------"
102  "---------------------\n");
103 
104  /* global stats */
105  uint32_t u = 0;
106  if (aft->statslog_ctx->flags & LOG_STATS_TOTALS) {
107  for (u = 0; u < st->nstats; u++) {
108  if (st->stats[u].name == NULL)
109  continue;
110 
111  char line[1024];
112  size_t len = snprintf(line, sizeof(line), "%-25s | %-25s | %-" PRIu64 "\n",
113  st->stats[u].name, st->stats[u].tm_name, st->stats[u].value);
114 
115  /* since we can have many threads, the buffer might not be big enough.
116  * Expand if necessary. */
117  if (MEMBUFFER_OFFSET(aft->buffer) + len > MEMBUFFER_SIZE(aft->buffer)) {
118  MemBufferExpand(&aft->buffer, OUTPUT_BUFFER_SIZE);
119  }
120 
121  MemBufferWriteString(aft->buffer, "%s", line);
122  }
123  }
124 
125  /* per thread stats */
126  if (st->tstats != NULL && aft->statslog_ctx->flags & LOG_STATS_THREADS) {
127  /* for each thread (store) */
128  uint32_t x;
129  for (x = 0; x < st->ntstats; x++) {
130  uint32_t offset = x * st->nstats;
131 
132  /* for each counter */
133  for (u = offset; u < (offset + st->nstats); u++) {
134  if (st->tstats[u].name == NULL)
135  continue;
136 
137  char line[1024];
138  size_t len = snprintf(line, sizeof(line), "%-25s | %-25s | %-" PRIu64 "\n",
139  st->tstats[u].name, st->tstats[u].tm_name, st->tstats[u].value);
140 
141  /* since we can have many threads, the buffer might not be big enough.
142  * Expand if necessary. */
143  if (MEMBUFFER_OFFSET(aft->buffer) + len > MEMBUFFER_SIZE(aft->buffer)) {
144  MemBufferExpand(&aft->buffer, OUTPUT_BUFFER_SIZE);
145  }
146 
147  MemBufferWriteString(aft->buffer, "%s", line);
148  }
149  }
150  }
151 
152  SCMutexLock(&aft->statslog_ctx->file_ctx->fp_mutex);
153  aft->statslog_ctx->file_ctx->Write((const char *)MEMBUFFER_BUFFER(aft->buffer),
154  MEMBUFFER_OFFSET(aft->buffer), aft->statslog_ctx->file_ctx);
155  SCMutexUnlock(&aft->statslog_ctx->file_ctx->fp_mutex);
156 
157  MemBufferReset(aft->buffer);
158 
159  SCReturnInt(0);
160 }

162 TmEcode LogStatsLogThreadInit(ThreadVars *t, void *initdata, void **data)
163 {
164  LogStatsLogThread *aft = SCMalloc(sizeof(LogStatsLogThread));
165  if (unlikely(aft == NULL))
166  return TM_ECODE_FAILED;
167  memset(aft, 0, sizeof(LogStatsLogThread));
168 
169  if(initdata == NULL)
170  {
171  SCLogDebug("Error getting context for HTTPLog. \"initdata\" argument NULL");
172  SCFree(aft);
173  return TM_ECODE_FAILED;
174  }
175 
176  aft->buffer = MemBufferCreateNew(OUTPUT_BUFFER_SIZE);
177  if (aft->buffer == NULL) {
178  SCFree(aft);
179  return TM_ECODE_FAILED;
180  }
181 
182  /* Use the Ouptut Context (file pointer and mutex) */
183  aft->statslog_ctx= ((OutputCtx *)initdata)->data;
184 
185  *data = (void *)aft;
186  return TM_ECODE_OK;
187 }

212 /** \brief Create a new http log LogFileCtx.
213  * \param conf Pointer to ConfNode containing this loggers configuration.
214  * \return NULL if failure, LogFileCtx* to the file_ctx if succesful
215  * */
216 OutputCtx *LogStatsLogInitCtx(ConfNode *conf)
217 {
218  LogFileCtx *file_ctx = LogFileNewCtx();
219  if (file_ctx == NULL) {
220  SCLogError(SC_ERR_HTTP_LOG_GENERIC, "couldn't create new file_ctx");
221  return NULL;
222  }
223 
224  if (SCConfLogOpenGeneric(conf, file_ctx, DEFAULT_LOG_FILENAME, 1) < 0) {
225  LogFileFreeCtx(file_ctx);
226  return NULL;
227  }
228 
229  LogStatsFileCtx *statslog_ctx = SCMalloc(sizeof(LogStatsFileCtx));
230  if (unlikely(statslog_ctx == NULL)) {
231  LogFileFreeCtx(file_ctx);
232  return NULL;
233  }
234  memset(statslog_ctx, 0x00, sizeof(LogStatsFileCtx));
235 
236  statslog_ctx->flags = LOG_STATS_TOTALS;
237 
238  if (conf != NULL) {
239  const char *totals = ConfNodeLookupChildValue(conf, "totals");
240  const char *threads = ConfNodeLookupChildValue(conf, "threads");
241  SCLogDebug("totals %s threads %s", totals, threads);
242 
243  if (totals != NULL && ConfValIsFalse(totals)) {
244  statslog_ctx->flags &= ~LOG_STATS_TOTALS;
245  }
246  if (threads != NULL && ConfValIsTrue(threads)) {
247  statslog_ctx->flags |= LOG_STATS_THREADS;
248  }
249  SCLogDebug("statslog_ctx->flags %08x", statslog_ctx->flags);
250  }
251 
252  statslog_ctx->file_ctx = file_ctx;
253 
254  OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
255  if (unlikely(output_ctx == NULL)) {
256  LogFileFreeCtx(file_ctx);
257  SCFree(statslog_ctx);
258  return NULL;
259  }
260 
261  output_ctx->data = statslog_ctx;
262  output_ctx->DeInit = LogStatsLogDeInitCtx;
263 
264  SCLogDebug("STATS log output initialized");
265 
266  return output_ctx;
267 }
