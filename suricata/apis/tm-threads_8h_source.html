137 /**
138  * \brief Process the rest of the functions (if any) and queue.
139  */
140 static inline TmEcode TmThreadsSlotProcessPkt(ThreadVars *tv, TmSlot *s, Packet *p)
141 {
142  TmEcode r = TM_ECODE_OK;
143 
144  if (s == NULL) {
145  tv->tmqh_out(tv, p);
146  return r;
147  }
148 
149  if (TmThreadsSlotVarRun(tv, p, s) == TM_ECODE_FAILED) {
150  TmqhOutputPacketpool(tv, p);
151  TmSlot *slot = s;
152  while (slot != NULL) {
153  SCMutexLock(&slot->slot_post_pq.mutex_q);
154  TmqhReleasePacketsToPacketPool(&slot->slot_post_pq);
155  SCMutexUnlock(&slot->slot_post_pq.mutex_q);
156 
157  slot = slot->slot_next;
158  }
159  TmThreadsSetFlag(tv, THV_FAILED);
160  r = TM_ECODE_FAILED;
161 
162  } else {
163  tv->tmqh_out(tv, p);
164 
165  /* post process pq */
166  TmSlot *slot = s;
167  while (slot != NULL) {
168  if (slot->slot_post_pq.top != NULL) {
169  while (1) {
170  SCMutexLock(&slot->slot_post_pq.mutex_q);
171  Packet *extra_p = PacketDequeue(&slot->slot_post_pq);
172  SCMutexUnlock(&slot->slot_post_pq.mutex_q);
173 
174  if (extra_p == NULL)
175  break;
176 
177  if (slot->slot_next != NULL) {
178  r = TmThreadsSlotVarRun(tv, extra_p, slot->slot_next);
179  if (r == TM_ECODE_FAILED) {
180  SCMutexLock(&slot->slot_post_pq.mutex_q);
181  TmqhReleasePacketsToPacketPool(&slot->slot_post_pq);
182  SCMutexUnlock(&slot->slot_post_pq.mutex_q);
183 
184  TmqhOutputPacketpool(tv, extra_p);
185  TmThreadsSetFlag(tv, THV_FAILED);
186  break;
187  }
188  }
189  tv->tmqh_out(tv, extra_p);
190  }
191  } /* if (slot->slot_post_pq.top != NULL) */
192  slot = slot->slot_next;
193  } /* while (slot != NULL) */
194  }
195 
196  return r;
197 }
