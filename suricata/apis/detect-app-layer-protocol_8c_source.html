39 {

49 static DetectAppLayerProtocolData *DetectAppLayerProtocolParse(const char *arg)
50 {
51  DetectAppLayerProtocolData *data;
52  AppProto alproto = ALPROTO_UNKNOWN;
53  uint8_t negated = 0;
54 
55  if (arg == NULL) {
56  SCLogError(SC_ERR_INVALID_SIGNATURE, "app-layer-protocol keyword "
57  "supplied with no arguments. This keyword needs "
58  "an argument.");
59  return NULL;
60  }
61 
62  while (*arg != '\0' && isspace((unsigned char)*arg))
63  arg++;
64 
65  if (arg[0] == '!') {
66  negated = 1;
67  arg++;
68  }
69 
70  while (*arg != '\0' && isspace((unsigned char)*arg))
71  arg++;
72 
73  alproto = AppLayerGetProtoByName((char *)arg);
74  if (alproto == ALPROTO_UNKNOWN) {
75  SCLogError(SC_ERR_INVALID_SIGNATURE, "app-layer-protocol "
76  "keyword supplied with unknown protocol \"%s\"", arg);
77  return NULL;
78  }
79 
80  data = SCMalloc(sizeof(DetectAppLayerProtocolData));
81  if (unlikely(data == NULL))
82  return NULL;
83  data->alproto = alproto;
84  data->negated = negated;
85 
86  return data;
87 }

89 int DetectAppLayerProtocolSetup(DetectEngineCtx *de_ctx, Signature *s,
90  char *arg)
91 {
92  DetectAppLayerProtocolData *data = NULL;
93  SigMatch *sm = NULL;
94 
95  if (s->alproto != ALPROTO_UNKNOWN) {
96  SCLogError(SC_ERR_CONFLICTING_RULE_KEYWORDS, "Either we already "
97  "have the rule match on an app layer protocol set through "
98  "other keywords that match on this protocol, or have "
99  "already seen a non-negated app-layer-protocol.");
100  goto error;
101  }
102 
103  data = DetectAppLayerProtocolParse(arg);
104  if (data == NULL)
105  goto error;
106 
107  if (!data->negated)
108  s->alproto = data->alproto;
109 
110  sm = SigMatchAlloc();
111  if (sm == NULL)
112  goto error;
113 
114  sm->type = DETECT_AL_APP_LAYER_PROTOCOL;
115  sm->ctx = (void *)data;
116 
117  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_AMATCH);
118  s->flags |= SIG_FLAG_APPLAYER;
119 
120  return 0;
121 
122 error:
123  if (data != NULL)
124  SCFree(data);
125  return -1;
126 }
