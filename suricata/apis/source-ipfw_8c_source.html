227 TmEcode ReceiveIPFWLoop(ThreadVars *tv, void *data, void *slot)
228 {
229  SCEnter();
230 
231  IPFWThreadVars *ptv = (IPFWThreadVars *)data;
232  IPFWQueueVars *nq = NULL;
233  uint8_t pkt[IP_MAXPACKET];
234  int pktlen=0;
235  struct pollfd IPFWpoll;
236  struct timeval IPFWts;
237  Packet *p = NULL;
238 
239  nq = IPFWGetQueue(ptv->ipfw_index);
240  if (nq == NULL) {
241  SCLogWarning(SC_ERR_INVALID_ARGUMENT, "Can't get thread variable");
242  SCReturnInt(TM_ECODE_FAILED);
243  }
244 
245  SCLogInfo("Thread '%s' will run on port %d (item %d)",
246  tv->name, nq->port_num, ptv->ipfw_index);
247  while (1) {
248  if (unlikely(suricata_ctl_flags != 0)) {
249  SCReturnInt(TM_ECODE_OK);
250  }
251 
252  IPFWpoll.fd = nq->fd;
253  IPFWpoll.events = POLLRDNORM;
254  /* Poll the socket for status */
255  if ( (poll(&IPFWpoll, 1, IPFW_SOCKET_POLL_MSEC)) > 0) {
256  if (!(IPFWpoll.revents & (POLLRDNORM | POLLERR)))
257  continue;
258  }
259 
260  if ((pktlen = recvfrom(nq->fd, pkt, sizeof(pkt), 0,
261  (struct sockaddr *)&nq->ipfw_sin,
262  &nq->ipfw_sinlen)) == -1) {
263  /* We received an error on socket read */
264  if (errno == EINTR || errno == EWOULDBLOCK) {
265  /* Nothing for us to process */
266  continue;
267  } else {
268  SCLogWarning(SC_WARN_IPFW_RECV,
269  "Read from IPFW divert socket failed: %s",
270  strerror(errno));
271  SCReturnInt(TM_ECODE_FAILED);
272  }
273  }
274  /* We have a packet to process */
275  memset (&IPFWts, 0, sizeof(struct timeval));
276  gettimeofday(&IPFWts, NULL);
277 
278  /* make sure we have at least one packet in the packet pool, to prevent
279  * us from alloc'ing packets at line rate */
280  PacketPoolWait();
281 
282  p = PacketGetFromQueueOrAlloc();
283  if (p == NULL) {
284  SCReturnInt(TM_ECODE_FAILED);
285  }
286  PKT_SET_SRC(p, PKT_SRC_WIRE);
287 
288  SCLogDebug("Received Packet Len: %d", pktlen);
289 
290  p->ts.tv_sec = IPFWts.tv_sec;
291  p->ts.tv_usec = IPFWts.tv_usec;
292 
293  ptv->pkts++;
294  ptv->bytes += pktlen;
295 
296  p->datalink = ptv->datalink;
297 
298  p->ipfw_v.ipfw_index = ptv->ipfw_index;
299 
300  PacketCopyData(p, pkt, pktlen);
301  SCLogDebug("Packet info: pkt_len: %" PRIu32 " (pkt %02x, pkt_data %02x)",
302  GET_PKT_LEN(p), *pkt, GET_PKT_DATA(p));
303 
304  if (TmThreadsSlotProcessPkt(tv, ((TmSlot *) slot)->slot_next, p)
305  != TM_ECODE_OK) {
306  TmqhOutputPacketpool(tv, p);
307  SCReturnInt(TM_ECODE_FAILED);
308  }
309 
310  StatsSyncCountersIfSignalled(tv);
311  }
312 
313  SCReturnInt(TM_ECODE_OK);
314 }

316 /**
317  * \brief Init function for RecieveIPFW.
318  *
319  * This is a setup function for recieving packets
320  * via ipfw divert, binds a socket, and prepares to
321  * to read from it.
322  *
323  * \param tv pointer to ThreadVars
324  * \param initdata pointer to the divert port passed from the user
325  * \param data pointer gets populated with IPFWThreadVars
326  *
327  */
328 TmEcode ReceiveIPFWThreadInit(ThreadVars *tv, void *initdata, void **data)
329 {
330  struct timeval timev;
331  int flag;
332  IPFWThreadVars *ntv = (IPFWThreadVars *) initdata;
333  IPFWQueueVars *nq = IPFWGetQueue(ntv->ipfw_index);
334 
335  sigset_t sigs;
336  sigfillset(&sigs);
337  pthread_sigmask(SIG_UNBLOCK, &sigs, NULL);
338 
339  SCEnter();
340 
341  IPFWMutexInit(nq);
342  /* We need a divert socket to play with */
343  if ((nq->fd = socket(PF_INET, SOCK_RAW, IPPROTO_DIVERT)) == -1) {
344  SCLogError(SC_ERR_IPFW_SOCK,"Can't create divert socket: %s", strerror(errno));
345  SCReturnInt(TM_ECODE_FAILED);
346  }
347 
348  /* set a timeout to the socket so we can check for a signal
349  * in case we don't get packets for a longer period. */
350  timev.tv_sec = 1;
351  timev.tv_usec = 0;
352 
353  if (setsockopt(nq->fd, SOL_SOCKET, SO_RCVTIMEO, &timev, sizeof(timev)) == -1) {
354  SCLogError(SC_ERR_IPFW_SETSOCKOPT,"Can't set IPFW divert socket timeout: %s", strerror(errno));
355  SCReturnInt(TM_ECODE_FAILED);
356  }
357 
358  /* set SO_BROADCAST on the divert socket, otherwise sendto()
359  * returns EACCES when reinjecting broadcast packets. */
360  flag = 1;
361 
362  if (setsockopt(nq->fd, SOL_SOCKET, SO_BROADCAST, &flag, sizeof(flag)) == -1) {
363  SCLogError(SC_ERR_IPFW_SETSOCKOPT,"Can't set IPFW divert socket broadcast flag: %s", strerror(errno));
364  SCReturnInt(TM_ECODE_FAILED);
365  }
366 
367  nq->ipfw_sinlen=sizeof(nq->ipfw_sin);
368  memset(&nq->ipfw_sin, 0, nq->ipfw_sinlen);
369  nq->ipfw_sin.sin_family = PF_INET;
370  nq->ipfw_sin.sin_addr.s_addr = INADDR_ANY;
371  nq->ipfw_sin.sin_port = htons(nq->port_num);
372 
373  /* Bind that SOB */
374  if (bind(nq->fd, (struct sockaddr *)&nq->ipfw_sin, nq->ipfw_sinlen) == -1) {
375  SCLogError(SC_ERR_IPFW_BIND,"Can't bind divert socket on port %d: %s",nq->port_num,strerror(errno));
376  SCReturnInt(TM_ECODE_FAILED);
377  }
378 
379  ntv->datalink = DLT_RAW;
380 
381  *data = (void *)ntv;
382 
383  SCReturnInt(TM_ECODE_OK);
384 }

430  * \brief This function passes off to link type decoders.
431  * \todo Unit tests are needed for this module.
432  *
433  * DecodeIPFW reads packets from the PacketQueue and passes
434  * them off to the proper link type decoder.
435  *

441 TmEcode DecodeIPFW(ThreadVars *tv, Packet *p, void *data, PacketQueue *pq, PacketQueue *postpq)
442 {
443  IPV4Hdr *ip4h = (IPV4Hdr *)GET_PKT_DATA(p);
444  IPV6Hdr *ip6h = (IPV6Hdr *)GET_PKT_DATA(p);
445  DecodeThreadVars *dtv = (DecodeThreadVars *)data;
446 
447  SCEnter();
448 
449  /* XXX HACK: flow timeout can call us for injected pseudo packets
450  * see bug: https://redmine.openinfosecfoundation.org/issues/1107 */
451  if (p->flags & PKT_PSEUDO_STREAM_END)
452  return TM_ECODE_OK;
453 
454  /* update counters */
455  DecodeUpdatePacketCounters(tv, dtv, p);
456 
457  /* Process IP packets */
458  if (IPV4_GET_RAW_VER(ip4h) == 4) {
459  SCLogDebug("DecodeIPFW ip4 processing");
460  DecodeIPV4(tv, dtv, p, GET_PKT_DATA(p), GET_PKT_LEN(p), pq);
461 
462  } else if(IPV6_GET_RAW_VER(ip6h) == 6) {
463  SCLogDebug("DecodeIPFW ip6 processing");
464  DecodeIPV6(tv, dtv, p, GET_PKT_DATA(p), GET_PKT_LEN(p), pq);
465 
466  } else {
467  /* We don't support anything besides IP packets for now, bridged packets? */
468  SCLogInfo("IPFW unknown protocol support %02x", *GET_PKT_DATA(p));
469  SCReturnInt(TM_ECODE_FAILED);
470  }
471 
472  PacketDecodeFinalize(tv, dtv, p);
473 
474  SCReturnInt(TM_ECODE_OK);
475 }

513  */
514 TmEcode IPFWSetVerdict(ThreadVars *tv, IPFWThreadVars *ptv, Packet *p)
515 {
516  uint32_t verdict;
517  struct pollfd IPFWpoll;
518  IPFWQueueVars *nq = NULL;
519 
520  SCEnter();
521 
522  if (p == NULL) {
523  SCLogWarning(SC_ERR_INVALID_ARGUMENT, "Packet is NULL");
524  SCReturnInt(TM_ECODE_FAILED);
525  }
526 
527  nq = IPFWGetQueue(p->ipfw_v.ipfw_index);
528  if (nq == NULL) {
529  SCLogWarning(SC_ERR_INVALID_ARGUMENT, "No thread found");
530  SCReturnInt(TM_ECODE_FAILED);
531  }
532 
533  IPFWpoll.fd = nq->fd;
534  IPFWpoll.events = POLLWRNORM;
535 
536  if (PACKET_TEST_ACTION(p, ACTION_DROP)) {
537  verdict = IPFW_DROP;
538  } else {
539  verdict = IPFW_ACCEPT;
540  }
541 
542  if (verdict == IPFW_ACCEPT) {
543  SCLogDebug("IPFW Verdict is to Accept");
544  ptv->accepted++;
545 
546  /* For divert sockets, accepting means writing the
547  * packet back to the socket for ipfw to pick up
548  */
549  SCLogDebug("IPFWSetVerdict writing to socket %d, %p, %u", nq->fd, GET_PKT_DATA(p),GET_PKT_LEN(p));
550 
551 #if 0
552  while ((poll(&IPFWpoll,1,IPFW_SOCKET_POLL_MSEC)) < 1) {
553  /* Did we receive a signal to shutdown */
554  if (TmThreadsCheckFlag(tv, THV_KILL) || TmThreadsCheckFlag(tv, THV_PAUSE)) {
555  SCLogInfo("Received ThreadShutdown: IPFW divert socket writing interrupted");
556  SCReturnInt(TM_ECODE_OK);
557  }
558  }
559 #endif
560 
561  IPFWMutexLock(nq);
562  if (sendto(nq->fd, GET_PKT_DATA(p), GET_PKT_LEN(p), 0,(struct sockaddr *)&nq->ipfw_sin, nq->ipfw_sinlen) == -1) {
563  int r = errno;
564  switch (r) {
565  default:
566  SCLogWarning(SC_WARN_IPFW_XMIT,"Write to ipfw divert socket failed: %s",strerror(r));
567  IPFWMutexUnlock(nq);
568  SCReturnInt(TM_ECODE_FAILED);
569  case EHOSTDOWN:
570  case ENETDOWN:
571  break;
572  }
573  }
574 
575  IPFWMutexUnlock(nq);
576 
577  SCLogDebug("Sent Packet back into IPFW Len: %d",GET_PKT_LEN(p));
578 
579  } /* end IPFW_ACCEPT */
580 
581 
582  if (verdict == IPFW_DROP) {
583  SCLogDebug("IPFW SetVerdict is to DROP");
584  ptv->dropped++;
585

595 /**
596  * \brief This function handles the Verdict processing
597  * \todo Unit tests are needed for this module.
598  *
599  *
600  * \param tv pointer to ThreadVars
601  * \param p pointer to the Packet
602  * \param data pointer that gets cast into IPFWThreadVars for ptv
603  * \param pq pointer for the Packet Queue access (Not used)
604  */
605 TmEcode VerdictIPFW(ThreadVars *tv, Packet *p, void *data, PacketQueue *pq, PacketQueue *postpq)
606 {
607  IPFWThreadVars *ptv = (IPFWThreadVars *)data;
608  TmEcode retval = TM_ECODE_OK;
609 
610  SCEnter();
611 
612  /* can't verdict a "fake" packet */
613  if (p->flags & PKT_PSEUDO_STREAM_END) {
614  SCReturnInt(TM_ECODE_OK);
615  }
616 
617  /* This came from NFQ.
618  * if this is a tunnel packet we check if we are ready to verdict
619  * already. */
620  if (IS_TUNNEL_PKT(p)) {
621  char verdict = 1;
622 
623  SCMutex *m = p->root ? &p->root->tunnel_mutex : &p->tunnel_mutex;
624  SCMutexLock(m);
625 
626  /* if there are more tunnel packets than ready to verdict packets,
627  * we won't verdict this one
628  */
629  if (TUNNEL_PKT_TPR(p) > TUNNEL_PKT_RTV(p)) {
630  SCLogDebug("VerdictIPFW: not ready to verdict yet: "
631  "TUNNEL_PKT_TPR(p) > TUNNEL_PKT_RTV(p) = %" PRId32
632  " > %" PRId32 "", TUNNEL_PKT_TPR(p), TUNNEL_PKT_RTV(p));
633  verdict = 0;
634  }
635 
636  SCMutexUnlock(m);
637 
638  /* don't verdict if we are not ready */
639  if (verdict == 1) {
640  SCLogDebug("Setting verdict on tunnel");
641  retval = IPFWSetVerdict(tv, ptv, p->root ? p->root : p);
642 
643  } else {
644  TUNNEL_INCR_PKT_RTV(p);
645  }
646  } else {
647  /* no tunnel, verdict normally */
648  SCLogDebug("Setting verdict on non-tunnel");
649  retval = IPFWSetVerdict(tv, ptv, p);
650  } /* IS_TUNNEL_PKT end */
651 
652  SCReturnInt(retval);
653 }

713  * \brief Add an IPFW divert
714  *
715  * \param string with the queue name
716  *
717  * \retval 0 on success.
718  * \retval -1 on failure.
719  */
720 int IPFWRegisterQueue(char *queue)
721 {
722  IPFWThreadVars *ntv = NULL;
723  IPFWQueueVars *nq = NULL;
724  /* Extract the queue number from the specified command line argument */
725  uint16_t port_num = 0;
726  if ((ByteExtractStringUint16(&port_num, 10, strlen(queue), queue)) < 0)
727  {
728  SCLogError(SC_ERR_INVALID_ARGUMENT, "specified queue number %s is not "
729  "valid", queue);
730  return -1;
731  }
732 
733  SCMutexLock(&ipfw_init_lock);
734  if (receive_port_num >= IPFW_MAX_QUEUE) {
735  SCLogError(SC_ERR_INVALID_ARGUMENT,
736  "too much IPFW divert port registered (%d)",
737  receive_port_num);
738  SCMutexUnlock(&ipfw_init_lock);
739  return -1;
740  }
741  if (receive_port_num == 0) {
742  memset(&ipfw_t, 0, sizeof(ipfw_t));
743  memset(&ipfw_q, 0, sizeof(ipfw_q));
744  }
745 
746  ntv = &ipfw_t[receive_port_num];
747  ntv->ipfw_index = receive_port_num;
748 
749  nq = &ipfw_q[receive_port_num];
750  nq->port_num = port_num;
751  receive_port_num++;
752  SCMutexUnlock(&ipfw_init_lock);
753  LiveRegisterDevice(queue);
754 
755  SCLogDebug("Queue \"%s\" registered.", queue);
756  return 0;
757 }
