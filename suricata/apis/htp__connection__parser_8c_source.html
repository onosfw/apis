47 void htp_connp_close(htp_connp_t *connp, const htp_time_t *timestamp) {
48  if (connp == NULL) return;
49 
50  // Close the underlying connection.
51  htp_conn_close(connp->conn, timestamp);
52 
53  // Update internal flags
54  if (connp->in_status != HTP_STREAM_ERROR)
55  connp->in_status = HTP_STREAM_CLOSED;
56  if (connp->out_status != HTP_STREAM_ERROR)
57  connp->out_status = HTP_STREAM_CLOSED;
58 
59  // Call the parsers one last time, which will allow them
60  // to process the events that depend on stream closure
61  htp_connp_req_data(connp, timestamp, NULL, 0);
62  htp_connp_res_data(connp, timestamp, NULL, 0);
63 }

65 htp_connp_t *htp_connp_create(htp_cfg_t *cfg) {
66  htp_connp_t *connp = calloc(1, sizeof (htp_connp_t));
67  if (connp == NULL) return NULL;
68 
69  // Use the supplied configuration structure
70  connp->cfg = cfg;
71 
72  // Create a new connection.
73  connp->conn = htp_conn_create();
74  if (connp->conn == NULL) {
75  free(connp);
76  return NULL;

78 
79  // Request parsing
80  connp->in_state = htp_connp_REQ_IDLE;
81  connp->in_status = HTP_STREAM_NEW;
82 
83  // Response parsing
84  connp->out_state = htp_connp_RES_IDLE;
85  connp->out_status = HTP_STREAM_NEW;
86 
87  return connp;
88 }

157 void htp_connp_open(htp_connp_t *connp, const char *client_addr, int client_port, const char *server_addr,
158  int server_port, htp_time_t *timestamp)
159 {
160  if (connp == NULL) return;
161 
162  // Check connection parser state first.
163  if ((connp->in_status != HTP_STREAM_NEW) || (connp->out_status != HTP_STREAM_NEW)) {
164  htp_log(connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0, "Connection is already open");
165  return;
166  }
167 
168  if (htp_conn_open(connp->conn, client_addr, client_port, server_addr, server_port, timestamp) != HTP_OK) {
169  return;
170  }
171 
172  connp->in_status = HTP_STREAM_OPEN;
173  connp->out_status = HTP_STREAM_OPEN;
174 }

181 htp_tx_t *htp_connp_tx_create(htp_connp_t *connp) {
182  if (connp == NULL) return NULL;
183 
184  // Detect pipelining.
185  if (htp_list_size(connp->conn->transactions) > connp->out_next_tx_index) {
186  connp->conn->flags |= HTP_CONN_PIPELINED;
187  }
188 
189  htp_tx_t *tx = htp_tx_create(connp);
190  if (tx == NULL) return NULL;
191 
192  connp->in_tx = tx;
193 
194  htp_connp_in_reset(connp);
195 
196  return tx;
197 }
