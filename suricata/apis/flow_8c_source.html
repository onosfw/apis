105 /** \brief Make sure we have enough spare flows.
106  *
107  * Enforce the prealloc parameter, so keep at least prealloc flows in the
108  * spare queue and free flows going over the limit.
109  *
110  * \retval 1 if the queue was properly updated (or if it already was in good shape)
111  * \retval 0 otherwise.
112  */
113 int FlowUpdateSpareFlows(void)
114 {
115  SCEnter();
116  uint32_t toalloc = 0, tofree = 0, len;
117 
118  FQLOCK_LOCK(&flow_spare_q);
119  len = flow_spare_q.len;
120  FQLOCK_UNLOCK(&flow_spare_q);
121 
122  if (len < flow_config.prealloc) {
123  toalloc = flow_config.prealloc - len;
124 
125  uint32_t i;
126  for (i = 0; i < toalloc; i++) {
127  Flow *f = FlowAlloc();
128  if (f == NULL)
129  return 0;
130 
131  FlowEnqueue(&flow_spare_q,f);
132  }
133  } else if (len > flow_config.prealloc) {
134  tofree = len - flow_config.prealloc;
135 
136  uint32_t i;
137  for (i = 0; i < tofree; i++) {
138  /* FlowDequeue locks the queue */
139  Flow *f = FlowDequeue(&flow_spare_q);
140  if (f == NULL)
141  return 1;
142 
143  FlowFree(f);
144  }
145  }
146 
147  return 1;
148 }

182 {

187  return TOSERVER;
188  } else {

190  }
191  } else {

193  return TOSERVER;
194  } else {
195  return TOCLIENT;
196  }
197  }

200  return TOSERVER;
201  } else {
202  return TOCLIENT;
203  }
204  }

260 /** \brief Update Packet and Flow
261  *
262  * Updates packet and flow based on the new packet.
263  *
264  * \param f locked flow
265  * \param p packet
266  *
267  * \note overwrites p::flowflags
268  */
269 void FlowHandlePacketUpdate(Flow *f, Packet *p)
270 {
271  SCLogDebug("packet %"PRIu64" -- flow %p", p->pcap_cnt, f);
272 
273  /* Point the Packet at the Flow */
274  FlowReference(&p->flow, f);
275 
276  /* update flags and counters */
277  if (FlowGetPacketDirection(f, p) == TOSERVER) {
278  f->todstpktcnt++;
279  f->todstbytecnt += GET_PKT_LEN(p);
280  p->flowflags = FLOW_PKT_TOSERVER;
281  if (!(f->flags & FLOW_TO_DST_SEEN)) {
282  if (FlowUpdateSeenFlag(p)) {
283  f->flags |= FLOW_TO_DST_SEEN;
284  p->flowflags |= FLOW_PKT_TOSERVER_FIRST;
285  }
286  }
287  } else {
288  f->tosrcpktcnt++;
289  f->tosrcbytecnt += GET_PKT_LEN(p);
290  p->flowflags = FLOW_PKT_TOCLIENT;
291  if (!(f->flags & FLOW_TO_SRC_SEEN)) {
292  if (FlowUpdateSeenFlag(p)) {
293  f->flags |= FLOW_TO_SRC_SEEN;
294  p->flowflags |= FLOW_PKT_TOCLIENT_FIRST;
295  }
296  }
297  }
298 
299  if ((f->flags & (FLOW_TO_DST_SEEN|FLOW_TO_SRC_SEEN)) == (FLOW_TO_DST_SEEN|FLOW_TO_SRC_SEEN)) {
300  SCLogDebug("pkt %p FLOW_PKT_ESTABLISHED", p);
301  p->flowflags |= FLOW_PKT_ESTABLISHED;
302 
303  if (f->proto != IPPROTO_TCP) {
304  SC_ATOMIC_SET(f->flow_state, FLOW_STATE_ESTABLISHED);
305  }
306  }
307 
308  /*set the detection bypass flags*/
309  if (f->flags & FLOW_NOPACKET_INSPECTION) {
310  SCLogDebug("setting FLOW_NOPACKET_INSPECTION flag on flow %p", f);
311  DecodeSetNoPacketInspectionFlag(p);
312  }
313  if (f->flags & FLOW_NOPAYLOAD_INSPECTION) {
314  SCLogDebug("setting FLOW_NOPAYLOAD_INSPECTION flag on flow %p", f);
315  DecodeSetNoPayloadInspectionFlag(p);
316  }
317 }

346  * \warning Not thread safe */
347 void FlowInitConfig(char quiet)
348 {
349  SCLogDebug("initializing flow engine...");
350 
351  memset(&flow_config, 0, sizeof(flow_config));
352  SC_ATOMIC_INIT(flow_flags);
353  SC_ATOMIC_INIT(flow_memuse);
354  SC_ATOMIC_INIT(flow_prune_idx);
355  FlowQueueInit(&flow_spare_q);
356  FlowQueueInit(&flow_recycle_q);
357 
358  unsigned int seed = RandomTimePreseed();
359  /* set defaults */
360  flow_config.hash_rand = (int)( FLOW_DEFAULT_HASHSIZE * (rand_r(&seed) / RAND_MAX + 1.0));
361 
362  flow_config.hash_size = FLOW_DEFAULT_HASHSIZE;
363  flow_config.memcap = FLOW_DEFAULT_MEMCAP;
364  flow_config.prealloc = FLOW_DEFAULT_PREALLOC;
365 
366  /* If we have specific config, overwrite the defaults with them,
367  * otherwise, leave the default values */
368  intmax_t val = 0;
369  if (ConfGetInt("flow.emergency-recovery", &val) == 1) {
370  if (val <= 100 && val >= 1) {
371  flow_config.emergency_recovery = (uint8_t)val;
372  } else {
373  SCLogError(SC_ERR_INVALID_VALUE, "flow.emergency-recovery must be in the range of 1 and 100 (as percentage)");
374  flow_config.emergency_recovery = FLOW_DEFAULT_EMERGENCY_RECOVERY;
375  }
376  } else {
377  SCLogDebug("flow.emergency-recovery, using default value");
378  flow_config.emergency_recovery = FLOW_DEFAULT_EMERGENCY_RECOVERY;
379  }
380 
381  /* Check if we have memcap and hash_size defined at config */
382  char *conf_val;
383  uint32_t configval = 0;
384 
385  /** set config values for memcap, prealloc and hash_size */
386  if ((ConfGet("flow.memcap", &conf_val)) == 1)
387  {
388  if (ParseSizeStringU64(conf_val, &flow_config.memcap) < 0) {
389  SCLogError(SC_ERR_SIZE_PARSE, "Error parsing flow.memcap "
390  "from conf file - %s. Killing engine",
391  conf_val);
392  exit(EXIT_FAILURE);
393  }
394  }
395  if ((ConfGet("flow.hash-size", &conf_val)) == 1)
396  {
397  if (ByteExtractStringUint32(&configval, 10, strlen(conf_val),
398  conf_val) > 0) {
399  flow_config.hash_size = configval;
400  }
401  }
402  if ((ConfGet("flow.prealloc", &conf_val)) == 1)
403  {
404  if (ByteExtractStringUint32(&configval, 10, strlen(conf_val),
405  conf_val) > 0) {
406  flow_config.prealloc = configval;
407  }
408  }
409  SCLogDebug("Flow config from suricata.yaml: memcap: %"PRIu64", hash-size: "
410  "%"PRIu32", prealloc: %"PRIu32, flow_config.memcap,
411  flow_config.hash_size, flow_config.prealloc);
412 
413  /* alloc hash memory */
414  uint64_t hash_size = flow_config.hash_size * sizeof(FlowBucket);
415  if (!(FLOW_CHECK_MEMCAP(hash_size))) {
416  SCLogError(SC_ERR_FLOW_INIT, "allocating flow hash failed: "
417  "max flow memcap is smaller than projected hash size. "
418  "Memcap: %"PRIu64", Hash table size %"PRIu64". Calculate "
419  "total hash size by multiplying \"flow.hash-size\" with %"PRIuMAX", "
420  "which is the hash bucket size.", flow_config.memcap, hash_size,
421  (uintmax_t)sizeof(FlowBucket));
422  exit(EXIT_FAILURE);
423  }
424  flow_hash = SCCalloc(flow_config.hash_size, sizeof(FlowBucket));
425  if (unlikely(flow_hash == NULL)) {
426  SCLogError(SC_ERR_FATAL, "Fatal error encountered in FlowInitConfig. Exiting...");
427  exit(EXIT_FAILURE);
428  }
429  memset(flow_hash, 0, flow_config.hash_size * sizeof(FlowBucket));
430 
431  uint32_t i = 0;
432  for (i = 0; i < flow_config.hash_size; i++) {
433  FBLOCK_INIT(&flow_hash[i]);
434  }
435  (void) SC_ATOMIC_ADD(flow_memuse, (flow_config.hash_size * sizeof(FlowBucket)));
436 
437  if (quiet == FALSE) {
438  SCLogInfo("allocated %llu bytes of memory for the flow hash... "
439  "%" PRIu32 " buckets of size %" PRIuMAX "",
440  SC_ATOMIC_GET(flow_memuse), flow_config.hash_size,
441  (uintmax_t)sizeof(FlowBucket));
442  }
443 
444  /* pre allocate flows */
445  for (i = 0; i < flow_config.prealloc; i++) {
446  if (!(FLOW_CHECK_MEMCAP(sizeof(Flow) + FlowStorageSize()))) {
447  SCLogError(SC_ERR_FLOW_INIT, "preallocating flows failed: "
448  "max flow memcap reached. Memcap %"PRIu64", "
449  "Memuse %"PRIu64".", flow_config.memcap,
450  ((uint64_t)SC_ATOMIC_GET(flow_memuse) + (uint64_t)sizeof(Flow)));
451  exit(EXIT_FAILURE);
452  }
453 
454  Flow *f = FlowAlloc();
455  if (f == NULL) {
456  SCLogError(SC_ERR_FLOW_INIT, "preallocating flow failed: %s", strerror(errno));
457  exit(EXIT_FAILURE);
458  }
459 
460  FlowEnqueue(&flow_spare_q,f);
461  }
462 
463  if (quiet == FALSE) {
464  SCLogInfo("preallocated %" PRIu32 " flows of size %" PRIuMAX "",
465  flow_spare_q.len, (uintmax_t)(sizeof(Flow) + + FlowStorageSize()));
466  SCLogInfo("flow memory usage: %llu bytes, maximum: %"PRIu64,
467  SC_ATOMIC_GET(flow_memuse), flow_config.memcap);
468  }
469 
470  FlowInitFlowProto();
471 
472  return;
473 }

486 /** \brief shutdown the flow engine
487  * \warning Not thread safe */
488 void FlowShutdown(void)
489 {
490  Flow *f;
491  uint32_t u;
492 
493  FlowPrintStats();
494 
495  /* free queues */
496  while((f = FlowDequeue(&flow_spare_q))) {
497  FlowFree(f);
498  }
499  while((f = FlowDequeue(&flow_recycle_q))) {
500  FlowFree(f);
501  }
502 
503  /* clear and free the hash */
504  if (flow_hash != NULL) {
505  /* clean up flow mutexes */
506  for (u = 0; u < flow_config.hash_size; u++) {
507  Flow *f = flow_hash[u].head;
508  while (f) {
509 #ifdef DEBUG_VALIDATION
510  BUG_ON(SC_ATOMIC_GET(f->use_cnt) != 0);
511 #endif
512  Flow *n = f->hnext;
513  uint8_t proto_map = FlowGetProtoMapping(f->proto);
514  FlowClearMemory(f, proto_map);
515  FlowFree(f);
516  f = n;
517  }
518 
519  FBLOCK_DESTROY(&flow_hash[u]);
520  }
521  SCFree(flow_hash);
522  flow_hash = NULL;
523  }
524  (void) SC_ATOMIC_SUB(flow_memuse, flow_config.hash_size * sizeof(FlowBucket));
525  FlowQueueDestroy(&flow_spare_q);
526  FlowQueueDestroy(&flow_recycle_q);
527 
528  SC_ATOMIC_DESTROY(flow_prune_idx);
529  SC_ATOMIC_DESTROY(flow_memuse);
530  SC_ATOMIC_DESTROY(flow_flags);
531  return;
532 }
