42 /** \brief allocate a flow
43  *
44  * We check against the memuse counter. If it passes that check we increment
45  * the counter first, then we try to alloc.
46  *
47  * \retval f the flow or NULL on out of memory
48  */
49 Flow *FlowAlloc(void)
50 {
51  Flow *f;
52  size_t size = sizeof(Flow) + FlowStorageSize();
53 
54  if (!(FLOW_CHECK_MEMCAP(size))) {
55  return NULL;
56  }
57 
58  (void) SC_ATOMIC_ADD(flow_memuse, size);
59 
60  f = SCMalloc(size);
61  if (unlikely(f == NULL)) {
62  (void)SC_ATOMIC_SUB(flow_memuse, size);
63  return NULL;
64  }
65  memset(f, 0, size);
66 
67  FLOW_INITIALIZE(f);
68  return f;
69 }

124 /* initialize the flow from the first packet
125  * we see from it. */
126 void FlowInit(Flow *f, const Packet *p)
127 {
128  SCEnter();
129  SCLogDebug("flow %p", f);
130 
131  f->proto = p->proto;
132  f->recursion_level = p->recursion_level;
133  f->vlan_id[0] = p->vlan_id[0];
134  f->vlan_id[1] = p->vlan_id[1];
135 
136  if (PKT_IS_IPV4(p)) {
137  FLOW_SET_IPV4_SRC_ADDR_FROM_PACKET(p, &f->src);
138  FLOW_SET_IPV4_DST_ADDR_FROM_PACKET(p, &f->dst);
139  f->flags |= FLOW_IPV4;
140  } else if (PKT_IS_IPV6(p)) {
141  FLOW_SET_IPV6_SRC_ADDR_FROM_PACKET(p, &f->src);
142  FLOW_SET_IPV6_DST_ADDR_FROM_PACKET(p, &f->dst);
143  f->flags |= FLOW_IPV6;
144  }
145 #ifdef DEBUG
146  /* XXX handle default */
147  else {
148  printf("FIXME: %s:%s:%" PRId32 "\n", __FILE__, __FUNCTION__, __LINE__);
149  }
150 #endif
151 
152  if (p->tcph != NULL) { /* XXX MACRO */
153  SET_TCP_SRC_PORT(p,&f->sp);
154  SET_TCP_DST_PORT(p,&f->dp);
155  } else if (p->udph != NULL) { /* XXX MACRO */
156  SET_UDP_SRC_PORT(p,&f->sp);
157  SET_UDP_DST_PORT(p,&f->dp);
158  } else if (p->icmpv4h != NULL) {
159  f->type = p->type;
160  f->code = p->code;
161  } else if (p->icmpv6h != NULL) {
162  f->type = p->type;
163  f->code = p->code;
164  } else if (p->sctph != NULL) { /* XXX MACRO */
165  SET_SCTP_SRC_PORT(p,&f->sp);
166  SET_SCTP_DST_PORT(p,&f->dp);
167  } /* XXX handle default */
168 #ifdef DEBUG
169  else {
170  printf("FIXME: %s:%s:%" PRId32 "\n", __FILE__, __FUNCTION__, __LINE__);
171  }
172 #endif
173  COPY_TIMESTAMP(&p->ts, &f->startts);
174 
175  f->protomap = FlowGetProtoMapping(f->proto);
176 
177  SCReturn;
178 }
