81  * \internal
82  * \brief Write meta data on a single line json record
83  */
84 static void FileWriteJsonRecord(JsonFileLogThread *aft, const Packet *p, const File *ff)
85 {
86  MemBuffer *buffer = (MemBuffer *)aft->buffer;
87  json_t *js = CreateJSONHeader((Packet *)p, 0, "fileinfo"); //TODO const
88  json_t *hjs = NULL;
89  if (unlikely(js == NULL))
90  return;
91 
92  /* reset */
93  MemBufferReset(buffer);
94 
95  switch (p->flow->alproto) {
96  case ALPROTO_HTTP:
97  hjs = JsonHttpAddMetadata(p->flow, ff->txid);
98  if (hjs)
99  json_object_set_new(js, "http", hjs);
100  break;
101  case ALPROTO_SMTP:
102  hjs = JsonSMTPAddMetadata(p->flow, ff->txid);
103  if (hjs)
104  json_object_set_new(js, "smtp", hjs);
105  hjs = JsonEmailAddMetadata(p->flow, ff->txid);
106  if (hjs)
107  json_object_set_new(js, "email", hjs);
108  break;
109  }
110 
111 
112  json_t *fjs = json_object();
113  if (unlikely(fjs == NULL)) {
114  json_decref(js);
115  return;
116  }
117 
118  char *s = BytesToString(ff->name, ff->name_len);
119  json_object_set_new(fjs, "filename", json_string(s));
120  if (s != NULL)
121  SCFree(s);
122  if (ff->magic)
123  json_object_set_new(fjs, "magic", json_string((char *)ff->magic));
124  switch (ff->state) {
125  case FILE_STATE_CLOSED:
126  json_object_set_new(fjs, "state", json_string("CLOSED"));
127 #ifdef HAVE_NSS
128  if (ff->flags & FILE_MD5) {
129  size_t x;
130  int i;
131  char s[256];
132  for (i = 0, x = 0; x < sizeof(ff->md5); x++) {
133  i += snprintf(&s[i], 255-i, "%02x", ff->md5[x]);
134  }
135  json_object_set_new(fjs, "md5", json_string(s));
136  }
137 #endif
138  break;
139  case FILE_STATE_TRUNCATED:
140  json_object_set_new(fjs, "state", json_string("TRUNCATED"));
141  break;
142  case FILE_STATE_ERROR:
143  json_object_set_new(fjs, "state", json_string("ERROR"));
144  break;
145  default:
146  json_object_set_new(fjs, "state", json_string("UNKNOWN"));
147  break;
148  }
149  json_object_set_new(fjs, "stored",
150  (ff->flags & FILE_STORED) ? json_true() : json_false());
151  if (ff->flags & FILE_STORED) {
152  json_object_set_new(fjs, "file_id", json_integer(ff->file_id));
153  }
154  json_object_set_new(fjs, "size", json_integer(ff->size));
155  json_object_set_new(fjs, "tx_id", json_integer(ff->txid));
156 
157  /* originally just 'file', but due to bug 1127 naming it fileinfo */
158  json_object_set_new(js, "fileinfo", fjs);
159  OutputJSONBuffer(js, aft->filelog_ctx->file_ctx, buffer);
160  json_object_del(js, "fileinfo");
161 
162  switch (p->flow->alproto) {
163  case ALPROTO_HTTP:
164  json_object_del(js, "http");
165  break;
166  case ALPROTO_SMTP:
167  json_object_del(js, "smtp");
168  json_object_del(js, "email");
169  break;
170  }
171 
172  json_object_clear(js);
173  json_decref(js);
174 }

176 static int JsonFileLogger(ThreadVars *tv, void *thread_data, const Packet *p, const File *ff)
177 {
178  SCEnter();
179  JsonFileLogThread *aft = (JsonFileLogThread *)thread_data;
180 
181  BUG_ON(ff->flags & FILE_LOGGED);
182 
183  SCLogDebug("ff %p", ff);
184 
185  FileWriteJsonRecord(aft, p, ff);

191 static TmEcode JsonFileLogThreadInit(ThreadVars *t, void *initdata, void **data)
192 {
193  JsonFileLogThread *aft = SCMalloc(sizeof(JsonFileLogThread));
194  if (unlikely(aft == NULL))
195  return TM_ECODE_FAILED;
196  memset(aft, 0, sizeof(JsonFileLogThread));
197 
198  if(initdata == NULL)
199  {
200  SCLogDebug("Error getting context for HTTPLog. \"initdata\" argument NULL");
201  SCFree(aft);
202  return TM_ECODE_FAILED;
203  }
204 
205  /* Use the Ouptut Context (file pointer and mutex) */
206  aft->filelog_ctx = ((OutputCtx *)initdata)->data;
207 
208  aft->buffer = MemBufferCreateNew(OUTPUT_BUFFER_SIZE);
209  if (aft->buffer == NULL) {
210  SCFree(aft);
211  return TM_ECODE_FAILED;
212  }
213 
214  *data = (void *)aft;
215  return TM_ECODE_OK;
216 }

240 /** \brief Create a new http log LogFileCtx.
241  * \param conf Pointer to ConfNode containing this loggers configuration.
242  * \return NULL if failure, LogFileCtx* to the file_ctx if succesful
243  * */
244 OutputCtx *OutputFileLogInitSub(ConfNode *conf, OutputCtx *parent_ctx)
245 {
246  OutputJsonCtx *ojc = parent_ctx->data;
247 
248  OutputFileCtx *output_file_ctx = SCMalloc(sizeof(OutputFileCtx));
249  if (unlikely(output_file_ctx == NULL))
250  return NULL;
251 
252  OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
253  if (unlikely(output_ctx == NULL)) {
254  SCFree(output_file_ctx);
255  return NULL;
256  }
257 
258  output_file_ctx->file_ctx = ojc->file_ctx;
259 
260  if (conf) {
261  const char *force_magic = ConfNodeLookupChildValue(conf, "force-magic");
262  if (force_magic != NULL && ConfValIsTrue(force_magic)) {
263  FileForceMagicEnable();
264  SCLogInfo("forcing magic lookup for logged files");
265  }
266 
267  const char *force_md5 = ConfNodeLookupChildValue(conf, "force-md5");
268  if (force_md5 != NULL && ConfValIsTrue(force_md5)) {
269 #ifdef HAVE_NSS
270  FileForceMd5Enable();
271  SCLogInfo("forcing md5 calculation for logged files");
272 #else
273  SCLogInfo("md5 calculation requires linking against libnss");
274 #endif
275  }
276  }
277 
278  output_ctx->data = output_file_ctx;
279  output_ctx->DeInit = OutputFileLogDeinitSub;
280 
281  FileForceTrackingEnable();
282  return output_ctx;
283 }
