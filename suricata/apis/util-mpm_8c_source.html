53  * \brief Register a new Mpm Context.
54  *
55  * \param name A new profile to be registered to store this MpmCtx.
56  *
57  * \retval id Return the id created for the new MpmCtx profile.

59 int32_t MpmFactoryRegisterMpmCtxProfile(DetectEngineCtx *de_ctx, const char *name, uint8_t flags)
60 {
61  void *ptmp;
62  /* the very first entry */
63  if (de_ctx->mpm_ctx_factory_container == NULL) {
64  de_ctx->mpm_ctx_factory_container = SCMalloc(sizeof(MpmCtxFactoryContainer));
65  if (de_ctx->mpm_ctx_factory_container == NULL) {
66  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");
67  exit(EXIT_FAILURE);
68  }
69  memset(de_ctx->mpm_ctx_factory_container, 0, sizeof(MpmCtxFactoryContainer));
70 
71  MpmCtxFactoryItem *item = SCMalloc(sizeof(MpmCtxFactoryItem));
72  if (unlikely(item == NULL)) {
73  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");
74  exit(EXIT_FAILURE);
75  }
76 
77  item[0].name = SCStrdup(name);
78  if (item[0].name == NULL) {
79  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");
80  exit(EXIT_FAILURE);
81  }
82 
83  /* toserver */
84  item[0].mpm_ctx_ts = SCMalloc(sizeof(MpmCtx));
85  if (item[0].mpm_ctx_ts == NULL) {
86  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");
87  exit(EXIT_FAILURE);
88  }
89  memset(item[0].mpm_ctx_ts, 0, sizeof(MpmCtx));
90  item[0].mpm_ctx_ts->global = 1;
91 
92  /* toclient */
93  item[0].mpm_ctx_tc = SCMalloc(sizeof(MpmCtx));
94  if (item[0].mpm_ctx_tc == NULL) {
95  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");
96  exit(EXIT_FAILURE);
97  }
98  memset(item[0].mpm_ctx_tc, 0, sizeof(MpmCtx));
99  item[0].mpm_ctx_tc->global = 1;
100 
101  /* our id starts from 0 always. Helps us with the ctx retrieval from
102  * the array */
103  item[0].id = 0;
104 
105  /* store the flag */
106  item[0].flags = flags;
107 
108  /* store the newly created item */
109  de_ctx->mpm_ctx_factory_container->items = item;
110  de_ctx->mpm_ctx_factory_container->no_of_items++;
111 
112  /* the first id is always 0 */
113  return item[0].id;
114  } else {
115  int i;
116  MpmCtxFactoryItem *items = de_ctx->mpm_ctx_factory_container->items;
117  for (i = 0; i < de_ctx->mpm_ctx_factory_container->no_of_items; i++) {
118  if (items[i].name != NULL && strcmp(items[i].name, name) == 0) {
119  /* looks like we have this mpm_ctx freed */
120  if (items[i].mpm_ctx_ts == NULL) {
121  items[i].mpm_ctx_ts = SCMalloc(sizeof(MpmCtx));
122  if (items[i].mpm_ctx_ts == NULL) {
123  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");
124  exit(EXIT_FAILURE);
125  }
126  memset(items[i].mpm_ctx_ts, 0, sizeof(MpmCtx));
127  items[i].mpm_ctx_ts->global = 1;
128  }
129  if (items[i].mpm_ctx_tc == NULL) {
130  items[i].mpm_ctx_tc = SCMalloc(sizeof(MpmCtx));
131  if (items[i].mpm_ctx_tc == NULL) {
132  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");
133  exit(EXIT_FAILURE);
134  }
135  memset(items[i].mpm_ctx_tc, 0, sizeof(MpmCtx));
136  items[i].mpm_ctx_tc->global = 1;
137  }
138  items[i].flags = flags;
139  return items[i].id;
140  }
141  }
142 
143  /* let's make the new entry */
144  ptmp = SCRealloc(items,
145  (de_ctx->mpm_ctx_factory_container->no_of_items + 1) * sizeof(MpmCtxFactoryItem));
146  if (unlikely(ptmp == NULL)) {
147  SCFree(items);
148  items = NULL;
149  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");
150  exit(EXIT_FAILURE);
151  }
152  items = ptmp;
153 
154  de_ctx->mpm_ctx_factory_container->items = items;
155 
156  MpmCtxFactoryItem *new_item = &items[de_ctx->mpm_ctx_factory_container->no_of_items];
157  new_item[0].name = SCStrdup(name);
158  if (new_item[0].name == NULL) {
159  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");
160  exit(EXIT_FAILURE);
161  }
162 
163  /* toserver */
164  new_item[0].mpm_ctx_ts = SCMalloc(sizeof(MpmCtx));
165  if (new_item[0].mpm_ctx_ts == NULL) {
166  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");
167  exit(EXIT_FAILURE);
168  }
169  memset(new_item[0].mpm_ctx_ts, 0, sizeof(MpmCtx));
170  new_item[0].mpm_ctx_ts->global = 1;
171 
172  /* toclient */
173  new_item[0].mpm_ctx_tc = SCMalloc(sizeof(MpmCtx));
174  if (new_item[0].mpm_ctx_tc == NULL) {
175  SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");
176  exit(EXIT_FAILURE);
177  }
178  memset(new_item[0].mpm_ctx_tc, 0, sizeof(MpmCtx));
179  new_item[0].mpm_ctx_tc->global = 1;
180 
181  new_item[0].id = de_ctx->mpm_ctx_factory_container->no_of_items;
182  new_item[0].flags = flags;
183  de_ctx->mpm_ctx_factory_container->no_of_items++;
184 
185  /* the newly created id */
186  return new_item[0].id;
187  }
188 }

190 int32_t MpmFactoryIsMpmCtxAvailable(DetectEngineCtx *de_ctx, MpmCtx *mpm_ctx)
191 {
192  if (mpm_ctx == NULL)
193  return 0;
194 
195  if (de_ctx->mpm_ctx_factory_container == NULL) {
196  return 0;
197  } else {
198  int i;
199  for (i = 0; i < de_ctx->mpm_ctx_factory_container->no_of_items; i++) {
200  if (mpm_ctx == de_ctx->mpm_ctx_factory_container->items[i].mpm_ctx_ts ||
201  mpm_ctx == de_ctx->mpm_ctx_factory_container->items[i].mpm_ctx_tc) {
202  return 1;
203  }
204  }
205  return 0;
206  }

246 void MpmFactoryDeRegisterAllMpmCtxProfiles(DetectEngineCtx *de_ctx)
247 {
248  if (de_ctx->mpm_ctx_factory_container == NULL)
249  return;
250 
251  int i = 0;
252  MpmCtxFactoryItem *items = de_ctx->mpm_ctx_factory_container->items;
253  for (i = 0; i < de_ctx->mpm_ctx_factory_container->no_of_items; i++) {
254  if (items[i].name != NULL)
255  SCFree(items[i].name);
256  if (items[i].mpm_ctx_ts != NULL) {
257  if (items[i].mpm_ctx_ts->mpm_type != MPM_NOTSET)
258  mpm_table[items[i].mpm_ctx_ts->mpm_type].DestroyCtx(items[i].mpm_ctx_ts);
259  SCFree(items[i].mpm_ctx_ts);
260  }
261  if (items[i].mpm_ctx_tc != NULL) {
262  if (items[i].mpm_ctx_tc->mpm_type != MPM_NOTSET)
263  mpm_table[items[i].mpm_ctx_tc->mpm_type].DestroyCtx(items[i].mpm_ctx_tc);
264  SCFree(items[i].mpm_ctx_tc);
265  }
266  }
267 
268  SCFree(de_ctx->mpm_ctx_factory_container->items);
269  SCFree(de_ctx->mpm_ctx_factory_container);
270  de_ctx->mpm_ctx_factory_container = NULL;
271 
272  return;

411 /**
412  * \brief Setup a pmq
413  *
414  * \param pmq Pattern matcher queue to be initialized
415  * \param maxid Max sig id to be matched on
416  * \param patmaxid Max pattern id to be matched on
417  *
418  * \retval -1 error
419  * \retval 0 ok
420  */
421 int PmqSetup(PatternMatcherQueue *pmq, uint32_t patmaxid)
422 {
423  SCEnter();
424  SCLogDebug("patmaxid %u", patmaxid);
425 
426  if (pmq == NULL) {
427  SCReturnInt(-1);
428  }
429 
430  memset(pmq, 0, sizeof(PatternMatcherQueue));
431 
432  if (patmaxid > 0) {
433  pmq->pattern_id_array_size = 32; /* Intial size, TODO Make this configure option */
434  pmq->pattern_id_array_cnt = 0;
435 
436  pmq->pattern_id_array = SCCalloc(pmq->pattern_id_array_size, sizeof(uint32_t));
437  if (pmq->pattern_id_array == NULL) {
438  SCReturnInt(-1);
439  }
440 
441  /* lookup bitarray */
442  pmq->pattern_id_bitarray_size = (patmaxid / 8) + 1;
443 
444  pmq->pattern_id_bitarray = SCMalloc(pmq->pattern_id_bitarray_size);
445  if (pmq->pattern_id_bitarray == NULL) {
446  SCReturnInt(-1);
447  }
448  memset(pmq->pattern_id_bitarray, 0, pmq->pattern_id_bitarray_size);
449 
450  SCLogDebug("pmq->pattern_id_array %p, pmq->pattern_id_bitarray %p",
451  pmq->pattern_id_array, pmq->pattern_id_bitarray);
452 
453  pmq->rule_id_array_size = 128; /* Initial size, TODO: Make configure option. */
454  pmq->rule_id_array_cnt = 0;
455 
456  size_t bytes = pmq->rule_id_array_size * sizeof(SigIntId);
457  pmq->rule_id_array = (SigIntId*)SCMalloc(bytes);
458  if (pmq->rule_id_array == NULL) {
459  pmq->rule_id_array_size = 0;
460  SCReturnInt(-1);
461  }
462  // Don't need to zero memory since it is always written first.
463  }
464 
465  SCReturnInt(0);
466 }

477 MpmAddSidsResize(PatternMatcherQueue *pmq, uint32_t new_size)

538 /** \brief Verify and store a match
539  *
540  * used at search runtime
541  *
542  * \param thread_ctx mpm thread ctx
543  * \param pmq storage for match results
544  * \param patid pattern ID being checked
545  * \param bitarray Array of bits for patterns IDs found in current search
546  * \param sids pointer to array of Signature IDs
547  * \param sids_size number of Signature IDs in sids array.
548  *
549  * \retval 0 no match after all
550  * \retval 1 (new) match
551  */
552 int
553 MpmVerifyMatch(MpmThreadCtx *thread_ctx, PatternMatcherQueue *pmq, uint32_t patid,
554  uint8_t *bitarray, SigIntId *sids, uint32_t sids_size)
555 {
556  SCEnter();
557 
558  /* Handle pattern id storage */
559  if (pmq != NULL && pmq->pattern_id_bitarray != NULL) {
560  SCLogDebug("using pattern id arrays, storing %"PRIu32, patid);
561 
562  if ((bitarray[(patid / 8)] & (1<<(patid % 8))) == 0) {
563  bitarray[(patid / 8)] |= (1<<(patid % 8));
564  /* flag this pattern id as being added now */
565  pmq->pattern_id_bitarray[(patid / 8)] |= (1<<(patid % 8));
566  /* append the pattern_id to the array with matches */
567  MpmAddPid(pmq, patid);
568  MpmAddSids(pmq, sids, sids_size);
569  }
570  }
571 
572  SCReturnInt(1);
573 }

575 /**
576  * \brief Merge two pmq's bitarrays
577  *
578  * \param src source pmq
579  * \param dst destination pmq to merge into
580  */
581 void PmqMerge(PatternMatcherQueue *src, PatternMatcherQueue *dst)
582 {
583  uint32_t u;
584 
585  if (src->pattern_id_array_cnt == 0)
586  return;
587 
588  for (u = 0; u < src->pattern_id_bitarray_size && u < dst->pattern_id_bitarray_size; u++) {
589  dst->pattern_id_bitarray[u] |= src->pattern_id_bitarray[u];
590  }
591

682 /** \brief Function to return the default hash size for the mpm algorithm,
683  * which has been defined by the user in the config file

686  * \retval hash_value returns the hash value as defined by user, otherwise

690 {
691  SCEnter();
692  uint32_t hash_value = HASHSIZE_LOW;
693 
694  if(strcmp(conf_val, "lowest") == 0) {
695  hash_value = HASHSIZE_LOWEST;
696  } else if(strcmp(conf_val, "low") == 0) {
697  hash_value = HASHSIZE_LOW;
698  } else if(strcmp(conf_val, "medium") == 0) {
699  hash_value = HASHSIZE_MEDIUM;
700  } else if(strcmp(conf_val, "high") == 0) {
701  hash_value = HASHSIZE_HIGH;
702  /* "highest" is supported in 1.0 to 1.0.2, so we keep supporting
703  * it for backwards compatibility */
704  } else if(strcmp(conf_val, "highest") == 0) {
705  hash_value = HASHSIZE_HIGHER;
706  } else if(strcmp(conf_val, "higher") == 0) {
707  hash_value = HASHSIZE_HIGHER;
708  } else if(strcmp(conf_val, "max") == 0) {
709  hash_value = HASHSIZE_MAX;
710  }
711 
712  SCReturnInt(hash_value);
713 }
