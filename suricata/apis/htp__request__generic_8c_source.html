43 /**
44  * Extract one request header. A header can span multiple lines, in
45  * which case they will be folded into one before parsing is attempted.
46  *
47  * @param[in] connp
48  * @param[in] data
49  * @param[in] len
50  * @return HTP_OK or HTP_ERROR
51  */
52 htp_status_t htp_process_request_header_generic(htp_connp_t *connp, unsigned char *data, size_t len) {
53  // Create a new header structure.
54  htp_header_t *h = calloc(1, sizeof (htp_header_t));
55  if (h == NULL) return HTP_ERROR;
56 
57  // Now try to parse the header.
58  if (htp_parse_request_header_generic(connp, h, data, len) != HTP_OK) {
59  free(h);
60  return HTP_ERROR;
61  }
62 
63  #ifdef HTP_DEBUG
64  fprint_bstr(stderr, "Header name", h->name);
65  fprint_bstr(stderr, "Header value", h->value);
66  #endif
67 
68  // Do we already have a header with the same name?
69  htp_header_t *h_existing = htp_table_get(connp->in_tx->request_headers, h->name);
70  if (h_existing != NULL) {
71  // TODO Do we want to have a list of the headers that are
72  // allowed to be combined in this way?
73 
74  // Add to the existing header.
75  bstr *new_value = bstr_expand(h_existing->value, bstr_len(h_existing->value) + 2 + bstr_len(h->value));
76  if (new_value == NULL) {
77  bstr_free(h->name);
78  bstr_free(h->value);
79  free(h);
80  return HTP_ERROR;
81  }
82 
83  h_existing->value = new_value;
84  bstr_add_mem_noex(h_existing->value, ", ", 2);
85  bstr_add_noex(h_existing->value, h->value);
86 
87  // The new header structure is no longer needed.
88  bstr_free(h->name);
89  bstr_free(h->value);
90  free(h);
91 
92  // Keep track of repeated same-name headers.
93  h_existing->flags |= HTP_FIELD_REPEATED;
94  } else {
95  // Add as a new header.
96  htp_table_add(connp->in_tx->request_headers, h->name, h);
97  }
98 
99  return HTP_OK;
100 }

102 /**
103  * Generic request header parser.
104  *
105  * @param[in] connp
106  * @param[in] h
107  * @param[in] data
108  * @param[in] len
109  * @return HTP_OK or HTP_ERROR
110  */
111 htp_status_t htp_parse_request_header_generic(htp_connp_t *connp, htp_header_t *h, unsigned char *data, size_t len) {
112  size_t name_start, name_end;
113  size_t value_start, value_end;
114 
115  htp_chomp(data, &len);
116 
117  name_start = 0;
118 
119  // Look for the colon.
120  size_t colon_pos = 0;
121  while ((colon_pos < len) && (data[colon_pos] != '\0') && (data[colon_pos] != ':')) colon_pos++;
122 
123  if ((colon_pos == len) || (data[colon_pos] == '\0')) {
124  // Missing colon.
125 
126  h->flags |= HTP_FIELD_UNPARSEABLE;
127 
128  // Log only once per transaction.
129  if (!(connp->in_tx->flags & HTP_FIELD_UNPARSEABLE)) {
130  connp->in_tx->flags |= HTP_FIELD_UNPARSEABLE;
131  htp_log(connp, HTP_LOG_MARK, HTP_LOG_WARNING, 0, "Request field invalid: colon missing");
132  }
133 
134  // We handle this case as a header with an empty name, with the value equal
135  // to the entire input string.
136 
137  // TODO Apache will respond to this problem with a 400.
138 
139  // Now extract the name and the value
140  h->name = bstr_dup_c("");
141  if (h->name == NULL) return HTP_ERROR;
142 
143  h->value = bstr_dup_mem(data, len);
144  if (h->value == NULL) {
145  bstr_free(h->name);
146  return HTP_ERROR;
147  }
148 
149  return HTP_OK;
150  }
151 
152  if (colon_pos == 0) {
153  // Empty header name.
154 
155  h->flags |= HTP_FIELD_INVALID;
156 
157  // Log only once per transaction.
158  if (!(connp->in_tx->flags & HTP_FIELD_INVALID)) {
159  connp->in_tx->flags |= HTP_FIELD_INVALID;
160  htp_log(connp, HTP_LOG_MARK, HTP_LOG_WARNING, 0, "Request field invalid: empty name");
161  }
162  }
163 
164  name_end = colon_pos;
165 
166  // Ignore LWS after field-name.
167  size_t prev = name_end;
168  while ((prev > name_start) && (htp_is_lws(data[prev - 1]))) {
169  // LWS after header name.
170 
171  prev--;
172  name_end--;
173 
174  h->flags |= HTP_FIELD_INVALID;
175 
176  // Log only once per transaction.
177  if (!(connp->in_tx->flags & HTP_FIELD_INVALID)) {
178  connp->in_tx->flags |= HTP_FIELD_INVALID;
179  htp_log(connp, HTP_LOG_MARK, HTP_LOG_WARNING, 0, "Request field invalid: LWS after name");
180  }
181  }
182 
183  // Header value.
184 
185  value_start = colon_pos;
186 
187  // Go over the colon.
188  if (value_start < len) {
189  value_start++;
190  }
191 
192  // Ignore LWS before field-content.
193  while ((value_start < len) && (htp_is_lws(data[value_start]))) {
194  value_start++;
195  }
196 
197  // Look for the end of field-content.
198  value_end = value_start;
199  while ((value_end < len) && (data[value_end] != '\0')) value_end++;
200 
201  // Ignore LWS after field-content.
202  prev = value_end - 1;
203  while ((prev > value_start) && (htp_is_lws(data[prev]))) {
204  prev--;
205  value_end--;
206  }
207 
208  // Check that the header name is a token.
209  size_t i = name_start;
210  while (i < name_end) {
211  if (!htp_is_token(data[i])) {
212  // Incorrectly formed header name.
213 
214  h->flags |= HTP_FIELD_INVALID;
215 
216  // Log only once per transaction.
217  if (!(connp->in_tx->flags & HTP_FIELD_INVALID)) {
218  connp->in_tx->flags |= HTP_FIELD_INVALID;
219  htp_log(connp, HTP_LOG_MARK, HTP_LOG_WARNING, 0, "Request header name is not a token");
220  }
221 
222  break;
223  }
224 
225  i++;
226  }
227 
228  // Now extract the name and the value
229  h->name = bstr_dup_mem(data + name_start, name_end - name_start);
230  if (h->name == NULL) return HTP_ERROR;
231 
232  h->value = bstr_dup_mem(data + value_start, value_end - value_start);
233  if (h->value == NULL) {
234  bstr_free(h->name);
235  return HTP_ERROR;
236  }
237 
238  return HTP_OK;
239 }

251 htp_status_t htp_parse_request_line_generic_ex(htp_connp_t *connp, int nul_terminates) {
252  htp_tx_t *tx = connp->in_tx;
253  unsigned char *data = bstr_ptr(tx->request_line);
254  size_t len = bstr_len(tx->request_line);
255  size_t pos = 0;
256  size_t mstart = 0;
257  size_t start;
258  size_t bad_delim;
259 
260  if (nul_terminates) {
261  // The line ends with the first NUL byte.
262 
263  size_t newlen = 0;
264  while ((pos < len) && (data[pos] != '\0')) {
265  pos++;
266  newlen++;
267  }
268 
269  // Start again, with the new length.
270  len = newlen;
271  pos = 0;
272  }
273 
274  // skip past leading whitespace. IIS allows this
275  while ((pos < len) && htp_is_space(data[pos])) pos++;
276  if (pos) {
277  htp_log(connp, HTP_LOG_MARK, HTP_LOG_WARNING, 0, "Request line: leading whitespace");
278  mstart = pos;
279 
280  if (connp->cfg->requestline_leading_whitespace_unwanted != HTP_UNWANTED_IGNORE) {
281  // reset mstart so that we copy the whitespace into the method
282  mstart = 0;
283  // set expected response code to this anomaly
284  tx->response_status_expected_number = connp->cfg->requestline_leading_whitespace_unwanted;
285  }
286  }
287 
288  // The request method starts at the beginning of the
289  // line and ends with the first whitespace character.
290  while ((pos < len) && (!htp_is_space(data[pos]))) pos++;
291 
292  // No, we don't care if the method is empty.
293 
294  tx->request_method = bstr_dup_mem(data + mstart, pos - mstart);
295  if (tx->request_method == NULL) return HTP_ERROR;
296 
297  #ifdef HTP_DEBUG
298  fprint_raw_data(stderr, __FUNCTION__, bstr_ptr(tx->request_method), bstr_len(tx->request_method));
299  #endif
300 
301  tx->request_method_number = htp_convert_method_to_number(tx->request_method);
302 
303  bad_delim = 0;
304  // Ignore whitespace after request method. The RFC allows
305  // for only one SP, but then suggests any number of SP and HT
306  // should be permitted. Apache uses isspace(), which is even
307  // more permitting, so that's what we use here.
308  while ((pos < len) && (isspace(data[pos]))) {
309  if (!bad_delim && data[pos] != 0x20) {
310  bad_delim++;
311  }
312  pos++;
313  }
314  if (bad_delim) {
315  htp_log(connp, HTP_LOG_MARK, HTP_LOG_WARNING, 0, "Request line: non-compliant delimiter between Method and URI");
316  }
317 
318  // Is there anything after the request method?
319  if (pos == len) {
320  // No, this looks like a HTTP/0.9 request.
321 
322  tx->is_protocol_0_9 = 1;
323  tx->request_protocol_number = HTP_PROTOCOL_0_9;
324 
325  return HTP_OK;
326  }
327 
328  start = pos;
329  bad_delim = 0;
330 
331  // The URI ends with the first whitespace.
332  while ((pos < len) && (data[pos] != 0x20)) {
333  if (!bad_delim && htp_is_space(data[pos])) {
334  bad_delim++;
335  }
336  pos++;
337  }
338  /* if we've seen some 'bad' delimiters, we retry with those */
339  if (bad_delim && pos == len) {
340  // special case: even though RFC's allow only SP (0x20), many
341  // implementations allow other delimiters, like tab or other
342  // characters that isspace() accepts.
343  pos = start;
344  while ((pos < len) && (!htp_is_space(data[pos]))) pos++;
345  }
346  if (bad_delim) {
347  // warn regardless if we've seen non-compliant chars
348  htp_log(connp, HTP_LOG_MARK, HTP_LOG_WARNING, 0, "Request line: URI contains non-compliant delimiter");
349  }
350 
351  tx->request_uri = bstr_dup_mem(data + start, pos - start);
352  if (tx->request_uri == NULL) return HTP_ERROR;
353 
354  #ifdef HTP_DEBUG
355  fprint_raw_data(stderr, __FUNCTION__, bstr_ptr(tx->request_uri), bstr_len(tx->request_uri));
356  #endif
357 
358  // Ignore whitespace after URI.
359  while ((pos < len) && (htp_is_space(data[pos]))) pos++;
360 
361  // Is there protocol information available?
362  if (pos == len) {
363  // No, this looks like a HTTP/0.9 request.
364 
365  tx->is_protocol_0_9 = 1;
366  tx->request_protocol_number = HTP_PROTOCOL_0_9;
367 
368  return HTP_OK;
369  }
370 
371  // The protocol information continues until the end of the line.
372  tx->request_protocol = bstr_dup_mem(data + pos, len - pos);
373  if (tx->request_protocol == NULL) return HTP_ERROR;
374 
375  tx->request_protocol_number = htp_parse_protocol(tx->request_protocol);
376 
377  #ifdef HTP_DEBUG
378  fprint_raw_data(stderr, __FUNCTION__, bstr_ptr(tx->request_protocol), bstr_len(tx->request_protocol));
379  #endif
380 
381  return HTP_OK;
382 }
