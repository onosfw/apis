65 static inline int HCBDCreateSpace(DetectEngineThreadCtx *det_ctx, uint64_t size)
66 {
67  if (size >= (USHRT_MAX - BUFFER_STEP))
68  return -1;
69 
70  void *ptmp;
71  if (size > det_ctx->hcbd_buffers_size) {
72  ptmp = SCRealloc(det_ctx->hcbd,
73  (det_ctx->hcbd_buffers_size + BUFFER_STEP) * sizeof(HttpReassembledBody));
74  if (ptmp == NULL) {
75  SCFree(det_ctx->hcbd);
76  det_ctx->hcbd = NULL;
77  det_ctx->hcbd_buffers_size = 0;
78  det_ctx->hcbd_buffers_list_len = 0;
79  return -1;
80  }
81  det_ctx->hcbd = ptmp;
82 
83  memset(det_ctx->hcbd + det_ctx->hcbd_buffers_size, 0, BUFFER_STEP * sizeof(HttpReassembledBody));
84  det_ctx->hcbd_buffers_size += BUFFER_STEP;
85 
86  uint16_t i;
87  for (i = det_ctx->hcbd_buffers_list_len; i < ((uint16_t)size); i++) {
88  det_ctx->hcbd[i].buffer_len = 0;
89  det_ctx->hcbd[i].offset = 0;
90  }
91  }
92 
93  return 0;
94 }

98 static uint8_t *DetectEngineHCBDGetBufferForTX(htp_tx_t *tx, uint64_t tx_id,
99  DetectEngineCtx *de_ctx,
100  DetectEngineThreadCtx *det_ctx,
101  Flow *f, HtpState *htp_state,
102  uint8_t flags,
103  uint32_t *buffer_len,
104  uint32_t *stream_start_offset)
105 {
106  int index = 0;
107  uint8_t *buffer = NULL;
108  *buffer_len = 0;
109  *stream_start_offset = 0;
110 
111  if (det_ctx->hcbd_buffers_list_len == 0) {
112  /* get the inspect id to use as a 'base id' */
113  uint64_t base_inspect_id = AppLayerParserGetTransactionInspectId(f->alparser, flags);
114  BUG_ON(base_inspect_id > tx_id);
115  /* see how many space we need for the current tx_id */
116  uint64_t txs = (tx_id - base_inspect_id) + 1;
117  if (HCBDCreateSpace(det_ctx, txs) < 0)
118  goto end;
119 
120  index = (tx_id - base_inspect_id);
121  det_ctx->hcbd_start_tx_id = base_inspect_id;
122  det_ctx->hcbd_buffers_list_len = txs;
123  } else {
124  if ((tx_id - det_ctx->hcbd_start_tx_id) < det_ctx->hcbd_buffers_list_len) {
125  if (det_ctx->hcbd[(tx_id - det_ctx->hcbd_start_tx_id)].buffer_len != 0) {
126  *buffer_len = det_ctx->hcbd[(tx_id - det_ctx->hcbd_start_tx_id)].buffer_len;
127  *stream_start_offset = det_ctx->hcbd[(tx_id - det_ctx->hcbd_start_tx_id)].offset;
128  return det_ctx->hcbd[(tx_id - det_ctx->hcbd_start_tx_id)].buffer;
129  }
130  } else {
131  uint64_t txs = (tx_id - det_ctx->hcbd_start_tx_id) + 1;
132  if (HCBDCreateSpace(det_ctx, txs) < 0)
133  goto end; /* let's consider it as stage not done for now */
134 
135  det_ctx->hcbd_buffers_list_len = txs;
136  }
137  index = (tx_id - det_ctx->hcbd_start_tx_id);
138  }
139 
140  HtpTxUserData *htud = (HtpTxUserData *)htp_tx_get_user_data(tx);
141  if (htud == NULL) {
142  SCLogDebug("no htud");
143  goto end;
144  }
145 
146  /* no new data */
147  if (htud->request_body.body_inspected == htud->request_body.content_len_so_far) {
148  SCLogDebug("no new data");
149  goto end;
150  }
151 
152  HtpBodyChunk *cur = htud->request_body.first;
153  if (cur == NULL) {
154  SCLogDebug("No http chunks to inspect for this transacation");
155  goto end;
156  }
157 
158  /* inspect the body if the transfer is complete or we have hit
159  * our body size limit */
160  if ((htp_state->cfg->request_body_limit == 0 ||
161  htud->request_body.content_len_so_far < htp_state->cfg->request_body_limit) &&
162  htud->request_body.content_len_so_far < htp_state->cfg->request_inspect_min_size &&
163  !(AppLayerParserGetStateProgress(IPPROTO_TCP, ALPROTO_HTTP, tx, flags) > HTP_REQUEST_BODY) &&
164  !(flags & STREAM_EOF)) {
165  SCLogDebug("we still haven't seen the entire request body. "
166  "Let's defer body inspection till we see the "
167  "entire body.");
168  goto end;
169  }
170 
171  int first = 1;
172  while (cur != NULL) {
173  /* see if we can filter out chunks */
174  if (htud->request_body.body_inspected > 0) {
175  if (cur->stream_offset < htud->request_body.body_inspected) {
176  if ((htud->request_body.body_inspected - cur->stream_offset) > htp_state->cfg->request_inspect_min_size) {
177  cur = cur->next;
178  continue;
179  } else {
180  /* include this one */
181  }
182  } else {
183  /* include this one */
184  }
185  }
186 
187  if (first) {
188  det_ctx->hcbd[index].offset = cur->stream_offset;
189  first = 0;
190  }
191 
192  /* see if we need to grow the buffer */
193  if (det_ctx->hcbd[index].buffer == NULL || (det_ctx->hcbd[index].buffer_len + cur->len) > det_ctx->hcbd[index].buffer_size) {
194  void *ptmp;
195  det_ctx->hcbd[index].buffer_size += cur->len * 2;
196 
197  if ((ptmp = SCRealloc(det_ctx->hcbd[index].buffer, det_ctx->hcbd[index].buffer_size)) == NULL) {
198  SCFree(det_ctx->hcbd[index].buffer);
199  det_ctx->hcbd[index].buffer = NULL;
200  det_ctx->hcbd[index].buffer_size = 0;
201  det_ctx->hcbd[index].buffer_len = 0;
202  goto end;
203  }
204  det_ctx->hcbd[index].buffer = ptmp;
205  }
206  memcpy(det_ctx->hcbd[index].buffer + det_ctx->hcbd[index].buffer_len, cur->data, cur->len);
207  det_ctx->hcbd[index].buffer_len += cur->len;
208 
209  cur = cur->next;
210  }
211 
212  /* update inspected tracker */
213  htud->request_body.body_inspected = htud->request_body.last->stream_offset + htud->request_body.last->len;
214 
215  buffer = det_ctx->hcbd[index].buffer;
216  *buffer_len = det_ctx->hcbd[index].buffer_len;
217  *stream_start_offset = det_ctx->hcbd[index].offset;
218  end:
219  return buffer;
220 }

308 static int RunTest (struct TestSteps *steps, const char *sig, const char *yaml)
309 {
310  TcpSession ssn;
311  Flow f;
312  Packet *p = NULL;
313  ThreadVars th_v;
314  DetectEngineCtx *de_ctx = NULL;
315  DetectEngineThreadCtx *det_ctx = NULL;
316  int result = 0;
317  AppLayerParserThreadCtx *alp_tctx = AppLayerParserThreadCtxAlloc();
318  memset(&th_v, 0, sizeof(th_v));
319  memset(&f, 0, sizeof(f));
320  memset(&ssn, 0, sizeof(ssn));
321 
322  if (yaml) {
323  ConfCreateContextBackup();
324  ConfInit();
325  HtpConfigCreateBackup();
326 
327  ConfYamlLoadString(yaml, strlen(yaml));
328  HTPConfigure();
329  }
330 
331  StreamTcpInitConfig(TRUE);
332 
333  de_ctx = DetectEngineCtxInit();
334  if (de_ctx == NULL)
335  goto end;
336 
337  FLOW_INITIALIZE(&f);
338  f.protoctx = (void *)&ssn;
339  f.proto = IPPROTO_TCP;
340  f.flags |= FLOW_IPV4;
341  f.alproto = ALPROTO_HTTP;
342 
343  SCLogDebug("sig %s", sig);
344  DetectEngineAppendSig(de_ctx, (char *)sig);
345 
346  de_ctx->flags |= DE_QUIET;
347 
348  if (de_ctx->sig_list == NULL)
349  goto end;
350 
351  SigGroupBuild(de_ctx);
352  DetectEngineThreadCtxInit(&th_v, (void *)de_ctx, (void *)&det_ctx);
353 
354  struct TestSteps *b = steps;
355  int i = 0;
356  while (b->input != NULL) {
357  SCLogDebug("chunk %p %d", b, i);
358  p = UTHBuildPacket(NULL, 0, IPPROTO_TCP);
359  if (p == NULL)
360  goto end;
361  p->flow = &f;
362  p->flowflags = (b->direction == STREAM_TOSERVER) ? FLOW_PKT_TOSERVER : FLOW_PKT_TOCLIENT;
363  p->flowflags |= FLOW_PKT_ESTABLISHED;
364  p->flags |= PKT_HAS_FLOW|PKT_STREAM_EST;
365 
366  SCMutexLock(&f.m);
367  int r = AppLayerParserParse(alp_tctx, &f, ALPROTO_HTTP, b->direction,
368  (uint8_t *)b->input,
369  b->input_size ? b->input_size : strlen((const char *)b->input));
370  if (r != 0) {
371  printf("toserver chunk %d returned %" PRId32 ", expected 0: ", i+1, r);
372  result = 0;
373  SCMutexUnlock(&f.m);
374  goto end;
375  }
376  SCMutexUnlock(&f.m);
377 
378  /* do detect */
379  SigMatchSignatures(&th_v, de_ctx, det_ctx, p);
380 
381  int match = PacketAlertCheck(p, 1);
382  if (b->expect != match) {
383  printf("rule matching mismatch: ");
384  goto end;
385  }
386 
387  UTHFreePackets(&p, 1);
388  p = NULL;
389  b++;
390  i++;
391  }
392  result = 1;
393 
394  end:
395  if (alp_tctx != NULL)
396  AppLayerParserThreadCtxFree(alp_tctx);
397  if (de_ctx != NULL)
398  SigGroupCleanup(de_ctx);
399  if (de_ctx != NULL)
400  SigCleanSignatures(de_ctx);
401  if (de_ctx != NULL)
402  DetectEngineCtxFree(de_ctx);
403 
404  StreamTcpFreeConfig(TRUE);
405  FLOW_DESTROY(&f);
406  UTHFreePackets(&p, 1);
407 
408  if (yaml) {
409  HtpConfigRestoreBackup();
410  ConfRestoreContextBackup();
411  }
412  return result;
413 }
