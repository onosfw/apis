103  /* ioctl is not defined, let's pretend returning 0 is ok */
104  return 0;

108 /**
109  * \brief output max packet size for a link
110  *
111  * This does a best effort to find the maximum packet size
112  * for the link. In case of uncertainty, it will output a
113  * majorant to be sure avoid the cost of dynamic allocation.
114  *
115  * \param Name of a network interface
116  * \retval 0 in case of error
117  */
118 int GetIfaceMaxPacketSize(const char *pcap_dev)
119 {
120  int ll_header = GetIfaceMaxHWHeaderLength(pcap_dev);
121  int mtu = 0;
122 
123  if ((pcap_dev == NULL) || strlen(pcap_dev) == 0)
124  return 0;
125 
126  mtu = GetIfaceMTU(pcap_dev);
127  switch (mtu) {
128  case 0:
129  case -1:
130  return 0;
131  }
132  if (ll_header == -1) {
133  /* be conservative, choose a big one */
134  ll_header = 16;
135  }
136  return ll_header + mtu;
137 }

139 /**
140  * \brief output offloading status of the link
141  *
142  * Test interface for GRO and LRO features. If one of them is
143  * activated then suricata mays received packets merge at reception.
144  * The result is oversized packets and this may cause some serious
145  * problem in some capture mode where the size of the packet is
146  * limited (AF_PACKET in V2 more for example).
147  *
148  * ETHTOOL_GGRO ETH_FLAG_LRO
149  *
150  * \param Name of link
151  * \retval -1 in case of error, 0 if none, 1 if some
152  */
153 int GetIfaceOffloading(const char *pcap_dev)
154 {
155 #if defined (ETHTOOL_GGRO) && defined (ETHTOOL_GFLAGS)
156  struct ifreq ifr;
157  int fd;
158  struct ethtool_value ethv;
159  int ret = 0;
160 
161  fd = socket(AF_INET, SOCK_DGRAM, 0);
162  if (fd == -1) {
163  return -1;
164  }
165  (void)strlcpy(ifr.ifr_name, pcap_dev, sizeof(ifr.ifr_name));
166 
167  /* First get GRO */
168  ethv.cmd = ETHTOOL_GGRO;
169  ifr.ifr_data = (void *) &ethv;
170  if (ioctl(fd, SIOCETHTOOL, (char *)&ifr) < 0) {
171  SCLogWarning(SC_ERR_SYSCALL,
172  "Failure when trying to get feature via ioctl: %s (%d)",
173  strerror(errno), errno);
174  close(fd);
175  return -1;
176  } else {
177  if (ethv.data) {
178  SCLogInfo("Generic Receive Offload is set on %s", pcap_dev);
179  ret = 1;
180  } else {
181  SCLogInfo("Generic Receive Offload is unset on %s", pcap_dev);
182  }
183  }
184 
185  /* Then get LRO which is set in a flag */
186  ethv.data = 0;
187  ethv.cmd = ETHTOOL_GFLAGS;
188  ifr.ifr_data = (void *) &ethv;
189  if (ioctl(fd, SIOCETHTOOL, (char *)&ifr) < 0) {
190  SCLogWarning(SC_ERR_SYSCALL,
191  "Failure when trying to get feature via ioctl: %s (%d)",
192  strerror(errno), errno);
193  close(fd);
194  return -1;
195  } else {
196  if (ethv.data & ETH_FLAG_LRO) {
197  SCLogInfo("Large Receive Offload is set on %s", pcap_dev);
198  ret = 1;
199  } else {
200  SCLogInfo("Large Receive Offload is unset on %s", pcap_dev);
201  }
202  }
203 
204  close(fd);
205 
206  return ret;
207 #else
208  /* ioctl is not defined, let's pretend returning 0 is ok */
209  return 0;
210 #endif
211 }
