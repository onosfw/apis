149 {
150  if (flags & TH_SYN)

153  json_object_set_new(js, "fin", json_true());
154  if (flags & TH_RST)
155  json_object_set_new(js, "rst", json_true());
156  if (flags & TH_PUSH)
157  json_object_set_new(js, "psh", json_true());
158  if (flags & TH_ACK)
159  json_object_set_new(js, "ack", json_true());
160  if (flags & TH_URG)
161  json_object_set_new(js, "urg", json_true());
162  if (flags & TH_ECN)
163  json_object_set_new(js, "ecn", json_true());
164  if (flags & TH_CWR)

180 json_t *CreateJSONHeader(Packet *p, int direction_sensitive, char *event_type)
181 {
182  char timebuf[64];
183  char srcip[46], dstip[46];
184  Port sp, dp;
185 
186  json_t *js = json_object();
187  if (unlikely(js == NULL))
188  return NULL;
189 
190  CreateIsoTimeString(&p->ts, timebuf, sizeof(timebuf));
191 
192  srcip[0] = '\0';
193  dstip[0] = '\0';
194  if (direction_sensitive) {
195  if ((PKT_IS_TOSERVER(p))) {
196  if (PKT_IS_IPV4(p)) {
197  PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p), srcip, sizeof(srcip));
198  PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p), dstip, sizeof(dstip));
199  } else if (PKT_IS_IPV6(p)) {
200  PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p), srcip, sizeof(srcip));
201  PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p), dstip, sizeof(dstip));
202  }
203  sp = p->sp;
204  dp = p->dp;
205  } else {
206  if (PKT_IS_IPV4(p)) {
207  PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p), srcip, sizeof(srcip));
208  PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p), dstip, sizeof(dstip));
209  } else if (PKT_IS_IPV6(p)) {
210  PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p), srcip, sizeof(srcip));
211  PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p), dstip, sizeof(dstip));
212  }
213  sp = p->dp;
214  dp = p->sp;
215  }
216  } else {
217  if (PKT_IS_IPV4(p)) {
218  PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p), srcip, sizeof(srcip));
219  PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p), dstip, sizeof(dstip));
220  } else if (PKT_IS_IPV6(p)) {
221  PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p), srcip, sizeof(srcip));
222  PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p), dstip, sizeof(dstip));
223  }
224  sp = p->sp;
225  dp = p->dp;
226  }
227 
228  char proto[16];
229  if (SCProtoNameValid(IP_GET_IPPROTO(p)) == TRUE) {
230  strlcpy(proto, known_proto[IP_GET_IPPROTO(p)], sizeof(proto));
231  } else {
232  snprintf(proto, sizeof(proto), "%03" PRIu32, IP_GET_IPPROTO(p));
233  }
234 
235  /* time & tx */
236  json_object_set_new(js, "timestamp", json_string(timebuf));
237 
238  CreateJSONFlowId(js, (const Flow *)p->flow);
239 
240  /* sensor id */
241  if (sensor_id >= 0)
242  json_object_set_new(js, "sensor_id", json_integer(sensor_id));
243 
244  /* input interface */
245  if (p->livedev) {
246  json_object_set_new(js, "in_iface", json_string(p->livedev->dev));
247  }
248 
249  /* pcap_cnt */
250  if (p->pcap_cnt != 0) {
251  json_object_set_new(js, "pcap_cnt", json_integer(p->pcap_cnt));
252  }
253 
254  if (event_type) {
255  json_object_set_new(js, "event_type", json_string(event_type));
256  }
257 
258  /* vlan */
259  if (p->vlan_idx > 0) {
260  json_t *js_vlan;
261  switch (p->vlan_idx) {
262  case 1:
263  json_object_set_new(js, "vlan",
264  json_integer(VLAN_GET_ID1(p)));
265  break;
266  case 2:
267  js_vlan = json_array();
268  if (unlikely(js != NULL)) {
269  json_array_append_new(js_vlan,
270  json_integer(VLAN_GET_ID1(p)));
271  json_array_append_new(js_vlan,
272  json_integer(VLAN_GET_ID2(p)));
273  json_object_set_new(js, "vlan", js_vlan);
274  }
275  break;
276  default:
277  /* shouldn't get here */
278  break;
279  }
280  }
281 
282  /* tuple */
283  json_object_set_new(js, "src_ip", json_string(srcip));
284  switch(p->proto) {
285  case IPPROTO_ICMP:
286  break;
287  case IPPROTO_UDP:
288  case IPPROTO_TCP:
289  case IPPROTO_SCTP:
290  json_object_set_new(js, "src_port", json_integer(sp));
291  break;
292  }
293  json_object_set_new(js, "dest_ip", json_string(dstip));
294  switch(p->proto) {
295  case IPPROTO_ICMP:
296  break;
297  case IPPROTO_UDP:
298  case IPPROTO_TCP:
299  case IPPROTO_SCTP:
300  json_object_set_new(js, "dest_port", json_integer(dp));
301  break;
302  }
303  json_object_set_new(js, "proto", json_string(proto));
304  switch (p->proto) {
305  case IPPROTO_ICMP:
306  if (p->icmpv4h) {
307  json_object_set_new(js, "icmp_type",
308  json_integer(p->icmpv4h->type));
309  json_object_set_new(js, "icmp_code",
310  json_integer(p->icmpv4h->code));
311  }
312  break;
313  case IPPROTO_ICMPV6:
314  if (p->icmpv6h) {
315  json_object_set_new(js, "icmp_type",
316  json_integer(p->icmpv6h->type));
317  json_object_set_new(js, "icmp_code",
318  json_integer(p->icmpv6h->code));
319  }
320  break;
321  }
322 
323  return js;
324 }

352 int OutputJSONBuffer(json_t *js, LogFileCtx *file_ctx, MemBuffer *buffer)
353 {
354  if (file_ctx->sensor_name) {
355  json_object_set_new(js, "host",
356  json_string(file_ctx->sensor_name));
357  }
358 
359  int r = json_dump_callback(js, MemBufferCallback, buffer,
360  JSON_PRESERVE_ORDER|JSON_COMPACT|JSON_ENSURE_ASCII|
361 #ifdef JSON_ESCAPE_SLASH
362  JSON_ESCAPE_SLASH
363 #else
364  0
365 #endif
366  );
367  if (r != 0)
368  return TM_ECODE_OK;
369 
370  LogFileWrite(file_ctx, buffer);
371  return 0;
372 }

419 /**
420  * \brief Create a new LogFileCtx for "fast" output style.
421  * \param conf The configuration node for this output.
422  * \return A LogFileCtx pointer on success, NULL on failure.
423  */
424 OutputCtx *OutputJsonInitCtx(ConfNode *conf)
425 {
426  OutputJsonCtx *json_ctx = SCCalloc(1, sizeof(OutputJsonCtx));;
427 
428  const char *sensor_name = ConfNodeLookupChildValue(conf, "sensor-name");
429 
430  if (unlikely(json_ctx == NULL)) {
431  SCLogDebug("AlertJsonInitCtx: Could not create new LogFileCtx");
432  return NULL;
433  }
434 
435  json_ctx->file_ctx = LogFileNewCtx();
436  if (unlikely(json_ctx->file_ctx == NULL)) {
437  SCLogDebug("AlertJsonInitCtx: Could not create new LogFileCtx");
438  SCFree(json_ctx);
439  return NULL;
440  }
441 
442  if (sensor_name) {
443  json_ctx->file_ctx->sensor_name = SCStrdup(sensor_name);
444  if (json_ctx->file_ctx->sensor_name == NULL) {
445  LogFileFreeCtx(json_ctx->file_ctx);
446  SCFree(json_ctx);
447  return NULL;
448  }
449  } else {
450  json_ctx->file_ctx->sensor_name = NULL;
451  }
452 
453  OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
454  if (unlikely(output_ctx == NULL)) {
455  LogFileFreeCtx(json_ctx->file_ctx);
456  SCFree(json_ctx);
457  return NULL;
458  }
459 
460  output_ctx->data = json_ctx;
461  output_ctx->DeInit = OutputJsonDeInitCtx;
462 
463  if (conf) {
464  const char *output_s = ConfNodeLookupChildValue(conf, "filetype");
465 
466  // Backwards compatibility
467  if (output_s == NULL) {
468  output_s = ConfNodeLookupChildValue(conf, "type");
469  }
470 
471  if (output_s != NULL) {
472  if (strcmp(output_s, "file") == 0 ||
473  strcmp(output_s, "regular") == 0) {
474  json_ctx->json_out = LOGFILE_TYPE_FILE;
475  } else if (strcmp(output_s, "syslog") == 0) {
476  json_ctx->json_out = LOGFILE_TYPE_SYSLOG;
477  } else if (strcmp(output_s, "unix_dgram") == 0) {
478  json_ctx->json_out = LOGFILE_TYPE_UNIX_DGRAM;
479  } else if (strcmp(output_s, "unix_stream") == 0) {
480  json_ctx->json_out = LOGFILE_TYPE_UNIX_STREAM;
481  } else if (strcmp(output_s, "redis") == 0) {
482 #ifdef HAVE_LIBHIREDIS
483  json_ctx->json_out = LOGFILE_TYPE_REDIS;
484 #else
485  SCLogError(SC_ERR_INVALID_ARGUMENT,
486  "redis JSON output option is not compiled");
487  exit(EXIT_FAILURE);
488 #endif
489  } else {
490  SCLogError(SC_ERR_INVALID_ARGUMENT,
491  "Invalid JSON output option: %s", output_s);
492  exit(EXIT_FAILURE);
493  }
494  }
495 
496  const char *prefix = ConfNodeLookupChildValue(conf, "prefix");
497  if (prefix != NULL)
498  {
499  json_ctx->file_ctx->prefix = SCStrdup(prefix);
500  if (json_ctx->file_ctx->prefix == NULL)
501  {
502  SCLogError(SC_ERR_MEM_ALLOC,
503  "Failed to allocate memory for eve-log.prefix setting.");
504  exit(EXIT_FAILURE);
505  }
506  }
507 
508  if (json_ctx->json_out == LOGFILE_TYPE_FILE ||
509  json_ctx->json_out == LOGFILE_TYPE_UNIX_DGRAM ||
510  json_ctx->json_out == LOGFILE_TYPE_UNIX_STREAM)
511  {
512  if (SCConfLogOpenGeneric(conf, json_ctx->file_ctx, DEFAULT_LOG_FILENAME, 1) < 0) {
513  LogFileFreeCtx(json_ctx->file_ctx);
514  SCFree(json_ctx);
515  SCFree(output_ctx);
516  return NULL;
517  }
518  OutputRegisterFileRotationFlag(&json_ctx->file_ctx->rotation_flag);
519 
520  const char *format_s = ConfNodeLookupChildValue(conf, "format");
521  if (format_s != NULL) {
522  if (strcmp(format_s, "indent") == 0) {
523  json_ctx->format = INDENT;
524  } else if (strcmp(format_s, "compact") == 0) {
525  json_ctx->format = COMPACT;
526  } else {
527  SCLogError(SC_ERR_INVALID_ARGUMENT,
528  "Invalid JSON format option: %s", format_s);
529  exit(EXIT_FAILURE);
530  }
531  }
532  } else if (json_ctx->json_out == LOGFILE_TYPE_SYSLOG) {
533  const char *facility_s = ConfNodeLookupChildValue(conf, "facility");
534  if (facility_s == NULL) {
535  facility_s = DEFAULT_ALERT_SYSLOG_FACILITY_STR;
536  }
537 
538  int facility = SCMapEnumNameToValue(facility_s, SCSyslogGetFacilityMap());
539  if (facility == -1) {
540  SCLogWarning(SC_ERR_INVALID_ARGUMENT, "Invalid syslog facility: \"%s\","
541  " now using \"%s\" as syslog facility", facility_s,
542  DEFAULT_ALERT_SYSLOG_FACILITY_STR);
543  facility = DEFAULT_ALERT_SYSLOG_FACILITY;
544  }
545 
546  const char *level_s = ConfNodeLookupChildValue(conf, "level");
547  if (level_s != NULL) {
548  int level = SCMapEnumNameToValue(level_s, SCSyslogGetLogLevelMap());
549  if (level != -1) {
550  json_ctx->file_ctx->syslog_setup.alert_syslog_level = level;
551  }
552  }
553 
554  const char *ident = ConfNodeLookupChildValue(conf, "identity");
555  /* if null we just pass that to openlog, which will then
556  * figure it out by itself. */
557 
558  openlog(ident, LOG_PID|LOG_NDELAY, facility);
559 
560  }
561 #ifdef HAVE_LIBHIREDIS
562  else if (json_ctx->json_out == LOGFILE_TYPE_REDIS) {
563  ConfNode *redis_node = ConfNodeLookupChild(conf, "redis");
564  if (!json_ctx->file_ctx->sensor_name) {
565  char hostname[1024];
566  gethostname(hostname, 1023);
567  json_ctx->file_ctx->sensor_name = SCStrdup(hostname);
568  }
569  if (json_ctx->file_ctx->sensor_name == NULL) {
570  LogFileFreeCtx(json_ctx->file_ctx);
571  SCFree(json_ctx);
572  SCFree(output_ctx);
573  return NULL;
574  }
575 
576  if (SCConfLogOpenRedis(redis_node, json_ctx->file_ctx) < 0) {
577  LogFileFreeCtx(json_ctx->file_ctx);
578  SCFree(json_ctx);
579  SCFree(output_ctx);
580  return NULL;
581  }
582  }
583 #endif
584 
585  const char *sensor_id_s = ConfNodeLookupChildValue(conf, "sensor-id");
586  if (sensor_id_s != NULL) {
587  if (ByteExtractStringUint64((uint64_t *)&sensor_id, 10, 0, sensor_id_s) == -1) {
588  SCLogError(SC_ERR_INVALID_ARGUMENT,
589  "Failed to initialize JSON output, "
590  "invalid sensor-is: %s", sensor_id_s);
591  exit(EXIT_FAILURE);
592  }
593  }
594 
595  json_ctx->file_ctx->type = json_ctx->json_out;
596  }
597 
598  SCLogDebug("returning output_ctx %p", output_ctx);
599  return output_ctx;
600 }
