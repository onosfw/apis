221 /** \brief dump stack from lua state to screen */
222 void LuaPrintStack(lua_State *state) {
223  int size = lua_gettop(state);
224  int i;
225 
226  for (i = 1; i <= size; i++) {
227  int type = lua_type(state, i);
228  printf("Stack size=%d, level=%d, type=%d, ", size, i, type);
229 
230  switch (type) {
231  case LUA_TFUNCTION:
232  printf("function %s", lua_tostring(state, i) ? "true" : "false");
233  break;
234  case LUA_TBOOLEAN:
235  printf("bool %s", lua_toboolean(state, i) ? "true" : "false");
236  break;
237  case LUA_TNUMBER:
238  printf("number %g", lua_tonumber(state, i));
239  break;
240  case LUA_TSTRING:
241  printf("string `%s'", lua_tostring(state, i));
242  break;
243  case LUA_TTABLE:
244  printf("table `%s'", lua_tostring(state, i));
245  break;
246  default:
247  printf("other %s", lua_typename(state, type));
248  break;
249 
250  }
251  printf("\n");
252  }
253 }

255 int LuaPushStringBuffer(lua_State *luastate, const uint8_t *input, size_t input_len)
256 {
257  if (input_len % 4 != 0) {
258  /* we're using a buffer sized at a multiple of 4 as lua_pushlstring generates
259  * invalid read errors in valgrind otherwise. Adding in a nul to be sure.
260  *
261  * Buffer size = len + 1 (for nul) + whatever makes it a multiple of 4 */
262  size_t buflen = input_len + 1 + ((input_len + 1) % 4);
263  uint8_t buf[buflen];
264  memset(buf, 0x00, buflen);
265  memcpy(buf, input, input_len);
266  buf[input_len] = '\0';
267 
268  /* return value through luastate, as a luastring */
269  lua_pushlstring(luastate, (char *)buf, input_len);
270  } else {
271  lua_pushlstring(luastate, (char *)input, input_len);
272  }
273  return 1;
274 }
