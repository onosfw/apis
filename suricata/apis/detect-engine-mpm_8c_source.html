71  * \brief check if a signature has patterns that are to be inspected
72  * against a packets payload (as opposed to the stream payload)
73  *
74  * \param s signature
75  *
76  * \retval 1 true
77  * \retval 0 false
78  */
79 int SignatureHasPacketContent(Signature *s)
80 {
81  SCEnter();
82 
83  if (s == NULL) {
84  SCReturnInt(0);
85  }
86 
87  if (!(s->proto.proto[IPPROTO_TCP / 8] & 1 << (IPPROTO_TCP % 8))) {
88  SCReturnInt(1);
89  }
90 
91  if (s->sm_lists[DETECT_SM_LIST_PMATCH] == NULL) {
92  SCLogDebug("no mpm");
93  SCReturnInt(0);
94  }
95 
96  if (!(s->flags & SIG_FLAG_REQUIRE_PACKET)) {
97  SCReturnInt(0);
98  }
99 
100  SCReturnInt(1);
101 }

138 /**
139  * \brief Function to return the multi pattern matcher algorithm to be
140  * used by the engine, based on the mpm-algo setting in yaml
141  * Use the default mpm if none is specified in the yaml file.
142  *
143  * \retval mpm algo value
144  */
145 uint16_t PatternMatchDefaultMatcher(void)
146 {
147  char *mpm_algo;
148  uint16_t mpm_algo_val = DEFAULT_MPM;
149 
150  /* Get the mpm algo defined in config file by the user */
151  if ((ConfGet("mpm-algo", &mpm_algo)) == 1) {
152  uint16_t u;
153 
154  if (mpm_algo != NULL) {
155  for (u = 0; u < MPM_TABLE_SIZE; u++) {
156  if (mpm_table[u].name == NULL)
157  continue;
158 
159  if (strcmp(mpm_table[u].name, mpm_algo) == 0) {
160  mpm_algo_val = u;
161  goto done;
162  }
163  }
164  }
165 
166  SCLogError(SC_ERR_INVALID_YAML_CONF_ENTRY, "Invalid mpm algo supplied "
167  "in the yaml conf file: \"%s\"", mpm_algo);
168  exit(EXIT_FAILURE);
169  }
170 
171  done:
172 #ifdef __tile__
173  if (mpm_algo_val == MPM_AC)
174  mpm_algo_val = MPM_AC_TILE;
175 #endif
176 
177  return mpm_algo_val;
178 }

204 /** \brief Pattern match -- searches for only one pattern per signature.
205  *
206  * \param det_ctx detection engine thread ctx
207  * \param p packet to inspect
208  *
209  * \retval ret number of matches
210  */
211 uint32_t PacketPatternSearch(DetectEngineThreadCtx *det_ctx, Packet *p)
212 {
213  SCEnter();
214 
215  uint32_t ret;
216  MpmCtx *mpm_ctx = NULL;
217 
218  if (p->proto == IPPROTO_TCP) {
219  if (p->flowflags & FLOW_PKT_TOSERVER) {
220  mpm_ctx = det_ctx->sgh->mpm_proto_tcp_ctx_ts;
221  } else {
222  mpm_ctx = det_ctx->sgh->mpm_proto_tcp_ctx_tc;
223  }
224  } else if (p->proto == IPPROTO_UDP) {
225  if (p->flowflags & FLOW_PKT_TOSERVER) {
226  mpm_ctx = det_ctx->sgh->mpm_proto_udp_ctx_ts;
227  } else {
228  mpm_ctx = det_ctx->sgh->mpm_proto_udp_ctx_tc;
229  }
230  } else {
231  mpm_ctx = det_ctx->sgh->mpm_proto_other_ctx;
232  }
233  if (unlikely(mpm_ctx == NULL))
234  SCReturnInt(0);
235 
236 #ifdef __SC_CUDA_SUPPORT__
237  if (p->cuda_pkt_vars.cuda_mpm_enabled && p->pkt_src == PKT_SRC_WIRE) {
238  ret = SCACCudaPacketResultsProcessing(p, mpm_ctx, &det_ctx->pmq);
239  } else {
240  ret = mpm_table[mpm_ctx->mpm_type].Search(mpm_ctx,
241  &det_ctx->mtc,
242  &det_ctx->pmq,
243  p->payload,
244  p->payload_len);
245  }
246 #else
247  ret = mpm_table[mpm_ctx->mpm_type].Search(mpm_ctx,
248  &det_ctx->mtc,
249  &det_ctx->pmq,
250  p->payload,
251  p->payload_len);
252 #endif
253 
254  SCReturnInt(ret);
255 }

1171 /** \brief Predict a strength value for patterns
1172  *
1173  * Patterns with high character diversity score higher.
1174  * Alpha chars score not so high
1175  * Other printable + a few common codes a little higher
1176  * Everything else highest.
1177  * Longer patterns score better than short patters.
1178  *
1179  * \param pat pattern
1180  * \param patlen length of the patternn
1181  *
1182  * \retval s pattern score
1183  */
1184 uint32_t PatternStrength(uint8_t *pat, uint16_t patlen)
1185 {
1186  uint8_t a[256];
1187  memset(&a, 0 ,sizeof(a));
1188 
1189  uint32_t s = 0;
1190  uint16_t u = 0;
1191  for (u = 0; u < patlen; u++) {
1192  if (a[pat[u]] == 0) {
1193  if (isalpha(pat[u]))
1194  s += 3;
1195  else if (isprint(pat[u]) || pat[u] == 0x00 || pat[u] == 0x01 || pat[u] == 0xFF)
1196  s += 4;
1197  else
1198  s += 6;
1199 
1200  a[pat[u]] = 1;
1201  } else {
1202  s++;
1203  }
1204  }
1205 
1206  return s;
1207 }
