141 /**
142  * Handles request parser state changes. At the moment, this function is used only
143  * to configure data receivers, which are sent raw connection data.
144  *
145  * @param[in] connp
146  * @return HTP_OK, or a value returned from a callback.
147  */
148 static htp_status_t htp_req_handle_state_change(htp_connp_t *connp) {
149  if (connp->in_state_previous == connp->in_state) return HTP_OK;
150 
151  if (connp->in_state == htp_connp_REQ_HEADERS) {
152  htp_status_t rc = HTP_OK;
153 
154  switch (connp->in_tx->request_progress) {
155  case HTP_REQUEST_HEADERS:
156  rc = htp_connp_req_receiver_set(connp, connp->in_tx->cfg->hook_request_header_data);
157  break;
158 
159  case HTP_REQUEST_TRAILER:
160  rc = htp_connp_req_receiver_set(connp, connp->in_tx->cfg->hook_request_trailer_data);
161  break;
162 
163  default:
164  // Do nothing; receivers are currently used only for header blocks.
165  break;
166  }
167 
168  if (rc != HTP_OK) return rc;
169  }
170 
171  // Initially, I had the finalization of raw data sending here, but that
172  // caused the last REQUEST_HEADER_DATA hook to be invoked after the
173  // REQUEST_HEADERS hook -- which I thought made no sense. For that reason,
174  // the finalization is now initiated from the request header processing code,
175  // which is less elegant but provides a better user experience. Having some
176  // (or all) hooks to be invoked on state change might work better.
177 
178  connp->in_state_previous = connp->in_state;
179 
180  return HTP_OK;
181 }

183 /**
184  * If there is any data left in the inbound data chunk, this function will preserve
185  * it for later consumption. The maximum amount accepted for buffering is controlled
186  * by htp_config_t::field_limit_hard.
187  *
188  * @param[in] connp
189  * @return HTP_OK, or HTP_ERROR on fatal failure.
190  */
191 static htp_status_t htp_connp_req_buffer(htp_connp_t *connp) {
192  if (connp->in_current_data == NULL) return HTP_OK;
193 
194  unsigned char *data = connp->in_current_data + connp->in_current_consume_offset;
195  size_t len = connp->in_current_read_offset - connp->in_current_consume_offset;
196 
197  if (len == 0)
198  return HTP_OK;
199 
200  // Check the hard (buffering) limit.
201 
202  size_t newlen = connp->in_buf_size + len;
203 
204  // When calculating the size of the buffer, take into account the
205  // space we're using for the request header buffer.
206  if (connp->in_header != NULL) {
207  newlen += bstr_len(connp->in_header);
208  }
209 
210  if (newlen > connp->in_tx->cfg->field_limit_hard) {
211  htp_log(connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0, "Request buffer over the limit: size %zd limit %zd.",
212  newlen, connp->in_tx->cfg->field_limit_hard);
213  return HTP_ERROR;
214  }
215 
216  // Copy the data remaining in the buffer.
217 
218  if (connp->in_buf == NULL) {
219  connp->in_buf = malloc(len);
220  if (connp->in_buf == NULL) return HTP_ERROR;
221  memcpy(connp->in_buf, data, len);
222  connp->in_buf_size = len;
223  } else {
224  size_t newsize = connp->in_buf_size + len;
225  unsigned char *newbuf = realloc(connp->in_buf, newsize);
226  if (newbuf == NULL) return HTP_ERROR;
227  connp->in_buf = newbuf;
228  memcpy(connp->in_buf + connp->in_buf_size, data, len);
229  connp->in_buf_size = newsize;
230  }
231 
232  // Reset the consumer position.
233  connp->in_current_consume_offset = connp->in_current_read_offset;
234 
235  return HTP_OK;
236 }

238 /**
239  * Returns to the caller the memory region that should be processed next. This function
240  * hides away the buffering process from the rest of the code, allowing it to work with
241  * non-buffered data that's in the inbound chunk, or buffered data that's in our structures.
242  *
243  * @param[in] connp
244  * @param[out] data
245  * @param[out] len
246  * @return HTP_OK
247  */
248 static htp_status_t htp_connp_req_consolidate_data(htp_connp_t *connp, unsigned char **data, size_t *len) {
249  if (connp->in_buf == NULL) {
250  // We do not have any data buffered; point to the current data chunk.
251  *data = connp->in_current_data + connp->in_current_consume_offset;
252  *len = connp->in_current_read_offset - connp->in_current_consume_offset;
253  } else {
254  // We already have some data in the buffer. Add the data from the current
255  // chunk to it, and point to the consolidated buffer.
256  if (htp_connp_req_buffer(connp) != HTP_OK) {
257  return HTP_ERROR;
258  }
259 
260  *data = connp->in_buf;
261  *len = connp->in_buf_size;
262  }
263 
264  return HTP_OK;
265 }

284  * parsing needs to be suspended.
285  *
286  * @param[in] connp
287  * @return HTP_OK if the request does not use CONNECT, HTP_DATA_OTHER if
288  * inbound parsing needs to be suspended until we hear from the
289  * other side
290  */
291 htp_status_t htp_connp_REQ_CONNECT_CHECK(htp_connp_t *connp) {
292  // If the request uses the CONNECT method, then there will
293  // not be a request body, but first we need to wait to see the
294  // response in order to determine if the tunneling request
295  // was a success.
296  if (connp->in_tx->request_method_number == HTP_M_CONNECT) {
297  connp->in_state = htp_connp_REQ_CONNECT_WAIT_RESPONSE;
298  connp->in_status = HTP_STREAM_DATA_OTHER;
299  return HTP_DATA_OTHER;
300  }
301 
302  // Continue to the next step to determine
303  // the presence of request body
304  connp->in_state = htp_connp_REQ_BODY_DETERMINE;
305 
306  return HTP_OK;

309 /**
310  * Determines whether inbound parsing needs to continue or stop. In
311  * case the data appears to be plain text HTTP, we try to continue.
312  *
313  * @param[in] connp
314  * @return HTP_OK if the parser can resume parsing, HTP_DATA_BUFFER if
315  * we need more data.
316  */
317 htp_status_t htp_connp_REQ_CONNECT_PROBE_DATA(htp_connp_t *connp) {
318  for (;;) {//;i < max_read; i++) {
319  IN_PEEK_NEXT(connp);
320  // Have we reached the end of the line? For some reason
321  // we can't test after IN_COPY_BYTE_OR_RETURN */
322  if (connp->in_next_byte == LF || connp->in_next_byte == 0x00)
323  break;
324 
325  IN_COPY_BYTE_OR_RETURN(connp);
326 
327  }
328 
329  unsigned char *data;
330  size_t len;
331  if (htp_connp_req_consolidate_data(connp, &data, &len) != HTP_OK) {
332  fprintf(stderr, "htp_connp_req_consolidate_data fail");
333  return HTP_ERROR;
334  }
335 #ifdef HTP_DEBUG
336  fprint_raw_data(stderr, "PROBING", data, len);
337 #endif
338 
339  size_t pos = 0;
340  size_t mstart = 0;
341  // skip past leading whitespace. IIS allows this
342  while ((pos < len) && htp_is_space(data[pos]))
343  pos++;
344  if (pos)
345  mstart = pos;
346  // The request method starts at the beginning of the
347  // line and ends with the first whitespace character.
348  while ((pos < len) && (!htp_is_space(data[pos])))
349  pos++;
350 
351  int methodi = HTP_M_UNKNOWN;
352  bstr *method = bstr_dup_mem(data + mstart, pos - mstart);
353  if (method) {
354  methodi = htp_convert_method_to_number(method);
355  bstr_free(method);
356  }
357  if (methodi != HTP_M_UNKNOWN) {
358 #ifdef HTP_DEBUG
359  fprint_raw_data(stderr, "htp_connp_REQ_CONNECT_PROBE_DATA: tunnel contains plain text HTTP", data, len);
360 #endif
361  connp->in_state = htp_connp_REQ_IDLE;
362  } else {
363 #ifdef HTP_DEBUG
364  fprint_raw_data(stderr, "htp_connp_REQ_CONNECT_PROBE_DATA: tunnel is not HTTP", data, len);
365 #endif
366  connp->in_status = HTP_STREAM_TUNNEL;
367  connp->out_status = HTP_STREAM_TUNNEL;
368  }
369 
370  // not calling htp_connp_req_clear_buffer, we're not consuming the data
371 
372  return HTP_OK;
373 }

375 /**
376  * Determines whether inbound parsing, which was suspended after
377  * encountering a CONNECT transaction, can proceed (after receiving
378  * the response).
379  *
380  * @param[in] connp
381  * @return HTP_OK if the parser can resume parsing, HTP_DATA_OTHER if
382  * it needs to continue waiting.
383  */
384 htp_status_t htp_connp_REQ_CONNECT_WAIT_RESPONSE(htp_connp_t *connp) {
385  // Check that we saw the response line of the current inbound transaction.
386  if (connp->in_tx->response_progress <= HTP_RESPONSE_LINE) {
387  return HTP_DATA_OTHER;
388  }
389 
390  // A 2xx response means a tunnel was established. Anything
391  // else means we continue to follow the HTTP stream.
392  if ((connp->in_tx->response_status_number >= 200) && (connp->in_tx->response_status_number <= 299)) {
393  // TODO Check that the server did not accept a connection to itself.
394 
395  // The requested tunnel was established: we are going
396  // to probe the remaining data on this stream to see
397  // if we need to ignore it or parse it
398  connp->in_state = htp_connp_REQ_CONNECT_PROBE_DATA;
399  } else {
400  // No tunnel; continue to the next transaction
401  connp->in_state = htp_connp_REQ_FINALIZE;
402  }
403 
404  return HTP_OK;
405 }

431 /**
432  * Processes a chunk of data.
433  *
434  * @param[in] connp
435  * @returns HTP_OK on state change, HTP_ERROR on error, or HTP_DATA when more data is needed.
436  */
437 htp_status_t htp_connp_REQ_BODY_CHUNKED_DATA(htp_connp_t *connp) {
438  // Determine how many bytes we can consume.
439  size_t bytes_to_consume;
440  if (connp->in_current_len - connp->in_current_read_offset >= connp->in_chunked_length) {
441  // Entire chunk available in the buffer; read all of it.
442  bytes_to_consume = connp->in_chunked_length;
443  } else {
444  // Partial chunk available in the buffer; read as much as we can.
445  bytes_to_consume = connp->in_current_len - connp->in_current_read_offset;
446  }
447 
448  #ifdef HTP_DEBUG
449  fprintf(stderr, "htp_connp_REQ_BODY_CHUNKED_DATA Consuming %zd bytes\n", bytes_to_consume);
450  #endif
451 
452  // If the input buffer is empty, ask for more data.
453  if (bytes_to_consume == 0) return HTP_DATA;
454 
455  // Consume the data.
456  htp_status_t rc = htp_tx_req_process_body_data_ex(connp->in_tx, connp->in_current_data + connp->in_current_read_offset, bytes_to_consume);
457  if (rc != HTP_OK) return rc;
458 
459  // Adjust counters.
460  connp->in_current_read_offset += bytes_to_consume;
461  connp->in_current_consume_offset += bytes_to_consume;
462  connp->in_stream_offset += bytes_to_consume;
463  connp->in_tx->request_message_len += bytes_to_consume;
464  connp->in_chunked_length -= bytes_to_consume;
465 
466  if (connp->in_chunked_length == 0) {
467  // End of the chunk.
468  connp->in_state = htp_connp_REQ_BODY_CHUNKED_DATA_END;
469  return HTP_OK;
470  }
471 
472  // Ask for more data.
473  return HTP_DATA;
474 }

476 /**
477  * Extracts chunk length.
478  *
479  * @param[in] connp
480  * @returns HTP_OK on state change, HTP_ERROR on error, or HTP_DATA when more data is needed.
481  */
482 htp_status_t htp_connp_REQ_BODY_CHUNKED_LENGTH(htp_connp_t *connp) {
483  for (;;) {
484  IN_COPY_BYTE_OR_RETURN(connp);
485 
486  // Have we reached the end of the line?
487  if (connp->in_next_byte == LF) {
488  unsigned char *data;
489  size_t len;
490 
491  if (htp_connp_req_consolidate_data(connp, &data, &len) != HTP_OK) {
492  return HTP_ERROR;
493  }
494 
495  connp->in_tx->request_message_len += len;
496 
497  #ifdef HTP_DEBUG
498  fprint_raw_data(stderr, "Chunk length line", data, len);
499  #endif
500 
501  htp_chomp(data, &len);
502 
503  connp->in_chunked_length = htp_parse_chunked_length(data, len);
504 
505  htp_connp_req_clear_buffer(connp);
506 
507  // Handle chunk length.
508  if (connp->in_chunked_length > 0) {
509  // More data available.
510  connp->in_state = htp_connp_REQ_BODY_CHUNKED_DATA;
511  } else if (connp->in_chunked_length == 0) {
512  // End of data.
513  connp->in_state = htp_connp_REQ_HEADERS;
514  connp->in_tx->request_progress = HTP_REQUEST_TRAILER;
515  } else {
516  // Invalid chunk length.
517  htp_log(connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0, "Request chunk encoding: Invalid chunk length");
518  return HTP_ERROR;
519  }
520 
521  return HTP_OK;
522  }
523  }
524 
525  return HTP_ERROR;
526 }

583  case HTP_CODING_IDENTITY:

585  connp->in_body_data_left = connp->in_content_length;

588  connp->in_state = htp_connp_REQ_BODY_IDENTITY;
589  connp->in_tx->request_progress = HTP_REQUEST_BODY;
590  } else {

592  }
593  break;
594 
595  case HTP_CODING_NO_BODY:
596  // This request does not have a body, which
597  // means that we're done with it

599  break;

602  // Should not be here
603  return HTP_ERROR;
604  break;
605  }
606

610 /**
611  * Parses request headers.
612  *
613  * @param[in] connp
614  * @returns HTP_OK on state change, HTP_ERROR on error, or HTP_DATA when more data is needed.
615  */
616 htp_status_t htp_connp_REQ_HEADERS(htp_connp_t *connp) {
617  for (;;) {
618  IN_COPY_BYTE_OR_RETURN(connp);
619 
620  // Have we reached the end of the line?
621  if (connp->in_next_byte == LF) {
622  unsigned char *data;
623  size_t len;
624 
625  if (htp_connp_req_consolidate_data(connp, &data, &len) != HTP_OK) {
626  return HTP_ERROR;
627  }
628 
629  #ifdef HTP_DEBUG
630  fprint_raw_data(stderr, __FUNCTION__, data, len);
631  #endif
632 
633  // Should we terminate headers?
634  if (htp_connp_is_line_terminator(connp, data, len)) {
635  // Parse previous header, if any.
636  if (connp->in_header != NULL) {
637  if (connp->cfg->process_request_header(connp, bstr_ptr(connp->in_header),
638  bstr_len(connp->in_header)) != HTP_OK) return HTP_ERROR;
639 
640  bstr_free(connp->in_header);
641  connp->in_header = NULL;
642  }
643 
644  htp_connp_req_clear_buffer(connp);
645 
646  // We've seen all the request headers.
647  return htp_tx_state_request_headers(connp->in_tx);
648  }
649 
650  htp_chomp(data, &len);
651 
652  // Check for header folding.
653  if (htp_connp_is_line_folded(data, len) == 0) {
654  // New header line.
655 
656  // Parse previous header, if any.
657  if (connp->in_header != NULL) {
658  if (connp->cfg->process_request_header(connp, bstr_ptr(connp->in_header),
659  bstr_len(connp->in_header)) != HTP_OK) return HTP_ERROR;
660 
661  bstr_free(connp->in_header);
662  connp->in_header = NULL;
663  }
664 
665  IN_PEEK_NEXT(connp);
666 
667  if (htp_is_folding_char(connp->in_next_byte) == 0) {
668  // Because we know this header is not folded, we can process the buffer straight away.
669  if (connp->cfg->process_request_header(connp, data, len) != HTP_OK) return HTP_ERROR;
670  } else {
671  // Keep the partial header data for parsing later.
672  connp->in_header = bstr_dup_mem(data, len);
673  if (connp->in_header == NULL) return HTP_ERROR;
674  }
675  } else {
676  // Folding; check that there's a previous header line to add to.
677  if (connp->in_header == NULL) {
678  // Invalid folding.
679 
680  // Warn only once per transaction.
681  if (!(connp->in_tx->flags & HTP_INVALID_FOLDING)) {
682  connp->in_tx->flags |= HTP_INVALID_FOLDING;
683  htp_log(connp, HTP_LOG_MARK, HTP_LOG_WARNING, 0, "Invalid request field folding");
684  }
685 
686  // Keep the header data for parsing later.
687  connp->in_header = bstr_dup_mem(data, len);
688  if (connp->in_header == NULL) return HTP_ERROR;
689  } else {
690  // Add to the existing header.
691  bstr *new_in_header = bstr_add_mem(connp->in_header, data, len);
692  if (new_in_header == NULL) return HTP_ERROR;
693  connp->in_header = new_in_header;
694  }
695  }
696 
697  htp_connp_req_clear_buffer(connp);
698  }
699  }
700 
701  return HTP_ERROR;
702 }

728  * @param[in] connp

733  size_t len;
734

848 int htp_connp_req_data(htp_connp_t *connp, const htp_time_t *timestamp, const void *data, size_t len) {
849  #ifdef HTP_DEBUG
850  fprintf(stderr, "htp_connp_req_data(connp->in_status %x)\n", connp->in_status);
851  fprint_raw_data(stderr, __FUNCTION__, data, len);
852  #endif
853 
854  // Return if the connection is in stop state.
855  if (connp->in_status == HTP_STREAM_STOP) {
856  htp_log(connp, HTP_LOG_MARK, HTP_LOG_INFO, 0, "Inbound parser is in HTP_STREAM_STOP");
857  return HTP_STREAM_STOP;
858  }
859 
860  // Return if the connection had a fatal error earlier
861  if (connp->in_status == HTP_STREAM_ERROR) {
862  htp_log(connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0, "Inbound parser is in HTP_STREAM_ERROR");
863 
864  #ifdef HTP_DEBUG
865  fprintf(stderr, "htp_connp_req_data: returning HTP_STREAM_DATA (previous error)\n");
866  #endif
867 
868  return HTP_STREAM_ERROR;
869  }
870 
871  // Sanity check: we must have a transaction pointer if the state is not IDLE (no inbound transaction)
872  if ((connp->in_tx == NULL)&&(connp->in_state != htp_connp_REQ_IDLE)) {
873  connp->in_status = HTP_STREAM_ERROR;
874 
875  htp_log(connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0, "Missing inbound transaction data");
876 
877  return HTP_STREAM_ERROR;
878  }
879 
880  // If the length of the supplied data chunk is zero, proceed
881  // only if the stream has been closed. We do not allow zero-sized
882  // chunks in the API, but we use them internally to force the parsers
883  // to finalize parsing.
884  if (((data == NULL) || (len == 0)) && (connp->in_status != HTP_STREAM_CLOSED)) {
885  htp_log(connp, HTP_LOG_MARK, HTP_LOG_ERROR, 0, "Zero-length data chunks are not allowed");
886 
887  #ifdef HTP_DEBUG
888  fprintf(stderr, "htp_connp_req_data: returning HTP_STREAM_DATA (zero-length chunk)\n");
889  #endif
890 
891  return HTP_STREAM_CLOSED;
892  }
893 
894  // Remember the timestamp of the current request data chunk
895  if (timestamp != NULL) {
896  memcpy(&connp->in_timestamp, timestamp, sizeof (*timestamp));
897  }
898 
899  // Store the current chunk information
900  connp->in_current_data = (unsigned char *) data;
901  connp->in_current_len = len;
902  connp->in_current_read_offset = 0;
903  connp->in_current_consume_offset = 0;
904  connp->in_current_receiver_offset = 0;
905  connp->in_chunk_count++;
906 
907  htp_conn_track_inbound_data(connp->conn, len, timestamp);
908 
909 
910  // Return without processing any data if the stream is in tunneling
911  // mode (which it would be after an initial CONNECT transaction).
912  if (connp->in_status == HTP_STREAM_TUNNEL) {
913  #ifdef HTP_DEBUG
914  fprintf(stderr, "htp_connp_req_data: returning HTP_STREAM_TUNNEL\n");
915  #endif
916 
917  return HTP_STREAM_TUNNEL;
918  }
919 
920  if (connp->out_status == HTP_STREAM_DATA_OTHER) {
921  connp->out_status = HTP_STREAM_DATA;
922  }
923 
924  // Invoke a processor, in a loop, until an error
925  // occurs or until we run out of data. Many processors
926  // will process a request, each pointing to the next
927  // processor that needs to run.
928  for (;;) {
929  #ifdef HTP_DEBUG
930  fprintf(stderr, "htp_connp_req_data: in state=%s, progress=%s\n",
931  htp_connp_in_state_as_string(connp),
932  htp_tx_request_progress_as_string(connp->in_tx));
933  #endif
934 
935  // Return if there's been an error or if we've run out of data. We are relying
936  // on processors to supply error messages, so we'll keep quiet here.
937  htp_status_t rc = connp->in_state(connp);
938  if (rc == HTP_OK) {
939  if (connp->in_status == HTP_STREAM_TUNNEL) {
940  #ifdef HTP_DEBUG
941  fprintf(stderr, "htp_connp_req_data: returning HTP_STREAM_TUNNEL\n");
942  #endif
943 
944  return HTP_STREAM_TUNNEL;
945  }
946 
947  rc = htp_req_handle_state_change(connp);
948  }
949 
950  if (rc != HTP_OK) {
951  // Do we need more data?
952  if ((rc == HTP_DATA) || (rc == HTP_DATA_BUFFER)) {
953  htp_connp_req_receiver_send_data(connp, 0 /* not last */);
954 
955  if (rc == HTP_DATA_BUFFER) {
956  if (htp_connp_req_buffer(connp) != HTP_OK) {
957  connp->in_status = HTP_STREAM_ERROR;
958  return HTP_STREAM_ERROR;
959  }
960  }
961 
962  #ifdef HTP_DEBUG
963  fprintf(stderr, "htp_connp_req_data: returning HTP_STREAM_DATA\n");
964  #endif
965 
966  connp->in_status = HTP_STREAM_DATA;
967 
968  return HTP_STREAM_DATA;
969  }
970 
971  // Check for suspended parsing.
972  if (rc == HTP_DATA_OTHER) {
973  // We might have actually consumed the entire data chunk?
974  if (connp->in_current_read_offset >= connp->in_current_len) {
975  // Do not send STREAM_DATE_DATA_OTHER if we've consumed the entire chunk.
976 
977  #ifdef HTP_DEBUG
978  fprintf(stderr, "htp_connp_req_data: returning HTP_STREAM_DATA (suspended parsing)\n");
979  #endif
980 
981  connp->in_status = HTP_STREAM_DATA;
982 
983  return HTP_STREAM_DATA;
984  } else {
985  // Partial chunk consumption.
986 
987  #ifdef HTP_DEBUG
988  fprintf(stderr, "htp_connp_req_data: returning HTP_STREAM_DATA_OTHER\n");
989  #endif
990 
991  connp->in_status = HTP_STREAM_DATA_OTHER;
992 
993  return HTP_STREAM_DATA_OTHER;
994  }
995  }
996 
997  // Check for the stop signal.
998  if (rc == HTP_STOP) {
999  #ifdef HTP_DEBUG
1000  fprintf(stderr, "htp_connp_req_data: returning HTP_STREAM_STOP\n");
1001  #endif
1002 
1003  connp->in_status = HTP_STREAM_STOP;
1004 
1005  return HTP_STREAM_STOP;
1006  }
1007 
1008  #ifdef HTP_DEBUG
1009  fprintf(stderr, "htp_connp_req_data: returning HTP_STREAM_ERROR\n");
1010  #endif
1011 
1012  // Permanent stream error.
1013  connp->in_status = HTP_STREAM_ERROR;
1014 
1015  return HTP_STREAM_ERROR;
1016  }
1017  }
1018 
1019  // Permanent stream error.
1020  connp->in_status = HTP_STREAM_ERROR;
1021 
1022  return HTP_STREAM_ERROR;
1023 }
