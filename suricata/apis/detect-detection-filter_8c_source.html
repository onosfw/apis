102 /**
103  * \internal
104  * \brief This function is used to parse detection_filter options passed via detection_filter: keyword
105  *
106  * \param rawstr Pointer to the user provided detection_filter options
107  *
108  * \retval df pointer to DetectThresholdData on success
109  * \retval NULL on failure
110  */
111 DetectThresholdData *DetectDetectionFilterParse (char *rawstr)
112 {
113  DetectThresholdData *df = NULL;
114 #define MAX_SUBSTRINGS 30
115  int ret = 0, res = 0;
116  int ov[MAX_SUBSTRINGS];
117  const char *str_ptr = NULL;
118  char *args[6] = { NULL, NULL, NULL, NULL, NULL, NULL};
119  char *copy_str = NULL, *df_opt = NULL;
120  int seconds_found = 0, count_found = 0, track_found = 0;
121  int seconds_pos = 0, count_pos = 0;
122  uint16_t pos = 0;
123  int i = 0;
124  char *saveptr = NULL;
125 
126  copy_str = SCStrdup(rawstr);
127  if (unlikely(copy_str == NULL)) {
128  goto error;
129  }
130 
131  for (pos = 0, df_opt = strtok_r(copy_str,",", &saveptr);
132  pos < strlen(copy_str) && df_opt != NULL;
133  pos++, df_opt = strtok_r(NULL,",", &saveptr))
134  {
135  if(strstr(df_opt,"count"))
136  count_found++;
137  if(strstr(df_opt,"second"))
138  seconds_found++;
139  if(strstr(df_opt,"track"))
140  track_found++;
141  }
142  SCFree(copy_str);
143  copy_str = NULL;
144 
145  if (count_found != 1 || seconds_found != 1 || track_found != 1)
146  goto error;
147 
148  ret = pcre_exec(parse_regex, parse_regex_study, rawstr, strlen(rawstr), 0, 0, ov, MAX_SUBSTRINGS);
149  if (ret < 5) {
150  SCLogError(SC_ERR_PCRE_MATCH, "pcre_exec parse error, ret %" PRId32 ", string %s", ret, rawstr);
151  goto error;
152  }
153 
154  df = SCMalloc(sizeof(DetectThresholdData));
155  if (unlikely(df == NULL))
156  goto error;
157 
158  memset(df,0,sizeof(DetectThresholdData));
159 
160  df->type = TYPE_DETECTION;
161 
162  for (i = 0; i < (ret - 1); i++) {
163  res = pcre_get_substring((char *)rawstr, ov, MAX_SUBSTRINGS, i + 1, &str_ptr);
164  if (res < 0) {
165  SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre_get_substring failed");
166  goto error;
167  }
168 
169  args[i] = (char *)str_ptr;
170 
171  if (strncasecmp(args[i],"by_dst",strlen("by_dst")) == 0)
172  df->track = TRACK_DST;
173  if (strncasecmp(args[i],"by_src",strlen("by_src")) == 0)
174  df->track = TRACK_SRC;
175  if (strncasecmp(args[i],"count",strlen("count")) == 0)
176  count_pos = i+1;
177  if (strncasecmp(args[i],"seconds",strlen("seconds")) == 0)
178  seconds_pos = i+1;
179  }
180 
181  if (args[count_pos] == NULL || args[seconds_pos] == NULL) {
182  goto error;
183  }
184 
185  if (ByteExtractStringUint32(&df->count, 10, strlen(args[count_pos]),
186  args[count_pos]) <= 0) {
187  goto error;
188  }
189 
190  if (ByteExtractStringUint32(&df->seconds, 10, strlen(args[seconds_pos]),
191  args[seconds_pos]) <= 0) {
192  goto error;
193  }
194 
195  if (df->count == 0 || df->seconds == 0) {
196  SCLogError(SC_ERR_INVALID_VALUE, "found an invalid value");
197  goto error;
198  }
199 
200  for (i = 0; i < 6; i++){
201  if (args[i] != NULL)
202  SCFree(args[i]);
203  }
204  return df;
205 
206 error:
207  for (i = 0; i < 6; i++){
208  if (args[i] != NULL)
209  SCFree(args[i]);
210  }
211  if (df != NULL)
212  SCFree(df);
213  return NULL;
214 }

216 /**
217  * \internal
218  * \brief this function is used to add the parsed detection_filter into the current signature
219  *
220  * \param de_ctx pointer to the Detection Engine Context
221  * \param s pointer to the Current Signature
222  * \param m pointer to the Current SigMatch
223  * \param rawstr pointer to the user provided detection_filter options
224  *
225  * \retval 0 on Success
226  * \retval -1 on Failure
227  */
228 int DetectDetectionFilterSetup (DetectEngineCtx *de_ctx, Signature *s, char *rawstr)
229 {
230  SCEnter();
231  DetectThresholdData *df = NULL;
232  SigMatch *sm = NULL;
233  SigMatch *tmpm = NULL;
234 
235  /* checks if there's a previous instance of threshold */
236  tmpm = SigMatchGetLastSMFromLists(s, 2,
237  DETECT_THRESHOLD, s->sm_lists_tail[DETECT_SM_LIST_MATCH]);
238  if (tmpm != NULL) {
239  SCLogError(SC_ERR_INVALID_SIGNATURE, "\"detection_filter\" and \"threshold\" are not allowed in the same rule");
240  SCReturnInt(-1);
241  }
242  /* checks there's no previous instance of detection_filter */
243  tmpm = SigMatchGetLastSMFromLists(s, 2,
244  DETECT_DETECTION_FILTER, s->sm_lists_tail[DETECT_SM_LIST_MATCH]);
245  if (tmpm != NULL) {
246  SCLogError(SC_ERR_INVALID_SIGNATURE, "At most one \"detection_filter\" is allowed per rule");
247  SCReturnInt(-1);
248  }
249 
250  df = DetectDetectionFilterParse(rawstr);
251  if (df == NULL)
252  goto error;
253 
254  sm = SigMatchAlloc();
255  if (sm == NULL)
256  goto error;
257 
258  sm->type = DETECT_DETECTION_FILTER;
259  sm->ctx = (SigMatchCtx *)df;
260 
261  SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_THRESHOLD);
262 
263  return 0;
264 
265 error:
266  if (df) SCFree(df);
267  if (sm) SCFree(sm);
268  return -1;
269 }
