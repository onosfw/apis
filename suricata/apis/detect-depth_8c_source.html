58 static int DetectDepthSetup (DetectEngineCtx *de_ctx, Signature *s, char *depthstr)
59 {
60  char *str = depthstr;
61  char dubbed = 0;
62  SigMatch *pm = NULL;
63  int ret = -1;
64 
65  /* strip "'s */
66  if (depthstr[0] == '\"' && depthstr[strlen(depthstr) - 1] == '\"') {
67  str = SCStrdup(depthstr + 1);
68  if (unlikely(str == NULL))
69  goto end;
70  str[strlen(depthstr) - 2] = '\0';
71  dubbed = 1;
72  }
73 
74  /* retrive the sm to apply the depth against */
75  if (s->list != DETECT_SM_LIST_NOTSET) {
76  pm = SigMatchGetLastSMFromLists(s, 2, DETECT_CONTENT, s->sm_lists_tail[s->list]);
77  } else {
78  pm = SigMatchGetLastSMFromLists(s, 28,
79  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_PMATCH],
80  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_UMATCH],
81  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HRUDMATCH],
82  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HCBDMATCH],
83  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_FILEDATA],
84  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HHDMATCH],
85  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HRHDMATCH],
86  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HMDMATCH],
87  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HCDMATCH],
88  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HSCDMATCH],
89  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HSMDMATCH],
90  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HUADMATCH],
91  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HHHDMATCH],
92  DETECT_CONTENT, s->sm_lists_tail[DETECT_SM_LIST_HRHHDMATCH]);
93  }
94  if (pm == NULL) {
95  SCLogError(SC_ERR_DEPTH_MISSING_CONTENT, "depth needs "
96  "preceding content, uricontent option, http_client_body, "
97  "http_server_body, http_header option, http_raw_header option, "
98  "http_method option, http_cookie, http_raw_uri, "
99  "http_stat_msg, http_stat_code, http_user_agent, "
100  "http_host, http_raw_host or "
101  "file_data/dce_stub_data sticky buffer options");
102  goto end;
103  }
104 
105  /* verify other conditions. */
106  DetectContentData *cd = (DetectContentData *)pm->ctx;
107 
108  if (cd->flags & DETECT_CONTENT_DEPTH) {
109  SCLogError(SC_ERR_INVALID_SIGNATURE, "can't use multiple depths for the same content.");
110  goto end;
111  }
112  if ((cd->flags & DETECT_CONTENT_WITHIN) || (cd->flags & DETECT_CONTENT_DISTANCE)) {
113  SCLogError(SC_ERR_INVALID_SIGNATURE, "can't use a relative "
114  "keyword like within/distance with a absolute "
115  "relative keyword like depth/offset for the same "
116  "content." );
117  goto end;
118  }
119  if (cd->flags & DETECT_CONTENT_NEGATED && cd->flags & DETECT_CONTENT_FAST_PATTERN) {
120  SCLogError(SC_ERR_INVALID_SIGNATURE, "can't have a relative "
121  "negated keyword set along with a fast_pattern");
122  goto end;
123  }
124  if (cd->flags & DETECT_CONTENT_FAST_PATTERN_ONLY) {
125  SCLogError(SC_ERR_INVALID_SIGNATURE, "can't have a relative "
126  "keyword set along with a fast_pattern:only;");
127  goto end;
128  }
129  if (str[0] != '-' && isalpha((unsigned char)str[0])) {
130  SigMatch *bed_sm = DetectByteExtractRetrieveSMVar(str, s);
131  if (bed_sm == NULL) {
132  SCLogError(SC_ERR_INVALID_SIGNATURE, "unknown byte_extract var "
133  "seen in depth - %s\n", str);
134  goto end;
135  }
136  cd->depth = ((DetectByteExtractData *)bed_sm->ctx)->local_id;
137  cd->flags |= DETECT_CONTENT_DEPTH_BE;
138  } else {
139  cd->depth = (uint32_t)atoi(str);
140  if (cd->depth < cd->content_len) {
141  SCLogError(SC_ERR_INVALID_SIGNATURE, "depth - %"PRIu16
142  " smaller than content length - %"PRIu32,
143  cd->depth, cd->content_len);
144  goto end;
145  }
146  /* Now update the real limit, as depth is relative to the offset */
147  cd->depth += cd->offset;
148  }
149  cd->flags |= DETECT_CONTENT_DEPTH;
150 
151  ret = 0;
152  end:
153  if (dubbed)
154  SCFree(str);
155  return ret;
156 }
