65 /**
66  * \brief this function is used to parse filedata options
67  * \brief into the current signature
68  *
69  * \param de_ctx pointer to the Detection Engine Context
70  * \param s pointer to the Current Signature
71  * \param str pointer to the user provided "filestore" option
72  *
73  * \retval 0 on Success
74  * \retval -1 on Failure
75  */
76 static int DetectFiledataSetup (DetectEngineCtx *de_ctx, Signature *s, char *str)
77 {
78  SCEnter();
79 
80  if (!DetectProtoContainsProto(&s->proto, IPPROTO_TCP) &&
81  s->alproto != ALPROTO_UNKNOWN && s->alproto != ALPROTO_HTTP &&
82  s->alproto != ALPROTO_SMTP) {
83  SCLogError(SC_ERR_CONFLICTING_RULE_KEYWORDS, "rule contains conflicting keywords.");
84  return -1;
85  }
86 
87  if (s->alproto == ALPROTO_HTTP && (s->init_flags & SIG_FLAG_INIT_FLOW) &&
88  (s->flags & SIG_FLAG_TOSERVER) && !(s->flags & SIG_FLAG_TOCLIENT)) {
89  SCLogError(SC_ERR_INVALID_SIGNATURE, "Can't use file_data with flow:to_server or from_client with http.");
90  return -1;
91  }
92 
93  if (s->alproto == ALPROTO_SMTP && (s->init_flags & SIG_FLAG_INIT_FLOW) &&
94  !(s->flags & SIG_FLAG_TOSERVER) && (s->flags & SIG_FLAG_TOCLIENT)) {
95  SCLogError(SC_ERR_INVALID_SIGNATURE, "Can't use file_data with flow:to_client or from_server with smtp.");
96  return -1;
97  }
98 
99  s->list = DETECT_SM_LIST_FILEDATA;
100 
101  return 0;
102 }
