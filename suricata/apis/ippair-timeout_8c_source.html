38 /** \internal
39  * \brief See if we can really discard this ippair. Check use_cnt reference.
40  *
41  * \param h ippair
42  * \param ts timestamp
43  *
44  * \retval 0 not timed out just yet
45  * \retval 1 fully timed out, lets kill it
46  */
47 static int IPPairTimedOut(IPPair *h, struct timeval *ts)
48 {
49  int vars = 0;
50 
51  /** never prune a ippair that is used by a packet
52  * we are currently processing in one of the threads */
53  if (SC_ATOMIC_GET(h->use_cnt) > 0) {
54  return 0;
55  }
56 
57  if (IPPairHasBits(h) && IPPairBitsTimedoutCheck(h, ts) == 0) {
58  vars = 1;
59  }
60 
61  if (vars) {
62  return 0;
63  }
64 
65  SCLogDebug("ippair %p timed out", h);
66  return 1;
67 }

69 /**
70  * \internal
71  *
72  * \brief check all ippairs in a hash row for timing out
73  *
74  * \param hb ippair hash row *LOCKED*
75  * \param h last ippair in the hash row
76  * \param ts timestamp
77  *
78  * \retval cnt timed out ippairs
79  */
80 static uint32_t IPPairHashRowTimeout(IPPairHashRow *hb, IPPair *h, struct timeval *ts)
81 {
82  uint32_t cnt = 0;
83 
84  do {
85  if (SCMutexTrylock(&h->m) != 0) {
86  h = h->hprev;
87  continue;
88  }
89 
90  IPPair *next_ippair = h->hprev;
91 
92  /* check if the ippair is fully timed out and
93  * ready to be discarded. */
94  if (IPPairTimedOut(h, ts) == 1) {
95  /* remove from the hash */
96  if (h->hprev != NULL)
97  h->hprev->hnext = h->hnext;
98  if (h->hnext != NULL)
99  h->hnext->hprev = h->hprev;
100  if (hb->head == h)
101  hb->head = h->hnext;
102  if (hb->tail == h)
103  hb->tail = h->hprev;
104 
105  h->hnext = NULL;
106  h->hprev = NULL;
107 
108  IPPairClearMemory (h);
109 
110  /* no one is referring to this ippair, use_cnt 0, removed from hash
111  * so we can unlock it and move it back to the spare queue. */
112  SCMutexUnlock(&h->m);
113 
114  /* move to spare list */
115  IPPairMoveToSpare(h);
116 
117  cnt++;
118  } else {
119  SCMutexUnlock(&h->m);
120  }
121 
122  h = next_ippair;
123  } while (h != NULL);
124 
125  return cnt;
126 }

128 /**
129  * \brief time out ippairs from the hash
130  *
131  * \param ts timestamp
132  *
133  * \retval cnt number of timed out ippair
134  */
135 uint32_t IPPairTimeoutHash(struct timeval *ts)
136 {
137  uint32_t idx = 0;
138  uint32_t cnt = 0;
139 
140  for (idx = 0; idx < ippair_config.hash_size; idx++) {
141  IPPairHashRow *hb = &ippair_hash[idx];
142 
143  if (HRLOCK_TRYLOCK(hb) != 0)
144  continue;
145 
146  /* ippair hash bucket is now locked */
147 
148  if (hb->tail == NULL) {
149  HRLOCK_UNLOCK(hb);

152

154  cnt += IPPairHashRowTimeout(hb, hb->tail, ts);
