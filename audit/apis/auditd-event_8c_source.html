102 int init_event(struct daemon_conf *config)
103 {
104  /* Store the netlink descriptor and config info away */
105  consumer_data.config = config;
106  consumer_data.log_fd = -1;
107 
108  /* Setup IPC mechanisms */
109  pthread_mutex_init(&consumer_data.queue_lock, NULL);
110  pthread_cond_init(&consumer_data.queue_nonempty, NULL);
111 
112  /* Reset the queue */
113  consumer_data.head = consumer_data.tail = NULL;
114 
115  /* Now open the log */
116  if (config->daemonize == D_BACKGROUND) {
117  if (open_audit_log(&consumer_data))
118  return 1;
119  } else {
120  consumer_data.log_fd = 1; // stdout
121  consumer_data.log_file = fdopen(consumer_data.log_fd, "a");
122  if (consumer_data.log_file == NULL) {
123  audit_msg(LOG_ERR,
124  "Error setting up stdout descriptor (%s)",
125  strerror(errno));
126  return 1;
127  }
128  /* Set it to line buffering */
129  setlinebuf(consumer_data.log_file);
130  }
131 
132  /* Create the worker thread */
133  if (pthread_create(&event_thread, NULL,
134  event_thread_main, &consumer_data) < 0) {
135  audit_msg(LOG_ERR, "Couldn't create event thread, exiting");
136  fclose(consumer_data.log_file);
137  return 1;
138  }
139 
140  if (config->daemonize == D_BACKGROUND) {
141  check_log_file_size(&consumer_data);
142  check_excess_logs(&consumer_data);
143  check_space_left(consumer_data.log_fd, &consumer_data);
144  }
145  format_buf = (char *)malloc(MAX_AUDIT_MESSAGE_LENGTH +
146  _POSIX_HOST_NAME_MAX);
147  if (format_buf == NULL) {
148  audit_msg(LOG_ERR, "No memory for formatting, exiting");
149  fclose(consumer_data.log_file);
150  return 1;
151  }
152  return 0;
153 }

157 void enqueue_event(struct auditd_reply_list *rep)
158 {
159  char *buf = NULL;
160  int len;
161 
162  rep->ack_func = 0;
163  rep->ack_data = 0;
164  rep->sequence_id = 0;
165 
166  if (rep->reply.type != AUDIT_DAEMON_RECONFIG) {
167  switch (consumer_data.config->log_format)
168  {
169  case LF_RAW:
170  buf = format_raw(&rep->reply, consumer_data.config);
171  break;
172  case LF_NOLOG:
173  // We need the rotate event to get enqueued
174  if (rep->reply.type != AUDIT_DAEMON_ROTATE ) {
175  // Internal DAEMON messages should be free'd
176  if (rep->reply.type >= AUDIT_FIRST_DAEMON &&
177  rep->reply.type <= AUDIT_LAST_DAEMON)
178  free((void *)rep->reply.message);
179  free(rep);
180  return;
181  }
182  break;
183  default:
184  audit_msg(LOG_ERR,
185  "Illegal log format detected %d",
186  consumer_data.config->log_format);
187  // Internal DAEMON messages should be free'd
188  if (rep->reply.type >= AUDIT_FIRST_DAEMON &&
189  rep->reply.type <= AUDIT_LAST_DAEMON)
190  free((void *)rep->reply.message);
191  free(rep);
192  return;
193  }
194 
195  if (buf) {
196  len = strlen(buf);
197  if (len < MAX_AUDIT_MESSAGE_LENGTH - 1)
198  memcpy(rep->reply.msg.data, buf, len+1);
199  else {
200  // FIXME: is truncation the right thing to do?
201  memcpy(rep->reply.msg.data, buf,
202  MAX_AUDIT_MESSAGE_LENGTH-1);
203  rep->reply.msg.data[MAX_AUDIT_MESSAGE_LENGTH-1] = 0;
204  }
205  }
206  }
207 
208  rep->next = NULL; /* new packet goes at end - so zero this */
209 
210  pthread_mutex_lock(&consumer_data.queue_lock);
211  if (consumer_data.head == NULL) {
212  consumer_data.head = consumer_data.tail = rep;
213  pthread_cond_signal(&consumer_data.queue_nonempty);
214  } else {
215  /* FIXME: wait for room on the queue */
216 
217  /* OK there's room...add it in */
218  consumer_data.tail->next = rep; /* link in at end */
219  consumer_data.tail = rep; /* move end to newest */
220  }
221  pthread_mutex_unlock(&consumer_data.queue_lock);
222 }

274 static void *event_thread_main(void *arg)
275 {
276  struct auditd_consumer_data *data = arg;
277  sigset_t sigs;
278 
279  /* This is a worker thread. Don't handle signals. */
280  sigemptyset(&sigs);
281  sigaddset(&sigs, SIGALRM);
282  sigaddset(&sigs, SIGTERM);
283  sigaddset(&sigs, SIGHUP);
284  sigaddset(&sigs, SIGUSR1);
285  sigaddset(&sigs, SIGUSR2);
286  pthread_sigmask(SIG_SETMASK, &sigs, NULL);
287 
288  while (1) {
289  struct auditd_reply_list *cur;
290  int stop_req = 0;
291 // FIXME: wait for data
292  pthread_mutex_lock(&data->queue_lock);
293  while (data->head == NULL) {
294  pthread_cond_wait(&data->queue_nonempty,
295  &data->queue_lock);
296  }
297 // FIXME: at this point we can use data->head unlocked since it won't change.
298  handle_event(data);
299  cur = data->head;
300 // FIXME: relock at this point
301  if (data->tail == data->head)
302  data->tail = NULL;
303  data->head = data->head->next;
304  if (data->head == NULL && stop &&
305  ( cur->reply.type == AUDIT_DAEMON_END ||
306  cur->reply.type == AUDIT_DAEMON_ABORT) )
307  stop_req = 1;
308  pthread_mutex_unlock(&data->queue_lock);
309 
310  /* Internal DAEMON messages should be free'd */
311  if (cur->reply.type >= AUDIT_FIRST_DAEMON &&
312  cur->reply.type <= AUDIT_LAST_DAEMON) {
313  free((void *)cur->reply.message);
314  }
315  free(cur);
316  if (stop_req)
317  break;
318  }
319  return NULL;
320 }

323 /* This function takes the newly dequeued event and handles it. */
324 static unsigned int count = 0L;
325 static void handle_event(struct auditd_consumer_data *data)
326 {
327  char *buf = data->head->reply.msg.data;
328 
329  if (data->head->reply.type == AUDIT_DAEMON_RECONFIG) {
330  reconfigure(data);
331  switch (consumer_data.config->log_format)
332  {
333  case LF_RAW:
334  buf = format_raw(&data->head->reply, consumer_data.config);
335  break;
336  case LF_NOLOG:
337  return;
338  default:
339  audit_msg(LOG_ERR,
340  "Illegal log format detected %d",
341  consumer_data.config->log_format);
342  return;
343  }
344  } else if (data->head->reply.type == AUDIT_DAEMON_ROTATE) {
345  rotate_logs_now(data);
346  if (consumer_data.config->log_format == LF_NOLOG)
347  return;
348  }
349  if (!logging_suspended) {
350 
351  write_to_log(buf, data);
352 
353  /* See if we need to flush to disk manually */
354  if (data->config->flush == FT_INCREMENTAL) {
355  count++;
356  if ((count % data->config->freq) == 0) {
357  int rc;
358  errno = 0;
359  do {
360  rc = fflush(data->log_file);
361  } while (rc < 0 && errno == EINTR);
362  if (errno) {
363  if (errno == ENOSPC &&
364  fs_space_left == 1) {
365  fs_space_left = 0;
366  do_disk_full_action(data);
367  } else
368  //EIO is only likely failure mode
369  do_disk_error_action("flush",
370  data->config, errno);
371  }
372 
373  /* EIO is only likely failure mode */
374  if ((data->config->daemonize == D_BACKGROUND)&&
375  (fsync(data->log_fd) != 0)) {
376  do_disk_error_action("fsync",
377  data->config, errno);
378  }
379  }
380  }
381  }

397 /* This function writes the given buf to the current log file */
398 static void write_to_log(const char *buf, struct auditd_consumer_data *data)
399 {
400  int rc;
401  FILE *f = data->log_file;
402  struct daemon_conf *config = data->config;
403  int ack_type = AUDIT_RMW_TYPE_ACK;
404  const char *msg = "";
405 
406  /* write it to disk */
407  rc = fprintf(f, "%s\n", buf);
408 
409  /* error? Handle it */
410  if (rc < 0) {
411  if (errno == ENOSPC) {
412  ack_type = AUDIT_RMW_TYPE_DISKFULL;
413  msg = "disk full";
414  send_ack(data, ack_type, msg);
415  if (fs_space_left == 1) {
416  fs_space_left = 0;
417  do_disk_full_action(data);
418  }
419  } else {
420  int saved_errno = errno;
421  ack_type = AUDIT_RMW_TYPE_DISKERROR;
422  msg = "disk write error";
423  send_ack(data, ack_type, msg);
424  do_disk_error_action("write", config, saved_errno);
425  }
426  } else {
427  /* check log file size & space left on partition */
428  if (config->daemonize == D_BACKGROUND) {
429  // If either of these fail, I consider it an
430  // inconvenience as opposed to something that is
431  // actionable. There may be some temporary condition
432  // that the system recovers from. The real error
433  // occurs on write.
434  log_size += rc;
435  check_log_file_size(data);
436  check_space_left(data->log_fd, data);
437  }
438 
439  if (fs_space_warning)
440  ack_type = AUDIT_RMW_TYPE_DISKLOW;
441  send_ack(data, ack_type, msg);
442  disk_err_warning = 0;
443  }
444 }

446 static void check_log_file_size(struct auditd_consumer_data *data)
447 {
448  struct daemon_conf *config = data->config;
449 
450  /* did we cross the size limit? */
451  off_t sz = log_size / MEGABYTE;
452 
453  if (sz >= config->max_log_size && (config->daemonize == D_BACKGROUND)) {
454  switch (config->max_log_size_action)
455  {
456  case SZ_IGNORE:
457  break;
458  case SZ_SYSLOG:
459  audit_msg(LOG_ERR,
460  "Audit daemon log file is larger than max size");
461  break;
462  case SZ_SUSPEND:
463  audit_msg(LOG_ERR,
464  "Audit daemon is suspending logging due to logfile size.");
465  logging_suspended = 1;
466  break;
467  case SZ_ROTATE:
468  if (data->config->num_logs > 1) {
469  audit_msg(LOG_NOTICE,
470  "Audit daemon rotating log files");
471  rotate_logs(data, 0);
472  }
473  break;
474  case SZ_KEEP_LOGS:
475  audit_msg(LOG_NOTICE,
476  "Audit daemon rotating log files with keep option");
477  shift_logs(data);
478  break;
479  default:
480  audit_msg(LOG_ALERT,
481  "Audit daemon log file is larger than max size and unknown action requested");
482  break;
483  }
484  }

487 static void check_space_left(int lfd, struct auditd_consumer_data *data)
488 {
489  int rc;
490  struct statfs buf;
491  struct daemon_conf *config = data->config;
492 
493  rc = fstatfs(lfd, &buf);
494  if (rc == 0) {
495  if (buf.f_bavail < 5) {
496  /* we won't consume the last 5 blocks */
497  fs_space_left = 0;
498  do_disk_full_action(data);
499  } else {
500  unsigned long blocks;
501  unsigned long block_size = buf.f_bsize;
502  blocks = config->space_left * (MEGABYTE/block_size);
503  if (buf.f_bavail < blocks) {
504  if (fs_space_warning == 0) {
505  do_space_left_action(data, 0);
506  fs_space_warning = 1;
507  }
508  } else if (fs_space_warning &&
509  config->space_left_action == FA_SYSLOG){
510  // Auto reset only if failure action is syslog
511  fs_space_warning = 0;
512  }
513  blocks=config->admin_space_left * (MEGABYTE/block_size);
514  if (buf.f_bavail < blocks) {
515  if (fs_admin_space_warning == 0) {
516  do_space_left_action(data, 1);
517  fs_admin_space_warning = 1;
518  }
519  } else if (fs_admin_space_warning &&
520  config->admin_space_left_action == FA_SYSLOG) {
521  // Auto reset only if failure action is syslog
522  fs_admin_space_warning = 0;
523  }
524  }
525  }
526  else audit_msg(LOG_DEBUG, "fstatfs returned:%d, %s", rc,
527  strerror(errno));
528 }

567  config->action_mail_acct);
568  audit_msg(LOG_ALERT,
569  "Audit daemon is very low on disk space for logging");
570  }
571  break;
572  case FA_EXEC:
573  if (admin)
574  safe_exec(config->admin_space_left_exe);
575  else
576  safe_exec(config->space_left_exe);
577  break;
578  case FA_SUSPEND:

587  break;
588  case FA_HALT:

591  change_runlevel(HALT);
592  break;
593  default:
594  audit_msg(LOG_ALERT,
595  "Audit daemon is low on disk space for logging and unknown action requested");

668  case FA_SINGLE:
669  audit_msg(LOG_ALERT,
670  "The audit daemon is now changing the system to single user mode due to previously mentioned write error");
671  change_runlevel(SINGLE);
672  break;
673  case FA_HALT:
674  audit_msg(LOG_ALERT,
675  "The audit daemon is now halting the system due to previously mentioned write error.");
676  change_runlevel(HALT);
677  break;
678  default:
679  audit_msg(LOG_ALERT,
680  "Unknown disk error action requested");
681  break;
682  }

729 static void rotate_logs(struct auditd_consumer_data *data,
730  unsigned int num_logs)
731 {
732  int rc;
733  unsigned int len, i;
734  char *oldname, *newname;
735 
736  if (data->config->max_log_size_action == SZ_ROTATE &&
737  data->config->num_logs < 2)
738  return;
739 
740  /* Close audit file. fchmod and fchown errors are not fatal because we
741  * already adjusted log file permissions and ownership when opening the
742  * log file. */
743  if (fchmod(data->log_fd, data->config->log_group ? S_IRUSR|S_IRGRP :
744  S_IRUSR) < 0) {
745  audit_msg(LOG_NOTICE, "Couldn't change permissions while "
746  "rotating log file (%s)", strerror(errno));
747  }
748  if (fchown(data->log_fd, 0, data->config->log_group) < 0) {
749  audit_msg(LOG_NOTICE, "Couldn't change ownership while "
750  "rotating log file (%s)", strerror(errno));
751  }
752  fclose(data->log_file);
753 
754  /* Rotate */
755  len = strlen(data->config->log_file) + 16;
756  oldname = (char *)malloc(len);
757  if (oldname == NULL) { /* Not fatal - just messy */
758  audit_msg(LOG_ERR, "No memory rotating logs");
759  logging_suspended = 1;
760  return;
761  }
762  newname = (char *)malloc(len);
763  if (newname == NULL) { /* Not fatal - just messy */
764  audit_msg(LOG_ERR, "No memory rotating logs");
765  free(oldname);
766  logging_suspended = 1;
767  return;
768  }
769 
770  /* If we are rotating, get number from config */
771  if (num_logs == 0)
772  num_logs = data->config->num_logs;
773 
774  /* Handle this case first since it will not enter the for loop */
775  if (num_logs == 2)
776  snprintf(oldname, len, "%s.1", data->config->log_file);
777 
778  for (i=num_logs - 1; i>1; i--) {
779  snprintf(oldname, len, "%s.%d", data->config->log_file, i-1);
780  snprintf(newname, len, "%s.%d", data->config->log_file, i);
781  /* if the old file exists */
782  rc = rename(oldname, newname);
783  if (rc == -1 && errno != ENOENT) {
784  // Likely errors: ENOSPC, ENOMEM, EBUSY
785  int saved_errno = errno;
786  audit_msg(LOG_ERR,
787  "Error rotating logs from %s to %s (%s)",
788  oldname, newname, strerror(errno));
789  if (saved_errno == ENOSPC && fs_space_left == 1) {
790  fs_space_left = 0;
791  do_disk_full_action(data);
792  } else
793  do_disk_error_action("rotate", data->config,
794  saved_errno);
795  }
796  }
797  free(newname);
798 
799  /* At this point, oldname should point to lowest number - use it */
800  newname = oldname;
801  rc = rename(data->config->log_file, newname);
802  if (rc == -1 && errno != ENOENT) {
803  // Likely errors: ENOSPC, ENOMEM, EBUSY
804  int saved_errno = errno;
805  audit_msg(LOG_ERR, "Error rotating logs from %s to %s (%s)",
806  data->config->log_file, newname, strerror(errno));
807  if (saved_errno == ENOSPC && fs_space_left == 1) {
808  fs_space_left = 0;
809  do_disk_full_action(data);
810  } else
811  do_disk_error_action("rotate2", data->config,
812  saved_errno);
813 
814  /* At this point, we've failed to rotate the original log.
815  * So, let's make the old log writable and try again next
816  * time */
817  chmod(data->config->log_file,
818  data->config->log_group ? S_IWUSR|S_IRUSR|S_IRGRP :
819  S_IWUSR|S_IRUSR);
820  }
821  free(newname);
822 
823  /* open new audit file */
824  if (open_audit_log(data)) {
825  int saved_errno = errno;
826  audit_msg(LOG_NOTICE,
827  "Could not reopen a log after rotating.");
828  logging_suspended = 1;
829  do_disk_error_action("reopen", data->config, saved_errno);
830  }
831 }

833 static int last_log = 1;
834 static void shift_logs(struct auditd_consumer_data *data)
835 {
836  // The way this has to work is to start scanning from .1 up until
837  // no file is found. Then do the rotate algorithm using that number
838  // instead of log_max.
839  unsigned int num_logs, len;
840  char *name;
841 
842  len = strlen(data->config->log_file) + 16;
843  name = (char *)malloc(len);
844  if (name == NULL) { /* Not fatal - just messy */
845  audit_msg(LOG_ERR, "No memory shifting logs");
846  return;
847  }
848 
849  // Find last log
850  num_logs = last_log;
851  while (num_logs) {
852  snprintf(name, len, "%s.%d", data->config->log_file,
853  num_logs);
854  if (access(name, R_OK) != 0)
855  break;
856  num_logs++;
857  }
858 
859  /* Our last known file disappeared, start over... */
860  if (num_logs <= last_log && last_log > 1) {
861  audit_msg(LOG_WARNING, "Last known log disappeared (%s)", name);
862  num_logs = last_log = 1;
863  while (num_logs) {
864  snprintf(name, len, "%s.%d", data->config->log_file,
865  num_logs);
866  if (access(name, R_OK) != 0)
867  break;
868  num_logs++;
869  }
870  audit_msg(LOG_INFO, "Next log to use will be %s", name);
871  }
872  last_log = num_logs;
873  rotate_logs(data, num_logs+1);
874  free(name);
875 }

877 /*
878  * This function handles opening a descriptor for the audit log
879  * file and ensuring the correct options are applied to the descriptor.
880  * It returns 0 on success and 1 on failure.
881  */
882 static int open_audit_log(struct auditd_consumer_data *data)
883 {
884  int flags, lfd;
885 
886  // Likely errors for open: Almost anything
887  // Likely errors on rotate: ENFILE, ENOMEM, ENOSPC
888 retry:
889  lfd = open(data->config->log_file, O_WRONLY|O_APPEND|O_NOFOLLOW);
890  if (lfd < 0) {
891  if (errno == ENOENT) {
892  lfd = create_log_file(data->config->log_file);
893  if (lfd < 0) {
894  audit_msg(LOG_ERR,
895  "Couldn't create log file %s (%s)",
896  data->config->log_file,
897  strerror(errno));
898  return 1;
899  }
900  close(lfd);
901  lfd = open(data->config->log_file,
902  O_WRONLY|O_APPEND|O_NOFOLLOW);
903  log_size = 0;
904  } else if (errno == ENFILE) {
905  // All system descriptors used, try again...
906  goto retry;
907  }
908  if (lfd < 0) {
909  audit_msg(LOG_ERR, "Couldn't open log file %s (%s)",
910  data->config->log_file, strerror(errno));
911  return 1;
912  }
913  } else {
914  // Get initial size
915  struct stat st;
916 
917  int rc = fstat(lfd, &st);
918  if (rc == 0)
919  log_size = st.st_size;
920  else {
921  close(lfd);
922  return 1;
923  }
924  }
925 
926  if (fcntl(lfd, F_SETFD, FD_CLOEXEC) == -1) {
927  close(lfd);
928  audit_msg(LOG_ERR, "Error setting log file CLOEXEC flag (%s)",
929  strerror(errno));
930  return 1;
931  }
932  if (data->config->flush == FT_DATA) {
933  flags = fcntl(lfd, F_GETFL);
934  if (flags < 0) {
935  audit_msg(LOG_ERR, "Couldn't get log file flags (%s)",
936  strerror(errno));
937  close(lfd);
938  return 1;
939  }
940  if (fcntl(lfd, F_SETFL, flags|O_DSYNC) < 0) {
941  audit_msg(LOG_ERR,
942  "Couldn't set data sync mode on log file (%s)",
943  strerror(errno));
944  close(lfd);
945  return 1;
946  }
947  }
948  else if (data->config->flush == FT_SYNC){
949  flags = fcntl(lfd, F_GETFL);
950  if (flags < 0) {
951  audit_msg(LOG_ERR, "Couldn't get log file flags (%s)",
952  strerror(errno));
953  close(lfd);
954  return 1;
955  }
956  if (fcntl(lfd, F_SETFL, flags|O_SYNC) < 0) {
957  audit_msg(LOG_ERR,
958  "Couldn't set sync mode on log file (%s)",
959  strerror(errno));
960  close(lfd);
961  return 1;
962  }
963  }
964  if (fchmod(lfd, data->config->log_group ? S_IRUSR|S_IWUSR|S_IRGRP :
965  S_IRUSR|S_IWUSR) < 0) {
966  audit_msg(LOG_ERR,
967  "Couldn't change permissions of log file (%s)",
968  strerror(errno));
969  close(lfd);
970  return 1;
971  }
972  if (fchown(lfd, 0, data->config->log_group) < 0) {
973  audit_msg(LOG_ERR, "Couldn't change ownership of log file (%s)",
974  strerror(errno));
975  close(lfd);
976  return 1;
977  }
978 
979  data->log_fd = lfd;
980  data->log_file = fdopen(lfd, "a");
981  if (data->log_file == NULL) {
982  audit_msg(LOG_ERR, "Error setting up log descriptor (%s)",
983  strerror(errno));
984  close(lfd);
985  return 1;
986  }
987 
988  /* Set it to line buffering */
989  setlinebuf(consumer_data.log_file);
990  return 0;
991 }

993 static void change_runlevel(const char *level)
994 {
995  char *argv[3];
996  int pid;
997  struct sigaction sa;
998  static const char *init_pgm = "/sbin/init";
999 
1000  pid = fork();
1001  if (pid < 0) {
1002  audit_msg(LOG_ALERT,
1003  "Audit daemon failed to fork switching runlevels");
1004  return;
1005  }
1006  if (pid) /* Parent */
1007  return;
1008  /* Child */
1009  sigfillset (&sa.sa_mask);
1010  sigprocmask (SIG_UNBLOCK, &sa.sa_mask, 0);
1011 
1012  argv[0] = (char *)init_pgm;
1013  argv[1] = (char *)level;
1014  argv[2] = NULL;
1015  execve(init_pgm, argv, NULL);
1016  audit_msg(LOG_ALERT, "Audit daemon failed to exec %s", init_pgm);
1017  exit(1);
1018 }

1051 /*

1056 static char *format_raw(const struct audit_reply *rep,

1098  "type=%s msg=%.*s", type, len, message);
1099

1101  ptr = format_buf;
1102  while ((ptr = strchr(ptr, 0x0A)) != NULL)
1103  *ptr = ' ';
1104 
1105  /* Trim trailing space off since it wastes space */
1106  if (format_buf[nlen-1] == ' ')
1107  format_buf[nlen-1] = 0;
1108  }

1112 static void reconfigure(struct auditd_consumer_data *data)
1113 {
1114  struct daemon_conf *nconf = data->head->reply.conf;
1115  struct daemon_conf *oconf = data->config;
1116  uid_t uid = nconf->sender_uid;
1117  pid_t pid = nconf->sender_pid;
1118  const char *ctx = nconf->sender_ctx;
1119  struct timeval tv;
1120  char txt[MAX_AUDIT_MESSAGE_LENGTH];
1121  char date[40];
1122  unsigned int seq_num;
1123  int need_size_check = 0, need_reopen = 0, need_space_check = 0;
1124 
1125  snprintf(txt, sizeof(txt),
1126  "config change requested by pid=%d auid=%u subj=%s",
1127  pid, uid, ctx);
1128  audit_msg(LOG_NOTICE, "%s", txt);
1129 
1130  /* Do the reconfiguring. These are done in a specific
1131  * order from least invasive to most invasive. We will
1132  * start with general system parameters. */
1133 
1134  // start with disk error action.
1135  oconf->disk_error_action = nconf->disk_error_action;
1136  free((char *)oconf->disk_error_exe);
1137  oconf->disk_error_exe = nconf->disk_error_exe;

1149

1170  (oconf->node_name && nconf->node_name &&

1200  do_disk_error_action("reconfig", data->config,
1201  saved_errno);
1202  }
1203  }
1204  // have one, but none after this
1205  else if (nconf->dispatcher == NULL) {
1206  shutdown_dispatcher();

1208  oconf->dispatcher = NULL;
1209  }

1211  else if (strcmp(oconf->dispatcher, nconf->dispatcher)) {
1212  shutdown_dispatcher();
1213  free((char *)oconf->dispatcher);
1214  oconf->dispatcher = strdup(nconf->dispatcher);
1215  if (oconf->dispatcher == NULL) {
1216  int saved_errno = errno;
1217  audit_msg(LOG_NOTICE,
1218  "Could not allocate dispatcher memory"
1219  " in reconfigure");
1220  // Likely errors: ENOMEM
1221  do_disk_error_action("reconfig", data->config,
1222  saved_errno);
1223  }

1225  int saved_errno = errno;
1226  audit_msg(LOG_NOTICE,
1227  "Could not start dispatcher %s"
1228  " in reconfigure", oconf->dispatcher);
1229  // Likely errors: Socketpairs or exec perms
1230  do_disk_error_action("reconfig", data->config,
1231  saved_errno);
1232  }
1233  }
1234  // they are the same app - just signal it
1235  else {
1236  reconfigure_dispatcher(oconf);
1237  free((char *)nconf->dispatcher);
1238  nconf->dispatcher = NULL;
1239  }
1240  }
1241 
1242  // network listener
1243  auditd_tcp_listen_reconfigure(nconf, oconf);

1246  * a single log file. */
1247 
1248  // max logfile action

1250  oconf->max_log_size_action = nconf->max_log_size_action;
1251  need_size_check = 1;

1255  if (oconf->max_log_size != nconf->max_log_size) {
1256  oconf->max_log_size = nconf->max_log_size;
1257  need_size_check = 1;
1258  }

1262  check_log_file_size(data);
1263  }

1266  if (oconf->flush != nconf->flush) {
1267  oconf->flush = nconf->flush;
1268  need_reopen = 1;
1269  }
1270 
1271  // logfile
1272  if (strcmp(oconf->log_file, nconf->log_file)) {
1273  free((void *)oconf->log_file);
1274  oconf->log_file = nconf->log_file;
1275  need_reopen = 1;
1276  need_space_check = 1; // might be on new partition
1277  } else
1278  free((void *)nconf->log_file);

1287  // Likely errors: ENOMEM, ENOSPC

1289  saved_errno);

1291  logging_suspended = 0;
1292  check_log_file_size(data);
1293  }
1294  }

1297  * related to the amount of space on the partition. */
