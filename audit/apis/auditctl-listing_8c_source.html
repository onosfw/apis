40  * Returns 1 if rule should be printed & 0 if not
41  */
42 int key_match(const struct audit_rule_data *r)
43 {
44  int i;
45  size_t boffset = 0;
46 
47  if (key[0] == 0)
48  return 1;
49 
50  // At this point, we have a key
51  for (i = 0; i < r->field_count; i++) {
52  int field = r->fields[i] & ~AUDIT_OPERATORS;
53  if (field == AUDIT_FILTERKEY) {
54  char *keyptr;
55  if (asprintf(&keyptr, "%.*s", r->values[i],
56  &r->buf[boffset]) < 0)
57  keyptr = NULL;
58  else if (strstr(keyptr, key)) {
59  free(keyptr);
60  return 1;
61  }
62  free(keyptr);
63  }
64  if (((field >= AUDIT_SUBJ_USER && field <= AUDIT_OBJ_LEV_HIGH)
65  && field != AUDIT_PPID) || field == AUDIT_WATCH ||
66  field == AUDIT_DIR || field == AUDIT_FILTERKEY) {
67  boffset += r->values[i];
68  }
69  }
70  return 0;
71 }

73 /*
74  * This function detects if we have a watch. A watch is detected when we
75  * have syscall == all and a perm field.
76  */
77 static int is_watch(const struct audit_rule_data *r)
78 {
79  int i, perm = 0, all = 1;
80 
81  for (i = 0; i < r->field_count; i++) {
82  int field = r->fields[i] & ~AUDIT_OPERATORS;
83  if (field == AUDIT_PERM)
84  perm = 1;
85  // Watches can have only 4 field types
86  if (field != AUDIT_PERM && field != AUDIT_FILTERKEY &&
87  field != AUDIT_DIR && field != AUDIT_WATCH)
88  return 0;
89  }
90 
91  if (((r->flags & AUDIT_FILTER_MASK) != AUDIT_FILTER_USER) &&
92  ((r->flags & AUDIT_FILTER_MASK) != AUDIT_FILTER_TASK) &&
93  ((r->flags & AUDIT_FILTER_MASK) != AUDIT_FILTER_EXCLUDE)) {
94  for (i = 0; i < (AUDIT_BITMASK_SIZE-1); i++) {
95  if (r->mask[i] != (uint32_t)~0) {
96  all = 0;
97  break;
98  }
99  }
100  }
101  if (perm && all)
102  return 1;
103  return 0;
104 }

131 static int print_syscall(const struct audit_rule_data *r, unsigned int *sc)
132 {
133  int count = 0;
134  int all = 1;
135  unsigned int i;
136  int machine = audit_detect_machine();
137 
138  /* Rules on the following filters do not take a syscall */
139  if (((r->flags & AUDIT_FILTER_MASK) == AUDIT_FILTER_USER) ||
140  ((r->flags & AUDIT_FILTER_MASK) == AUDIT_FILTER_TASK) ||
141  ((r->flags &AUDIT_FILTER_MASK) == AUDIT_FILTER_EXCLUDE))
142  return 0;
143 
144  /* See if its all or specific syscalls */
145  for (i = 0; i < (AUDIT_BITMASK_SIZE-1); i++) {
146  if (r->mask[i] != (uint32_t)~0) {
147  all = 0;
148  break;
149  }
150  }
151 
152  if (all) {
153  printf(" -S all");
154  count = i;
155  } else for (i = 0; i < AUDIT_BITMASK_SIZE * 32; i++) {
156  int word = AUDIT_WORD(i);
157  int bit = AUDIT_BIT(i);
158  if (r->mask[word] & bit) {
159  const char *ptr;
160  if (_audit_elf)
161  machine = audit_elf_to_machine(_audit_elf);
162  if (machine < 0)
163  ptr = NULL;
164  else
165  ptr = audit_syscall_to_name(i, machine);
166  if (!count)
167  printf(" -S ");
168  if (ptr)
169  printf("%s%s", !count ? "" : ",", ptr);
170  else
171  printf("%s%d", !count ? "" : ",", i);
172  count++;
173  *sc = i;
174  }
175  }
176  return count;
177 }

286 /*
287  * This function prints 1 rule from the kernel reply
288  */
289 static void print_rule(const struct audit_rule_data *r)
290 {
291  unsigned int i, count = 0, sc = 0;
292  size_t boffset = 0;
293  int mach = -1, watch = is_watch(r);
294  unsigned long long a0 = 0, a1 = 0;
295 
296  if (!watch) { /* This is syscall auditing */
297  printf("-a %s,%s",
298  audit_action_to_name((int)r->action),
299  audit_flag_to_name(r->flags));
300 
301  // Now find the arch and print it
302  for (i = 0; i < r->field_count; i++) {
303  int field = r->fields[i] & ~AUDIT_OPERATORS;
304  if (field == AUDIT_ARCH) {
305  int op = r->fieldflags[i] & AUDIT_OPERATORS;
306  mach = print_arch(r->values[i], op);
307  }
308  }
309  // And last do the syscalls
310  count = print_syscall(r, &sc);
311  }
312 
313  // Now iterate over the fields
314  for (i = 0; i < r->field_count; i++) {
315  const char *name;
316  int op = r->fieldflags[i] & AUDIT_OPERATORS;
317  int field = r->fields[i] & ~AUDIT_OPERATORS;
318 
319  if (field == AUDIT_ARCH)
320  continue; // already printed
321 
322  name = audit_field_to_name(field);
323  if (name) {
324  // Special cases to print the different field types
325  // in a meaningful way.
326  if (field == AUDIT_MSGTYPE) {
327  if (!audit_msg_type_to_name(r->values[i]))
328  printf(" -F %s%s%d", name,
329  audit_operator_to_symbol(op),
330  r->values[i]);
331  else
332  printf(" -F %s%s%s", name,
333  audit_operator_to_symbol(op),
334  audit_msg_type_to_name(
335  r->values[i]));
336  } else if ((field >= AUDIT_SUBJ_USER &&
337  field <= AUDIT_OBJ_LEV_HIGH)
338  && field != AUDIT_PPID) {
339  printf(" -F %s%s%.*s", name,
340  audit_operator_to_symbol(op),
341  r->values[i], &r->buf[boffset]);
342  boffset += r->values[i];
343  } else if (field == AUDIT_WATCH) {
344  if (watch)
345  printf("-w %.*s", r->values[i],
346  &r->buf[boffset]);
347  else
348  printf(" -F path=%.*s", r->values[i],
349  &r->buf[boffset]);
350  boffset += r->values[i];
351  } else if (field == AUDIT_DIR) {
352  if (watch)
353  printf("-w %.*s/", r->values[i],
354  &r->buf[boffset]);
355  else
356  printf(" -F dir=%.*s", r->values[i],
357  &r->buf[boffset]);
358 
359  boffset += r->values[i];
360  } else if (field == AUDIT_FILTERKEY) {
361  char *rkey, *ptr, *saved;
362  if (asprintf(&rkey, "%.*s", r->values[i],
363  &r->buf[boffset]) < 0)
364  rkey = NULL;
365  boffset += r->values[i];
366  ptr = strtok_r(rkey, key_sep, &saved);
367  while (ptr) {
368  if (watch)
369  printf(" -k %s", ptr);
370  else
371  printf(" -F key=%s", ptr);
372  ptr = strtok_r(NULL, key_sep, &saved);
373  }
374  free(rkey);
375  } else if (field == AUDIT_PERM) {
376  char perms[5];
377  int val=r->values[i];
378  perms[0] = 0;
379  if (val & AUDIT_PERM_READ)
380  strcat(perms, "r");
381  if (val & AUDIT_PERM_WRITE)
382  strcat(perms, "w");
383  if (val & AUDIT_PERM_EXEC)
384  strcat(perms, "x");
385  if (val & AUDIT_PERM_ATTR)
386  strcat(perms, "a");
387  if (watch)
388  printf(" -p %s", perms);
389  else
390  printf(" -F perm=%s", perms);
391  } else if (field == AUDIT_INODE) {
392  // This is unsigned
393  printf(" -F %s%s%u", name,
394  audit_operator_to_symbol(op),
395  r->values[i]);
396  } else if (field == AUDIT_FIELD_COMPARE) {
397  print_field_cmp(r->values[i], op);
398  } else if (field >= AUDIT_ARG0 && field <= AUDIT_ARG3){
399  if (field == AUDIT_ARG0)
400  a0 = r->values[i];
401  else if (field == AUDIT_ARG1)
402  a1 = r->values[i];
403 
404  // Show these as hex
405  if (count > 1 || interpret == 0)
406  printf(" -F %s%s0x%X", name,
407  audit_operator_to_symbol(op),
408  r->values[i]);
409  else { // Use ignore to mean interpret
410  const char *out;
411  idata id;
412  char val[32];
413  int type;
414 
415  id.syscall = sc;
416  id.machine = mach;
417  id.a0 = a0;
418  id.a1 = a1;
419  id.name = name;
420  snprintf(val, 32, "%x", r->values[i]);
421  id.val = val;
422  type = auparse_interp_adjust_type(
423  AUDIT_SYSCALL, name, val);
424  out = auparse_do_interpretation(type,
425  &id);
426  printf(" -F %s%s%s", name,
427  audit_operator_to_symbol(op),
428  out);
429  free((void *)out);
430  }
431  } else if (field == AUDIT_EXIT) {
432  int e = abs((int)r->values[i]);
433  const char *err = audit_errno_to_name(e);
434 
435  if (((int)r->values[i] < 0) && err)
436  printf(" -F %s%s-%s", name,
437  audit_operator_to_symbol(op),
438  err);
439  else
440  printf(" -F %s%s%d", name,
441  audit_operator_to_symbol(op),
442  (int)r->values[i]);
443  } else {
444  // The default is signed decimal
445  printf(" -F %s%s%d", name,
446  audit_operator_to_symbol(op),
447  r->values[i]);
448  }
449  } else {
450  // The field name is unknown
451  printf(" f%d%s%d", r->fields[i],
452  audit_operator_to_symbol(op),
453  r->values[i]);
454  }
455  }
456  printf("\n");
457 }

495 /*
496  * This function interprets the reply and prints it to stdout. It returns
497  * 0 if no more should be read and 1 to indicate that more messages of this
498  * type may need to be read.
499  */
500 int audit_print_reply(struct audit_reply *rep, int fd)
501 {
502  _audit_elf = 0;
503 
504  switch (rep->type) {
505  case NLMSG_NOOP:
506  return 1;
507  case NLMSG_DONE:
508  // Close the socket so kernel can do other things
509  audit_close(fd);
510  if (printed == 0)
511  printf("No rules\n");
512  else {
513  lnode *n;
514  list_first(&l);
515  n = l.cur;
516  while (n) {
517  print_rule(n->r);
518  n = list_next(&l);
519  }
520  list_clear(&l);
521  }
522  break;
523  case NLMSG_ERROR:
524  printf("NLMSG_ERROR %d (%s)\n",
525  -rep->error->error,
526  strerror(-rep->error->error));
527  printed = 1;
528  break;
529  case AUDIT_GET:
530  if (interpret)
531  printf("enabled %s\nfailure %s\n",
532  get_enable(rep->status->enabled),
533  get_failure(rep->status->failure));
534  else
535  printf("enabled %u\nfailure %u\n",
536  rep->status->enabled, rep->status->failure);
537  printf("pid %u\nrate_limit %u\nbacklog_limit %u\n"
538  "lost %u\nbacklog %u\n",
539  rep->status->pid, rep->status->rate_limit,
540  rep->status->backlog_limit, rep->status->lost,
541  rep->status->backlog);
542 #if HAVE_DECL_AUDIT_VERSION_BACKLOG_WAIT_TIME
543  printf("backlog_wait_time %u\n",
544  rep->status->backlog_wait_time);
545 #endif
546  printed = 1;
547  break;
548 #if HAVE_DECL_AUDIT_FEATURE_VERSION
549  case AUDIT_GET_FEATURE:
550  {
551  uint32_t mask = AUDIT_FEATURE_TO_MASK(AUDIT_FEATURE_LOGINUID_IMMUTABLE);
552  if (rep->features->mask & mask)
553  printf("loginuid_immutable %u %s\n",
554  !!(rep->features->features & mask),
555  rep->features->lock & mask ? "locked" :
556  "unlocked");
557  }
558  printed = 1;
559  break;
560 #endif
561  case AUDIT_LIST_RULES:
562  list_requested = 0;
563  if (key_match(rep->ruledata))
564  list_append(&l, rep->ruledata,
565  sizeof(struct audit_rule_data) +
566  rep->ruledata->buflen);
567  printed = 1;
568  return 1;
569  default:
570  printf("Unknown: type=%d, len=%d\n", rep->type,
571  rep->nlh->nlmsg_len);
572  printed = 1;
573  break;
574  }
575  return 0;
576 }
