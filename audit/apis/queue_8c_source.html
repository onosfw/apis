90 static void do_overflow_action(struct daemon_conf *config)
91 {
92  switch (config->overflow_action)
93  {
94  case O_IGNORE:
95  break;
96  case O_SYSLOG:
97  if (queue_full_warning < QUEUE_FULL_LIMIT) {
98  syslog(LOG_ERR,
99  "queue is full - dropping event");
100  queue_full_warning++;
101  if (queue_full_warning == QUEUE_FULL_LIMIT)
102  syslog(LOG_ERR,
103  "audispd queue full reporting "
104  "limit reached - ending "
105  "dropped event notifications");
106  }
107  break;
108  case O_SUSPEND:
109  syslog(LOG_ALERT,
110  "Audispd is suspending event processing due to overflowing its queue.");
111  processing_suspended = 1;
112  break;
113  case O_SINGLE:
114  syslog(LOG_ALERT,
115  "Audisp is now changing the system to single user mode due to overflowing its queue");
116  change_runlevel(SINGLE);
117  break;
118  case O_HALT:
119  syslog(LOG_ALERT,
120  "Audispd is now halting the system due to overflowing its queue");
121  change_runlevel(HALT);
122  break;
123  default:
124  syslog(LOG_ALERT, "Unknown overflow action requested");
125  break;
126  }
127 }

129 void enqueue(event_t *e, struct daemon_conf *config)
130 {
131  unsigned int n, retry_cnt = 0;
132 
133  if (processing_suspended) {
134  free(e);
135  return;
136  }
137 
138 retry:
139  // We allow 3 retries and then its over
140  if (retry_cnt > 3) {
141  do_overflow_action(config);
142  free(e);
143  return;
144  }
145  pthread_mutex_lock(&queue_lock);
146 
147  // OK, have lock add event
148  n = q_next%q_depth;
149  if (q[n] == NULL) {
150  q[n] = e;
151  q_next = (n+1) % q_depth;
152  pthread_cond_signal(&queue_nonempty);
153  pthread_mutex_unlock(&queue_lock);
154  } else {
155  pthread_mutex_unlock(&queue_lock);
156  struct timespec ts;
157  ts.tv_sec = 0;
158  ts.tv_nsec = 2 * 1000 * 1000; // 2 milliseconds
159  nanosleep(&ts, NULL); /* Let other thread try to log it. */
160  retry_cnt++;
161  goto retry;
162  }

181

185  q[n] = NULL;

187  } else
188  e = NULL;
189 
190  pthread_mutex_unlock(&queue_lock);
191 
192  // Process the event
193  return e;
