484 void print_per_event_item(llist *l)
485 {
486  char buf[128];
487  char name[64];
488  char date[32];
489  struct tm *tv;
490 
491  // The beginning is common to all reports
492  tv = localtime(&l->e.sec);
493  strftime(date, sizeof(date), "%x %T", tv);
494  if (report_type != RPT_AVC) {
495  line_item++;
496  printf("%u. %s ", line_item, date);
497  }
498 
499  switch (report_type)
500  {
501  case RPT_AVC:
502  alist_find_avc(l->s.avc);
503  do {
504  anode *an = l->s.avc->cur;
505  line_item++;
506  printf("%u. %s ", line_item, date);
507  // command subject syscall action obj res event
508  safe_print_string(l->s.comm ? l->s.comm : "?", 0);
509  printf(" %s %s %s %s %s %s %lu\n",
510  an->scontext,
511  aulookup_syscall(l, buf,sizeof(buf)),
512  an->avc_class, an->avc_perm,
513  an->tcontext, aulookup_result(an->avc_result),
514  l->e.serial);
515 //printf("items:%d\n", l->s.avc->cnt);
516  } while (alist_next_avc(l->s.avc));
517  break;
518  case RPT_CONFIG:
519  // FIXME:who, action, what, outcome, event
520  // NOW: type auid success event
521  printf("%s %s %s %lu\n",
522  audit_msg_type_to_name(l->head->type),
523  aulookup_uid(l->s.loginuid, name, sizeof(name)),
524  aulookup_success(l->s.success), l->e.serial);
525  break;
526  case RPT_AUTH:
527  // who, addr, terminal, exe, success, event
528  // Special note...uid is used here because that is
529  // the way that the message works. This is because
530  // on failed logins, loginuid is not set.
531  safe_print_string(l->s.acct ? l->s.acct :
532  aulookup_uid(l->s.uid, name, sizeof(name)), 0);
533  printf(" %s %s %s %s %lu\n",
534  l->s.hostname, l->s.terminal,
535  l->s.exe, aulookup_success(l->s.success),
536  l->e.serial);
537  break;
538  case RPT_LOGIN:
539  // who, addr, terminal, exe, success, event
540  // Special note...uid is used here because that is
541  // the way that the message works. This is because
542  // on failed logins, loginuid is not set.
543  safe_print_string(((l->s.success == S_FAILED) &&
544  l->s.acct) ? l->s.acct :
545  aulookup_uid(l->s.uid, name, sizeof(name)), 0);
546  printf(" %s %s %s %s %lu\n",
547  l->s.hostname, l->s.terminal,
548  l->s.exe, aulookup_success(l->s.success),
549  l->e.serial);
550  break;
551  case RPT_ACCT_MOD:
552  // who, addr, terminal, exe, success, event
553  safe_print_string(
554  aulookup_uid(l->s.loginuid, name,
555  sizeof(name)), 0);
556  printf(" %s %s %s %s %s %lu\n",
557  l->s.hostname ? l->s.hostname : "?",
558  l->s.terminal ? l->s.terminal : "?",
559  l->s.exe ? l->s.exe : "?",
560  l->s.acct ? l->s.acct : "?",
561  aulookup_success(l->s.success),
562  l->e.serial);
563  break;
564  case RPT_EVENT: // report_detail == D_DETAILED
565  // event, type, who, success
566  printf("%lu %s ",
567  l->e.serial,
568  audit_msg_type_to_name(l->head->type));
569  safe_print_string(aulookup_uid(l->s.loginuid, name,
570  sizeof(name)), 0);
