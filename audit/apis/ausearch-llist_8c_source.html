128 int list_find_item(llist *l, unsigned int i)
129 {
130  register lnode* window;
131 
132  if (l->cur && (l->cur->item <= i))
133  window = l->cur; /* Try to use where we are */
134  else
135  window = l->head; /* Can't, start over */
136 
137  while (window) {
138  if (window->item == i) {
139  l->cur = window;
140  return 1;
141  } else
142  window = window->next;
143  }
144  return 0;
145 }
