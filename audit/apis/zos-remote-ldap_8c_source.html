92 int submit_request_s(ZOS_REMOTE *zos_remote, BerElement *ber)
93 {
94  int rc, retry = 1; /* retry once and give up */
95  struct berval bv;
96 
97  rc = ber_flatten2(ber, &bv, 0); /* 0 = Use ber's buffer */
98  if (rc == -1) {
99  log_err("Error flattening BER element");
100  return ICTX_E_ABORT;
101  }
102 
103 retry:
104  rc = submit_xop_s(zos_remote, &bv);
105  switch (rc) {
106  case ICTX_SUCCESS:
107  break;
108  case ICTX_E_TRYAGAIN:
109  /*
110  * Usually means that the server connection timed-out
111  * So we flush the LDAP connection by unsetting the
112  * 'connected' flag and trying again.
113  */
114  if (retry > 0) {
115  log_debug("Connection seems down - retrying");
116  retry--;
117  _zos_remote_destroy(zos_remote);
118  rc = _zos_remote_init(zos_remote);
119  if (rc != ICTX_SUCCESS)
120  log_err("Error - failed to re-initialize LDAP session");
121  else
122  goto retry; /* go to submit_xop_s once more */
123  }
124  log_err("Can't establish connection");
125  break;
126  case ICTX_E_ABORT:
127  break;
128  default:
129  log_err("Event resulted failure, code: 0x%x", rc);
130  }
131 
132  return rc;
133 }

176 static int _zos_remote_init(ZOS_REMOTE *zos_remote)
177 {
178  int version, rc;
179  char *uri = NULL;
180 
181 #ifdef LDAP_DEPRECATED
182 
183  log_debug("Initializing z/OS Remote-services LDAP connection at ldap://%s:%d",
184  zos_remote->server, zos_remote->port);
185  zos_remote->ld = ldap_init(zos_remote->server
186  zos_remote->port ? zos_remote->port : LDAP_PORT);
187  if (zos_remote->ld == NULL) {
188  log_err("Error initializing LDAP session: %s",
189  strerror(errno));
190  rc = ICTX_E_FATAL;
191  goto end;
192  }
193 #else
194  /* build ldap URI */
195  if (zos_remote->port == 0 || zos_remote->port == LDAP_PORT)
196  rc = asprintf(&uri, "ldap://%s", zos_remote->server);
197  else
198  rc = asprintf(&uri, "ldap://%s:%d", zos_remote->server,
199  zos_remote->port);
200 
201  if (rc == -1) {
202  log_err("Out of memory building LDAP server URI");
203  rc = ICTX_E_FATAL;
204  uri = NULL;
205  goto end;
206  }
207 
208  log_debug("Initializing z/OS Remote-services LDAP connection at %s", uri);
209  /* Get a handle to an LDAP connection */
210  rc = ldap_initialize(&zos_remote->ld, uri);
211  if (rc != LDAP_SUCCESS) {
212  log_err("Error initializing LDAP session: %s",
213  ldap_err2string(rc));
214  rc = ICTX_E_FATAL;
215  goto free_uri;
216  }
217 #endif
218 
219  /*
220  * Ensure the LDAP protocol version supported by the client
221  * to 3. (Extended operations are part of version 3).
222  */
223  rc = ldap_get_option(zos_remote->ld, LDAP_OPT_PROTOCOL_VERSION,
224  &version);
225  if (rc != LDAP_OPT_SUCCESS) {
226  log_err("Error getting LDAP session options");
227  rc = ICTX_E_FATAL;
228  goto unbind;
229  }
230 
231  if (version < LDAP_VERSION3) {
232  log_debug("Setting LDAP session version to %d",
233  LDAP_VERSION3);
234  version = LDAP_VERSION3;
235  rc = ldap_set_option(zos_remote->ld, LDAP_OPT_PROTOCOL_VERSION,
236  &version);
237  if (rc != LDAP_OPT_SUCCESS) {
238  log_err("Error setting LDAP session version");
239  rc = ICTX_E_FATAL;
240  goto unbind;
241  }
242  }
243 
244  goto free_uri;
245 
246 unbind:
247  ldap_unbind_ext_s(zos_remote->ld, NULL, NULL);
248  zos_remote->ld = NULL;
249 
250 free_uri:

252 
253 end:

322 /*
323  * Sync-submit extended operation given in *bv
324  * return ICTX_SUCCESS if submission (and response)
325  * succeeded.
326  * Log errors using log_err() functions
327  */
328 int submit_xop_s(ZOS_REMOTE *zos_remote, struct berval *bv)
329 {
330  LDAPMessage *result;
331  audit_response_t response;
332  int rc, errcode, msgId;
333  unsigned int i;
334  char *errmsg, *oid;
335  struct berval *bv_response;
336  struct timeval t;
337 
338  if (zos_remote->connected == 0) {
339  rc = zos_remote_connect(zos_remote);
340  if (rc != ICTX_SUCCESS)
341  return rc;
342  }
343 
344  /* call LDAP - won't block */
345  rc = ldap_extended_operation(zos_remote->ld, ICTX_OIDAUDITREQUEST,
346  bv, NULL, NULL, &msgId);
347  if (rc == LDAP_SERVER_DOWN) {
348  zos_remote->connected = 0;
349  return ICTX_E_TRYAGAIN;
350  } else if (rc != LDAP_SUCCESS) {
351  log_err("LDAP extended operation submission failure: %s",
352  ldap_err2string(rc));
353  return ICTX_E_ABORT;
354  } else {
355  log_debug("Sent LDAP extended operation request, msgId=0x%x",
356  msgId);
357  }
358 
359  /* call blocking ldap_result with specified timeout */
360  t.tv_sec = zos_remote->timeout;
361  t.tv_usec = 0;
362  rc = ldap_result(zos_remote->ld, msgId, 1, &t, &result);
363 
364  if (rc == -1) {
365  /* error in ldap operation */
366  ldap_get_option(zos_remote->ld, LDAP_OPT_ERROR_NUMBER, &errcode);
367  switch (errcode) {
368  case LDAP_SERVER_DOWN:
369  /* Connection may have timed out, let's retry */
370  zos_remote->connected = 0;
371  rc = ICTX_E_TRYAGAIN;
372  break;
373  default:
374  log_err("ldap_result unexpected failure: %s (0x%x)",
375  ldap_err2string(rc), rc);
376  rc = ICTX_E_ABORT;
377  }
378  goto end;
379  } else if (rc == 0) {
380  /* timeout reached */
381  log_warn("LDAP extended operation timed out");
382  rc = ICTX_E_ABORT;
383  goto end;
384  } else if (rc != LDAP_RES_EXTENDED) {
385  /* not an extended operation response! */
386  log_err("LDAP extended operation resulted in unexpected answer: 0x%x", rc);
387  rc = ICTX_E_ABORT;
388  goto free_result;
389  }
390 
391  log_debug("Got LDAP Extended result");
392  /*
393  * we have an extended operation result
394  * first parse_result will check for errcode, later
395  * parse_extended_result will give us the oid and the BER value
396  */
397  rc = ldap_parse_result(zos_remote->ld, result, &errcode, NULL,
398  &errmsg, NULL, NULL, 0);
399  if (rc != LDAP_SUCCESS) {
400  log_err("LDAP parse result internal failure (code 0x%x)",
401  rc);
402  rc = ICTX_E_ABORT;
403  goto free_result;
404  }
405 
406  if (errcode != LDAP_SUCCESS) {
407  log_err("LDAP extended operation failed: %s", errmsg);
408  rc = ICTX_E_ABORT;
409  goto free_errmsg;
410  }
411 
412  rc = ldap_parse_extended_result(zos_remote->ld, result, &oid,
413  &bv_response, 0);
414  if (rc != LDAP_SUCCESS) {
415  log_err("Failed to parse ldap extended result (code 0x%x)",
416  rc);
417  rc = ICTX_E_ABORT;
418  goto free_errmsg;
419  }
420 
421  if (oid && strcmp(oid, ICTX_OIDAUDITRESPONSE) != 0) {
422  /* oid == null shouldn't be a problem to log_err */
423  log_err("LDAP extended operation returned an invalid oid: %s", oid);
424  rc = ICTX_E_ABORT;
425  goto free_bv;
426  }
427 
428  rc = decode_response(&response, bv_response);
429  if (rc != ICTX_SUCCESS) {
430  log_err("Error decoding extended operation response");
431  goto free_bv;
432  }
433 
434  if (response.respMajor == ZOS_REMOTE_MAJOR_SUCCESS) {
435  /* submission was successful, no further processing needed */
436  log_debug("Successfully submited Remote audit Request");
437  rc = ICTX_SUCCESS;
438  goto free_response;
439  } else if (response.respMajor == ZOS_REMOTE_MAJOR_EMPTY) {
440  /* something is going on. Set error and stop processing */
441  log_warn("Warning - LDAP extended operation returned empty result");
442  rc = ICTX_E_ABORT;
443  goto free_response;
444  } else if (response.respMajor == ZOS_REMOTE_MAJOR_WARNINGMODE ||
445  response.respMajor == ZOS_REMOTE_MAJOR_NOTREQ)
446  rc = ICTX_SUCCESS; /* don't fail, but continue processing */
447  else
448  rc = ICTX_E_ABORT; /* set return code and continue processing */
449 
450  /* If it's not success nor empty, let's check for errors in the response */
451  for (i = 0; i < response.numItems; i++) {
452  switch ((response.itemList[i])->majorCode) {
453  /* 0 <= Major Code <= 14 */
454  case ZOS_REMOTE_MAJOR_SUCCESS:
455  break;
456  case ZOS_REMOTE_MAJOR_WARNINGMODE:
457  case ZOS_REMOTE_MAJOR_NOTREQ:
458  log_debug("Warning - LDAP extended operation returned '%s' for item %d",
459  zos_remote_err2string((response.itemList[i])->majorCode),
460  (response.itemList[i])->itemTag);
461  log_debug("SAF code: 0x%x, RACF code: 0x%x, RACF reason: 0x%x",
462  (response.itemList[i])->minorCode1,
463  (response.itemList[i])->minorCode2,
464  (response.itemList[i])->minorCode3);
465  break;
466  case ZOS_REMOTE_MAJOR_UNDETERMINED:
467  case ZOS_REMOTE_MAJOR_UNAUTHORIZED:
468  case ZOS_REMOTE_MAJOR_RACROUTE:
469  log_err("Error - LDAP extended operation returned '%s' for item %d",
470  zos_remote_err2string((response.itemList[i])->majorCode),
471  (response.itemList[i])->itemTag);
472  log_err("SAF code: 0x%x, RACF code: 0x%x, RACF reason: 0x%x",
473  (response.itemList[i])->minorCode1,
474  (response.itemList[i])->minorCode2,
475  (response.itemList[i])->minorCode3);
476  break;
477  /* 16 <= Major Code <= 20 */
478  case ZOS_REMOTE_MAJOR_VAL_ERR:
479  case ZOS_REMOTE_MAJOR_ENC_ERR:
480  log_err("Error - LDAP extended operation returned '%s' for item %d",
481  zos_remote_err2string((response.itemList[i])->majorCode),
482  (response.itemList[i])->itemTag);
483  log_err("Item field: %d, reson %d",
484  (response.itemList[i])->
485  minorCode1,
486  (response.itemList[i])->minorCode2);
487  break;
488  /* 24 <= Major code <= 100 */
489  case ZOS_REMOTE_MAJOR_UNSUF_AUTH:
490  case ZOS_REMOTE_MAJOR_EMPTY:
491  case ZOS_REMOTE_MAJOR_INVALID_VER:
492  case ZOS_REMOTE_MAJOR_INTERNAL_ERR:
493  log_err("Error - LDAP extended operation returned '%s' for item %d",
494  zos_remote_err2string((response.itemList[i])->majorCode),
495  (response.itemList[i])->itemTag);
496  break;
497  default:
498  log_err("Error - LDAP extended operation returned an unknown Major code for item %d",
499  (response.itemList[i])->majorCode);
500  }
501  }
502 
503 free_response:
504  for (; response.numItems > 0; response.numItems--)
505  free(response.itemList[response.numItems - 1]);
506  free(response.itemList);
507 
508 free_bv:
509  if (bv_response)
510  ber_bvfree(bv_response);
511  if (oid)
512  ldap_memfree(oid);
513 
514 free_errmsg:
515  ldap_memfree(errmsg);
516 
517 free_result:
518  ldap_msgfree(result);
519 
520 end:
521  return rc;
522 }

524 static int decode_response(audit_response_t * r, struct berval *bv)
525 {
526  BerElement *ber;
527  ber_len_t len;
528  int rc;
529 
530  if (!bv) {
531  log_err("LDAP extended operation returned NULL message");
532  return ICTX_E_ABORT;
533  } else if ((ber = ber_init(bv)) == NULL) {
534  log_err("Error initializing BER response data");
535  return ICTX_E_ABORT;
536  }
537 
538  log_debug("---Got an encoded request response:");
539  debug_bv(bv);
540 
541  r->respVersion = 0;
542  r->respMajor = 0;
543  r->numItems = 0;
544  r->itemList = NULL;
545 
546  rc = ber_scanf(ber, "{ii", &r->respVersion, &r->respMajor);
547  if (r->respVersion != ICTX_REQUESTVER) {
548  log_err("Invalid version returned by z/OS Remote-services server");
549  log_err("Should be %d, got %d", ICTX_REQUESTVER,
550  r->respVersion);
551  rc = ICTX_E_ABORT;
552  goto free_ber;
553  }
554 
555  if (r->respMajor == ZOS_REMOTE_MAJOR_SUCCESS ||
556  r->respMajor == ZOS_REMOTE_MAJOR_EMPTY) {
557  rc = ICTX_SUCCESS;
558  /* No further processing required */
559  goto free_ber;
560  }
561 
562  /* Inspect ber response otherwise */
563  while (ber_peek_tag(ber, &len) == LBER_SEQUENCE) {
564  r->numItems++;
565  r->itemList = (audit_resp_item_t **) realloc(r->itemList,
566  r->numItems *
567  sizeof
568  (audit_resp_item_t
569  *));
570  if (errno == ENOMEM) {
571  if (r->itemList)
572  free(r->itemList);
573  rc = ICTX_E_FATAL;
574  goto free_ber;
575  }
576 
577  audit_resp_item_t *item = (audit_resp_item_t *)
578  malloc(sizeof(audit_resp_item_t));
579 
580  if (!item) {
581  rc = ICTX_E_FATAL;
582  goto free_ber;
583  }
584 
585  rc |= ber_scanf(ber, "{{iiiiii}}",
586  &item->version,
587  &item->itemTag,
588  &item->majorCode,
589  &item->minorCode1, &item->minorCode2,
590  &item->minorCode3);
591  r->itemList[r->numItems - 1] = item;
592  }
593  rc |= ber_scanf(ber, "}");
594 
595  if (rc == -1) {
596  for (; r->numItems > 0; r->numItems--)
597  free(r->itemList[r->numItems - 1]);
598  free(r->itemList);
599  rc = ICTX_E_ABORT;
600  }
601  else
602  rc = ICTX_SUCCESS;
603 
604 free_ber:
605  ber_free(ber, 1);
606 
607  return rc;
608 }
