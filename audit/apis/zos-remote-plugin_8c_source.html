95 /*
96  * The submission thread
97  * It's job is to dequeue the events from the queue
98  * and sync submit them to ITDS
99  */
100 static void *submission_thread_main(void *arg)
101 {
102  int rc;
103 
104  UNUSED(arg);
105  log_debug("Starting event submission thread");
106 
107  rc = zos_remote_init(&zos_remote_inst, conf.server,
108  conf.port, conf.user,
109  conf.password,
110  conf.timeout);
111 
112  if (rc != ICTX_SUCCESS) {
113  log_err("Error - Failed to initialize session to z/OS ITDS Server");
114  stop = 1;
115  return 0;
116  }
117 
118  while (stop == 0) {
119  /* block until we have an event */
120  BerElement *ber = dequeue();
121 
122  if (ber == NULL) {
123  if (hup) {
124  break;
125  }
126  continue;
127  }
128  debug_ber(ber);
129  rc = submit_request_s(&zos_remote_inst, ber);
130  if (rc == ICTX_E_FATAL) {
131  log_err("Error - Fatal error in event submission. Aborting");
132  stop = 1;
133  } else if (rc != ICTX_SUCCESS) {
134  log_warn("Warning - Event submission failure - event dropped");
135  }
136  else {
137  log_debug("Event submission success");
138  }
139  ber_free(ber, 1); /* also free BER buffer */
140  }
141  log_debug("Stopping event submission thread");
142  zos_remote_destroy(&zos_remote_inst);
143 
144  return 0;
145 }

149  * auparse library callback that's called when an event is ready
150  */
151 void
152 push_event(auparse_state_t * au, auparse_cb_event_t cb_event_type,
153  void *user_data)
154 {
155  int rc;
156  BerElement *ber;
157  int qualifier;
158  char timestamp[26];
159  char linkValue[ZOS_REMOTE_LINK_VALUE_SIZE];
160  char logString[ZOS_REMOTE_LOGSTRING_SIZE];
161  unsigned long linkValue_tmp;
162 
163  UNUSED(user_data);
164  if (cb_event_type != AUPARSE_CB_EVENT_READY)
165  return;
166 
167  const au_event_t *e = auparse_get_timestamp(au);
168  if (e == NULL)
169  return;
170  /*
171  * we have an event. Each record will result in a different 'Item'
172  * (refer ASN.1 definition in zos-remote-ldap.h)
173  */
174 
175  /*
176  * Create a new BER element to encode the request
177  */
178  ber = ber_alloc_t(LBER_USE_DER);
179  if (ber == NULL) {
180  log_err("Error allocating memory for BER element");
181  goto fatal;
182  }
183 
184  /*
185  * Collect some information to fill in every item
186  */
187  const char *node = auparse_get_node(au);
188  const char *orig_type = auparse_find_field(au, "type");
189  /* roll back event to get 'success' */
190  auparse_first_record(au);
191  const char *success = auparse_find_field(au, "success");
192  /* roll back event to get 'res' */
193  auparse_first_record(au);
194  const char *res = auparse_find_field(au, "res");
195 
196  /* check if this event is a success or failure one */
197  if (success) {
198  if (strncmp(success, "0", 1) == 0 ||
199  strncmp(success, "no", 2) == 0)
200  qualifier = ZOS_REMOTE_QUALIF_FAIL;
201  else
202  qualifier = ZOS_REMOTE_QUALIF_SUCCESS;
203  } else if (res) {
204  if (strncmp(res, "0", 1) == 0
205  || strncmp(res, "failed", 6) == 0)
206  qualifier = ZOS_REMOTE_QUALIF_FAIL;
207  else
208  qualifier = ZOS_REMOTE_QUALIF_SUCCESS;
209  } else
210  qualifier = ZOS_REMOTE_QUALIF_INFO;
211 
212  /* get timestamp text */
213  ctime_r(&e->sec, timestamp);
214  timestamp[24] = '\0'; /* strip \n' */
215 
216  /* prepare linkValue which will be used for every item */
217  linkValue_tmp = htonl(e->serial); /* padronize to use network
218  * byte order
219  */
220  memset(&linkValue, 0, ZOS_REMOTE_LINK_VALUE_SIZE);
221  memcpy(&linkValue, &linkValue_tmp, sizeof(unsigned long));
222 
223  /*
224  * Prepare the logString with some meaningful text
225  * We assume the first record type found is the
226  * 'originating' audit record
227  */
228  sprintf(logString, "Linux (%s): type: %s", node, orig_type);
229  free((void *)node);
230 
231  /*
232  * Start writing to BER element.
233  * There's only one field (version) out of the item sequence.
234  * Also open item sequence
235  */
236  rc = ber_printf(ber, "{i{", ICTX_REQUESTVER);
237  if (rc < 0)
238  goto skip_event;
239 
240  /*
241  * Roll back to first record and iterate through all records
242  */
243  auparse_first_record(au);
244  do {
245  const char *type = auparse_find_field(au, "type");
246  if (type == NULL)
247  goto skip_event;
248 
249  log_debug("got record: %s", auparse_get_record_text(au));
250 
251  /*
252  * First field is item Version, same as global version
253  */
254  rc = ber_printf(ber, "{i", ICTX_REQUESTVER);
255 
256  /*
257  * Second field is the itemTag
258  * use our internal event counter, increasing it
259  */
260  rc |= ber_printf(ber, "i", conf.counter++);
261 
262  /*
263  * Third field is the linkValue
264  * using ber_put_ostring since it is not null-terminated
265  */
266  rc |= ber_put_ostring(ber, linkValue,
267  ZOS_REMOTE_LINK_VALUE_SIZE,
268  LBER_OCTETSTRING);
269  /*
270  * Fourth field is the violation
271  * Don't have anything better yet to put here
272  */
273  rc |= ber_printf(ber, "b", 0);
274 
275  /*
276  * Fifth field is the event.
277  * FIXME: this might be the place to switch on the
278  * audit record type and map to a more meaningful
279  * SMF type 83, subtype 4 event here
280  */
281  rc |= ber_printf(ber, "i", ZOS_REMOTE_EVENT_AUTHORIZATION);
282 
283  /*
284  * Sixth field is the qualifier. We map 'success' or
285  * 'res' to this field
286  */
287  rc |= ber_printf(ber, "i", qualifier);
288 
289  /*
290  * Seventh field is the Class
291  * always use '@LINUX' for this version
292  * max size ZOS_REMOTE_CLASS_SIZE
293  */
294  rc |= ber_printf(ber, "t", ASN1_IA5STRING_TAG);
295  rc |= ber_printf(ber, "s", "@LINUX");
296 
297  /*
298  * Eighth field is the resource
299  * use the record type (name) as the resource
300  * max size ZOS_REMOTE_RESOURCE_SIZE
301  */
302  rc |= ber_printf(ber, "t", ASN1_IA5STRING_TAG);
303  rc |= ber_printf(ber, "s", type);
304 
305  /*
306  * Nineth field is the LogString
307  * we try to put something meaningful here
308  * we also start the relocations sequence
309  */
310  rc |= ber_printf(ber, "t", ASN1_IA5STRING_TAG);
311  rc |= ber_printf(ber, "s{", logString);
312 
313  /*
314  * Now we start adding the relocations.
315  * Let's add the timestamp as the first one
316  * so it's out of the field loop
317  */
318  rc |= ber_printf(ber, "{i", ZOS_REMOTE_RELOC_TIMESTAMP);
319  rc |= ber_printf(ber, "t", ASN1_IA5STRING_TAG);
320  rc |= ber_printf(ber, "s}", timestamp);
321 
322  /*
323  * Check that encoding is going OK until now
324  */
325  if (rc < 0)
326  goto skip_event;
327 
328  /*
329  * Now go to first field,
330  * and iterate through all fields
331  */
332  auparse_first_field(au);
333  do {
334  /*
335  * we set a maximum of 1024 chars for
336  * relocation data (field=value pairs)
337  * Hopefuly this wont overflow too often
338  */
339  char data[1024];
340  const char *name = auparse_get_field_name(au);
341  const char *value = auparse_interpret_field(au);
342  if (name == NULL || value == NULL)
343  goto skip_event;
344 
345  /*
346  * First reloc field is the Relocation type
347  * We use 'OTHER' here since we don't have
348  * anything better
349  */
350  rc |= ber_printf(ber, "{i", ZOS_REMOTE_RELOC_OTHER);
351 
352  /*
353  * Second field is the relocation data
354  * We use a 'name=value' pair here
355  * Use up to 1023 chars (one char left for '\0')
356  */
357  snprintf(data, 1023, "%s=%s", name, value);
358  rc |= ber_printf(ber, "t", ASN1_IA5STRING_TAG);
359  rc |= ber_printf(ber, "s}", data);
360 
361  /*
362  * Check encoding status
363  */
364  if (rc < 0)
365  goto skip_event;
366  } while (auparse_next_field(au) > 0);
367 
368  /*
369  * After adding all relocations we are done with
370  * this item - finalize relocs and item
371  */
372  rc |= ber_printf(ber, "}}");
373 
374  /*
375  * Check if we are doing well with encoding
376  */
377  if (rc < 0)
378  goto skip_event;
379 
380  } while (auparse_next_record(au) > 0);
381 
382  /*
383  * We have all items in - finalize item sequence & request
384  */
385  rc |= ber_printf(ber, "}}");
386 
387  /*
388  * Check if everything went alright with encoding
389  */
390  if (rc < 0)
391  goto skip_event;
392 
393  /*
394  * finally, enqueue request and let the other
395  * thread process it
396  */
397  log_debug("Encoding done, enqueuing event");
398  enqueue(ber);
399 
400  return;
401 
402 skip_event:
403  log_warn("Warning - error encoding request, skipping event");
404  ber_free(ber, 1); /* free it since we're not enqueuing it */
405  return;
406 
407 fatal:
408  log_err("Error - Fatal error while encoding request. Aborting");
409  stop = 1;
410 }

412 int main(int argc, char *argv[])
413 {
414  int rc;
415  const char *cpath;
416  char buf[1024];
417  struct sigaction sa;
418  sigset_t ss;
419  auparse_state_t *au;
420  ssize_t len;
421 
422  mypid = getpid();
423 
424  log_info("starting with pid=%d", mypid);
425 
426  /*
427  * install signal handlers
428  */
429  sa.sa_flags = 0;
430  sigemptyset(&sa.sa_mask);
431  sa.sa_handler = term_handler;
432  sigaction(SIGTERM, &sa, NULL);
433  sa.sa_handler = hup_handler;
434  sigaction(SIGHUP, &sa, NULL);
435  sa.sa_handler = alarm_handler;
436  sigaction(SIGALRM, &sa, NULL);
437 
438  /*
439  * the main program accepts a single (optional) argument:
440  * it's configuration file (this is NOT the plugin configuration
441  * usually located at /etc/audisp/plugin.d)
442  * We use the default (def_config_file) if no arguments are given
443  */
444  if (argc == 1) {
445  cpath = def_config_file;
446  log_warn("No configuration file specified - using default (%s)", cpath);
447  } else if (argc == 2) {
448  cpath = argv[1];
449  log_info("Using configuration file: %s", cpath);
450  } else {
451  log_err("Error - invalid number of parameters passed. Aborting");
452  return 1;
453  }
454 
455  /* initialize record counter */
456  conf.counter = 1;
457 
458  /* initialize configuration with default values */
459  plugin_clear_config(&conf);
460 
461  /* initialize the submission queue */
462  if (init_queue(conf.q_depth) != 0) {
463  log_err("Error - Can't initialize event queue. Aborting");
464  return -1;
465  }
466 
467 #ifdef HAVE_LIBCAP_NG
468  // Drop all capabilities
469  capng_clear(CAPNG_SELECT_BOTH);
470  capng_apply(CAPNG_SELECT_BOTH);
471 #endif
472 
473  /* set stdin to O_NONBLOCK */
474  if (fcntl(0, F_SETFL, O_NONBLOCK) == -1) {
475  log_err("Error - Can't set input to Non-blocking mode: %s. Aborting",
476  strerror(errno));
477  return -1;
478  }
479 
480  do {
481 
482  hup = 0; /* don't flush unless hup == 1 */
483 
484  /*
485  * initialization is done in 4 steps:
486  */
487 
488  /*
489  * load configuration and
490  * increase queue depth if needed
491  */
492  rc = plugin_load_config(&conf, cpath);
493  if (rc != 0) {
494  log_err("Error - Can't load configuration. Aborting");
495  return -1;
496  }
497  increase_queue_depth(conf.q_depth); /* 1 */
498 
499  /* initialize auparse */
500  au = auparse_init(AUSOURCE_FEED, 0); /* 2 */
501  if (au == NULL) {
502  log_err("Error - exiting due to auparse init errors");
503  return -1;
504  }
505 
506  /*
507  * Block signals for everyone,
508  * Initialize submission thread, and
509  * Unblock signals for this thread
510  */
511  sigfillset(&ss);
512  pthread_sigmask(SIG_BLOCK, &ss, NULL);
513  pthread_create(&submission_thread, NULL,
514  submission_thread_main, NULL);
515  pthread_sigmask(SIG_UNBLOCK, &ss, NULL); /* 3 */
516 
517  /* add our event consumer callback */
518  auparse_add_callback(au, push_event, NULL, NULL); /* 4 */
519 
520  /* main loop */
521  while (hup == 0 && stop == 0) {
522  fd_set rfds;
523  struct timeval tv;
524 
525  FD_ZERO(&rfds);
526  FD_SET(0, &rfds);
527  tv.tv_sec = 5;
528  tv.tv_usec = 0;
529  rc = select(1, &rfds, NULL, NULL, &tv);
530  if (rc == -1) {
531  if (errno == EINTR) {
532  log_debug("Select call interrupted");
533  continue;
534  }
535  else {
536  log_err("Error - Fatal error while monitoring input: %s. Aborting",
537  strerror(errno));
538  stop = 1;
539  }
540  }
541  else if (rc) {
542  len = read(0, buf, 1024);
543  if (len > 0)
544  /* let our callback know of the new data */
545  auparse_feed(au, buf, len);
546  else if (len == 0) {
547  log_debug("End of input - Exiting");
548  stop = 1;
549  }
550  else {
551  /* ignore interrupted call or empty pipe */
552  if (errno != EINTR && errno != EAGAIN) {
553  log_err("Error - Fatal error while reading input: %s. Aborting",
554  strerror(errno));
555  stop = 1;
556  }
557  else {
558  log_debug("Ignoring read interruption: %s",
559  strerror(errno));
560  }
561  }
562  }
563  }
564  /* flush everything, in order */
565  auparse_flush_feed(au); /* 4 */
566  alarm(10); /* 10 seconds to clear the queue */
567  pthread_join(submission_thread, NULL); /* 3 */
568  alarm(0); /* cancel any pending alarm */
569  auparse_destroy(au); /* 2 */
570  plugin_free_config(&conf); /* 1 */
571  }
572  while (hup && stop == 0);
573 
574  /* destroy queue before leaving */
575  destroy_queue();
576 
577  log_info("Exiting");
578 
579  return 0;
580 }
