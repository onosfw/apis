227 int audit_name_to_msg_type(const char *msg_type)
228 {
229  int rc;
230 
231  if (msg_type_s2i(msg_type, &rc) != 0)
232  return rc;
233 
234  /* Take a stab at converting */
235  if (strncmp(msg_type, "UNKNOWN[", 8) == 0) {
236  int len;
237  char buf[8];
238  const char *end = strchr(msg_type + 8, ']');
239  if (end == NULL)
240  return -1;
241 
242  len = end - (msg_type + 8);
243  if (len > 7)
244  len = 7;
245  memset(buf, 0, sizeof(buf));
246  strncpy(buf, msg_type + 8, len);
247  errno = 0;
248  return strtol(buf, NULL, 10);
249  } else if (isdigit(*msg_type)) {
250  errno = 0;
251  return strtol(msg_type, NULL, 10);
252  }
253 
254  return -1;
255 }
