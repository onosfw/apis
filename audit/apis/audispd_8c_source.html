164 static int start_one_plugin(lnode *conf)
165 {
166  if (conf->p->restart_cnt > daemon_config.max_restarts)
167  return 1;
168 
169  if (conf->p->type == S_BUILTIN)
170  start_builtin(conf->p);

172  if (safe_exec(conf->p)) {
173  syslog(LOG_ERR,

209 static int reconfigure(void)
210 {
211  int rc;
212  daemon_conf_t tdc;
213  conf_llist tmp_plugin;
214  lnode *tpconf;
215 
216  /* Read new daemon config */
217  rc = load_config(&tdc, config_file);
218  if (rc == 0) {
219  if (tdc.q_depth > daemon_config.q_depth) {
220  increase_queue_depth(tdc.q_depth);
221  daemon_config.q_depth = tdc.q_depth;
222  }
223  daemon_config.overflow_action = tdc.overflow_action;
224  reset_suspended();
225  /* We just fill these in because they are used by this
226  * same thread when we return
227  */
228  daemon_config.node_name_format = tdc.node_name_format;
229  free((char *)daemon_config.name);
230  daemon_config.name = tdc.name;
231  }
232 
233  /* The idea for handling SIGHUP to children goes like this:
234  * 1) load the current config in temp list
235  * 2) mark all in real list unchecked
236  * 3) for each one in tmp list, scan old list
237  * 4) if new, start it, append to list, mark done
238  * 5) else check if there was a change to active state
239  * 6) if so, copy config over and start
240  * 7) If no change, send sighup to non-builtins and mark done
241  * 8) Finally, scan real list for unchecked, terminate and deactivate
242  */
243  syslog(LOG_INFO, "Starting reconfigure");
244  load_plugin_conf(&tmp_plugin);
245  plist_mark_all_unchecked(&plugin_conf);
246 
247  plist_first(&tmp_plugin);
248  tpconf = plist_get_cur(&tmp_plugin);
249  while (tpconf && tpconf->p) {
250  lnode *opconf;
251 
252  opconf = plist_find_name(&plugin_conf, tpconf->p->name);
253  if (opconf == NULL) {
254  /* We have a new service */
255  if (tpconf->p->active == A_YES) {
256  tpconf->p->checked = 1;
257  plist_last(&plugin_conf);
258  plist_append(&plugin_conf, tpconf->p);
259  free(tpconf->p);
260  tpconf->p = NULL;
261  start_one_plugin(plist_get_cur(&plugin_conf));
262  }
263  } else {
264  if (opconf->p->active == tpconf->p->active) {
265  /* If active and no state change, sighup it */
266  if (opconf->p->type == S_ALWAYS &&
267  opconf->p->active == A_YES) {
268  if (opconf->p->inode==tpconf->p->inode)
269  kill(opconf->p->pid, SIGHUP);
270  else {
271  /* Binary changed, restart */
272  syslog(LOG_INFO,
273  "Restarting %s since binary changed",
274  opconf->p->path);
275  kill(opconf->p->pid, SIGTERM);
276  usleep(50000); // 50 msecs
277  close(opconf->p->plug_pipe[1]);
278  opconf->p->plug_pipe[1] = -1;
279  opconf->p->pid = 0;
280  start_one_plugin(opconf);
281  opconf->p->inode =
282  tpconf->p->inode;
283  }
284  }
285  opconf->p->checked = 1;
286  } else {
287  /* A change in state */
288  if (tpconf->p->active == A_YES) {
289  /* starting - copy config and exec */
290  free_pconfig(opconf->p);
291  free(opconf->p);
292  opconf->p = tpconf->p;
293  opconf->p->checked = 1;
294  start_one_plugin(opconf);
295  tpconf->p = NULL;
296  }
297  }
298  }
299 
300  tpconf = plist_next(&tmp_plugin);
301  }
302 
303  /* Now see what's left over */
304  while ( (tpconf = plist_find_unchecked(&plugin_conf)) ) {
305  /* Anything not checked is something removed from the config */
306  tpconf->p->active = A_NO;
307  syslog(LOG_INFO, "Terminating %s because its now inactive",
308  tpconf->p->path);
309  if (tpconf->p->type == S_ALWAYS) {
310  kill(tpconf->p->pid, SIGTERM);
311  close(tpconf->p->plug_pipe[1]);
312  } else
313  stop_builtin(tpconf->p);
314  tpconf->p->plug_pipe[1] = -1;
315  tpconf->p->pid = 0;
316  tpconf->p->checked = 1;
317  }
318 
319  /* Release memory from temp config */
320  plist_first(&tmp_plugin);
321  tpconf = plist_get_cur(&tmp_plugin);
322  while (tpconf) {
323  free_pconfig(tpconf->p);
324  tpconf = plist_next(&tmp_plugin);
325  }
326  plist_clear(&tmp_plugin);
327  return plist_count_active(&plugin_conf);
328 }

330 int main(int argc, char *argv[])
331 {
332  lnode *conf;
333  struct sigaction sa;
334  int i;
335 
336 #ifndef DEBUG
337  /* Make sure we are root */
338  if (getuid() != 0) {
339  fprintf(stderr, "You must be root to run this program.\n");
340  return 4;
341  }
342 #endif
343  set_aumessage_mode(MSG_SYSLOG, DBG_YES);
344 
345  /* Clear any procmask set by libev */
346  sigfillset (&sa.sa_mask);
347  sigprocmask (SIG_UNBLOCK, &sa.sa_mask, 0);
348 
349  /* Register sighandlers */
350  sa.sa_flags = 0;
351  sigemptyset(&sa.sa_mask);
352  /* Ignore all signals by default */
353  sa.sa_handler = SIG_IGN;
354  for (i=1; i<NSIG; i++)
355  sigaction(i, &sa, NULL);
356  /* Set handler for the ones we care about */
357  sa.sa_handler = term_handler;
358  sigaction(SIGTERM, &sa, NULL);
359  sa.sa_handler = hup_handler;
360  sigaction(SIGHUP, &sa, NULL);
361  sa.sa_handler = alarm_handler;
362  sigaction(SIGALRM, &sa, NULL);
363  sa.sa_handler = child_handler;
364  sigaction(SIGCHLD, &sa, NULL);
365 
366  /* move stdin to its own fd */
367  if (argc == 3 && strcmp(argv[1], "--input") == 0)
368  audit_fd = open(argv[2], O_RDONLY);
369  else
370  audit_fd = dup(0);
371  if (audit_fd < 0) {
372  syslog(LOG_ERR, "Failed setting up input, exiting");
373  return 1;
374  }
375 
376  /* Make all descriptors point to dev null */
377  i = open("/dev/null", O_RDWR);
378  if (i >= 0) {
379  if (dup2(0, i) < 0 || dup2(1, i) < 0 || dup2(2, i) < 0) {
380  syslog(LOG_ERR, "Failed duping /dev/null %s, exiting",
381  strerror(errno));
382  return 1;
383  }
384  close(i);
385  } else {
386  syslog(LOG_ERR, "Failed opening /dev/null %s, exiting",
387  strerror(errno));
388  return 1;
389  }
390  if (fcntl(audit_fd, F_SETFD, FD_CLOEXEC) < 0) {
391  syslog(LOG_ERR, "Failed protecting input %s, exiting",
392  strerror(errno));
393  return 1;
394  }
395 
396  /* init the daemon's config */
397  if (load_config(&daemon_config, config_file))
398  return 6;
399 
400  load_plugin_conf(&plugin_conf);
401 
402  /* if no plugins - exit */
403  if (plist_count(&plugin_conf) == 0) {
404  syslog(LOG_NOTICE, "No plugins found, exiting");
405  return 0;
406  }
407 
408  /* Plugins are started with the auditd priority */
409  i = start_plugins(&plugin_conf);
410 
411  /* Now boost priority to make sure we are getting time slices */
412  if (daemon_config.priority_boost != 0) {
413  int rc;
414 
415  errno = 0;
416  rc = nice((int)-daemon_config.priority_boost);
417  if (rc == -1 && errno) {
418  syslog(LOG_ERR, "Cannot change priority (%s)",
419  strerror(errno));
420  /* Stay alive as this is better than stopping */
421  }
422  }
423 
424  /* Let the queue initialize */
425  init_queue(daemon_config.q_depth);
426  syslog(LOG_INFO,
427  "audispd initialized with q_depth=%d and %d active plugins",
428  daemon_config.q_depth, i);
429 
430  /* Tell it to poll the audit fd */
431  if (add_event(audit_fd, process_inbound_event) < 0) {
432  syslog(LOG_ERR, "Cannot add event, exiting");
433  return 1;
434  }
435 
436  /* Create inbound thread */
437  pthread_create(&inbound_thread, NULL, inbound_thread_main, NULL);
438 
439  /* Start event loop */
440  while (event_loop()) {
441  hup = 0;
442  if (reconfigure() == 0) {
443  syslog(LOG_INFO,
444  "After reconfigure, there are no active plugins, exiting");
445  break;
446  }
447  }
448 
449  /* Tell plugins we are going down */
450  signal_plugins(SIGTERM);
451 
452  /* Cleanup builtin plugins */
453  destroy_af_unix();
454  destroy_syslog();
455 
456  /* Give it 5 seconds to clear the queue */
457  alarm(5);
458  pthread_join(inbound_thread, NULL);
459 
460  /* Release configs */
461  plist_first(&plugin_conf);
462  conf = plist_get_cur(&plugin_conf);
463  while (conf) {
464  free_pconfig(conf->p);
465  conf = plist_next(&plugin_conf);
466  }
467  plist_clear(&plugin_conf);
468 
469  /* Cleanup the queue */
470  destroy_queue();
471  free_config(&daemon_config);
472 
473  return 0;
474 }

476 static int safe_exec(plugin_conf_t *conf)
477 {
478  char *argv[MAX_PLUGIN_ARGS+2];
479  int pid, i;
480 
481  /* Set up IPC with child */
482  if (socketpair(AF_UNIX, SOCK_STREAM, 0, conf->plug_pipe) != 0)
483  return -1;
484 
485  pid = fork();
486  if (pid > 0) {
487  conf->pid = pid;
488  return 0; /* Parent...normal exit */
489  }
490  if (pid < 0) {
491  close(conf->plug_pipe[0]);
492  close(conf->plug_pipe[1]);
493  conf->pid = 0;
494  return -1; /* Failed to fork */
495  }
496 
497  /* Set up comm with child */
498  dup2(conf->plug_pipe[0], 0);
499  for (i=3; i<24; i++) /* Arbitrary number */
500  close(i);
501 
502  /* Child */
503  argv[0] = (char *)conf->path;
504  for (i=1; i<(MAX_PLUGIN_ARGS+1); i++)
505  argv[i] = conf->args[i];
506  argv[i] = NULL;
507  execve(conf->path, argv, NULL);
508  exit(1); /* Failed to exec */
509 }

548 /* Returns 0 on stop, and 1 on HUP */
549 static int event_loop(void)
550 {
551  char *name = NULL, tmp_name[255];
552 
553  /* Get the host name representation */
554  switch (daemon_config.node_name_format)
555  {
556  case N_NONE:
557  break;
558  case N_HOSTNAME:
559  if (gethostname(tmp_name, sizeof(tmp_name))) {
560  syslog(LOG_ERR, "Unable to get machine name");
561  name = strdup("?");
562  } else
563  name = strdup(tmp_name);
564  break;
565  case N_USER:
566  if (daemon_config.name)
567  name = strdup(daemon_config.name);
568  else {
569  syslog(LOG_ERR, "User defined name missing");
570  name = strdup("?");
571  }
572  break;
573  case N_FQD:
574  if (gethostname(tmp_name, sizeof(tmp_name))) {
575  syslog(LOG_ERR, "Unable to get machine name");
576  name = strdup("?");
577  } else {
578  int rc;
579  struct addrinfo *ai;
580  struct addrinfo hints;
581 
582  memset(&hints, 0, sizeof(hints));
583  hints.ai_flags = AI_ADDRCONFIG | AI_CANONNAME;
584  hints.ai_socktype = SOCK_STREAM;
585 
586  rc = getaddrinfo(tmp_name, NULL, &hints, &ai);
587  if (rc != 0) {
588  syslog(LOG_ERR,
589  "Cannot resolve hostname %s (%s)",
590  tmp_name, gai_strerror(rc));
591  name = strdup("?");
592  break;
593  }
594  name = strdup(ai->ai_canonname);
595  freeaddrinfo(ai);
596  }
597  break;
598  case N_NUMERIC:
599  if (gethostname(tmp_name, sizeof(tmp_name))) {
600  syslog(LOG_ERR, "Unable to get machine name");
601  name = strdup("?");
602  } else {
603  int rc;
604  struct addrinfo *ai;
605  struct addrinfo hints;
606 
607  memset(&hints, 0, sizeof(hints));
608  hints.ai_flags = AI_ADDRCONFIG | AI_PASSIVE;
609  hints.ai_socktype = SOCK_STREAM;
610 
611  rc = getaddrinfo(tmp_name, NULL, &hints, &ai);
612  if (rc != 0) {
613  syslog(LOG_ERR,
614  "Cannot resolve hostname %s (%s)",
615  tmp_name, gai_strerror(rc));
616  name = strdup("?");
617  break;
618  }
619  inet_ntop(ai->ai_family,
620  ai->ai_family == AF_INET ?
621  (void *) &((struct sockaddr_in *)ai->ai_addr)->sin_addr :
622  (void *) &((struct sockaddr_in6 *)ai->ai_addr)->sin6_addr,
623  tmp_name, INET6_ADDRSTRLEN);
624  freeaddrinfo(ai);
625  name = strdup(tmp_name);
626  }
627  break;
628  }
629 
630  /* Figure out the format for the af_unix socket */
631  while (stop == 0) {
632  event_t *e;
633  const char *type;
634  char *v, *ptr, unknown[32];
635  unsigned int len;
636  lnode *conf;
637 
638  /* This is where we block until we have an event */
639  e = dequeue();
640  if (e == NULL) {
641  if (hup) {
642  free(name);
643  return 1;
644  }
645  continue;
646  }
647 
648  /* Get the event formatted */
649  type = audit_msg_type_to_name(e->hdr.type);
650  if (type == NULL) {
651  snprintf(unknown, sizeof(unknown),
652  "UNKNOWN[%d]", e->hdr.type);
653  type = unknown;
654  }
655 
656  if (daemon_config.node_name_format != N_NONE) {
657  len = asprintf(&v, "node=%s type=%s msg=%.*s\n",
658  name, type, e->hdr.size, e->data);
659  } else
660  len = asprintf(&v, "type=%s msg=%.*s\n",
661  type, e->hdr.size, e->data);
662  if (len <= 0) {
663  v = NULL;
664  free(e); /* Either corrupted event or no memory */
665  continue;
666  }
667 
668  /* Strip newlines from event record */
669  ptr = v;
670  while ((ptr = strchr(ptr, 0x0A)) != NULL) {
671  if (ptr != &v[len-1])
672  *ptr = ' ';
673  else
674  break; /* Done - exit loop */
675  }
676 
677  /* Distribute event to the plugins */
678  plist_first(&plugin_conf);
679  conf = plist_get_cur(&plugin_conf);
680  do {
681  if (conf == NULL || conf->p == NULL)
682  continue;
683  if (conf->p->active == A_NO || stop)
684  continue;
685 
686  /* Now send the event to the right child */
687  if (conf->p->type == S_SYSLOG)
688  send_syslog(v);
689  else if (conf->p->type == S_AF_UNIX) {
690  if (conf->p->format == F_STRING)
691  send_af_unix_string(v, len);
692  else
693  send_af_unix_binary(e);
694  } else if (conf->p->type == S_ALWAYS && !stop) {
695  int rc;
696  rc = write_to_plugin(e, v, len, conf);
697  if (rc < 0 && errno == EPIPE) {
698  /* Child disappeared ? */
699  syslog(LOG_ERR,
700  "plugin %s terminated unexpectedly",
701  conf->p->path);
702  conf->p->pid = 0;
703  conf->p->restart_cnt++;
704  if (conf->p->restart_cnt >
705  daemon_config.max_restarts) {
706  syslog(LOG_ERR,
707  "plugin %s has exceeded max_restarts",
708  conf->p->path);
709  }
710  close(conf->p->plug_pipe[1]);
711  conf->p->plug_pipe[1] = -1;

804 static void process_inbound_event(int fd)
805 {
806  int rc;
807  struct iovec vec;
808  event_t *e = malloc(sizeof(event_t));
809  if (e == NULL)
810  return;
811  memset(e, 0, sizeof(event_t));
812 
813  /* Get header first. It is fixed size */
814  vec.iov_base = &e->hdr;
815  vec.iov_len = sizeof(struct audit_dispatcher_header);
816  do {
817  rc = readv(fd, &vec, 1);
818  } while (rc < 0 && errno == EINTR);
819 
820  if (rc <= 0) {
821  if (rc == 0)
822  stop = 1; // End of File
823  free(e);
824  return;
825  }
826 
827  if (rc > 0) {
828  /* Sanity check */
829  if (e->hdr.ver != AUDISP_PROTOCOL_VER ||
830  e->hdr.hlen != sizeof(e->hdr) ||
831  e->hdr.size > MAX_AUDIT_MESSAGE_LENGTH) {
832  free(e);
833  syslog(LOG_ERR,
834  "Dispatcher protocol mismatch, exiting");
835  exit(1);
836  }
837 
838  /* Next payload */
839  vec.iov_base = e->data;
840  vec.iov_len = e->hdr.size;
841  do {
842  rc = readv(fd, &vec, 1);
843  } while (rc < 0 && errno == EINTR);
844 
845  if (rc > 0)
846  enqueue(e, &daemon_config);
847  else {
848  if (rc == 0)
849  stop = 1; // End of File
850  free(e);
851  }
852  }
