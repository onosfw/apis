227 int load_config(remote_conf_t *config, const char *file)
228 {
229  int fd, rc, mode, lineno = 1;
230  struct stat st;
231  FILE *f;
232  char buf[128];
233 
234  clear_config(config);
235 
236  /* open the file */
237  mode = O_RDONLY;
238  rc = open(file, mode);
239  if (rc < 0) {
240  if (errno != ENOENT) {
241  syslog(LOG_ERR, "Error opening %s (%s)", file,
242  strerror(errno));
243  return 1;
244  }
245  syslog(LOG_WARNING,
246  "Config file %s doesn't exist, skipping", file);
247  return 0;
248  }
249  fd = rc;
250 
251  /* check the file's permissions: owned by root, not world writable,
252  * not symlink.
253  */
254  if (fstat(fd, &st) < 0) {
255  syslog(LOG_ERR, "Error fstat'ing config file (%s)",
256  strerror(errno));
257  close(fd);
258  return 1;
259  }
260  if (st.st_uid != 0) {
261  syslog(LOG_ERR, "Error - %s isn't owned by root",
262  file);
263  close(fd);
264  return 1;
265  }
266  if ((st.st_mode & S_IWOTH) == S_IWOTH) {
267  syslog(LOG_ERR, "Error - %s is world writable",
268  file);
269  close(fd);
270  return 1;
271  }
272  if (!S_ISREG(st.st_mode)) {
273  syslog(LOG_ERR, "Error - %s is not a regular file",
274  file);
275  close(fd);
276  return 1;
277  }
278 
279  /* it's ok, read line by line */
280  f = fdopen(fd, "rm");
281  if (f == NULL) {
282  syslog(LOG_ERR, "Error - fdopen failed (%s)",
283  strerror(errno));
284  close(fd);
285  return 1;
286  }
287 
288  while (get_line(f, buf)) {
289  // convert line into name-value pair
290  const struct kw_pair *kw;
291  struct nv_pair nv;
292  rc = nv_split(buf, &nv);
293  switch (rc) {
294  case 0: // fine
295  break;
296  case 1: // not the right number of tokens.
297  syslog(LOG_ERR,
298  "Wrong number of arguments for line %d in %s",
299  lineno, file);
300  break;
301  case 2: // no '=' sign
302  syslog(LOG_ERR,
303  "Missing equal sign for line %d in %s",
304  lineno, file);
305  break;
306  default: // something else went wrong...
307  syslog(LOG_ERR,
308  "Unknown error for line %d in %s",
309  lineno, file);
310  break;
311  }
312  if (nv.name == NULL) {
313  lineno++;
314  continue;
315  }
316  if (nv.value == NULL) {
317  fclose(f);
318  return 1;
319  }
320 
321  /* identify keyword or error */
322  kw = kw_lookup(nv.name);
323  if (kw->name == NULL) {
324  syslog(LOG_ERR,
325  "Unknown keyword \"%s\" in line %d of %s",
326  nv.name, lineno, file);
327  fclose(f);
328  return 1;
329  }
330 
331  /* Check number of options */
332  if (kw->max_options == 0 && nv.option != NULL) {
333  syslog(LOG_ERR,
334  "Keyword \"%s\" has invalid option "
335  "\"%s\" in line %d of %s",
336  nv.name, nv.option, lineno, file);
337  fclose(f);
338  return 1;
339  }
340 
341  /* dispatch to keyword's local parser */
342  rc = kw->parser(&nv, lineno, config);
343  if (rc != 0) {
344  fclose(f);
345  return 1; // local parser puts message out
346  }
347 
348  lineno++;
349  }
350 
351  fclose(f);
352  if (lineno > 1)
353  return sanity_check(config, file);
354  return 0;
355 }

369 static int nv_split(char *buf, struct nv_pair *nv)
370 {
371  /* Get the name part */
372  char *ptr, *saved;
373 
374  nv->name = NULL;
375  nv->value = NULL;
376  nv->option = NULL;
377  ptr = strtok_r(buf, " ", &saved);
378  if (ptr == NULL)
379  return 0; /* If there's nothing, go to next line */
380  if (ptr[0] == '#')
381  return 0; /* If there's a comment, go to next line */
382  nv->name = ptr;
383 
384  /* Check for a '=' */
385  ptr = strtok_r(NULL, " ", &saved);
386  if (ptr == NULL)
387  return 1;
388  if (strcmp(ptr, "=") != 0)
389  return 2;
390 
391  /* get the value */
392  ptr = strtok_r(NULL, " ", &saved);
393  if (ptr == NULL)
394  return 1;
395  nv->value = ptr;
396 
397  /* See if there's an option */
398  ptr = strtok_r(NULL, " ", &saved);
399  if (ptr) {
400  nv->option = ptr;
401 
402  /* Make sure there's nothing else */
403  ptr = strtok_r(NULL, " ", &saved);
404  if (ptr)
405  return 1;
406  }
407 
408  /* Everything is OK */
409  return 0;
410 }

423 static int check_exe_name(const char *val, int line)
424 {
425  struct stat buf;
426 
427  if (val == NULL) {
428  syslog(LOG_ERR, "Executable path needed for line %d", line);
429  return -1;
430  }
431 
432  if (*val != '/') {
433  syslog(LOG_ERR, "Absolute path needed for %s - line %d",
434  val, line);
435  return -1;
436  }
437 
438  if (stat(val, &buf) < 0) {
439  syslog(LOG_ERR, "Unable to stat %s (%s) - line %d", val,
440  strerror(errno), line);
441  return -1;
442  }
443  if (!S_ISREG(buf.st_mode)) {
444  syslog(LOG_ERR, "%s is not a regular file - line %d", val,
445  line);
446  return -1;
447  }
448  if (buf.st_uid != 0) {
449  syslog(LOG_ERR, "%s is not owned by root - line %d", val,
450  line);
451  return -1;
452  }
453  if ((buf.st_mode & (S_IRWXU|S_IRWXG|S_IWOTH)) !=
454  (S_IRWXU|S_IRGRP|S_IXGRP)) {
455  syslog(LOG_ERR, "%s permissions should be 0750 - line %d", val,
456  line);
457  return -1;
458  }
459  return 0;
460 }

472 static int parse_uint (const struct nv_pair *nv, int line, unsigned int *valp,
473  unsigned int min, unsigned int max)
474 {
475  const char *ptr = nv->value;
476  unsigned int i;
477 
478  /* check that all chars are numbers */
479  for (i=0; ptr[i]; i++) {
480  if (!isdigit(ptr[i])) {
481  syslog(LOG_ERR,
482  "Value %s should only be numbers - line %d",
483  nv->value, line);
484  return 1;
485  }
486  }
487 
488  /* convert to unsigned int */
489  errno = 0;
490  i = strtoul(nv->value, NULL, 10);
491  if (errno) {
492  syslog(LOG_ERR,
493  "Error converting string to a number (%s) - line %d",
494  strerror(errno), line);
495  return 1;
496  }
497  /* Check its range */
498  if (min != 0 && i < (int)min) {
499  syslog(LOG_ERR,
500  "Error - converted number (%s) is too small - line %d",
501  nv->value, line);
502  return 1;
503  }
504  if (max != 0 && i > max) {
505  syslog(LOG_ERR,
506  "Error - converted number (%s) is too large - line %d",
507  nv->value, line);
508  return 1;
509  }
510  *valp = (unsigned int)i;
511  return 0;
512 }
