233 static int do_action (const char *desc, const char *message,
234  int log_level,
235  failure_action_t action, const char *exe)
236 {
237  switch (action)
238  {
239  case FA_IGNORE:
240  return 0;
241  case FA_SYSLOG:
242  syslog (log_level, "%s, %s", desc, message);
243  return 0;
244  case FA_EXEC:
245  safe_exec (exe, message);
246  return 0;
247  case FA_SUSPEND:
248  syslog (log_level,
249  "suspending remote logging due to %s", desc);
250  suspend = 1;
251  return 0;
252  case FA_RECONNECT:
253  syslog (log_level,
254  "remote logging disconnected due to %s, will attempt reconnection",
255  desc);
256  return 0;
257  case FA_SINGLE:
258  syslog (log_level,
259  "remote logging is switching system to single user mode due to %s",
260  desc);
261  change_runlevel(SINGLE);
262  return -1;
263  case FA_HALT:
264  syslog (log_level,
265  "remote logging halting system due to %s", desc);
266  change_runlevel(HALT);
267  return -1;
268  case FA_STOP:
269  syslog (log_level, "remote logging stopping due to %s, %s",
270  desc, message);
271  stop = 1;
272  return -1;
273  }
274  syslog (log_level, "unhandled action %d for %s", action, desc);
275  return -1;
276 }

396 /* Send a record from QUEUE to the remote system */
397 static void send_one(struct queue *queue)
398 {
399  char event[MAX_AUDIT_MESSAGE_LENGTH];
400  int len;
401 
402  if (suspend || !transport_ok)
403  return;
404 
405  len = q_peek(queue, event, sizeof(event));
406  if (len == 0)
407  return;
408  if (len < 0) {
409  queue_error();
410  return;
411  }
412 
413  /* We send len -1 to remove trailing \n */
414  if (relay_event(event, len-1) < 0)
415  return;
416 
417  if (q_drop_head(queue) != 0)
418  queue_error();
419 }

421 int main(int argc, char *argv[])
422 {
423  struct sigaction sa;
424  struct queue *queue;
425  int rc;
426  size_t q_len;
427 
428  /* Register sighandlers */
429  sa.sa_flags = 0;
430  sigemptyset(&sa.sa_mask);
431  /* Set handler for the ones we care about */
432  sa.sa_handler = term_handler;
433  sigaction(SIGTERM, &sa, NULL);
434  sa.sa_handler = hup_handler;
435  sigaction(SIGHUP, &sa, NULL);
436  sa.sa_handler = user1_handler;
437  sigaction(SIGUSR1, &sa, NULL);
438  sa.sa_handler = user2_handler;
439  sigaction(SIGUSR2, &sa, NULL);
440  sa.sa_handler = child_handler;
441  sigaction(SIGCHLD, &sa, NULL);
442  if (load_config(&config, CONFIG_FILE))
443  return 6;
444 
445  (void) umask( umask( 077 ) | 027 );
446  // ifd = open("test.log", O_RDONLY);
447  ifd = 0;
448  fcntl(ifd, F_SETFL, O_NONBLOCK);
449 
450  /* We fail here if the transport can't be initialized because of some
451  * permanent (i.e. operator) problem, such as misspelled host name. */
452  rc = init_transport();
453  if (rc == ET_PERMANENT)
454  return 1;
455  queue = init_queue();
456  if (queue == NULL) {
457  syslog(LOG_ERR, "Error initializing audit record queue: %m");
458  return 1;
459  }
460 
461 #ifdef HAVE_LIBCAP_NG
462  // Drop capabilities
463  capng_clear(CAPNG_SELECT_BOTH);
464  if (config.local_port && config.local_port < 1024)
465  capng_update(CAPNG_ADD, CAPNG_EFFECTIVE|CAPNG_PERMITTED,
466  CAP_NET_BIND_SERVICE);
467  capng_apply(CAPNG_SELECT_BOTH);
468 #endif
469  syslog(LOG_NOTICE, "Audisp-remote started with queue_size: %zu",
470  q_queue_length(queue));
471 
472  while (stop == 0) { //FIXME break out when socket is closed
473  fd_set rfd, wfd;
474  struct timeval tv;
475  char event[MAX_AUDIT_MESSAGE_LENGTH];
476  int n, fds = ifd + 1;
477 
478  /* Load configuration */
479  if (hup)
480  reload_config();
481 
482  if (dump)
483  dump_stats(queue);
484 
485  /* Setup select flags */
486  FD_ZERO(&rfd);
487  FD_SET(ifd, &rfd); // input fd
488  FD_ZERO(&wfd);
489  if (sock > 0) {
490  // Setup socket to read acks from server
491  FD_SET(sock, &rfd); // remote socket
492  if (sock > ifd)
493  fds = sock + 1;
494  // If we have anything in the queue,
495  // find out if we can send it
496  if (q_queue_length(queue) && !suspend && transport_ok)
497  FD_SET(sock, &wfd);
498  }
499 
500  if (config.heartbeat_timeout > 0) {
501  tv.tv_sec = config.heartbeat_timeout;
502  tv.tv_usec = 0;
503  n = select(fds, &rfd, &wfd, NULL, &tv);
504  } else
505  n = select(fds, &rfd, &wfd, NULL, NULL);
506  if (n < 0)
507  continue; // If here, we had some kind of problem
508 
509  if ((config.heartbeat_timeout > 0) && n == 0 && !remote_ended) {
510  /* We attempt a hearbeat if select fails, which
511  * may give us more heartbeats than we need. This
512  * is safer than too few heartbeats. */
513  quiet = 1;
514  send_heartbeat();
515  quiet = 0;
516  continue;
517  }
518 
519  // See if we got a shutdown message from the server
520  if (sock > 0 && FD_ISSET(sock, &rfd))
521  check_message();
522 
523  // If we broke out due to one of these, cycle to start
524  if (hup != 0 || stop != 0)
525  continue;
526 
527  // See if input fd is also set
528  if (FD_ISSET(ifd, &rfd)) {
529  do {
530  if (remote_fgets(event, sizeof(event), ifd)) {
531  if (!transport_ok && remote_ended &&
532  config.remote_ending_action ==
533  FA_RECONNECT) {
534  quiet = 1;
535  if (init_transport() ==
536  ET_SUCCESS)
537  remote_ended = 0;
538  quiet = 0;
539  }
540  /* Strip out EOE records */
541  if (*event == 't') {
542  if (strncmp(event,
543  "type=EOE", 8) == 0)
544  continue;
545  } else {
546  char *ptr = strchr(event, ' ');
547  if (ptr) {
548  ptr++;
549  if (strncmp(ptr,
550  "type=EOE",
551  8) == 0)
552  continue;
553  } else
554  continue; //malformed
555  }
556  if (q_append(queue, event) != 0) {
557  if (errno == ENOSPC)
558  do_overflow_action();
559  else
560  queue_error();
561  }
562  } else if (remote_fgets_eof())
563  stop = 1;
564  } while (remote_fgets_more(sizeof(event)));
565  }
566  // See if output fd is also set
567  if (sock > 0 && FD_ISSET(sock, &wfd)) {
568  // If so, try to drain backlog
569  while (q_queue_length(queue) && !suspend &&
570  !stop && transport_ok)
571  send_one(queue);
572  }
573  }
574  if (sock >= 0) {
575  shutdown(sock, SHUT_RDWR);
576  close(sock);
577  }
578  free_config(&config);
579  q_len = q_queue_length(queue);
580  q_close(queue);
581  if (stop)
582  syslog(LOG_NOTICE, "audisp-remote is exiting on stop request, queue_size: %zu", q_len);
583 
584  return q_len ? 1 : 0;
585 }

589 /* Communications under GSS is done by token exchanges. Each "token" may
590  contain a message, perhaps signed, perhaps encrypted. The messages within
591  are what we're interested in, but the network sees the tokens. The
592  protocol we use for transferring tokens is to send the length first,
593  four bytes MSB first, then the token data. We return nonzero on error. */
594 static int recv_token(int s, gss_buffer_t tok)
595 {
596  int ret;
597  unsigned char lenbuf[4];
598  unsigned int len;
599 
600  ret = ar_read(s, (char *) lenbuf, 4);
601  if (ret < 0) {
602  syslog(LOG_ERR, "GSS-API error reading token length");
603  return -1;
604  } else if (!ret) {
605  return 0;
606  } else if (ret != 4) {
607  syslog(LOG_ERR, "GSS-API error reading token length");
608  return -1;
609  }
610 
611  len = ( ((uint32_t)(lenbuf[0] & 0xFF) << 24)
612  | ((uint32_t)(lenbuf[1] & 0xFF) << 16)
613  | ((uint32_t)(lenbuf[2] & 0xFF) << 8)
614  | (uint32_t)(lenbuf[3] & 0xFF));
615 
616  if (len > MAX_AUDIT_MESSAGE_LENGTH) {
617  syslog(LOG_ERR,
618  "GSS-API error: event length excedes MAX_AUDIT_LENGTH");
619  return -1;
620  }
621  tok->length = len;
622  tok->value = (char *) malloc(tok->length ? tok->length : 1);
623  if (tok->length && tok->value == NULL) {
624  syslog(LOG_ERR, "Out of memory allocating token data %zd %zx",
625  tok->length, tok->length);
626  return -1;
627  }
628 
629  ret = ar_read(s, (char *) tok->value, tok->length);
630  if (ret < 0) {
631  syslog(LOG_ERR, "GSS-API error reading token data");
632  free(tok->value);
633  return -1;
634  } else if (ret != (int) tok->length) {
635  syslog(LOG_ERR, "GSS-API error reading token data");
636  free(tok->value);
637  return -1;
638  }
639 
640  return 0;
641 }

643 /* Same here. */
644 int send_token(int s, gss_buffer_t tok)
645 {
646  int ret;
647  unsigned char lenbuf[4];
648  unsigned int len;
649 
650  if (tok->length > 0xffffffffUL)
651  return -1;
652 
653  len = tok->length;
654  lenbuf[0] = (len >> 24) & 0xff;
655  lenbuf[1] = (len >> 16) & 0xff;
656  lenbuf[2] = (len >> 8) & 0xff;
657  lenbuf[3] = len & 0xff;
658 
659  ret = ar_write(s, (char *) lenbuf, 4);
660  if (ret < 0) {
661  syslog(LOG_ERR, "GSS-API error sending token length");
662  return -1;
663  } else if (ret != 4) {
664  syslog(LOG_ERR, "GSS-API error sending token length");
665  return -1;
666  }
667 
668  ret = ar_write(s, tok->value, tok->length);
669  if (ret < 0) {
670  syslog(LOG_ERR, "GSS-API error sending token data");
671  return -1;
672  } else if (ret != (int) tok->length) {
673  syslog(LOG_ERR, "GSS-API error sending token data");
674  return -1;
675  }
676 
677  return 0;
678 }

719  of kinit here. When our tickets expire, the usual close/open/retry logic
720  has us calling here again, where we re-init and get new tickets. */

730  OM_uint32 ret_flags;

732  /* Getting an initial ticket is outside the scope of GSS, so
733  we use Kerberos calls here. */
734

736  krb5_context kcontext = NULL;
737  char *realm_name;
738  krb5_principal audit_princ;
739  krb5_ccache ccache = NULL;
740  krb5_creds my_creds;
741  krb5_get_init_creds_opt options;
742  krb5_keytab keytab = NULL;
743  const char *krb5_client_name;
744  char *slashptr;
745  char host_name[255];
746  struct stat st;
747  const char *key_file;
748 
749  token_ptr = GSS_C_NO_BUFFER;
750  *gss_context = GSS_C_NO_CONTEXT;
751  recv_tok.value = NULL;
752

754  KCHECK (krberr, "krb5_init_context");
755

757  key_file = config.krb5_key_file;
758  else

761  setenv ("KRB5_KTNAME", key_file, 1);
762

764  if ((st.st_mode & 07777) != 0400) {
765  if (!quiet)
766  syslog (LOG_ERR,
767  "%s is not mode 0400 (it's %#o) - compromised key?",
768  key_file, st.st_mode & 07777);
769  return -1;
770  }
771  if (st.st_uid != 0) {
772  if (!quiet)
773  syslog (LOG_ERR,
774  "%s is not owned by root (it's %d) - compromised key?",
775  key_file, st.st_uid);
776  return -1;
777  }
778  }

781  /etc/krb5.conf (or wherever) */
782  krberr = krb5_get_default_realm (kcontext, &realm_name);
783  KCHECK (krberr, "krb5_get_default_realm");
784 
785  krb5_client_name = config.krb5_client_name ?
786  config.krb5_client_name : "auditd";

788  if (!quiet)
789  syslog (LOG_ERR,
790  "gethostname: host name longer than %ld characters?",
791  sizeof (host_name));
792  return -1;
793  }
794 
795  syslog (LOG_ERR, "kerberos principal: %s/%s@%s\n",
796  krb5_client_name, host_name, realm_name);
797  /* Encode our own "name" as auditd/remote@EXAMPLE.COM. */
798  krberr = krb5_build_principal (kcontext, &audit_princ,
799  strlen(realm_name), realm_name,

801  KCHECK (krberr, "krb5_build_principal");
802 
803  /* Locate our machine's key table, where our private key is
804  * held. */
805  krberr = krb5_kt_resolve (kcontext, key_file, &keytab);
806  KCHECK (krberr, "krb5_kt_resolve");
807 
808  /* Identify a cache to hold the key in. The GSS wrappers look
809  up our credentials here. */
810  krberr = krb5_cc_resolve (kcontext, CCACHE_NAME, &ccache);
811  KCHECK (krberr, "krb5_cc_resolve");
812 
813  setenv("KRB5CCNAME", CCACHE_NAME, 1);
814

816  memset(&options, 0, sizeof(options));
817  krb5_get_init_creds_opt_set_address_list(&options, NULL);
818  krb5_get_init_creds_opt_set_forwardable(&options, 0);
819  krb5_get_init_creds_opt_set_proxiable(&options, 0);
820  krb5_get_init_creds_opt_set_tkt_life(&options, 24*60*60);
821 
822  /* Load our credentials from the key table. */
823  krberr = krb5_get_init_creds_keytab(kcontext, &my_creds, audit_princ,
824  keytab, 0, NULL,

826  KCHECK (krberr, "krb5_get_init_creds_keytab");
827 
828  /* Create the cache... */
829  krberr = krb5_cc_initialize(kcontext, ccache, audit_princ);
830  KCHECK (krberr, "krb5_cc_initialize");
831 
832  /* ...and store our credentials in it. */
833  krberr = krb5_cc_store_cred(kcontext, ccache, &my_creds);
834  KCHECK (krberr, "krb5_cc_store_cred");
835

837  I.e. we know who "we" are. Now we talk to the server to
838  get its credentials and set up a security context for encryption. */
839  if (config.krb5_principal == NULL) {
840  const char *name = config.krb5_client_name ?
841  config.krb5_client_name : "auditd";
842  config.krb5_principal = (char *) malloc (strlen (name) + 1
843  + strlen (config.remote_server) + 1);
844  sprintf((char *)config.krb5_principal, "%s@%s",
845  name, config.remote_server);
846  }
847  slashptr = strchr (config.krb5_principal, '/');
848  if (slashptr)
849  *slashptr = '@';
850 
851  name_buf.value = (char *)config.krb5_principal;
852  name_buf.length = strlen(name_buf.value) + 1;
853  major_status = gss_import_name(&minor_status, &name_buf,
854  (gss_OID) gss_nt_service_name, &service_name_e);
855  if (major_status != GSS_S_COMPLETE) {
856  gss_failure("importing name", major_status, minor_status);
857  return -1;
858  }
859 
860  /* Someone has to go first. In this case, it's us. */
861  if (send_token(sock, empty_token) < 0) {
862  (void) gss_release_name(&minor_status, &service_name_e);
863  return -1;
864  }
865 
866  /* The server starts this loop with the token we just sent
867  (the empty one). We start this loop with "no token". */
868  token_ptr = GSS_C_NO_BUFFER;
869  *gss_context = GSS_C_NO_CONTEXT;
870 
871  do {
872  /* Give GSS a chance to digest what we have so far. */
873  major_status = gss_init_sec_context(&init_sec_min_stat,
874  GSS_C_NO_CREDENTIAL, gss_context,
875  service_name_e, NULL, REQ_FLAGS, 0,
876  NULL, /* no channel bindings */
877  token_ptr, NULL, /* ignore mech type */
878  &send_tok, &ret_flags, NULL); /* ignore time_rec */
879 
880  if (token_ptr != GSS_C_NO_BUFFER)
881  free(recv_tok.value);
882 
883  /* Send the server any tokens requested of us. */
884  if (send_tok.length != 0) {
885  if (send_token(sock, &send_tok) < 0) {
886  (void) gss_release_buffer(&minor_status,
887  &send_tok);
888  (void) gss_release_name(&minor_status,
889  &service_name_e);
890  return -1;
891  }
892  }
893  (void) gss_release_buffer(&minor_status, &send_tok);
894 
895  if (major_status != GSS_S_COMPLETE
896  && major_status != GSS_S_CONTINUE_NEEDED) {
897  gss_failure("initializing context", major_status,
898  init_sec_min_stat);
899  (void) gss_release_name(&minor_status, &service_name_e);
900  if (*gss_context != GSS_C_NO_CONTEXT)
901  gss_delete_sec_context(&minor_status,
902  gss_context, GSS_C_NO_BUFFER);
903  return -1;
904  }
905 
906  /* Now get any tokens the sever sends back. We use
907  these back at the top of the loop. */
908  if (major_status == GSS_S_CONTINUE_NEEDED) {
909  if (recv_token(sock, &recv_tok) < 0) {
910  (void) gss_release_name(&minor_status,
911  &service_name_e);
912  return -1;
913  }
914  token_ptr = &recv_tok;
915  }
916  } while (major_status == GSS_S_CONTINUE_NEEDED);
917 
918  (void) gss_release_name(&minor_status, &service_name_e);
919 
920 #if 0
921  major_status = gss_inquire_context (&minor_status, &my_context, NULL,
922  &service_name_e, NULL, NULL,
923  NULL, NULL, NULL);
924  if (major_status != GSS_S_COMPLETE) {
925  gss_failure("inquiring target name", major_status, minor_status);
926  return -1;
927  }
928  major_status = gss_display_name(&minor_status, service_name_e,
929  &recv_tok, NULL);
930  gss_release_name(&minor_status, &service_name_e);
931  if (major_status != GSS_S_COMPLETE) {
932  gss_failure("displaying name", major_status, minor_status);
933  return -1;
934  }

936  (char *)recv_tok.value);
937 #endif

970 static int init_sock(void)
971 {

974  struct addrinfo hints;
975  char remote[BUF_SIZE];
976  int one=1;
977 
978  if (sock >= 0) {
979  syslog(LOG_NOTICE, "socket already setup");
980  transport_ok = 1;
981  return ET_SUCCESS;
982  }
983  memset(&hints, '\0', sizeof(hints));
984  hints.ai_flags = AI_ADDRCONFIG|AI_NUMERICSERV;
985  hints.ai_socktype = SOCK_STREAM;

988  if (rc) {
989  if (!quiet)
990  syslog(LOG_ERR,
991  "Error looking up remote host: %s - exiting",
992  gai_strerror(rc));
993  if (rc == EAI_NONAME || rc == EAI_NODATA)
994  return ET_PERMANENT;
995  else
996  return ET_TEMPORARY;
997  }
998  sock = socket(ai->ai_family, ai->ai_socktype, ai->ai_protocol);
999  if (sock < 0) {
1000  if (!quiet)
1001  syslog(LOG_ERR, "Error creating socket: %s",
1002  strerror(errno));
1003  freeaddrinfo(ai);
1004  return ET_TEMPORARY;
1005  }
1006 
1007  setsockopt(sock, SOL_SOCKET, SO_REUSEADDR, (char *)&one, sizeof (int));
1008 
1009  if (config.local_port != 0) {
1010  struct sockaddr_in address;
1011 
1012  memset (&address, 0, sizeof(address));
1013  address.sin_family = AF_INET;
1014  address.sin_port = htons(config.local_port);
1015  address.sin_addr.s_addr = htonl(INADDR_ANY);
1016 
1017  if (bind(sock, (struct sockaddr *)&address, sizeof(address))) {
1018  if (!quiet)
1019  syslog(LOG_ERR,
1020  "Cannot bind local socket to port %d",
1021  config.local_port);
1022  stop_sock();
1023  freeaddrinfo(ai);
1024  return ET_TEMPORARY;
1025  }
1026 
1027  }
1028  if (connect(sock, ai->ai_addr, ai->ai_addrlen)) {
1029  if (!quiet)
1030  syslog(LOG_ERR, "Error connecting to %s: %s",
1031  config.remote_server, strerror(errno));
1032  freeaddrinfo(ai);
1033  stop_sock();
1034  return ET_TEMPORARY;
1035  }
1036 
1037  freeaddrinfo(ai);
1038  setsockopt(sock, SOL_SOCKET, SO_KEEPALIVE, (char *)&one, sizeof (int));

1041  and the ACK, assuming that individual messages are
1042  infrequent enough that we can ignore the inefficiency of
1043  sending the header and message in separate packets. */
1044  if (config.format == F_MANAGED)
1045  setsockopt(sock, IPPROTO_TCP, TCP_NODELAY,
1046  (char *)&one, sizeof (int));
1047 
1048 #ifdef USE_GSSAPI
1049  if (USE_GSS) {
1050  if (negotiate_credentials ())
1051  return ET_PERMANENT;
1052  }
1053 #endif
1054 
1055  transport_ok = 1;
1056  syslog(LOG_NOTICE, "Connected to %s", config.remote_server);
1057  return ET_SUCCESS;
1058 }

1079 static int ar_write (int sk, const void *buf, int len)
1080 {
1081  int rc = 0, r;
1082  while (len > 0) {
1083  do {
1084  r = write(sk, buf, len);
1085  } while (r < 0 && errno == EINTR);
1086  if (r < 0) {
1087  if (errno == EPIPE)
1088  stop_sock();
1089  return r;
1090  }
1091  if (r == 0)
1092  break;
1093  rc += r;
1094  buf = (void *)((char *)buf + r);
1095  len -= r;
1096  }
1097  return rc;
1098 }

1419  if (type == AUDIT_RMW_TYPE_ENDING)

1433  if (type == AUDIT_RMW_TYPE_DISKLOW)
