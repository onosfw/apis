258 int ausearch_time_start(const char *da, const char *ti)
259 {
260 /* If da == NULL, use current date */
261 /* If ti == NULL, then use midnight 00:00:00 */
262  int rc = 0;
263  struct tm d, t;
264  char *ret;
265 
266  if (da == NULL)
267  set_tm_now(&d);
268  else {
269  if (lookup_and_set_time(da, &d) < 0) {
270  ret = strptime(da, "%x", &d);
271  if (ret == NULL) {
272  fprintf(stderr,
273  "Invalid start date (%s). Month, Day, and Year are required.\n",
274  da);
275  return 1;
276  }
277  if (*ret != 0) {
278  fprintf(stderr,
279  "Error parsing start date (%s)\n", da);
280  return 1;
281  }
282  } else {
283  int keyword=lookup_time(da);
284  if (keyword == T_RECENT || keyword == T_NOW) {
285  if (ti == NULL || strcmp(ti, "00:00:00") == 0)
286  goto set_it;
287  }
288  }
289  }
290 
291  if (ti != NULL) {
292  char tmp_t[36];
293 
294  if (strlen(ti) <= 5) {
295  snprintf(tmp_t, sizeof(tmp_t), "%s:00", ti);
296  } else {
297  tmp_t[0]=0;
298  strncat(tmp_t, ti, sizeof(tmp_t)-1);
299  }
300  ret = strptime(tmp_t, "%X", &t);
301  if (ret == NULL) {
302  fprintf(stderr,
303  "Invalid start time (%s). Hour, Minute, and Second are required.\n",
304  ti);
305  return 1;
306  }
307  if (*ret != 0) {
308  fprintf(stderr, "Error parsing start time (%s)\n", ti);
309  return 1;
310  }
311  } else
312  clear_tm(&t);
313 
314  add_tm(&d, &t);
315  if (d.tm_year < 104) {
316  fprintf(stderr, "Error - year is %d\n", d.tm_year+1900);
317  return -1;
318  }
319 set_it:
320  // printf("start is: %s\n", ctime(&start_time));
321  start_time = mktime(&d);
322  if (start_time == -1) {
323  fprintf(stderr, "Error converting start time\n");
324  rc = -1;
325  }
326  return rc;
327 }
