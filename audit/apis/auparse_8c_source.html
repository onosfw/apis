48 static int setup_log_file_array(auparse_state_t *au)
49 {
50  struct daemon_conf config;
51  char *filename, **tmp;
52  int len, num = 0, i = 0;
53 
54  /* Load config so we know where logs are */
55  set_aumessage_mode(MSG_STDERR, DBG_NO);
56  load_config(&config, TEST_SEARCH);
57 
58  /* for each file */
59  len = strlen(config.log_file) + 16;
60  filename = malloc(len);
61  if (!filename) {
62  fprintf(stderr, "No memory\n");
63  free_config(&config);
64  return 1;
65  }
66  /* Find oldest log file */
67  snprintf(filename, len, "%s", config.log_file);
68  do {
69  if (access(filename, R_OK) != 0)
70  break;
71  num++;
72  snprintf(filename, len, "%s.%d", config.log_file, num);
73  } while (1);
74 
75  if (num == 0) {
76  fprintf(stderr, "No log file\n");
77  free_config(&config);
78  free(filename);
79  return 1;
80  }
81  num--;
82  tmp = malloc((num+2)*sizeof(char *));
83 
84  /* Got it, now process logs from last to first */
85  if (num > 0)
86  snprintf(filename, len, "%s.%d", config.log_file, num);
87  else
88  snprintf(filename, len, "%s", config.log_file);
89  do {
90  tmp[i++] = strdup(filename);
91 
92  /* Get next log file */
93  num--;
94  if (num > 0)
95  snprintf(filename, len, "%s.%d", config.log_file, num);
96  else if (num == 0)
97  snprintf(filename, len, "%s", config.log_file);
98  else
99  break;
100  } while (1);
101  free_config(&config);
102  free(filename);
103 
104  // Terminate the list
105  tmp[i] = NULL;
106  au->source_list = tmp;
107  return 0;
108 }

111 auparse_state_t *auparse_init(ausource_t source, const void *b)
112 {
113  char **tmp, **bb = (char **)b, *buf = (char *)b;
114  int n, i;
115  size_t size, len;
116 
117  auparse_state_t *au = malloc(sizeof(auparse_state_t));
118  if (au == NULL) {
119  errno = ENOMEM;
120  return NULL;
121  }
122 
123  au->in = NULL;
124  au->source_list = NULL;
125  databuf_init(&au->databuf, 0, 0);
126  au->callback = NULL;
127  au->callback_user_data = NULL;
128  au->callback_user_data_destroy = NULL;
129  switch (source)
130  {
131  case AUSOURCE_LOGS:
132  if (geteuid()) {
133  errno = EPERM;
134  goto bad_exit;
135  }
136  setup_log_file_array(au);
137  break;
138  case AUSOURCE_FILE:
139  if (access(b, R_OK))
140  goto bad_exit;
141  tmp = malloc(2*sizeof(char *));
142  tmp[0] = strdup(b);
143  tmp[1] = NULL;
144  au->source_list = tmp;
145  break;
146  case AUSOURCE_FILE_ARRAY:
147  n = 0;
148  while (bb[n]) {
149  if (access(bb[n], R_OK))
150  goto bad_exit;
151  n++;
152  }
153  tmp = malloc((n+1)*sizeof(char *));
154  for (i=0; i<n; i++)
155  tmp[i] = strdup(bb[i]);
156  tmp[n] = NULL;
157  au->source_list = tmp;
158  break;
159  case AUSOURCE_BUFFER:
160  buf = buf;
161  len = strlen(buf);
162  if (databuf_init(&au->databuf, len,
163  DATABUF_FLAG_PRESERVE_HEAD) < 0)
164  goto bad_exit;
165  if (databuf_append(&au->databuf, buf, len) < 0)
166  goto bad_exit;
167  break;
168  case AUSOURCE_BUFFER_ARRAY:
169  size = 0;
170  for (n = 0; (buf = bb[n]); n++) {
171  len = strlen(bb[n]);
172  if (bb[n][len-1] != '\n') {
173  size += len + 1;
174  } else {
175  size += len;
176  }
177  }
178  if (databuf_init(&au->databuf, size,
179  DATABUF_FLAG_PRESERVE_HEAD) < 0)
180  goto bad_exit;
181  for (n = 0; (buf = bb[n]); n++) {
182  len = strlen(buf);
183  if (databuf_append(&au->databuf, buf, len) < 0)
184  goto bad_exit;
185  }
186  break;
187  case AUSOURCE_DESCRIPTOR:
188  n = (long)b;
189  au->in = fdopen(n, "rm");
190  break;
191  case AUSOURCE_FILE_POINTER:
192  au->in = (FILE *)b;
193  break;
194  case AUSOURCE_FEED:
195  if (databuf_init(&au->databuf, 0, 0) < 0) goto bad_exit;
196  break;
197  default:
198  errno = EINVAL;
199  goto bad_exit;
200  break;
201  }
202  au->source = source;
203  au->list_idx = 0;
204  au->line_number = 0;
205  au->next_buf = NULL;
206  au->off = 0;
207  au->cur_buf = NULL;
208  au->line_pushed = 0;
209  aup_list_create(&au->le);
210  au->parse_state = EVENT_EMPTY;
211  au->expr = NULL;
212  au->find_field = NULL;
213  au->search_where = AUSEARCH_STOP_EVENT;
214 
215  return au;
216 bad_exit:
217  databuf_free(&au->databuf);
218  free(au);
219  return NULL;
220 }

335 static int add_expr(auparse_state_t *au, struct expr *expr, ausearch_rule_t how)
336 {
337  if (au->expr == NULL)
338  au->expr = expr;
339  else if (how == AUSEARCH_RULE_CLEAR) {
340  expr_free(au->expr);
341  au->expr = expr;
342  } else {
343  struct expr *e;

346  au->expr, expr);

348  int err;
349

351  expr_free(expr);
352  errno = err;
353  return -1;
354  }
355  au->expr = e;
356  }

361  const char *op, const char *value, ausearch_rule_t how, unsigned op_eq,
362  unsigned op_ne)
363 {

365 
366  // Make sure there's a field

375  if (strcmp(op, "exists") == 0)

391  return -1;
392  if (add_expr(au, expr, how) != 0)

394  return 0;

567 /* alloc a new buffer, cur_buf which contains a null terminated line
568  * without a newline (note, this implies the line may be empty (strlen == 0)) if
569  * successfully read a blank line (e.g. containing only a single newline).
570  * cur_buf will have been newly allocated with malloc.
571  *
572  * Note: cur_buf will be freed the next time this routine is called if
573  * cur_buf is not NULL, callers who retain a reference to the cur_buf
574  * pointer will need to set cur_buf to NULL to cause the previous cur_buf
575  * allocation to persist.
576  *
577  * Returns:
578  * 1 if successful (errno == 0)

580  * -1 if error (errno contains non-zero error code)
581  * -2 if EOF (errno == 0)

584 static int readline_file(auparse_state_t *au)
585 {
586  ssize_t rc;
587  char *p_last_char;
588  size_t n = 0;
589 
590  if (au->cur_buf != NULL) {
591  free(au->cur_buf);
592  au->cur_buf = NULL;
593  }
594  if (au->in == NULL) {
595  errno = EBADF;
596  return -1;
597  }
598  if ((rc = getline(&au->cur_buf, &n, au->in)) <= 0) {
599  // Note: getline always malloc's if lineptr==NULL or n==0,
600  // on failure malloc'ed memory is left uninitialized,
601  // caller must free it.
602  free(au->cur_buf);
603  au->cur_buf = NULL;
604 
605  // Note: feof() does not set errno
606  if (feof(au->in)) {
607  // return EOF condition
608  errno = 0;
609  return -2;
610  }
611  // return error condition, error code in errno
612  return -1;
613  }
614  p_last_char = au->cur_buf + (rc-1);
615  if (*p_last_char == '\n') { /* nuke newline */
616  *p_last_char = 0;
617  }
618  // return success
619  errno = 0;
620  return 1;
621 }

642 static int readline_buf(auparse_state_t *au)
643 {
644  char *p_newline=NULL;
645  size_t line_len;
646 
647  if (au->cur_buf != NULL) {
648  free(au->cur_buf);
649  au->cur_buf = NULL;
650  }
651 
652  //if (debug) databuf_print(&au->databuf, 1, "readline_buf");
653  if (au->databuf.len == 0) {
654  // return EOF condition
655  errno = 0;
656  return -2;
657  }
658 
659  if ((p_newline = strnchr(databuf_beg(&au->databuf), '\n',
660  au->databuf.len)) != NULL) {
661  line_len = p_newline - databuf_beg(&au->databuf);
662 
663  /* dup the line */
664  au->cur_buf = malloc(line_len+1); // +1 for null terminator
665  if (au->cur_buf == NULL)
666  return -1; // return error condition, errno set
667  strncpy(au->cur_buf, databuf_beg(&au->databuf), line_len);
668  au->cur_buf[line_len] = 0;
669 
670  if (databuf_advance(&au->databuf, line_len+1) < 0)
671  return -1;
672  // return success
673  errno = 0;
674  return 1;
675 
676  } else {
677  // return no data available
678  errno = 0;
679  return 0;
680  }
681 }

778 /* This function will figure out how to get the next line of input.
779  * storing it cur_buf. cur_buf will be NULL terminated but will not
780  * contain a trailing newline. This implies a successful read
781  * (result == 1) may result in a zero length cur_buf if a blank line
782  * was read.
783  *
784  * cur_buf will have been allocated with malloc. The next time this
785  * routine is called if cur_buf is non-NULL cur_buf will be freed,
786  * thus if the caller wishes to retain a reference to malloc'ed
787  * cur_buf data it should copy the cur_buf pointer and set cur_buf to
788  * NULL.
789  *
790  * Returns:
791  * 1 if successful (errno == 0)
792  * 0 if non-blocking input unavailable (errno == 0)
793  * -1 if error (errno contains non-zero error code)
794  * -2 if EOF (errno == 0)
795  */

797 static int retrieve_next_line(auparse_state_t *au)
798 {
799  int rc;
800 
801  // If line was pushed back for re-reading return that
802  if (au->line_pushed) {
803  // Starting new event, clear previous event data,
804  // previous line is returned again for new parsing
805  au->line_pushed = 0;
806  au->line_number++;
807  return 1;
808  }
809 
810  switch (au->source)
811  {
812  case AUSOURCE_DESCRIPTOR:
813  case AUSOURCE_FILE_POINTER:
814  rc = readline_file(au);
815  if (rc > 0) au->line_number++;
816  return rc;
817  case AUSOURCE_LOGS:
818  case AUSOURCE_FILE:
819  case AUSOURCE_FILE_ARRAY:
820  // if the first time through, open file
821  if (au->list_idx == 0 && au->in == NULL &&
822  au->source_list != NULL) {
823  if (au->source_list[au->list_idx] == NULL) {
824  errno = 0;
825  return -2;
826  }
827  au->line_number = 0;
828  au->in = fopen(au->source_list[au->list_idx],
829  "rm");
830  if (au->in == NULL)
831  return -1;
832  __fsetlocking(au->in, FSETLOCKING_BYCALLER);
833  }
834 
835  // loop reading lines from a file
836  while (au->in) {
837  if ((rc = readline_file(au)) == -2) {
838  // end of file, open next file,
839  // try readline again
840  fclose(au->in);
841  au->in = NULL;
842  au->list_idx++;
843  au->line_number = 0;
844  if (au->source_list[au->list_idx]) {
845  au->in = fopen(
846  au->source_list[au->list_idx],
847  "rm");
848  if (au->in == NULL)
849  return -1;
850  __fsetlocking(au->in,
851  FSETLOCKING_BYCALLER);
852  }
853  } else {
854  if (rc > 0)
855  au->line_number++;
856  return rc;
857  }
858  }
859  return -2; // return EOF
860  case AUSOURCE_BUFFER:
861  case AUSOURCE_BUFFER_ARRAY:
862  rc = readline_buf(au);
863  if (rc > 0)
864  au->line_number++;
865  return rc;
866  case AUSOURCE_FEED:
867  rc = readline_buf(au);
868  // No such thing as EOF for feed, translate EOF
869  // to data not available
870  if (rc == -2)
871  return 0;
872  else
873  if (rc > 0)
874  au->line_number++;
875  return rc;
876  default:
877  return -1;
878  }
879  return -1; /* should never reach here */
880 }

956 int auparse_next_event(auparse_state_t *au)
957 {

963  // clear previous event data in preparation to accumulate
964  // new event data
