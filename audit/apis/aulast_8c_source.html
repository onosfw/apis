51 /* This outputs a line of text reporting the login/out times */
52 static void report_session(lnode* cur)
53 {
54  int notime = 0;
55 
56  // Don't list failed logins
57  if (cur == NULL)
58  return;
59 
60  if (cur->result != bad)
61  return;
62 
63  if (cur->name) {
64  // This is a reboot record
65  printf("%-8.8s ", cur->name);
66  if (cur->end == 0) {
67  cur->end = time(NULL);
68  notime = 1;
69  }
70  } else {
71  struct passwd *p = getpwuid(cur->auid);
72  if (p)
73  printf("%-8.8s ", p->pw_name);
74  else
75  printf("%-8.u ", cur->auid);
76  }
77  if (strncmp("/dev/", cur->term, 5) == 0)
78  printf("%-12.12s ", cur->term+5);
79  else
80  printf("%-12.12s ", cur->term);
81  printf("%-16.16s ", cur->host ? cur->host : "?");
82  printf("%-16.16s ", ctime(&cur->start));
83  switch(cur->status)
84  {
85  case SESSION_START:
86  printf(" still logged in\n");
87  break;
88  case DOWN:
89  printf("- down\n");
90  break;
91  case CRASH:
92  printf("- crash\n");
93  break;
94  case GONE:
95  printf(" gone - no logout\n");
96  break;
97  case LOG_OUT: {
98  time_t secs;
99  int mins, hours, days;
100  if (notime)
101  printf("- %-7.5s", " ");
102  else
103  printf("- %-7.5s", ctime(&cur->end) + 11);
104  secs = cur->end - cur->start;
105  mins = (secs / 60) % 60;
106  hours = (secs / 3600) % 24;
107  days = secs / 86400;
108  if (days)
109  printf("(%d+%02d:%02d)\n", days, hours, mins);
110  else
111  printf("(%02d:%02d)\n", hours, mins);
112  }
113  break;
114  default:
115  printf("\n");
116  break;
117  }
118  if (proof) {
119  char start[32], end[32];
120  struct tm *btm;
121 
122  if (cur->loginuid_proof == 0 && cur->result == 1) // Bad login
123  printf(" audit event proof serial number:"
124  " %lu\n", cur->user_login_proof);
125  else
126  printf(" audit event proof serial numbers:"
127  " %lu, %lu, %lu\n", cur->loginuid_proof,
128  cur->user_login_proof, cur->user_end_proof);
129  printf(" Session data can be found with this search:\n");
130  btm = localtime(&cur->start);
131  strftime(start, sizeof(start), "%x %T", btm);
132  if (cur->end != 0) {
133  btm = localtime(&cur->end);
134  strftime(end, sizeof(end), "%x %T", btm);
135  printf(" ausearch --start %s --end %s",
136  start, end);
137  } else {
138  printf(" ausearch --start %s", start);
139  }
140  if (cur->name == NULL)
141  printf(" --session %d", cur->session);
142  if (cur->loginuid_proof == 0 && cur->result == 1) // Bad login
143  printf(" -a %lu", cur->user_login_proof);
144  printf("\n\n");
145  }
146 }

226 {

236

239  if (tses)
240  ses = auparse_get_field_int(au);
241 
242  // Get second uid field - we should be positioned past the first one
243  // gdm sends uid, everything else sends id, we try acct as last resort
244  tuid = auparse_find_field(au, "uid");
245  if (tuid)
246  uid = auparse_get_field_int(au);
247  else {
248  auparse_first_record(au);
249  tuid = auparse_find_field(au, "id");
250  if (tuid)
251  uid = auparse_get_field_int(au);
252  else {
253  auparse_first_record(au);
254  tuid = auparse_find_field(au, "acct");
255  if (tuid) {
256  const char *tacct = auparse_interpret_field(au);
257  struct passwd *pw = getpwnam (tacct);
258  if (pw != NULL)
259  uid = pw->pw_uid;
260  } else
261  auparse_first_record(au);
262  }
263  }
264 
265  start = auparse_get_time(au);
266 
267  host = auparse_find_field(au, "hostname");
268  if (host && strcmp(host, "?") == 0)
269  host = auparse_find_field(au, "addr");
270 
271  term = auparse_find_field(au, "terminal");
272  if (term == NULL)
273  term = "?";
274  tres = auparse_find_field(au, "res");
275  if (tres)
276  tres = auparse_interpret_field(au);
277  if (tres) {
278  if (strcmp(tres, "success") == 0)
279  result = 0;
280  else
281  result = 1;
282  }
283  // We only get tacct when its a bad login
284  if (result == 1) {
285  auparse_first_record(au);
286  tacct = auparse_find_field(au, "acct");
287  if (tacct)
288  tacct = auparse_interpret_field(au);
289  } else {
290  // Check that they are valid
291  if (pid == -1 || uid ==-1 || ses == -1) {
292  if (debug)
293  fprintf(stderr,
294  "Bad user login for event: %lu\n",
295  auparse_get_serial(au));
296  return;
297  }
298  }
299 
300  // See if this session is already open
301  if (result == 0)
302  cur = list_find_auid(&l, uid, pid, ses);
303  else
304  cur = NULL;
305  if (cur) {
306  // If we are limited to a specific terminal and
307  // we find out the session is not associated with
308  // the terminal of interest, delete the current node
309  if (cterm && strstr(term, cterm) == NULL) {
310  list_delete_cur(&l);
311  if (debug)
312  fprintf(stderr,
313  "User login limited to %s for event: %lu\n",
314  cterm, auparse_get_serial(au));
315  return;
316  }
317 
318  // This means we have an open session - update it
319  list_update_start(&l, start, host, term, result,
320  auparse_get_serial(au));
321 
322  // If the results were failed, we can close it out
323  /* FIXME: result cannot be true. This is dead code.
324  if (result) {
325  report_session(cur);
326  list_delete_cur(&l);
327  } */
328  } else if (bad == 1 && result == 1) {
329  // If it were a bad login and we are wanting bad logins
330  // create the record and report it.
331  lnode n;
332 
333  n.start = start;
334  n.end = start;
335  n.auid = uid;
336  n.name = tacct;
337  n.term = term;
338  n.host = host;
339  n.result = result;
340  n.status = LOG_OUT;
341  n.loginuid_proof = 0;
342  n.user_login_proof = auparse_get_serial(au);
343  n.user_end_proof = 0;
344  report_session(&n);
345  } else if (debug)
346  printf("Session not found or updated\n");
347 }

395 static void process_bootup(auparse_state_t *au)
396 {
397  lnode *cur;
398  int start;
399 
400  // See if we have unclosed boot up and make into CRASH record
401  list_first(&l);
402  cur = list_get_cur(&l);
403  while (cur) {
404  if (cur->name) {
405  cur->user_end_proof = auparse_get_serial(au);
406  cur->status = CRASH;
407  cur->end = auparse_get_time(au);
408  report_session(cur);
409  }
410  cur = list_next(&l);
411  }
412 
413  // Logout and process anyone still left in the machine
414  list_first(&l);
415  cur = list_get_cur(&l);
416  while (cur) {
417  if (cur->status != CRASH) {
418  cur->user_end_proof = auparse_get_serial(au);
419  cur->status = DOWN;
420  cur->end = auparse_get_time(au);
421  report_session(cur);
422  }
423  cur = list_next(&l);
424  }
425 
426  // Since this is a boot message, all old entries should be gone
427  list_clear(&l);
428  list_create(&l);
429 
430  // make reboot record - user:reboot, tty:system boot, host: kernel
431  start = auparse_get_time(au);
432  list_create_session(&l, 0, 0, 0, auparse_get_serial(au));
433  cur = list_get_cur(&l);
434  cur->start = start;
435  cur->name = strdup("reboot");
436  cur->term = strdup("system boot");
437  if (kernel)
438  cur->host = strdup(kernel);
439  cur->result = 0;
440 }

472 int main(int argc, char *argv[])
473 {
474  int i, use_stdin = 0;
475  char *user = NULL, *file = NULL;
476  struct passwd *p;
477  auparse_state_t *au;
478 
479  setlocale (LC_ALL, "");
480  for (i=1; i<argc; i++) {
481  if (argv[i][0] != '-') {
482  //take input and lookup as if it were a user name
483  //if that fails assume its a tty
484  if (user == NULL) {
485  p = getpwnam(argv[i]);
486  if (p) {
487  cuid = p->pw_uid;
488  user = argv[i];
489  continue;
490  }
491  }
492  if (cterm == NULL) {
493  cterm = argv[i];
494  } else {
495  usage();
496  return 1;
497  }
498  } else {
499  if (strcmp(argv[i], "-f") == 0) {
500  if (use_stdin == 0) {
501  i++;
502  file = argv[i];
503  } else {
504  fprintf(stderr,"stdin already given\n");
505  return 1;
506  }
507  } else if (strcmp(argv[i], "--bad") == 0) {
508  bad = 1;
509  } else if (strcmp(argv[i], "--proof") == 0) {
510  proof = 1;
511  } else if (strcmp(argv[i], "--extract") == 0) {
512  f = fopen("aulast.log", "wt");
513  } else if (strcmp(argv[i], "--stdin") == 0) {
514  if (file == NULL)
515  use_stdin = 1;
516  else {
517  fprintf(stderr, "file already given\n");
518  return 1;
519  }
520  } else if (strcmp(argv[i], "--debug") == 0) {
521  debug = 1;
522  } else {
523  usage();
524  return 1;
525  }
526  }
527  }
528  list_create(&l);
529 
530  // Search for successful user logins
531  if (file)
532  au = auparse_init(AUSOURCE_FILE, file);
533  else if (use_stdin)
534  au = auparse_init(AUSOURCE_FILE_POINTER, stdin);
535  else {
536  if (getuid()) {
537  fprintf(stderr, "You probably need to be root for this to work\n");
538  }
539  au = auparse_init(AUSOURCE_LOGS, NULL);
540  }
541  if (au == NULL) {
542  fprintf(stderr, "Error - %s\n", strerror(errno));
543  goto error_exit_1;
544  }
545 
546  // The theory: iterate though events
547  // 1) when LOGIN is found, create a new session node
548  // 2) if that session number exists, close out the old one
549  // 3) when USER_LOGIN is found, update session node
550  // 4) When USER_END is found update session node and close it out
551  // 5) When BOOT record found make new record and check for previous
552  // 6) If previous boot found, set status to crash and logout everyone
553  // 7) When SHUTDOWN found, close out reboot record
554 
555  while (auparse_next_event(au) > 0) {
556  // We will take advantage of the fact that all events
557  // of interest are one record long
558  int type = auparse_get_type(au);
559  if (type < 0)
560  continue;
561  switch (type)
562  {
563  case AUDIT_LOGIN:
564  create_new_session(au);
565  extract_record(au);
566  break;
567  case AUDIT_USER_LOGIN:
568  update_session_login(au);
569  extract_record(au);
570  break;
571  case AUDIT_USER_END:
572  update_session_logout(au);
573  extract_record(au);
574  break;
575  case AUDIT_SYSTEM_BOOT:
576  process_bootup(au);
577  extract_record(au);
578  break;
579  case AUDIT_SYSTEM_SHUTDOWN:
580  process_shutdown(au);
581  extract_record(au);
582  break;
583  case AUDIT_DAEMON_START:
584  process_kernel(au);
585  extract_record(au);
586  break;
587  }
588  }
589  auparse_destroy(au);
590 
591  // Now output the leftovers
592  list_first(&l);
593  do {
594  lnode *cur = list_get_cur(&l);
595  report_session(cur);
596  } while (list_next(&l));
597 
598  free(kernel);
599  list_clear(&l);
600  if (f)
601  fclose(f);
602  return 0;
603 
604 error_exit_1:
605  list_clear(&l);
606  if (f)
607  fclose(f);
608  return 1;
609 }
