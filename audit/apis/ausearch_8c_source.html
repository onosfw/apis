82 int main(int argc, char *argv[])
83 {
84  struct rlimit limit;
85  int rc;
86  struct stat sb;
87 
88  /* Check params and build regexpr */
89  setlocale (LC_ALL, "");
90  if (check_params(argc, argv))
91  return 1;
92 
93  /* Raise the rlimits in case we're being started from a shell
94  * with restrictions. Not a fatal error. */
95  limit.rlim_cur = RLIM_INFINITY;
96  limit.rlim_max = RLIM_INFINITY;
97  setrlimit(RLIMIT_FSIZE, &limit);
98  setrlimit(RLIMIT_CPU, &limit);
99  set_aumessage_mode(MSG_STDERR, DBG_NO);
100  (void) umask( umask( 077 ) | 027 );
101 
102  /* Load the checkpoint file if requested */
103  if (checkpt_filename) {
104  rc = load_ChkPt(checkpt_filename);
105  /*
106  * If < -1, then some load/parse error
107  * If == -1 then no file present (OK)
108  * If == 0, then checkpoint has data
109  */
110  if (rc < -1) {
111  (void)free((void *)checkpt_filename);
112  free_ChkPtMemory();
113  return 10; /* bad checkpoint status file */
114  } else if (rc == -1) {
115  /*
116  * No file, so no checking required. This just means
117  * we have never checkpointed before and this is the
118  * first time.
119  */
120  have_chkpt_data = 0;
121  } else {
122  /* We will need to check */
123  have_chkpt_data++;
124  }
125  }
126 
127  lol_create(&lo);
128  if (user_file) {
129  if (stat(user_file, &sb) == -1) {
130  perror("stat");
131  return 1;
132  }
133  switch (sb.st_mode & S_IFMT) {
134  case S_IFDIR:
135  userfile_is_dir = 1;
136  rc = process_logs();
137  break;
138  case S_IFREG:
139  default:
140  rc = process_file(user_file);
141  break;
142  }
143  } else if (force_logs)
144  rc = process_logs();
145  else if (is_pipe(0))
146  rc = process_stdin();
147  else
148  rc = process_logs();
149 
150  /* Generate a checkpoint if required */
151  if (checkpt_filename) {
152  /* Providing we found something and haven't failed */
153  if (!checkpt_failure && found)
154  save_ChkPt(checkpt_filename);
155  free_ChkPtMemory();
156  free((void *)checkpt_filename);
157  /*
158  * A checkpoint failure at this point means either
159  * - we failed in attempting to create the checkpouint file
160  * and so we will return 11
161  * - we had a corrupted checkpoint file and so we will return 12
162  */
163  if (checkpt_failure) {
164  rc = ((checkpt_failure & CP_CORRUPTED) ==
165  CP_CORRUPTED) ? 12 : 11;
166  }
167  }
168 
169  lol_clear(&lo);
170  ilist_clear(event_type);
171  free(event_type);
172  free(user_file);
173  free((char *)event_key);
174  auparse_destroy(NULL);
175  if (rc)
176  return rc;
177  if (!found) {
178  if (report_format != RPT_RAW)
179  fprintf(stderr, "<no matches>\n");
180  return 1;
181  }
182  return 0;
183 }

203  fprintf(stderr, "NOTE - using logs in %s\n", config.log_file);
204  }
205  else {
206  /* Load config so we know where logs are */
207  if (load_config(&config, TEST_SEARCH)) {
208  fprintf(stderr,
209  "NOTE - using built-in logs: %s\n",
210  config.log_file);
211  }
212  }
213 
214  /* for each file */
215  len = strlen(config.log_file) + 16;
216  filename = malloc(len);
217  if (!filename) {
218  fprintf(stderr, "No memory\n");
219  free_config(&config);
220  return 1;
221  }
222  /* Find oldest log file */
223  snprintf(filename, len, "%s", config.log_file);
224  do {
225  if (access(filename, R_OK) != 0)
226  break;
227 
228  /*
229  * If we have prior checkpoint data, we ignore files till we
230  * find the file we last checkpointed from
231  */
232  if (checkpt_filename && have_chkpt_data) {
233  struct stat sbuf;

253  */

256  break;
257  }
258  }
259  }
260 
261  num++;

263  } while (1);
264

266  * are not only just checking the timestamp from the checkpoint file,
267  * we need to error */
268  if (checkpt_filename && have_chkpt_data && found_chkpt_file == -1

273  }

276

278  files_to_process = num;
279 
280  /* Got it, now process logs from last to first */
281  if (num > 0)

285  do {

289  return ret;

292  break;

294

296  num--;

298  snprintf(filename, len, "%s.%d", config.log_file, num);
299  else if (num == 0)
300  snprintf(filename, len, "%s", config.log_file);
301  else
302  break;
303  } while (1);

306  * file details - ie remember the last file processed
307  */
308  ret = 0;
309  if (checkpt_filename)
310  ret = set_ChkPtFileDetails(filename);
311 
312  free(filename);
313  free_config(&config);

317 /*
318  * Decide if we should start outputing events given we loaded a checkpoint.
319  *
320  * The previous checkpoint will have recorded the last event outputted,
321  * if there was one. If nothing is to be output, either the audit.log file
322  * is empty, all the events in it were incomplete, or ???
323  *
324  * We can return
325  * 0 no output
326  * 1 can output
327  * 2 can output but not this event
328  * 3 we have found an event whose time is > MAX_EVENT_DELTA_SECS secs
329  * past our checkpoint time, which means this particulare event is
330  * complete. This should not happen, for we should have found our
331  * checkpoint event before ANY other completed event.
332  *
333  */
334 static int chkpt_output_decision(event * e)
335 {
336  static int can_output = 0;
337 
338  /* Short cut. Once we made the decision, it's made for good */
339  if (can_output)
340  return 1;
341 
342  /* If there was no checkpoint file, we turn on output */
343  if (have_chkpt_data == 0) {
344  can_output = 1;
345  return 1; /* can output on this event */
346  }
347 
348  /*
349  * If the previous checkpoint had no recorded output, then
350  * we assume everything was partial so we turn on output
351  */
352  if (chkpt_input_levent.sec == 0) {
353  can_output = 1;
354  return 1; /* can output on this event */
355  }
356 
357  /*
358  * If we are ignoring all but event time from within the checkpoint
359  * file, then we output if the current event's time is greater than
360  * or equal to the checkpoint time.
361  */
362  if (checkpt_timeonly) {
363  if ( (chkpt_input_levent.sec < e->sec) ||
364  ( (chkpt_input_levent.sec == e->sec) &&
365  (chkpt_input_levent.milli <= e->milli) ) ) {
366  can_output = 1;
367  return 1; /* can output on this event */
368  }
369  }
370 
371  if (chkpt_input_levent.sec == e->sec &&
372  chkpt_input_levent.milli == e->milli &&
373  chkpt_input_levent.serial == e->serial &&
374  chkpt_input_levent.type == e->type ) {
375 
376  /* So far a match, so now check the nodes */
377  if (chkpt_input_levent.node == NULL && e->node == NULL) {
378  can_output = 1;
379  return 2; /* output after this event */
380  }
381  if (chkpt_input_levent.node && e->node &&
382  (strcmp(chkpt_input_levent.node, e->node) == 0) ) {
383  can_output = 1;
384  return 2; /* output after this event */
385  }
386  /*
387  * The nodes are different. Drop through to further checks.
388  */
389  }
390  /*
391  * If the event we are looking at is more than MAX_EVENT_DELTA_SECS
392  * seconds past our checkpoint event, then by definition we should
393  * have had a complete event (ie a complete event is one where at
394  * least MAX_EVENT_DELTA_SECS seconds have passed since it's last
395  * output record).
396  * This means there is a problem, for the recorded checkpoint event was
397  * the last complete event in the file when we last processed it.
398  * Normally we see this if the checkpoint is very old and the system
399  * has used the same inode again in an audit log file.
400  */
401  if ( (chkpt_input_levent.sec < e->sec) &&
402  ((e->sec - chkpt_input_levent.sec) > MAX_EVENT_DELTA_SECS) ) {
403 /* fprintf(stderr, "%s %lu.%03u:%lu vs %s %lu.%03u:%lu\n",
404  chkpt_input_levent.host ? chkpt_input_levent.host : "-",
405  chkpt_input_levent.sec, chkpt_input_levent.milli,
406  chkpt_input_levent.serial,
407  e->host, e->sec, e->milli, e->serial); */
408  return 3;
409  }
410 
411  return 0;
412 }

414 static int process_log_fd(void)
415 {
416  llist *entries; // entries in a record
417  int ret;
418  int do_output = 1;
419 
420  /* For each record in file */
421  do {
422  ret = get_record(&entries);
423  if ((ret != 0)||(entries->cnt == 0)) {
424  break;
425  }
426  /*
427  * We flush all events on the last log file being processed.
428  * Thus incomplete events are 'carried forward' to be
429  * completed from the rest of it's records we expect to find
430  * in the next file we are about to process.
431  */
432  if (match(entries)) {
433  /*
434  * If we are checkpointing, decide if we output
435  * this event
436  */
437  if (checkpt_filename)
438  do_output = chkpt_output_decision(&entries->e);
439 
440  if (do_output == 1) {
441  found = 1;
442  output_record(entries);
443 
444  /* Remember this event if checkpointing */
445  if (checkpt_filename) {
446  if (set_ChkPtLastEvent(&entries->e)) {
447  list_clear(entries);
448  free(entries);
449  fclose(log_fd);
450  return 4; /* no memory */
451  }
452  }
453  } else if (do_output == 3) {
454  fprintf(stderr,
455  "Corrupted checkpoint file. Inode match, but newer complete event (%lu.%03u:%lu) found before loaded checkpoint %lu.%03u:%lu\n",
456  entries->e.sec, entries->e.milli,
457  entries->e.serial,
458  chkpt_input_levent.sec,
459  chkpt_input_levent.milli,
460  chkpt_input_levent.serial);
461  checkpt_failure |= CP_CORRUPTED;
462  list_clear(entries);
463  free(entries);
464  fclose(log_fd);
465  return 10;
466  }
467  if (just_one) {
468  list_clear(entries);
469  free(entries);
470  break;
471  }
472  if (line_buffered)
473  fflush(stdout);
474  }
475  list_clear(entries);
476  free(entries);
477  } while (ret == 0);
478  fclose(log_fd);
479 
480  return 0;
481 }

513 /*
514  * This function returns a malloc'd buffer of the next record in the audit
515  * logs. It returns 0 on success, 1 on eof, -1 on error.
516  */
517 static int get_record(llist **l)
518 {
519  char *rc;
520  char *buff = NULL;
521  int rcount = 0, timer_running = 0;
522 
523  /*
524  * If we have any events ready to print ie have all records that
525  * make up the event, we just return. If not, we read more lines
526  * from the files until we get a complete event or finish reading
527  * input
528  */
529  *l = get_ready_event(&lo);
530  if (*l)
531  return 0;
532 
533  while (1) {
534  rcount++;
535 
536  if (!buff) {
537  buff = malloc(MAX_AUDIT_MESSAGE_LENGTH);
538  if (!buff)
539  return -1;
540  }
541 
542  if (input_is_pipe && rcount > 1) {
543  timer_running = 1;
544  alarm(timeout_interval);
545  }
546 
547  rc = fgets_unlocked(buff, MAX_AUDIT_MESSAGE_LENGTH,
548  log_fd);
549 
550  if (timer_running) {
551  /* timer may have fired but thats ok */
552  timer_running = 0;
553  alarm(0);
554  }
555 
556  if (rc) {
557  if (lol_add_record(&lo, buff)) {
558  *l = get_ready_event(&lo);
559  if (*l)
560  break;
561  }
562  } else {
563  free(buff);
564  /*
565  * If we get an EINTR error or we are at EOF, we check
566  * to see if we have any events to print and return
567  * appropriately. If we are the last file being
568  * processed, we mark all incomplete events as
569  * complete so they will be printed.
570  */
571  if ((ferror_unlocked(log_fd) &&
572  errno == EINTR) || feof_unlocked(log_fd)) {
573  /*
574  * Only mark all events as L_COMPLETE if we are
575  * the last file being processed.
576  * We DO NOT do this if we are checkpointing.
577  */
578  if (files_to_process == 0) {
579  if (!checkpt_filename)
580  terminate_all_events(&lo);
581  }
582  *l = get_ready_event(&lo);
583  if (*l)
584  return 0;
585  else
586  return 1;
587  } else
588  return -1; /* all other errors are terminal */
589  }
590  }
591  free(buff);
592  return 0;
593 }
